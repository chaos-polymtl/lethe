15template <int dim>
16void
17write_checkpoint(
18 TimerOutput &computing_timer,
19 const DEMSolverParameters<dim> &parameters,
20 std::shared_ptr<SimulationControl> &simulation_control,
21 PVDHandler &particles_pvdhandler,
22 PVDHandler &grid_pvdhandler,
23 parallel::distributed::Triangulation<dim> &triangulation,
24 Particles::ParticleHandler<dim> &particle_handler,
25 std::shared_ptr<Insertion<dim>> &insertion_object,
26 std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> &solid_objects,
27 const ConditionalOStream &pcout,
28 MPI_Comm &mpi_communicator,
29 const CheckpointControl &checkpoint_controller)
30{
31 TimerOutput::Scope timer(computing_timer, "Write checkpoint");
32
33 pcout << "Writing restart file" << std::endl;
34
35 std::string prefix =
36 checkpoint_controller.get_filename() + "_" +
37 Utilities::int_to_string(checkpoint_controller.get_next_checkpoint_id());
38
39 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
40 {
41 simulation_control->save(prefix);
42 particles_pvdhandler.save(prefix);
43
44 if (parameters.post_processing.lagrangian_post_processing_enabled)
45 {
46 grid_pvdhandler.save(prefix + "_lagrangian_postprocessing");
47 }
48 }
49
50 // Prepare the particle handler for checkpointing
51 particle_handler.prepare_for_serialization();
52
53 std::ostringstream oss;
54 boost::archive::text_oarchive oa(oss, boost::archive::no_header);
55 oa << particle_handler;
56 std::string triangulation_name = prefix + ".triangulation";
57 triangulation.save(prefix + ".triangulation");
58
59 // Write additional particle information for deserialization
60 std::string particle_filename = prefix + ".particles";
61 std::ofstream output(particle_filename.c_str());
62 output << oss.str() << std::endl;
63
64 // Prepare the insertion object for checkpointing
65 std::string insertion_object_filename = prefix + ".insertion_object";
66 std::ofstream oss_insertion_obj(insertion_object_filename);
67 boost::archive::text_oarchive oa_insertion_obj(oss_insertion_obj,
68 boost::archive::no_header);
69 insertion_object->serialize(oa_insertion_obj, 0);
70
71 // Checkpoint the serial solid objects one by one
72 for (unsigned int i = 0; i < solid_objects.size(); ++i)
73 {
74 solid_objects[i]->write_checkpoint(prefix);
75 }
76
77 // Prepare the checkpoint controller for checkpointing
78 // We don't use the same prefix, since this file needs to have the same name
79 // regardless of the checkpoint id being used. This file is giving the
80 // information of which checkpoint id to use when restarting.
81 std::string checkpoint_controller_object_filename =
82 checkpoint_controller.get_filename() + ".checkpoint_controller";
83 std::ofstream oss_checkpoint_controller_obj(
84 checkpoint_controller_object_filename);
85 boost::archive::text_oarchive oa_checkpoint_controller_obj(
86 oss_checkpoint_controller_obj, boost::archive::no_header);
87 checkpoint_controller.serialize(oa_checkpoint_controller_obj);
88}
