14template <int dim, typename PropertiesIndex>
15void
16write_checkpoint(
17 TimerOutput &computing_timer,
18 const DEMSolverParameters<dim> &parameters,
19 std::shared_ptr<SimulationControl> &simulation_control,
20 PVDHandler &particles_pvdhandler,
21 PVDHandler &grid_pvdhandler,
22 parallel::distributed::Triangulation<dim> &triangulation,
23 Particles::ParticleHandler<dim> &particle_handler,
24 std::shared_ptr<Insertion<dim, PropertiesIndex>> &insertion_object,
25 std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> &solid_objects,
26 const ConditionalOStream &pcout,
27 MPI_Comm &mpi_communicator,
28 const CheckpointControl &checkpoint_controller)
29{
30 TimerOutput::Scope timer(computing_timer, "Write checkpoint");
31
32 pcout << "Writing restart file" << std::endl;
33
34 std::string prefix =
35 checkpoint_controller.get_filename() + "_" +
36 Utilities::int_to_string(checkpoint_controller.get_next_checkpoint_id());
37
38 if (Utilities::MPI::this_mpi_process(mpi_communicator) == 0)
39 {
40 simulation_control->save(prefix);
41 particles_pvdhandler.save(prefix);
42
43 if (parameters.post_processing.lagrangian_post_processing_enabled)
44 {
45 grid_pvdhandler.save(prefix + "_lagrangian_postprocessing");
46 }
47 }
48
49 // Prepare the particle handler for checkpointing
50 particle_handler.prepare_for_serialization();
51
52 std::ostringstream oss;
53 boost::archive::text_oarchive oa(oss, boost::archive::no_header);
54 oa << particle_handler;
55 std::string triangulation_name = prefix + ".triangulation";
56 triangulation.save(prefix + ".triangulation");
57
58 // Write additional particle information for deserialization
59 std::string particle_filename = prefix + ".particles";
60 std::ofstream output(particle_filename.c_str());
61 output << oss.str() << std::endl;
62
63 // Prepare the insertion object for checkpointing
64 std::string insertion_object_filename = prefix + ".insertion_object";
65 std::ofstream oss_insertion_obj(insertion_object_filename);
66 boost::archive::text_oarchive oa_insertion_obj(oss_insertion_obj,
67 boost::archive::no_header);
68 insertion_object->serialize(oa_insertion_obj);
69
70 // Checkpoint the serial solid objects one by one
71 for (unsigned int i = 0; i < solid_objects.size(); ++i)
72 {
73 solid_objects[i]->write_checkpoint(prefix);
74 }
75
76 // Prepare the checkpoint controller for checkpointing
77 // We don't use the same prefix, since this file needs to have the same name
78 // regardless of the checkpoint id being used. This file is giving the
79 // information of which checkpoint id to use when restarting.
80 std::string checkpoint_controller_object_filename =
81 checkpoint_controller.get_filename() + ".checkpoint_controller";
82 std::ofstream oss_checkpoint_controller_obj(
83 checkpoint_controller_object_filename);
84 boost::archive::text_oarchive oa_checkpoint_controller_obj(
85 oss_checkpoint_controller_obj, boost::archive::no_header);
86 checkpoint_controller.serialize(oa_checkpoint_controller_obj);
87}
