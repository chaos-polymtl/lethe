16template <int dim>
17VolumeOfFluid<dim>::VolumeOfFluid(
18 MultiphysicsInterface<dim> *multiphysics_interface,
19 const SimulationParameters<dim> &p_simulation_parameters,
20 std::shared_ptr<parallel::DistributedTriangulationBase<dim>> p_triangulation,
21 std::shared_ptr<SimulationControl> p_simulation_control)
22 : AuxiliaryPhysics<dim, GlobalVectorType>(
23 p_simulation_parameters.non_linear_solver.at(PhysicsID::VOF))
24 , multiphysics(multiphysics_interface)
25 , computing_timer(p_triangulation->get_mpi_communicator(),
26 this->pcout,
27 TimerOutput::summary,
28 TimerOutput::wall_times)
29 , simulation_parameters(p_simulation_parameters)
30 , triangulation(p_triangulation)
31 , simulation_control(std::move(p_simulation_control))
32 , dof_handler(*triangulation)
33 , sharpening_threshold(simulation_parameters.multiphysics.vof_parameters
34 .regularization_method.sharpening.threshold)
35{
36 AssertThrow(
37 simulation_parameters.physical_properties_manager.get_number_of_fluids() ==
38 2,
39 InvalidNumberOfFluid(simulation_parameters.physical_properties_manager
40 .get_number_of_fluids()));
41
42 AssertThrow(((simulation_parameters.fem_parameters.VOF_uses_dg &&
43 simulation_parameters.multiphysics.vof_parameters
44 .regularization_method.regularization_method_type ==
45 Parameters::RegularizationMethodType::none) ||
46 !simulation_parameters.fem_parameters.VOF_uses_dg),
47 UnsupportedRegularization());
48
49 if (simulation_parameters.fem_parameters.VOF_uses_dg &&
50 this->simulation_parameters.post_processing.calculate_mass_conservation)
51 {
52 this->pcout
53 << "Warning: DG-VOF and the geometric surface and volume computations are not compatible. Only the global volume will be monitored."
54 << std::endl;
55 }
56
57 if (simulation_parameters.mesh.simplex)
58 {
59 // for simplex meshes
60 fe = std::make_shared<FE_SimplexP<dim>>(
61 simulation_parameters.fem_parameters.VOF_order);
62 mapping = std::make_shared<MappingFE<dim>>(*fe);
63 cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
64 face_quadrature =
65 std::make_shared<QGaussSimplex<dim - 1>>(fe->degree + 1);
66 }
67 else
68 {
69 // Usual case, for quad/hex meshes
70 if (simulation_parameters.fem_parameters.VOF_uses_dg)
71 {
72 fe = std::make_shared<FE_DGQ<dim>>(
73 simulation_parameters.fem_parameters.VOF_order);
74 }
75 else
76 {
77 fe = std::make_shared<FE_Q<dim>>(
78 simulation_parameters.fem_parameters.VOF_order);
79 }
80 // Mapping has to be at least Q1, but DGQ0 is allowed
81 mapping = std::make_shared<MappingQ<dim>>(std::max(fe->degree, uint(1)));
82 cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
83 face_quadrature = std::make_shared<QGauss<dim - 1>>(fe->degree + 1);
84 }
85
86 // Allocate solution transfer
87 solution_transfer =
88 std::make_shared<SolutionTransfer<dim, GlobalVectorType>>(dof_handler);
89
90 // Set size of previous solutions using BDF schemes information
91 previous_solutions.resize(maximum_number_of_previous_solutions());
92
93 // Prepare previous solutions transfer
94 previous_solutions_transfer.reserve(previous_solutions.size());
95 for (unsigned int i = 0; i < previous_solutions.size(); ++i)
96 {
97 previous_solutions_transfer.emplace_back(
98 SolutionTransfer<dim, GlobalVectorType>(this->dof_handler));
99 }
100
101 // Check the value of interface sharpness
102 if (simulation_parameters.multiphysics.vof_parameters.regularization_method
103 .sharpening.interface_sharpness < 1.0)
104 this->pcout
105 << "Warning: interface sharpness values smaller than 1 smooth the interface instead of sharpening it."
106 << std::endl
107 << "The interface sharpness value should be set between 1 and 2"
108 << std::endl;
109
110
111 // Change the behavior of the timer for situations when you don't want
112 // outputs
113 if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
114 this->computing_timer.disable_output();
115
116 // Initialize the interface object for subequations to solve
117 this->vof_subequations_interface =
118 std::make_shared<VOFSubequationsInterface<dim>>(this->simulation_parameters,
119 this->pcout,
120 this->triangulation,
121 this->simulation_control);
122
123
124 if (simulation_parameters.multiphysics.vof_parameters.regularization_method
125 .geometric_interface_reinitialization.enable)
126 {
127 /* For the VOF solver, the interface is defined as the iso-contour
128 \f$\phi\f = 0.5$. Hence, for the SignedDistanceSolver, we set the
129 iso-level to 0.5. We also define the inside part of the domain as
130 \f$\phi>0.5\f$. Since the SignedDistanceSolver considers the inside
131 of the domain as \f$d<0\f$, we scale the phase fraction with a factor
132 of -1.*/
133 this->signed_distance_solver = std::make_shared<
134 InterfaceTools::SignedDistanceSolver<dim, GlobalVectorType>>(
135 triangulation,
136 fe,
137 simulation_parameters.multiphysics.vof_parameters.regularization_method
138 .geometric_interface_reinitialization.max_reinitialization_distance,
139 0.5,
140 -1.0,
141 simulation_parameters.multiphysics.vof_parameters.regularization_method
142 .verbosity);
143 this->signed_distance_transformation =
144 SignedDistanceTransformationBase::model_cast(
145 simulation_parameters.multiphysics.vof_parameters
146 .regularization_method.geometric_interface_reinitialization);
147 }
148}

159template <int dim>
160void
161VolumeOfFluid<dim>::setup_assemblers()
162{
163 AssertThrow(
164 is_sdirk(this->simulation_control->get_assembly_method()) == false,
165 ExcMessage("The SDIRK scheme is not yet supported for this physics"));
166
167 this->assemblers.clear();
168
169 // Time-stepping schemes
170 if (is_bdf(this->simulation_control->get_assembly_method()))
171 {
172 this->assemblers.emplace_back(
173 std::make_shared<VOFAssemblerBDF<dim>>(this->simulation_control));
174 }
175
176 // If the VOF solver uses DG, a different set of assemblers is used
177 if (simulation_parameters.fem_parameters.VOF_uses_dg)
178 {
179 this->assemblers.emplace_back(
180 std::make_shared<VOFAssemblerDGCore<dim>>());
181 this->inner_face_assembler = std::make_shared<VOFAssemblerSIPG<dim>>();
182 }
183 else
184 {
185 // Core assembler
186 this->assemblers.emplace_back(std::make_shared<VOFAssemblerCore<dim>>(
187 this->simulation_control,
188 this->simulation_parameters.fem_parameters,
189 this->simulation_parameters.multiphysics.vof_parameters));
190
191 // DCDD shock-capturing assembler
192 if (this->simulation_parameters.stabilization.vof_dcdd_stabilization)
193 this->assemblers.emplace_back(
194 std::make_shared<VOFAssemblerDCDDStabilization<dim>>(
195 this->simulation_control));
196 }
197}

244template <int dim>
245void
246VolumeOfFluid<dim>::assemble_system_matrix_dg()
247{
248 const DoFHandler<dim> *dof_handler_fluid =
249 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
250
251 auto scratch_data =
252 VOFScratchData<dim>(this->simulation_control,
253 this->simulation_parameters.physical_properties_manager,
254 *this->fe,
255 *this->cell_quadrature,
256 *this->face_quadrature,
257 *this->mapping,
258 dof_handler_fluid->get_fe());
259
260 StabilizedDGMethodsCopyData copy_data(this->fe->n_dofs_per_cell(),
261 this->cell_quadrature->size());
262
263 // We first wrap the assembly of the matrix within a cell_worker lambda
264 // function. This is only done for compatibility reasons with the MeshWorker
265 // paradigm and does not have any functional purpose.
266 const auto cell_worker =
267 [&](const typename DoFHandler<dim>::active_cell_iterator &cell,
268 VOFScratchData<dim> &scratch_data,
269 StabilizedDGMethodsCopyData &copy_data) {
270 this->assemble_local_system_matrix(cell, scratch_data, copy_data);
271 };
272
273 const auto boundary_worker =
274 [&]([[maybe_unused]] const typename DoFHandler<dim>::active_cell_iterator
275 &cell,
276 [[maybe_unused]] const unsigned int &face_no,
277 [[maybe_unused]] VOFScratchData<dim> &scratch_data,
278 [[maybe_unused]] StabilizedDGMethodsCopyData &copy_data) {};
279
280 const auto face_worker =
281 [&](const typename DoFHandler<dim>::active_cell_iterator &cell,
282 const unsigned int &face_no,
283 const unsigned int &sub_face_no,
284 const typename DoFHandler<dim>::active_cell_iterator &neigh_cell,
285 const unsigned int &neigh_face_no,
286 const unsigned int &neigh_sub_face_no,
287 VOFScratchData<dim> &scratch_data,
288 StabilizedDGMethodsCopyData &copy_data) {
289 scratch_data.reinit_internal_face(cell,
290 face_no,
291 sub_face_no,
292 neigh_cell,
293 neigh_face_no,
294 neigh_sub_face_no,
295 this->evaluation_point);
296
297 // Pad copy_data memory for the internal faces elementary matrices
298 // BB note : Array could be pre-allocated
299 copy_data.face_data.emplace_back();
300 auto &copy_data_face = copy_data.face_data.back();
301 copy_data_face.face_matrix.reinit(scratch_data.n_interface_dofs,
302 scratch_data.n_interface_dofs);
303 copy_data_face.joint_dof_indices =
304 scratch_data.fe_interface_values_vof.get_interface_dof_indices();
305
306 // Gather velocity information at the face to advect properly
307 // Get the cell that corresponds to the fluid dynamics
308 typename DoFHandler<dim>::active_cell_iterator velocity_cell(
309 &(*triangulation), cell->level(), cell->index(), dof_handler_fluid);
310
311 // Reinit the internal face velocity within the scratch data
312 reinit_face_velocity_with_adequate_solution(velocity_cell,
313 face_no,
314 scratch_data);
315
316 this->inner_face_assembler->assemble_matrix(scratch_data, copy_data);
317 };
318
319 const auto copier = [&](const StabilizedDGMethodsCopyData &copy_data) {
320 this->copy_local_matrix_to_global_matrix(copy_data);
321
322 const AffineConstraints<double> &constraints_used = this->zero_constraints;
323
324 for (const auto &cdf : copy_data.face_data)
325 {
326 constraints_used.distribute_local_to_global(cdf.face_matrix,
327 cdf.joint_dof_indices,
328 system_matrix);
329 }
330 };
331
332 MeshWorker::mesh_loop(this->dof_handler.begin_active(),
333 this->dof_handler.end(),
334 cell_worker,
335 copier,
336 scratch_data,
337 copy_data,
338 MeshWorker::assemble_own_cells |
339 MeshWorker::assemble_boundary_faces |
340 MeshWorker::assemble_own_interior_faces_once |
341 MeshWorker::assemble_ghost_faces_both,
342 boundary_worker,
343 face_worker);
344}

347template <int dim>
348void
349VolumeOfFluid<dim>::assemble_local_system_matrix(
350 const typename DoFHandler<dim>::active_cell_iterator &cell,
351 VOFScratchData<dim> &scratch_data,
352 StabilizedMethodsCopyData &copy_data)
353{
354 copy_data.cell_is_local = cell->is_locally_owned();
355 if (!cell->is_locally_owned())
356 return;
357
358 scratch_data.reinit(cell, this->evaluation_point, this->previous_solutions);
359
360 const DoFHandler<dim> *dof_handler_fd =
361 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
362
363 typename DoFHandler<dim>::active_cell_iterator velocity_cell(
364 &(*this->triangulation), cell->level(), cell->index(), dof_handler_fd);
365
366 if (multiphysics->fluid_dynamics_is_block())
367 {
368 // Check if the post processed variable needs to be calculated with the
369 // average velocity profile or the fluid solution.
370 if (this->simulation_parameters.initial_condition->type ==
371 Parameters::FluidDynamicsInitialConditionType::
372 average_velocity_profile &&
373 !this->simulation_parameters.multiphysics.fluid_dynamics &&
374 simulation_control->get_current_time() >
375 this->simulation_parameters.post_processing
376 .initial_time_for_average_velocities)
377 {
378 scratch_data.reinit_velocity(
379 velocity_cell,
380 *multiphysics->get_block_time_average_solution(
381 PhysicsID::fluid_dynamics),
382 *multiphysics->get_block_previous_solutions(
383 PhysicsID::fluid_dynamics),
384 this->simulation_parameters.ale);
385 }
386 else
387 {
388 scratch_data.reinit_velocity(
389 velocity_cell,
390 *multiphysics->get_block_solution(PhysicsID::fluid_dynamics),
391 *multiphysics->get_block_previous_solutions(
392 PhysicsID::fluid_dynamics),
393 this->simulation_parameters.ale);
394 }
395 }
396 else
397 {
398 // Check if the post-processed variable needs to be calculated with the
399 // average velocity profile or the fluid solution.
400 if (this->simulation_parameters.initial_condition->type ==
401 Parameters::FluidDynamicsInitialConditionType::
402 average_velocity_profile &&
403 !this->simulation_parameters.multiphysics.fluid_dynamics &&
404 simulation_control->get_current_time() >
405 this->simulation_parameters.post_processing
406 .initial_time_for_average_velocities)
407 {
408 scratch_data.reinit_velocity(
409 velocity_cell,
410 *multiphysics->get_time_average_solution(PhysicsID::fluid_dynamics),
411 *multiphysics->get_previous_solutions(PhysicsID::fluid_dynamics),
412 this->simulation_parameters.ale);
413 }
414 else
415 {
416 scratch_data.reinit_velocity(
417 velocity_cell,
418 *multiphysics->get_solution(PhysicsID::fluid_dynamics),
419 *multiphysics->get_previous_solutions(PhysicsID::fluid_dynamics),
420 this->simulation_parameters.ale);
421 }
422 }
423
424 copy_data.reset();
425
426 for (auto &assembler : this->assemblers)
427 {
428 assembler->assemble_matrix(scratch_data, copy_data);
429 }
430
431 cell->get_dof_indices(copy_data.local_dof_indices);
432}

436VolumeOfFluid<dim>::copy_local_matrix_to_global_matrix(

766template <int dim>
767double
768VolumeOfFluid<dim>::calculate_L2_error()
769{
770 auto mpi_communicator = this->triangulation->get_mpi_communicator();
771
772 FEValues<dim> fe_values_vof(*this->mapping,
773 *this->fe,
774 *this->cell_quadrature,
775 update_values | update_quadrature_points |
776 update_JxW_values);
777
778 const unsigned int n_q_points = this->cell_quadrature->size();
779 std::vector<double> phase_exact_solution(n_q_points);
780 std::vector<double> phase_values(n_q_points);
781
782 auto &exact_solution = simulation_parameters.analytical_solution->phase;
783 exact_solution.set_time(this->simulation_control->get_current_time());
784
785 double l2error = 0.;
786
787 for (const auto &cell : this->dof_handler.active_cell_iterators())
788 {
789 if (cell->is_locally_owned())
790 {
791 fe_values_vof.reinit(cell);
792 fe_values_vof.get_function_values(this->present_solution,
793 phase_values);
794
795 // Get the exact solution at all gauss points
796 exact_solution.value_list(fe_values_vof.get_quadrature_points(),
797 phase_exact_solution);
798
799 for (unsigned int q = 0; q < n_q_points; q++)
800 {
801 double sim = phase_values[q];
802 double exact = phase_exact_solution[q];
803 l2error += (sim - exact) * (sim - exact) * fe_values_vof.JxW(q);
804 }
805 }
806 }
807 l2error = Utilities::MPI::sum(l2error, mpi_communicator);
808 return l2error;
809}

811template <int dim>
812template <typename VectorType>
813std::pair<Tensor<1, dim>, Tensor<1, dim>>
814VolumeOfFluid<dim>::calculate_barycenter(const GlobalVectorType &solution,
815 const VectorType &solution_fd)
816{
817 const MPI_Comm mpi_communicator = this->triangulation->get_mpi_communicator();
818
819 FEValues<dim> fe_values_vof(*this->mapping,
820 *this->fe,
821 *this->cell_quadrature,
822 update_values | update_quadrature_points |
823 update_JxW_values);
824
825 std::shared_ptr<VolumeOfFluidFilterBase> filter =
826 VolumeOfFluidFilterBase::model_cast(
827 this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
828
829 const DoFHandler<dim> *dof_handler_fd =
830 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
831
832 FEValues<dim> fe_values_fd(*this->mapping,
833 dof_handler_fd->get_fe(),
834 *this->cell_quadrature,
835 update_values);
836
837 const unsigned int n_q_points = this->cell_quadrature->size();
838 std::vector<double> phase_values(n_q_points);
839 std::vector<Tensor<1, dim>> velocity_values(n_q_points);
840 std::vector<Point<dim>> quadrature_locations(n_q_points);
841
842 const FEValuesExtractors::Vector velocity(0);
843
844 Tensor<1, dim> barycenter_location;
845 Tensor<1, dim> barycenter_velocity;
846 double volume = 0;
847
848
849 std::map<field, std::vector<double>> fields;
850
851 for (const auto &cell : this->dof_handler.active_cell_iterators())
852 {
853 if (cell->is_locally_owned())
854 {
855 fe_values_vof.reinit(cell);
856 quadrature_locations = fe_values_vof.get_quadrature_points();
857 fe_values_vof.get_function_values(solution, phase_values);
858
859 // Get fluid dynamics active cell iterator
860 typename DoFHandler<dim>::active_cell_iterator cell_fd(
861 &(*(this->triangulation)),
862 cell->level(),
863 cell->index(),
864 dof_handler_fd);
865
866 fe_values_fd.reinit(cell_fd);
867 fe_values_fd[velocity].get_function_values(solution_fd,
868 velocity_values);
869
870 for (unsigned int q = 0; q < n_q_points; q++)
871 {
872 const double JxW = fe_values_vof.JxW(q);
873 const double filtered_phase_value =
874 filter->filter_phase(phase_values[q]);
875
876 volume += (filtered_phase_value)*JxW;
877 barycenter_location +=
878 (filtered_phase_value)*quadrature_locations[q] * JxW;
879 barycenter_velocity +=
880 (filtered_phase_value)*velocity_values[q] * JxW;
881 }
882 }
883 }
884
885 volume = Utilities::MPI::sum(volume, mpi_communicator);
886 barycenter_location =
887 Utilities::MPI::sum(barycenter_location, mpi_communicator) / volume;
888 barycenter_velocity =
889 Utilities::MPI::sum(barycenter_velocity, mpi_communicator) / volume;
890
891 return std::pair<Tensor<1, dim>, Tensor<1, dim>>(barycenter_location,
892 barycenter_velocity);
893}

918template <int dim>
919template <typename VectorType>
920void
921VolumeOfFluid<dim>::calculate_volume_and_mass(
922 const GlobalVectorType &solution,
923 const VectorType &current_solution_fd,
924 const Parameters::FluidIndicator monitored_fluid)
925{
926 const MPI_Comm mpi_communicator = this->triangulation->get_mpi_communicator();
927
928 FEValues<dim> fe_values_vof(*this->mapping,
929 *this->fe,
930 *this->cell_quadrature,
931 update_values | update_JxW_values);
932
933 const DoFHandler<dim> *dof_handler_fd =
934 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
935 QGauss<dim> quadrature_formula(dof_handler_fd->get_fe().degree + 1);
936
937 FEValues<dim> fe_values_fd(*this->mapping,
938 dof_handler_fd->get_fe(),
939 quadrature_formula,
940 update_values);
941
942 const FEValuesExtractors::Scalar pressure(dim);
943
944 const unsigned int n_q_points = this->cell_quadrature->size();
945 std::vector<double> phase_values(n_q_points);
946 std::vector<double> density_0(n_q_points);
947 std::vector<double> density_1(n_q_points);
948 std::vector<double> pressure_values(n_q_points);
949
950 this->volume_monitored = 0.;
951 this->mass_monitored = 0.;
952
953 // Physical properties
954 const auto density_models =
955 this->simulation_parameters.physical_properties_manager
956 .get_density_vector();
957 std::map<field, std::vector<double>> fields;
958 fields.insert(
959 std::pair<field, std::vector<double>>(field::pressure, n_q_points));
960
961 for (const auto &cell_vof : this->dof_handler.active_cell_iterators())
962 {
963 if (cell_vof->is_locally_owned())
964 {
965 fe_values_vof.reinit(cell_vof);
966 fe_values_vof.get_function_values(solution, phase_values);
967
968 if (this->simulation_parameters.physical_properties_manager
969 .field_is_required(field::pressure))
970 {
971 // Get fluid dynamics active cell iterator
972 typename DoFHandler<dim>::active_cell_iterator cell_fd(
973 &(*(this->triangulation)),
974 cell_vof->level(),
975 cell_vof->index(),
976 dof_handler_fd);
977
978 fe_values_fd.reinit(cell_fd);
979 fe_values_fd[pressure].get_function_values(current_solution_fd,
980 pressure_values);
981 set_field_vector(field::pressure, pressure_values, fields);
982 }
983 // Calculate physical properties for the cell
984 density_models[0]->vector_value(fields, density_0);
985 density_models[1]->vector_value(fields, density_1);
986
987 for (unsigned int q = 0; q < n_q_points; q++)
988 {
989 switch (monitored_fluid)
990 {
991 case Parameters::FluidIndicator::fluid0:
992 {
993 this->volume_monitored +=
994 fe_values_vof.JxW(q) * (1 - phase_values[q]);
995 this->mass_monitored += fe_values_vof.JxW(q) *
996 (1 - phase_values[q]) *
997 density_0[q];
998 break;
999 }
1000 case Parameters::FluidIndicator::fluid1:
1001 {
1002 this->volume_monitored +=
1003 fe_values_vof.JxW(q) * phase_values[q];
1004 this->mass_monitored +=
1005 fe_values_vof.JxW(q) * phase_values[q] * density_1[q];
1006 break;
1007 }
1008 default:
1009 throw std::runtime_error(
1010 "Unsupported number of fluids (>2)");
1011 }
1012 }
1013 }
1014 }
1015
1016 this->volume_monitored =
1017 Utilities::MPI::sum(this->volume_monitored, mpi_communicator);
1018 this->mass_monitored =
1019 Utilities::MPI::sum(this->mass_monitored, mpi_communicator);
1020}

1707template <int dim>
1708double
1709VolumeOfFluid<dim>::find_sharpening_threshold()
1710{
1711 // Sharpening threshold (st) search range extrema
1712 double st_min =
1713 0.5 - this->simulation_parameters.multiphysics.vof_parameters
1714 .regularization_method.sharpening.threshold_max_deviation;
1715 double st_max =
1716 0.5 + this->simulation_parameters.multiphysics.vof_parameters
1717 .regularization_method.sharpening.threshold_max_deviation;
1718
1719 // Useful definitions for readability
1720 const double mass_deviation_tol =
1721 this->simulation_parameters.multiphysics.vof_parameters
1722 .regularization_method.sharpening.tolerance *
1723 this->mass_first_iteration;
1724 const unsigned int max_iterations =
1725 this->simulation_parameters.multiphysics.vof_parameters
1726 .regularization_method.sharpening.max_iterations;
1727
1728 const Parameters::FluidIndicator monitored_fluid =
1729 this->simulation_parameters.multiphysics.vof_parameters
1730 .regularization_method.sharpening.monitored_fluid;
1731
1732 unsigned int nb_search_ite = 0;
1733 // Local variable for the tested sharpening_threshold values
1734
1735 double mass_deviation_min = calculate_mass_deviation(monitored_fluid, st_min);
1736 double mass_deviation_max = calculate_mass_deviation(monitored_fluid, st_max);
1737 double mass_deviation_avg = 0.;
1738 double st_avg = 0.;
1739
1740
1741 // Bissection algorithm to calculate an interface sharpening value that would
1742 // ensure mass conservation of the monitored phase (do-while loop, see
1743 // condition below)
1744 do
1745 {
1746 nb_search_ite++;
1747 // Calculate middle point value
1748 st_avg = (st_min + st_max) / 2.;
1749
1750 mass_deviation_avg = calculate_mass_deviation(monitored_fluid, st_avg);
1751
1752 if (this->simulation_parameters.multiphysics.vof_parameters
1753 .regularization_method.verbosity != Parameters::Verbosity::quiet)
1754 {
1755 this->pcout
1756 << " ... step " << nb_search_ite
1757 << " of the search algorithm, min, avg, max mass deviation is : "
1758 << mass_deviation_min << " " << mass_deviation_avg << " "
1759 << mass_deviation_max << std::endl;
1760 }
1761
1762 if (mass_deviation_avg * mass_deviation_min < 0)
1763 {
1764 st_max = st_avg;
1765 mass_deviation_max = mass_deviation_avg;
1766 }
1767 else
1768 {
1769 st_min = st_avg;
1770 mass_deviation_min = mass_deviation_avg;
1771 }
1772 }
1773 while (std::abs(mass_deviation_avg) > mass_deviation_tol &&
1774 nb_search_ite < max_iterations);
1775
1776 // Take minimum deviation in between the two endpoints of the last
1777 // interval searched, if out of the do-while loop because max_iterations is
1778 // reached
1779 if (std::abs(mass_deviation_avg) > mass_deviation_tol)
1780 {
1781 double mass_deviation_endpoint = 0.;
1782 double st_tested = 0;
1783 if (st_min == st_avg)
1784 st_tested = st_max;
1785 else if (st_max == st_avg)
1786 st_tested = st_min;
1787
1788 mass_deviation_endpoint =
1789 calculate_mass_deviation(monitored_fluid, st_tested);
1790
1791 // Retake st_ave value if mass deviation is not lowered at endpoint
1792 // values
1793 if (std::abs(mass_deviation_endpoint) > std::abs(mass_deviation_avg))
1794 {
1795 st_tested = st_avg;
1796 }
1797
1798 // Output message
1799 if (std::abs(mass_deviation_endpoint) > mass_deviation_tol)
1800 {
1801 this->pcout
1802 << " WARNING: Maximum number of iterations (" << nb_search_ite
1803 << ") reached in the " << std::endl
1804 << " adaptive sharpening threshold algorithm, remaining error"
1805 << std::endl
1806 << " on mass conservation is: "
1807 << (this->mass_monitored - this->mass_first_iteration) /
1808 this->mass_first_iteration
1809 << std::endl
1810 << " Consider increasing the sharpening threshold range or the "
1811 << std::endl
1812 << " number of iterations to reach the mass conservation tolerance."
1813 << std::endl;
1814 }
1815 }
1816
1817 // Output message that mass conservation condition is reached
1818 if (this->simulation_parameters.multiphysics.vof_parameters
1819 .regularization_method.verbosity != Parameters::Verbosity::quiet)
1820 {
1821 this->pcout << " ... search algorithm took : " << nb_search_ite
1822 << " step(s) " << std::endl
1823 << " ... error on mass conservation reached: "
1824 << (this->mass_monitored - this->mass_first_iteration) /
1825 this->mass_first_iteration
1826 << std::endl;
1827 }
1828
1829 return st_avg;
1830}

1906template <int dim>
1907void
1908VolumeOfFluid<dim>::assemble_projection_phase_fraction(
1909 GlobalVectorType &solution)
1910{
1911 // Get fe values of VOF phase fraction
1912 FEValues<dim> fe_values_phase_fraction(*this->mapping,
1913 *this->fe,
1914 *this->cell_quadrature,
1915 update_values | update_JxW_values |
1916 update_gradients);
1917
1918 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
1919
1920 const unsigned int n_q_points = this->cell_quadrature->size();
1921 FullMatrix<double> local_matrix_phase_fraction(dofs_per_cell, dofs_per_cell);
1922 Vector<double> local_rhs_phase_fraction(dofs_per_cell);
1923 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
1924
1925 std::vector<double> phi_phase_fraction(dofs_per_cell);
1926 std::vector<Tensor<1, dim>> phi_phase_fraction_gradient(dofs_per_cell);
1927
1928 std::vector<double> phase_values(n_q_points);
1929
1930 // Reinitialize system matrix and rhs for the projected phase fraction
1931 system_rhs_phase_fraction = 0;
1932 system_matrix_phase_fraction = 0;
1933
1934 double h;
1935 double cell_measure;
1936
1937 const double phase_fraction_diffusion_factor =
1938 this->simulation_parameters.initial_condition
1939 ->projection_step_diffusion_factor;
1940
1941 for (const auto &cell : this->dof_handler.active_cell_iterators())
1942 {
1943 if (cell->is_locally_owned())
1944 {
1945 fe_values_phase_fraction.reinit(cell);
1946
1947 // Compute cell diameter
1948 cell_measure = compute_cell_measure_with_JxW(
1949 fe_values_phase_fraction.get_JxW_values());
1950 h = compute_cell_diameter<dim>(cell_measure, fe->degree);
1951
1952 local_matrix_phase_fraction = 0;
1953 local_rhs_phase_fraction = 0;
1954
1955 // Get phase fraction values
1956 fe_values_phase_fraction.get_function_values(solution, phase_values);
1957
1958 double color_function = 0.0;
1959 for (unsigned int q = 0; q < n_q_points; ++q)
1960 {
1961 color_function +=
1962 phase_values[q] * fe_values_phase_fraction.JxW(q);
1963 }
1964
1965 color_function /= cell_measure;
1966
1967 for (unsigned int q = 0; q < n_q_points; ++q)
1968 {
1969 for (unsigned int k = 0; k < dofs_per_cell; ++k)
1970 {
1971 phi_phase_fraction[k] =
1972 fe_values_phase_fraction.shape_value(k, q);
1973 phi_phase_fraction_gradient[k] =
1974 fe_values_phase_fraction.shape_grad(k, q);
1975 }
1976
1977 for (unsigned int i = 0; i < dofs_per_cell; ++i)
1978 {
1979 // Matrix assembly
1980 for (unsigned int j = 0; j < dofs_per_cell; ++j)
1981 {
1982 local_matrix_phase_fraction(i, j) +=
1983 (phi_phase_fraction[j] * phi_phase_fraction[i] +
1984 phase_fraction_diffusion_factor * h * h *
1985 scalar_product(phi_phase_fraction_gradient[i],
1986 phi_phase_fraction_gradient[j])) *
1987 fe_values_phase_fraction.JxW(q);
1988 }
1989
1990 // rhs
1991 local_rhs_phase_fraction(i) +=
1992 phi_phase_fraction[i] * color_function *
1993 fe_values_phase_fraction.JxW(q);
1994 }
1995 }
1996
1997 cell->get_dof_indices(local_dof_indices);
1998 this->nonzero_constraints.distribute_local_to_global(
1999 local_matrix_phase_fraction,
2000 local_rhs_phase_fraction,
2001 local_dof_indices,
2002 system_matrix_phase_fraction,
2003 system_rhs_phase_fraction);
2004 }
2005 }

2662 // on the problem itself and needs to be chosen large enough (for example,

2666 GlobalVectorType lambda(this->locally_owned_dofs,
2667 this->triangulation->get_mpi_communicator());
2668

2670
2671 complete_system_matrix_phase_fraction.residual(lambda,
2672 nodal_phase_fraction_owned,
2673 system_rhs_phase_fraction);
2674
2675 this->bounding_constraints.clear();
2676
2677 std::vector<bool> dof_touched(this->dof_handler.n_dofs(), false);
2678
2679 for (const auto &cell : this->dof_handler.active_cell_iterators())

2693 nodal_phase_fraction_owned(dof_index);
2694 if (lambda(dof_index) +
2695 penalty_parameter *
2696 mass_matrix_phase_fraction(dof_index, dof_index) *

2698 0)

2701 this->bounding_constraints.set_inhomogeneity(
2702 dof_index, this->phase_upper_bound);

2704 this->phase_upper_bound;
2705 lambda(dof_index) = 0;
2706 }
2707 else if (lambda(dof_index) +
2708 penalty_parameter *
2709 mass_matrix_phase_fraction(dof_index,

2711 (solution_value - this->phase_lower_bound) <
2712 0)
2713 {
2714 this->bounding_constraints.add_line(dof_index);
2715 this->bounding_constraints.set_inhomogeneity(

2718 this->phase_lower_bound;
2719 lambda(dof_index) = 0;

2730template <int dim>
2731void
2732VolumeOfFluid<dim>::assemble_L2_projection_interface_sharpening(
2733 GlobalVectorType &solution,
2734 const double sharpening_threshold)
2735{
2736 const double interface_sharpness =
2737 this->simulation_parameters.multiphysics.vof_parameters
2738 .regularization_method.sharpening.interface_sharpness;
2739
2740 FEValues<dim> fe_values_vof(*this->mapping,
2741 *this->fe,
2742 *this->cell_quadrature,
2743 update_values | update_JxW_values);
2744
2745 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
2746 const unsigned int n_q_points = this->cell_quadrature->size();
2747 FullMatrix<double> local_matrix_phase_fraction(dofs_per_cell, dofs_per_cell);
2748 Vector<double> local_rhs_phase_fraction(dofs_per_cell);
2749 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
2750 std::vector<double> phi_phase(dofs_per_cell);
2751
2752 std::vector<double> phase_values(n_q_points);
2753
2754 system_rhs_phase_fraction = 0;
2755 system_matrix_phase_fraction = 0;
2756
2757 for (const auto &cell : this->dof_handler.active_cell_iterators())
2758 {
2759 if (cell->is_locally_owned())
2760 {
2761 fe_values_vof.reinit(cell);
2762
2763 local_matrix_phase_fraction = 0;
2764 local_rhs_phase_fraction = 0;
2765
2766 fe_values_vof.get_function_values(solution, phase_values);
2767
2768 for (unsigned int q = 0; q < n_q_points; ++q)
2769 {
2770 auto phase_value = phase_values[q];
2771 // phase_value = std::min(std::max(phase_value, 0.), 1.);
2772
2773 for (unsigned int k = 0; k < dofs_per_cell; ++k)
2774 {
2775 phi_phase[k] = fe_values_vof.shape_value(k, q);
2776 }
2777 for (unsigned int i = 0; i < dofs_per_cell; ++i)
2778 {
2779 // Matrix assembly
2780 for (unsigned int j = 0; j < dofs_per_cell; ++j)
2781 {
2782 local_matrix_phase_fraction(i, j) +=
2783 (phi_phase[j] * phi_phase[i]) * fe_values_vof.JxW(q);
2784 }
2785
2786 // $$ (if 0 <= \phi <= c) {\Phi = c ^ (1 - \alpha) * (\phi
2787 // ^ \alpha)}$$
2788 // $$ (if c < \phi <= 1) {\Phi = 1 - (1 - c) ^ (1 -
2789 // \alpha)
2790 // * (1 - \phi) ^ \alpha}
2791 if (phase_value >= 0 && phase_value <= sharpening_threshold)
2792 local_rhs_phase_fraction(i) +=
2793 std::pow(sharpening_threshold,
2794 (1. - interface_sharpness)) *
2795 std::pow(phase_value, interface_sharpness) *
2796 phi_phase[i] * fe_values_vof.JxW(q);
2797 else
2798 {
2799 local_rhs_phase_fraction(i) +=
2800 (1 -
2801 std::pow((1. - sharpening_threshold),
2802 (1. - interface_sharpness)) *
2803 std::pow((1. - phase_value), interface_sharpness)) *
2804 phi_phase[i] * fe_values_vof.JxW(q);
2805 }
2806 }
2807 }
2808
2809 cell->get_dof_indices(local_dof_indices);
2810 this->nonzero_constraints.distribute_local_to_global(
2811 local_matrix_phase_fraction,
2812 local_rhs_phase_fraction,
2813 local_dof_indices,
2814 system_matrix_phase_fraction,
2815 system_rhs_phase_fraction);
2816 }
2817 }
2818
2819 system_matrix_phase_fraction.compress(VectorOperation::add);
2820 system_rhs_phase_fraction.compress(VectorOperation::add);
2821}
