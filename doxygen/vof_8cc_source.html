32template <int dim>
33void
34VolumeOfFluid<dim>::setup_assemblers()
35{
36 this->assemblers.clear();
37
38 // Time-stepping schemes
39 if (is_bdf(this->simulation_control->get_assembly_method()))
40 {
41 this->assemblers.emplace_back(
42 std::make_shared<VOFAssemblerBDF<dim>>(this->simulation_control));
43 }
44
45 // Core assembler
46 this->assemblers.emplace_back(std::make_shared<VOFAssemblerCore<dim>>(
47 this->simulation_control,
48 this->simulation_parameters.fem_parameters,
49 this->simulation_parameters.multiphysics.vof_parameters));
50
51 // DCDD shock-capturing assembler
52 if (this->simulation_parameters.stabilization.vof_dcdd_stabilization)
53 this->assemblers.emplace_back(
54 std::make_shared<VOFAssemblerDCDDStabilization<dim>>(
55 this->simulation_control));
56}

92template <int dim>
93void
94VolumeOfFluid<dim>::assemble_local_system_matrix(
95 const typename DoFHandler<dim>::active_cell_iterator &cell,
96 VOFScratchData<dim> &scratch_data,
97 StabilizedMethodsCopyData &copy_data)
98{
99 copy_data.cell_is_local = cell->is_locally_owned();
100 if (!cell->is_locally_owned())
101 return;
102
103 scratch_data.reinit(cell, this->evaluation_point, this->previous_solutions);
104
105 const DoFHandler<dim> *dof_handler_fd =
106 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
107
108 typename DoFHandler<dim>::active_cell_iterator velocity_cell(
109 &(*this->triangulation), cell->level(), cell->index(), dof_handler_fd);
110
111 if (multiphysics->fluid_dynamics_is_block())
112 {
113 // Check if the post processed variable needs to be calculated with the
114 // average velocity profile or the fluid solution.
115 if (this->simulation_parameters.initial_condition->type ==
116 Parameters::InitialConditionType::average_velocity_profile &&
117 !this->simulation_parameters.multiphysics.fluid_dynamics &&
118 simulation_control->get_current_time() >
119 this->simulation_parameters.post_processing
120 .initial_time_for_average_velocities)
121 {
122 scratch_data.reinit_velocity(
123 velocity_cell,
124 *multiphysics->get_block_time_average_solution(
125 PhysicsID::fluid_dynamics),
126 *multiphysics->get_block_previous_solutions(
127 PhysicsID::fluid_dynamics),
128 this->simulation_parameters.ale);
129 }
130 else
131 {
132 scratch_data.reinit_velocity(
133 velocity_cell,
134 *multiphysics->get_block_solution(PhysicsID::fluid_dynamics),
135 *multiphysics->get_block_previous_solutions(
136 PhysicsID::fluid_dynamics),
137 this->simulation_parameters.ale);
138 }
139 }
140 else
141 {
142 // Check if the post-processed variable needs to be calculated with the
143 // average velocity profile or the fluid solution.
144 if (this->simulation_parameters.initial_condition->type ==
145 Parameters::InitialConditionType::average_velocity_profile &&
146 !this->simulation_parameters.multiphysics.fluid_dynamics &&
147 simulation_control->get_current_time() >
148 this->simulation_parameters.post_processing
149 .initial_time_for_average_velocities)
150 {
151 scratch_data.reinit_velocity(
152 velocity_cell,
153 *multiphysics->get_time_average_solution(PhysicsID::fluid_dynamics),
154 *multiphysics->get_previous_solutions(PhysicsID::fluid_dynamics),
155 this->simulation_parameters.ale);
156 }
157 else
158 {
159 scratch_data.reinit_velocity(
160 velocity_cell,
161 *multiphysics->get_solution(PhysicsID::fluid_dynamics),
162 *multiphysics->get_previous_solutions(PhysicsID::fluid_dynamics),
163 this->simulation_parameters.ale);
164 }
165 }
166
167 copy_data.reset();
168
169 for (auto &assembler : this->assemblers)
170 {
171 assembler->assemble_matrix(scratch_data, copy_data);
172 }
173
174 cell->get_dof_indices(copy_data.local_dof_indices);
175}

366template <int dim>
367double
368VolumeOfFluid<dim>::calculate_L2_error()
369{
370 auto mpi_communicator = this->triangulation->get_communicator();
371
372 FEValues<dim> fe_values_vof(*this->mapping,
373 *this->fe,
374 *this->cell_quadrature,
375 update_values | update_quadrature_points |
376 update_JxW_values);
377
378 const unsigned int n_q_points = this->cell_quadrature->size();
379 std::vector<double> phase_exact_solution(n_q_points);
380 std::vector<double> phase_values(n_q_points);
381
382 auto &exact_solution = simulation_parameters.analytical_solution->phase;
383 exact_solution.set_time(this->simulation_control->get_current_time());
384
385 double l2error = 0.;
386
387 for (const auto &cell : this->dof_handler.active_cell_iterators())
388 {
389 if (cell->is_locally_owned())
390 {
391 fe_values_vof.reinit(cell);
392 fe_values_vof.get_function_values(this->present_solution,
393 phase_values);
394
395 // Get the exact solution at all gauss points
396 exact_solution.value_list(fe_values_vof.get_quadrature_points(),
397 phase_exact_solution);
398
399 for (unsigned int q = 0; q < n_q_points; q++)
400 {
401 double sim = phase_values[q];
402 double exact = phase_exact_solution[q];
403 l2error += (sim - exact) * (sim - exact) * fe_values_vof.JxW(q);
404 }
405 }
406 }
407 l2error = Utilities::MPI::sum(l2error, mpi_communicator);
408 return l2error;
409}

411template <int dim>
412template <typename VectorType>
413std::pair<Tensor<1, dim>, Tensor<1, dim>>
414VolumeOfFluid<dim>::calculate_barycenter(const GlobalVectorType &solution,
415 const VectorType &solution_fd)
416{
417 const MPI_Comm mpi_communicator = this->triangulation->get_communicator();
418
419 FEValues<dim> fe_values_vof(*this->mapping,
420 *this->fe,
421 *this->cell_quadrature,
422 update_values | update_quadrature_points |
423 update_JxW_values);
424
425 std::shared_ptr<VolumeOfFluidFilterBase> filter =
426 VolumeOfFluidFilterBase::model_cast(
427 this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
428
429 const DoFHandler<dim> *dof_handler_fd =
430 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
431
432 FEValues<dim> fe_values_fd(*this->mapping,
433 dof_handler_fd->get_fe(),
434 *this->cell_quadrature,
435 update_values);
436
437 const unsigned int n_q_points = this->cell_quadrature->size();
438 std::vector<double> phase_values(n_q_points);
439 std::vector<Tensor<1, dim>> velocity_values(n_q_points);
440 std::vector<Point<dim>> quadrature_locations(n_q_points);
441
442 const FEValuesExtractors::Vector velocity(0);
443
444 Tensor<1, dim> barycenter_location;
445 Tensor<1, dim> barycenter_velocity;
446 double volume = 0;
447
448
449 std::map<field, std::vector<double>> fields;
450
451 for (const auto &cell : this->dof_handler.active_cell_iterators())
452 {
453 if (cell->is_locally_owned())
454 {
455 fe_values_vof.reinit(cell);
456 quadrature_locations = fe_values_vof.get_quadrature_points();
457 fe_values_vof.get_function_values(solution, phase_values);
458
459 // Get fluid dynamics active cell iterator
460 typename DoFHandler<dim>::active_cell_iterator cell_fd(
461 &(*(this->triangulation)),
462 cell->level(),
463 cell->index(),
464 dof_handler_fd);
465
466 fe_values_fd.reinit(cell_fd);
467 fe_values_fd[velocity].get_function_values(solution_fd,
468 velocity_values);
469
470 for (unsigned int q = 0; q < n_q_points; q++)
471 {
472 const double JxW = fe_values_vof.JxW(q);
473 const double filtered_phase_value =
474 filter->filter_phase(phase_values[q]);
475
476 volume += (filtered_phase_value)*JxW;
477 barycenter_location +=
478 (filtered_phase_value)*quadrature_locations[q] * JxW;
479 barycenter_velocity +=
480 (filtered_phase_value)*velocity_values[q] * JxW;
481 }
482 }
483 }
484
485 volume = Utilities::MPI::sum(volume, mpi_communicator);
486 barycenter_location =
487 Utilities::MPI::sum(barycenter_location, mpi_communicator) / volume;
488 barycenter_velocity =
489 Utilities::MPI::sum(barycenter_velocity, mpi_communicator) / volume;
490
491 return std::pair<Tensor<1, dim>, Tensor<1, dim>>(barycenter_location,
492 barycenter_velocity);
493}

518template <int dim>
519template <typename VectorType>
520void
521VolumeOfFluid<dim>::calculate_volume_and_mass(
522 const GlobalVectorType &solution,
523 const VectorType &current_solution_fd,
524 const Parameters::FluidIndicator monitored_fluid)
525{
526 const MPI_Comm mpi_communicator = this->triangulation->get_communicator();
527
528 FEValues<dim> fe_values_vof(*this->mapping,
529 *this->fe,
530 *this->cell_quadrature,
531 update_values | update_JxW_values);
532
533 const DoFHandler<dim> *dof_handler_fd =
534 multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
535 QGauss<dim> quadrature_formula(dof_handler_fd->get_fe().degree + 1);
536
537 FEValues<dim> fe_values_fd(*this->mapping,
538 dof_handler_fd->get_fe(),
539 quadrature_formula,
540 update_values);
541
542 const FEValuesExtractors::Scalar pressure(dim);
543
544 const unsigned int n_q_points = this->cell_quadrature->size();
545 std::vector<double> phase_values(n_q_points);
546 std::vector<double> density_0(n_q_points);
547 std::vector<double> density_1(n_q_points);
548 std::vector<double> pressure_values(n_q_points);
549
550 this->volume_monitored = 0.;
551 this->mass_monitored = 0.;
552
553 // Physical properties
554 const auto density_models =
555 this->simulation_parameters.physical_properties_manager
556 .get_density_vector();
557 std::map<field, std::vector<double>> fields;
558 fields.insert(
559 std::pair<field, std::vector<double>>(field::pressure, n_q_points));
560
561 for (const auto &cell_vof : this->dof_handler.active_cell_iterators())
562 {
563 if (cell_vof->is_locally_owned())
564 {
565 fe_values_vof.reinit(cell_vof);
566 fe_values_vof.get_function_values(solution, phase_values);
567
568 if (this->simulation_parameters.physical_properties_manager
569 .field_is_required(field::pressure))
570 {
571 // Get fluid dynamics active cell iterator
572 typename DoFHandler<dim>::active_cell_iterator cell_fd(
573 &(*(this->triangulation)),
574 cell_vof->level(),
575 cell_vof->index(),
576 dof_handler_fd);
577
578 fe_values_fd.reinit(cell_fd);
579 fe_values_fd[pressure].get_function_values(current_solution_fd,
580 pressure_values);
581 set_field_vector(field::pressure, pressure_values, fields);
582 }
583 // Calculate physical properties for the cell
584 density_models[0]->vector_value(fields, density_0);
585 density_models[1]->vector_value(fields, density_1);
586
587 for (unsigned int q = 0; q < n_q_points; q++)
588 {
589 switch (monitored_fluid)
590 {
591 case Parameters::FluidIndicator::fluid0:
592 {
593 this->volume_monitored +=
594 fe_values_vof.JxW(q) * (1 - phase_values[q]);
595 this->mass_monitored += fe_values_vof.JxW(q) *
596 (1 - phase_values[q]) *
597 density_0[q];
598 break;
599 }
600 case Parameters::FluidIndicator::fluid1:
601 {
602 this->volume_monitored +=
603 fe_values_vof.JxW(q) * phase_values[q];
604 this->mass_monitored +=
605 fe_values_vof.JxW(q) * phase_values[q] * density_1[q];
606 break;
607 }
608 default:
609 throw std::runtime_error(
610 "Unsupported number of fluids (>2)");
611 }
612 }
613 }
614 }
615
616 this->volume_monitored =
617 Utilities::MPI::sum(this->volume_monitored, mpi_communicator);
618 this->mass_monitored =
619 Utilities::MPI::sum(this->mass_monitored, mpi_communicator);
620}

692
693
694 // Calculate physical properties for the cell
695 density_models[0]->vector_value(fields, density_0);
696 density_models[1]->vector_value(fields, density_1);
697
698 for (unsigned int q = 0; q < n_q_points; q++)
699 {
700 switch (monitored_fluid)
701 {
702 case Parameters::FluidIndicator::fluid0:
703 {
704 total_momentum += fe_values_vof.JxW(q) *
705 (1 - phase_values[q]) *
706 velocity_values[q] * density_0[q];
707 break;
708 }
709 case Parameters::FluidIndicator::fluid1:
710 {
711 total_momentum += fe_values_vof.JxW(q) * phase_values[q] *
712 velocity_values[q] * density_1[q];
713 break;
714 }
715 default:
716 throw std::runtime_error(
717 "Unsupported number of fluids (>2)");
718 }
719 }
720 }
721 }
722
723 total_momentum = Utilities::MPI::sum(total_momentum, mpi_communicator);
724
725 return total_momentum;
726}

1223template <int dim>
1224double
1225VolumeOfFluid<dim>::find_sharpening_threshold()
1226{
1227 // Sharpening threshold (st) search range extrema
1228 double st_min = 0.5 - this->simulation_parameters.multiphysics.vof_parameters
1229 .sharpening.threshold_max_deviation;
1230 double st_max = 0.5 + this->simulation_parameters.multiphysics.vof_parameters
1231 .sharpening.threshold_max_deviation;
1232
1233 // Useful definitions for readability
1234 const double mass_deviation_tol = this->simulation_parameters.multiphysics
1235 .vof_parameters.sharpening.tolerance *
1236 this->mass_first_iteration;
1237 const unsigned int max_iterations =
1238 this->simulation_parameters.multiphysics.vof_parameters.sharpening
1239 .max_iterations;
1240
1241 const Parameters::FluidIndicator monitored_fluid =
1242 this->simulation_parameters.multiphysics.vof_parameters.sharpening
1243 .monitored_fluid;
1244
1245 unsigned int nb_search_ite = 0;
1246 // Local variable for the tested sharpening_threshold values
1247
1248 double mass_deviation_min = calculate_mass_deviation(monitored_fluid, st_min);
1249 double mass_deviation_max = calculate_mass_deviation(monitored_fluid, st_max);
1250 double mass_deviation_avg = 0.;
1251 double st_avg = 0.;
1252
1253
1254 // Bissection algorithm to calculate an interface sharpening value that would
1255 // ensure mass conservation of the monitored phase (do-while loop, see
1256 // condition below)
1257 do
1258 {
1259 nb_search_ite++;
1260 // Calculate middle point value
1261 st_avg = (st_min + st_max) / 2.;
1262
1263 mass_deviation_avg = calculate_mass_deviation(monitored_fluid, st_avg);
1264
1265 if (this->simulation_parameters.multiphysics.vof_parameters.sharpening
1266 .verbosity != Parameters::Verbosity::quiet)
1267 {
1268 this->pcout
1269 << " ... step " << nb_search_ite
1270 << " of the search algorithm, min, avg, max mass deviation is : "
1271 << mass_deviation_min << " " << mass_deviation_avg << " "
1272 << mass_deviation_max << std::endl;
1273 }
1274
1275 if (mass_deviation_avg * mass_deviation_min < 0)
1276 {
1277 st_max = st_avg;
1278 mass_deviation_max = mass_deviation_avg;
1279 }
1280 else
1281 {
1282 st_min = st_avg;
1283 mass_deviation_min = mass_deviation_avg;
1284 }
1285 }
1286 while (std::abs(mass_deviation_avg) > mass_deviation_tol &&
1287 nb_search_ite < max_iterations);
1288
1289 // Take minimum deviation in between the two endpoints of the last
1290 // interval searched, if out of the do-while loop because max_iterations is
1291 // reached
1292 if (std::abs(mass_deviation_avg) > mass_deviation_tol)
1293 {
1294 double mass_deviation_endpoint = 0.;
1295 double st_tested = 0;
1296 if (st_min == st_avg)
1297 st_tested = st_max;
1298 else if (st_max == st_avg)
1299 st_tested = st_min;
1300
1301 mass_deviation_endpoint =
1302 calculate_mass_deviation(monitored_fluid, st_tested);
1303
1304 // Retake st_ave value if mass deviation is not lowered at endpoint
1305 // values
1306 if (std::abs(mass_deviation_endpoint) > std::abs(mass_deviation_avg))
1307 {
1308 st_tested = st_avg;
1309 }
1310
1311 // Output message
1312 if (std::abs(mass_deviation_endpoint) > mass_deviation_tol)
1313 {
1314 this->pcout
1315 << " WARNING: Maximum number of iterations (" << nb_search_ite
1316 << ") reached in the " << std::endl
1317 << " adaptive sharpening threshold algorithm, remaining error"
1318 << std::endl
1319 << " on mass conservation is: "
1320 << (this->mass_monitored - this->mass_first_iteration) /
1321 this->mass_first_iteration
1322 << std::endl
1323 << " Consider increasing the sharpening threshold range or the "
1324 << std::endl
1325 << " number of iterations to reach the mass conservation tolerance."
1326 << std::endl;
1327 }
1328 }
1329
1330 // Output message that mass conservation condition is reached
1331 if (this->simulation_parameters.multiphysics.vof_parameters.sharpening
1332 .verbosity != Parameters::Verbosity::quiet)
1333 {
1334 this->pcout << " ... search algorithm took : " << nb_search_ite
1335 << " step(s) " << std::endl
1336 << " ... error on mass conservation reached: "
1337 << (this->mass_monitored - this->mass_first_iteration) /
1338 this->mass_first_iteration
1339 << std::endl;
1340 }
1341
1342 return st_avg;
1343}

1346template <int dim>
1347double
1348VolumeOfFluid<dim>::calculate_mass_deviation(
1349 const Parameters::FluidIndicator monitored_fluid,
1350 const double sharpening_threshold)
1351{
1352 // Copy present solution VOF
1353 evaluation_point = this->present_solution;
1354
1355 // Sharpen interface using the tested threshold value
1356 sharpen_interface(evaluation_point, sharpening_threshold, false);
1357
1358 // Calculate mass of the monitored phase
1359 calculate_volume_and_mass(evaluation_point,
1360 *multiphysics->get_solution(
1361 PhysicsID::fluid_dynamics),
1362 monitored_fluid);
1363

1365 double mass_deviation = (this->mass_monitored - this->mass_first_iteration) /
1366 this->mass_first_iteration;

1371template <int dim>
1372void
1373VolumeOfFluid<dim>::sharpen_interface(GlobalVectorType &solution,
1374 const double sharpening_threshold,
1375 const bool sharpen_previous_solutions)
1376{
1377 // Limit the phase fractions between 0 and 1
1378 update_solution_and_constraints(solution);
1379 if (sharpen_previous_solutions)
1380 {
1381 for (auto &previous_solution : previous_solutions)
1382 update_solution_and_constraints(previous_solution);
1383 }
1384
1385 // Assemble matrix and solve the system for interface sharpening
1386 assemble_L2_projection_interface_sharpening(solution, sharpening_threshold);
1387 solve_interface_sharpening(solution);
1388
1389 if (sharpen_previous_solutions)
1390 {
1391 for (auto &previous_solution : previous_solutions)
1392 {
1393 assemble_L2_projection_interface_sharpening(previous_solution,
1394 sharpening_threshold);
1395 solve_interface_sharpening(previous_solution);
1396 }
1397 }
1398
1399 // Re limit the phase fractions between 0 and 1 after interface
1400 // sharpening
1401 update_solution_and_constraints(solution);
1402 if (sharpen_previous_solutions)
1403 {
1404 for (auto &previous_solution : previous_solutions)
1405 update_solution_and_constraints(previous_solution);
1406 }

1419template <int dim>
1420void
1421VolumeOfFluid<dim>::assemble_projection_phase_fraction(
1422 GlobalVectorType &solution)
1423{
1424 // Get fe values of VOF phase fraction
1425 FEValues<dim> fe_values_phase_fraction(*this->mapping,
1426 *this->fe,
1427 *this->cell_quadrature,
1428 update_values | update_JxW_values |
1429 update_gradients);
1430
1431 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
1432
1433 const unsigned int n_q_points = this->cell_quadrature->size();
1434 FullMatrix<double> local_matrix_phase_fraction(dofs_per_cell, dofs_per_cell);
1435 Vector<double> local_rhs_phase_fraction(dofs_per_cell);
1436 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
1437
1438 std::vector<double> phi_phase_fraction(dofs_per_cell);
1439 std::vector<Tensor<1, dim>> phi_phase_fraction_gradient(dofs_per_cell);
1440
1441 std::vector<double> phase_values(n_q_points);
1442
1443 // Reinitialize system matrix and rhs for the projected phase fraction
1444 system_rhs_phase_fraction = 0;
1445 system_matrix_phase_fraction = 0;
1446
1447 double h;
1448 double cell_measure;
1449
1450 const double phase_fraction_diffusion_factor =
1451 this->simulation_parameters.initial_condition
1452 ->projection_step_diffusion_factor;
1453
1454 for (const auto &cell : this->dof_handler.active_cell_iterators())
1455 {
1456 if (cell->is_locally_owned())
1457 {
1458 fe_values_phase_fraction.reinit(cell);
1459
1460 // Compute cell diameter
1461 cell_measure = compute_cell_measure_with_JxW(
1462 fe_values_phase_fraction.get_JxW_values());
1463 h = compute_cell_diameter<dim>(cell_measure, fe->degree);
1464
1465 local_matrix_phase_fraction = 0;
1466 local_rhs_phase_fraction = 0;
1467
1468 // Get phase fraction values
1469 fe_values_phase_fraction.get_function_values(solution, phase_values);
1470
1471 double color_function = 0.0;
1472 for (unsigned int q = 0; q < n_q_points; ++q)
1473 {
1474 color_function +=
1475 phase_values[q] * fe_values_phase_fraction.JxW(q);
1476 }
1477
1478 color_function /= cell_measure;
1479
1480 for (unsigned int q = 0; q < n_q_points; ++q)
1481 {
1482 for (unsigned int k = 0; k < dofs_per_cell; ++k)
1483 {
1484 phi_phase_fraction[k] =
1485 fe_values_phase_fraction.shape_value(k, q);
1486 phi_phase_fraction_gradient[k] =
1487 fe_values_phase_fraction.shape_grad(k, q);
1488 }
1489
1490 for (unsigned int i = 0; i < dofs_per_cell; ++i)
1491 {
1492 // Matrix assembly
1493 for (unsigned int j = 0; j < dofs_per_cell; ++j)
1494 {
1495 local_matrix_phase_fraction(i, j) +=
1496 (phi_phase_fraction[j] * phi_phase_fraction[i] +
1497 phase_fraction_diffusion_factor * h * h *
1498 scalar_product(phi_phase_fraction_gradient[i],
1499 phi_phase_fraction_gradient[j])) *
1500 fe_values_phase_fraction.JxW(q);
1501 }
1502
1503 // rhs
1504 local_rhs_phase_fraction(i) +=
1505 phi_phase_fraction[i] * color_function *
1506 fe_values_phase_fraction.JxW(q);
1507 }
1508 }
1509
1510 cell->get_dof_indices(local_dof_indices);
1511 this->nonzero_constraints.distribute_local_to_global(
1512 local_matrix_phase_fraction,
1513 local_rhs_phase_fraction,
1514 local_dof_indices,
1515 system_matrix_phase_fraction,
1516 system_rhs_phase_fraction);
1517 }
1518 }
1519 system_matrix_phase_fraction.compress(VectorOperation::add);
1520 system_rhs_phase_fraction.compress(VectorOperation::add);
1521}

1530

1534 SolverControl solver_control(

1761template <int dim>
1762void
1763VolumeOfFluid<dim>::setup_dofs()
1764{
1765 verify_consistency_of_boundary_conditions();
1766
1767 auto mpi_communicator = triangulation->get_communicator();
1768
1769 // Setup DoFs for all active subequations
1770 this->subequations->setup_dofs();
1771
1772 this->dof_handler.distribute_dofs(*this->fe);
1773 DoFRenumbering::Cuthill_McKee(this->dof_handler);
1774
1775 this->locally_owned_dofs = this->dof_handler.locally_owned_dofs();
1776
1777 this->locally_relevant_dofs =
1778 DoFTools::extract_locally_relevant_dofs(this->dof_handler);
1779
1780 this->present_solution.reinit(this->locally_owned_dofs,
1781 this->locally_relevant_dofs,
1782 mpi_communicator);
1783
1784 this->filtered_solution.reinit(this->locally_owned_dofs,
1785 this->locally_relevant_dofs,
1786 mpi_communicator);
1787
1788 // Previous solutions for transient schemes
1789 for (auto &solution : this->previous_solutions)
1790 {
1791 solution.reinit(this->locally_owned_dofs,
1792 this->locally_relevant_dofs,
1793 mpi_communicator);
1794 }
1795
1796 this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
1797
1798 this->newton_update.reinit(this->locally_owned_dofs, mpi_communicator);
1799
1800 this->local_evaluation_point.reinit(this->locally_owned_dofs,
1801 mpi_communicator);
1802
1803 define_non_zero_constraints();
1804
1805 // Boundary conditions for Newton correction
1806 define_zero_constraints();
1807
1808 // Sparse matrices initialization
1809 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
1810 DoFTools::make_sparsity_pattern(this->dof_handler,
1811 dsp,
1812 this->nonzero_constraints,
1813 /*keep_constrained_dofs = */ true);
1814
1815 SparsityTools::distribute_sparsity_pattern(dsp,
1816 this->locally_owned_dofs,
1817 mpi_communicator,
1818 this->locally_relevant_dofs);
1819
1820 // Initialization of phase fraction matrices for interface sharpening.
1821 // system_matrix_phase_fraction is used in
1822 // assemble_L2_projection_interface_sharpening for assembling the system for
1823 // sharpening the interface, while complete_system_matrix_phase_fraction is
1824 // used in update_solution_and_constraints to limit the phase fraction
1825 // values between 0 and 1. According to step-41, to limit the phase fractions
1826 // we compute the Lagrange multiplier as the residual of the original linear
1827 // system, given via the variables complete_system_matrix_phase_fraction and
1828 // complete_system_rhs_phase_fraction
1829 system_matrix_phase_fraction.reinit(this->locally_owned_dofs,
1830 this->locally_owned_dofs,
1831 dsp,
1832 mpi_communicator);
1833
1834 complete_system_matrix_phase_fraction.reinit(this->locally_owned_dofs,
1835 this->locally_owned_dofs,
1836 dsp,
1837 mpi_communicator);
1838
1839 complete_system_rhs_phase_fraction.reinit(this->locally_owned_dofs,
1840 mpi_communicator);
1841
1842 // In update_solution_and_constraints (which limits the phase fraction
1843 // between 0 and 1) nodal_phase_fraction_owned copies the solution, then
1844 // limits it, and finally updates (rewrites) the solution.
1845 nodal_phase_fraction_owned.reinit(this->locally_owned_dofs, mpi_communicator);
1846
1847 // Right hand side of the interface sharpening problem (used in
1848 // assemble_L2_projection_interface_sharpening).
1849 system_rhs_phase_fraction.reinit(this->locally_owned_dofs, mpi_communicator);
1850
1851 this->system_matrix.reinit(this->locally_owned_dofs,
1852 this->locally_owned_dofs,
1853 dsp,
1854 mpi_communicator);
1855
1856 this->pcout << " Number of VOF degrees of freedom: "
1857 << this->dof_handler.n_dofs() << std::endl;
1858
1859 // Provide the VOF dof_handler and solution pointers to the
1860 // multiphysics interface
1861 multiphysics->set_dof_handler(PhysicsID::VOF, &this->dof_handler);
1862 multiphysics->set_solution(PhysicsID::VOF, &this->present_solution);
1863 multiphysics->set_filtered_solution(PhysicsID::VOF, &this->filtered_solution);
1864 multiphysics->set_previous_solutions(PhysicsID::VOF,
1865 &this->previous_solutions);
1866
1867
1868 mass_matrix_phase_fraction.reinit(this->locally_owned_dofs,
1869 this->locally_owned_dofs,
1870 dsp,
1871 mpi_communicator);
1872
1873 assemble_mass_matrix(mass_matrix_phase_fraction);
1874}

2047VolumeOfFluid<dim>::solve_linear_system(const bool initial_step,
2048 const bool /*renewed_matrix*/)
2049{

2051
2052 auto mpi_communicator = this->triangulation->get_communicator();
2053
2054 const AffineConstraints<double> &constraints_used =
2055 initial_step ? this->nonzero_constraints : this->zero_constraints;

2058 simulation_parameters.linear_solver.at(PhysicsID::VOF).minimum_residual;
