28template <int dim, typename PropertiesIndex>
29void
30InsertionFile<dim, PropertiesIndex>::insert(
31 Particles::ParticleHandler<dim> &particle_handler,
32 const parallel::distributed::Triangulation<dim> &triangulation,
33 const DEMSolverParameters<dim> &dem_parameters)
34{
35 if (remaining_particles_of_each_type == 0 &&
36 this->current_inserting_particle_type !=
37 dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
38 {
39 remaining_particles_of_each_type =
40 dem_parameters.lagrangian_physical_properties.number.at(
41 ++this->current_inserting_particle_type);
42 }
43
44 if (remaining_particles_of_each_type > 0)
45 {
46 if (this->removing_particles_in_region)
47 {
48 if (this->mark_for_update)
49 {
50 this->find_cells_in_removing_box(triangulation);
51 this->mark_for_update = false;
52 }
53 this->remove_particles_in_box(particle_handler);
54 }
55
56 // Read the input file
57 std::map<std::string, std::vector<double>> particles_data;
58 fill_vectors_from_file(particles_data,
59 insertion_files.at(current_file_id),
60 ";");
61 current_file_id++;
62 current_file_id = current_file_id % number_of_files;
63
64 // Number of particles in the file
65 unsigned int n_total_particles_to_insert = particles_data["p_x"].size();
66
67 // Adjusting the value in case we exceed the maximum number of particle in
68 // the simulation.
69 n_total_particles_to_insert =
70 std::min(remaining_particles_of_each_type, n_total_particles_to_insert);
71
72 // Processor 0 will be the only one inserting particles
73 MPI_Comm communicator = triangulation.get_mpi_communicator();
74 auto this_mpi_process = Utilities::MPI::this_mpi_process(communicator);
75 const unsigned int n_particles_to_insert_this_proc =
76 this_mpi_process == 0 ? n_total_particles_to_insert : 0;
77
78 std::vector<Point<dim>> insertion_points_on_proc_this_step;
79 insertion_points_on_proc_this_step.reserve(
80 n_particles_to_insert_this_proc);
81
82 if (this_mpi_process == 0)
83 {
84 for (unsigned int p = 0; p < n_particles_to_insert_this_proc; ++p)
85 {
86 if constexpr (dim == 2)
87 {
88 insertion_points_on_proc_this_step.emplace_back(Point<dim>(
89 {particles_data["p_x"][p], particles_data["p_y"][p]}));
90 }
91
92 if constexpr (dim == 3)
93 {
94 insertion_points_on_proc_this_step.emplace_back(
95 Point<dim>({particles_data["p_x"][p],
96 particles_data["p_y"][p],
97 particles_data["p_z"][p]}));
98 }
99 }
100 }
101
102 // Obtain global bounding boxes
103 const auto my_bounding_box =
104 GridTools::compute_mesh_predicate_bounding_box(
105 triangulation, IteratorFilters::LocallyOwnedCell());
106 const auto global_bounding_boxes =
107 Utilities::MPI::all_gather(communicator, my_bounding_box);
108
109 // A vector of vectors, which contains all the properties of all particles
110 // about to get inserted
111 std::vector<std::vector<double>> particle_properties;
112
113 // Assign inserted particles properties
114 this->assign_particle_properties_for_file_insertion(
115 dem_parameters,
116 n_particles_to_insert_this_proc,
117 particles_data,
118 particle_properties);
119
120 // Insert the particles using the points and assigned properties
121 particle_handler.insert_global_particles(
122 insertion_points_on_proc_this_step,
123 global_bounding_boxes,
124 particle_properties);
125
126 // Update number of particle remaining to be inserted
127 remaining_particles_of_each_type -= n_total_particles_to_insert;
128
129
130 ConditionalOStream pcout(
131 std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
132 this->print_insertion_info(n_total_particles_to_insert,
133 remaining_particles_of_each_type,
134 this->current_inserting_particle_type,
135 pcout);
136 }
137}

139template <int dim, typename PropertiesIndex>
140void
141InsertionFile<dim, PropertiesIndex>::
142 assign_particle_properties_for_file_insertion(
143 const DEMSolverParameters<dim> &dem_parameters,
144 const unsigned int &inserted_this_step_this_proc,
145 std::map<std::string, std::vector<double>> &particles_data,
146 std::vector<std::vector<double>> &particle_properties)
147{
148 // Clearing and resizing particle_properties
149 particle_properties.reserve(inserted_this_step_this_proc);
150
151 // Getting properties as local parameters
152 auto physical_properties = dem_parameters.lagrangian_physical_properties;
153
154 // A loop is defined over the number of particles which are going to be
155 // inserted at this step
156 for (unsigned int particle_counter = 0;
157 particle_counter < inserted_this_step_this_proc;
158 ++particle_counter)
159 {
160 double type = this->current_inserting_particle_type;
161 double diameter = particles_data["diameters"][particle_counter];
162 double density =
163 physical_properties
164 .density_particle[this->current_inserting_particle_type];
165 double vel_x = particles_data["v_x"][particle_counter];
166 double vel_y = particles_data["v_y"][particle_counter];
167 double vel_z = particles_data["v_z"][particle_counter];
168 double omega_x = particles_data["w_x"][particle_counter];
169 double omega_y = particles_data["w_y"][particle_counter];
170 double omega_z = particles_data["w_z"][particle_counter];
171 double mass = density * 4. / 3. * M_PI *
172 Utilities::fixed_power<3, double>(diameter * 0.5);
173
174 std::vector<double> properties_of_one_particle{
175 type, diameter, mass, vel_x, vel_y, vel_z, omega_x, omega_y, omega_z};
176
177 if constexpr (std::is_same_v<PropertiesIndex,
178 DEM::DEMMPProperties::PropertiesIndex>)
179 {
180 double T = particles_data["T"][particle_counter];
181 double specific_heat =
182 physical_properties
183 .specific_heat_particle[this->current_inserting_particle_type];
184 properties_of_one_particle.push_back(T);
185 properties_of_one_particle.push_back(specific_heat);
186 }
187
188 if constexpr (std::is_same_v<PropertiesIndex,
189 DEM::CFDDEMProperties::PropertiesIndex>)
190 {
191 // Push back all zero variables for the CFD-DEM coupling properties
192 for (unsigned int i = properties_of_one_particle.size();
193 i < PropertiesIndex::n_properties;
194 ++i)
195 properties_of_one_particle.push_back(0.);
196 }
197
198 particle_properties.push_back(properties_of_one_particle);
199 properties_of_one_particle.clear();
200 }
201}
