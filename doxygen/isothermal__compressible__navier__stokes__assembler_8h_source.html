14/**
15 * @brief Class that assembles the core of the isothermal compressible Navier-Stokes equations.
16 * According to the following weak form:
17 * \f$\nabla \cdot (\rho \mathbf{u}) + \rho \mathbf{u} \cdot \nabla \mathbf{u} -
18 * \nabla p - \mu \nabla^2 \mathbf{u} = 0\f$ with a full GLS stabilization
19 * including the laplacian of the test function.
20 *
21 * @tparam dim An integer that denotes the number of spatial dimensions
22 *
23 * @ingroup assemblers
24 */
25
26template <int dim>
27class GLSIsothermalCompressibleNavierStokesAssemblerCore
28 : public NavierStokesAssemblerBase<dim>
29{
30public:
31 GLSIsothermalCompressibleNavierStokesAssemblerCore(
32 const std::shared_ptr<SimulationControl> &simulation_control)
33 : simulation_control(simulation_control)
34 {}
35
36 /**
37 * @brief assemble_matrix Assembles the matrix
38 * @param scratch_data (see base class)
39 * @param copy_data (see base class)
40 */
41 virtual void
42 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
43 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
44
45 /**
46 * @brief assemble_rhs Assembles the rhs
47 * @param scratch_data (see base class)
48 * @param copy_data (see base class)
49 */
50 virtual void
51 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
52 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
53
54 const std::shared_ptr<SimulationControl> simulation_control;
55};

63 * \frac{(\psi \mathbf{p})^{t+\Delta t}-(\psi \mathbf{p})^{t}}{\Delta t} \f$
64 *
65 * @tparam dim An integer that denotes the number of spatial dimensions
66 *
67 * @ingroup assemblers
