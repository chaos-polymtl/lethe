12/**
13 * @brief Abstract class that allows to calculate the
14 * non-newtonian viscosity on each quadrature point and the shear rate
15 * magnitude.
16 */
17class RheologicalModel : public PhysicalPropertyModel
18{
19public:
20 /**
21 * @brief Default constructor
22 */
23 RheologicalModel()
24 {}
25
26 /**
27 * @brief Instanciates and returns a pointer to a RheologicalModel object by
28 * casting it to the proper child class
29 *
30 * @param material_properties Parsed physical properties that will provide
31 * either the model rheological model being used or say it is a
32 * Newtonian flow
33 */
34 static std::shared_ptr<RheologicalModel>
35 model_cast(const Parameters::Material &material_properties);
36
37 /**
38 * @brief Returns the value of the n parameters of the model, if the model
39 * has one.
40 */
41 virtual double
42 get_n() const
43 {
44 return 1.0;
45 }
46
47 /**
48 * @brief Sets a new value to the n parameter, if the models has one.
49 */
50 virtual void
51 set_n(const double &)
52 {}
53
54 /**
55 * @brief Returns the value of the kinematic viscosity, if the model has one.
56 */
57 virtual double
58 get_kinematic_viscosity() const
59 {
60 return 1.0;
61 }
62
63 /**
64 * @brief Returns the kinematic viscosity scale. This is used to get the
65 * kinematic viscosity scale in the problem when it's not constant. This is
66 * useful for weak boundary conditions.
67 */
68 virtual double
69 get_kinematic_viscosity_scale() const
70 {
71 return 1.0;
72 }
73
74 /**
75 * @brief Sets a new kinematic viscosity value, if the models has one.
76 * @param p_kinematic_viscosity The kinematic viscosity value
77 */
78 virtual void
79 set_kinematic_viscosity(const double & /*p_kinematic_viscosity*/)
80 {}
81
82 /**
83 * @brief Sets a new dynamic viscosity value, if the models has a constant
84 * dynamic viscosity value.
85 * @param p_density_ref The density of the fluid at the reference state
86 */
87 virtual void
88 set_dynamic_viscosity(const double & /*p_density_ref*/)
89 {}
90
91 /**
92 * @brief Calculates the dynamic viscosity, if a kinematic viscosity
93 * and a reference density value is specified.
94 * @param p_density_ref The density of the fluid at the reference state
95 * @param field_values Value of the various fields on which the property may
96 * depend.
97 */
98 virtual double
99 get_dynamic_viscosity(const double &p_density_ref,
100 const std::map<field, double> &field_values) const = 0;
101
102 /**
103 * @brief Calculates the vector values of the dynamic viscosity.
104 * @param p_density_ref The density of the fluid at the reference state
105 * @param field_vectors Values of the field on which the dynamic viscosity may
106 * depend on.
107 */
108 virtual void
109 get_dynamic_viscosity_vector(
110 const double &p_density_ref,
111 const std::map<field, std::vector<double>> &field_vectors,
112 std::vector<double> &property_vector) = 0;
113
114 /**
115 * @brief Calculates the kinematic viscosity used in PSPG and SUPG stabilization terms.
116 * @param[in] field_values Value of the various fields on which the property
117 * may depend.
118 */
119 virtual double
120 get_kinematic_viscosity_for_stabilization(
121 const std::map<field, double> &field_values)
122 {
123 return value(field_values);
124 }
125
126 /**
127 * @brief Calculates the vector values of the kinematic viscosity used in PSPG and SUPG stabilization terms.
128 * @param[in] field_vectors Value of the fields on which the property may
129 * depend on.
130 * @param[out] property_vector Vector of computed viscosities.
131 */
132 virtual void
133 get_kinematic_viscosity_for_stabilization_vector(
134 const std::map<field, std::vector<double>> &field_vectors,
135 std::vector<double> &property_vector)
136 {
137 vector_value(field_vectors, property_vector);
138 }
139
140 /**
141 * @brief Calculates the dynamic viscosity used in PSPG and SUPG stabilization terms.
142 * @param[in] p_density_ref The density of the fluid at the reference state
143 * @param[in] field_values Value of the various fields on which the property
144 * may depend.
145 */
146 virtual double
147 get_dynamic_viscosity_for_stabilization(
148 const double &p_density_ref,
149 const std::map<field, double> &field_values)
150 {
151 return get_dynamic_viscosity(p_density_ref, field_values);
152 }
153
154 /**
155 * @brief Calculates the vector values of the dynamic viscosity used in PSPG and SUPG stabilization terms.
156 * @param[in] p_density_ref The density of the fluid at the reference state
157 * @param[in] field_vectors Value of the fields on which the property may
158 * depend on.
159 * @param[out] property_vector Vector of computed viscosities.
160 */
161 virtual void
162 get_dynamic_viscosity_for_stabilization_vector(
163 const double &p_density_ref,
164 const std::map<field, std::vector<double>> &field_vectors,
165 std::vector<double> &property_vector)
166 {
167 get_dynamic_viscosity_vector(p_density_ref, field_vectors, property_vector);
168 }
169
170 /**
171 * @brief is_non_newtonian_rheological_model Returns a boolean indicating if
172 * the model is a non-Newtonian rheological model.
173 * @return Boolean value of if the model corresponds to a non newtonian
174 * rheological model.
175 */
176 bool
177 is_non_newtonian_rheological_model() const
178 {
179 return non_newtonian_rheological_model;
180 }
181
182 /**
183 * @brief Returns a boolean indicating if
184 * the model is a phase change rheological model.
185 * @return Boolean value of if the model corresponds to a phase change
186 * rheological model.
187 */
188 bool
189 is_phase_change_rheological_model() const
190 {
191 return phase_change_rheological_model;
192 }
193
194protected:
195 /// Boolean to indicate if the rheological model is a non-Newtonian model.
196 bool non_newtonian_rheological_model = false;
197
198 /// Boolean to indicate if the rheological model is a phase change model.
199 bool phase_change_rheological_model = false;
200};

370 * viscosity may depend on. For this model, it only depends on the magnitude
371 * of the shear rate tensor
372 *
373 * Source : Morrison, F. A. (2001). No Memory: Generalized Newtonian Fluids.
374 * Understanding Rheology. Raymond F. Boyer Librabry Collection, Oxford
375 * University Press.
376 */
377 double
378 value(const std::map<field, double> &field_values) override;
379
380 /**
381 * @brief vector_value Calculates the vector values of the kinematic
382 * viscosity.
383 * @param field_vectors Values of the field on which the kinematic viscosity
384 * may depend on. The power-law kinematic viscosity depends on the shear rate.
385 */
386 void
387 vector_value(const std::map<field, std::vector<double>> & /*field_vectors*/,
388 std::vector<double> &property_vector) override;
389
390 /**

396 * should be calculated
397 * @return value of the partial derivative of the kinematic viscosity with
398 * respect to the field.
399 */
400
401 double
402 jacobian(const std::map<field, double> & /*field_values*/,
403 field /*id*/) override;
404
405 /**
406 * @brief vector_jacobian Calculate the derivative of the with respect to a
407 * field
408 * @param field_vectors Vector for the values of the fields used to evaluate
409 * the property
410 * @param id Identifier of the field with respect to which a derivative
411 * should be calculated
412 * @param jacobian Vector of the value of the derivative of the kinematic
413 * viscosity with respect to the field id
414 */
415
416 void
417 vector_jacobian(
418 const std::map<field, std::vector<double>> & /*field_vectors*/,
419 const field /*id*/,
420 std::vector<double> &jacobian_vector) override;
421
422 double
423 get_n() const override
424 {
425 return n;
426 }
427
428 void
429 set_n(const double &p_n) override
430 {
431 n = p_n;
432 }
433
434 double
435 get_kinematic_viscosity() const override
436 {
437 return K;
438 }
439
440 double
441 get_kinematic_viscosity_scale() const override
442 {
443 return 1.0 > shear_rate_min ? K * std::pow(1.0, n - 1) :
444 K * std::pow(shear_rate_min, n - 1);
445 }
446
447 void
448 set_kinematic_viscosity(const double &p_kinematic_viscosity) override
449 {
450 K = p_kinematic_viscosity;
451 }
452
453 /**
454 * @brief Calculates the dynamic viscosity, if a kinematic viscosity
455 * and a reference density value is specified.
456 * @param p_density_ref The density of the fluid at the reference state
457 * @param field_values Value of the various fields on which the property may
458 * depend. The power law dynamic viscosity depends on the shear rate
459 * magnitude.
460 */
461 double
462 get_dynamic_viscosity(
463 const double &p_density_ref,
464 const std::map<field, double> &field_values) const override
465 {
466 Assert(field_values.find(field::shear_rate) != field_values.end(),
467 PhysicialPropertyModelFieldUndefined("PowerLaw", "shear_rate"));
468 const double shear_rate_magnitude = field_values.at(field::shear_rate);
469 return calculate_kinematic_viscosity(shear_rate_magnitude) * p_density_ref;
470 }
471
472 /**
473 * @brief Calculates the vector values of the dynamic viscosity.
474 * @param p_density_ref The density of the fluid at the reference state
475 * @param field_vectors Values of the field on which the dynamic viscosity may
476 * depend on.
477 */
478 void
479 get_dynamic_viscosity_vector(
480 const double &p_density_ref,
481 const std::map<field, std::vector<double>> &field_vectors,
482 std::vector<double> &property_vector) override
483 {
484 Assert(field_vectors.find(field::shear_rate) != field_vectors.end(),
485 PhysicialPropertyModelFieldUndefined("PowerLaw", "shear_rate"));
486 const std::vector<double> &shear_rate_magnitude =
487 field_vectors.at(field::shear_rate);
488
489 for (unsigned int i = 0; i < property_vector.size(); ++i)
490 property_vector[i] =
491 calculate_kinematic_viscosity(shear_rate_magnitude[i]) * p_density_ref;
492 }
493
494private:
495 inline double
496 calculate_kinematic_viscosity(const double shear_rate_magnitude) const
497 {
498 return shear_rate_magnitude > shear_rate_min ?
499 K * std::pow(shear_rate_magnitude, n - 1) :
500 K * std::pow(shear_rate_min, n - 1);
501 }
502
503 inline double
504 calculate_derivative(const double shear_rate_magnitude)
505 {
506 return shear_rate_magnitude > shear_rate_min ?
507 (n - 1) * K * std::pow(shear_rate_magnitude, n - 2) :
508 0;
509 }
510
511 double K;
512 double n;
513 const double shear_rate_min;
514};

675/**
676 * @brief Calculates the magnitude of the shear rate tensor given as input
677 *
678 * @param shear_rate A shear rate tensor
679 */
680template <int dim>
681inline double
682calculate_shear_rate_magnitude(const Tensor<2, dim> shear_rate)
683{
684 double shear_rate_magnitude = 0;
685 for (unsigned int i = 0; i < dim; ++i)
686 {
687 for (unsigned int j = 0; j < dim; ++j)
688 {
689 shear_rate_magnitude += (shear_rate[i][j] * shear_rate[j][i]);
690 }
691 }
692 shear_rate_magnitude = sqrt(0.5 * shear_rate_magnitude);
693 return shear_rate_magnitude;
694}
