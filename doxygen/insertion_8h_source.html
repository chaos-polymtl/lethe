39/**
40 * @brief Base interface for classes that carry out the insertion of particles
41 * in the system.
42 * @tparam dim An integer that denotes the number of spatial dimensions.
43 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
44 */
45template <int dim, typename PropertiesIndex>
46class Insertion
47{
48public:
49 /**
50 * @brief Carries out the insertion of particles. This is the base class of
51 * volume_insertion, plane_insertion, list_insertion and file_insertion
52 * classes.
53 *
54 * @param size_distribution_object_container Contains all distribution for each
55 * particle type
56 * @param triangulation Triangulation to access the cells in which the
57 * particles are inserted
58 * @param dem_parameters DEM parameters declared in the .prm file
59 */
60 Insertion(const std::vector<std::shared_ptr<Distribution>>
61 &size_distribution_object_container,
62 const parallel::distributed::Triangulation<dim> &triangulation,
63 const DEMSolverParameters<dim> &dem_parameters);
64 /**
65 * @brief Carries out the destruction of the insertion class.
66 *
67 */
68 virtual ~Insertion()
69 {
70 // This boost signal needs to be disconnected before being destroyed,
71 // otherwise a segfault could occur when the triangulation will be
72 // destroyed.
73 this->change_to_triangulation.disconnect();
74 };
75
76 /**
77 * @brief This function is overridden by the volume_insertion, plane_insertion,
78 * list_insertion and file_insertion classes.
79 *
80 * @param particle_handler The particle handler of particles which are being
81 * inserted
82 * @param triangulation Triangulation to access the cells in which the
83 * particles are inserted
84 * @param dem_parameters DEM parameters declared in the .prm file
85 */
86 virtual void
87 insert(Particles::ParticleHandler<dim> &particle_handler,
88 const parallel::distributed::Triangulation<dim> &triangulation,
89 const DEMSolverParameters<dim> &dem_parameters) = 0;
90
91 /**
92 * @brief Serialize the insertion object to an output archive. Is being used
93 * when checkpointing a simulation. This function is overridden by
94 * volume_insertion, plane_insertion, list_insertion and file_insertion
95 * classes.
96 *
97 * @param ar Output archive where the attributes are stored.
98 *
99 */
100 virtual void
101 serialize(boost::archive::text_oarchive &ar) = 0;
102
103
104 /**
105 * @brief Deserialize an input archive to the insertion object. Is being used
106 * when restarting a simulation. This function is overridden by
107 * volume_insertion, plane_insertion, list_insertion and from_file_insertion
108 * classes.
109 *
110 * @param ar Input archive where the attributes are stored.
111 *
112 */
113 virtual void
114 deserialize(boost::archive::text_iarchive &ar) = 0;
115
116protected:
117 /**
118 * @brief Print information about the particles that have been inserted during an
119 * insertion time step.
120 *
121 * @param inserted_this_step Number of particles that are inserted
122 * at each insertion step.
123 * @param remained_particles Number of remained particles going to be inserted
124 * in future insertions
125 * @param particle_type Inserted particle type
126 * @param pcout Printing in parallel
127 */
128 void
129 print_insertion_info(const unsigned int &inserted_this_step,
130 const unsigned int &remained_particles,
131 const unsigned int &particle_type,
132 const ConditionalOStream &pcout);
133
134 /**
135 * @brief Carries out assigning the properties of inserted particles.
136 *
137 * @param dem_parameters DEM parameters declared in the .prm file
138 * @param inserted_this_step_this_proc Number of particles that are inserted
139 * at each insertion step on each processor. This value can change in the last
140 * insertion step to reach the desired number of particles
141 * @param current_inserting_particle_type Type of inserting particles
142 * @param particle_properties Properties of all inserted particles at each insertion step
143 */
144 void
145 assign_particle_properties(
146 const DEMSolverParameters<dim> &dem_parameters,
147 const unsigned int &inserted_this_step_this_proc,
148 const unsigned int &current_inserting_particle_type,
149 const std::vector<Point<dim>> &insertion_points,
150 std::vector<std::vector<double>> &particle_properties);
151
152 /**
153 * @brief Carries out finding the maximum number of inserted particles based on the
154 * insertion box size. If the requested number of particles for insertion in
155 * each insertion step is larger than this maximum, it is limited to this
156 * value and a warning is printed.
157 *
158 * @param dem_parameters DEM parameters declared in the .prm file
159 * @param pcout Printing in parallel
160 */
161 void
162 calculate_insertion_domain_maximum_particle_number(
163 const DEMSolverParameters<dim> &dem_parameters,
164 const ConditionalOStream &pcout);
165
166 /**
167 * @brief Find every cell that are completely and partially inside de the
168 * clearing box. For the cell completely inside the box, all the particles
169 * will be deleted. For those partially inside, a fine verification is
170 * required.
171 *
172 * @param triangulation Triangulation to access the cells in which the
173 * particles deleted.
174 */
175 void
176 find_cells_in_removing_box(
177 const parallel::distributed::Triangulation<dim> &triangulation);
178
179 /**
180 * @brief Remove every particle located inside a predefined zone. Currently,
181 * the zone is defined by a box.
182 *
183 * @param particle_handler The particle handler of particles which are being
184 * removed.
185 */
186 void
187 remove_particles_in_box(Particles::ParticleHandler<dim> &particle_handler);
188
189 // Number of particles inserted at each insertion time step. This value can
190 // change in the last insertion step to reach the desired number of particles
191 unsigned int inserted_this_step;
192
193 // Number of insertion points in the x, y and z directions
194 std::vector<int> number_of_particles_directions;
195
196 // Minimum and maximum number of inserted particles based on the insertion box
197 // size and the direction order (it means that axis 0 is not necessarily x
198 // etc...) It depends of the order of the insertion direction.
199 std::vector<double> axis_min, axis_max;
200
201 // Maximum particle diameter
202 double maximum_diameter;
203
204 // Inserted number of particles at this step on this processor
205 unsigned int inserted_this_step_this_proc;
206
207 // A distribution object that carries out the attribution of diameter to every
208 // particle during an insertion time step
209 std::vector<std::shared_ptr<Distribution>> distributions_objects;
210
211 // Clearing bool
212 const bool removing_particles_in_region;
213
214 // Clearing box coordinates
215 Point<dim> p_min, p_max;
216
217 // Cell iterator containers for clearing
218 std::set<typename Triangulation<dim>::active_cell_iterator> in_removal_box;
219 std::set<typename Triangulation<dim>::active_cell_iterator>
220 edge_of_removal_box;
221
222 // For when the triangulation has changed (i.e. when load balancing)
223 bool mark_for_update;
224 boost::signals2::connection change_to_triangulation;
225
226private:
227 // Stores particles diameters
228 std::vector<double> particle_sizes;
229};
