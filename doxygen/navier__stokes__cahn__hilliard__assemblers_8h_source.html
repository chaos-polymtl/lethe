15/**
16 * @brief Class that assembles the core of the Navier-Stokes equation with
17 * free surface using Cahn-Hilliard modeling.
18 * According to the following weak form:
19 * \f$ \nabla \cdot \mathbf{u} + \rho \mathbf{u} \cdot \nabla
20 * \mathbf{u} + \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
21 * (\nabla \mathbf{u})^T)=0 \f$ with an SUPG and PSPG stabilization
22 *
23 * @tparam dim An integer that denotes the number of spatial dimensions
24 *
25 * @ingroup assemblers
26 */
27template <int dim>
28class GLSNavierStokesCahnHilliardAssemblerCore
29 : public NavierStokesAssemblerBase<dim>
30{
31public:
32 GLSNavierStokesCahnHilliardAssemblerCore(
33 const std::shared_ptr<SimulationControl> &simulation_control,
34 const SimulationParameters<dim> &nsparam)
35 : simulation_control(simulation_control)
36 , cahn_hilliard_parameters(nsparam.multiphysics.cahn_hilliard_parameters)
37 {}
38
39 /**
40 * @brief assemble_matrix Assembles the matrix
41 * @param scratch_data (see base class)
42 * @param copy_data (see base class)
43 */
44 virtual void
45 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
46 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
47
48 /**
49 * @brief assemble_rhs Assembles the rhs
50 * @param scratch_data (see base class)
51 * @param copy_data (see base class)
52 */
53 virtual void
54 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
55 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
56
57 const bool SUPG = true;
58
59 const std::shared_ptr<SimulationControl> simulation_control;
60 const Parameters::CahnHilliard cahn_hilliard_parameters;
61};
