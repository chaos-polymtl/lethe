14/**
15 * @brief A pure virtual class that serves as an interface for all
16 * of the assemblers for the Cahn-Hilliard equations.
17 *
18 * @tparam dim Integer that denotes the number of spatial dimensions.
19 *
20 * @ingroup assemblers
21 */
22template <int dim>
23using CahnHilliardAssemblerBase =
24 PhysicsAssemblerBase<CahnHilliardScratchData<dim>, StabilizedMethodsCopyData>;

27/**
28 * @brief Class that assembles the core of the Cahn-Hilliard equation :
29 * \f$ \frac{d \phi}{dt} + (u \cdot \nabla) \phi - \nabla \cdot (M(\phi)\nabla
30 * \eta) = 0 \\ \eta - \frac{\lambda}{\epsilon^2}(\phi^3 - \phi) + \lambda
31 * \nabla^2 \phi + \xi \nabla^2 \eta = 0 \f$
32 *
33 * @tparam dim An integer that denotes the number of spatial dimensions
34 *
35 * @ingroup assemblers
36 */
37template <int dim>
38class CahnHilliardAssemblerCore : public CahnHilliardAssemblerBase<dim>
39{
40public:
41 CahnHilliardAssemblerCore(
42 const std::shared_ptr<SimulationControl> &simulation_control,
43 const Parameters::CahnHilliard cahn_hilliard_parameters,
44 const double epsilon)
45 : simulation_control(simulation_control)
46 , cahn_hilliard_parameters(cahn_hilliard_parameters)
47 , epsilon(epsilon)
48 {}
49
50 /**
51 * @brief assemble_matrix Assembles the matrix
52 * @param scratch_data (see base class)
53 * @param copy_data (see base class)
54 */
55 virtual void
56 assemble_matrix(const CahnHilliardScratchData<dim> &scratch_data,
57 StabilizedMethodsCopyData &copy_data) override;
58
59 /**
60 * @brief assemble_rhs Assembles the rhs
61 * @param scratch_data (see base class)
62 * @param copy_data (see base class)
63 */
64 virtual void
65 assemble_rhs(const CahnHilliardScratchData<dim> &scratch_data,
66 StabilizedMethodsCopyData &copy_data) override;
67
68 const std::shared_ptr<SimulationControl> simulation_control;
69 const Parameters::CahnHilliard cahn_hilliard_parameters;
70 // Epsilon is a coefficient that depends on the mesh size. The thickness of
71 // the interface between the two phases is proportional to epsilon
72 const double epsilon;
73};
