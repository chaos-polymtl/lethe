12template <int dim>
13void
14VOFAlgebraicInterfaceReinitialization<dim>::setup_dofs()
15{
16 // Get MPI communicator
17 auto mpi_communicator = this->triangulation->get_mpi_communicator();
18
19 // Distribute and renumber DoFs
20 this->dof_handler->distribute_dofs(*this->fe);
21 DoFRenumbering::Cuthill_McKee(*this->dof_handler);
22
23 // Get locally owned and relevant DoFs
24 this->locally_owned_dofs = this->dof_handler->locally_owned_dofs();
25 this->locally_relevant_dofs =
26 DoFTools::extract_locally_relevant_dofs(*this->dof_handler);
27
28 // Constraints
29 define_non_zero_constraints();
30 define_zero_constraints();
31
32 // Sparsity pattern
33 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
34 DoFTools::make_sparsity_pattern(*this->dof_handler,
35 dsp,
36 this->nonzero_constraints,
37 false);
38 SparsityTools::distribute_sparsity_pattern(dsp,
39 this->locally_owned_dofs,
40 mpi_communicator,
41 this->locally_relevant_dofs);
42
43 // Reinitialize system matrix and right-hand side (rhs)
44 this->system_matrix.reinit(this->locally_owned_dofs,
45 this->locally_owned_dofs,
46 dsp,
47 mpi_communicator);
48 this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
49
50 // Reinitialize solution vectors
51 this->present_solution->reinit(this->locally_owned_dofs,
52 this->locally_relevant_dofs,
53 mpi_communicator);
54 this->previous_solution.reinit(this->locally_owned_dofs,
55 this->locally_relevant_dofs,
56 mpi_communicator);
57 this->newton_update.reinit(this->locally_owned_dofs, mpi_communicator);
58 this->local_evaluation_point.reinit(this->locally_owned_dofs,
59 mpi_communicator);
60 this->previous_local_evaluation_point.reinit(this->locally_owned_dofs,
61 mpi_communicator);
62 this->evaluation_point = *this->present_solution;
63
64 if (this->subequation_verbosity != Parameters::Verbosity::quiet)
65 {
66 std::string subequation_string =
67 this->subequations_interface.get_subequation_string(
68 this->subequation_id);
69 this->pcout << " Number of " << subequation_string
70 << " degrees of freedom: " << this->dof_handler->n_dofs()
71 << std::endl;
72 }
73
74 // Provide DoFHandler and solutions to the subequations interface
75 this->subequations_interface.set_dof_handler(this->subequation_id,
76 this->dof_handler);
77 this->subequations_interface.set_solution(this->subequation_id,
78 this->present_solution);
79}

82template <int dim>
83void
84VOFAlgebraicInterfaceReinitialization<dim>::define_zero_constraints()
85{
86 this->zero_constraints.clear();
87 this->zero_constraints.reinit(this->locally_owned_dofs,
88 this->locally_relevant_dofs);
89
90 DoFTools::make_hanging_node_constraints(*this->dof_handler,
91 this->zero_constraints);
92
93 for (auto const &[id, type] :
94 this->simulation_parameters.boundary_conditions.type)
95 {
96 if (type == BoundaryConditions::BoundaryType::periodic)
97 {
98 DoFTools::make_periodicity_constraints(
99 *this->dof_handler,
100 id,
101 this->simulation_parameters.boundary_conditions.periodic_neighbor_id
102 .at(id),
103 this->simulation_parameters.boundary_conditions.periodic_direction
104 .at(id),
105 this->zero_constraints);
106 }
107 }
108 for (auto const &[id, type] :
109 this->simulation_parameters.boundary_conditions_vof.type)
110 {
111 if (type == BoundaryConditions::BoundaryType::vof_dirichlet)
112 {
113 VectorTools::interpolate_boundary_values(
114 *this->dof_handler,
115 id,
116 Functions::ZeroFunction<dim>(),
117 this->zero_constraints);
118 }
119 }
120 this->zero_constraints.close();
121}

167template <int dim>
168void
169VOFAlgebraicInterfaceReinitialization<dim>::set_initial_conditions()
170{
171 // Get VOF DoFHandler
172 const DoFHandler<dim> &dof_handler_vof =
173 this->subequations_interface.get_vof_dof_handler();
174
175 // Interpolate VOF solution to algebraic interface reinitialization
176 FETools::interpolate(dof_handler_vof,
177 this->subequations_interface.get_vof_solution(),
178 *this->dof_handler,
179 this->nonzero_constraints,
180 this->newton_update);
181
182 *this->present_solution = this->newton_update;
183 this->previous_solution =
184 *this->present_solution; // We only have 1 previous solution (bdf1)
185 this->previous_local_evaluation_point =
186 this->previous_solution; // For steady-state criterion evaluation
187
188 // Initial condition
189 if (this->simulation_parameters.multiphysics.vof_parameters
190 .regularization_method.algebraic_interface_reinitialization
191 .output_reinitialization_steps)
192 {
193 this->pvdhandler.times_and_names.clear();
194 write_output_results(0);
195 }
196}

199template <int dim>
200void
201VOFAlgebraicInterfaceReinitialization<dim>::assemble_system_matrix()
202{
203 // Reinitialize system matrix
204 this->system_matrix = 0;
205
206 // Get projected VOF phase gradient and curvature DoFHandler
207 const DoFHandler<dim> &dof_handler_vof_phase_fraction_gradient =
208 this->subequations_interface.get_dof_handler(
209 VOFSubequationsID::phase_gradient_projection);
210 const DoFHandler<dim> &dof_handler_curvature =
211 this->subequations_interface.get_dof_handler(
212 VOFSubequationsID::curvature_projection);
213
214 // Initialize FEValues for algebraic interface reinitialization, VOF
215 // phase fraction gradient and curvature projections
216 FEValues<dim> fe_values_algebraic_reinitialization(*this->mapping,
217 *this->fe,
218 *this->cell_quadrature,
219 update_values |
220 update_gradients |
221 update_JxW_values);
222 FEValues<dim> fe_values_phase_gradient_projection(
223 *this->mapping,
224 dof_handler_vof_phase_fraction_gradient.get_fe(),
225 *this->cell_quadrature,
226 update_values);
227 FEValues<dim> fe_values_curvature_projection(*this->mapping,
228 dof_handler_curvature.get_fe(),
229 *this->cell_quadrature,
230 update_values);
231
232 // Initialize size of arrays
233 const double n_q_points =
234 fe_values_algebraic_reinitialization.get_quadrature().size();
235 const double n_dofs_per_cell =
236 fe_values_algebraic_reinitialization.get_fe().n_dofs_per_cell();
237
238 // Initialize local matrix
239 FullMatrix<double> local_matrix(n_dofs_per_cell, n_dofs_per_cell);
240
241 // Initialize local dof indices array
242 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
243
244 // Extractor for phase fraction gradient vector
245 FEValuesExtractors::Vector phase_fraction_gradients(0);
246
247 // Initialize phase fraction, projected phase gradient and projected curvature
248 // solution arrays
249 std::vector<double> present_phase_fraction_values(n_q_points);
250 std::vector<Tensor<1, dim>> present_phase_gradient_projection_values(
251 n_q_points);
252 std::vector<double> present_curvature_values(n_q_points);
253 std::vector<Tensor<1, dim>> present_reinitialized_phase_gradient_values(
254 n_q_points);
255
256 // Initialize shape function arrays
257 std::vector<double> phi(n_dofs_per_cell);
258 std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
259
260 // Compute diffusivity coefficient with the smallest cell size
261 const double h_min = identify_minimum_cell_size(
262 *this->mapping,
263 this->subequations_interface.get_dof_handler(
264 VOFSubequationsID::algebraic_interface_reinitialization),
265 *this->cell_quadrature,
266 this->triangulation->get_mpi_communicator());
267 const double diffusivity_coefficient = compute_diffusivity(h_min);
268
269 // BDF coefficients for pseudo time-stepping
270 const Parameters::SimulationControl::TimeSteppingMethod method =
271 Parameters::SimulationControl::TimeSteppingMethod::bdf1;
272 const Vector<double> bdf_coefficient_vector =
273 calculate_bdf_coefficients(method, this->time_step_vector);
274
275 // Assemble system matrix
276 for (const auto &cell : this->dof_handler->active_cell_iterators())
277 {
278 if (cell->is_locally_owned())
279 {
280 // Reinitialize local matrix
281 local_matrix = 0;
282
283 // Get phase gradient projection and curvature cell iterators
284 typename DoFHandler<dim>::active_cell_iterator
285 vof_phase_gradient_projection_cell(
286 &(*this->triangulation),
287 cell->level(),
288 cell->index(),
289 &dof_handler_vof_phase_fraction_gradient);
290 typename DoFHandler<dim>::active_cell_iterator
291 curvature_projection_cell(&(*this->triangulation),
292 cell->level(),
293 cell->index(),
294 &dof_handler_curvature);
295
296 // Reinitialize FEValues with corresponding cell
297 fe_values_algebraic_reinitialization.reinit(cell);
298 fe_values_phase_gradient_projection.reinit(
299 vof_phase_gradient_projection_cell);
300 fe_values_curvature_projection.reinit(curvature_projection_cell);
301
302 // Get vector of JxW
303 std::vector<double> JxW_vec =
304 fe_values_algebraic_reinitialization.get_JxW_values();
305
306 // Get get present phase fraction, projected VOF phase gradient and
307 // curvature values
308 fe_values_algebraic_reinitialization.get_function_values(
309 this->evaluation_point, present_phase_fraction_values);
310 fe_values_phase_gradient_projection[phase_fraction_gradients]
311 .get_function_values(
312 this->subequations_interface.get_solution(
313 VOFSubequationsID::phase_gradient_projection),
314 present_phase_gradient_projection_values);
315 fe_values_curvature_projection.get_function_values(
316 this->subequations_interface.get_solution(
317 VOFSubequationsID::curvature_projection),
318 present_curvature_values);
319 fe_values_algebraic_reinitialization.get_function_gradients(
320 this->evaluation_point,
321 present_reinitialized_phase_gradient_values);
322
323 // Set tolerance to avoid division by zero
324 const double tolerance = 1e-12;
325
326 // Loop over quadrature points
327 for (unsigned int q = 0; q < n_q_points; ++q)
328 {
329 // Shape functions
330 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
331 {
332 phi[k] =
333 fe_values_algebraic_reinitialization.shape_value(k, q);
334 grad_phi[k] =
335 fe_values_algebraic_reinitialization.shape_grad(k, q);
336 }
337
338 // Extract phase fraction value, projected phase fraction gradient
339 // and curvature
340 const double phase_fraction = present_phase_fraction_values[q];
341 const Tensor<1, dim> projected_vof_phase_gradient =
342 present_phase_gradient_projection_values[q];
343 const double curvature = present_curvature_values[q];
344 const Tensor<1, dim> reinitialized_phase_gradient =
345 present_reinitialized_phase_gradient_values[q];
346
347 // Compute normal vector with projected phase gradient
348 const double projected_vof_phase_gradient_norm =
349 projected_vof_phase_gradient.norm() + tolerance;
350 const Tensor<1, dim> interface_normal =
351 (projected_vof_phase_gradient /
352 projected_vof_phase_gradient_norm);
353
354 // Local matrix assembly
355 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
356 {
357 for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
358 {
359 local_matrix(i, j) +=
360 (
361 // Time-stepping term
362 phi[i] * phi[j] * bdf_coefficient_vector[0] +
363 // Compressive term
364 phi[i] *
365 (scalar_product(grad_phi[j], interface_normal) -
366 2 * phase_fraction *
367 scalar_product(grad_phi[j], interface_normal) -
368 2 * phi[j] *
369 scalar_product(reinitialized_phase_gradient,
370 interface_normal) -
371 (phi[j] - 2 * phase_fraction * phi[j]) *
372 curvature) +
373 // Diffusive term
374 scalar_product(grad_phi[i],
375 (diffusivity_coefficient *
376 scalar_product(grad_phi[j],
377 interface_normal) *
378 interface_normal))) *
379 JxW_vec[q];
380 }
381 }
382 }
383 // Distribute the local contributions to the global system
384 cell->get_dof_indices(local_dof_indices);
385 this->zero_constraints.distribute_local_to_global(
386 local_matrix, local_dof_indices, this->system_matrix);
387 }
388 }
389 this->system_matrix.compress(VectorOperation::add);
390}

393template <int dim>
394void
395VOFAlgebraicInterfaceReinitialization<dim>::assemble_system_rhs()
396{
397 // Reinitialize system right-hand side (rhs)
398 this->system_rhs = 0;
399
400 // Get projected phase gradient, curvature and reinitialized phase gradient
401 // DoFHandler
402 const DoFHandler<dim> &dof_handler_vof_phase_fraction_gradient =
403 this->subequations_interface.get_dof_handler(
404 VOFSubequationsID::phase_gradient_projection);
405 const DoFHandler<dim> &dof_handler_curvature =
406 this->subequations_interface.get_dof_handler(
407 VOFSubequationsID::curvature_projection);
408
409 // Initialize FEValues for algebraic interface reinitialization, VOF phase
410 // fraction gradient and curvature projections
411 FEValues<dim> fe_values_algebraic_reinitialization(*this->mapping,
412 *this->fe,
413 *this->cell_quadrature,
414 update_values |
415 update_gradients |
416 update_JxW_values);
417 FEValues<dim> fe_values_vof_phase_gradient_projection(
418 *this->mapping,
419 dof_handler_vof_phase_fraction_gradient.get_fe(),
420 *this->cell_quadrature,
421 update_values);
422 FEValues<dim> fe_values_curvature_projection(*this->mapping,
423 dof_handler_curvature.get_fe(),
424 *this->cell_quadrature,
425 update_values);
426
427 // Initialize size of arrays
428 const double n_q_points =
429 fe_values_algebraic_reinitialization.get_quadrature().size();
430 const double n_dofs_per_cell =
431 fe_values_algebraic_reinitialization.get_fe().n_dofs_per_cell();
432
433 // Initialize local rhs
434 Vector<double> local_rhs(n_dofs_per_cell);
435
436 // Initialize local dof indices array
437 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
438
439 // Extractor for phase fraction gradient vector
440 FEValuesExtractors::Vector phase_fraction_gradients(0);
441
442 // Initialize phase fraction and projected phase fraction gradient and
443 // curvature solution arrays
444 std::vector<double> present_phase_fraction_values(n_q_points);
445 std::vector<double> previous_phase_fraction_values(n_q_points);
446 std::vector<Tensor<1, dim>> present_vof_phase_gradient_projection_values(
447 n_q_points);
448 std::vector<double> present_curvature_values(n_q_points);
449 std::vector<Tensor<1, dim>> present_reinitialized_phase_gradient_values(
450 n_q_points);
451
452 // Initialize shape function arrays
453 std::vector<double> phi(n_dofs_per_cell);
454 std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
455
456 // Compute diffusivity coefficient with the smallest cell size
457 const double h_min = identify_minimum_cell_size(
458 *this->mapping,
459 this->subequations_interface.get_dof_handler(
460 VOFSubequationsID::algebraic_interface_reinitialization),
461 *this->cell_quadrature,
462 this->triangulation->get_mpi_communicator());
463 const double diffusivity_coefficient = compute_diffusivity(h_min);
464
465 // BDF coefficients for pseudo time-stepping
466 const Parameters::SimulationControl::TimeSteppingMethod method =
467 Parameters::SimulationControl::TimeSteppingMethod::bdf1;
468 const Vector<double> bdf_coefficient_vector =
469 calculate_bdf_coefficients(method, this->time_step_vector);
470
471 for (const auto &cell : this->dof_handler->active_cell_iterators())
472 {
473 if (cell->is_locally_owned())
474 {
475 // Reinitialize local rhs
476 local_rhs = 0;
477
478 // Get phase gradient projection cell iterator
479 typename DoFHandler<dim>::active_cell_iterator
480 vof_phase_gradient_projection_cell(
481 &(*this->triangulation),
482 cell->level(),
483 cell->index(),
484 &dof_handler_vof_phase_fraction_gradient);
485 typename DoFHandler<dim>::active_cell_iterator
486 curvature_projection_cell(&(*this->triangulation),
487 cell->level(),
488 cell->index(),
489 &dof_handler_curvature);
490
491 // Reinitialize FEValues with corresponding cells
492 fe_values_algebraic_reinitialization.reinit(cell);
493 fe_values_vof_phase_gradient_projection.reinit(
494 vof_phase_gradient_projection_cell);
495 fe_values_curvature_projection.reinit(curvature_projection_cell);
496
497 // Get vector of Jacobi determinant times the quadrature weights
498 std::vector<double> JxW_vec =
499 fe_values_algebraic_reinitialization.get_JxW_values();
500
501 // Get VOF phase fraction and projected phase gradient values
502 fe_values_algebraic_reinitialization.get_function_values(
503 this->evaluation_point, present_phase_fraction_values);
504 fe_values_algebraic_reinitialization.get_function_values(
505 this->previous_solution, previous_phase_fraction_values);
506 fe_values_vof_phase_gradient_projection[phase_fraction_gradients]
507 .get_function_values(
508 this->subequations_interface.get_solution(
509 VOFSubequationsID::phase_gradient_projection),
510 present_vof_phase_gradient_projection_values);
511 fe_values_curvature_projection.get_function_values(
512 this->subequations_interface.get_solution(
513 VOFSubequationsID::curvature_projection),
514 present_curvature_values);
515 fe_values_algebraic_reinitialization.get_function_gradients(
516 this->evaluation_point,
517 present_reinitialized_phase_gradient_values);
518
519 // Set tolerance to avoid division by zero
520 const double tolerance = 1e-12;
521
522 // Loop over quadrature points
523 for (unsigned int q = 0; q < n_q_points; ++q)
524 {
525 // Shape functions
526 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
527 {
528 phi[k] =
529 fe_values_algebraic_reinitialization.shape_value(k, q);
530 grad_phi[k] =
531 fe_values_algebraic_reinitialization.shape_grad(k, q);
532 }
533
534 // Extract present and previous phase fraction values and
535 // projected phase gradient values
536 std::vector<double> phase_fraction_values(2);
537 phase_fraction_values[0] = present_phase_fraction_values[q];
538 const double phase_fraction = phase_fraction_values[0];
539 phase_fraction_values[1] = previous_phase_fraction_values[q];
540 const Tensor<1, dim> projected_vof_phase_gradient =
541 present_vof_phase_gradient_projection_values[q];
542 const double curvature = present_curvature_values[q];
543 const Tensor<1, dim> reinitialized_phase_gradient =
544 present_reinitialized_phase_gradient_values[q];
545
546 // Compute normal vector with projected phase gradient
547 const double projected_vof_phase_gradient_norm =
548 projected_vof_phase_gradient.norm() + tolerance;
549 const Tensor<1, dim> interface_normal =
550 (projected_vof_phase_gradient /
551 projected_vof_phase_gradient_norm);
552
553 // Local rhs assembly
554 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
555 {
556 // Time-stepping term (bdf1)
557 for (unsigned int p = 0; p < 2; ++p)
558 {
559 local_rhs(i) -= bdf_coefficient_vector[p] *
560 (phase_fraction_values[p] * phi[i]) *
561 JxW_vec[q];
562 }
563
564 local_rhs(i) -=
565 (
566 // Compressive term
567 phi[i] *
568 (scalar_product(reinitialized_phase_gradient,
569 interface_normal) -
570 2 * phase_fraction *
571 scalar_product(reinitialized_phase_gradient,
572 interface_normal) -
573 phase_fraction * (1 - phase_fraction) * curvature) +
574 // Diffusive term
575 (scalar_product(
576 grad_phi[i],
577 (diffusivity_coefficient *
578 scalar_product(reinitialized_phase_gradient,
579 interface_normal) *
580 interface_normal)))) *
581 JxW_vec[q];
582 }
583 }
584 // Distribute the local contributions to the global system
585 cell->get_dof_indices(local_dof_indices);
586 this->zero_constraints.distribute_local_to_global(local_rhs,
587 local_dof_indices,
588 this->system_rhs);
589 }
590 }
591 this->system_rhs.compress(VectorOperation::add);
592}

595template <int dim>
596void
597VOFAlgebraicInterfaceReinitialization<dim>::solve_linear_system(
598 const bool initial_step,
599 const bool /*renewed_matrix*/)
600{
601 auto mpi_communicator = this->triangulation->get_mpi_communicator();
602
603 const AffineConstraints<double> &constraints_used =
604 initial_step ? this->nonzero_constraints : this->zero_constraints;
605
606 const bool verbose(
607 this->simulation_parameters.vof_subequations_linear_solvers
608 .at(VOFSubequationsID::algebraic_interface_reinitialization)
609 .verbosity != Parameters::Verbosity::quiet);
610
611 // Get residual conditions
612 const double absolute_residual =
613 this->simulation_parameters.vof_subequations_linear_solvers
614 .at(VOFSubequationsID::algebraic_interface_reinitialization)
615 .minimum_residual;
616 const double relative_residual =
617 this->simulation_parameters.vof_subequations_linear_solvers
618 .at(VOFSubequationsID::algebraic_interface_reinitialization)
619 .relative_residual;
620
621 // Set linear solver tolerance
622 const double linear_solver_tolerance =
623 std::max(relative_residual * this->system_rhs.l2_norm(), absolute_residual);
624
625 if (verbose)
626 {
627 this->pcout << " -Tolerance of iterative solver is: "
628 << linear_solver_tolerance << std::endl;
629 }
630
631 // ILU preconditioner
632 const double ilu_fill =
633 this->simulation_parameters.vof_subequations_linear_solvers
634 .at(VOFSubequationsID::algebraic_interface_reinitialization)
635 .ilu_precond_fill;
636 const double ilu_atol =
637 this->simulation_parameters.vof_subequations_linear_solvers
638 .at(VOFSubequationsID::algebraic_interface_reinitialization)
639 .ilu_precond_atol;
640 const double ilu_rtol =
641 this->simulation_parameters.vof_subequations_linear_solvers
642 .at(VOFSubequationsID::algebraic_interface_reinitialization)
643 .ilu_precond_rtol;
644 TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
645 ilu_fill, ilu_atol, ilu_rtol, 0);
646
647 TrilinosWrappers::PreconditionILU ilu_preconditioner;
648
649 ilu_preconditioner.initialize(this->system_matrix, preconditionerOptions);
650
651 GlobalVectorType completely_distributed_solution(this->locally_owned_dofs,
652 mpi_communicator);
653

688template <int dim>
689void
690VOFAlgebraicInterfaceReinitialization<dim>::solve()
691{
692 // Check if all required solutions are valid
693 check_dependencies_validity();
694
695 // Compute time-step
696 this->current_time_step = compute_time_step();
697
698 double current_time_step_inv = 1. / this->current_time_step;
699 this->time_step_vector[0] = this->current_time_step;
700
701 // Set initial conditions from the VOF solution (step 0)
702 set_initial_conditions();
703
704 // Reinitialization steps counter
705 unsigned int step = 1;
706
707 // Define verbosity
708 const bool verbose(
709 this->subequation_verbosity != Parameters::Verbosity::quiet ||
710 this->simulation_parameters.vof_subequations_non_linear_solvers
711 .at(VOFSubequationsID::algebraic_interface_reinitialization)
712 .verbosity != Parameters::Verbosity::quiet);
713 std::string subequation_string =
714 this->subequations_interface.get_subequation_string(this->subequation_id);
715
716 if (verbose)
717 {
718 announce_string(this->pcout, "VOF algebraic interface reinitialization");
719
720 this->pcout << "Current algebraic reinitialization time-step: "
721 << this->current_time_step << std::endl;
722 }
723
724 // For debugging purposes
725 if (this->simulation_parameters.multiphysics.vof_parameters
726 .regularization_method.algebraic_interface_reinitialization
727 .output_reinitialization_steps)
728 write_output_results(step);
729
730 // Iterate until a stop criterion is met
731 while (continue_iterating(current_time_step_inv, step))
732 {
733 if (verbose)
734 {
735 this->pcout << "-Solving " << subequation_string << ", step " << step
736 << std::endl;
737 }
738
739 if (step == 1)
740 {
741 // Solve non-linear equation
742 this->solve_non_linear_system(true);
743 }
744 else
745 {
746 // Update previous solution
747 this->previous_solution = *this->present_solution;
748 this->previous_local_evaluation_point = this->local_evaluation_point;
749
750 // Update non-zero constraints
751 define_non_zero_constraints();
752
753 // Solve non-linear equation
754 this->solve_non_linear_system(false);
755
756 // For debugging purposes
757 if (this->simulation_parameters.multiphysics.vof_parameters
758 .regularization_method.algebraic_interface_reinitialization
759 .output_reinitialization_steps)
760 write_output_results(step);
761 }
762
763 // Increment step number
764 step++;
765 }
766
767 if (verbose)
768 this->pcout << "The solver took: " << step - 1 << " reinitialization steps"
769 << std::endl;
770
771 this->subequations_interface.set_solution_valid(this->subequation_id);
772}
