14template <int dim>
15void
16VOFAlgebraicInterfaceReinitialization<dim>::setup_dofs()
17{
18 // Get MPI communicator
19 auto mpi_communicator = this->triangulation->get_mpi_communicator();
20
21 // Distribute and renumber DoFs
22 this->dof_handler->distribute_dofs(*this->fe);
23 DoFRenumbering::Cuthill_McKee(*this->dof_handler);
24
25 // Get locally owned and relevant DoFs
26 this->locally_owned_dofs = this->dof_handler->locally_owned_dofs();
27 this->locally_relevant_dofs =
28 DoFTools::extract_locally_relevant_dofs(*this->dof_handler);
29
30 // Constraints
31 define_non_zero_constraints();
32 define_zero_constraints();
33
34 // Sparsity pattern
35 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
36 DoFTools::make_sparsity_pattern(*this->dof_handler,
37 dsp,
38 this->nonzero_constraints,
39 false);
40 SparsityTools::distribute_sparsity_pattern(dsp,
41 this->locally_owned_dofs,
42 mpi_communicator,
43 this->locally_relevant_dofs);
44
45 // Reinitialize system matrix and right-hand side (rhs)
46 this->system_matrix.reinit(this->locally_owned_dofs,
47 this->locally_owned_dofs,
48 dsp,
49 mpi_communicator);
50 this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
51
52 // Reinitialize solution vectors
53 this->present_solution->reinit(this->locally_owned_dofs,
54 this->locally_relevant_dofs,
55 mpi_communicator);
56 this->previous_solution.reinit(this->locally_owned_dofs,
57 this->locally_relevant_dofs,
58 mpi_communicator);
59 this->newton_update.reinit(this->locally_owned_dofs, mpi_communicator);
60 this->local_evaluation_point.reinit(this->locally_owned_dofs,
61 mpi_communicator);
62 this->previous_local_evaluation_point.reinit(this->locally_owned_dofs,
63 mpi_communicator);
64 this->evaluation_point = *this->present_solution;
65
66 if (this->subequation_verbosity != Parameters::Verbosity::quiet)
67 {
68 std::string subequation_string =
69 this->subequations_interface.get_subequation_string(
70 this->subequation_id);
71 this->pcout << " Number of " << subequation_string
72 << " degrees of freedom: " << this->dof_handler->n_dofs()
73 << std::endl;
74 }
75
76 // Provide DoFHandler and solutions to the subequations interface
77 this->subequations_interface.set_dof_handler(this->subequation_id,
78 this->dof_handler);
79 this->subequations_interface.set_solution(this->subequation_id,
80 this->present_solution);
81}

84template <int dim>
85void
86VOFAlgebraicInterfaceReinitialization<dim>::define_zero_constraints()
87{
88 this->zero_constraints.clear();
89 this->zero_constraints.reinit(this->locally_owned_dofs,
90 this->locally_relevant_dofs);
91
92 DoFTools::make_hanging_node_constraints(*this->dof_handler,
93 this->zero_constraints);
94
95 for (auto const &[id, type] :
96 this->simulation_parameters.boundary_conditions.type)
97 {
98 if (type == BoundaryConditions::BoundaryType::periodic)
99 {
100 DoFTools::make_periodicity_constraints(
101 *this->dof_handler,
102 id,
103 this->simulation_parameters.boundary_conditions.periodic_neighbor_id
104 .at(id),
105 this->simulation_parameters.boundary_conditions.periodic_direction
106 .at(id),
107 this->zero_constraints);
108 }
109 }
110 for (auto const &[id, type] :
111 this->simulation_parameters.boundary_conditions_vof.type)
112 {
113 if (type == BoundaryConditions::BoundaryType::vof_dirichlet)
114 {
115 VectorTools::interpolate_boundary_values(
116 *this->dof_handler,
117 id,
118 Functions::ZeroFunction<dim>(),
119 this->zero_constraints);
120 }
121 }
122 this->zero_constraints.close();
123}

169template <int dim>
170void
171VOFAlgebraicInterfaceReinitialization<dim>::set_initial_conditions()
172{
173 // Get VOF DoFHandler
174 const DoFHandler<dim> &dof_handler_vof =
175 this->subequations_interface.get_vof_dof_handler();
176
177 // Interpolate VOF solution to algebraic interface reinitialization
178 FETools::interpolate(dof_handler_vof,
179 this->subequations_interface.get_vof_solution(),
180 *this->dof_handler,
181 this->nonzero_constraints,
182 this->newton_update);
183
184 *this->present_solution = this->newton_update;
185 this->previous_solution =
186 *this->present_solution; // We only have 1 previous solution (bdf1)
187 this->previous_local_evaluation_point =
188 this->previous_solution; // For steady-state criterion evaluation
189
190 // Initial condition
191 if (this->simulation_parameters.multiphysics.vof_parameters
192 .regularization_method.algebraic_interface_reinitialization
193 .output_reinitialization_steps)
194 {
195 this->pvdhandler.times_and_names.clear();
196 write_output_results(0);
197 }
198}

201template <int dim>
202void
203VOFAlgebraicInterfaceReinitialization<dim>::assemble_system_matrix()
204{
205 // Reinitialize system matrix
206 this->system_matrix = 0;
207
208 // Get projected VOF phase gradient and curvature DoFHandler
209 const DoFHandler<dim> &dof_handler_vof_phase_fraction_gradient =
210 this->subequations_interface.get_dof_handler(
211 VOFSubequationsID::phase_gradient_projection);
212 const DoFHandler<dim> &dof_handler_curvature =
213 this->subequations_interface.get_dof_handler(
214 VOFSubequationsID::curvature_projection);
215
216 // Initialize FEValues for algebraic interface reinitialization, VOF
217 // phase fraction gradient and curvature projections
218 FEValues<dim> fe_values_algebraic_reinitialization(*this->mapping,
219 *this->fe,
220 *this->cell_quadrature,
221 update_values |
222 update_gradients |
223 update_JxW_values);
224 FEValues<dim> fe_values_phase_gradient_projection(
225 *this->mapping,
226 dof_handler_vof_phase_fraction_gradient.get_fe(),
227 *this->cell_quadrature,
228 update_values);
229 FEValues<dim> fe_values_curvature_projection(*this->mapping,
230 dof_handler_curvature.get_fe(),
231 *this->cell_quadrature,
232 update_values);
233
234 // Initialize size of arrays
235 const unsigned int n_q_points =
236 fe_values_algebraic_reinitialization.get_quadrature().size();
237 const unsigned int n_dofs_per_cell =
238 fe_values_algebraic_reinitialization.get_fe().n_dofs_per_cell();
239
240 // Initialize local matrix
241 FullMatrix<double> local_matrix(n_dofs_per_cell, n_dofs_per_cell);
242
243 // Initialize local dof indices array
244 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
245
246 // Extractor for phase fraction gradient vector
247 FEValuesExtractors::Vector phase_fraction_gradients(0);
248
249 // Initialize phase fraction, projected phase gradient and projected curvature
250 // solution arrays
251 std::vector<double> present_phase_fraction_values(n_q_points);
252 std::vector<Tensor<1, dim>> present_phase_gradient_projection_values(
253 n_q_points);
254 std::vector<double> present_curvature_values(n_q_points);
255 std::vector<Tensor<1, dim>> present_reinitialized_phase_gradient_values(
256 n_q_points);
257
258 // Initialize shape function arrays
259 std::vector<double> phi(n_dofs_per_cell);
260 std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
261
262 // Compute diffusivity coefficient with the smallest cell size
263 const double h_min = identify_minimum_cell_size(
264 *this->mapping,
265 this->subequations_interface.get_dof_handler(
266 VOFSubequationsID::algebraic_interface_reinitialization),
267 *this->cell_quadrature,
268 this->triangulation->get_mpi_communicator());
269 const double diffusivity_coefficient = compute_diffusivity(h_min);
270
271 // BDF coefficients for pseudo time-stepping
272 const Parameters::SimulationControl::TimeSteppingMethod method =
273 Parameters::SimulationControl::TimeSteppingMethod::bdf1;
274 const Vector<double> bdf_coefficient_vector =
275 calculate_bdf_coefficients(method, this->time_step_vector);
276
277 // Assemble system matrix
278 for (const auto &cell : this->dof_handler->active_cell_iterators())
279 {
280 if (cell->is_locally_owned())
281 {
282 // Reinitialize local matrix
283 local_matrix = 0;
284
285 // Get phase gradient projection and curvature cell iterators
286 typename DoFHandler<dim>::active_cell_iterator
287 vof_phase_gradient_projection_cell(
288 &(*this->triangulation),
289 cell->level(),
290 cell->index(),
291 &dof_handler_vof_phase_fraction_gradient);
292 typename DoFHandler<dim>::active_cell_iterator
293 curvature_projection_cell(&(*this->triangulation),
294 cell->level(),
295 cell->index(),
296 &dof_handler_curvature);
297
298 // Reinitialize FEValues with corresponding cell
299 fe_values_algebraic_reinitialization.reinit(cell);
300 fe_values_phase_gradient_projection.reinit(
301 vof_phase_gradient_projection_cell);
302 fe_values_curvature_projection.reinit(curvature_projection_cell);
303
304 // Get vector of JxW
305 std::vector<double> JxW_vec =
306 fe_values_algebraic_reinitialization.get_JxW_values();
307
308 // Get get present phase fraction, projected VOF phase gradient and
309 // curvature values
310 fe_values_algebraic_reinitialization.get_function_values(
311 this->evaluation_point, present_phase_fraction_values);
312 fe_values_phase_gradient_projection[phase_fraction_gradients]
313 .get_function_values(
314 this->subequations_interface.get_solution(
315 VOFSubequationsID::phase_gradient_projection),
316 present_phase_gradient_projection_values);
317 fe_values_curvature_projection.get_function_values(
318 this->subequations_interface.get_solution(
319 VOFSubequationsID::curvature_projection),
320 present_curvature_values);
321 fe_values_algebraic_reinitialization.get_function_gradients(
322 this->evaluation_point,
323 present_reinitialized_phase_gradient_values);
324
325 // Set tolerance to avoid division by zero
326 const double tolerance = 1e-12;
327
328 // Loop over quadrature points
329 for (unsigned int q = 0; q < n_q_points; ++q)
330 {
331 // Shape functions
332 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
333 {
334 phi[k] =
335 fe_values_algebraic_reinitialization.shape_value(k, q);
336 grad_phi[k] =
337 fe_values_algebraic_reinitialization.shape_grad(k, q);
338 }
339
340 // Extract phase fraction value, projected phase fraction gradient
341 // and curvature
342 const double phase_fraction = present_phase_fraction_values[q];
343 const Tensor<1, dim> projected_vof_phase_gradient =
344 present_phase_gradient_projection_values[q];
345 const double curvature = present_curvature_values[q];
346 const Tensor<1, dim> reinitialized_phase_gradient =
347 present_reinitialized_phase_gradient_values[q];
348
349 // Compute normal vector with projected phase gradient
350 const double projected_vof_phase_gradient_norm =
351 projected_vof_phase_gradient.norm() + tolerance;
352 const Tensor<1, dim> interface_normal =
353 (projected_vof_phase_gradient /
354 projected_vof_phase_gradient_norm);
355
356 // Local matrix assembly
357 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
358 {
359 for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
360 {
361 local_matrix(i, j) +=
362 (
363 // Time-stepping term
364 phi[i] * phi[j] * bdf_coefficient_vector[0] +
365 // Compressive term
366 phi[i] *
367 (scalar_product(grad_phi[j], interface_normal) -
368 2 * phase_fraction *
369 scalar_product(grad_phi[j], interface_normal) -
370 2 * phi[j] *
371 scalar_product(reinitialized_phase_gradient,
372 interface_normal) -
373 (phi[j] - 2 * phase_fraction * phi[j]) *
374 curvature) +
375 // Diffusive term
376 scalar_product(grad_phi[i],
377 (diffusivity_coefficient *
378 scalar_product(grad_phi[j],
379 interface_normal) *
380 interface_normal))) *
381 JxW_vec[q];
382 }
383 }
384 }
385 // Distribute the local contributions to the global system
386 cell->get_dof_indices(local_dof_indices);
387 this->zero_constraints.distribute_local_to_global(
388 local_matrix, local_dof_indices, this->system_matrix);
389 }
390 }
391 this->system_matrix.compress(VectorOperation::add);
392}

395template <int dim>
396void
397VOFAlgebraicInterfaceReinitialization<dim>::assemble_system_rhs()
398{
399 // Reinitialize system right-hand side (rhs)
400 this->system_rhs = 0;
401
402 // Get projected phase gradient, curvature and reinitialized phase gradient
403 // DoFHandler
404 const DoFHandler<dim> &dof_handler_vof_phase_fraction_gradient =
405 this->subequations_interface.get_dof_handler(
406 VOFSubequationsID::phase_gradient_projection);
407 const DoFHandler<dim> &dof_handler_curvature =
408 this->subequations_interface.get_dof_handler(
409 VOFSubequationsID::curvature_projection);
410
411 // Initialize FEValues for algebraic interface reinitialization, VOF phase
412 // fraction gradient and curvature projections
413 FEValues<dim> fe_values_algebraic_reinitialization(*this->mapping,
414 *this->fe,
415 *this->cell_quadrature,
416 update_values |
417 update_gradients |
418 update_JxW_values);
419 FEValues<dim> fe_values_vof_phase_gradient_projection(
420 *this->mapping,
421 dof_handler_vof_phase_fraction_gradient.get_fe(),
422 *this->cell_quadrature,
423 update_values);
424 FEValues<dim> fe_values_curvature_projection(*this->mapping,
425 dof_handler_curvature.get_fe(),
426 *this->cell_quadrature,
427 update_values);
428
429 // Initialize size of arrays
430 const unsigned int n_q_points =
431 fe_values_algebraic_reinitialization.get_quadrature().size();
432 const unsigned int n_dofs_per_cell =
433 fe_values_algebraic_reinitialization.get_fe().n_dofs_per_cell();
434
435 // Initialize local rhs
436 Vector<double> local_rhs(n_dofs_per_cell);
437
438 // Initialize local dof indices array
439 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
440
441 // Extractor for phase fraction gradient vector
442 FEValuesExtractors::Vector phase_fraction_gradients(0);
443
444 // Initialize phase fraction and projected phase fraction gradient and
445 // curvature solution arrays
446 std::vector<double> present_phase_fraction_values(n_q_points);
447 std::vector<double> previous_phase_fraction_values(n_q_points);
448 std::vector<Tensor<1, dim>> present_vof_phase_gradient_projection_values(
449 n_q_points);
450 std::vector<double> present_curvature_values(n_q_points);
451 std::vector<Tensor<1, dim>> present_reinitialized_phase_gradient_values(
452 n_q_points);
453
454 // Initialize shape function arrays
455 std::vector<double> phi(n_dofs_per_cell);
456 std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
457
458 // Compute diffusivity coefficient with the smallest cell size
459 const double h_min = identify_minimum_cell_size(
460 *this->mapping,
461 this->subequations_interface.get_dof_handler(
462 VOFSubequationsID::algebraic_interface_reinitialization),
463 *this->cell_quadrature,
464 this->triangulation->get_mpi_communicator());
465 const double diffusivity_coefficient = compute_diffusivity(h_min);
466
467 // BDF coefficients for pseudo time-stepping
468 const Parameters::SimulationControl::TimeSteppingMethod method =
469 Parameters::SimulationControl::TimeSteppingMethod::bdf1;
470 const Vector<double> bdf_coefficient_vector =
471 calculate_bdf_coefficients(method, this->time_step_vector);
472
473 for (const auto &cell : this->dof_handler->active_cell_iterators())
474 {
475 if (cell->is_locally_owned())
476 {
477 // Reinitialize local rhs
478 local_rhs = 0;
479
480 // Get phase gradient projection cell iterator
481 typename DoFHandler<dim>::active_cell_iterator
482 vof_phase_gradient_projection_cell(
483 &(*this->triangulation),
484 cell->level(),
485 cell->index(),
486 &dof_handler_vof_phase_fraction_gradient);
487 typename DoFHandler<dim>::active_cell_iterator
488 curvature_projection_cell(&(*this->triangulation),
489 cell->level(),
490 cell->index(),
491 &dof_handler_curvature);
492
493 // Reinitialize FEValues with corresponding cells
494 fe_values_algebraic_reinitialization.reinit(cell);
495 fe_values_vof_phase_gradient_projection.reinit(
496 vof_phase_gradient_projection_cell);
497 fe_values_curvature_projection.reinit(curvature_projection_cell);
498
499 // Get vector of Jacobi determinant times the quadrature weights
500 std::vector<double> JxW_vec =
501 fe_values_algebraic_reinitialization.get_JxW_values();
502
503 // Get VOF phase fraction and projected phase gradient values
504 fe_values_algebraic_reinitialization.get_function_values(
505 this->evaluation_point, present_phase_fraction_values);
506 fe_values_algebraic_reinitialization.get_function_values(
507 this->previous_solution, previous_phase_fraction_values);
508 fe_values_vof_phase_gradient_projection[phase_fraction_gradients]
509 .get_function_values(
510 this->subequations_interface.get_solution(
511 VOFSubequationsID::phase_gradient_projection),
512 present_vof_phase_gradient_projection_values);
513 fe_values_curvature_projection.get_function_values(
514 this->subequations_interface.get_solution(
515 VOFSubequationsID::curvature_projection),
516 present_curvature_values);
517 fe_values_algebraic_reinitialization.get_function_gradients(
518 this->evaluation_point,
519 present_reinitialized_phase_gradient_values);
520
521 // Set tolerance to avoid division by zero
522 const double tolerance = 1e-12;
523
524 // Loop over quadrature points
525 for (unsigned int q = 0; q < n_q_points; ++q)
526 {
527 // Shape functions
528 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
529 {
530 phi[k] =
531 fe_values_algebraic_reinitialization.shape_value(k, q);
532 grad_phi[k] =
533 fe_values_algebraic_reinitialization.shape_grad(k, q);
534 }
535
536 // Extract present and previous phase fraction values and
537 // projected phase gradient values
538 std::vector<double> phase_fraction_values(2);
539 phase_fraction_values[0] = present_phase_fraction_values[q];
540 const double phase_fraction = phase_fraction_values[0];
541 phase_fraction_values[1] = previous_phase_fraction_values[q];
542 const Tensor<1, dim> projected_vof_phase_gradient =
543 present_vof_phase_gradient_projection_values[q];
544 const double curvature = present_curvature_values[q];
545 const Tensor<1, dim> reinitialized_phase_gradient =
546 present_reinitialized_phase_gradient_values[q];
547
548 // Compute normal vector with projected phase gradient
549 const double projected_vof_phase_gradient_norm =
550 projected_vof_phase_gradient.norm() + tolerance;
551 const Tensor<1, dim> interface_normal =
552 (projected_vof_phase_gradient /
553 projected_vof_phase_gradient_norm);
554
555 // Local rhs assembly
556 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
557 {
558 // Time-stepping term (bdf1)
559 for (unsigned int p = 0; p < 2; ++p)
560 {
561 local_rhs(i) -= bdf_coefficient_vector[p] *
562 (phase_fraction_values[p] * phi[i]) *
563 JxW_vec[q];
564 }
565
566 local_rhs(i) -=
567 (
568 // Compressive term
569 phi[i] *
570 (scalar_product(reinitialized_phase_gradient,
571 interface_normal) -
572 2 * phase_fraction *
573 scalar_product(reinitialized_phase_gradient,
574 interface_normal) -
575 phase_fraction * (1 - phase_fraction) * curvature) +
576 // Diffusive term
577 (scalar_product(
578 grad_phi[i],
579 (diffusivity_coefficient *
580 scalar_product(reinitialized_phase_gradient,
581 interface_normal) *
582 interface_normal)))) *
583 JxW_vec[q];
584 }
585 }
586 // Distribute the local contributions to the global system
587 cell->get_dof_indices(local_dof_indices);
588 this->zero_constraints.distribute_local_to_global(local_rhs,
589 local_dof_indices,
590 this->system_rhs);
591 }
592 }
593 this->system_rhs.compress(VectorOperation::add);
594}

597template <int dim>
598void
599VOFAlgebraicInterfaceReinitialization<dim>::solve_linear_system()
600{
601 auto mpi_communicator = this->triangulation->get_mpi_communicator();
602
603 const AffineConstraints<double> &constraints_used = this->zero_constraints;
604
605 const bool verbose(
606 this->simulation_parameters.vof_subequations_linear_solvers
607 .at(VOFSubequationsID::algebraic_interface_reinitialization)
608 .verbosity != Parameters::Verbosity::quiet);
609
610 // Get residual conditions
611 const double absolute_residual =
612 this->simulation_parameters.vof_subequations_linear_solvers
613 .at(VOFSubequationsID::algebraic_interface_reinitialization)
614 .minimum_residual;
615 const double relative_residual =
616 this->simulation_parameters.vof_subequations_linear_solvers
617 .at(VOFSubequationsID::algebraic_interface_reinitialization)
618 .relative_residual;
619
620 // Set linear solver tolerance
621 const double rescale_metric = this->get_residual_rescale_metric();
622 const double current_residual = this->system_rhs.l2_norm() / rescale_metric;
623 const double linear_solver_tolerance =
624 std::max(relative_residual * current_residual, absolute_residual);
625 const double non_rescaled_linear_solver_tolerance =
626 linear_solver_tolerance * rescale_metric;
627
628 if (verbose)
629 {
630 this->pcout << " -Tolerance of iterative solver is: "
631 << linear_solver_tolerance << std::endl;
632 }
633
634 // ILU preconditioner
635 const unsigned int ilu_fill =
636 this->simulation_parameters.vof_subequations_linear_solvers
637 .at(VOFSubequationsID::algebraic_interface_reinitialization)
638 .ilu_precond_fill;
639 const double ilu_atol =
640 this->simulation_parameters.vof_subequations_linear_solvers
641 .at(VOFSubequationsID::algebraic_interface_reinitialization)
642 .ilu_precond_atol;
643 const double ilu_rtol =
644 this->simulation_parameters.vof_subequations_linear_solvers
645 .at(VOFSubequationsID::algebraic_interface_reinitialization)
646 .ilu_precond_rtol;
647 TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
648 ilu_fill, ilu_atol, ilu_rtol, 0);
649
650 TrilinosWrappers::PreconditionILU ilu_preconditioner;
651
652 ilu_preconditioner.initialize(this->system_matrix, preconditionerOptions);
653
654 GlobalVectorType completely_distributed_solution(this->locally_owned_dofs,
655 mpi_communicator);
656
657 SolverControl solver_control(
658 this->simulation_parameters.vof_subequations_linear_solvers
659 .at(VOFSubequationsID::algebraic_interface_reinitialization)
660 .max_iterations,
661 non_rescaled_linear_solver_tolerance,
662 true,
663 true);
664
665 TrilinosWrappers::SolverGMRES::AdditionalData solver_parameters(
666 false,
667 this->simulation_parameters.vof_subequations_linear_solvers
668 .at(VOFSubequationsID::algebraic_interface_reinitialization)
669 .max_krylov_vectors);
670
671 TrilinosWrappers::SolverGMRES solver(solver_control, solver_parameters);
672
673 solver.solve(this->system_matrix,
674 completely_distributed_solution,
675 this->system_rhs,
676 ilu_preconditioner);
677
678 if (verbose)

680 this->pcout << " -Iterative solver took: " << solver_control.last_step()
681 << " steps to reach a residual norm of "
682 << solver_control.last_value() / rescale_metric << std::endl;
683 }
684
685 // Update constraints and newton vectors
686 constraints_used.distribute(completely_distributed_solution);
687 this->newton_update = completely_distributed_solution;

691template <int dim>
692void
693VOFAlgebraicInterfaceReinitialization<dim>::solve()
694{
695 // Check if all required solutions are valid
696 check_dependencies_validity();
697
698 // Compute time-step
699 this->current_time_step = compute_time_step();
700
701 double current_time_step_inv = 1. / this->current_time_step;
702 this->time_step_vector[0] = this->current_time_step;
703
704 // Set initial conditions from the VOF solution (step 0)
705 set_initial_conditions();
706
707 // Reinitialization steps counter
708 unsigned int step = 1;
709
710 // Define verbosity
711 const bool verbose(
712 this->subequation_verbosity != Parameters::Verbosity::quiet ||
713 this->simulation_parameters.vof_subequations_non_linear_solvers
714 .at(VOFSubequationsID::algebraic_interface_reinitialization)
715 .verbosity != Parameters::Verbosity::quiet);
716 std::string subequation_string =
717 this->subequations_interface.get_subequation_string(this->subequation_id);
718
719 if (verbose)
720 {
721 announce_string(this->pcout, "VOF algebraic interface reinitialization");
722
723 this->pcout << "Current algebraic reinitialization time-step: "
724 << this->current_time_step << std::endl;
725 }
726
727 // For debugging purposes
728 if (this->simulation_parameters.multiphysics.vof_parameters
729 .regularization_method.algebraic_interface_reinitialization
730 .output_reinitialization_steps)
731 write_output_results(step);
732
733 // Iterate until a stop criterion is met
734 while (continue_iterating(current_time_step_inv, step))
735 {
736 if (verbose)
737 {
738 this->pcout << "-Solving " << subequation_string << ", step " << step
739 << std::endl;
740 }
741
742 // Update previous solution
743 this->previous_solution = *this->present_solution;
744 this->previous_local_evaluation_point = this->local_evaluation_point;
745
746 // Update non-zero constraints
747 define_non_zero_constraints();
748
749 // Solve non-linear equation
750 this->solve_governing_system();
751
752 // For debugging purposes
753 if (this->simulation_parameters.multiphysics.vof_parameters
754 .regularization_method.algebraic_interface_reinitialization
755 .output_reinitialization_steps)
756 write_output_results(step);
757
758 // Increment step number
759 step++;
760 }
761
762 if (verbose)
763 this->pcout << "The solver took: " << step - 1 << " reinitialization steps"
764 << std::endl;
765
766 this->subequations_interface.set_solution_valid(this->subequation_id);
767}
