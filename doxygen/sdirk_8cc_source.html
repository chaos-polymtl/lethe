8// Matrix of coefficients for the SDIRK methods
9// The lines store the information required for each step
10// Column 0 always refer to outcome of the step that is being calculated
11// Column 1 always refer to step n
12// Column 2+ refer to intermediary steps
13FullMatrix<double>
14sdirk_coefficients(unsigned int order, double time_step)
15{
16 FullMatrix<double> sdirk_coefs;
17 double sdt = 1. / time_step;
18
19
20 if (order == 2)
21 {
22 const double alpha = (2. - std::sqrt(2)) / 2.;
23 sdirk_coefs.reinit(2, 3);
24 sdirk_coefs[0][0] = 1. / alpha * sdt;
25 sdirk_coefs[0][1] = -1. / alpha * sdt;
26 sdirk_coefs[1][0] = 1. / alpha * sdt;
27 sdirk_coefs[1][1] = -(2 * alpha - 1) / alpha / alpha * sdt;
28 sdirk_coefs[1][2] = -(1 - alpha) / alpha / alpha * sdt;
29 }
30
31 if (order == 3)
32 {
33 sdirk_coefs.reinit(3, 4);
34
35 sdirk_coefs[0][0] = 2.29428036027904 * sdt;
36 sdirk_coefs[0][1] = -2.29428036027904 * sdt;
37 sdirk_coefs[1][0] = 2.29428036027904 * sdt;
38 sdirk_coefs[1][1] = -0.809559354637498 * sdt;
39 sdirk_coefs[1][2] = -1.48472100564154 * sdt;
40 sdirk_coefs[2][0] = 2.29428036027904 * sdt;
41 sdirk_coefs[2][1] = 2.87009860433106 * sdt;
42 sdirk_coefs[2][2] = -8.55612780155264 * sdt;
43 sdirk_coefs[2][3] = 3.39174883694255 * sdt;
44 }
45
46 return sdirk_coefs;
47}
