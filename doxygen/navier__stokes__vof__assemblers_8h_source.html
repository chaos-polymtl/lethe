16/**
17 * @brief Assembles the core of the Navier-Stokes equation with
18 * free surface using VOF modeling.
19 * According to the following weak form:
20 * \f$ \nabla \cdot \mathbf{u} + \rho \mathbf{u} \cdot \nabla
21 * \mathbf{u} + \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
22 * (\nabla \mathbf{u})^T) = 0 \f$ with an SUPG and PSPG stabilization
23 *
24 * @tparam dim An integer that denotes the number of spatial dimensions
25 *
26 * @ingroup assemblers
27 */
28template <int dim>
29class GLSNavierStokesVOFAssemblerCore : public NavierStokesAssemblerBase<dim>
30{
31public:
32 GLSNavierStokesVOFAssemblerCore(
33 const std::shared_ptr<SimulationControl> &simulation_control,
34 const SimulationParameters<dim> &nsparam)
35 : simulation_control(simulation_control)
36 , vof_parameters(nsparam.multiphysics.vof_parameters)
37 {}
38
39 /**
40 * @brief assemble_matrix Assembles the matrix
41 * @param scratch_data (see base class)
42 * @param copy_data (see base class)
43 */
44 virtual void
45 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
46 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
47
48 /**
49 * @brief assemble_rhs Assembles the rhs
50 * @param scratch_data (see base class)
51 * @param copy_data (see base class)
52 */
53 virtual void
54 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
55 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
56
57 const bool SUPG = true;

61};

106/**
107 * @brief Assembles a phase change Darcy forcing term for a VOF two-fluid
108 * simulation. The term \f$-\phi \beta_D \mathbf{u} \f$ is added to the right
109 * hand-side of the Navier-Stokes equations to prohibit the motion of a
110 * material. In the phase change model, the value of the \f$ \beta_D \f$
111 * coefficient depends on the temperature field and the material (fluid)
112 * properties. Generally, this is used to impose the stasis in the solid
113 * phase. This generally leads to a better conditioning of the linear
114 * system than increasing the viscosity of the solid phase.
115 *
116 *
117 * @tparam dim An integer that denotes the number of spatial dimensions
118 *
119 * @ingroup assemblers
120 */
121template <int dim>
122class PhaseChangeDarcyVOFAssembler : public NavierStokesAssemblerBase<dim>
123{
124public:
125 PhaseChangeDarcyVOFAssembler(
126 const std::vector<Parameters::PhaseChange> &phase_change_parameters_vector)
127 : phase_change_parameters_vector(phase_change_parameters_vector)
128 {}
129
130 /**
131 * @brief assemble_matrix Assembles the matrix of: \f$-\beta_D \mathbf{u}\f$.
132 * @param scratch_data (see base class)
133 * @param copy_data (see base class)
134 */
135 virtual void
136 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
137 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
138
139
140 /**
141 * @brief assemble_rhs Assembles the weak form of: \f$-\beta_D \mathbf{u} \f$
142 * @param scratch_data (see base class)
143 * @param copy_data (see base class)
144 */
145 virtual void
146 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
147 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
148
149private:

151 * Phase change parameters are kept within the assembler and are used to
152 * calculate, on the fly, the inverse permeability (\f$ \beta_D \f$).

306/**
307 * @brief Assembles the core of the Navier-Stokes equation
308 * using a Rheological model to predict non-Newtonian behaviors
309 *
310 * @tparam dim An integer that denotes the number of spatial dimensions
311 *
312 * @ingroup assemblers
313 */
314template <int dim>
315class GLSNavierStokesVOFAssemblerNonNewtonianCore
316 : public NavierStokesAssemblerBase<dim>
317{
318public:
319 GLSNavierStokesVOFAssemblerNonNewtonianCore(
320 const std::shared_ptr<SimulationControl> &simulation_control,
321 const SimulationParameters<dim> &nsparam)
322 : simulation_control(simulation_control)
323 , vof_parameters(nsparam.multiphysics.vof_parameters)
324 {}

326 /**
327 * @brief Calculates an approximation of the gradient of the viscosity
328 * @param velocity_gradient The velocity gradient tensor on the quadrature point
329 @param velocity_hessians The velocity hessian tensor on the quadrture point
330 @param non_newtonian_viscosity The viscosity at which the gradient is calculated
331 @param d_gamma_dot Th difference in the shear rate magnitude to approximate the
332 viscosity variation with a slight change in the shear_rate magnitude
333 */
334 inline Tensor<1, dim>
335 get_kinematic_viscosity_gradient(
336 const Tensor<2, dim> &velocity_gradient,
337 const Tensor<3, dim> &velocity_hessians,
338 const double shear_rate_magnitude,
339 const double grad_kinematic_viscosity_shear_rate) const
340 {
341 // Calculates an approximation of the shear rate magnitude gradient using
342 // the derived form, since it does not change with rheological models
343 Tensor<1, dim> grad_shear_rate;
344 for (int d = 0; d < dim; ++d)
345 {
346 if (dim == 2)
347 {
348 for (unsigned int k = 0; k < dim; ++k)
349 {
350 grad_shear_rate[d] +=
351 2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
352 shear_rate_magnitude;
353 }
354 grad_shear_rate[d] +=
355 (velocity_gradient[0][1] + velocity_gradient[1][0]) *
356 (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
357 shear_rate_magnitude;
358 }
359 else
360 {
361 for (unsigned int k = 0; k < dim; ++k)
362 {
363 grad_shear_rate[d] +=
364 2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
365 shear_rate_magnitude +
366 (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
367 velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
368 (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
369 velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
370 shear_rate_magnitude;
371 }
372 }
373 }
374 return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
375 };
