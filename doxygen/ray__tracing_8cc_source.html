18template <int dim>
19RayTracingSolver<dim>::RayTracingSolver(
20 RayTracingSolverParameters<dim> &parameters,
21 DEMSolverParameters<dim> &dem_parameters)
22 : mpi_communicator(MPI_COMM_WORLD)
23 , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
24 , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
25 , pcout(std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
26 , parameters(parameters)
27 , dem_parameters(dem_parameters)
28 , triangulation(this->mpi_communicator)
29 , mapping(1)
30 , particle_handler(triangulation,
31 mapping,
32 DEMProperties::PropertiesIndex::n_properties)
33 , photon_handler(triangulation, mapping, 6)
34 , computing_timer(this->mpi_communicator,
35 this->pcout,
36 TimerOutput::summary,
37 TimerOutput::wall_times)
38 , background_dh(triangulation)
39 , displacement_distance(0)
40{
41 AssertThrow(!(parameters.model_parameters.load_balance_method ==
42 Parameters::Lagrangian::ModelParameters<
43 dim>::LoadBalanceMethod::dynamic_with_sparse_contacts ||
44 parameters.model_parameters.load_balance_method ==
45 Parameters::Lagrangian::ModelParameters<
46 dim>::LoadBalanceMethod::dynamic),
47 dealii::ExcMessage(
48 "The \"dynamic\" and \"dynamic_with_sparse_contacts\""
49 "load balancing methods are not supported"));
50}

52template <int dim>
53void
54RayTracingSolver<dim>::setup_parameters()
55{
56 // Print simulation starting information
57 pcout << std::endl;
58 std::string msg =
59 "Running on " + std::to_string(n_mpi_processes) + " rank(s)";
60 announce_string(pcout, msg, '*');
61
62 if (parameters.timer.type == Parameters::Timer::Type::none)
63 computing_timer.disable_output();
64
65 // Get the pointer of the only instance of the action manager
66 action_manager = DEMActionManager::get_action_manager();
67
68 // Set the simulation control as SimulationControlRayTracing
69 simulation_control =
70 std::make_shared<SimulationControlRayTracing>(parameters.simulation_control,
71 photon_handler);
72
73 // Setup load balancing parameters and attach the correct functions to the
74 // signals inside the triangulation
75 load_balancing.set_parameters(parameters.model_parameters);
76
77 AdaptiveSparseContacts<dim, DEMProperties::PropertiesIndex>
78 dummy_sparse_contacts_object;
79 load_balancing.copy_references(simulation_control,
80 triangulation,
81 photon_handler,
82 dummy_sparse_contacts_object);
83 load_balancing.connect_weight_signals();
84}

86template <int dim>
87std::shared_ptr<Insertion<dim, DEMProperties::PropertiesIndex>>
88RayTracingSolver<dim>::set_particle_insertion_type()
89{
90 using namespace Parameters::Lagrangian;
91 typename InsertionInfo<dim>::InsertionMethod insertion_method =
92 parameters.particle_insertion_info.insertion_method;
93
94 std::vector<std::shared_ptr<Distribution>>
95 dummy_size_distribution_object_container;
96
97 switch (insertion_method)
98 {
99 case InsertionInfo<dim>::InsertionMethod::file:
100 return std::make_shared<
101 InsertionFile<dim, DEMProperties::PropertiesIndex>>(
102 dummy_size_distribution_object_container,
103 triangulation,
104 dem_parameters);
105 case InsertionInfo<dim>::InsertionMethod::list:
106 return std::make_shared<
107 InsertionList<dim, DEMProperties::PropertiesIndex>>(
108 dummy_size_distribution_object_container,
109 triangulation,
110 dem_parameters);
111 default:
112 AssertThrow(
113 false,
114 dealii::ExcMessage(
115 "For lethe-particles-ray-tracing simulation, only the list"
116 " and file insertion are supported."));
117 return nullptr;
118 }

130template <int dim>
131void
132RayTracingSolver<dim>::load_balance()
133{
134 load_balancing.check_load_balance_iteration();
135
136 if (!action_manager->check_load_balance())
137 return;
138
139 TimerOutput::Scope t(this->computing_timer, "Load balancing");
140 // Prepare particle handler for the adaptation of the triangulation to the
141 // load
142 photon_handler.prepare_for_coarsening_and_refinement();
143 particle_handler.prepare_for_coarsening_and_refinement();
144
145 pcout << "-->Repartitioning triangulation" << std::endl;
146 triangulation.repartition();
147
148 // Unpack the photon handler after the mesh has been repartitioned
149 photon_handler.unpack_after_coarsening_and_refinement();
150 particle_handler.unpack_after_coarsening_and_refinement();
151
152 const auto average_minimum_maximum_cells =
153 Utilities::MPI::min_max_avg(triangulation.n_active_cells(),
154 mpi_communicator);
155
156 const auto average_minimum_maximum_photons =
157 Utilities::MPI::min_max_avg(photon_handler.n_locally_owned_particles(),
158 mpi_communicator);
159
160 pcout << "Load balance finished " << std::endl;
161 pcout
162 << "Average, minimum and maximum number of photons on the processors are "
163 << average_minimum_maximum_photons.avg << " , "
164 << average_minimum_maximum_photons.min << " and "
165 << average_minimum_maximum_photons.max << std::endl;
166 pcout << "Minimum and maximum number of cells owned by the processors are "
167 << average_minimum_maximum_cells.min << " and "
168 << average_minimum_maximum_cells.max << std::endl;
169
170 setup_background_dofs();
171
172 // Particles don't move, thus we sort them at the end of a load balance.
173 // Photons are being sorted every pseudo time step, thus no need to sort them
174 // here.
175 particle_handler.sort_particles_into_subdomains_and_cells();
176
177 // Exchange ghost particles
178 particle_handler.exchange_ghost_particles(true);
179
180 cells_local_neighbor_list.clear();
181 cells_ghost_neighbor_list.clear();
182
183 // Update cell neighbors
184 find_cell_neighbors<dim, true>(triangulation,
185 cells_local_neighbor_list,
186 cells_ghost_neighbor_list);
187}

202template <int dim>
203void
204RayTracingSolver<dim>::insert_particles_and_photons()
205{
206 // Insert particles using the insertion object.
207 particle_insertion_object->insert(particle_handler,
208 triangulation,
209 dem_parameters);
210
211 // A vector of points where the insertion location of every photon will be
212 // stored.
213 std::vector<Point<3>> insertion_points_on_proc;
214
215 // For each photon, we store its insertion/initial location as a property.
216 // This initial location is used to identify the right intersection
217 // point when multiple intersections are found. The logic used therein is that
218 // the intersection point which is closest to the initial location is the
219 // correct one. We also store the displacement direction unit vector in this
220 // vector since each photon has its own slightly offset from the reference
221 // displacement direction.
222 std::vector<std::vector<double>> photon_properties;
223
224 // Create variables for readability
225 const std::vector<unsigned int> &n_photons_each_directions =
226 parameters.ray_tracing_info.n_photons_each_directions;
227
228 // Starting point
229 const Point<dim> starting_insertion_point =
230 parameters.ray_tracing_info.starting_point;
231
232 // Insertion directions
233 const Tensor<1, dim> first_dir =
234 parameters.ray_tracing_info.insertion_directions_units_vector.at(0);
235 const Tensor<1, dim> second_dir =
236 parameters.ray_tracing_info.insertion_directions_units_vector.at(1);
237 const Tensor<1, dim> third_dir =
238 parameters.ray_tracing_info.insertion_directions_units_vector.at(2);
239
240 // Steps
241 const double step_first_dir =
242 parameters.ray_tracing_info.step_between_photons_each_directions.at(0);
243 const double step_second_dir =
244 parameters.ray_tracing_info.step_between_photons_each_directions.at(1);
245 const double step_third_dir =
246 parameters.ray_tracing_info.step_between_photons_each_directions.at(2);
247
248 // Reference displacement direction
249 const Tensor<1, 3> ref_displacement_dir =
250 parameters.ray_tracing_info.ref_displacement_tensor_unit;
251
252 // Total number of photons to insert
253 const unsigned int n_total_photons_to_insert =
254 n_photons_each_directions.at(0) * n_photons_each_directions.at(1) *
255 n_photons_each_directions.at(2);
256
257 // Distributing photons between processors
258 const unsigned int base_photons_per_proc =
259 n_total_photons_to_insert / n_mpi_processes;
260 const unsigned int remainder = n_total_photons_to_insert % n_mpi_processes;
261
262 // We add one photon to the first processors so that the total number of
263 // photon inserted match the requested number written in the prm file.
264 const unsigned int n_photons_to_insert_this_proc =
265 base_photons_per_proc + (this_mpi_process < remainder ? 1 : 0);
266
267 // First and last photon id on this processor. This will be used to find the
268 // insertion location on this proc.
269 const unsigned int first_id =
270 this_mpi_process * base_photons_per_proc +
271 (this_mpi_process < remainder ? this_mpi_process : remainder);
272 const unsigned int last_id = first_id + n_photons_to_insert_this_proc - 1;
273
274 // Generate the random offsets for the insertion position and the displacement
275 // direction. For the position, the offset uses the same logic as other
276 // insertion mechanism in Lethe. For the displacement direction of each
277 // photon, we need to generate one random angular offset (theta) relative to
278 // the reference displacement vector. We then need to decide in which
279 // direction (phi) around this reference displacement direction vector the
280 // offset will be applied.
281 std::vector<double> random_number_angular_1; // From 0 to 2*pi
282 std::vector<double> random_number_angular_2; // From 0 to max_angular_offset
283 std::vector<double> random_number_position; // From 0 to max_insertion_offset
284
285 // Reserve the size of the vectors
286 random_number_angular_1.reserve(n_photons_to_insert_this_proc);
287 random_number_angular_2.reserve(n_photons_to_insert_this_proc);
288 random_number_position.reserve(3 * n_photons_to_insert_this_proc);
289
290 create_random_number_container(
291 random_number_angular_1,
292 n_photons_to_insert_this_proc,
293 2.0 * M_PI,
294 parameters.ray_tracing_info.prn_seed_photon_displacement);
295
296 create_random_number_container(
297 random_number_angular_2,
298 n_photons_to_insert_this_proc,
299 parameters.ray_tracing_info.max_angular_offset,
300 parameters.ray_tracing_info.prn_seed_photon_displacement);
301
302 create_random_number_container(
303 random_number_position,
304 3 * n_photons_to_insert_this_proc,
305 parameters.ray_tracing_info.max_insertion_offset,
306 parameters.ray_tracing_info.prn_seed_photon_insertion);
307
308 // For the displacement direction randomness, we need to find two vectors
309 // normal to ref_displacement_dir.
310 // We make sure that our first vector is not parallel to the
311 // ref_displacement_dir
312 const Tensor<1, 3> temp_normal_vector =
313 (std::fabs(ref_displacement_dir[0]) < 0.9) ? Tensor<1, 3>({1, 0, 0}) :
314 Tensor<1, 3>({0, 1, 0});
315
316 Tensor<1, 3> u = temp_normal_vector -
317 temp_normal_vector *
318 scalar_product(temp_normal_vector, ref_displacement_dir);
319 u = u / u.norm();
320
321 // We find the second one using the cross product
322 const Tensor<1, 3> v = cross_product_3d(ref_displacement_dir, u);
323
324 // Create the insertion location of every photon from their IDs.
325 // Temporary variable
326 Point<3> temp_point;
327 Tensor<1, 3> temp_dir;
328 std::vector<double> properties_of_one_photon(6);
329
330 // Prepare the containers.
331 insertion_points_on_proc.reserve(n_photons_to_insert_this_proc);
332
333 // Fill the photon properties container with vectors of size 6.
334 photon_properties.resize(n_photons_to_insert_this_proc,
335 std::vector<double>(6));
336 // This loop is equivalent of having three nested loops over the three
337 // directions. However, this way we can easily distribute the photons over
338 // the processors using the first and last ID.
339 for (unsigned int id = first_id; id <= last_id; ++id)
340 {
341 // nth position in the z direction
342 const unsigned int iz = id / (n_photons_each_directions.at(0) *
343 n_photons_each_directions.at(1));
344
345 const unsigned int rem = id % (n_photons_each_directions.at(0) *
346 n_photons_each_directions.at(1));
347 // nth position in the y direction
348 const unsigned int iy = rem / n_photons_each_directions.at(0);
349
350 // nth position in the x direction
351 const unsigned int ix = rem % n_photons_each_directions.at(0);
352
353 // ID relative to this processor. This is used to write at the right
354 // location in each container.
355 const unsigned int id_on_proc = id - first_id;
356 temp_point =
357 starting_insertion_point +
358 (ix * step_first_dir + random_number_position.at(3 * id_on_proc)) *
359 first_dir +
360 (iy * step_second_dir + random_number_position.at(3 * id_on_proc + 1)) *
361 second_dir +
362 (iz * step_third_dir + random_number_position.at(3 * id_on_proc + 2)) *
363 third_dir;
364 insertion_points_on_proc.push_back(temp_point);
365
366 temp_dir = ref_displacement_dir +
367 std::sin(random_number_angular_2.at(id_on_proc)) *
368 (std::cos(random_number_angular_1.at(id_on_proc)) * u +
369 std::sin(random_number_angular_1.at(id_on_proc)) * v);
370
371 for (unsigned int d = 0; d < 3; ++d)
372 {
373 properties_of_one_photon[d] = temp_point[d];
374 properties_of_one_photon[d + 3] = temp_dir[d];
375 }
376 photon_properties[id_on_proc] = properties_of_one_photon;
377 }
378 MPI_Comm communicator = triangulation.get_mpi_communicator();
379 // Obtain global bounding boxes
380 const auto my_bounding_box = GridTools::compute_mesh_predicate_bounding_box(
381 triangulation, IteratorFilters::LocallyOwnedCell());
382 const auto global_bounding_boxes =
383 Utilities::MPI::all_gather(communicator, my_bounding_box);
384
385 // Insert the photons using the points and the assigned properties.
386 photon_handler.insert_global_particles(insertion_points_on_proc,
387 global_bounding_boxes,
388 photon_properties);
389
390 ConditionalOStream pcout(std::cout,
391 Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) ==
392 0);
393 print_insertion_info(n_total_photons_to_insert, pcout);
394}

396template <int dim>
397void
398RayTracingSolver<dim>::write_output_results(
399 const std::vector<Point<dim>> &points)
400{
401 TimerOutput::Scope t(this->computing_timer, "Output VTU");
402
403 // Check if the output directory exists
404 std::string output_dir_name = parameters.simulation_control.output_folder;
405 struct stat buffer;
406
407 // If output directory does not exist, create it
408 if (this_mpi_process == 0)
409 {
410 if (stat(output_dir_name.c_str(), &buffer) != 0)
411 {
412 create_output_folder(output_dir_name);
413 }
414 }
415
416 // Create a temporary particle handler on the existing triangulation
417 Particles::ParticleHandler<dim> temp_handler(triangulation, mapping, 0);
418
419 MPI_Comm communicator = triangulation.get_mpi_communicator();
420 // Obtain global bounding boxes
421 const auto my_bounding_box = GridTools::compute_mesh_predicate_bounding_box(
422 triangulation, IteratorFilters::LocallyOwnedCell());
423 const auto global_bounding_boxes =
424 Utilities::MPI::all_gather(communicator, my_bounding_box);
425
426 std::vector<std::vector<double>> photon_properties(points.size());
427
428 // Insert the photons using the points and the assigned properties.
429 temp_handler.insert_global_particles(points,
430 global_bounding_boxes,
431 photon_properties);
432
433 // Write particles to a VTU file
434 Particles::DataOut<dim> data_out;
435 data_out.build_patches(temp_handler);
436
437 const std::string filename = parameters.simulation_control.output_folder +
438 "/" + parameters.simulation_control.output_name +
439 ".vtu";
440
441 data_out.write_vtu_in_parallel(filename, this->mpi_communicator);
442}

469template <int dim>
470template <typename NeighborListType>
471void
472RayTracingSolver<dim>::find_intersection(
473 NeighborListType &cell_list,
474 ankerl::unordered_dense::map<
475 types::particle_index,
476 std::tuple<double, Point<dim>, Particles::ParticleIterator<dim>>>
477 &photon_intersection_points_map)
478{
479 // Loop over each local cell in the triangulation. To do this, we loop
480 // over each cell neighbor list. Each local cell has a list and the first
481 // iterator is the main cell itself.
482 for (auto cell_neighbor_list_iterator = cell_list.begin();
483 cell_neighbor_list_iterator != cell_list.end();
484 ++cell_neighbor_list_iterator)
485 {
486 // The main cell
487 const auto main_cell = cell_neighbor_list_iterator->begin();
488
489 // Photons in main cell
490 typename Particles::ParticleHandler<dim>::particle_iterator_range
491 photons_in_main_cell = photon_handler.particles_in_cell(*main_cell);
492
493 // Loop on each photon in the main cell
494 for (auto current_photon = photons_in_main_cell.begin();
495 current_photon != photons_in_main_cell.end();
496 ++current_photon)
497 {
498 // Get the photon properties. We need the properties to find the
499 // insertion position of the current photon. This will be used to
500 // differentiate between intersection points when there's more
501 // than one.
502 auto photon_properties = current_photon->get_properties();
503
504 // Current photon insertion point.
505 const Point<3> photon_insertion_point(
506 {photon_properties[0], photon_properties[1], photon_properties[2]});
507
508 // Displacement vector of the current photon
509 Tensor<1, 3> photon_displacement_vector(
510 {photon_properties[3], photon_properties[4], photon_properties[5]});
511
512 const Point<dim> current_photon_location =
513 current_photon->get_location();
514
515 // Loop over the neighboring cells of the main cell. This includes
516 // the main cell itself.
517 auto &cell_neighbor_list = *cell_neighbor_list_iterator;
518
519 auto starting_iterator = cell_neighbor_list.begin();
520
521 // The intersection between the photon and the particles in the main
522 // cell is checked when we are looping on the local cells, thus we
523 // skip the main cell when we are looping on the ghost cells.
524 if constexpr (std::is_same_v<NeighborListType, ghost_neighbor_list>)
525 ++starting_iterator;
526 for (auto current_neighboring_cell = starting_iterator;
527 current_neighboring_cell != cell_neighbor_list.end();
528 ++current_neighboring_cell)
529 {
530 // Particle iterators in the current neighboring cell
531 typename Particles::ParticleHandler<dim>::particle_iterator_range
532 particle_in_neighboring_cell =
533 particle_handler.particles_in_cell(*current_neighboring_cell);
534
535 // This vector will be reused for every line-sphere
536 // intersection search. There is maximum 2 intersection points
537 // between a line and a sphere.
538 std::vector<Point<dim>> current_intersection_points;
539 current_intersection_points.reserve(2);
540
541 // Loop of the particles in the current neighboring cells
542 for (auto current_particle = particle_in_neighboring_cell.begin();
543 current_particle != particle_in_neighboring_cell.end();
544 ++current_particle)
545 {
546 // Check if the current photon intersects the current
547 // particle
548 const Point<dim> particle_position =
549 current_particle->get_location();
550 const double particle_diameter =
551 current_particle
552 ->get_properties()[DEMProperties::PropertiesIndex::dp];
553
554 current_intersection_points =
555 LetheGridTools::find_line_sphere_intersection(
556 photon_insertion_point,
557 photon_displacement_vector,
558 particle_position,
559 particle_diameter);
560
561 // If the size of the vector is 0, there's no intersection
562 // point, thus we go to the next particle.
563 if (current_intersection_points.size() == 0)
564 continue;
565
566 // Otherwise, there is at least one intersection point.
567 // Initializing the needed variables.
568 double new_distance;
569 Point<dim> new_closest_point;
570
571 // If the size of the vector is 1, we have one
572 // intersection point.
573 if (current_intersection_points.size() == 1)
574 {
575 new_distance = (photon_insertion_point -
576 current_intersection_points[0])
577 .norm();
578 new_closest_point = current_intersection_points[0];
579 }
580 // If the size is equal to 2, we need to find which one
581 // between the two new intersection points is the closest
582 // to the photon insertion point.
583 else
584 {
585 const double distance_0 = (photon_insertion_point -
586 current_intersection_points[0])
587 .norm();
588 const double distance_1 = (photon_insertion_point -
589 current_intersection_points[1])
590 .norm();
591
592 if (distance_0 < distance_1)
593 {
594 new_distance = distance_0;
595 new_closest_point = current_intersection_points[0];
596 }
597 else
598 {
599 new_distance = distance_1;
600 new_closest_point = current_intersection_points[1];
601 }
602 }
603 // We need to add the intersection point in the
604 // photon_insertion_point map. We first need to check
605 // if an intersection point already exist in the map
606 // for that given photon.
607 if (photon_intersection_points_map.find(
608 current_photon->get_id()) ==
609 photon_intersection_points_map.end())
610 {
611 // If it doesn't exist, we just add the new one.
612 photon_intersection_points_map[current_photon->get_id()] =
613 {new_distance, new_closest_point, current_photon};
614 }
615 else
616 {
617 // If it does exist, we need to check which
618 // intersection point is the closest to the
619 // current photon insertion point. We replace the
620 // value only if the new one is closer.
621 const double old_distance = std::get<0>(
622 photon_intersection_points_map[current_photon
623 ->get_id()]);
624
625 if (new_distance < old_distance)
626 {
627 // if the new distance is smaller, we replace the
628 // old one in the map. Otherwise, we do nothing.
629 photon_intersection_points_map[current_photon
630 ->get_id()] = {
631 new_distance, new_closest_point, current_photon};
632 }
633 }
634 }
635 }
636 // Even if we are removing photon at the end on this pseudo-time
637 // step, we move the photons here since we are looping on each of
638 // them in this loop. This way, we don't need to loop on each of
639 // them at the end.
640 if constexpr (std::is_same_v<NeighborListType, local_neighbor_list>)
641 {
642 const Point<dim> new_photon_location =
643 current_photon_location +
644 photon_displacement_vector * displacement_distance;
645 current_photon->set_location(new_photon_location);
646 }
647 }
648 }
649}

652template <int dim>
653void
654RayTracingSolver<dim>::solve()
655{
656 // Set up the parameters
657 setup_parameters();
658
659 // Reading mesh
660 read_mesh(parameters.mesh,
661 action_manager->check_restart_simulation(),
662 pcout,
663 triangulation,
664 dem_parameters.boundary_conditions);
665
666 displacement_distance = 0.5 * GridTools::minimal_cell_diameter(triangulation);
667
668 // Set particle insertion object
669 particle_insertion_object = set_particle_insertion_type();
670
671 // Insert photon and particles
672 insert_particles_and_photons();
673
674 // Particles won't move, thus we can sort them.
675 particle_handler.sort_particles_into_subdomains_and_cells();
676 // Exchange ghost particles
677 particle_handler.exchange_ghost_particles(true);
678
679 // Every intersection point from the beginning of the simulation.
680 std::vector<Point<dim>> total_intersection_points;
681 total_intersection_points.reserve(photon_handler.n_global_particles());
682
683 // Map :
684 // <photon_iterator, (distance, intersection point<dim>, iterator to remove)>
685 ankerl::unordered_dense::map<
686 types::particle_index,
687 std::tuple<double, Point<dim>, Particles::ParticleIterator<dim>>>
688 photon_intersection_points_map;
689
690 photon_intersection_points_map.reserve(
691 photon_handler.n_locally_owned_particles());
692
693 // Update cell neighbors
694 find_cell_neighbors<dim, true>(triangulation,
695 cells_local_neighbor_list,
696 cells_ghost_neighbor_list);
697
698 // Particle don't move, thus we sort them at the end of a load balance.
699 // Photons are being sorted every pseudo time step, thus no need to sort them
700 // here.
701 particle_handler.sort_particles_into_subdomains_and_cells();
702
703 // Exchange ghost particles
704 particle_handler.exchange_ghost_particles(true);
705
706 while (simulation_control->integrate())
707 {
708 simulation_control->print_progression(pcout);
709
710 // Load balancing (if needed)
711 load_balance();
712 // Since last pseudo time step, there is a high chance that the photons
713 // have changed cell considering the size of their displacement relative
714 // to the cell size. Thus, we sort them.
715
716 photon_handler.sort_particles_into_subdomains_and_cells();
717
718 find_intersection(cells_local_neighbor_list,
719 photon_intersection_points_map);
720
721 find_intersection(cells_ghost_neighbor_list,
722 photon_intersection_points_map);
723
724 // Remove all the photon that have found their intersection. In other
725 // words, every photon that is present in the map. We also store the
726 // intersection points in the appropriate vector.
727 std::vector<Particles::ParticleIterator<dim>> photon_iterators_to_remove;
728 photon_iterators_to_remove.reserve(photon_intersection_points_map.size());
729
730 for (auto it = photon_intersection_points_map.begin();
731 it != photon_intersection_points_map.end();
732 ++it)
733 {
734 auto photon_tuple = it->second;
735 total_intersection_points.push_back(std::get<1>(photon_tuple));
736 photon_iterators_to_remove.push_back(std::get<2>(photon_tuple));
737 }
738
739 // Removes the photon and clear the containers.
740 photon_handler.remove_particles(photon_iterators_to_remove);
741 photon_iterators_to_remove.clear();
742 photon_intersection_points_map.clear();
743
744 action_manager->reset_triggers();
745 }
746
747 // We don't want to write an output file if we are in test mode.
748 if (!parameters.test.enabled)
749 write_output_results(total_intersection_points);
750
751 finish_simulation(total_intersection_points);
752}
