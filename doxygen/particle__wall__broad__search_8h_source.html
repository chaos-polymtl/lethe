201/**
202 * @brief Store the candidate particle-wall contact pairs.
203 *
204 * @param particle_iterator The particle iterator to start boundary id.
205 * @param boundary_cells_content The info of the current boundary cell to
206 * store.
207 * @param contact_pair_candidates A map which will contain all the
208 * particle/wall pairs candidate.
209 */
210template <int dim>
211inline void
212store_candidates(
213 const typename Particles::ParticleHandler<
214 dim>::particle_iterator_range::iterator &particle_iterator,
215 const boundary_cells_info_struct<dim> &boundary_cells_content,
216 typename DEM::dem_data_structures<dim>::particle_wall_candidates
217 &contact_pair_candidates)
218{
219 // Find the contact candidate container of the particle
220 const types::particle_index particle_id = particle_iterator->get_id();
221 auto candidates_container_it = contact_pair_candidates.find(particle_id);
222
223 // Reserve arbitrary vector capacity and store if the particle does not have
224 // contact candidate yet
225 if (candidates_container_it == contact_pair_candidates.end())
226 {
227 auto pair_it_bool = contact_pair_candidates.emplace(
228 particle_id,
229 ankerl::unordered_dense::map<
230 DEM::global_face_id,
231 std::tuple<Particles::ParticleIterator<dim>,
232 Tensor<1, dim>,
233 Point<dim>,
234 DEM::global_face_id>>());
235
236 candidates_container_it = pair_it_bool.first;
237 }
238
239 // Store particle ids from the selected particle iterator
240 candidates_container_it->second.emplace(
241 boundary_cells_content.global_face_id,
242 std::make_tuple(particle_iterator,
243 boundary_cells_content.normal_vector,
244 boundary_cells_content.point_on_face,
245 boundary_cells_content.boundary_id));
246}
