32/**
33 * @brief Class that stores the information required by the assembly procedure
34 * for a Navier-Stokes equation. Consequently, this class calculates
35 * the velocity (values, gradients, laplacians) and the shape function
36 * (values, gradients, laplacians) at all the gauss points for all degrees
37 * of freedom and stores it into arrays. Additionally, the use can request
38 * that this class gathers additional fields for physics which are coupled
39 * to the Navier-Stokes equation, such as the VOF. This class
40 * serves as a separation between the evaluation at the gauss point of the
41 * variables of interest and their use in the assembly, which is carried out
42 * by the assembler functions. For more information on this design, the reader
43 * can consult deal.II step-9
44 * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
45 * example, the scratch is a struct instead of a templated class because of the
46 * simplicity of step-9.
47 *
48 * @tparam dim An integer that denotes the dimension of the space in which
49 * the flow is solved
50 *
51 * @ingroup solvers
52 **/
53
54template <int dim>
55class NavierStokesScratchData : public PhysicsScratchDataBase
56{
57public:
58 /**
59 * @brief Constructor. The constructor creates the fe_values that will be used
60 * to fill the member variables of the scratch. It also allocated the
61 * necessary memory for all member variables. However, it does not do any
62 * evaluation, since this needs to be done at the cell level.
63 *
64 * @param simulation_control The SimulationControl object that holds
65 * information related to the control of the steady-state or transient
66 * simulation. This is used to extrapolate auxiliary physics solutions in time
67 * for transient simulation.
68 *
69 * @param properties_manager The PhysicalPropertiesManager object that stores
70 * physical property models.
71 *
72 * @param fe The FESystem used to solve the Navier-Stokes equations
73 *
74 * @param quadrature The quadrature to use for the assembly
75 *
76 * @param mapping The mapping of the domain in which the Navier-Stokes
77 * equations are solved
78 */
79 NavierStokesScratchData(
80 const std::shared_ptr<SimulationControl> &simulation_control,
81 const PhysicalPropertiesManager &properties_manager,
82 const FESystem<dim> &fe,
83 const Quadrature<dim> &quadrature,
84 const Mapping<dim> &mapping,
85 const Quadrature<dim - 1> &face_quadrature)
86 : simulation_control(simulation_control)
87 , properties_manager(properties_manager)
88 , fe_values(mapping,
89 fe,
90 quadrature,
91 update_values | update_quadrature_points | update_JxW_values |
92 update_gradients | update_hessians)
93 , fe_face_values(mapping,
94 fe,
95 face_quadrature,
96 update_values | update_quadrature_points |
97 update_JxW_values | update_gradients | update_hessians |
98 update_normal_vectors)
99 , sdirk_table(
100 time_stepping_is_sdirk(simulation_control->get_assembly_method()) ?
101 ::sdirk_table(simulation_control->get_assembly_method()) :
102 SDIRKTable())

104 allocate();
105
106 // By default, the assembly of variables belonging to auxiliary physics is
107 // disabled.
108 gather_vof = false;
109 gather_projected_phase_fraction_gradient = false;
110 gather_curvature = false;
111 gather_void_fraction = false;
112 gather_particles_information = false;
113 gather_temperature = false;
114 gather_cahn_hilliard = false;
115 gather_mortar = false;
116 gather_particle_field_project = false;
117 gather_hessian = properties_manager.is_non_newtonian();

142 update_normal_vectors)

161 enable_vof(sd.fe_values_vof->get_fe(),

163 sd.fe_values_vof->get_mapping(),

213 /**
214 * @brief Reinitializes the content of the scratch.
215 *
216 * Using the FeValues and the content of the solutions and previous solutions,
217 * fills all of the class member of the scratch.
218 *
219 * @param[in] cell The cell over which the assembly is being carried.
220 * This cell must be compatible with the FE which is used to fill the
221 * FeValues.
222 *
223 * @param[in] current_solution The present value of the solution for [u,p].
224 *
225 * @param[in] previous_solutions The solutions at the previous time steps.
226 *
227 * @param[in] forcing_function The function describing the momentum/mass
228 * source term.
229 *
230 * @param[in] beta_force The additional force for flow control.
231 * TODO : Deprecate this argument and pass it to the constructor of the
232 * assembler
233 */
234
235 template <typename VectorType>
236 void
237 reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
238 const VectorType &current_solution,
239 const std::vector<VectorType> &previous_solutions,
240 const VectorType &sum_over_previous_stages,
241 std::shared_ptr<Function<dim>> forcing_function,
242 Tensor<1, dim> beta_force,
243 const double pressure_scaling_factor)
244 {
245 this->fe_values.reinit(cell);
246
247 quadrature_points = this->fe_values.get_quadrature_points();
248 auto &fe = this->fe_values.get_fe();
249
250 forcing_function->vector_value_list(quadrature_points, this->rhs_force);
251
252 // Establish the force vector
253 for (unsigned int q = 0; q < n_q_points; ++q)
254 {
255 for (int d = 0; d < dim; ++d)
256 {
257 const unsigned int component_i =
258 fe.system_to_component_index(d).first;
259 this->force[q][d] = this->rhs_force[q](component_i);
260 }
261
262 const unsigned int component_mass =
263 fe.system_to_component_index(dim).first;
264 this->mass_source[q] = this->rhs_force[q](component_mass);
265
266 // Correct force to include the dynamic forcing term for flow
267 // control
268 force[q] = force[q] + beta_force;
269 }
270
271 for (const unsigned int k : fe_values.dof_indices())
272 {
273 components[k] = fe.system_to_component_index(k).first;
274 }
275
276 // Compute cell diameter
277 double cell_measure =
278 compute_cell_measure_with_JxW(this->fe_values.get_JxW_values());
279 this->cell_size = compute_cell_diameter<dim>(cell_measure, fe.degree);
280
281 // For the SDIRK methods, \sum_{j=1}^{i-1} a_{ij} k_j is needed for the
282 // assembler
283 if (this->simulation_control->is_sdirk())
284 {
285 this->fe_values[velocities].get_function_values(
286 sum_over_previous_stages, this->sdirk_stage_sum);
287 }
288
289 // Gather velocity (values, gradient and laplacian)
290 this->fe_values[velocities].get_function_values(current_solution,
291 this->velocity_values);
292 this->fe_values[velocities].get_function_gradients(
293 current_solution, this->velocity_gradients);
294 this->fe_values[velocities].get_function_laplacians(
295 current_solution, this->velocity_laplacians);
296 if (gather_hessian)
297 this->fe_values[velocities].get_function_hessians(
298 current_solution, this->velocity_hessians);
299
300 // Gather velocity for stabilization (same as velocity_values unless ALE is
301 // enabled. The ALE correction is made within the correponding reinit
302 // function)
303 this->velocity_for_stabilization = this->velocity_values;
304
305 for (unsigned int q = 0; q < this->n_q_points; ++q)
306 {
307 this->velocity_divergences[q] = trace(this->velocity_gradients[q]);
308 }
309
310 // Gather pressure (values, gradient)
311 fe_values[pressure].get_function_values(current_solution,
312 this->pressure_values);
313 fe_values[pressure].get_function_gradients(current_solution,
314 this->pressure_gradients);
315 this->pressure_scaling_factor = pressure_scaling_factor;
316
317 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
318 {
319 this->fe_values[velocities].get_function_values(
320 previous_solutions[p], previous_velocity_values[p]);
321 }
322
323 // Only gather the pressure when a pressure history is necessary
324 // (compressible Navier-Stokes)
325 if (!this->properties_manager.density_is_constant())
326 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
327 {
328 this->fe_values[pressure].get_function_values(
329 previous_solutions[p], previous_pressure_values[p]);
330 }
331
332 for (unsigned int q = 0; q < n_q_points; ++q)
333 {
334 this->JxW[q] = this->fe_values.JxW(q);
335 for (unsigned int k = 0; k < n_dofs; ++k)
336 {
337 // Velocity
338 this->phi_u[q][k] = this->fe_values[velocities].value(k, q);
339 this->div_phi_u[q][k] =
340 this->fe_values[velocities].divergence(k, q);
341 this->grad_phi_u[q][k] = this->fe_values[velocities].gradient(k, q);
342 this->hess_phi_u[q][k] = this->fe_values[velocities].hessian(k, q);
343 for (int d = 0; d < dim; ++d)
344 this->laplacian_phi_u[q][k][d] = trace(this->hess_phi_u[q][k][d]);
345 // Pressure
346 this->phi_p[q][k] = this->fe_values[pressure].value(k, q);
347 this->grad_phi_p[q][k] = this->fe_values[pressure].gradient(k, q);
348 }
349 }
350
351 is_boundary_cell = cell->at_boundary();
352 if (is_boundary_cell)
353 {
354 n_faces = cell->n_faces();
355 is_boundary_face = std::vector<bool>(n_faces, false);
356 n_faces_q_points = fe_face_values.get_quadrature().size();
357 boundary_face_id = std::vector<unsigned int>(n_faces);
358
359 face_JxW.reinit(n_faces, n_faces_q_points);
360
361 // Velocity and pressure values
362 // First vector is face number, second quadrature point
363 this->face_velocity_values = std::vector<std::vector<Tensor<1, dim>>>(
364 n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
365
366 this->face_velocity_divergences = std::vector<std::vector<double>>(
367 n_faces, std::vector<double>(n_faces_q_points));
368
369 this->face_velocity_gradients =
370 std::vector<std::vector<Tensor<2, dim>>>(
371 n_faces, std::vector<Tensor<2, dim>>(n_faces_q_points));
372
373 this->face_velocity_laplacians =
374 std::vector<std::vector<Tensor<1, dim>>>(
375 n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
376
377 this->face_pressure_values = std::vector<std::vector<double>>(
378 n_faces, std::vector<double>(n_faces_q_points));
379
380 this->face_pressure_gradients =
381 std::vector<std::vector<Tensor<1, dim>>>(
382 n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
383
384 this->face_normal.reinit(n_faces, n_faces_q_points);
385
386 this->face_quadrature_points = std::vector<std::vector<Point<dim>>>(
387 n_faces, std::vector<Point<dim>>(n_faces_q_points));
388
389 this->face_div_phi_u.reinit(n_faces, n_faces_q_points, n_dofs);
390
391 this->face_phi_u.reinit(n_faces, n_faces_q_points, n_dofs);
392 this->face_hess_phi_u.reinit(n_faces, n_faces_q_points, n_dofs);
393
394 this->face_laplacian_phi_u.reinit(n_faces, n_faces_q_points, n_dofs);
395
396 this->face_grad_phi_u.reinit(n_faces, n_faces_q_points, n_dofs);
397 this->face_phi_p.reinit(n_faces, n_faces_q_points, n_dofs);
398 this->face_grad_phi_p.reinit(n_faces, n_faces_q_points, n_dofs);
399
400 for (const auto face : cell->face_indices())
401 {
402 is_boundary_face[face] = cell->face(face)->at_boundary();
403 if (is_boundary_face[face])
404 {
405 fe_face_values.reinit(cell, face);
406 n_dofs = fe_face_values.get_fe().n_dofs_per_cell();
407 boundary_face_id[face] = cell->face(face)->boundary_id();
408 // Shape functions
409 // First vector is face number, second quadrature point, third
410 // DOF
411
412 // Gather velocity (values, gradient and laplacian)
413 this->fe_face_values[velocities].get_function_values(
414 current_solution, this->face_velocity_values[face]);
415 this->fe_face_values[velocities].get_function_gradients(
416 current_solution, this->face_velocity_gradients[face]);
417 this->fe_face_values[velocities].get_function_laplacians(
418 current_solution, this->face_velocity_laplacians[face]);
419
420 // Gather pressure (values, gradient)
421 this->fe_face_values[pressure].get_function_values(
422 current_solution, this->face_pressure_values[face]);
423 this->fe_face_values[pressure].get_function_gradients(
424 current_solution, this->face_pressure_gradients[face]);
425
426 for (unsigned int q = 0; q < n_faces_q_points; ++q)
427 {
428 this->face_JxW[face][q] = this->fe_face_values.JxW(q);
429 this->face_normal[face][q] =
430 this->fe_face_values.normal_vector(q);
431 this->face_quadrature_points[face][q] =
432 this->fe_face_values.quadrature_point(q);
433 for (const unsigned int k : fe_face_values.dof_indices())
434 {
435 // Velocity
436 this->face_phi_u[face][q][k] =
437 this->fe_face_values[velocities].value(k, q);
438 this->face_div_phi_u[face][q][k] =
439 this->fe_face_values[velocities].divergence(k, q);
440 this->face_grad_phi_u[face][q][k] =
441 this->fe_face_values[velocities].gradient(k, q);
442 this->face_hess_phi_u[face][q][k] =
443 this->fe_face_values[velocities].hessian(k, q);
444 for (int d = 0; d < dim; ++d)
445 this->face_laplacian_phi_u[face][q][k][d] =
446 trace(this->face_hess_phi_u[face][q][k][d]);
447 // Pressure
448 this->face_phi_p[face][q][k] =
449 this->fe_face_values[pressure].value(k, q);
450 this->face_grad_phi_p[face][q][k] =
451 this->fe_face_values[pressure].gradient(k, q);
452 }
453 }
454 }
455 }
456 }
457 }

537 this->filtered_phase_values);

539 current_filtered_solution, this->filtered_phase_gradient_values);
540 this->fe_values_vof->get_function_gradients(current_solution,
541 this->phase_gradient_values);

543 // Gather previous phase fraction values
544 for (unsigned int p = 0; p < previous_solutions.size(); ++p)

546 this->fe_values_vof->get_function_values(previous_solutions[p],

842 * calculated based on the filtered VOF solution interpolated at the
843 * location of the particles. These properties are used in the forces
844 * calculations in the VANS equations.
845 */

847 void
848 calculate_fluid_properties_at_particle_location()
849 {
850 AssertThrow(
851 properties_manager.density_is_constant(),
852 ExcMessage(
853 "The lethe-fluid-particles solver only supports constant densities"));
854 AssertThrow(
855 !properties_manager.get_rheology()
856 ->is_non_newtonian_rheological_model() &&
857 !properties_manager.get_rheology()

868 filtered_phase_values_at_particle_location[i_particle],

872 calculate_point_property(

899 /**
900 * @brief Calculates the velocity of the fluid relative to that of the particle and
901 * the particle Reynolds number at the location of the particles. These will
902 * be used in forces calculations in the vans equations.
903 */
904
905 void
906 calculate_force_parameters_at_particle_location()
907 {
908 // Tolerance for the calculation of the particle Reynolds number.
909 // TODO -> Revisit if that is not a too high value.
910 const double Re_particle_tolerance = 1e-3;
911 unsigned int i_particle = 0;
912 average_fluid_particle_relative_velocity = 0;
913
914 for (auto &particle : pic)
915 {
916 auto particle_properties = particle.get_properties();
917
918 fluid_particle_relative_velocity_at_particle_location[i_particle] =
919 fluid_velocity_at_particle_location[i_particle] -
920 particle_velocity[i_particle];
921 average_fluid_particle_relative_velocity +=
922 fluid_particle_relative_velocity_at_particle_location[i_particle];
923
924 Re_particle[i_particle] =
925 Re_particle_tolerance +
926 cell_void_fraction[i_particle] *
927 fluid_particle_relative_velocity_at_particle_location[i_particle]
928 .norm() *
929 particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp] /
930 (kinematic_viscosity_at_particle_location[i_particle] + DBL_MIN);
931 i_particle++;
932 }
933
934 average_fluid_particle_relative_velocity =
935 average_fluid_particle_relative_velocity / i_particle;
936 }
