43 /**
44 * @brief Scalar function defined by the DOF values of a single cell. Based on
45 * the CellWiseFunction and RefSpaceFEFieldFunction of dealii.
46 *
47 * @tparam dim An integer that denotes the dimension of the space in which
48 * the problem is solved.
49 *
50 * @tparam VectorType The vector type of the solution vector.
51 *
52 * @tparam FEType The finite element type used to discretize the problem.
53 */
54 template <int dim,
55 typename VectorType = Vector<double>,
56 typename FEType = FE_Q<dim>>
57 class CellWiseFunction : public Function<dim>
58 {
59 public:
60 /**
61 * @brief Constructor.
62 *
63 * @param[in] p_fe_degree Finite element degree discretizing the field we
64 * want to convert to a CellWiseFunction.
65 *
66 */
67 CellWiseFunction(const unsigned int p_fe_degree);
68
69 /**
70 * @brief Set the cell that the function should be evaluated on.
71 *
72 * @param[in] in_local_dof_values Cell's DOF values
73 *
74 */
75 void
76 set_active_cell(const VectorType &in_local_dof_values);
77
78 /**
79 * @brief Return the value of the function at the given point in the reference cell.
80 *
81 * @param[in] point Coordinates of the point in the reference cell
82 *
83 * @param[in] component Index of the component of the function for which the
84 * value is computed
85 *
86 * @return Value of the function (or the component of interest) at the given point
87 *
88 */
89 double
90 value(const Point<dim> &point,
91 const unsigned int component = 0) const override;
92
93 /**
94 * @brief Return the function gradient of the specified component at the
95 * given point in the reference cell.
96 *
97 * @param[in] point Coordinates of the point in the reference cell
98 *
99 * @param[in] component Index of the component of the function for which the
100 * gradient is computed
101 *
102 * @return Value of the gradient of the specified component at the given point
103 *
104 */
105 Tensor<1, dim>
106 gradient(const Point<dim> &point,
107 const unsigned int component = 0) const override;
108
109 /**
110 * @brief Return the function Hessian of the specified component at the given point in the reference cell.
111 *
112 * @param[in] point Coordinates of the point in the reference cell
113 *
114 * @param[in] component Index of the component of the function for which the
115 * Hessian is computed
116 *
117 * @return Value of the Hessian of the specified component at the given point
118 *
119 */
120 SymmetricTensor<2, dim>
121 hessian(const Point<dim> &point,
122 const unsigned int component = 0) const override;
123
124 private:
125 /// Finite element discretizing the field of interest
126 FEType fe;
127
128 /// Number of DoFs per element
129 unsigned int n_cell_wise_dofs;
130
131 /// Value of the level-set field at the DoFs of the cell
132 VectorType cell_dof_values;
133 };
134
135 template <int dim, typename VectorType, typename FEType>
136 CellWiseFunction<dim, VectorType, FEType>::CellWiseFunction(
137 const unsigned int p_fe_degree)
138 : fe(p_fe_degree)
139 {
140 n_cell_wise_dofs = fe.dofs_per_cell;
141 }
142
143 template <int dim, typename VectorType, typename FEType>
144 inline void
145 CellWiseFunction<dim, VectorType, FEType>::set_active_cell(
146 const VectorType &in_local_dof_values)
147 {
148 cell_dof_values = in_local_dof_values;
149 }
150
151 template <int dim, typename VectorType, typename FEType>
152 inline double
153 CellWiseFunction<dim, VectorType, FEType>::value(
154 const Point<dim> &point,
155 const unsigned int component) const
156 {
157 double value = 0;
158 for (unsigned int i = 0; i < n_cell_wise_dofs; ++i)
159 value +=
160 cell_dof_values[i] * fe.shape_value_component(i, point, component);
161
162 return value;
163 }
164
165 template <int dim, typename VectorType, typename FEType>
166 inline Tensor<1, dim>
167 CellWiseFunction<dim, VectorType, FEType>::gradient(
168 const Point<dim> &point,
169 const unsigned int component) const
170 {
171 Tensor<1, dim> gradient;
172 for (unsigned int i = 0; i < n_cell_wise_dofs; ++i)
173 gradient +=
174 cell_dof_values[i] * fe.shape_grad_component(i, point, component);
175
176 return gradient;
177 }
178
179 template <int dim, typename VectorType, typename FEType>
180 inline SymmetricTensor<2, dim>
181 CellWiseFunction<dim, VectorType, FEType>::hessian(
182 const Point<dim> &point,
183 const unsigned int component) const
184 {
185 Tensor<2, dim> hessian;
186 for (unsigned int i = 0; i < n_cell_wise_dofs; ++i)
187 hessian +=
188 cell_dof_values[i] * fe.shape_grad_grad_component(i, point, component);
189
190 return symmetrize(hessian);
191 }

399 * @brief Solver to compute the signed distance from a given level of a level-set field. It is based on the method proposed in the following article: Ausas, R.F., Dari, E.A. and Buscaglia, G.C. 605 /**

669 /**
670 * @brief Return the face transformation jacobian (dim-1 x dim-1). This is
671 * required because the distance minimization problem is resolved in the
672 * reference face space (dim-1).
673 *
674 * @param[in] cell_transformation_jac transformation jacobian of the cell
675 * (dim x dim)
676 *
677 * @param[in] local_face_id local id of the face
678 *
679 * @param[out] face_transformation_jac face transformation jacobian (dim-1 x
680 * dim-1) faces
681 */
682 inline void
683 get_face_transformation_jacobian(
684 const DerivativeForm<1, dim, dim> &cell_transformation_jac,
685 const unsigned int local_face_id,
686 DerivativeForm<1, dim - 1, dim> &face_transformation_jac)
687 {
688 for (unsigned int i = 0; i < dim; ++i)
689 {
690 unsigned int k = 0;
691 for (unsigned int j = 0; j < dim; ++j)
692 {
693 if (local_face_id / 2 == j)
694 continue;
695 face_transformation_jac[i][k] = cell_transformation_jac[i][j];
696 k += 1;
697 }
698 }
699 };

701 /**
702 * @brief
703 * Transform a point dim-1 in a reference face to a point dim in the
704 * reference cell. This is required because the distance minimization
705 * problem is resolved in the reference face space (dim-1).
706 *
707 * @param[in] x_ref_face point dim-1 in the reference face
708 *
709 * @param[in] local_face_id local id of the face
710 *
711 * @return Point dim in the reference cell
712 */
713 inline Point<dim>
714 transform_ref_face_point_to_ref_cell(const Point<dim - 1> &x_ref_face,
715 const unsigned int local_face_id)
716 {
717 Point<dim> x_ref_cell;
718
719 unsigned int j = 0;
720 for (unsigned int i = 0; i < dim; ++i)
721 {
722 if (local_face_id / 2 == i)
723 {
724 x_ref_cell[i] = double(local_face_id % 2);
725 continue;
726 }
727 x_ref_cell[i] = x_ref_face[j];
728 j += 1;
729 }
730
731 return x_ref_cell;
732 };

734 /**
735 * @brief Compute the residual at the point x_n of the distance minimization
736 * problem for the DoF x_I in the reference face space (dim - 1). In the
737 * real space the residual correspond to: R = grad(d) - (x_I - x_n)/||x_I -
738 * x_n||
739 *
740 *
741 * @param[in] x_n_to_x_I_real vector from x_n to x_I in the real space
742 *
743 * @param[in] distance_gradient gradient of the distance at the point x_n
744 *
745 * @param[in] transformation_jac transformation jacobian of the face
746 *
747 * @param[out] residual_ref residual in the reference face space
748 */
749 inline void
750 compute_residual(const Tensor<1, dim> &x_n_to_x_I_real,
751 const Tensor<1, dim> &distance_gradient,
752 const DerivativeForm<1, dim - 1, dim> &transformation_jac,
753 Tensor<1, dim - 1> &residual_ref)
754 {
755 Tensor<1, dim> residual_real =
756 distance_gradient - (1.0 / x_n_to_x_I_real.norm()) * x_n_to_x_I_real;
757
758 DerivativeForm<1, dim, dim - 1> transformation_jac_transpose =
759 transformation_jac.transpose();
760
761 for (unsigned int i = 0; i < dim - 1; ++i)
762 for (unsigned int j = 0; j < dim; ++j)
763 residual_ref[i] +=
764 transformation_jac_transpose[i][j] * residual_real[j];
765 };

767 /**
768 * @brief Compute the stencil for the numerical jacobian of the distance
769 * minimization problem in the face where the problem is solved
770 *
771 * @param[in] x_ref point in the reference cell space where the jacobian is
772 * evaluated
773 *
774 * @param[in] local_face_id local id of the face in which the numerical
775 * jacobian is required
776 *
777 * @param[in] perturbation value of the perturbation use for the
778 * numerical jacobian computation
779 *
780 * @return vector containing the 2*dim - 1 stencil point. The entries of the
781 * vector are the following:
782 * 4
783 *
784 * 1 0 2
785 *
786 * 3
787 * The entry 0 is the current evaluation point x_ref
788 */
789 inline void
790 compute_numerical_jacobian_stencil(const Point<dim> x_ref,
791 const unsigned int local_face_id,
792 const double perturbation,
793 std::vector<Point<dim>> &stencil)
794 {
795 for (unsigned int i = 0; i < 2 * dim - 1; ++i)
796 {
797 stencil[i] = x_ref;
798 }
799
800 unsigned int skip_index = local_face_id / 2;
801 std::vector<unsigned int> j_index(dim - 1);
802
803 // Set the coordinates (x,y,z) to be skipped (the coordinates not
804 // perturbed)
805 unsigned int j = 0;
806 for (unsigned int i = 0; i < dim; ++i)
807 {
808 if (i == skip_index)
809 continue;
810 j_index[j] = i;
811 j += 1;
812 }
813
814 for (unsigned int i = 1; i < dim; ++i)
815 {
816 j = j_index[i - 1];
817 stencil[2 * i - 1][j] -= perturbation;
818 stencil[2 * i][j] += perturbation;
819 }
820 };
821
822 /**
823 * @brief
824 * Transform the Newton correction in a reference face to a tensor (dim) in
825 * the reference cell. This is required because the distance minimization
826 * problem is resolved in the reference face space (dim-1).
827 *
828 * @param[in] correction_ref_face Newton correction in the reference face
829 *
830 * @param[in] local_face_id local id of the face
831 *
832 * @return tensor dim in the reference cell
833 */
834 inline Tensor<1, dim>
835 transform_ref_face_correction_to_ref_cell(
836 const Vector<double> &correction_ref_face,
837 const unsigned int local_face_id)
838 {
839 Tensor<1, dim> correction_ref_cell;
840
841 unsigned int j = 0;
842 for (unsigned int i = 0; i < dim; ++i)
843 {
844 if (local_face_id / 2 == i)
845 {
846 correction_ref_cell[i] = 0.0;
847 continue;
848 }
849 correction_ref_cell[i] = correction_ref_face[j];
850 j += 1;
851 }
852
853 return correction_ref_cell;
854 };

856 /**
857 * @brief
858 * Compute the numerical jacobian at the point x_n of the distance
859 * minimization problem for the DoF x_I in the reference face space (dim -
860 * 1).
861 *
862 * @param[in] stencil_real stencil in the real space. The evaluation point
863 * x_n corresponds to the first entry.
864 *
865 * @param[in] x_I_real coordinate of the DoF x_I in the real space
866 *
867 * @param[in] distance_gradients vector storing the distance gradients at
868 * each stencil points
869 *
870 * @param[in] transformation_jacobians vector storing the face
871 * transformation jacobians at each stencil points
872 *
873 * @param[in] perturbation value of the perturbation use for the
874 * numerical jacobian computation
875 *
876 * @param[out] jacobian_matrix jacobian matrix of the minimization problem
877 */
878 inline void
879 compute_numerical_jacobian(
880 const std::vector<Point<dim>> &stencil_real,
881 const Point<dim> &x_I_real,
882 const std::vector<Tensor<1, dim>> &distance_gradients,
883 const std::vector<DerivativeForm<1, dim - 1, dim>>
884 &transformation_jacobians,
885 const double perturbation,
886 LAPACKFullMatrix<double> &jacobian_matrix)
887 {
888 for (unsigned int i = 0; i < dim - 1; ++i)
889 {
890 const Tensor<1, dim> x_n_to_x_I_real_m1 =
891 x_I_real - stencil_real[2 * i + 1];
892
893 Tensor<1, dim - 1> residual_ref_m1;
894 compute_residual(x_n_to_x_I_real_m1,
895 distance_gradients[2 * i + 1],
896 transformation_jacobians[2 * i + 1],
897 residual_ref_m1);
898
899 const Tensor<1, dim> x_n_to_x_I_real_p1 =
900 x_I_real - stencil_real[2 * i + 2];
901
902 Tensor<1, dim - 1> residual_ref_p1;
903 compute_residual(x_n_to_x_I_real_p1,
904 distance_gradients[2 * i + 2],
905 transformation_jacobians[2 * i + 2],
906 residual_ref_p1);
907
908 for (unsigned int j = 0; j < dim - 1; ++j)
909 {
910 jacobian_matrix.set(j,
911 i,
912 (residual_ref_p1[j] - residual_ref_m1[j]) /
913 (2.0 * perturbation));
914 }
915 }
916 };

918 /**
919 * @brief
920 * Compute the distance according to: d(x_I) = d(x_n) + ||x_I - x_n||
921 *
922 * @param[in] x_n_to_x_I_real vector from x_n to x_I in the real space
923 *
924 * @param[in] distance value of the distance at the point x_n
925 *
926 * @return distance between x_I and the interface
927 */
928 inline double
929 compute_distance(const Tensor<1, dim> &x_n_to_x_I_real,
930 const double distance)
931 {
932 return distance + x_n_to_x_I_real.norm();
933 };
