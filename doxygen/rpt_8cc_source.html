22template <int dim>
23void
24RPT<dim>::setup_and_calculate()
25{
26 // Reading and assigning positions of detectors and particles
27 detectors = assign_detector_positions<dim>(rpt_parameters.detector_param);
28 particle_positions = assign_particle_positions<dim>(
29 rpt_parameters.rpt_param.particle_positions_file);
30
31 // Calculate count for every particle-detector pair and transfer it to
32 // calculated_counts
33 calculate_counts();
34
35 // Export results in .csv if enable
36 if (rpt_parameters.rpt_param.export_counts)
37 export_data();
38
39 // Calculate the cost function for parameters tuning
40 if (rpt_parameters.tuning_param.tuning)
41 {
42 std::vector<double> measured_counts =
43 read_counts<dim>(rpt_parameters.tuning_param.experimental_file);
44 AssertThrow(
45 measured_counts.size() == particle_positions.size(),
46 ExcMessage(
47 "Prior tuning parameters, the number of particle positions provided"
48 " has to be the same number of counts of experimental data. "
49 "Note : The experimental counts also have to be at the same positions which can not be verified."));
50
51 double cost_function =
52 calculate_cost_function(measured_counts, calculated_counts);
53 std::cout << cost_function << std::endl;
54 }
55}

57template <int dim>
58void
59RPT<dim>::calculate_counts()
60{
61 // Calculate count for every particle-detector pair
62 for (unsigned int i_particle = 0; i_particle < particle_positions.size();
63 i_particle++)
64 {
65 for (unsigned int i_detector = 0; i_detector < detectors.size();
66 i_detector++)
67 {
68 // Create the particle-detector interaction object
69 ParticleDetectorInteractions<dim> particle_detector_interactions(
70 particle_positions[i_particle],
71 detectors[i_detector],
72
73 rpt_parameters.rpt_param);
74
75 double count = particle_detector_interactions.calculate_count();
76 calculated_counts.push_back(count);
77
78 if (rpt_parameters.rpt_param.verbosity ==
79 Parameters::Verbosity::verbose)
80 std::cout << "Count for particle position " << i_particle
81 << " and detector " << i_detector << " : " << count
82 << std::endl;
83 }
84 }
85}

88template <int dim>
89void
90RPT<dim>::export_data()
91{
92 // Open a file
93 std::ofstream myfile;
94 std::string sep;
95
96 std::string filename = rpt_parameters.rpt_param.export_counts_file;
97
98 // check if an extension is specified in the filename, if not add ".csv"
99 std::size_t csv_file = filename.find(".csv");
100 std::size_t dat_file = filename.find("dat");
101 if ((csv_file == std::string::npos) && (dat_file == std::string::npos))
102 filename += ".csv";
103
104 myfile.open(filename);
105 if (filename.substr(filename.find_last_of('.') + 1) == ".dat")
106 {
107 myfile
108 << "particle_positions_x particle_positions_y particle_positions_z detector_id counts"
109 << std::endl;
110 sep = " ";
111 }
112 else // .csv is default
113 {
114 myfile
115 << "particle_positions_x,particle_positions_y,particle_positions_z,detector_id,counts"
116 << std::endl;
117 sep = ",";
118 }
119
120 for (unsigned int i_particle = 0; i_particle < particle_positions.size();
121 i_particle++)
122 {
123 for (unsigned int i_detector = 0; i_detector < detectors.size();
124 i_detector++)
125 {
126 myfile
127 << particle_positions[i_particle].get_position()[0] << sep
128 << particle_positions[i_particle].get_position()[1] << sep
129 << particle_positions[i_particle].get_position()[2] << sep
130 << detectors[i_detector].get_id() << sep
131 << calculated_counts[i_particle * detectors.size() + i_detector]
132 << std::endl;
133 }
134 }
135 myfile.close();
136}

138template <int dim>
139double
140RPT<dim>::calculate_cost_function(std::vector<double> &measured_counts,
141 std::vector<double> &calculated_counts)
142{
143 double cost_function = 0;
144
145 if (rpt_parameters.tuning_param.cost_function_type ==
146 Parameters::RPTTuningParameters::CostFunctionType::larachi)
147 {
148 for (unsigned int i = 0; i < measured_counts.size(); i++)
149 {
150 cost_function +=
151 std::pow((calculated_counts[i] - measured_counts[i]) /
152 (calculated_counts[i] + measured_counts[i]),
153 2);
154 }
155 }
156 else if (rpt_parameters.tuning_param.cost_function_type ==
157 Parameters::RPTTuningParameters::CostFunctionType::l1)
158 {
159 for (unsigned int i = 0; i < measured_counts.size(); i++)
160 cost_function += abs(calculated_counts[i] - measured_counts[i]);
161
162 cost_function /= measured_counts.size();
163 }
164 else if (rpt_parameters.tuning_param.cost_function_type ==
165 Parameters::RPTTuningParameters::CostFunctionType::l2)
166 {
167 for (unsigned int i = 0; i < measured_counts.size(); i++)
168 cost_function +=
169 std::pow((calculated_counts[i] - measured_counts[i]), 2);
170
171 cost_function /= measured_counts.size();
172 }
173
174 return cost_function;
175}
