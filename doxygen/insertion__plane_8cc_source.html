10// The constructor of plane insertion class. In the constructor, we find which
11// cells are going to be use for the insertion and we also find the centers of
12// those cells.
13template <int dim, typename PropertiesIndex>
14InsertionPlane<dim, PropertiesIndex>::InsertionPlane(
15 const std::vector<std::shared_ptr<Distribution>>
16 &size_distribution_object_container,
17 const parallel::distributed::Triangulation<dim> &triangulation,
18 const DEMSolverParameters<dim> &dem_parameters)
19 : Insertion<dim, PropertiesIndex>(size_distribution_object_container,
20 triangulation,
21 dem_parameters)
22 , particles_of_each_type_remaining(
23 dem_parameters.lagrangian_physical_properties.number.at(0))
24{
25 // Initializing current inserting particle type
26 current_inserting_particle_type = 0;
27
28 // Finding which cells are inplane
29 this->find_inplane_cells(
30 triangulation,
31 dem_parameters.insertion_info.insertion_plane_point,
32 dem_parameters.insertion_info.insertion_plane_normal_vector);
33
34 // Finding the center of those cells
35 this->find_centers_of_inplane_cells();
36
37 // Initializing the variable for the random position of particle
38 maximum_range_for_randomness =
39 dem_parameters.insertion_info.insertion_maximum_offset /
40 static_cast<double>(RAND_MAX);
41}

43template <int dim, typename PropertiesIndex>
44void
45InsertionPlane<dim, PropertiesIndex>::find_inplane_cells(
46 const parallel::distributed::Triangulation<dim> &triangulation,
47 Point<3> plane_point,
48 Tensor<1, 3> plane_normal_vector)
49{
50 plane_cells_for_insertion.clear();
51
52 // Looping through cells
53 for (const auto &cell : triangulation.active_cell_iterators())
54 {
55 // If the cell is owned by the processor
56 if (cell->is_locally_owned())
57 {
58 // Vector that goes from the plane reference point to the vertex 0.
59 Tensor<1, 3> connecting_vector_ref =
60 point_nd_to_3d(cell->vertex(0)) - plane_point;
61
62 // Normal distance between vertex 0 and the plane.
63 double vertex_wall_distance_ref =
64 connecting_vector_ref * plane_normal_vector;
65
66 // Loop over all the vertices of the cell
67 for (unsigned int vertex_id = 1; vertex_id < cell->n_vertices();
68 ++vertex_id)
69 {
70 // Vector that goes from the plane reference point to the vertex
71 // n-th
72 Tensor<1, 3> connecting_vector =
73 point_nd_to_3d(cell->vertex(vertex_id)) - plane_point;
74
75 // Normal distance between vertex n-th and the plane
76 double vertex_wall_distance =
77 connecting_vector * plane_normal_vector;
78
79 // If the multiplication of the two distances is negative, then at
80 // least one of vertex of the cell is on both sides of the plane.
81 // If the multiplication is equal to zero, then one vertex is on
82 // the plane.
83 if (vertex_wall_distance_ref * vertex_wall_distance <= 0)
84 {
85 plane_cells_for_insertion.insert(cell);
86 break; // If so, we break the loop, and go to the next cell
87 }
88 }
89 }
90 }
91}

93template <int dim, typename PropertiesIndex>
94void
95InsertionPlane<dim, PropertiesIndex>::find_centers_of_inplane_cells()
96{
97 cells_centers.clear();
98 for (const auto &cell : plane_cells_for_insertion)
99 {
100 cells_centers.insert({cell->active_cell_index(), cell->center()});
101 }
102}

104// The main insertion function, insert_particle, is utilized to insert the
105// particle at the cell center with a random shifts particles
106template <int dim, typename PropertiesIndex>
107void
108InsertionPlane<dim, PropertiesIndex>::insert(
109 Particles::ParticleHandler<dim> &particle_handler,
110 const parallel::distributed::Triangulation<dim> &triangulation,
111 const DEMSolverParameters<dim> &dem_parameters)
112{
113 if (particles_of_each_type_remaining == 0 &&
114 this->current_inserting_particle_type !=
115 dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
116 {
117 particles_of_each_type_remaining =
118 dem_parameters.lagrangian_physical_properties.number.at(
119 ++current_inserting_particle_type);
120 }
121 if (particles_of_each_type_remaining > 0)
122 {
123 if (this->mark_for_update)
124 {
125 find_inplane_cells(
126 triangulation,
127 dem_parameters.insertion_info.insertion_plane_point,
128 dem_parameters.insertion_info.insertion_plane_normal_vector);
129 find_centers_of_inplane_cells();
130
131 if (this->removing_particles_in_region)
132 {
133 this->find_cells_in_removing_box(triangulation);
134 }
135 this->mark_for_update = false;
136 }
137 if (this->removing_particles_in_region)
138 {
139 this->remove_particles_in_box(particle_handler);
140 }
141
142 MPI_Comm communicator = triangulation.get_mpi_communicator();
143 ConditionalOStream pcout(
144 std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
145
146 unsigned int this_mpi_process =
147 Utilities::MPI::this_mpi_process(communicator);
148 unsigned int total_mpi_process =
149 Utilities::MPI::n_mpi_processes(communicator);
150 // List of the empty cells at this time step
151 std::set<typename Triangulation<dim>::active_cell_iterator>
152 empty_cells_on_proc;
153
154 // Loop over the cell inplane
155 for (const auto &cell : plane_cells_for_insertion)
156 {
157 // if the cell is empty ...
158 if (particle_handler.n_particles_in_cell(cell) == 0)
159 {
160 // ... we add the cell in std::set
161 empty_cells_on_proc.insert(cell);
162 }
163 }
164
165 // Processor zero knows how many particles every processor wants to insert
166 auto number_of_particles_to_insert_per_core =
167 Utilities::MPI::gather(communicator, empty_cells_on_proc.size(), 0);
168
169 std::vector<int> starting_IDs_on_every_proc(total_mpi_process);
170 std::vector<int> remained_particles_for_every_proc(total_mpi_process);
171 // Only processor zero does this
172 if (this_mpi_process == 0)
173 {
174 int starting_id_on_proc =
175 particle_handler.get_next_free_particle_index();
176
177 // we loop over the vector that we have created using the gather.
178 // Every integer in that vector gives the number of empty_cell of a
179 // certain processor.
180 for (unsigned int i = 0; i < total_mpi_process; ++i)
181 {
182 // We subtract those integer from the following variable
183 particles_of_each_type_remaining -=
184 number_of_particles_to_insert_per_core[i];
185
186 // If the particles_of_each_type_remaining variable become
187 // negative, this means that we are trying to insert too many
188 // particle in the simulation. Thus, we need to decrease the
189 // number of empty_cell we want to insert in at this time step.
190 if (particles_of_each_type_remaining < 0)
191 {
192 // At this point, we know the current processor have to many
193 // available cell to insert in (or just the right amount). The
194 // excess number of available cell on this processor is equal
195 // to the positive value of the
196 // particles_of_each_type_remaining variable since that
197 // variable became negative on this processor.
198
199 // Thus, we decrease the number of particle to insert on the
200 // current processor. To do so, we add the negative variable
201 // to the number of available cells which guarantee that the
202 // number of inserted particles in the simulation will reach
203 // its maximum on this processor.
204
205 number_of_particles_to_insert_per_core[i] +=
206 particles_of_each_type_remaining;
207
208 // We put the particles_of_each_type_remaining variable back
209 // to zero since we have just lower the number of cell
210 // available on this processor by the exceeding amount.
211 particles_of_each_type_remaining = 0;
212
213 starting_IDs_on_every_proc[i] = starting_id_on_proc;
214 starting_id_on_proc +=
215 number_of_particles_to_insert_per_core[i];
216
217 // Loop over the remaining processor and put their number of
218 // available cell to zero.
219 for (unsigned int j = i + 1; j < (total_mpi_process); ++j)
220 {
221 number_of_particles_to_insert_per_core[j] = 0;
222 starting_IDs_on_every_proc[j] = 0;
223 }
224 break;
225 }
226 starting_IDs_on_every_proc[i] = starting_id_on_proc;
227 starting_id_on_proc += number_of_particles_to_insert_per_core[i];
228 }
229 std::fill(remained_particles_for_every_proc.begin(),
230 remained_particles_for_every_proc.end(),
231 particles_of_each_type_remaining);
232 }
233
234 // Now, processor zero knows how many particles will be inserted by every
235 // processor without exceeding the maximum number of particles.
236
237 // We now need to scatter 3 information per processor, first how
238 // many particles it needs to insert. Second what is the ID of the
239 // first particle it will insert at this insertion step. The starting ID
240 // is mandatory, since we don't want two particles to have the same ID,
241 // and we don't want to use the global_insertion method because it uses an
242 // all_gather, which has a high computational cost. Third, the
243 // particles_of_each_type_remaining variable needs to be updated on
244 // processors other than processor 0 otherwise the insert function will
245 // become incoherent.
246
247 // This scatters the number of particles to insert
248 unsigned int number_of_particles_to_insert_on_this_core =
249 Utilities::MPI::scatter(communicator,
250 number_of_particles_to_insert_per_core,
251 0);
252
253 // This scatters the starting ID on every proc for the insertion
254 unsigned int starting_ID_on_proc =
255 Utilities::MPI::scatter(communicator, starting_IDs_on_every_proc, 0);
256
257 // This scatters particles_of_each_type_remaining on every proc
258 particles_of_each_type_remaining =
259 Utilities::MPI::scatter(communicator,
260 remained_particles_for_every_proc,
261 0);
262
263 // We yet didn't choose which empty cells won't be used anymore.
264
265 while (empty_cells_on_proc.size() >
266 number_of_particles_to_insert_on_this_core)
267 {
268 auto it = empty_cells_on_proc
269 .begin(); // Get the iterator of the first element
270 empty_cells_on_proc.erase(it); // Erase the first element
271 }
272
273 std::vector<Point<dim>> insertion_points_on_proc;
274 insertion_points_on_proc.reserve(
275 number_of_particles_to_insert_on_this_core);
276
277 // Loop over the empty cells we have kept.
278 for (const auto &cell : empty_cells_on_proc)
279 {
280 // Position of the center of the cell
281 Point<dim> insertion_location =
282 cells_centers.at(cell->active_cell_index());
283
284 // Generate the random point of insertion
285 insertion_location(0) +=
286 static_cast<double>(rand()) * maximum_range_for_randomness;
287
288 insertion_location(1) +=
289 static_cast<double>(rand()) * maximum_range_for_randomness;
290
291 if constexpr (dim == 3)
292 {
293 insertion_location(2) +=
294 static_cast<double>(rand()) * maximum_range_for_randomness;
295 }
296
297 insertion_points_on_proc.push_back(insertion_location);
298 }
299
300 // A vector of vectors, which contains all the properties of all inserted
301 // particles at each insertion step
302 std::vector<std::vector<double>> particle_properties;
303
304 this->assign_particle_properties(
305 dem_parameters,
306 number_of_particles_to_insert_on_this_core,
307 current_inserting_particle_type,
308 insertion_points_on_proc,
309 particle_properties);
310
311 // This is to iterate over the particle_properties vector
312 unsigned int i = 0;
313
314 // Insert the particles using the points and assigned properties
315 for (const auto &cell : empty_cells_on_proc)
316 {
317 // Use the pre-calculated insertion positions
318 Point<dim> insertion_location = insertion_points_on_proc[i];
319
320 // Insertion
321 Point<dim> ref_point;
322 particle_handler.insert_particle(insertion_location,
323 ref_point,
324 starting_ID_on_proc++,
325 cell,
326 particle_properties.at(i++));
327 }
328 }
329}
