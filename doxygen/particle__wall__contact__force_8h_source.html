24/**

130protected:
131 /**
132 * @brief Update the contact pair information for contact calculations.
133 *
134 * @param[in,out] contact_info Contact information of a particle-wall pair.
135 * @param[out] tangential_relative_velocity Tangential relative velocity.
136 * @param[out] normal_relative_velocity_value Normal relative contact
137 * velocity value.
138 * @param[in] particle_location Location of particle in 3d.
139 * @param[in] particle_properties Properties of particle.
140 * @param[in] dt DEM time step.
141 */
142 inline void
143 update_contact_information(particle_wall_contact_info<dim> &contact_info,
144 Tensor<1, 3> &tangential_relative_velocity,
145 double &normal_relative_velocity_value,
146 const Point<3> &particle_location,
147 const ArrayView<const double> &particle_properties,
148 const double dt)
149 {
150 // i is the particle, j is the wall.
151 // There is a minus sign in front of the normal_vector to respect the
152 // convention i -> j
153 Tensor<1, 3> normal_vector = -contact_info.normal_vector;
154 const unsigned int boundary_id = contact_info.boundary_id;
155
156 Tensor<1, 3> particle_velocity{{particle_properties[PropertiesIndex::v_x],
157 particle_properties[PropertiesIndex::v_y],
158 particle_properties[PropertiesIndex::v_z]}};
159
160 Tensor<1, 3> particle_angular_velocity{
161 {particle_properties[PropertiesIndex::omega_x],
162 particle_properties[PropertiesIndex::omega_y],
163 particle_properties[PropertiesIndex::omega_z]}};
164
165 // Calculating approximation of the contact point using the normal vector
166 Point<3> contact_point =
167 particle_location +
168 0.5 * particle_properties[PropertiesIndex::dp] * normal_vector;
169
170 // Getting vector pointing from the contact point to the origin of the
171 // rotation axis
172 Tensor<1, 3> vector_to_rotating_axis =
173 contact_point - this->point_on_rotation_vector[boundary_id];
174
175 // Removing the rotating axis component of that vector
176 vector_to_rotating_axis = vector_to_rotating_axis -
177 (vector_to_rotating_axis *
178 this->boundary_rotational_vector[boundary_id]) *
179 this->boundary_rotational_vector[boundary_id];
180
181 // Defining relative contact velocity using the convention
182 // v_ij = v_j - v_i
183 Tensor<1, 3> contact_relative_velocity =
184 this->boundary_translational_velocity_map[boundary_id] -
185 particle_velocity +
186 cross_product_3d((-0.5 * particle_properties[PropertiesIndex::dp] *
187 particle_angular_velocity),
188 normal_vector) +
189 cross_product_3d(this->boundary_rotational_speed_map[boundary_id] *
190 this->boundary_rotational_vector[boundary_id],
191 vector_to_rotating_axis);
192
193 // Calculating normal relative velocity
194 normal_relative_velocity_value = contact_relative_velocity * normal_vector;
195
196 // Calculating tangential relative velocity
197 tangential_relative_velocity =
198 contact_relative_velocity -
199 (normal_relative_velocity_value * normal_vector);
200
201 // Calculating new tangential_displacement
202 // Since this value is history-dependent, it needs the value at previous
203 // time-step. This variable is the main reason that we have iteration over
204 // two different vectors (pairs_in_contact and contact_pair_candidates):
205 // tangential_displacement of the particles which were already in contact
206 // (pairs_in_contact) needs to be modified using its history, while the
207 // tangential_displacements of new particles are equal to zero.
208 contact_info.tangential_displacement += tangential_relative_velocity * dt;
209 }

211 /**
212 * @brief Update the contact pair information for particle-solid object contacts.
213 *
214 * @param[in,out] contact_info Contact information of a particle-wall pair.
215 * @param[out] tangential_relative_velocity Tangential relative velocity.
216 * @param[out] normal_relative_velocity_value Normal relative contact
217 * velocity value.
218 * @param[in] particle_properties Properties of particle.
219 * @param[in] dt DEM time step.
220 * @param[in] cut_cell_translational_velocity Translational velocity of the
221 * cut cell.
222 * @param[in] cut_cell_rotational_velocity Rotational velocity of the cut
223 * cell.
224 * @param[in] center_of_rotation_particle_distance Distance between the
225 * particle and the center of rotation of the floating mesh.
226 */
227 inline void
228 update_particle_solid_object_contact_information(
229 particle_wall_contact_info<dim> &contact_info,
230 Tensor<1, 3> &tangential_relative_velocity,
231 double &normal_relative_velocity_value,
232 const ArrayView<const double> &particle_properties,
233 const double dt,
234 const Tensor<1, 3> &cut_cell_translational_velocity,
235 const Tensor<1, 3> &cut_cell_rotational_velocity,
236 const double center_of_rotation_particle_distance)
237 {
238 // i is the particle, j is the wall.
239 // we need to put a minus sign in front of the normal_vector to respect the
240 // convention (i -> j)
241 const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
242
243 // Using velocity and angular velocity of particle as
244 // local vectors
245 Tensor<1, 3> particle_velocity{{particle_properties[PropertiesIndex::v_x],
246 particle_properties[PropertiesIndex::v_y],
247 particle_properties[PropertiesIndex::v_z]}};
248
249 Tensor<1, 3> particle_angular_velocity{
250 {particle_properties[PropertiesIndex::omega_x],
251 particle_properties[PropertiesIndex::omega_y],
252 particle_properties[PropertiesIndex::omega_z]}};
253
254 // Defining relative contact velocity
255 // v_ij = v_j - v_i
256 Tensor<1, 3> contact_relative_velocity =
257 cut_cell_translational_velocity - particle_velocity +
258 cross_product_3d((center_of_rotation_particle_distance *
259 cut_cell_rotational_velocity -
260 0.5 * particle_properties[PropertiesIndex::dp] *
261 particle_angular_velocity),
262 normal_vector);
263
264 // Calculation of normal relative velocity
265 normal_relative_velocity_value = contact_relative_velocity * normal_vector;
266
267 // Calculation of tangential relative velocity
268 tangential_relative_velocity =
269 contact_relative_velocity -
270 normal_relative_velocity_value * normal_vector;
271
272 // Calculate the new tangential_displacement
273 // Since this value is history-dependent, it needs the value at
274 // previous time-step.
275 contact_info.tangential_displacement += tangential_relative_velocity * dt;
276 }

278 /**
279 * @brief This function is used to find the projection of vector_a on
280 * vector_b
281 * @param[in] vector_a A vector which is going to be projected on vector_b
282 * @param[in] vector_b The projection vector of vector_a
283 * @return The projection of vector_a on vector_b
284 */
285 static inline Tensor<1, 3>
286 find_projection(const Tensor<1, 3> &vector_a, const Tensor<1, 3> &vector_b)
287 {
288 Tensor<1, 3> vector_c;
289 vector_c = ((vector_a * vector_b) / (vector_b.norm_square())) * vector_b;
290
291 return vector_c;
292 }

398 /**
399 * @brief Calculate the minimum overlap at which particle-wall forces are
400 * computed. This function is useful for non-contact cohesive force models
401 * such as the DMT.
402 *
403 * @return minimum overlap for the force calculation.
404 */
405 inline double
406 get_force_calculation_threshold_distance()
407 {
408 if constexpr (contact_model ==
409 Parameters::Lagrangian::ParticleWallContactForceModel::DMT)
410 {
411 // We are looking for the maximum Hamaker constant and minimum surface
412 // energy to compute the biggest distance at which force will be
413 // computed. In other words, we are maximising the delta_0.
414 // This way, force_calculation_threshold_distance can be set
415 // to const variable, which should make the code faster.
416 const double max_effective_hamaker_constant =
417 *(std::max_element(this->effective_hamaker_constant.begin(),
418 this->effective_hamaker_constant.end()));
419 const double min_effective_surface_energy =
420 *(std::min_element(this->effective_surface_energy.begin(),
421 this->effective_surface_energy.end()));
422
423 // Return the critical delta_0. We put a minus sign in front of since a
424 // positive overlap means that there is contact.
425 return -std::sqrt(
426 max_effective_hamaker_constant /
427 (12. * M_PI * min_effective_surface_energy * dmt_cut_off_threshold));
428 }
429 return 0.;
430 }

436 *
437 * @param[in] normal_force Contact normal force.
438 * @param[in] tangential_force Contact tangential force.
439 * @param[in] tangential_torque Contact torque induced by the tangential
440 * force.

442 * @param[in,out] particle_torque Torque acting on particle.

445 inline void
446 apply_force_and_torque(const Tensor<1, 3> &normal_force,
447 const Tensor<1, 3> &tangential_force,
448 const Tensor<1, 3> &tangential_torque,
449 const Tensor<1, 3> &rolling_resistance_torque,

452 {

454 Tensor<1, 3> total_force = normal_force + tangential_force;
455

457 // Since the force was calculated on the wall, we use -= operator
458 particle_force -= total_force;
459 // Since the torque was directly calculated on the particle, we use +=
460 // operator
461 particle_torque += tangential_torque + rolling_resistance_torque;
462 }

534 /**
535 * @brief Calculate the particle-wall linear contact force and torque
536 * based on the updated values in contact_info.
537 *
538 * @param[in,out] contact_info A container that contains the required
539 * information for calculation of the contact force for a particle-wall pair.
540 * @param[in] tangential_relative_velocity Tangential relative velocity.
541 * @param[in] normal_relative_velocity_value Normal relative velocity value.
542 * @param[in] normal_overlap Contact normal overlap.
543 * @param[in] dt DEM time step.
544 * @param[in] particle_properties Properties of particle.
545 * @param[out] normal_force Contact normal force.
546 * @param[out] tangential_force Contact tangential force.
547 * @param[out] tangential_torque Contact tangential torque.
548 * @param[out] rolling_resistance_torque Contact rolling resistance torque.
549 */
550 inline void
551 calculate_linear_contact(particle_wall_contact_info<dim> &contact_info,
552 const Tensor<1, 3> &tangential_relative_velocity,
553 const double normal_relative_velocity_value,
554 const double normal_overlap,
555 const double dt,
556 const ArrayView<const double> &particle_properties,
557 Tensor<1, 3> &normal_force,
558 Tensor<1, 3> &tangential_force,
559 Tensor<1, 3> &tangential_torque,
560 Tensor<1, 3> &rolling_resistance_torque)
561 {
562 // i is the particle, j is the wall.
563 // there is a minus sign in front of the normal_vector to respect the
564 // convention (i -> j), the forces are thus calculated on the wall
565 const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
566 const unsigned int particle_type =
567 static_cast<unsigned int>(particle_properties[PropertiesIndex::type]);
568 const double particle_radius =
569 particle_properties[PropertiesIndex::dp] * 0.5;
570
571 const double youngs_modulus = this->effective_youngs_modulus[particle_type];
572 const double beta = this->model_parameter_beta[particle_type];
573 const double friction_coeff =
574 this->effective_coefficient_of_friction[particle_type];
575 const double rolling_viscous_damping_coeff =
576 this->effective_coefficient_of_rolling_viscous_damping[particle_type];
577 const double rolling_friction_coeff =
578 this->effective_coefficient_of_rolling_friction[particle_type];
579
580 // Calculation of normal and tangential spring and dashpot constants
581 // using particle properties
582 const double rp_sqrt = sqrt(particle_radius);
583
584 const double normal_spring_constant =
585 1.0667 * rp_sqrt * youngs_modulus *
586 pow((0.9375 * particle_properties[PropertiesIndex::mass] * 1.0 *
587 1.0 / // Characteristic velocity is set to 1.0
588 (rp_sqrt * youngs_modulus)),
589 0.2);
590
591 // There is no minus sign here since model_parameter_beta is negative or
592 // equal to zero.
593 const double normal_damping_constant =
594 2 * beta *
595 sqrt(particle_properties[PropertiesIndex::mass] * normal_spring_constant);
596
597 // REF : R. Garg, J. Galvin-Carney, T. Li, and S. Pannala, Documentation
598 // of open-source MFIXDEM software for gas-solids flows, Tingwen Li Dr.,
599 // p. 10, Sep. 2012. There is a minus sign since the tangential force is
600 // applied in the opposite direction of the tangential_displacement
601 const double tangential_spring_constant = -normal_spring_constant * 0.4;
602
603 const double tangential_damping_constant =
604 normal_damping_constant * 0.6324555320336759; // sqrt(0.4)
605
606 // Calculation of normal force using spring and dashpot normal forces
607 normal_force = (normal_spring_constant * normal_overlap +
608 normal_damping_constant * normal_relative_velocity_value) *
609 normal_vector;
610
611 // Calculation of tangential force
612 tangential_force =
613 (tangential_spring_constant * contact_info.tangential_displacement +
614 tangential_damping_constant * tangential_relative_velocity);
615
616 const double coulomb_threshold = friction_coeff * normal_force.norm();
617 // Check for gross sliding
618 if (tangential_force.norm() > coulomb_threshold)
619 {
620 // Gross sliding occurs and the tangential displacement and tangential
621 // force are limited to Coulomb's criterion
622 tangential_force =
623 coulomb_threshold * (tangential_force / tangential_force.norm());
624
625 contact_info.tangential_displacement =
626 tangential_force / (tangential_spring_constant + DBL_MIN);
627 }
628
629 // Calculation torque caused by tangential force
630 // We add the minus sign here since the tangential_force is applied on the
631 // particle in the opposite direction
632 tangential_torque =
633 cross_product_3d((particle_radius * normal_vector), -tangential_force);
634
635 rolling_resistance_torque = calculate_rolling_resistance_torque(
636 particle_radius,
637 particle_properties,
638 rolling_friction_coeff,
639 rolling_viscous_damping_coeff,
640 dt,
641 normal_spring_constant,
642 normal_force.norm(),
643 contact_info.normal_vector,
644 contact_info.rolling_resistance_spring_torque);
645 }

647 /**
648 * @brief Calculate the particle-wall non-linear contact force and torque
649 * based on the updated values in contact_info.
650 *

656 * @param[in] dt DEM time step.
657 * @param[in] particle_properties Properties of particle.
658 * @param[out] normal_force Contact normal force.
659 * @param[out] tangential_force Contact tangential force.

663 inline void
664 calculate_nonlinear_contact(
665 particle_wall_contact_info<dim> &contact_info,
666 const Tensor<1, 3> &tangential_relative_velocity,
667 const double normal_relative_velocity_value,
668 const double normal_overlap,
669 const double dt,
670 const ArrayView<const double> &particle_properties,
671 Tensor<1, 3> &normal_force,
672 Tensor<1, 3> &tangential_force,
673 Tensor<1, 3> &tangential_torque,
674 Tensor<1, 3> &rolling_resistance_torque)
675 {
676 // i is the particle, j is the wall.
677 // there is a minus sign in front of the normal_vector to respect the
678 // convention (i -> j), the forces are thus calculated on the wall
679 const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
680 const double particle_radius =
681 particle_properties[PropertiesIndex::dp] * 0.5;
682 const unsigned int particle_type =
683 static_cast<unsigned int>(particle_properties[PropertiesIndex::type]);
684
685 const double youngs_modulus = this->effective_youngs_modulus[particle_type];
686 const double shear_modulus = this->effective_shear_modulus[particle_type];
687 const double beta = this->model_parameter_beta[particle_type];
688 const double friction_coeff =
689 this->effective_coefficient_of_friction[particle_type];
690 const double rolling_viscous_damping_coeff =
691 this->effective_coefficient_of_rolling_viscous_damping[particle_type];
692 const double rolling_friction_coeff =
693 this->effective_coefficient_of_rolling_friction[particle_type];
694
695 // Calculate intermediate model parameters
696 // These values are used to consider non-linear relation of the contact
697 // force to the normal overlap
698 const double radius_times_overlap_sqrt =
699 sqrt(particle_radius * normal_overlap);
700 const double model_parameter_sn =
701 2.0 * youngs_modulus * radius_times_overlap_sqrt;
702 const double model_parameter_st =
703 8.0 * shear_modulus * radius_times_overlap_sqrt;
704
705 // Calculation of normal and tangential spring and dashpot constants
706 // using particle and wall properties
707 const double normal_spring_constant =
708 1.3333 * youngs_modulus * radius_times_overlap_sqrt;
709
710 // There is no minus sign here since beta is negative or
711 // equal to zero.
712 const double normal_damping_constant =
713 1.8257 * beta *
714 sqrt(model_parameter_sn * particle_properties[PropertiesIndex::mass]);
715
716 // There is a minus sign since the tangential force is applied in the
717 // opposite direction of the tangential_displacement
718 const double tangential_spring_constant =
719 -8.0 * shear_modulus * radius_times_overlap_sqrt + DBL_MIN;
720
721 const double tangential_damping_constant =
722 normal_damping_constant * sqrt(model_parameter_st / model_parameter_sn);
723
724 // Calculation of normal force using spring and dashpot normal forces
725 normal_force = (normal_spring_constant * normal_overlap +
726 normal_damping_constant * normal_relative_velocity_value) *
727 normal_vector;
728
729 // Calculation of tangential force. Since we need damping tangential force
730 // in the gross sliding again, we define it as a separate variable
731 const Tensor<1, 3> damping_tangential_force =
732 tangential_damping_constant * tangential_relative_velocity;
733 tangential_force =
734 tangential_spring_constant * contact_info.tangential_displacement +
735 damping_tangential_force;
736
737 const double coulomb_threshold = friction_coeff * normal_force.norm();
738
739 // Check for gross sliding
740 const double tangential_force_norm = tangential_force.norm();
741 if (tangential_force_norm > coulomb_threshold)
742 {
743 // Gross sliding occurs and the tangential displacement and tangential
744 // force are limited to Coulomb's criterion
745 contact_info.tangential_displacement =
746 (coulomb_threshold *
747 (tangential_force / (tangential_force_norm + DBL_MIN)) -
748 damping_tangential_force) /
749 (tangential_spring_constant + DBL_MIN);
750
751 tangential_force =
752 (tangential_spring_constant * contact_info.tangential_displacement) +
753 damping_tangential_force;
754 }
755
756 // Calculating torque caused by tangential force
757 // We add the minus sign here since the tangential_force applied on the
758 // particle is in the opposite direction
759 tangential_torque =
760 cross_product_3d((particle_radius * normal_vector), -tangential_force);
761
762 // Rolling resistance torque
763 rolling_resistance_torque = calculate_rolling_resistance_torque(
764 particle_radius,
765 particle_properties,
766 rolling_friction_coeff,
767 rolling_viscous_damping_coeff,
768 dt,
769 normal_spring_constant,
770 normal_force.norm(),
771 contact_info.normal_vector,
772 contact_info.rolling_resistance_spring_torque);
773 }

775 /**
776 * @brief Calculate the particle-wall JKR contact force and torque
777 * based on the updated values in contact_info.
778 *
779 * @param[in,out] contact_info A container that contains the required
780 * information for calculation of the contact force for a particle-wall pair.
781 * @param[in] tangential_relative_velocity Tangential relative velocity.
782 * @param[in] normal_relative_velocity_value Normal relative velocity value.
783 * @param[in] normal_overlap Contact normal overlap.
784 * @param[in] dt DEM time step.
785 * @param[in] particle_properties Properties of particle.
786 * @param[out] normal_force Contact normal force.
787 * @param[out] tangential_force Contact tangential force.
788 * @param[out] tangential_torque Contact tangential torque.
789 * @param[out] rolling_resistance_torque Contact rolling resistance torque.
790 */
791 inline void
792 calculate_JKR_contact(particle_wall_contact_info<dim> &contact_info,
793 const Tensor<1, 3> &tangential_relative_velocity,
794 const double normal_relative_velocity_value,
795 const double normal_overlap,
796 const double dt,
797 const ArrayView<const double> &particle_properties,
798 Tensor<1, 3> &normal_force,
799 Tensor<1, 3> &tangential_force,
800 Tensor<1, 3> &tangential_torque,
801 Tensor<1, 3> &rolling_resistance_torque)
802 {
803 // i is the particle, j is the wall.
804 // there is a minus sign in front of the normal_vector to respect the
805 // convention (i -> j), the forces are thus calculated on the wall
806 const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
807 const double particle_radius =
808 0.5 * particle_properties[PropertiesIndex::dp];
809 const unsigned int particle_type =
810 static_cast<unsigned int>(particle_properties[PropertiesIndex::type]);
811
812 const double youngs_modulus = this->effective_youngs_modulus[particle_type];
813 const double shear_modulus = this->effective_shear_modulus[particle_type];
814 const double beta = this->model_parameter_beta[particle_type];
815 const double friction_coeff =
816 this->effective_coefficient_of_friction[particle_type];
817 const double rolling_viscous_damping_coeff =
818 this->effective_coefficient_of_rolling_viscous_damping[particle_type];
819 const double rolling_friction_coeff =
820 this->effective_coefficient_of_rolling_friction[particle_type];
821 const double surface_energy = this->effective_surface_energy[particle_type];
822
823 // Calculate intermediate model parameters
824 // These values are used to consider non-linear relation of the contact
825 // force to the normal overlap
826 const double radius_times_overlap_sqrt =
827 sqrt(particle_radius * normal_overlap);
828 const double model_parameter_sn =
829 2.0 * youngs_modulus * radius_times_overlap_sqrt;
830 const double model_parameter_st =
831 8.0 * shear_modulus * radius_times_overlap_sqrt;
832
833 // Calculation of the contact patch radius (a) using the analytical solution
834 // described in the theory guide.
835 const double c0 =
836 Utilities::fixed_power<2>(particle_radius * normal_overlap);
837 const double c1 = -2. * Utilities::fixed_power<2>(particle_radius) * M_PI *
838 surface_energy / youngs_modulus;
839 const double c2 = -2. * normal_overlap * particle_radius;
840 const double P = -Utilities::fixed_power<2>(c2) / 12. - c0;
841 const double Q = -Utilities::fixed_power<3>(c2) / 108. + c0 * c2 / 3. -
842 Utilities::fixed_power<2>(c1) * 0.125;
843 const double root1 = std::max(0.,
844 (0.25 * Utilities::fixed_power<2>(Q)) +
845 (Utilities::fixed_power<3>(P) / 27.));
846 const double U = std::cbrt(-0.5 * Q + std::sqrt(root1));
847 const double s = -c2 * (5. / 6.) + U - P / (3. * U);
848 const double w = std::sqrt(std::max(1e-16, c2 + 2. * s));
849 const double lambda = 0.5 * c1 / w;
850 const double root2 = std::max(1e-16, w * w - 4. * (c2 + s + lambda));
851 const double a = 0.5 * (w + std::sqrt(root2));
852
853 // Calculation of normal damping and tangential spring and dashpot constants
854 // using particle and wall properties.
855 // There is no minus sign here since model_parameter_beta is negative or
856 // equal to zero.
857 const double normal_damping_constant =
858 1.8257 * beta * // 2. * sqrt(5./6.)
859 sqrt(model_parameter_sn * particle_properties[PropertiesIndex::mass]);
860
861 // Tangential spring constant is set as a negative just like in the other
862 // particle-wall models. This must be taken into account for the square root
863 // in the tangential_damping_calculation.
864 const double tangential_spring_constant =
865 -8.0 * shear_modulus * radius_times_overlap_sqrt;
866
867 // There is no minus sign here since model_parameter_beta is negative or
868 // equal to zero.
869 const double tangential_damping_constant =
870 normal_damping_constant *
871 sqrt(model_parameter_st / (model_parameter_sn + DBL_MIN));
872
873 // Calculation of the normal force coefficient (F_n_JKR)
874 const double normal_force_norm =
875 4. * youngs_modulus * Utilities::fixed_power<3>(a) /
876 (3. * particle_radius) -
877 std::sqrt(8. * M_PI * surface_energy * youngs_modulus *
878 Utilities::fixed_power<3>(a)) +
879 normal_damping_constant * normal_relative_velocity_value;
880
881 // Calculation of normal force using the normal_force_norm and the
882 // normal vector.
883 normal_force = normal_force_norm * normal_vector;
884
885 // Calculation of tangential forces.
886 Tensor<1, 3> damping_tangential_force =
887 tangential_damping_constant * tangential_relative_velocity;
888 tangential_force =
889 tangential_spring_constant * contact_info.tangential_displacement +
890 damping_tangential_force;
891
892 // JKR theory says that the coulomb threshold must be modified with the
893 // pull-out force. (Thornton 1991)
894 const double modified_coulomb_threshold =
895 (normal_force_norm + 3. * M_PI * surface_energy * particle_radius) *
896 friction_coeff;
897
898 // Check for gross sliding
899 const double tangential_force_norm = tangential_force.norm();
900 if (tangential_force_norm > modified_coulomb_threshold)
901 {
902 // Gross sliding occurs and the tangential displacement and tangential
903 // force are limited to Coulomb's criterion
904 contact_info.tangential_displacement =
905 (modified_coulomb_threshold *
906 (tangential_force / (tangential_force_norm + DBL_MIN)) -
907 damping_tangential_force) /
908 (tangential_spring_constant + DBL_MIN);
909
910 tangential_force =
911 (tangential_spring_constant * contact_info.tangential_displacement) +
912 damping_tangential_force;
913 }
914
915 // Calculation of torque caused by tangential force (tangential_torque)
916 // We add the minus sign here since the tangential_force applied on the
917 // particle is in the opposite direction
918 tangential_torque =
919 cross_product_3d((particle_radius * normal_vector), -tangential_force);
920
921 // We need to compute the normal spring constant for case where we use the
922 // EPSD rolling resistance model.
923 const double normal_spring_constant = 0.66665 * model_parameter_sn;
924
925 // Rolling resistance torque
926 rolling_resistance_torque = calculate_rolling_resistance_torque(
927 particle_radius,
928 particle_properties,
929 rolling_friction_coeff,
930 rolling_viscous_damping_coeff,
931 dt,
932 normal_spring_constant,
933 normal_force.norm(),
934 contact_info.normal_vector,
935 contact_info.rolling_resistance_spring_torque);
936 }

938 /**
939 * @brief Calculate the particle-wall cohesive force and
940 * contact torque based on the updated values in contact_info. It uses the DMT
941 * cohesive force model.
942 *
943 * @param[in,out] contact_info A container that contains the required
944 * information for calculation of the contact force for a particle-wall.
945 * @param[in] tangential_relative_velocity Tangential relative velocity.
946 * @param[in] normal_relative_velocity_value Normal relative velocity value.
947 * @param[in] normal_overlap Contact normal overlap.
948 * @param[in] dt DEM time step.
949 * @param[in] particle_properties Properties of particle.
950 * @param[out] normal_force Contact normal force.
951 * @param[out] tangential_force Contact tangential force.
952 * @param[out] tangential_torque Contact tangential torque.
953 * @param[out] rolling_resistance_torque Contact rolling resistance torque.
954 */
955 inline void
956 calculate_DMT_contact(particle_wall_contact_info<dim> &contact_info,
957 const Tensor<1, 3> &tangential_relative_velocity,
958 const double normal_relative_velocity_value,
959 const double normal_overlap,
960 const double dt,
961 const ArrayView<const double> &particle_properties,
962 Tensor<1, 3> &normal_force,
963 Tensor<1, 3> &tangential_force,
964 Tensor<1, 3> &tangential_torque,
965 Tensor<1, 3> &rolling_resistance_torque)
966 {
967 constexpr double M_2PI = 2. * M_PI;
968
969 // i is the particle, j is the wall.
970 // there is a minus sign in front of the normal_vector to respect the
971 // convention (i -> j), the forces are thus calculated on the wall
972 const Tensor<1, 3> normal_vector = -contact_info.normal_vector;
973 const double particle_radius =
974 0.5 * particle_properties[PropertiesIndex::dp];
975 const unsigned int particle_type =
976 static_cast<unsigned int>(particle_properties[PropertiesIndex::type]);
977
978 const double surface_energy = this->effective_surface_energy[particle_type];
979 const double hamaker_constant =
980 this->effective_hamaker_constant[particle_type];
981
982 const double F_po = M_2PI * particle_radius * surface_energy;
983
984 const double delta_0 =
985 -std::sqrt(hamaker_constant * particle_radius / (6. * F_po));
986
987 // Cohesive force.
988 double cohesive_term;
989
990 // Contact particle-wall + constant cohesive force.
991 if (normal_overlap > 0.)
992 {
993 cohesive_term = -F_po;
994
995 calculate_nonlinear_contact(contact_info,
996 tangential_relative_velocity,
997 normal_relative_velocity_value,
998 normal_overlap,
999 dt,
1000 particle_properties,
1001 normal_force,
1002 tangential_force,
1003 tangential_torque,
1004 rolling_resistance_torque);
1005 }
1006 // No contact, but still in the constant zone for the cohesive force.
1007 else if (normal_overlap > delta_0)
1008 {
1009 cohesive_term = -F_po;
1010 contact_info.tangential_displacement.clear();
1011 contact_info.rolling_resistance_spring_torque.clear();
1012 }
1013 // No contact. The cohesive force is not constant. It needs to be computed.
1014 else
1015 {
1016 cohesive_term = -hamaker_constant * particle_radius /
1017 (6. * Utilities::fixed_power<2>(normal_overlap));
1018 contact_info.tangential_displacement.clear();
1019 contact_info.rolling_resistance_spring_torque.clear();
1020 }
1021 normal_force += cohesive_term * normal_vector;
1022 }
