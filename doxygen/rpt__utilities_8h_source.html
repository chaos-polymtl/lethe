22/**
23 * @brief Create grid of the reactor vessel (cylinder).
24 */
25template <int dim>
26void
27attach_grid_to_triangulation(Triangulation<dim> &triangulation,
28 const Parameters::RPTParameters &parameters,
29 const unsigned int n_refinement)
30{
31 // Generate cylinder (needs rotation and shift to get origin at the bottom
32 // with z towards top)
33 GridGenerator::subdivided_cylinder(triangulation,
34 5,
35 parameters.reactor_radius,
36 parameters.reactor_height / 2.);
37 Tensor<1, dim> axis({0, 1, 0});
38 GridTools::rotate(axis, M_PI_2, triangulation);
39 const Tensor<1, dim> shift_vector({0, 0, parameters.reactor_height / 2.});
40 GridTools::shift(shift_vector, triangulation);
41
42 // Add cylindrical manifold
43 const Tensor<1, dim> direction({0, 0, 1});
44 const CylindricalManifold<dim, dim> manifold(direction, {0, 0, 1});
45 triangulation.set_manifold(0, manifold);
46
47 // Refine all cells
48 triangulation.prepare_coarsening_and_refinement();
49 triangulation.refine_global(n_refinement);
50}

53/**
54 * @brief Read text file for positions
55 *
56 * @param positions_filename Data filename for particle positions
57 */
58template <int dim>
59std::vector<Point<dim>>
60read_positions(std::string &positions_filename)
61{
62 // Read text file with positions and store it in vector
63 std::ifstream particle_file(positions_filename);
64
65 std::string skip;
66 std::getline(particle_file, skip); // Skip header line
67 std::vector<double> values;
68 std::copy(std::istream_iterator<double>(particle_file),
69 std::istream_iterator<double>(),
70 std::back_inserter(values));
71
72 unsigned int number_of_positions = values.size() / dim;
73
74 // Extract positions
75 std::vector<Point<dim>> positions;
76 for (unsigned int i = 0; i < number_of_positions; i++)
77 {
78 Point<dim> point(values[dim * i],
79 values[dim * i + 1],
80 values[dim * i + 2]);
81
82 positions.push_back(point);
83 }
84
85 return positions;
86}

88/**
89 * @brief Read text file for particle positions and assign them to particle
90 * objects
91 *
92 * @param particle_positions_filename Data filename for particle positions
93 */
94template <int dim>
95std::vector<RadioParticle<dim>>
96assign_particle_positions(std::string &particle_positions_filename)
97{
98 std::vector<Point<dim>> positions =
99 read_positions<dim>(particle_positions_filename);
100
101 // Extract positions, create point objects and radioactive particles
102 std::vector<RadioParticle<dim>> particle_positions;
103 for (unsigned int i = 0; i < positions.size(); i++)
104 {
105 RadioParticle<dim> position(positions[i], i);
106 particle_positions.push_back(position);
107 }
108
109 return particle_positions;
110}

113/**
114 * @brief Read text file for detector positions and assign them to detector
115 * objects
116 *
117 * @param detector_parameters All parameters related to detector
118 */
119
120template <int dim>
121std::vector<Detector<dim>>
122assign_detector_positions(Parameters::DetectorParameters &detector_parameters)
123{
124 // Read text file with detector positions and store it in vector
125 std::ifstream detector_file(detector_parameters.detector_positions_file);
126
127 std::string skip;
128 std::getline(detector_file, skip);
129 // Skip header line if the header is present
130 if (!isalpha(skip[0]))
131 {
132 detector_file.seekg(0, std::ios::beg);
133 }
134 std::vector<double> values;
135 std::copy(std::istream_iterator<double>(detector_file),
136 std::istream_iterator<double>(),
137 std::back_inserter(values));
138
139 // Get the number of detector (2 positions for 1 detector, face and middle)
140 int number_of_detector = values.size() / (2 * dim);
141
142 // Extract positions, create point objects and detectors
143 std::vector<Detector<dim>> detectors;
144 for (int i = 0; i < number_of_detector; i++)
145 {
146 Point<dim> face_point(values[2 * dim * i],
147 values[2 * dim * i + 1],
148 values[2 * dim * i + 2]);
149 Point<dim> middle_point(values[2 * dim * i + dim],
150 values[2 * dim * i + dim + 1],
151 values[2 * dim * i + dim + 2]);
152
153 Detector<dim> detector(detector_parameters, i, face_point, middle_point);
154
155 detectors.push_back(detector);
156 }
157 return detectors;
158}

187/**
188 * @brief Read text file for count data and sort them to match all counts
189 * to a detector
190 * @param counts_filename Data filename for counts
191 */
192
193template <int dim>
194std::vector<std::vector<double>>
195read_detectors_counts(std::string &counts_filename, unsigned int n_detectors)
196{
197 std::vector<double> counts = read_counts<dim>(counts_filename);
198
199 std::vector<std::vector<double>> detectors_counts;
200 for (unsigned int i = 0; i < counts.size(); i += n_detectors)
201 {
202 std::vector<double>::const_iterator first = counts.begin() + i;
203 std::vector<double>::const_iterator last =
204 counts.begin() + i + n_detectors;
205
206 std::vector<double> particle_counts(first, last);
207
208 detectors_counts.push_back(particle_counts);
209 }
210
211 return detectors_counts;
212}
