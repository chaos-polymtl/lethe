13 /**
14 * @brief Class used to rescale physical properties for cases where the main dimensions
15 * of the problems (Length, Mass, Time and Temperature) are not expressed in
16 * SI, but are expressed in another set of units. For example, this can be
17 * used to carry out a simulation where the mesh is expressed in millimeters
18 * instead of meters. This class is currently only used to rescale the
19 * physical properties in the FEM-based physics (e.g. all of those that are
20 * solvers). Support for DEM, CFD-DEM and rescaling of boundary conditions is
21 * planned in the midterm.
22 */
23 class Dimensionality
24 {
25 public:
26 /**
27 * @brief
28 * Default constructor that assumes that the units used in the simulation
29 * are consistent.
30 */
31 Dimensionality()
32 : length(1)
33 , mass(1)
34 , time(1)
35 , temperature(1)
36 {
37 define_all_scales();
38 }
39
40 /**
41 * @brief
42 * Defines all the sub-scales that are relevant to simulations. This is used
43 * to define scaling laws for kinematic viscosity or other quantities which
44 * appear multiple times in the simulation. This prevents the recalculation
45 * of these scales (and thus, human errors).
46 */
47 void
48 define_all_scales();
49
50 /**
51 * @brief
52 * Declares the fundamental dimensions of the parameter file
53 *
54 * @param prm The parameter handler being used
55 */
56 static void
57 declare_parameters(ParameterHandler &prm);
58
59 /**
60 * @brief
61 * Parses the fundamental dimensions from the parameter file
62 *
63 * @param prm The parameter handler being used
64 */
65 void
66 parse_parameters(ParameterHandler &prm);
67
68 /*
69 * Fundamental dimensions. These are the dimensions that are read from the
70 * parameter file
71 */
72 double length;
73 double mass;
74 double time;
75 double temperature;
76
77 /*
78 * Scaling laws. These are constants that are used to rescale quantities
79 * arising frequently. Some of these quantities are not trivial to calculate
80 * (e.g. thermal conductivity). Thus, it is better to pre-calculate these
81 * scalings and to reuse them instead of calculating them on the fly
82 * everywhere.
83 */
84 double density_scaling;
85 double specific_gas_constant_scaling;
86 double viscosity_scaling;
87 double specific_heat_scaling;
88 double thermal_conductivity_scaling;
89 double enthalpy_scaling;
90 double diffusivity_scaling;
91 double thermal_expansion_scaling;
92 double surface_tension_scaling;
93 double surface_tension_gradient_scaling;
94 double cahn_hilliard_mobility_scaling;
95 double cahn_hilliard_epsilon_scaling;
96 };
97} // namespace Parameters
