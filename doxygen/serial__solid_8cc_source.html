22template <int dim, int spacedim>
23SerialSolid<dim, spacedim>::SerialSolid(
24 std::shared_ptr<Parameters::RigidSolidObject<spacedim>> &param,
25 const unsigned int id)
26 : mpi_communicator(MPI_COMM_WORLD)
27 , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
28 , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
29 , param(param)
30 , id(id)
31 , output_bool(param->output_bool)
32 , translational_velocity(param->translational_velocity)
33 , angular_velocity(param->angular_velocity)
34 , center_of_rotation(param->center_of_rotation)
35 , solid_temperature(param->solid_temperature)
36 , thermal_boundary_type(param->thermal_boundary_type)
37{
38 if (param->solid_mesh.simplex)
39 {
40 // for simplex meshes
41 fe = std::make_shared<FE_SimplexP<dim, spacedim>>(1);
42 solid_mapping = std::make_shared<MappingFE<dim, spacedim>>(*fe);
43 solid_tria = std::make_shared<Triangulation<dim, spacedim>>();
44
45 displacement_fe =
46 std::make_shared<FESystem<dim, spacedim>>(FE_SimplexP<dim, spacedim>(1),
47 spacedim);
48 }
49 else
50 {
51 // Usual case, for quad/hex meshes
52 fe = std::make_shared<FE_Q<dim, spacedim>>(1);
53 solid_mapping = std::make_shared<MappingQGeneric<dim, spacedim>>(1);
54 solid_tria = std::make_shared<Triangulation<dim, spacedim>>(
55 typename Triangulation<dim, spacedim>::MeshSmoothing(
56 Triangulation<dim, spacedim>::smoothing_on_refinement |
57 Triangulation<dim, spacedim>::smoothing_on_coarsening));
58
59 displacement_fe =
60 std::make_shared<FESystem<dim, spacedim>>(FE_Q<dim, spacedim>(1),
61 spacedim);
62 }
63
64 // Dof handler associated with mesh displacement
65 displacement_dh.clear();
66 displacement_dh.reinit(*this->solid_tria);
67
68 // Load triangulation
69 initial_setup();
70
71 if constexpr (dim == 2)
72 {
73 // Set-up containers
74 if (param->solid_mesh.simplex)
75 setup_containers();
76 }
77}

79template <int dim, int spacedim>
80std::vector<
81 std::pair<typename Triangulation<spacedim>::active_cell_iterator,
82 typename Triangulation<dim, spacedim>::active_cell_iterator>>
83SerialSolid<dim, spacedim>::map_solid_in_background_triangulation(
84 const parallel::TriangulationBase<spacedim> &background_tr)
85{
86 std::vector<
87 std::pair<typename Triangulation<spacedim>::active_cell_iterator,
88 typename Triangulation<dim, spacedim>::active_cell_iterator>>
89 mapped_solid;
90
91 if constexpr (dim == 2)
92 {
93 // Gather the amount of vertices per cell in the solid cells once to have
94 // a static memory allocation of the vector. This prevents reallocating
95 // the vector dynamically using push_back if the solid triangulation
96 // has many triangles
97 auto temporary_solid_cell = solid_tria->begin();
98 std::vector<Point<spacedim>> triangle(temporary_solid_cell->n_vertices());
99
100 // Calculate distance from cell center to solid_cell
101 for (const auto &background_cell : background_tr.active_cell_iterators())
102 {
103 // If the cell is owned by the processor
104 if (background_cell->is_locally_owned())
105 {
106 // Calculate the characteristic size of the background cell
107 const double bg_cell_length = background_cell->diameter();
108
109 // Calculate the center of the cell
110 Point<spacedim> bg_cell_center = background_cell->center();
111
112 // Calculate distance from center of the cell to triangle
113 for (auto &solid_cell : solid_tria->active_cell_iterators())
114 {
115 // Gather triangle vertices
116 for (unsigned int v = 0; v < solid_cell->n_vertices(); ++v)
117 {
118 triangle[v] = solid_cell->vertex(v);
119 }
120
121 // Calculate distance between triangle and center
122 const double distance =
123 LetheGridTools::find_point_triangle_distance(
124 triangle, bg_cell_center);
125
126 if (distance < bg_cell_length)
127 {
128 mapped_solid.push_back(
129 std::make_pair(background_cell, solid_cell));
130 }
131 }
132 }
133 }
134 }
135 else if constexpr (dim == 3)
136 {
137 throw std::runtime_error(
138 "Solid volumes are not yet implemented in Lethe. Please, remove the whole \"solid volumes\" subsection from your parameter file.");
139 }
140
141
142 reset_displacement_monitoring();
143
144 return mapped_solid;
145}

161template <int dim, int spacedim>
162void
163SerialSolid<dim, spacedim>::setup_triangulation(const bool restart)
164{
165 if (param->solid_mesh.type == Parameters::Mesh::Type::gmsh)
166 {
167 // Grid creation
168 GridIn<dim, spacedim> grid_in;
169 // Attach triangulation
170 grid_in.attach_triangulation(*solid_tria);
171 // Read input gmsh file
172 std::ifstream input_file(param->solid_mesh.file_name);
173
174 grid_in.read_msh(input_file);
175 }
176 else if (param->solid_mesh.type == Parameters::Mesh::Type::dealii)
177 {
178 if (param->solid_mesh.simplex)
179 {
180 Triangulation<dim, spacedim> temporary_quad_triangulation;
181 GridGenerator::generate_from_name_and_arguments(
182 temporary_quad_triangulation,
183 param->solid_mesh.grid_type,
184 param->solid_mesh.grid_arguments);
185
186 // initial refinement
187 const int initial_refinement = param->solid_mesh.initial_refinement;
188 temporary_quad_triangulation.refine_global(initial_refinement);
189 // flatten the triangulation
190 Triangulation<dim, spacedim> flat_temp_quad_triangulation;
191 GridGenerator::flatten_triangulation(temporary_quad_triangulation,
192 flat_temp_quad_triangulation);
193
194 GridGenerator::convert_hypercube_to_simplex_mesh(
195 flat_temp_quad_triangulation, *solid_tria);
196 }
197 else
198 { // deal.ii creates grid and attaches the solid triangulation
199 GridGenerator::generate_from_name_and_arguments(
200 *solid_tria,
201 param->solid_mesh.grid_type,
202 param->solid_mesh.grid_arguments);
203 }
204 }
205 else
206 throw std::runtime_error(
207
208 "Unsupported mesh type - solid mesh will not be created");
209
210 // Rotate the triangulation
211 rotate_grid(param->solid_mesh.rotation_angle,
212 param->solid_mesh.rotation_axis);
213
214 // Translate the triangulation
215 translate_grid(param->solid_mesh.translation);
216
217 // Refine the solid triangulation to its initial size
218 // NB: solid_tria should not be refined if loaded from a restart file
219 // afterward
220 if (!restart)
221 {
222 if (param->solid_mesh.simplex)
223 {
224 // Simplex triangulation refinement isn't possible yet
225 }
226 else
227 {
228 solid_tria->refine_global(param->solid_mesh.initial_refinement);
229 }
230 }
231}

331template <int dim, int spacedim>
332void
333SerialSolid<dim, spacedim>::move_solid_triangulation(const double time_step,
334 const double initial_time)
335{
336 // First we calculate the displacement that the solid triangulation
337 // will feel during this iteration.
338 // Then, we apply this displacement to the triangulation
339 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
340 const std::vector<bool> locally_owned_vertices =
341 GridTools::get_locally_owned_vertices(*this->solid_tria);
342 std::vector<bool> vertex_moved(this->solid_tria->n_vertices(), false);
343 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
344
345 translational_velocity->set_time(initial_time);
346 angular_velocity->set_time(initial_time);
347
348 for (int comp_i = 0; comp_i < spacedim; ++comp_i)
349 {
350 current_translational_velocity[comp_i] =
351 translational_velocity->value(center_of_rotation, comp_i);
352 current_angular_velocity[comp_i] =
353 angular_velocity->value(center_of_rotation, comp_i);
354 }
355
356 for (const auto &cell : displacement_dh.active_cell_iterators())
357 {
358 if (cell->is_locally_owned())
359 {
360 for (unsigned int vertex = 0;
361 vertex < cell->reference_cell().n_vertices();
362 ++vertex)
363 {
364 const auto dof_index = cell->vertex_dof_index(vertex, 0);
365 Point<spacedim> &vertex_position = cell->vertex(vertex);
366 const unsigned global_vertex_no = cell->vertex_index(vertex);
367 if (vertex_moved[global_vertex_no] ||
368 !locally_owned_vertices[global_vertex_no])
369 continue;
370
371 Tensor<1, spacedim> distance_vector =
372 vertex_position - center_of_rotation;
373
374 Tensor<1, spacedim> local_velocity =
375 current_translational_velocity;
376 if constexpr (spacedim == 2)
377 {
378 local_velocity[0] +=
379 -distance_vector[1] * current_angular_velocity[2];
380 local_velocity[1] +=
381 distance_vector[0] * current_angular_velocity[2];
382 }
383 if constexpr (spacedim == 3)
384 {
385 local_velocity +=
386 cross_product_3d(current_angular_velocity, distance_vector);
387 }
388
389 auto vertex_displacement = time_step * local_velocity;
390
391 vertex_position += vertex_displacement;
392
393 for (int d = 0; d < spacedim; ++d)
394 {
395 displacement[dof_index + d] =
396 displacement[dof_index + d] + vertex_displacement[d];
397 displacement_since_mapped[dof_index + d] +=
398 vertex_displacement[d];
399 }
400
401 vertex_moved[global_vertex_no] = true;
402 }
403 }
404 }
405
406 center_of_rotation += current_translational_velocity * time_step;
407}

410template <int dim, int spacedim>
411void
412SerialSolid<dim, spacedim>::displace_solid_triangulation()
413{
414 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
415 std::set<unsigned int> dof_vertex_displaced;
416 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
417
418 for (const auto &cell : displacement_dh.active_cell_iterators())
419 {
420 if (cell->is_locally_owned())
421 {
422 for (unsigned int vertex = 0;
423 vertex < cell->reference_cell().n_vertices();
424 ++vertex)
425 {
426 if (!dof_vertex_displaced.contains(cell->vertex_index(vertex)))
427 {
428 const auto dof_index = cell->vertex_dof_index(vertex, 0);
429 Point<spacedim> &vertex_position = cell->vertex(vertex);
430
431 for (int d = 0; d < spacedim; ++d)
432 {
433 vertex_position[d] =
434 vertex_position[d] + displacement[dof_index + d];
435 }
436
437 dof_vertex_displaced.insert(cell->vertex_index(vertex));
438 }
439 }
440 }
441 }
442}

444template <int dim, int spacedim>
445void
446SerialSolid<dim, spacedim>::move_solid_triangulation_with_displacement()
447{
448 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
449
450 // Set of vertices which were displaced already. A set is used
451 // for efficiency when searching over it.
452 std::set<unsigned int> dof_vertex_displaced;
453 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
454
455 // Loop over all cells
456 for (const auto &cell : displacement_dh.active_cell_iterators())
457 {
458 for (unsigned int vertex = 0;
459 vertex < cell->reference_cell().n_vertices();
460 ++vertex)
461 {
462 if (!dof_vertex_displaced.contains(cell->vertex_index(vertex)))
463 {
464 const auto dof_index = cell->vertex_dof_index(vertex, 0);
465 Point<spacedim> &vertex_position = cell->vertex(vertex);
466
467 for (int d = 0; d < spacedim; ++d)
468 {
469 vertex_position[d] += displacement[dof_index + d];
470 }
471
472 dof_vertex_displaced.insert(cell->vertex_index(vertex));
473 }
474 }
475 }

481SerialSolid<dim, spacedim>::write_output_results(
482 const std::shared_ptr<SimulationControl> &simulation_control)
483{
484 if (!output_bool)
485 return;
486
487 DataOut<dim, spacedim> data_out;
488 data_out.attach_dof_handler(displacement_dh);
489
490 {
491 std::vector<std::string> displacement_names(spacedim, "displacement");
492 std::vector<DataComponentInterpretation::DataComponentInterpretation>
493 data_component_interpretation(
494 spacedim, DataComponentInterpretation::component_is_part_of_vector);
495
496 data_out.add_data_vector(displacement,
497 displacement_names,
498 DataOut<dim, spacedim>::type_dof_data,
499 data_component_interpretation);
500 }
501
502 {
503 std::vector<std::string> displacement_names(spacedim,
504 "displacement_since_mapped");
505 std::vector<DataComponentInterpretation::DataComponentInterpretation>
506 data_component_interpretation(
507 spacedim, DataComponentInterpretation::component_is_part_of_vector);
508
509 data_out.add_data_vector(displacement_since_mapped,
510 displacement_names,
511 DataOut<dim, spacedim>::type_dof_data,
512 data_component_interpretation);
513 }
514
515 data_out.build_patches();
516
517 const std::string folder = simulation_control->get_output_path();
518 const std::string solution_name = [&] {
519 if constexpr (dim == 1)
520 {
521 return simulation_control->get_output_name() + ".solid_line." +
522 Utilities::int_to_string(id, 2);
523 }
524 if constexpr (dim == 2)
525 {
526 return simulation_control->get_output_name() + ".solid_surface." +
527 Utilities::int_to_string(id, 2);
528 }
529 if constexpr (dim == 3)
530 {
531 return simulation_control->get_output_name() + ".solid_volume." +
532 Utilities::int_to_string(id, 2);
533 }
534 }();
535 const unsigned int iter = simulation_control->get_step_number();
536 const double time = simulation_control->get_current_time();
537 const unsigned int group_files = simulation_control->get_group_files();
538
539 write_vtu_and_pvd<dim, spacedim>(pvdhandler,
540 data_out,
541 folder,
542 solution_name,
543 time,
544 iter,
545 group_files,
546 this->mpi_communicator);
547}

617template <int dim, int spacedim>
618void
619SerialSolid<dim, spacedim>::setup_containers()
620{
621 // Map : Vertices ID --> Cell iterator vector
622 std::map<
623 unsigned int,
624 std::set<typename Triangulation<dim, spacedim>::active_cell_iterator>>
625 vertices_cell_map;
626 LetheGridTools::vertices_cell_mapping(*solid_tria, vertices_cell_map);
627
628 // Loop on every cell
629 for (const auto &main_cell : solid_tria->active_cell_iterators())
630 {
631 // Initializing the containers associated with the main cell
632 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>
633 es_main_cell, vs_main_cell;
634
635 // Number of vertices of the main cell
636 const unsigned int n_vertices_main_cell = main_cell->n_vertices();
637
638 // Store the vertices index of the main cell in a vector
639 std::vector<unsigned int> main_cell_vertex_indices;
640 main_cell_vertex_indices.reserve(n_vertices_main_cell);
641 for (unsigned int v = 0; v < n_vertices_main_cell; ++v)
642 main_cell_vertex_indices.push_back(main_cell->vertex_index(v));
643
644 // Find the cell iterators around the main cell. This includes the main
645 // cell itself.
646 auto neighboring_cells =
647 LetheGridTools::find_cells_around_cell<dim, spacedim>(vertices_cell_map,
648 main_cell);
649
650 // Loop on the neighboring cells of the main cell
651 for (const auto &neigh_cell : neighboring_cells)
652 {
653 // Skip the first iterator since it is the main cell itself
654 if (neigh_cell == main_cell)
655 continue;
656
657 // Number of sharing vertices between main cell and neighboring
658 // cell. We know it is high either 1 or 2.
659 unsigned int n_sharing_vertices = 0;
660
661 // Loop on the vertices of that of the neighboring cell.
662 const unsigned int n_vertices_neigh_cell = neigh_cell->n_vertices();
663
664 for (unsigned int v = 0; v < n_vertices_neigh_cell; ++v)
665 {
666 unsigned int neigh_cell_vertex_index =
667 neigh_cell->vertex_index(v);
668 if (std::ranges::find(main_cell_vertex_indices,
669 neigh_cell_vertex_index) !=
670 main_cell_vertex_indices.end())
671 n_sharing_vertices++;
672 }
673 // Sharing 1 vertex
674 if (n_sharing_vertices == 1)
675 vs_main_cell.push_back(neigh_cell);
676 else // Sharing 2 vertex
677 es_main_cell.push_back(neigh_cell);
678 }
679 es_neighbors.insert(std::make_pair(main_cell, es_main_cell));
680 vs_neighbors.insert(std::make_pair(main_cell, vs_main_cell));
681 }
682}
