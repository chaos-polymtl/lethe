22template <int dim, int spacedim>
23SerialSolid<dim, spacedim>::SerialSolid(
24 std::shared_ptr<Parameters::RigidSolidObject<spacedim>> &param,
25 unsigned int id)
26 : mpi_communicator(MPI_COMM_WORLD)
27 , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
28 , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
29 , param(param)
30 , id(id)
31 , output_bool(param->output_bool)
32 , translational_velocity(param->translational_velocity)
33 , angular_velocity(param->angular_velocity)
34 , center_of_rotation(param->center_of_rotation)
35 , solid_temperature(param->solid_temperature)
36 , thermal_boundary_type(param->thermal_boundary_type)
37{
38 if (param->solid_mesh.simplex)
39 {
40 // for simplex meshes
41 fe = std::make_shared<FE_SimplexP<dim, spacedim>>(1);
42 solid_mapping = std::make_shared<MappingFE<dim, spacedim>>(*fe);
43 solid_tria = std::make_shared<Triangulation<dim, spacedim>>();
44
45 displacement_fe =
46 std::make_shared<FESystem<dim, spacedim>>(FE_SimplexP<dim, spacedim>(1),
47 spacedim);
48 }
49 else
50 {
51 // Usual case, for quad/hex meshes
52 fe = std::make_shared<FE_Q<dim, spacedim>>(1);
53 solid_mapping = std::make_shared<MappingQGeneric<dim, spacedim>>(1);
54 solid_tria = std::make_shared<Triangulation<dim, spacedim>>(
55 typename Triangulation<dim, spacedim>::MeshSmoothing(
56 Triangulation<dim, spacedim>::smoothing_on_refinement |
57 Triangulation<dim, spacedim>::smoothing_on_coarsening));
58
59 displacement_fe =
60 std::make_shared<FESystem<dim, spacedim>>(FE_Q<dim, spacedim>(1),
61 spacedim);
62 }
63
64 // Dof handler associated with mesh displacement
65 displacement_dh.clear();
66 displacement_dh.reinit(*this->solid_tria);
67
68 // Load triangulation
69 initial_setup();
70}

72template <int dim, int spacedim>
73std::vector<
74 std::pair<typename Triangulation<spacedim>::active_cell_iterator,
75 typename Triangulation<dim, spacedim>::active_cell_iterator>>
76SerialSolid<dim, spacedim>::map_solid_in_background_triangulation(
77 const parallel::TriangulationBase<spacedim> &background_tr)
78{
79 std::vector<
80 std::pair<typename Triangulation<spacedim>::active_cell_iterator,
81 typename Triangulation<dim, spacedim>::active_cell_iterator>>
82 mapped_solid;
83
84 if constexpr (dim == 2)
85 {
86 // Gather the amount of vertices per cell in the solid cells once to have
87 // a static memory allocation of the vector. This prevents reallocating
88 // the vector dynamically using push_back if the solid triangulation
89 // has many triangles
90 auto temporary_solid_cell = solid_tria->begin();
91 std::vector<Point<spacedim>> triangle(temporary_solid_cell->n_vertices());
92
93 // Calculate distance from cell center to solid_cell
94 for (const auto &background_cell : background_tr.active_cell_iterators())
95 {
96 // If the cell is owned by the processor
97 if (background_cell->is_locally_owned())
98 {
99 // Calculate the characteristic size of the background cell
100 const double bg_cell_length = background_cell->diameter();
101
102 // Calculate the center of the cell
103 Point<spacedim> bg_cell_center = background_cell->center();
104
105 // Calculate distance from center of the cell to triangle
106 for (auto &solid_cell : solid_tria->active_cell_iterators())
107 {
108 // Gather triangle vertices
109 for (unsigned int v = 0; v < solid_cell->n_vertices(); ++v)
110 {
111 triangle[v] = solid_cell->vertex(v);
112 }
113
114 // Calculate distance between triangle and center
115 const double distance =
116 LetheGridTools::find_point_triangle_distance(
117 triangle, bg_cell_center);
118
119 if (distance < bg_cell_length)
120 {
121 mapped_solid.push_back(
122 std::make_pair(background_cell, solid_cell));
123 }
124 }
125 }
126 }
127 }
128 else if constexpr (dim == 3)
129 {
130 throw std::runtime_error(
131 "Solid volumes are not yet implemented in Lethe. Please, remove the whole \"solid volumes\" subsection from your parameter file.");
132 }
133
134
135 reset_displacement_monitoring();
136
137 return mapped_solid;
138}

154template <int dim, int spacedim>
155void
156SerialSolid<dim, spacedim>::setup_triangulation(const bool restart)
157{
158 if (param->solid_mesh.type == Parameters::Mesh::Type::gmsh)
159 {
160 // Grid creation
161 GridIn<dim, spacedim> grid_in;
162 // Attach triangulation
163 grid_in.attach_triangulation(*solid_tria);
164 // Read input gmsh file
165 std::ifstream input_file(param->solid_mesh.file_name);
166
167 grid_in.read_msh(input_file);
168 }
169 else if (param->solid_mesh.type == Parameters::Mesh::Type::dealii)
170 {
171 if (param->solid_mesh.simplex)
172 {
173 Triangulation<dim, spacedim> temporary_quad_triangulation;
174 GridGenerator::generate_from_name_and_arguments(
175 temporary_quad_triangulation,
176 param->solid_mesh.grid_type,
177 param->solid_mesh.grid_arguments);
178
179 // initial refinement
180 const int initial_refinement = param->solid_mesh.initial_refinement;
181 temporary_quad_triangulation.refine_global(initial_refinement);
182 // flatten the triangulation
183 Triangulation<dim, spacedim> flat_temp_quad_triangulation;
184 GridGenerator::flatten_triangulation(temporary_quad_triangulation,
185 flat_temp_quad_triangulation);
186
187 GridGenerator::convert_hypercube_to_simplex_mesh(
188 flat_temp_quad_triangulation, *solid_tria);
189 }
190 else
191 { // deal.ii creates grid and attaches the solid triangulation
192 GridGenerator::generate_from_name_and_arguments(
193 *solid_tria,
194 param->solid_mesh.grid_type,
195 param->solid_mesh.grid_arguments);
196 }
197 }
198 else
199 throw std::runtime_error(
200
201 "Unsupported mesh type - solid mesh will not be created");
202
203 // Rotate the triangulation
204 rotate_grid(param->solid_mesh.rotation_angle,
205 param->solid_mesh.rotation_axis);
206
207 // Translate the triangulation
208 translate_grid(param->solid_mesh.translation);
209
210 // Refine the solid triangulation to its initial size
211 // NB: solid_tria should not be refined if loaded from a restart file
212 // afterwards
213 if (!restart)
214 {
215 if (param->solid_mesh.simplex)
216 {
217 // Simplex triangulation refinement isn't possible yet
218 }
219 else
220 {
221 solid_tria->refine_global(param->solid_mesh.initial_refinement);
222 }
223 }
224}

322template <int dim, int spacedim>
323void
324SerialSolid<dim, spacedim>::move_solid_triangulation(const double time_step,
325 const double initial_time)
326{
327 // First we calculate the displacement that the solid triangulation
328 // will feel during this iteration.
329 // Then, we apply this displacement to the triangulation
330 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
331 const std::vector<bool> locally_owned_vertices =
332 GridTools::get_locally_owned_vertices(*this->solid_tria);
333 std::vector<bool> vertex_moved(this->solid_tria->n_vertices(), false);
334 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
335
336 translational_velocity->set_time(initial_time);
337 angular_velocity->set_time(initial_time);
338
339 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
340 {
341 current_translational_velocity[comp_i] =
342 translational_velocity->value(center_of_rotation, comp_i);
343 current_angular_velocity[comp_i] =
344 angular_velocity->value(center_of_rotation, comp_i);
345 }
346
347 for (const auto &cell : displacement_dh.active_cell_iterators())
348 {
349 if (cell->is_locally_owned())
350 {
351 for (unsigned int vertex = 0;
352 vertex < cell->reference_cell().n_vertices();
353 ++vertex)
354 {
355 const auto dof_index = cell->vertex_dof_index(vertex, 0);
356 Point<spacedim> &vertex_position = cell->vertex(vertex);
357 const unsigned global_vertex_no = cell->vertex_index(vertex);
358 if (vertex_moved[global_vertex_no] ||
359 !locally_owned_vertices[global_vertex_no])
360 continue;
361
362 Tensor<1, spacedim> distance_vector =
363 vertex_position - center_of_rotation;
364
365 Tensor<1, spacedim> local_velocity =
366 current_translational_velocity;
367 if constexpr (spacedim == 2)
368 {
369 local_velocity[0] +=
370 -distance_vector[1] * current_angular_velocity[2];
371 local_velocity[1] +=
372 distance_vector[0] * current_angular_velocity[2];
373 }
374 if constexpr (spacedim == 3)
375 {
376 local_velocity +=
377 cross_product_3d(current_angular_velocity, distance_vector);
378 }
379
380 auto vertex_displacement = time_step * local_velocity;
381
382 vertex_position += vertex_displacement;
383
384 for (unsigned d = 0; d < spacedim; ++d)
385 {
386 displacement[dof_index + d] =
387 displacement[dof_index + d] + vertex_displacement[d];
388 displacement_since_mapped[dof_index + d] +=
389 vertex_displacement[d];
390 }
391
392 vertex_moved[global_vertex_no] = true;
393 }
394 }
395 }
396
397 center_of_rotation += current_translational_velocity * time_step;
398}

401template <int dim, int spacedim>
402void
403SerialSolid<dim, spacedim>::displace_solid_triangulation()
404{
405 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
406 std::set<unsigned int> dof_vertex_displaced;
407 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
408
409 for (const auto &cell : displacement_dh.active_cell_iterators())
410 {
411 if (cell->is_locally_owned())
412 {
413 for (unsigned int vertex = 0;
414 vertex < cell->reference_cell().n_vertices();
415 ++vertex)
416 {
417 if (!dof_vertex_displaced.count(cell->vertex_index(vertex)))
418 {
419 const auto dof_index = cell->vertex_dof_index(vertex, 0);
420 Point<spacedim> &vertex_position = cell->vertex(vertex);
421
422 for (unsigned d = 0; d < spacedim; ++d)
423 {
424 vertex_position[d] =
425 vertex_position[d] + displacement[dof_index + d];
426 }
427
428 dof_vertex_displaced.insert(cell->vertex_index(vertex));
429 }
430 }
431 }
432 }
433}

435template <int dim, int spacedim>
436void
437SerialSolid<dim, spacedim>::move_solid_triangulation_with_displacement()
438{
439 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
440
441 // Set of vertices which were displaced already. A set is used
442 // for efficiency when searching over it.
443 std::set<unsigned int> dof_vertex_displaced;
444 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
445
446 // Loop over all cells
447 for (const auto &cell : displacement_dh.active_cell_iterators())
448 {
449 for (unsigned int vertex = 0;
450 vertex < cell->reference_cell().n_vertices();
451 ++vertex)
452 {
453 if (!dof_vertex_displaced.count(cell->vertex_index(vertex)))
454 {
455 const auto dof_index = cell->vertex_dof_index(vertex, 0);
456 Point<spacedim> &vertex_position = cell->vertex(vertex);
457
458 for (unsigned d = 0; d < spacedim; ++d)
459 {
460 vertex_position[d] += displacement[dof_index + d];
461 }
462
463 dof_vertex_displaced.insert(cell->vertex_index(vertex));
464 }
465 }
466 }

470template <int dim, int spacedim>
471void
472SerialSolid<dim, spacedim>::write_output_results(
473 const std::shared_ptr<SimulationControl> &simulation_control)
474{
475 if (!output_bool)
476 return;
477
478 DataOut<dim, spacedim> data_out;
479 data_out.attach_dof_handler(displacement_dh);
480
481 {
482 std::vector<std::string> displacement_names(spacedim, "displacement");
483 std::vector<DataComponentInterpretation::DataComponentInterpretation>
484 data_component_interpretation(
485 spacedim, DataComponentInterpretation::component_is_part_of_vector);
486
487 data_out.add_data_vector(displacement,
488 displacement_names,
489 DataOut<dim, spacedim>::type_dof_data,
490 data_component_interpretation);
491 }
492
493 {
494 std::vector<std::string> displacement_names(spacedim,
495 "displacement_since_mapped");
496 std::vector<DataComponentInterpretation::DataComponentInterpretation>
497 data_component_interpretation(
498 spacedim, DataComponentInterpretation::component_is_part_of_vector);
499
500 data_out.add_data_vector(displacement_since_mapped,
501 displacement_names,
502 DataOut<dim, spacedim>::type_dof_data,
503 data_component_interpretation);
504 }
505
506 data_out.build_patches();
507
508 const std::string folder = simulation_control->get_output_path();
509 const std::string solution_name = [&] {
510 if constexpr (dim == 1)
511 {
512 return simulation_control->get_output_name() + ".solid_line." +
513 Utilities::int_to_string(id, 2);
514 }
515 if constexpr (dim == 2)
516 {
517 return simulation_control->get_output_name() + ".solid_surface." +
518 Utilities::int_to_string(id, 2);
519 }
520 if constexpr (dim == 3)
521 {
522 return simulation_control->get_output_name() + ".solid_volume." +
523 Utilities::int_to_string(id, 2);
524 }
525 }();
526 const unsigned int iter = simulation_control->get_step_number();
527 const double time = simulation_control->get_current_time();
528 const unsigned int group_files = simulation_control->get_group_files();
529
530 write_vtu_and_pvd<dim, spacedim>(pvdhandler,
531 data_out,
532 folder,
533 solution_name,
534 time,
535 iter,
536 group_files,
537 this->mpi_communicator);
538}
