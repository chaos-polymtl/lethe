7/**
8 * @brief IDs associated to the different subequations solved in Lethe.
9 */
10enum class CLSSubequationsID : unsigned int
11{
12 /// CLS phase fraction gradient L2 projection
13 phase_gradient_projection = 0,
14 /// CLS curvature L2 projection
15 curvature_projection = 1,
16 /// CLS algebraic interface reinitialization
17 algebraic_interface_reinitialization = 2
18};

20/**
21 * @brief Get the CLSSubequationsID associated to a specific CLS subequation
22 * from a string.
23 *
24 * @param[in] cls_subequation_name String corresponding to a particular CLS
25 * subequation.
26 *
27 * @return CLSSubequationsID corresponding to @p cls_subequation_name.
28 */
29inline CLSSubequationsID
30get_cls_subequation_id(const std::string &cls_subequation_name)
31{
32 if (cls_subequation_name == "CLS phase gradient L2 projection")
33 return CLSSubequationsID::phase_gradient_projection;
34 else if (cls_subequation_name == "CLS curvature L2 projection")
35 return CLSSubequationsID::curvature_projection;
36 else if (cls_subequation_name == "CLS PDE-based interface reinitialization")
37 return CLSSubequationsID::algebraic_interface_reinitialization;
38 else
39 throw(std::invalid_argument("Invalid CLS subequation name. Options are: \n"
40 " <CLS phase gradient L2 projection>\n"
41 " <CLS curvature L2 projection>\n"
42 " <CLS PDE-based interface reinitialization>"));
43}
