12 template <int dim, typename PropertiesIndex>
13 std::shared_ptr<ParticleParticleContactForceBase<dim, PropertiesIndex>>
14 set_particle_particle_contact_force_model(
15  const DEMSolverParameters<dim> &dem_parameters)
16 {
17  ParticleParticleContactForceModel particle_particle_contact_force_model =
18  dem_parameters.model_parameters.particle_particle_contact_force_model;
19 
20  std::shared_ptr<ParticleParticleContactForceBase<dim, PropertiesIndex>>
21  particle_particle_contact_force_object;
22 
23  switch (particle_particle_contact_force_model)
24  {
25  case ParticleParticleContactForceModel::linear:
26  {
27  set_rolling_resistance_model<
28  dim,
29  PropertiesIndex,
30  ParticleParticleContactForceModel::linear>(
31  dem_parameters, particle_particle_contact_force_object);
32  break;
33  }
34  case ParticleParticleContactForceModel::hertz_mindlin_limit_overlap:
35  {
36  set_rolling_resistance_model<
37  dim,
38  PropertiesIndex,
39  ParticleParticleContactForceModel::hertz_mindlin_limit_overlap>(
40  dem_parameters, particle_particle_contact_force_object);
41  break;
42  }
43  case ParticleParticleContactForceModel::hertz_mindlin_limit_force:
44  {
45  set_rolling_resistance_model<
46  dim,
47  PropertiesIndex,
48  ParticleParticleContactForceModel::hertz_mindlin_limit_force>(
49  dem_parameters, particle_particle_contact_force_object);
50  break;
51  }
52  case ParticleParticleContactForceModel::hertz:
53  {
54  set_rolling_resistance_model<
55  dim,
56  PropertiesIndex,
57  ParticleParticleContactForceModel::hertz>(
58  dem_parameters, particle_particle_contact_force_object);
59  break;
60  }
61  case ParticleParticleContactForceModel::hertz_JKR:
62  {
63  set_rolling_resistance_model<
64  dim,
65  PropertiesIndex,
66  ParticleParticleContactForceModel::hertz_JKR>(
67  dem_parameters, particle_particle_contact_force_object);
68  break;
69  }
70  case ParticleParticleContactForceModel::DMT:
71  {
72  set_rolling_resistance_model<dim,
73  PropertiesIndex,
74  ParticleParticleContactForceModel::DMT>(
75  dem_parameters, particle_particle_contact_force_object);
76  break;
77  }
78  default:
79  {
80  throw std::runtime_error(
81  "The chosen particle-particle contact force model is invalid");
82  break;
83  }
84  }
85  return particle_particle_contact_force_object;
86 }

89 template <int dim,
90  typename PropertiesIndex,
91  Parameters::Lagrangian::ParticleParticleContactForceModel
92  particle_particle_contact_force_model>
93 void
94 set_rolling_resistance_model(
95  const DEMSolverParameters<dim> &dem_parameters,
96  std::shared_ptr<ParticleParticleContactForceBase<dim, PropertiesIndex>>
97  &particle_particle_contact_force_object)
98 {
99  RollingResistanceMethod rolling_resistance_method =
100  dem_parameters.model_parameters.rolling_resistance_method;
101 
102  switch (rolling_resistance_method)
103  {
104  case RollingResistanceMethod::no_resistance:
105  {
106  particle_particle_contact_force_object =
107  std::make_shared<ParticleParticleContactForce<
108  dim,
109  PropertiesIndex,
110  particle_particle_contact_force_model,
111  RollingResistanceMethod::no_resistance>>(dem_parameters);
112  break;
113  }
114  case RollingResistanceMethod::constant_resistance:
115  {
116  particle_particle_contact_force_object =
117  std::make_shared<ParticleParticleContactForce<
118  dim,
119  PropertiesIndex,
120  particle_particle_contact_force_model,
121  RollingResistanceMethod::constant_resistance>>(dem_parameters);
122  break;
123  }
124  case RollingResistanceMethod::viscous_resistance:
125  {
126  particle_particle_contact_force_object =
127  std::make_shared<ParticleParticleContactForce<
128  dim,
129  PropertiesIndex,
130  particle_particle_contact_force_model,
131  RollingResistanceMethod::viscous_resistance>>(dem_parameters);
132  break;
133  }
134  default:
135  throw std::runtime_error("Invalid rolling resistance method");
136  }
137 }
