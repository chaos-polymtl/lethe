18/**
19 * @brief Generates a 3D mesh for the Birmingham fluidized bed geometry.
20 *
21 * The geometry is taken from:
22 * P. Fede, O. Simonin, and A. Ingram, "3D numerical simulation of a
23 * lab-scale pressurized dense fluidized bed focussing on the effect of
24 * the particle-particle restitution coefficient and particle-wall boundary
25 * conditions," Chemical Engineering Science, vol. 142, pp. 215-235, 2016.
26 * DOI: 10.1016/j.ces.2015.11.016
27 *
28 * The geometry consists of three or four sections joined along the x-axis:
29 * - A bottom cylinder of small radius (0.077 m),
30 * - A truncated cone expanding from the small to a larger radius (0.127 m),
31 * - A top cylinder of the larger radius,
32 * - (optional) A rectangular chimney pipe extending from an off-center cell
33 * of the top end cap, serving as the outlet.
34 *
35 * The chimney is controlled by the @p enable_chimney constructor argument.
36 *
37 * Boundary IDs are assigned as follows:
38 * - 0: wall surfaces (curved cylinder/cone laterals; when the chimney is
39 * enabled this also includes the chimney walls and the annular top wall
40 * around the chimney opening),
41 * - 1: inlet end cap (x = -inlet_offset),
42 * - 2: outlet end cap (x = total length), or chimney top face when the
43 * chimney is enabled.
44 *
45 * A CylindricalManifold is attached to the curved lateral surfaces so that
46 * mesh refinement preserves the circular cross-sections and the cone taper.
47 * When the chimney is enabled, its walls and the annular top wall remain
48 * flat.
49 *
50 * @tparam dim The dimension of the mesh, must be 3.
51 * @tparam spacedim The dimension of the space, must be 3.
52 */
53template <int dim, int spacedim>
54class GridBirminghamFluidizedBed
55{
56public:
57 /**
58 * @brief Constructor that parses grid parameters from a colon-separated
59 * string.
60 *
61 * @param[in] grid_arguments A string with the following format:
62 * @code "enable_chimney : inlet_offset" @endcode
63 *
64 * | # | Field | Format | Required | Description |
65 * |----|----------------|------------|----------|----------------------------------------------|
66 * | 0 | enable_chimney | true/false | no | Add chimney outlet (default: true) |
67 * | 1 | inlet_offset | double | no | Extend bottom cylinder into -x (default: 0.0) |
68 *
69 * Example: @code "true : 0.1" @endcode
70 */
71 GridBirminghamFluidizedBed(const std::string &grid_arguments);
72
73 /**
74 * @brief Generate the Birmingham fluidized bed mesh.
75 *
76 * The cylindrical geometry (bottom cylinder, truncated cone, top cylinder)
77 * is first merged and refined with a CylindricalManifold to obtain a
78 * finer mesh on the top end cap. The refined mesh is then flattened and
79 * an off-center face on the top end cap is selected for the chimney
80 * attachment. A hexahedral chimney pipe is built from that face's exact
81 * vertices and merged with the flattened mesh. The number of pre-chimney
82 * refinements is controlled by the hardcoded parameter
83 * @p n_chimney_refinements.
84 *
85 * @param[out] triangulation The triangulation to fill with the mesh.
86 */
87 void
88 make_grid(Triangulation<dim, spacedim> &triangulation);
89
90private:
91 /// Arguments used to generate the grid
92 std::string grid_arguments;
93
94 /// Flag used to indicate if the chimney is added or not to the mesh
95 bool enable_chimney;
96
97 /// Length by which the bottom cylinder extends into negative x
98 double inlet_offset;
99};
