31 /**
32  * @brief Calculate the area of intersection between a circular (2D) particle and a circle.
33  *
34  * @param[in] r_particle Radius of the particle
35  *
36  * @param[in] r_circle Radius of the circle
37  *
38  * @param[in] neighbor_distance Distance between the particle and the circle
39  */
40 inline double
41 particle_circle_intersection_2d(double r_particle,
42  double r_circle,
43  double neighbor_distance)
44 {
45  return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
46  cos((pow(neighbor_distance, 2) +
47  pow(r_particle, 2) - pow(r_circle, 2)) /
48  (2 * neighbor_distance * r_particle))) +
49  Utilities::fixed_power<2, double>(r_circle) *
50  Utilities::fixed_power<-1, double>(
51  cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
52  pow(r_circle, 2)) /
53  (2 * neighbor_distance * r_circle))) -
54  0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
55  (neighbor_distance + r_particle - r_circle) *
56  (neighbor_distance - r_particle + r_circle) *
57  (neighbor_distance + r_particle + r_circle));
58 }

60 /**
61  * @brief Calculate the volume of intersection between a spherical (3D) particle and a sphere.
62  *
63  * @param[in] r_particle Radius of the particle
64  *
65  * @param[in] r_sphere Radius of the sphere
66  *
67  * @param[in] neighbor_distance Distance between the particle and the sphere
68  */
69 
70 inline double
71 particle_sphere_intersection_3d(double r_particle,
72  double r_sphere,
73  double neighbor_distance)
74 {
75  return M_PI *
76  Utilities::fixed_power<2, double>(r_sphere + r_particle -
77  neighbor_distance) *
78  (Utilities::fixed_power<2, double>(neighbor_distance) +
79  (2 * neighbor_distance * r_particle) -
80  (3 * Utilities::fixed_power<2, double>(r_particle)) +
81  (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
82  (3 * Utilities::fixed_power<2, double>(r_sphere))) /
83  (12 * neighbor_distance);
84 }

87  * @brief Void fraction calculator. This class manages the calculation of the
88  * void fraction which is used as an auxiliary field for the solvers that solve
89  * the Volume-Averaged Navier-Stokes equations. The present architecture
90  * of the class support all of the different void fraction calculation methods
91  * within a single class instead of building a class hierarchy. This is
92  * because there are only a few void fraction calculation strategies.

106  * @param particle_handler The particle handler used to manage the particles in the simulation. This object must be provided even if the particles are not used to establish the void fraction.

281  /**
282  * @brief Calculate and return the periodic offset distance vector of the domain which is needed
283  * for the periodic boundary conditions using the QCM or SPM for void fraction
284  * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
285  * periodic boundaries. This periodic boundary is then used as the reference
286  * for all particle locations shifted by this offset vector.
287  *
288  * @param[in] boundary_id The ID of one of the periodic boundaries
289  *
290  * @return The periodic offset vector.
291  */
292  inline Tensor<1, dim>
293  get_periodic_offset_distance(unsigned int boundary_id) const
294  {
295  Tensor<1, dim> offset;
296 
297  // Iterating over the active cells in the triangulation
298  for (const auto &cell : (*this->triangulation).active_cell_iterators())
299  {
300  if (cell->is_locally_owned() || cell->is_ghost())
301  {
302  if (cell->at_boundary())
303  {
304  // Iterating over cell faces
305  for (unsigned int face_id = 0; face_id < cell->n_faces();
306  ++face_id)
307  {
308  unsigned int face_boundary_id =
309  cell->face(face_id)->boundary_id();
310 
311  // Check if face is on the boundary, if so, get
312  // the periodic offset distance for one pair of periodic
313  // faces only since periodic boundaries are aligned with the
314  // direction and only axis are currently allowed
315  if (face_boundary_id == boundary_id)
316  {
317  Point<dim> face_center = cell->face(face_id)->center();
318  auto periodic_cell = cell->periodic_neighbor(face_id);
319  unsigned int periodic_face_id =
320  cell->periodic_neighbor_face_no(face_id);
321  Point<dim> periodic_face_center =
322  periodic_cell->face(periodic_face_id)->center();
323 
324  offset = periodic_face_center - face_center;
325 
326  return offset;
327  }
328  }
329  }
330  }
331  }
332 
333  // A zero tensor is returned in case no cells are found on the periodic
334  // boundaries on this processor. This processor won't handle particle in
335  // cells at periodic boundaries, so it won't affect any computation.
336  return offset;
337  }
