100  /**
101  * @brief Sets the load balancing iteration check function according
102  * to the chosen load balancing method. During the setup of the parameters,
103  * the validation of the method is already checked, so the default value is
104  * when the method is `none`.
105  *
106  * @return Return a function that returns a bool indicating if the current
107  * time step is a load balance iteration.
108  */
109  inline std::function<void()>
110  set_iteration_check_function()
111  {
112  using namespace Parameters::Lagrangian;
113 
114  switch (load_balance_method)
115  {
116  case ModelParameters::LoadBalanceMethod::once:
117  return [&] { check_load_balance_once(); };
118  case ModelParameters::LoadBalanceMethod::frequent:
119  return [&] { check_load_balance_frequent(); };
120  case ModelParameters::LoadBalanceMethod::dynamic:
121  return [&] { check_load_balance_dynamic(); };
122  case ModelParameters::LoadBalanceMethod::dynamic_with_sparse_contacts:
123  return [&] { check_load_balance_with_sparse_contacts(); };
124  default: // Default is no load balance (none)
125  return [&]() { return; };
126  }
127  }

173  /**
174  * @brief Connects the weight signals of the cells to the triangulation.
175  *
176  * In order to consider the particles when repartitioning the triangulation
177  * the algorithm needs to know three things:
178  * 1. How much weight to assign to each cell (how many particles are in
179  * there)
180  * 2. How to pack the particles before shipping data around
181  * 3. How to unpack the particles after repartitioning
182  * Attach the correct functions to the signals inside
183  * parallel::distributed::Triangulation, which will be called every time the
184  * load balancing or refinement functions are called.
185  *
186  * TODO: Remove preprocessor directives after deal.ii v.9.6 release
187  */

189  connect_weight_signals()
190  {
191 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
192  triangulation->signals.weight.connect(
193  [&](const typename Triangulation<dim>::cell_iterator &,
194  const typename Triangulation<dim>::CellStatus) -> unsigned int {
195  return cell_weight;
196  });
197 
198  triangulation->signals.weight.connect(
199  [&](const typename parallel::distributed::Triangulation<
200  dim>::cell_iterator &cell,
201  const typename parallel::distributed::Triangulation<dim>::CellStatus
202  status) -> unsigned int {
203  return this->calculate_total_cell_weight(cell, status);
204  });

206 #else

268 /**
269  * @brief Indicates to the triangulation how much computational work is
270  * expected to happen on this cell, and consequently how the domain needs to
271  * be partitioned.
272  *
273  * Every MPI rank receives a roughly equal amount of work (potentially not an
274  * equal number of cells). While the function is called from the outside,
275  * it is connected to the corresponding signal from inside this class,
276  * therefore it can be private. This function is the key component that allows
277  * dynamically balance the computational load. The function attributes a
278  * weight to every cell that represents the computational work on this cell.
279  * Here the majority of work is expected to happen on the particles, therefore
280  * the return value of this function is calculated based on the number of
281  * particles in the current cell. The function is connected to the
282  * cell_weight() signal inside the triangulation, and will be called once per
283  * cell, whenever the triangulation repartitions the domain between ranks.
284  *
285  * @param[in] cell The cell for which the load is calculated.
286  *
287  * @param[in] status The status of the cell related to the coarsening level.
288  *
289  * TODO: Remove preprocessor directives after deal.ii v.9.6 release
290  *
291  * @return The total weight of the cell.
292  */

308  /**
309  * @brief Indicates to the triangulation how much computational work is
310  * expected to happen on this cell, and consequently how the domain needs to
311  * be partitioned. Only used with the Adaptive Sparse Contacts mechanism.
312  *
313  * Similar to the calculate_total_cell_weight(), this function is used when
314  * the cell weight is adapted to the mobility status. For instance, if the
315  * cell is inactive, its computational load will be significantly lower than
316  * if it is a mobile cell since there is no force calculation and no velocity
317  * integration for the particles that lie within it. The weight of the cells
318  * must thus be adapted to the status of the cell.
319  *
320  * total cell weight = cell weight +
321  * load balancing factor * n particles * particle weight
322  *
323  * TODO: Remove preprocessor directives after deal.ii v.9.6 release
324  *
325  * @param[in] cell The cell for which the load is calculated
326  * @param[in] status The status of the cell related to the coarsening level
327  *
328  * @return The total weight of the cell
329  */
