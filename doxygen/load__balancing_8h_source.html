17/**
18 * @brief Manages the load balancing (repartitioning), of the domain according
19 * to the computational load of the cells and the particles.
20 * It is used by the DEM and the coupling CFD-DEM solvers.
21 *
22 * @tparam dim An integer that denotes the number of spatial dimensions.
23 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.

92 /**
93 * @brief Checks if the current iteration is a load balance iteration.
94 *
95 * @return bool indicating if this is a load balance iteration.
96 */
97 inline void
98 check_load_balance_iteration() const
99 {
100 return iteration_check_function();
101 }

103 /**
104 * @brief Sets the load balancing iteration check function according
105 * to the chosen load balancing method. During the setup of the parameters,
106 * the validation of the method is already checked, so the default value is
107 * when the method is `none`.
108 *
109 * @return Return a function that returns a bool indicating if the current
110 * time step is a load balance iteration.
111 */
112 inline std::function<void()>
113 set_iteration_check_function()
114 {
115 using namespace Parameters::Lagrangian;
116
117 switch (load_balance_method)
118 {
119 case ModelParameters<dim>::LoadBalanceMethod::once:
120 return [&] { check_load_balance_once(); };
121 case ModelParameters<dim>::LoadBalanceMethod::frequent:
122 return [&] { check_load_balance_frequent(); };
123 case ModelParameters<dim>::LoadBalanceMethod::dynamic:
124 return [&] { check_load_balance_dynamic(); };
125 case ModelParameters<
126 dim>::LoadBalanceMethod::dynamic_with_sparse_contacts:
127 return [&] { check_load_balance_with_sparse_contacts(); };
128 default: // Default is no load balance (none)
129 return [&]() { return; };
130 }
131 }

177 /**
178 * @brief Connects the weight signals of the cells to the triangulation.
179 *
180 * In order to consider the particles when repartitioning the triangulation
181 * the algorithm needs to know three things:
182 * 1. How much weight to assign to each cell (how many particles are in
183 * there)
184 * 2. How to pack the particles before shipping data around
185 * 3. How to unpack the particles after repartitioning
186 * Attach the correct functions to the signals inside
187 * parallel::distributed::Triangulation, which will be called every time the
188 * load balancing or refinement functions are called.
189 *
190 * TODO: Remove preprocessor directives after deal.ii v.9.6 release
191 */
192 inline void
193 connect_weight_signals()
194 {
195#if DEAL_II_VERSION_GTE(9, 7, 0)
196
197 triangulation->signals.weight.connect(
198 [&](const typename Triangulation<dim>::cell_iterator &cell,
199 const typename Triangulation<dim>::CellStatus) -> unsigned int {
200 return static_cast<int>(cell_weight_function->value(cell->center()));
201 });
202
203 triangulation->signals.weight.connect(
204 [&](const typename parallel::distributed::Triangulation<
205 dim>::cell_iterator &cell,
206 const typename parallel::distributed::Triangulation<dim>::CellStatus
207 status) -> unsigned int {
208 return this->calculate_total_cell_weight(cell, status);
209 });
210
211#elif (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 7)
212 triangulation->signals.weight.connect(
213 [&](const typename Triangulation<dim>::cell_iterator &,
214 const typename Triangulation<dim>::CellStatus) -> unsigned int {
215 return cell_weight;
216 });
217
218 triangulation->signals.weight.connect(
219 [&](const typename parallel::distributed::Triangulation<
220 dim>::cell_iterator &cell,
221 const typename parallel::distributed::Triangulation<dim>::CellStatus
222 status) -> unsigned int {
223 return this->calculate_total_cell_weight(cell, status);
224 });
225
226#else
227 // Connect the default cell weight function
228 triangulation->signals.weight.connect(
229 [&](const typename Triangulation<dim>::cell_iterator &,
230 const CellStatus) -> unsigned int { return cell_weight; });
231
232 triangulation->signals.weight.connect(
233 [&](const typename parallel::distributed::Triangulation<
234 dim>::cell_iterator &cell,
235 const CellStatus status) -> unsigned int {
236 return this->calculate_total_cell_weight(cell, status);
237 });
238#endif
239 }
240
241 /**
242 * @brief Connects the weight signals of the cells to the triangulation with
243 * the mobility status. Cells are reconnected when load balancing is checked
244 * with the new mobility status.
245 *
246 * TODO: Remove preprocessor directives after deal.ii v.9.6 release
247 */
248 inline void
249 connect_mobility_status_weight_signals()
250 {
251 // Clear and connect a new cell weight function
252 triangulation->signals.weight.disconnect_all_slots();
253
254 // Connect, or reconnect, the default cell weight function
255 triangulation->signals.weight.connect(
256 [&](const typename Triangulation<dim>::cell_iterator &,
257 const CellStatus) -> unsigned int { return cell_weight; });
258
259 triangulation->signals.weight.connect(
260 [&](const typename parallel::distributed::Triangulation<
261 dim>::cell_iterator &cell,
262 const CellStatus status) -> unsigned int {
263 return this->calculate_total_cell_weight_with_mobility_status(cell,
264 status);
265 });
266 }

269 /**
270 * @brief Indicates to the triangulation how much computational work is
271 * expected to happen on this cell, and consequently how the domain needs to
272 * be partitioned.
273 *
274 * Every MPI rank receives a roughly equal amount of work (potentially not an
275 * equal number of cells). While the function is called from the outside,
276 * it is connected to the corresponding signal from inside this class,
277 * therefore it can be private. This function is the key component that allows
278 * dynamically balance the computational load. The function attributes a
279 * weight to every cell that represents the computational work on this cell.
280 * Here the majority of work is expected to happen on the particles, therefore
281 * the return value of this function is calculated based on the number of
282 * particles in the current cell. The function is connected to the
283 * cell_weight() signal inside the triangulation, and will be called once per
284 * cell, whenever the triangulation repartition the domain between ranks.
285 *
286 * @param[in] cell The cell for which the load is calculated.
287 *
288 * @param[in] status The status of the cell related to the coarsening level.
289 *
290 * TODO: Remove preprocessor directives after deal.ii v.9.6 release
291 *
292 * @return The total weight of the cell.
293 */
294 unsigned int
295 calculate_total_cell_weight(
296 const typename parallel::distributed::Triangulation<dim>::cell_iterator
297 &cell,
298 const CellStatus status) const;
299
300 /**
301 * @brief Indicates to the triangulation how much computational work is
302 * expected to happen on this cell, and consequently how the domain needs to
303 * be partitioned. Only used with the Adaptive Sparse Contacts mechanism.
304 *
305 * Similar to the calculate_total_cell_weight(), this function is used when
306 * the cell weight is adapted to the mobility status. For instance, if the
307 * cell is inactive, its computational load will be significantly lower than
308 * if it is a mobile cell since there is no force calculation and no velocity
309 * integration for the particles that lie within it. The weight of the cells
310 * must thus be adapted to the status of the cell.
311 *
312 * total cell weight = cell weight +
313 * load balancing factor * n particles * particle weight
314 *
315 *
316 * @param[in] cell The cell for which the load is calculated
317 * @param[in] status The status of the cell related to the coarsening level
318 *
319 * @return The total weight of the cell
320 */
321 unsigned int
322 calculate_total_cell_weight_with_mobility_status(
323 const typename parallel::distributed::Triangulation<dim>::cell_iterator
324 &cell,
325 const CellStatus status) const;
