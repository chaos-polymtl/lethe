11template <int dim>
12void
13VOFLinearSubequationsSolver<dim>::setup_dofs()
14{
15 // Get MPI communicator
16 auto mpi_communicator = this->triangulation->get_mpi_communicator();
17
18 // Distribute and renumber DoFs
19 this->dof_handler->distribute_dofs(*this->fe);
20 DoFRenumbering::Cuthill_McKee(*this->dof_handler);
21
22 // Get locally owned and relevant DoFs
23 this->locally_owned_dofs = this->dof_handler->locally_owned_dofs();
24 this->locally_relevant_dofs =
25 DoFTools::extract_locally_relevant_dofs(*this->dof_handler);
26
27 // Constraints
28 this->constraints.clear();
29 this->constraints.reinit(this->locally_owned_dofs,
30 this->locally_relevant_dofs);
31 DoFTools::make_hanging_node_constraints(*this->dof_handler,
32 this->constraints);
33
34 // Add periodic boundary conditions for the linear subequation
35 for (auto const &[id, type] :
36 this->simulation_parameters.boundary_conditions_vof.type)
37 {
38 if (type == BoundaryConditions::BoundaryType::periodic)
39 {
40 DoFTools::make_periodicity_constraints(
41 *this->dof_handler,
42 id,
43 this->simulation_parameters.boundary_conditions_vof
44 .periodic_neighbor_id.at(id),
45 this->simulation_parameters.boundary_conditions_vof
46 .periodic_direction.at(id),
47 this->constraints);
48 }
49 }
50
51 this->constraints.close();
52
53 // Sparsity pattern
54 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
55 DoFTools::make_sparsity_pattern(*this->dof_handler,
56 dsp,
57 this->constraints,
58 false);
59 SparsityTools::distribute_sparsity_pattern(dsp,
60 this->locally_owned_dofs,
61 mpi_communicator,
62 this->locally_relevant_dofs);
63
64 // Reinitialize system matrix and right-hand side (rhs)
65 this->system_matrix.reinit(this->locally_owned_dofs,
66 this->locally_owned_dofs,
67 dsp,
68 mpi_communicator);
69 this->system_rhs.reinit(this->locally_owned_dofs, mpi_communicator);
70
71 // Reinitialize solution vectors
72 this->present_solution->reinit(this->locally_owned_dofs,
73 this->locally_relevant_dofs,
74 mpi_communicator);
75 this->evaluation_point = *this->present_solution;
76
77
78 if (this->subequation_verbosity != Parameters::Verbosity::quiet)
79 {
80 std::string subequation_string =
81 this->subequations_interface.get_subequation_string(
82 this->subequation_id);
83 this->pcout << " Number of " << subequation_string
84 << " degrees of freedom: " << this->dof_handler->n_dofs()
85 << std::endl;
86 }
87
88 // Provide DoFHandler and solutions to the subequations interface
89 this->subequations_interface.set_dof_handler(this->subequation_id,
90 this->dof_handler);
91 this->subequations_interface.set_solution(this->subequation_id,
92 this->present_solution);
93}

96template <int dim>
97void
98VOFLinearSubequationsSolver<dim>::solve_void_fraction_linear_system()
99{
100 auto mpi_communicator = this->triangulation->get_mpi_communicator();
101
102 const AffineConstraints<double> &constraints_used = this->constraints;
103
104 const bool verbose(
105 this->subequation_verbosity != Parameters::Verbosity::quiet &&
106 this->linear_solver_verbosity != Parameters::Verbosity::quiet);
107
108 if (verbose)
109 {
110 std::string subequation_string =
111 this->subequations_interface.get_subequation_string(
112 this->subequation_id);
113
114 this->pcout << " -Solving " << subequation_string << ":" << std::endl;
115 }
116
117 // Set tolerance
118 const double linear_solver_tolerance =
119 this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
120 .minimum_residual;
121
122 // Solution vector
123 GlobalVectorType completely_distributed_solution(this->locally_owned_dofs,
124 mpi_communicator);
125
126 if (verbose)
127 {
128 this->pcout << " -Tolerance of iterative solver is: "
129 << linear_solver_tolerance << std::endl;
130 }
131
132 // ILU preconditioner
133 const unsigned int ilu_fill =
134 this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
135 .ilu_precond_fill;
136 const double ilu_atol =
137 this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
138 .ilu_precond_atol;
139 const double ilu_rtol =
140 this->simulation_parameters.linear_solver.at(PhysicsID::VOF)
141 .ilu_precond_rtol;
142 TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
143 ilu_fill, ilu_atol, ilu_rtol, 0);
144
145 this->ilu_preconditioner =
146 std::make_shared<TrilinosWrappers::PreconditionILU>();
147
148 this->ilu_preconditioner->initialize(this->system_matrix,
149 preconditionerOptions);
150
151 // CG solver
152 SolverControl solver_control(
153 this->simulation_parameters.linear_solver.at(PhysicsID::VOF).max_iterations,
154 linear_solver_tolerance,
155 true,
156 true);
157
158 TrilinosWrappers::SolverCG solver(solver_control);
159
160 solver.solve(this->system_matrix,
161 completely_distributed_solution,
162 this->system_rhs,
163 *this->ilu_preconditioner);
164
165 if (verbose)
166 {
167 this->pcout << " -Iterative solver took " << solver_control.last_step()
168 << " steps to reach a residual norm of "
169 << solver_control.last_value() << std::endl;
170 }
171
172 // Update constraints vector
173 constraints_used.distribute(completely_distributed_solution);
174
175 // Update solution vectors
176 *this->present_solution = completely_distributed_solution;
177 this->evaluation_point = *this->present_solution;
178}

189}
