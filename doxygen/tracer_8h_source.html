33/*
34 * Implementation of tracer as an auxiliary physics.
35 * Equation solved:
36 * dT/dt + u * gradT = D * div(grad T) + f
37 * with T the tracer function, D the diffusivity and f the forcing
38 */
39
40DeclException1(
41 TracerBoundaryConditionMissing,
42 types::boundary_id,
43 << "The boundary id: " << arg1
44 << " is defined in the triangulation, but not as a boundary condition for the tracer physics. Lethe does not assign a default boundary condition to boundary ids. Every boundary id defined within the triangulation must have a corresponding boundary condition defined in the input file.");

47template <int dim>
48class Tracer : public AuxiliaryPhysics<dim, GlobalVectorType>

50public:
51 Tracer(MultiphysicsInterface<dim> *multiphysics_interface,
52 const SimulationParameters<dim> &p_simulation_parameters,
53 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
54 p_triangulation,
55 std::shared_ptr<SimulationControl> p_simulation_control)
56 : AuxiliaryPhysics<dim, GlobalVectorType>(
57 p_simulation_parameters.non_linear_solver.at(PhysicsID::tracer))
58 , multiphysics(multiphysics_interface)
59 , computing_timer(p_triangulation->get_mpi_communicator(),
60 this->pcout,
61 TimerOutput::summary,
62 TimerOutput::wall_times)
63 , simulation_parameters(p_simulation_parameters)
64 , triangulation(p_triangulation)
65 , simulation_control(p_simulation_control)
66 , dof_handler(*triangulation)
67 {
68 if (simulation_parameters.mesh.simplex)
69 {
70 // for simplex meshes
71 fe = std::make_shared<FE_SimplexP<dim>>(
72 simulation_parameters.fem_parameters.tracer_order);
73 mapping = std::make_shared<MappingFE<dim>>(*fe);
74 cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
75 face_quadrature =
76 std::make_shared<QGaussSimplex<dim - 1>>(fe->degree + 1);
77 if (simulation_parameters.fem_parameters.tracer_uses_dg)
78 AssertThrow(
79 false,
80 ExcMessage(
81 "Discontinuous Galerkin elements cannot be simplex in the tracer physics."));
82 }
83 else
84 {
85 // Usual case, for quad/hex meshes
86 if (simulation_parameters.fem_parameters.tracer_uses_dg)
87 fe = std::make_shared<FE_DGQ<dim>>(
88 simulation_parameters.fem_parameters.tracer_order);
89 else
90 fe = std::make_shared<FE_Q<dim>>(
91 simulation_parameters.fem_parameters.tracer_order);
92 // The mapping must be a minimum of degree 1. This is necessary in the
93 // DG method since DG_Q(0) elements are adequate but a mapping of degree
94 // 0 does not make sense.
95 mapping = std::make_shared<MappingQ<dim>>(std::max(int(fe->degree), 1));
96 cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
97 face_quadrature = std::make_shared<QGauss<dim - 1>>(fe->degree + 1);
98 }
99
100 // Allocate solution transfer
101 solution_transfer =
102 std::make_shared<SolutionTransfer<dim, GlobalVectorType>>(dof_handler);
103
104 // Set size of previous solutions using BDF schemes information
105 previous_solutions.resize(maximum_number_of_previous_solutions());
106
107 // Prepare previous solutions transfer
108 previous_solutions_transfer.reserve(previous_solutions.size());
109 for (unsigned int i = 0; i < previous_solutions.size(); ++i)
110 {
111 previous_solutions_transfer.emplace_back(
112 SolutionTransfer<dim, GlobalVectorType>(this->dof_handler));
113 }
114
115 // Change the behavior of the timer for situations when you don't want
116 // outputs
117 if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
118 this->computing_timer.disable_output();
119 }

293 /**
294 * @brief Verify consistency of the input parameters for boundary
295 * conditions to ensure that for every boundary condition within the
296 * triangulation, a boundary condition has been specified in the input file.
297 */
298 void
299 verify_consistency_of_boundary_conditions()
300 {
301 // Sanity check all the boundary conditions of the triangulation to
302 // ensure that they have a type.
303 std::vector<types::boundary_id> boundary_ids_in_triangulation =
304 this->triangulation->get_boundary_ids();
305 for (auto const &boundary_id_in_tria : boundary_ids_in_triangulation)
306 {
307 AssertThrow(
308 simulation_parameters.boundary_conditions_tracer.type.find(
309 boundary_id_in_tria) !=
310 simulation_parameters.boundary_conditions_tracer.type.end(),
311 TracerBoundaryConditionMissing(boundary_id_in_tria));

477 /**
478 * @brief Helper function to reinit the face velocity with the adequate solution.
479 * This prevents code duplication throughout the tracer. The function looks at
480 * the multiphysics interface to decide if the velocity is a block velocity or
481 * a regular velocity. Furthermore, it also checks if a time-averaged solution
482 * is required. Otherwise the code here would be copied four times.
483 *
484 * @param[in] velocity_cell the iterator to the cell where the velocity is to
485 * be reinitialized.
486 *
487 * @param[in] face_no the number of the face where the velocity is to be
488 * reinitialized.
489 *
490 * @param[in,out] scratch_data the scratch data to be used for the
491 * reinitialization.
492 */
493 inline void
494 reinit_face_velocity_with_adequate_solution(
495 const typename DoFHandler<dim>::active_cell_iterator &velocity_cell,
496 const unsigned int &face_no,
497 TracerScratchData<dim> &scratch_data)
498 {
499 if (multiphysics->fluid_dynamics_is_block())
500 {
501 // Check if the post processed variable needs to be calculated with the
502 // average velocity profile or the fluid solution.
503 if (this->simulation_parameters.initial_condition->type ==
504 Parameters::FluidDynamicsInitialConditionType::
505 average_velocity_profile &&
506 !this->simulation_parameters.multiphysics.fluid_dynamics &&
507 simulation_control->get_current_time() >
508 this->simulation_parameters.post_processing
509 .initial_time_for_average_velocities)
510 {
511 scratch_data.reinit_face_velocity(
512 velocity_cell,
513 face_no,
514 *multiphysics->get_block_time_average_solution(
515 PhysicsID::fluid_dynamics),
516 this->simulation_parameters.ale,
517 this->simulation_parameters.tracer_drift_velocity.drift_velocity);
518 }
519 else
520 {
521 scratch_data.reinit_face_velocity(
522 velocity_cell,
523 face_no,
524 *multiphysics->get_block_solution(PhysicsID::fluid_dynamics),
525 this->simulation_parameters.ale,
526 this->simulation_parameters.tracer_drift_velocity.drift_velocity);
527 }
528 }
529 else
530 {
531 // Check if the post processed variable needs to be calculated with the
532 // average velocity profile or the fluid solution.
533 if (this->simulation_parameters.initial_condition->type ==
534 Parameters::FluidDynamicsInitialConditionType::
535 average_velocity_profile &&
536 !this->simulation_parameters.multiphysics.fluid_dynamics &&
537 simulation_control->get_current_time() >
538 this->simulation_parameters.post_processing
539 .initial_time_for_average_velocities)
540 {
541 scratch_data.reinit_face_velocity(
542 velocity_cell,
543 face_no,
544 *multiphysics->get_time_average_solution(
545 PhysicsID::fluid_dynamics),
546 this->simulation_parameters.ale,
547 this->simulation_parameters.tracer_drift_velocity.drift_velocity);
548 }
549 else
550 {
551 scratch_data.reinit_face_velocity(
552 velocity_cell,
553 face_no,
554 *multiphysics->get_solution(PhysicsID::fluid_dynamics),
555 this->simulation_parameters.ale,
556 this->simulation_parameters.tracer_drift_velocity.drift_velocity);
557 }
558 }
559 }
