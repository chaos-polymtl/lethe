20 FlowControl<dim>::calculate_beta(const double &average_velocity_n,
21  const double &dt,
22  const unsigned int &step_number)
23 {
24  double beta_n1;
25  // Check if disabling no_force and modify threshold factor (> step time 1):
26  // If average velocity target is reached without any beta force applied,
27  // the "no_force" is disabled. It means that the average velocity is now
28  // reached the target value without force OR that slowing down the flow with
29  // the only pressure drop (no force applied) is ineffective (too big pressure
30  // drop). In case the average velocity is in the initial threshold but the
31  // pressure drop is too small to slow down the flow, the threshold factor
32  // decreases by 2. Otherwise, the average velocity may take a lot of time to
33  // reach the target value only with pressure drop.
34  if (abs(beta_n) < 1e-6 && step_number > 1)
35  {
36  if (abs(average_velocity_n) < abs(average_velocity_0))
37  no_force = false;
38  else if (abs(average_velocity_1n - average_velocity_n) <
39  abs(average_velocity_n - average_velocity_0) &&
40  threshold_factor > 1.00125)
41  threshold_factor = 0.5 * (1 + threshold_factor);
42  }
43 
44  if (step_number == 0)
45  {
46  // No force applied at first time step.
47  // At this moment, calculate_beta is not called at this time step but
48  // the condition prevents a beta calculation if the way it's called
49  // changes.
50  beta_n1 = 0.0;
51  }
52  else if (step_number == 1)
53  {
54  // Apply initial beta force if user defined (> than 0)
55  // otherwise, it is calculated by the proportional controller
56  beta_n1 = (abs(beta_0) > 1e-6 ?
57  beta_0 :
58  proportional_flow_controller(average_velocity_n, dt));
59  }
60  else if (abs(average_velocity_n) > abs(average_velocity_0) &&
61  abs(average_velocity_n) <
62  threshold_factor * abs(average_velocity_0) &&
63  no_force)
64  {
65  // If the average velocity is between targeted value and the threshold
66  // (meaning it is sightly over the value) and it has not reached once the
67  // value during the simulation (no_force is enable), it decreases by
68  // itself (pressure drop).
69  beta_n1 = 0.0;
70  }
71  else if (step_number == 2)
72  {
73  // Since the main controller take the previous beta value, it is better
74  // to use the proportional controller for this second time step since
75  // it won't be affected by the first beta value that may cause overshoot
76  // or undershoot.
77  beta_n1 = proportional_flow_controller(average_velocity_n, dt);
78  }
79  else
80  {
81  // Main controller
82  beta_n1 = main_flow_controller(average_velocity_n, dt);
83  }
84 
85  // If the new beta is in the user defined threshold of the old beta, the new
86  // beta is kept as the previous beta. This could allow avoiding unnecessary
87  // reassembly of the matrix because of the force term when reuse matrix option
88  // is enable for the non-linear solver
89  if (abs(beta_n1 - beta_n) < abs(beta_threshold * beta_n))
90  beta_n1 = beta_n;
91 
92  // Setting beta coefficient to the tensor according to the flow direction.
93  beta[flow_direction] = beta_n1;
94 
95  // Assigning values of this time step as previous values prior next
96  // calculation.
97  beta_n = beta_n1;
98  average_velocity_1n = average_velocity_n;
99 }
