21template <int dim>
22void
23FlowControl<dim>::calculate_beta(const double &average_velocity_n,
24 const double &dt,
25 const unsigned int &step_number)
26{
27 double beta_n1;
28 // Check if disabling no_force and modify threshold factor (> step time 1):
29 // If average velocity target is reached without any beta force applied,
30 // the "no_force" is disabled. It means that the average velocity is now
31 // reached the target value without force OR that slowing down the flow with
32 // the only pressure drop (no force applied) is ineffective (too big pressure
33 // drop). In case the average velocity is in the initial threshold but the
34 // pressure drop is too small to slow down the flow, the threshold factor
35 // decreases by 2. Otherwise, the average velocity may take a lot of time to
36 // reach the target value only with pressure drop.
37 if (abs(beta_n) < 1e-6 && step_number > 1)
38 {
39 if (abs(average_velocity_n) < abs(average_velocity_0))
40 no_force = false;
41 else if (abs(average_velocity_1n - average_velocity_n) <
42 abs(average_velocity_n - average_velocity_0) &&
43 threshold_factor > 1.00125)
44 threshold_factor = 0.5 * (1 + threshold_factor);
45 }
46
47 if (step_number == 0)
48 {
49 // No force applied at first time step.
50 // At this moment, calculate_beta is not called at this time step but
51 // the condition prevents a beta calculation if the way it's called
52 // changes.
53 beta_n1 = 0.0;
54 }
55 else if (step_number == 1)
56 {
57 // Apply initial beta force if user defined (> than 0)
58 // otherwise, it is calculated by the proportional controller
59 beta_n1 = (abs(beta_0) > 1e-6 ?
60 beta_0 :
61 proportional_flow_controller(average_velocity_n, dt));
62 }
63 else if (abs(average_velocity_n) > abs(average_velocity_0) &&
64 abs(average_velocity_n) <
65 threshold_factor * abs(average_velocity_0) &&
66 no_force)
67 {
68 // If the average velocity is between targeted value and the threshold
69 // (meaning it is sightly over the value) and it has not reached once the
70 // value during the simulation (no_force is enable), it decreases by
71 // itself (pressure drop).
72 beta_n1 = 0.0;
73 }
74 else if (step_number == 2)
75 {
76 // Since the main controller take the previous beta value, it is better
77 // to use the proportional controller for this second time step since
78 // it won't be affected by the first beta value that may cause overshoot
79 // or undershoot.
80 beta_n1 = proportional_flow_controller(average_velocity_n, dt);
81 }
82 else
83 {
84 // Main controller
85 beta_n1 = main_flow_controller(average_velocity_n, dt);
86 }
87
88 // If the new beta is in the user defined threshold of the old beta, the new
89 // beta is kept as the previous beta. This could allow avoiding unnecessary
90 // reassembly of the matrix because of the force term when reuse matrix option
91 // is enable for the non-linear solver
92 if (abs(beta_n1 - beta_n) < abs(beta_threshold * beta_n))
93 beta_n1 = beta_n;
94
95 // Setting beta coefficient to the tensor according to the flow direction.
96 beta[flow_direction] = beta_n1;
97
98 // Assigning values of this time step as previous values prior next
99 // calculation.
100 beta_n = beta_n1;
101 average_velocity_1n = average_velocity_n;
102}

104template <int dim>
105void
106FlowControl<dim>::save(const std::string &prefix)
107{
108 std::string filename = prefix + ".flowcontrol";
109 std::ofstream output(filename.c_str());
110 output << "Flow control" << std::endl;
111 output << "Previous_beta " << beta_n << std::endl;
112 output << "Previous_average_velocity " << average_velocity_1n << std::endl;
113 output << "No_force " << no_force << std::endl;
114 output << "Threshold_factor " << threshold_factor << std::endl;
115}

122 std::ifstream input(filename.c_str());
123 AssertThrow(input, ExcFileNotOpen(filename));
124
125 std::string buffer;
126 std::getline(input, buffer);
127 input >> buffer >> beta_n;
128 input >> buffer >> average_velocity_1n;
129 input >> buffer >> no_force;
130 input >> buffer >> threshold_factor;
131}
