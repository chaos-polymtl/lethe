17/**
18 * @brief A pure virtual class that serves as an interface for all
19 * of the assemblers for the particle_fluid interactions of the
20 * VANS equations
21 *
22 * @tparam dim An integer that denotes the number of spatial dimensions
23 *
24 * @ingroup assemblers
25 */
26template <int dim>
27class ParticleFluidAssemblerBase
28{
29public:
30 /**
31 * @brief Calculates the particle-fluid interaction for all of the particles.
32 * The forces calculated on the particles is stored within the fem_force. The
33 * overall beta coefficient for the cell, which is defined as the drag
34 * coefficient divided by the cell volume, is also calculated within this
35 * function.
36 * @param scratch_data Scratch data containing the Navier-Stokes information.
37 * It is important to note that the scratch data has to have been re-inited
38 * before calling for matrix assembly.
39 */
40
41 virtual void
42 calculate_particle_fluid_interactions(
43 NavierStokesScratchData<dim> &scratch_data) = 0;
44
45 /**
46 * @brief Default destructor.
47 */
48 virtual ~ParticleFluidAssemblerBase() = default;
49};

284/**
285 * @brief Class that assembles the drag force using the Koch and Hill drag model for the
286 * VANS equations where the momentum exchange coefficient
287 * beta = ((18 * mu * cell_void_fraction^2 *
288 * (1 - cell_void_fraction)) / pow(dp, 2)) *
289 * (f0 + 0.5 * f3 * cell_void_fraction * re) *
290 * Vp /(1 - cell_void_fraction) where f0 and f3 are functions given by:
291 * if ((1 - cell_void_fraction) < 0.4)
292 * {
293 * f0 = (1 + 3 * sqrt((1 - cell_void_fraction) / 2) +
294 * (135.0 / 64) * (1 - cell_void_fraction) *
295 * log(1 - cell_void_fraction) +
296 * 16.14 * (1 - cell_void_fraction)) /
297 * (1 + 0.681 * (1 - cell_void_fraction) -
298 * 8.48 * (1 - cell_void_fraction)^2 +
299 * 8.14 * (1 - cell_void_fraction)^3);
300 * }
301 * else if ((1 - cell_void_fraction) >= 0.4)
302 * {
303 * f0 = 10 * (1 - cell_void_fraction) / cell_void_fraction^3;
304 * }
305 * f3 = 0.0673 + 0.212 * (1 - cell_void_fraction) +
306 * 0.0232 / pow(cell_void_fraction, 5);
307 * @tparam dim An integer that denotes the number of spatial dimensions
308 *
309 * @ingroup assemblers
310 */
311
312template <int dim>
313class VANSAssemblerKochHill : public ParticleFluidAssemblerBase<dim>
314{
315public:
316 VANSAssemblerKochHill(const Parameters::CFDDEM &cfd_dem)
317 : cfd_dem(cfd_dem)
318 {}

333/**
334 * @brief Class that assembles the drag force using Beetstra model for the
335 * VANS equations where the
336 * normalized drag force = 10 * (1 - cell_void_fraction) /
337 * (pow(cell_void_fraction, 2)) + pow(cell_void_fraction, 2) * (1 + 1.5
338 * * pow((1 - cell_void_fraction), 0.5)) + 0.413 * re / (24 *
339 * pow(cell_void_fraction, 2)) *
340 * ((1 / cell_void_fraction) + 3 * (1 - cell_void_fraction) *
341 * cell_void_fraction

344 * -(1 + 4 * (1 - cell_void_fraction)) * 0.5));
345 * the drag coefficient = normalized_drag_force * 24 / Re_p
346 * The reference for this formulation of the Beestra model is given in the
347 * article *Complete liquid-solid momentum coupling for unresolved CFD-DEM

349 * https://www.sciencedirect.com/science/article/pii/S0301932220305346

375/**
376 * @brief Class that assembles the drag force using Gidaspow model for the
377 * VANS equations where the momentum_transfer_coefficient is calculated
378 according to the following Marchelli et al. (2020):
379 if (cell_void_fraction > 0.8)
380 {
381 momentum_transfer_coefficient =
382 (18 * pow(cell_void_fraction, -3.65) *
383 (1 + 0.15 * pow(Re_p[particle_number], 0.687))) *
384 (particle_properties[DEM::PropertiesIndex<DEM::SolverType::cfd_dem>::mass]
385 * viscosity /
386 (pow(particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp],
387 2) * particle_density));
388 }
389 else
390 {
391 // Assuming the sphericity of particles = 1
392 momentum_transfer_coefficient =
393 (150 * (1 - cell_void_fraction) / pow(cell_void_fraction, 2) +
394 1.75 * Re_p[particle_number] / pow(cell_void_fraction, 2)) *
395 (particle_properties[DEM::PropertiesIndex<DEM::SolverType::cfd_dem>::mass]
396 * viscosity /
397 (pow(particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp],
398 2) * particle_density));
399 }

403 *
404 * @ingroup assemblers
