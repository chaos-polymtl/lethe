17/**
18 * @brief A pure virtual class that serves as an interface for all
19 * of the assemblers for the particle_fluid interactions of the
20 * VANS equations
21 *
22 * @tparam dim An integer that denotes the number of spatial dimensions
23 *
24 * @ingroup assemblers
25 */
26template <int dim>
27class ParticleFluidAssemblerBase
28{
29public:
30 /**
31 * @brief calculate_particle_fluid_interactions calculates the particle-fluid interaction for all of the particles.
32 * The forces calculated on the particles is stored within the fem_force. The
33 * overall beta coefficient for the cell, whcih is defined as the drag
34 * coefficient divided by the cell volume, is also calculated within this
35 * function.
36 * @param scratch_data Scratch data containing the Navier-Stokes information.
37 * It is important to note that the scratch data has to have been re-inited
38 * before calling for matrix assembly.
39 * @param copy_data Destination where the local_rhs and loc
40 */
41
42 virtual void
43 calculate_particle_fluid_interactions(
44 NavierStokesScratchData<dim> &scratch_data) = 0;
45};

281/**
282 * @brief Class that assembles the drag force using the Koch and Hill drag model for the
283 * VANS equations where the momentum exchange coefficient
284 * beta = ((18 * mu * cell_void_fraction^2 *
285 (1 - cell_void_fraction)) / pow(dp, 2)) *
286 (f0 + 0.5 * f3 * cell_void_fraction * re) *
287 Vp /(1 - cell_void_fraction) where f0 and f3 are functions given by:
288 if ((1 - cell_void_fraction) < 0.4)
289 {
290 f0 = (1 + 3 * sqrt((1 - cell_void_fraction) / 2) +
291 (135.0 / 64) * (1 - cell_void_fraction) *
292 log(1 - cell_void_fraction) +
293 16.14 * (1 - cell_void_fraction)) /
294 (1 + 0.681 * (1 - cell_void_fraction) -
295 8.48 * (1 - cell_void_fraction)^2 +
296 8.14 * (1 - cell_void_fraction)^3);
297 }
298 else if ((1 - cell_void_fraction) >= 0.4)
299 {
300 f0 = 10 * (1 - cell_void_fraction) / cell_void_fraction^3;
301 }
302
303 f3 = 0.0673 + 0.212 * (1 - cell_void_fraction) +
304 0.0232 / pow(cell_void_fraction, 5);
305 * @tparam dim An integer that denotes the number of spatial dimensions
306 *
307 * @ingroup assemblers
308 */
309
310template <int dim>
311class VANSAssemblerKochHill : public ParticleFluidAssemblerBase<dim>
312{
313public:
314 VANSAssemblerKochHill(const Parameters::CFDDEM &cfd_dem)
315 : cfd_dem(cfd_dem)
316 {}

371/**
372 * @brief Class that assembles the drag force using Gidaspow model for the
373 * VANS equations where the momentum_transfer_coefficient is calculated
374 according to the following Marchelli et al. (2020):
375 if (cell_void_fraction > 0.8)
376 {
377 momentum_transfer_coefficient =
378 (18 * pow(cell_void_fraction, -3.65) *
379 (1 + 0.15 * pow(Re_p[particle_number], 0.687))) *
380 (particle_properties[DEM::PropertiesIndex<DEM::SolverType::cfd_dem>::mass]
381 * viscosity /
382 (pow(particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp],
383 2) * particle_density));
384 }
385 else
386 {
387 // Assuming the sphericity of particles = 1
388 momentum_transfer_coefficient =
389 (150 * (1 - cell_void_fraction) / pow(cell_void_fraction, 2) +
390 1.75 * Re_p[particle_number] / pow(cell_void_fraction, 2)) *
391 (particle_properties[DEM::PropertiesIndex<DEM::SolverType::cfd_dem>::mass]
392 * viscosity /
393 (pow(particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp],
394 2) * particle_density));
395 }

694/**
695 * @brief Calculate gamma grad-div stabilization constant for the VANS equations
696 *
697 * @param[in] velocity Magnitude of the velocity at the quadrature point
698 * @param[in] kinematic_viscosity
699 * @param[in] h The element size
700 * @param[in] c_star Scaling constant with units of length
701 */
702inline double
703calculate_gamma(double velocity,
704 double kinematic_viscosity,
705 double /*h*/,
706 double c_star)
707{
708 return kinematic_viscosity + c_star * velocity;
709}
