12/**
13 * Copies a two-dimensional tensor in a three-dimensional tensor. The third
14 * element of the three-dimensional tensor (in z direction) is set to zero.
15 * If the tensor given is already 3d, do nothing and return the tensor.
16 *
17 * @param tensor the tensor that should be transformed in 3d
18 * @return tensor_3d Three-dimensional output tensor
19 *
20 */
21
22template <int dim>
23inline Tensor<1, 3>
24tensor_nd_to_3d(const Tensor<1, dim> &tensor)
25{
26 Tensor<1, 3> tensor_3d;
27 tensor_3d[0] = tensor[0];
28 tensor_3d[1] = tensor[1];
29 if constexpr (dim > 2)
30 tensor_3d[2] = tensor[2];
31
32 if constexpr (dim == 2)
33 tensor_3d[2] = 0;
34
35 return tensor_3d;
36}
