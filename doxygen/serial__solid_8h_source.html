27/**
28 * A class that generates an object to manage a serial triangulation that
29 * represents a solid object. This class can represent solid objects for which
30 * the dimension do not match the spatial dimension. For example, a 2D surface
31 * defined in 3D or a line defined in 2D.
32 *
33 * @tparam dim An integer that denotes the dimensionality of the geometry
34 * @tparam spacedim An integer that denotes the dimension of the space occupied
35 * by the embedded solid
36 *
37 */
38
39template <int dim, int spacedim = dim>
40class SerialSolid
41{
42public:
43 SerialSolid(std::shared_ptr<Parameters::RigidSolidObject<spacedim>> &param,
44 unsigned int id);

96 Vector<double> &

100 * @brief Calculates the max displacement of the solid object. The max displacement is defined as

102 * maximal displacement in any direction.

104 * @return The reference to the vector of the displacement since the mapping
105 */
106 double
107 get_max_displacement_since_mapped();

109 /**
110 * @brief Returns translational velocity of the center of mass of the solid object
111 *
112 * @return The translational velocity of the solid object
113 */
114 inline Tensor<1, 3>
115 get_translational_velocity() const
116 {
117 if constexpr (spacedim == 3)
118 return this->current_translational_velocity;
119
120 if constexpr (spacedim == 2)
121 return tensor_nd_to_3d(this->current_translational_velocity);
122 }
123
124 /**
125 * @brief Returns the angular velocity of the center of mass of the solid object
126 *
127 * @return The angular velocity of the solid object
128 */
129 inline Tensor<1, 3>
130 get_angular_velocity() const
131 {
132 return this->current_angular_velocity;
133 }
134
135 /**
136 * @brief Returns the center of rotation of the center of mass of the solid object
137 *
138 * @return The center of rotation of the solid object
139 */
140 inline Point<3>
141 get_center_of_rotation() const
142 {
143 if constexpr (spacedim == 3)
144 return this->center_of_rotation;
145
146 if constexpr (spacedim == 2)
147 return point_nd_to_3d(this->center_of_rotation);
148 }
149
150 /**
151 * @brief Returns the temperature of the solid object
152 *
153 * @return The temperature of the solid object
154 */
155 inline double
156 get_temperature() const
157 {
158 return this->current_solid_temperature;
159 }
160
161 /**
162 * @brief Returns the thermal boundary type of the solid object
163 *
164 * @return The thermal boundary type of the solid object

168 {
169 return this->thermal_boundary_type;
170 }
171
172 /**
173 * @brief Returns the solid id of the solid object.
174 * This ID is an unsigned integer which is given to the solid object at
175 * compile time and is mostly used when writing files.
176 *
177 * @return id of the solid
178 */
179 inline unsigned int
180 get_solid_id() const

182 return id;
183 }

185 /**
186 * @brief Update the temperature of the solid object, which is a function of time.
187 *
188 * @param initial_time The initial time (time t) of the timestep. This is used to
189 * set the time of the temperature function.
190 */
191 void
192 update_solid_temperature(const double initial_time);
193
194 /**
195 * @brief Moves the vertices of the solid triangulation. This function
196 * uses explicitEuler scheme time integrator to displace the vertices
197 * of the solid triangulation and stores the displacement in an array
198 * in order to allow correct checkpointing of the triangulation.
199 *
200 * @param time_step The time_step value for this iteration
201 *
202 * @param initial_time The initial time (time t) of the timestep. This is used to
203 * set the time of the velocity function.
204 */
205 void
206 move_solid_triangulation(double time_step, double initial_time);
207
208 /**

232 void
