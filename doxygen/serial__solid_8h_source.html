23/**
24 * A class that generates an object to manage a serial triangulation that
25 * represents a solid object. This class can represent solid objects for which
26 * the dimension do not match the spatial dimension. For example, a 2D surface
27 * defined in 3D or a line defined in 2D.
28 *
29 * @tparam dim An integer that denotes the dimensionality of the geometry
30 * @tparam spacedim An integer that denotes the dimension of the space occupied
31 * by the embedded solid
32 *
33 */
34
35template <int dim, int spacedim = dim>
36class SerialSolid
37{
38public:
39 SerialSolid(std::shared_ptr<Parameters::RigidSolidObject<spacedim>> &param,
40 unsigned int id);
41

48 map_solid_in_background_triangulation(
49 const parallel::TriangulationBase<spacedim> &background_tr);

78 setup_containers();

104 /**
105 * @brief Calculates the max displacement of the solid object. The max

108 *
109 * @return The reference to the vector of the displacement since the mapping

111 double

114 /**
115 * @brief Returns translational velocity of the center of mass of the solid object
116 *
117 * @return The translational velocity of the solid object

119 inline Tensor<1, 3>

123 }
124
125 /**
126 * @brief Returns the angular velocity of the center of mass of the solid object
127 *

129 */
130 inline Tensor<1, 3>
131 get_angular_velocity() const

133 return this->current_angular_velocity;
134 }
135
136 /**
137 * @brief Returns the center of rotation of the center of mass of the solid object
138 *
139 * @return The center of rotation of the solid object
140 */
141 inline Point<3>
142 get_center_of_rotation() const
143 {
144 if constexpr (spacedim == 3)
145 return this->center_of_rotation;
146
147 else
148 return point_nd_to_3d(this->center_of_rotation);
149 }

173 /**
174 * @brief Returns the solid id of the solid object.

186 typedef std::map<
187 typename Triangulation<dim, spacedim>::active_cell_iterator,
188 std::vector<typename Triangulation<dim, spacedim>::active_cell_iterator>,
189 cut_cell_comparison<spacedim>>
190 triangulation_cell_neighbors_map;

192 /**
193 * @brief Returns the neighboring cells maps of the solid object triangulation.
194 * These maps store for each cell of the solid object which neighboring cells

196 * vertex-sharing.
197 *
198 * @return A tuple containing the four neighboring cells maps.
199 */
200 inline std::tuple<triangulation_cell_neighbors_map,

204 return std::tie(es_neighbors, vs_neighbors);
205 }

207 /**
208 * @brief Update the temperature of the solid object, which is a function of time.
209 *
210 * @param initial_time The initial time (time t) of the timestep. This is used to
211 * set the time of the temperature function.
212 */
213 void
214 update_solid_temperature(const double initial_time);
215
216 /**
217 * @brief Moves the vertices of the solid triangulation. This function
218 * uses explicitEuler scheme time integrator to displace the vertices
219 * of the solid triangulation and stores the displacement in an array
220 * in order to allow correct checkpointing of the triangulation.
221 *
222 * @param time_step The time_step value for this iteration
223 * @param initial_time The initial time (time t) of the timestep. This is used to
224 * set the time of the velocity function.
225 */
226 void
227 move_solid_triangulation(double time_step, double initial_time);
