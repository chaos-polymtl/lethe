371using namespace Parameters::Lagrangian;
372inline void
373setup_distributions(const LagrangianPhysicalProperties &lpp,
374 std::vector<std::shared_ptr<Distribution>>
375 &size_distribution_object_container,
376 double &maximum_particle_diameter,
377 unsigned int mpi_process_id,
378 const ConditionalOStream &pcout)
379{
380 for (unsigned int particle_type = 0; particle_type < lpp.particle_type_number;
381 particle_type++)
382 {
383 switch (lpp.distribution_type.at(particle_type))
384 {
385 case SizeDistributionType::uniform:
386 size_distribution_object_container[particle_type] =
387 std::make_shared<UniformDistribution>(
388 lpp.particle_average_diameter.at(particle_type));
389 break;
390 case SizeDistributionType::normal:
391 size_distribution_object_container[particle_type] =
392 std::make_shared<NormalDistribution>(
393 lpp.particle_average_diameter.at(particle_type),
394 lpp.particle_size_std.at(particle_type),
395 lpp.seed_for_distributions[particle_type] + mpi_process_id,
396 lpp.diameter_min_cutoff.at(particle_type),
397 lpp.diameter_max_cutoff.at(particle_type),
398 lpp.distribution_weighting_type.at(particle_type));
399 break;
400 case SizeDistributionType::lognormal:
401 size_distribution_object_container[particle_type] =
402 std::make_shared<LogNormalDistribution>(
403 lpp.particle_average_diameter.at(particle_type),
404 lpp.particle_size_std.at(particle_type),
405 lpp.seed_for_distributions[particle_type] + mpi_process_id,
406 lpp.diameter_min_cutoff.at(particle_type),
407 lpp.diameter_max_cutoff.at(particle_type),
408 lpp.distribution_weighting_type.at(particle_type));
409 break;
410 case SizeDistributionType::custom:
411 size_distribution_object_container[particle_type] =
412 std::make_shared<CustomDistribution>(
413 lpp.particle_custom_diameter.at(particle_type),
414 lpp.particle_custom_probability.at(particle_type),
415 lpp.seed_for_distributions[particle_type] + mpi_process_id,
416 // lpp.diameter_min_cutoff.at(particle_type),
417 // lpp.diameter_max_cutoff.at(particle_type),
418 lpp.distribution_weighting_type.at(particle_type));
419 break;
420 }
421 size_distribution_object_container[particle_type]
422 ->print_psd_declaration_string(particle_type, pcout);
423
424 maximum_particle_diameter = std::max(
425 maximum_particle_diameter,
426 size_distribution_object_container[particle_type]->find_max_diameter());
427 }
428}
