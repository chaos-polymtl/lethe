764  const double laser_end_time = laser_parameters->end_time;
765  const double beam_radius = laser_parameters->beam_radius;

770  {
771  // Get laser path

773  laser_scan_path.set_time(current_time);
774 
775  // For the laser heat source calculations, we need the radial distance, r,
776  // (in a dim-1 dimensional plane perpendicular to the laser beam
777  // direction) between the laser focal point and the quadrature points, as
778  // well as the axial distance, z, (in a laser emission direction) between
779  // the laser focal point and the quadrature points, separately. Hence, we
780  // get the laser location (laser_location) as a Point<dim>, in which the
781  // first and second components show the position of the laser focal point
782  // in a plane perpendicular to the emission direction, and the (dim-1)th
783  // component denotes the position of the laser focal point in the
784  // direction of emission. Then we use dim-1 auxiliary variables
785  // (Point<dim-1> laser_location_on_surface) to store the position of the
786  // laser focal point in the perpendicular plane to the emission direction,
787  // and double laser_location_in_depth to store the position of the laser
788  // focal point in the direction of emission.
789 
790  // Get laser location
791  Point<dim> laser_location;
792  laser_location[0] = laser_scan_path.value(
793  laser_location, laser_parameters->perpendicular_plane_coordinate_one);
794  if constexpr (dim == 3)
795  {
796  laser_location[1] = laser_scan_path.value(
797  laser_location,
798  laser_parameters->perpendicular_plane_coordinate_two);
799  }
800 
801  // Get laser location in depth (direction of emission).
802  double laser_location_in_depth = 0.0;
803  laser_location[dim - 1] =
804  laser_scan_path.value(laser_location,
805  laser_parameters->beam_orientation_coordinate);
806  laser_location_in_depth = laser_location[dim - 1];
807 
808  // Get laser location on the operation surface
809  Point<dim - 1> laser_location_on_surface;
810  for (unsigned int d = 0; d < dim - 1; d++)
811  laser_location_on_surface[d] = laser_location[d];
812 
813  const unsigned int n_q_points = scratch_data.n_q_points;
814  const unsigned int n_dofs = scratch_data.n_dofs;
815 
816  // Copy data elements
817  auto &strong_residual = copy_data.strong_residual;
818  auto &local_rhs = copy_data.local_rhs;
819 
820  // assembling right hand side
821  for (unsigned int q = 0; q < n_q_points; ++q)
822  {
823  // Get quadrature point location on surface to calculate its distance
824  // from the laser focal point in a perpendicular plane to the
825  // direction of emission
826  Point<dim - 1> quadrature_point_on_surface;
827  quadrature_point_on_surface[0] =
828  scratch_data.quadrature_points
829  [q][laser_parameters->perpendicular_plane_coordinate_one];
830  if constexpr (dim == 3)
831  {
832  quadrature_point_on_surface[1] =
833  scratch_data.quadrature_points
834  [q][laser_parameters->perpendicular_plane_coordinate_two];
835  }
836 
837  // Get quadrature point depth to calculate its distance from the laser
838  // focal point in the direction of emission. Note that in
839  // two-dimensional simulations, this variable is always equal to zero
840  const double quadrature_point_depth =
841  scratch_data.quadrature_points[q][laser_parameters
842  ->beam_orientation_coordinate];
843

845  // quadrature_point_depth is smaller than the laser_location_in_depth,
846  // or the laser beam is in positive direction and
847  // quadrature_point_depth is larger than laser_location_in_depth we
848  // need to apply the laser on the quadrature point, otherwise it is
849  // zero
850  double laser_quadrature_point_distance_in_depth = 0.0;
851  if ((beam_direction == 0 &&
852  quadrature_point_depth <= laser_location_in_depth) |
853  (beam_direction == 1 &&
854  quadrature_point_depth >= laser_location_in_depth))
855  laser_quadrature_point_distance_in_depth =
856  std::abs(quadrature_point_depth - laser_location_in_depth);
857 
858  // Store JxW in local variable for faster access
859  const double JxW = scratch_data.fe_values_T.JxW(q);

862  const double laser_heat_source =
863  (concentration_factor * absorptivity * laser_power /
864  (M_PI * beam_radius * beam_radius * penetration_depth)) *
865  exp(-1.0 * concentration_factor *
866  std::pow(laser_location_on_surface.distance(
867  quadrature_point_on_surface),
868  2.0) /
869  (beam_radius * beam_radius)) *
870  exp(-1.0 * laser_quadrature_point_distance_in_depth /
871  penetration_depth);
872  strong_residual[q] -= laser_heat_source;
873

875  {
876  const auto phi_T_i = scratch_data.phi_T[q][i];
877 
878  // rhs for : eta * alpha * P / (pi * R^2 * mu) * exp(-eta * r^2 /
879  // R^2) * exp(-|z| / mu) where eta, alpha, P, R, mu, r and z
880  // denote concentration factor, absorptivity, laser power, beam
881  // radius, penetration depth, radial distance from the laser focal
882  // point, and axial distance from the laser focal point,
883  // respectively.
884  local_rhs(i) += laser_heat_source * phi_T_i * JxW;
885  }
886  } // end loop on quadrature points
887  }
