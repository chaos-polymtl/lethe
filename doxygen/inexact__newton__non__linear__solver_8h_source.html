11/**
12 * @brief Non-linear solver for non-linear systems of equations which uses an inexact
13 * Newton method with \f$\alpha\f$ relaxation to ensure that the residual is
14 * monotonically decreasing. It allows to reuse the matrix and avoid its
15 * reassembly for every Newton step.
16 *
17 */
18template <typename VectorType>
19class InexactNewtonNonLinearSolver : public NonLinearSolver<VectorType>
20{
21public:
22 /**
23 * @brief Constructor.
24 *
25 * @param[in] physics_solver A pointer to the physics solver to which the
26 * non-linear solver is attached.
27 *
28 * @param[in] param Non-linear solver parameters as specified in the
29 * simulation parameter file.
30 *
31 */
32 InexactNewtonNonLinearSolver(PhysicsSolver<VectorType> *physics_solver,
33 const Parameters::NonLinearSolver &param);
34
35
36 /**
37 * @brief Solve the non-linear system of equations.
38 *
39 * @param[in] is_initial_step Boolean variable that controls which constraints
40 * are going to be applied to the equations depending on the time step.
41 *
42 */
43 void
44 solve(const bool is_initial_step) override;
45
46private:
47 bool matrix_requires_assembly;
48};

58template <typename VectorType>
59void
60InexactNewtonNonLinearSolver<VectorType>::solve(const bool is_initial_step)
61{
62 double global_res;
63 double current_res;
64 double last_res;
65 bool first_step = is_initial_step;
66 unsigned int outer_iteration = 0;
67 last_res = 1e6;
68 current_res = 1e6;
69 global_res = 1e6;
70
71 // current_res and global_res are different as one is defined based on the l2
72 // norm of the residual vector (current_res) and the other (global_res) is
73 // defined by the physical solver and may differ from the l2_norm of the
74 // residual vector. Only the global_res is compared to the tolerance in order
75 // to evaluate if the nonlinear system is solved. Only current_res is used for
76 // the alpha scheme as this scheme only monitors the convergence of the
77 // non-linear system of equation (the matrix problem).
78
79 PhysicsSolver<VectorType> *solver = this->physics_solver;
80
81 auto &evaluation_point = solver->get_evaluation_point();
82 auto &present_solution = solver->get_present_solution();
83
84 if (!this->params.reuse_matrix)
85 matrix_requires_assembly = true;
86
87 while ((global_res > this->params.tolerance) &&
88 outer_iteration < this->params.max_iterations)
89 {
90 evaluation_point = present_solution;
91
92 if (matrix_requires_assembly)
93 {
94 solver->assemble_system_matrix();
95 solver->setup_preconditioner();
96 }
97
98 if (this->params.force_rhs_calculation || outer_iteration == 0)
99 solver->assemble_system_rhs();
100
101 if (outer_iteration == 0)
102 {
103 auto &system_rhs = solver->get_system_rhs();
104 current_res = system_rhs.l2_norm();
105 last_res = current_res;
106 }
107
108 if (this->params.verbosity != Parameters::Verbosity::quiet)
109 {
110 solver->pcout << "Newton iteration: " << outer_iteration
111 << " - Residual: " << current_res << std::endl;
112 }
113
114 solver->solve_linear_system(first_step);
115 double last_alpha_res = current_res;
116
117 unsigned int alpha_iter = 0;
118 for (double alpha = 1.0; alpha > 1e-1; alpha *= 0.5)
119 {
120 auto &local_evaluation_point = solver->get_local_evaluation_point();
121 auto &newton_update = solver->get_newton_update();
122 local_evaluation_point = present_solution;
123 local_evaluation_point.add(alpha, newton_update);
124 solver->apply_constraints();
125 evaluation_point = local_evaluation_point;
126 solver->assemble_system_rhs();
127
128 auto &system_rhs = solver->get_system_rhs();
129 current_res = system_rhs.l2_norm();
130
131 if (this->params.verbosity != Parameters::Verbosity::quiet)
132 {
133 solver->pcout << "\talpha = " << std::setw(6) << alpha
134 << std::setw(0) << " res = "
135 << std::setprecision(this->params.display_precision)
136 << std::setw(6) << current_res;
137
138 solver->output_newton_update_norms(
139 this->params.display_precision);
140 }
141
142 // If it's not the first iteration of alpha check if the residual is
143 // smaller then the last alpha iteration. If it's not smaller we fall
144 // back to the last alpha iteration, which is generally the first one.
145 if (alpha_iter != 0 and current_res > last_alpha_res)
146 {
147 alpha = 2 * alpha;
148 local_evaluation_point = present_solution;
149 local_evaluation_point.add(alpha, newton_update);
150 solver->apply_constraints();
151 evaluation_point = local_evaluation_point;
152
153 if (this->params.verbosity != Parameters::Verbosity::quiet)
154 {
155 solver->pcout
156 << "\t\talpha value was kept at alpha = " << alpha
157 << " since alpha = " << alpha / 2
158 << " increased the residual" << std::endl;
159 }
160 current_res = last_alpha_res;
161 break;
162 }
163 if (current_res < this->params.step_tolerance * last_res ||
164 last_res < this->params.tolerance)
165 {
166 // If the current residual has decreased by less than
167 // the last residual * matrix_tolerance or more than one alpha
168 // iteration was required, the jacobian approximation has become
169 // insufficiently performant and, consequently, it is preferable
170 // to renew the jacobian matrix
171 if (current_res > this->params.matrix_tolerance * last_res ||
172 alpha_iter > 0)
173 matrix_requires_assembly = true;
174 else
175 matrix_requires_assembly = false;
176
177 break;
178 }
179 last_alpha_res = current_res;
180 alpha_iter++;
181 }
182
183 global_res = solver->get_current_residual();
184 present_solution = evaluation_point;
185 last_res = current_res;
186 ++outer_iteration;
187 }
188
189 // If the non-linear solver has not converged abort simulation if
190 // abort_at_convergence_failure=true
191 if ((global_res > this->params.tolerance) &&
192 outer_iteration >= this->params.max_iterations &&
193 this->params.abort_at_convergence_failure)
194 {
195 throw(std::runtime_error(
196 "Stopping simulation because the non-linear solver has failed to converge"));
197 }
198}
