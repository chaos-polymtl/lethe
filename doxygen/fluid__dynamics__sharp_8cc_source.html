74template <int dim>
75void
76FluidDynamicsSharp<dim>::generate_cut_cells_map()
77{
78 // Potential amelioration of this function. This function could try to use the
79 // information of the previous cut cell mapping as an initial guess of the new
80 // cell mapping by looking at the parent level of the cells to look for new
81 // cells cut.
82
83
84
85 // check all the cells if they are cut or not. Put the information in a map
86 // with the key being the cell.
87 TimerOutput::Scope t(this->computing_timer, "Map cut cells");
88
89
90 // A vector of the unordered map. Each map stores if a point is inside or
91 // outside of a given particle.
92 std::vector<std::unordered_map<types::global_dof_index, bool>>
93 inside_outside_support_point_vector(particles.size());
94 std::map<types::global_dof_index, Point<dim>> support_points =
95 DoFTools::map_dofs_to_support_points(*this->mapping, this->dof_handler);
96
97 // When the finite element order > 1, overconstrained cells are impossible
98 // since there is always at least a DOF inside the element that is not
99 // overconstrained. We therefore only have to check when the velocity order
100 // == 1.
101 const bool mapping_overconstrained_cells =
102 this->simulation_parameters.fem_parameters.velocity_order == 1;
103
104 cut_cells_map.clear();
105 cells_inside_map.clear();
106 overconstrained_fluid_cell_map.clear();
107 dof_with_more_then_one_particle.clear();
108 if (mapping_overconstrained_cells)
109 {
110 local_dof_overconstrained.reinit(this->locally_owned_dofs,
111 this->mpi_communicator);
112 dof_overconstrained.reinit(this->locally_owned_dofs,
113 this->locally_relevant_dofs,
114 this->mpi_communicator);
115 }
116
117 const auto &cell_iterator = this->dof_handler.active_cell_iterators();
118 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
119 const unsigned int dofs_per_face = this->fe->dofs_per_face;
120
121 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
122 std::vector<types::global_dof_index> local_face_dof_indices(dofs_per_face);
123
124 auto &v_x_fe = this->fe->get_sub_fe(0, 1);
125 const unsigned int dofs_per_cell_local_v_x = v_x_fe.dofs_per_cell;
126
127
128 // // Loop on all the cells and check if they are cut.
129 for (const auto &cell : cell_iterator)
130 {
131 if (cell->is_locally_owned() || cell->is_ghost())
132 {
133 bool cell_is_cut = false;
134 bool cell_is_inside = false;
135 unsigned int particle_id_which_cuts_this_cell = 0;
136 unsigned int particle_id_in_which_this_cell_is_embedded = 0;
137 std::vector<unsigned int> particles_cutting_this_cell;
138 // is the particle index that cuts the cell if it's cut. If the cell
139 // is not cut the default value is stored (0). If the cell is not cut
140 // this value will never be used.
141 // particles_cutting_this_cell count the number of particles
142 // that cut a cell if multiple particles cut this cell. This is used
143 // to treat cells that are cut by multiple particles differently.
144 cell->get_dof_indices(local_dof_indices);
145
146 for (unsigned int p = 0; p < particles.size(); ++p)
147 {
148 // Particles defined from a stl or an iges don't have a signed
149 // distance function. To identify cells that are cut a special
150 // algorithm is required. For this type of particle, no cell are
151 // identified as being inside the particle. This means that the
152 // fluid is always solved inside the particle.
153 if (particles[p].shape->additional_info_on_shape == "iges")
154 {
155 cell_is_cut =
156 cell_cut_by_p_absolute_distance(cell, support_points, p);
157 particle_id_which_cuts_this_cell = p;
158 if (cell_is_cut)
159 {
160 particles_cutting_this_cell.push_back(p);
161 }
162 }
163 else
164 {
165 unsigned int nb_dof_inside = 0;
166 for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
167 {
168 // Count the number of DOFs that are inside
169 // of the particles. If all the DOfs are on one side
170 // the cell is not cut by the boundary.
171 if (0 == this->fe->system_to_component_index(j).first)
172 {
173 // Check if we already have checked this point in a
174 // previously evaluated cell. If we didn't find it in
175 // the previous evaluation, we assess whether the DOF
176 // is inside or outside the shape.
177 if (particles[p].get_levelset(
178 support_points[local_dof_indices[j]], cell) <=
179 0)
180 {
181 ++nb_dof_inside;
182 inside_outside_support_point_vector
183 [p][local_dof_indices[j]] = true;
184 }
185 }
186 }
187
188 // If some DOFs are inside the boundary, the cell is inside
189 // the particle or cut by the particle.
190 if (nb_dof_inside != 0)
191 {
192 // If all the DOFs are inside the boundary this cell is
193 // inside the particle. Otherwise, the particle is cut.
194 if (nb_dof_inside == dofs_per_cell_local_v_x)
195 {
196 // We register only the particle with the lowest id
197 // as the particle in which this cell is embedded if
198 // the cell is in multiple particles.
199 if (particles_cutting_this_cell.size() == 0)
200 {
201 cell_is_cut = false;
202 particle_id_which_cuts_this_cell = 0;
203 cell_is_inside = true;
204 particle_id_in_which_this_cell_is_embedded = p;
205 }
206 break;
207 }
208 else
209 {
210 // We only register the particle with the lowest id as
211 // the particle by this cell is cut if the cell is cut
212 // in multiple particles.
213 if (particles_cutting_this_cell.size() == 0)
214 {
215 cell_is_cut = true;
216 particle_id_which_cuts_this_cell = p;
217 cell_is_inside = false;
218 particle_id_in_which_this_cell_is_embedded = 0;
219 }
220 particles_cutting_this_cell.push_back(p);
221 if (particles_cutting_this_cell.size() > 1)
222 {
223 for (const auto &dof_index : local_dof_indices)
224 {
225 dof_with_more_then_one_particle[dof_index] =
226 true;
227 }
228 }
229 }
230 }
231 else
232 {
233 if (particles_cutting_this_cell.size() == 0)
234 {
235 cell_is_cut = false;
236 particle_id_which_cuts_this_cell = 0;
237 cell_is_inside = false;
238 particle_id_in_which_this_cell_is_embedded = 0;
239 }
240 }
241 }
242 }
243
244 if (mapping_overconstrained_cells)
245 {
246 // If a cell is cut, we register its DOFs as "cut" also. This
247 // will allow us to detect fluid cells that have all their DOFs
248 // cut, and consider them as cut cells.
249 if (cell_is_cut)
250 {
251 for (const auto &id : local_dof_indices)
252 {
253 local_dof_overconstrained(id) = 1;
254 }
255 }
256
257 // We loop on every face of the cell, and if the face is at a
258 // boundary we count it as constrained.
259 size_t nb_faces = cell->n_faces();
260 for (unsigned int f = 0; f < nb_faces; ++f)
261 {
262 const auto face = cell->face(f);
263 if (face->at_boundary())
264 {
265 face->get_dof_indices(local_face_dof_indices);
266 for (const auto &id : local_face_dof_indices)
267 {
268 local_dof_overconstrained(id) = 1;
269 }
270 }
271 }
272 }
273
274 cut_cells_map[cell] = {cell_is_cut,
275 particle_id_which_cuts_this_cell,
276 particles_cutting_this_cell};
277 cells_inside_map[cell] = {cell_is_inside,
278 particle_id_in_which_this_cell_is_embedded};
279 }
280 }
281
282 if (mapping_overconstrained_cells)
283 {
284 local_dof_overconstrained.compress(VectorOperation::insert);
285 dof_overconstrained = local_dof_overconstrained;
286
287
288 for (const auto &cell : cell_iterator)
289 {
290 if (cell->is_locally_owned() || cell->is_ghost())
291 {
292 bool cell_is_cut;
293 bool cell_is_inside;
294 cell->get_dof_indices(local_dof_indices);
295 std::tie(cell_is_cut, std::ignore, std::ignore) =
296 cut_cells_map[cell];
297 std::tie(cell_is_inside, std::ignore) = cells_inside_map[cell];
298 if (!cell_is_cut && !cell_is_inside)
299 {
300 unsigned int number_of_vertices_in_cell =
301 local_dof_indices.size();
302 unsigned int number_of_vertices_cut = 0;
303 unsigned int particle_current_id =
304 std::numeric_limits<int>::max();
305
306 double particle_current_distance(
307 std::numeric_limits<double>::max());
308 double particle_candidate_distance(
309 std::numeric_limits<double>::max());
310
311 unsigned int global_dof_id;
312 // We count the number of vertices that are constrained.
313 for (unsigned int j = 0; j < number_of_vertices_in_cell; ++j)
314 {
315 global_dof_id = local_dof_indices[j];
316 if (dof_overconstrained(global_dof_id) > 0)
317 {
318 number_of_vertices_cut += 1;
319 }
320 }
321 // We check which particle is the closest to the cell, so
322 // that we know which one to use for applying the immersed
323 // boundary constraints.
324 if (number_of_vertices_in_cell == number_of_vertices_cut)
325 {
326 for (unsigned int p = 0; p < particles.size(); p++)
327 {
328 particle_candidate_distance =
329 particles[p].get_levelset(cell->barycenter(), cell);
330 if (abs(particle_candidate_distance) <
331 abs(particle_current_distance))
332 {
333 particle_current_id = p;
334 particle_current_distance =
335 particle_candidate_distance;
336 }
337 }
338
339 for (const auto &local_dof_index : local_dof_indices)
340 {
341 dof_with_more_then_one_particle[local_dof_index] =
342 true;
343 }
344
345 overconstrained_fluid_cell_map[cell] = {
346 true, particle_current_id, particle_current_distance};
347 }
348 }
349 }
350 }
351 }

354template <int dim>
355void
356FluidDynamicsSharp<dim>::refinement_control(const bool initial_refinement)
357{
358 // We add a post-refinement check to update precalculations only if needed.
359 // Two booleans are used: the first one is used to update the precalculations,
360 // and the other one is used to remove superfluous nodes. We make this
361 // separation between the steps to avoid removing useful information until all
362 // triangulation refinement steps are done.
363 bool update_precalculations_flag = false;
364 bool remove_superfluous_data_flag = false;
365 this->triangulation->signals.post_refinement.connect(
366 [&update_precalculations_flag]() { update_precalculations_flag = true; });
367
368 // This function applies the various refinement steps depending on the
369 // parameters and the state.
370 if (initial_refinement)
371 {
372 // Apply the initial box refinement
373 this->box_refine_mesh(false);
374 update_precalculations_for_ib();
375 }
376 if (this->simulation_parameters.particlesParameters
377 ->time_extrapolation_of_refinement_zone ||
378 initial_refinement)
379 {
380 // Stores variable for refinement around the particle.
381 double temp_refine =
382 this->simulation_parameters.mesh_adaptation.variables.begin()
383 ->second.refinement_fraction;
384 double temp_coarse =
385 this->simulation_parameters.mesh_adaptation.variables.begin()
386 ->second.coarsening_fraction;
387 this->simulation_parameters.mesh_adaptation.variables.begin()
388 ->second.refinement_fraction = 0;
389 this->simulation_parameters.mesh_adaptation.variables.begin()
390 ->second.coarsening_fraction = 0;
391
392 for (unsigned int i = 0;
393 i <
394 this->simulation_parameters.particlesParameters->initial_refinement;
395 ++i)
396 {
397 update_precalculations_flag = false;
398 this->pcout << "Initial refinement around IB particles - Step : "
399 << i + 1 << " of "
400 << this->simulation_parameters.particlesParameters
401 ->initial_refinement
402 << std::endl;
403 refine_ib(initial_refinement);
404 NavierStokesBase<dim, GlobalVectorType, IndexSet>::refine_mesh();
405 if (update_precalculations_flag)
406 {
407 remove_superfluous_data_flag = true;
408 update_precalculations_for_ib();
409 }
410 }
411 this->simulation_parameters.mesh_adaptation.variables.begin()
412 ->second.refinement_fraction = temp_refine;
413 this->simulation_parameters.mesh_adaptation.variables.begin()
414 ->second.coarsening_fraction = temp_coarse;
415 }
416 if (initial_refinement == false)
417 {
418 update_precalculations_flag = false;
419 refine_ib(initial_refinement);
420 NavierStokesBase<dim, GlobalVectorType, IndexSet>::refine_mesh();
421 if (update_precalculations_flag)
422 {
423 remove_superfluous_data_flag = true;
424 update_precalculations_for_ib();
425 }
426 }
427 if (remove_superfluous_data_flag || initial_refinement)
428 {
429 for (unsigned int p_i = 0; p_i < particles.size(); ++p_i)
430 {
431 TimerOutput::Scope t(this->computing_timer, "Remove RBF nodes");
432 particles[p_i].remove_superfluous_data(
433 this->dof_handler, particles[p_i].mesh_based_precalculations);
434 }
435 }

438template <int dim>
439bool
440FluidDynamicsSharp<dim>::cell_cut_by_p_absolute_distance(
441 const typename DoFHandler<dim>::active_cell_iterator &cell,
442 std::map<types::global_dof_index, Point<dim>> &support_points,
443 unsigned int p)
444{
445 // This function aims at defining if a cell is cut when the level set used to
446 // define the particle is not signed. This function works by analyzing the
447 // location of the projection of the support point and the cell centroid on
448 // the particle's surface. If the projected point at the surface of the
449 // particle is inside the cell then the cell is cut.
450
451 // First step: define useful variables.
452 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
453 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
454 cell->get_dof_indices(local_dof_indices);
455 bool cell_is_cut = false;
456 Point<dim> centroid_of_cell = cell->barycenter();
457 Point<dim> projected_point;
458 particles[p].closest_surface_point(centroid_of_cell, projected_point, cell);
459
460 // Check the centroid of the cell first.
461 if (cell->point_inside(projected_point))
462 {
463 cell_is_cut = true;
464 return cell_is_cut;
465 }
466 else
467 {
468 // If the projection of the centroid is not inside the cell we also check
469 // for each dof support point of the cell.
470 for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
471 {
472 // Only check for the support point of the velocity in x.
473 if (0 == this->fe->system_to_component_index(j).first)
474 {
475 particles[p].closest_surface_point(
476 support_points[local_dof_indices[j]], projected_point, cell);
477
478 if (cell->point_inside(projected_point))
479 {
480 cell_is_cut = true;
481 break;
482 }
483 if ((projected_point - support_points[local_dof_indices[j]])
484 .norm() < cell->diameter() * 0.1)
485 {
486 cell_is_cut = true;
487 break;
488 }
489 }
490 }
491 }
492 return cell_is_cut;
493}

495template <int dim>
496void
497FluidDynamicsSharp<dim>::optimized_generate_cut_cells_map()
498{
499 TimerOutput::Scope t(this->computing_timer, "Optmize cut cells mapping");
500 MappingQ1<dim> mapping;
501 unsigned int max_children = GeometryInfo<dim>::max_children_per_cell;
502
503 std::set<typename DoFHandler<dim>::cell_iterator> all_candidate_cells;
504 std::set<typename DoFHandler<dim>::cell_iterator>
505 previous_all_candidate_cells;
506
507 cut_cells_map.clear();
508 cells_inside_map.clear();
509 for (const auto &cell : this->dof_handler.active_cell_iterators())
510 {
511 if (cell->is_locally_owned() || cell->is_ghost())
512 {
513 cut_cells_map[cell] = {false, 0, std::vector<unsigned int>()};
514 cells_inside_map[cell] = {false, 0};
515 }
516 }
517
518 // Loop over particles in reverse.
519 // This is done in reverse to guarantee that the lowest particle ID is
520 // associated to the cut cell or the inside particle cell. For instance, if
521 // you have two particles cutting the same cell, it guarantees that the lowest
522 // ID particle is associated with the cell in the map.
523 for (int p = particles.size() - 1; p >= 0; --p)
524 {
525 bool empty = true;
526 unsigned int lvl_iter = 0;
527
528 // Fix max level search
529 unsigned int max_lvl_search =
530 this->dof_handler.get_triangulation().n_levels() - 2;
531 if (max_lvl_search < 4)
532 max_lvl_search = this->dof_handler.get_triangulation().n_levels();
533
534 // Search for candidates until at least one is found or until the
535 // max_lvl_search is reached
536 while (empty || lvl_iter < max_lvl_search)
537 {
538 const auto &cell_iterator =
539 this->dof_handler.cell_iterators_on_level(lvl_iter);
540
541 // Loop over the cells on level lvl_iter of the mesh
542 for (const auto &cell : cell_iterator)
543 {
544 auto is_candidate = generate_cut_cell_candidates(cell, p);
545
546 // check whether the cell is candidate for inside of cut
547 if (is_candidate.first)
548 {
549 all_candidate_cells.insert(cell);
550 }
551 if (is_candidate.second)
552 {
553 all_candidate_cells.insert(cell);
554 }
555 }
556 if (all_candidate_cells.size() > 0)
557 empty = false;
558
559 lvl_iter += 1;
560 }
561
562 // Store list with all candidate cells
563 previous_all_candidate_cells = all_candidate_cells;
564
565 // If there are candidate cells
566 if (all_candidate_cells.size() != 0)
567 {
568 bool all_cells_are_active = false;
569
570 // Loop until candidate cells are active
571 while (!all_cells_are_active)
572 {
573 all_cells_are_active = true;
574 all_candidate_cells.clear();
575
576 // loop over the last set of candidates to check if it is still a
577 // candidate
578 for (auto cell : previous_all_candidate_cells)
579 {
580 auto is_candidate = generate_cut_cell_candidates(cell, p);
581
582 // Check if cell is inside particles
583 if (is_candidate.first)
584 {
585 // If it is an active cell, add it to cells_inside_map
586 if (cell->is_active())
587 {
588 cells_inside_map[cell] = {true, p};
589 }
590 else
591 {
592 // If we are here, the cell has children.
593 all_cells_are_active = false;
594 for (unsigned int j = 0; j < max_children; ++j)
595 {
596 // Store the children of the cell for
597 // further checks.
598 all_candidate_cells.insert(cell->child(j));
599 }
600 }
601 }
602
603 if (is_candidate.second)
604 {
605 // If it is an active cell, add it to cells_cut_map
606 if (cell->is_active())
607 {
608 cut_cells_map[cell] = {true,
609 p,
610 std::vector<unsigned int>()};
611 }
612 else
613 {
614 // If we are here, the cell has children.
615 all_cells_are_active = false;
616 for (unsigned int j = 0; j < max_children; ++j)
617 {
618 // Store the children of the cell for
619 // further checks.
620 all_candidate_cells.insert(cell->child(j));
621 }
622 }
623 }
624 }
625 previous_all_candidate_cells.clear();
626 previous_all_candidate_cells = all_candidate_cells;
627 }
628 }
629 }

633template <int dim>
634std::pair<bool, bool>
635FluidDynamicsSharp<dim>::generate_cut_cell_candidates(
636 const typename DoFHandler<dim>::cell_iterator &cell,
637 const unsigned int p_id)
638{
639 bool cell_is_inside = false;
640 bool cell_is_cut = false;
641
642 double search_radius = particles[p_id].radius * (1.0 - 1e-07);
643 auto position = particles[p_id].position;
644
645 bool point_inside_cell = cell->point_inside(position);
646
647 // Check how many vertices are inside the particle
648 unsigned int nb_vertices_inside = 0;
649 for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_cell; ++i)
650 {
651 if ((cell->vertex(i) - position).norm() < search_radius)
652 ++nb_vertices_inside;
653 }
654
655 // If vertices are found inside the particle
656 if (nb_vertices_inside > 0)
657 {
658 // If the number of vertices inside the cell is equal to the number of
659 // vertices per cell, the cell is inside the particle
660 if (nb_vertices_inside == GeometryInfo<dim>::vertices_per_cell)
661 {
662 cell_is_inside = true;
663 cell_is_cut = false;
664 }
665 // Otherwise, the cell is cut
666 else
667 {
668 cell_is_inside = false;
669 cell_is_cut = true;
670 }
671 return {cell_is_inside, cell_is_cut};
672 }
673
674 // If the particle intersect the cell and is not known whether all
675 // vertices are inside the particles, set all true by default
676 if (point_inside_cell)
677 {
678 cell_is_inside = true;
679 cell_is_cut = true;
680 return {cell_is_inside, cell_is_cut};
681 }
682
683 // The last check consists of projecting the particle's position on the cells'
684 // face and checking whether the projected points are within the particle. If
685 // one of the projected points is inside the particle, either the cell is
686 // inside or is cut by the particles. If this is true, the next
687 // level will be tested again by this function.
688
689 // Initialize superpoint of manifold
690 std::vector<Point<dim>> manifold_points(
691 GeometryInfo<dim - 1>::vertices_per_cell);
692
693 // Loop through faces
694 for (const auto face : cell->face_indices())
695 {
696 auto face_iter = cell->face(face);
697 auto &local_face_manifold = face_iter->get_manifold();
698
699 // Loop through face vertices
700 for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_face; ++i)
701 {
702 // Assign vertex to manifold superpoint
703 manifold_points[i] = face_iter->vertex(i);
704 }
705
706 // Create array of points on face
707 auto surrounding_face_points =
708 make_array_view(manifold_points.begin(), manifold_points.end());
709
710 // Loop through face vertices
711 for (unsigned int i = 0; i < GeometryInfo<dim>::vertices_per_face; ++i)
712 {
713 // Project points to face
714 Point<dim> projected_point =
715 local_face_manifold.project_to_manifold(surrounding_face_points,
716 position);
717
718 bool projected_point_over_face = true;
719 Tensor<1, dim> projected_point_tensor(projected_point);
720
721 // Check whether the projected point is within the cell
722 try
723 {
724 projected_point_over_face = cell->point_inside(projected_point);
725
726 if ((position - projected_point).norm() < search_radius &&
727 projected_point_over_face)
728 {
729 cell_is_inside = true;
730 cell_is_cut = true;
731 return {cell_is_inside, cell_is_cut};
732 }
733 }
734 // If the method crashes, change default for false
735 catch (...)
736 {
737 projected_point_over_face = false;
738 }
739
740 for (unsigned int d = 0; d != dim; d++)
741 {
742 if (projected_point_tensor[d] < 0 ||
743 projected_point_tensor[d] > 1)
744 {
745 projected_point_over_face = false;
746 }
747 }
748 }
749 }
750 return {cell_is_inside, cell_is_cut};
751}

811template <int dim>
812void
813FluidDynamicsSharp<dim>::refine_ib(const bool initial_refinement)
814{
815 bool refinement_step;
816 if (this->simulation_parameters.mesh_adaptation.refinement_at_frequency)
817 refinement_step = this->simulation_control->get_step_number() %
818 this->simulation_parameters.mesh_adaptation.frequency ==
819 0;
820 else
821 refinement_step = this->simulation_control->get_step_number() == 0;
822
823 // If this is not the initial refinement cycle nor a usual refinement step,
824 // we refine the cells around immersed solids only if they have moved since
825 // the last time step.
826 if (!initial_refinement && !refinement_step)
827 {
828 bool stationary_particles = true;
829 for (unsigned int p = 0; p < particles.size(); ++p)
830 {
831 if ((particles[p].position - particles[p].previous_positions[0])
832 .norm() > 1e-16 ||
833 (particles[p].orientation - particles[p].previous_orientation[0])
834 .norm() > 1e-16)
835 {
836 stationary_particles = false;
837 break;
838 }
839 }
840 if (stationary_particles)
841 return;
842 }
843
844 TimerOutput::Scope t(this->computing_timer,
845 "Refine around immersed boundary");
846 Point<dim> center_immersed;
847 std::map<types::global_dof_index, Point<dim>> support_points =
848 DoFTools::map_dofs_to_support_points(*this->mapping, this->dof_handler);
849
850 double dt = this->simulation_control->get_time_steps_vector()[0];
851
852 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
853 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
854
855 bool extrapolate_particle_position =
856 this->simulation_parameters.particlesParameters
857 ->time_extrapolation_of_refinement_zone;
858
859 double smallest_cut_cell = std::numeric_limits<double>::max();
860 bool minimal_crown_refinement_enabled =
861 abs(this->simulation_parameters.particlesParameters->outside_radius - 1) <
862 1e-16 &&
863 abs(this->simulation_parameters.particlesParameters->inside_radius - 1) <
864 1e-16;
865 if (minimal_crown_refinement_enabled)
866 {
867 const auto &cell_iterator_smallest_cell =
868 this->dof_handler.active_cell_iterators();
869 for (const auto &cell : cell_iterator_smallest_cell)
870 {
871 if (cell->is_locally_owned())
872 smallest_cut_cell = std::min(smallest_cut_cell, cell->diameter());
873 }
874 smallest_cut_cell =
875 Utilities::MPI::min(smallest_cut_cell, this->mpi_communicator);
876 }
877
878 const auto &cell_iterator = this->dof_handler.active_cell_iterators();
879 for (const auto &cell : cell_iterator)
880 {
881 if (cell->is_locally_owned())
882 {
883 cell->get_dof_indices(local_dof_indices);
884 for (unsigned int p = 0; p < particles.size(); ++p)
885 {
886 unsigned int count_small = 0;
887 center_immersed = particles[p].position;
888 Tensor<1, dim> r;
889 r[0] = particles[p].radius;
890
891 Point<dim> particle_position;
892 Tensor<1, 3> particle_orientation;
893 if (extrapolate_particle_position)
894 {
895 particle_position = particles[p].position;
896 particle_orientation = particles[p].orientation;
897 particles[p].orientation =
898 particles[p].orientation + particles[p].omega * dt;
899 particles[p].position[0] =
900 particles[p].position[0] + particles[p].velocity[0] * dt;
901 particles[p].position[1] =
902 particles[p].position[1] + particles[p].velocity[1] * dt;
903
904 if constexpr (dim == 3)
905 {
906 particles[p].position[2] = particles[p].position[2] +
907 particles[p].velocity[2] * dt;
908 }
909 particles[p].set_position(particles[p].position);
910 particles[p].set_orientation(particles[p].orientation);
911 }
912 // Check if a point on the random point on the IB is contained in
913 // that cell. If the particle is much smaller than the cell, all
914 // its vertices may be outside of the particle. In that case the
915 // cell won't be refined. To prevent that, we check if a random
916 // point on the boundary is contained in the cell.
917 bool cell_as_ib_inside =
918 cell->point_inside(particles[p].position + r);
919 for (unsigned int j = 0; j < local_dof_indices.size(); ++j)
920 {
921 // Only check the dof of velocity in x.
922 if (this->fe->system_to_component_index(j).first == 0)
923 {
924 // Count the number of DOFs that fall in the refinement
925 // zone around the particle. To fall in the zone, the
926 // radius of the DOFs to the center of the particle must
927 // be bigger than the inside radius of the refinement zone
928 // and smaller than the outside radius of the refinement
929 // zone.
930 bool is_inside_crown;
931 if (minimal_crown_refinement_enabled)
932 {
933 // The factor should be equal to the length of the
934 // current smallest cell. Since cell->diameter()
935 // returns the longest diagonal, we divide the
936 // diameter by the diagonal of a unit hypercube to
937 // obtain the correct length.
938 double factor = 1 / sqrt(dim);
939 is_inside_crown = particles[p].is_inside_crown(
940 support_points[local_dof_indices[j]],
941 factor * smallest_cut_cell,
942 -factor * smallest_cut_cell,
943 true, // indicates that we use the absolute
944 // distance definition
945 cell);
946 }
947 else
948 {
949 is_inside_crown = particles[p].is_inside_crown(
950 support_points[local_dof_indices[j]],
951 this->simulation_parameters.particlesParameters
952 ->outside_radius,
953 this->simulation_parameters.particlesParameters
954 ->inside_radius,
955 false, // indicates that we use the
956 // radius relative distance definition
957 cell);
958 }
959 if (is_inside_crown)
960 ++count_small;
961 }
962 }
963
964 if (extrapolate_particle_position)
965 {
966 particles[p].position = particle_position;
967 particles[p].orientation = particle_orientation;
968 particles[p].set_position(particles[p].position);
969 particles[p].set_orientation(particles[p].orientation);
970 }
971 if (count_small > 0 || cell_as_ib_inside)
972 {
973 cell->set_refine_flag();
974 break;
975 }
976 }
977 }
978 }
979}

1118 {
1119 // Loop over all cut cell faces'.
1120 for (const auto face : cell->face_indices())
1121 {
1122 auto local_face = cell->face(face);
1123 cell->face(face)->get_dof_indices(local_face_dof_indices);
1124
1125 // Check if the face is cut
1126 unsigned int nb_dof_inside = 0;
1127 for (const auto &dof_index : local_face_dof_indices)
1128 {
1129 // Count the number of DOFs that are inside
1130 // of the particles. If all the DOfs are on one side
1131 // the cell is not cut by the boundary.
1132 if (particles[p].get_levelset(
1133 support_points[dof_index], cell) <= 0)
1134 ++nb_dof_inside;
1135 }

1137 // If the face is not cut and the face is outside the
1138 // IB, the face of the cell is on the boundary of the
1139 // computational domain.
1140 if (nb_dof_inside == 0)
1141 {
1142 // Create a vector to approximate the normal of the
1143 // cell to orient the projection of the face.
1144 Tensor<1, dim> approximate_surface_cell_normal;
1145 // Project the face on the surface of the IB. This
1146 // creates a surface cell where we can evaluate the
1147 // solution. Define the triangulation of the surface
1148 // cell.

1150 {
1151 Point<dim> vertex_projection;
1152 particles[p].closest_surface_point(
1153 local_face->vertex(i), vertex_projection, cell);

1155 (local_face->vertex(i) - vertex_projection) /
1156 ((local_face->vertex(i) - vertex_projection)
1157 .norm() +
1158 DBL_MIN);

1161 // Create the list of vertices
1162 for (unsigned int j = 0; j < dim; ++j)
1163 {
1164 vertices_of_face_projection[i][j] =
1165 vertex_projection[j];
1166 }
1167 // Create the connectivity of the vertices of the
1168 // cell
1169 local_face_cell_data[0].vertices[i] = i;

1172 approximate_surface_cell_normal / vertices_per_face;

1176 local_face_projection_triangulation =
1177 Triangulation<dim - 1, dim>();

1179 // triangulation of the projection of the face on the
1180 // IB. This creates the surface cell on the IB
1181 local_face_projection_triangulation
1182 .create_triangulation(vertices_of_face_projection,
1183 local_face_cell_data,
1184 SubCellData());

1188 0, *particles[p].shape->get_shape_manifold());

1190 local_face_dof_handler.reinit(
1191 local_face_projection_triangulation);

1194 // Define the solution on IB surface cell by using
1195 // the IB stencil to extrapolate the fluid stress
1196 // tensor.
1197 std::vector<Tensor<2, dim>>
1198 local_face_viscous_stress_tensor(dofs_per_face);
1199 std::vector<Tensor<2, dim>>
1200 local_face_pressure_tensor(dofs_per_face);

1206 this->fe->face_system_to_component_index(i)
1207 .first;

1209 // extrapolate the fluid stress tensor on the IB
1210 // surface.

1214 [local_face_dof_indices[i]] == true)

1219 // location;
1220

1222
1223 auto [point, interpolation_points] =
1224 stencil
1225 .support_points_for_interpolation(
1226 order,
1227 length_ratio,
1228 particles[p],
1229 support_points
1230 [local_face_dof_indices[i]],
1231 cell);

1234 ib_done[local_face_dof_indices[i]]
1235 .second;

1237 // to define the IB constraint of that
1238 // dof. We always have that information
1239 // except if the dof is not owned.
1240 if (ib_done[local_face_dof_indices[i]]
1241 .first == false or
1242 dof_with_more_then_one_particle
1243 [local_face_dof_indices[i]] ==
1244 true)
1245 {
1246 // Get the cell use for the
1247 // extrapolation

1249 stencil.point_for_cell_detection(
1250 particles[p],
1251 support_points
1252 [local_face_dof_indices[i]],
1253 cell);
1254
1255 try
1256 {
1257 cell_2 = LetheGridTools::
1258 find_cell_around_point_with_neighbors<
1259 dim>(this->dof_handler,
1260 vertices_to_cell,
1261 cell,
1262 point_to_find_cell);
1263 }
1264 catch (...)
1265 {
1266 cell_2 = cell;
1267 }
1268 }

1270 cell_2->get_dof_indices(
1271 local_dof_indices_2);
1272
1273 unite_cell_interpolation_points[0] =
1274 this->mapping
1275 ->transform_real_to_unit_cell(cell_2,
1276 point);
1277 for (unsigned int j = 1;
1278 j < ib_coef.size();
1279 ++j)
1280 {
1281 unite_cell_interpolation_points[j] =
1282 this->mapping
1283 ->transform_real_to_unit_cell(
1284 cell_2,
1285 interpolation_points[j - 1]);
1286 }
1287
1288 fluid_viscous_stress_at_ib = 0;
1289 fluid_pressure_stress_at_ib = 0;
1290

1292 // the IB stencil
1293 Quadrature<dim> q_local(
1294 unite_cell_interpolation_points,
1295 ib_coef);

1297 *this->fe,
1298 q_local,
1299 update_quadrature_points |
1300 update_gradients | update_values);
1301
1302 // Evaluate the relevant information at
1303 // the quadrature points to do the
1304 // extrapolation.
1305 fe_values_cell2.reinit(cell_2);
1306 fe_values_cell2[velocities]
1307 .get_function_gradients(
1308 this->evaluation_point,
1309 velocity_gradients);
1310 fe_values_cell2[pressure]
1311 .get_function_values(
1312 this->evaluation_point,
1313 pressure_values);
1314
1315 // Extrapolate the fluid stress tensor on
1316 // the surface of the IB.
1317
1318 for (unsigned int k = 0;
1319 k < ib_coef.size();
1320 ++k)
1321 {
1322 fluid_pressure = 0;
1323 for (int d = 0; d < dim; ++d)
1324 {
1325 fluid_pressure[d][d] =
1326 pressure_values[k];
1327 }
1328 shear_rate =
1329 velocity_gradients[k] +
1330 transpose(velocity_gradients[k]);
1331
1332
1333 const double shear_rate_magnitude =
1334 calculate_shear_rate_magnitude(
1335 shear_rate);
1336
1337 std::map<field, double> field_values;
1338 field_values[field::shear_rate] =
1339 shear_rate_magnitude;
1340
1341 kinematic_viscosity =
1342 rheological_model->value(
1343 field_values);
1344
1345 fluid_viscous_stress =
1346 -kinematic_viscosity * shear_rate;
1347
1348 fluid_viscous_stress_at_ib -=
1349 fluid_viscous_stress * ib_coef[k];
1350
1351 fluid_pressure_stress_at_ib -=
1352 fluid_pressure * ib_coef[k];
1353 }
1354 // Store the stress tensor that results
1355 // from the extrapolation in the local
1356 // evaluation vector of the IB surface
1357 // cell and in a map that is used if the
1358 // same extrapolation is needed in another
1359 // face.
1360 local_face_viscous_stress_tensor[i] =
1361 fluid_viscous_stress_at_ib;
1362 local_face_pressure_tensor[i] =
1363 fluid_pressure_stress_at_ib;
1364
1365
1366 force_eval[local_face_dof_indices[i]] =
1367 std::make_pair(
1368 fluid_viscous_stress_at_ib,
1369 fluid_pressure_stress_at_ib);
1370 }
1371 }
1372 else
1373 {
1374 // Use the results from a previously evaluated
1375 // extrapolation. This step comes with an
1376 // error due to the curvature of the surface
1377 // in Q2 and higher order elements.
1378 local_face_viscous_stress_tensor[i] =
1379 force_eval[local_face_dof_indices[i]].first;
1380 local_face_pressure_tensor[i] =
1381 force_eval[local_face_dof_indices[i]]
1382 .second;
1383 }
1384 }
1385 // Use the extrapolation of fluid stress tensor at the
1386 // dof location of the IB surface cell to integrate
1387 // the stress tensor on the surface of the IB
1388 auto local_face_viscous_stress_tensor_old =
1389 local_face_viscous_stress_tensor;
1390 auto local_face_pressure_tensor_old =
1391 local_face_pressure_tensor;
1392
1393 local_face_viscous_stress_tensor.clear();
1394 local_face_pressure_tensor.clear();
1395
1396 local_face_viscous_stress_tensor.resize(
1397 local_face_dof_indices.size());
1398 local_face_pressure_tensor.resize(
1399 local_face_dof_indices.size());
1400
1401 for (const auto &projection_cell_face :
1402 local_face_dof_handler.active_cell_iterators())
1403 {
1404 fe_face_projection_values.reinit(
1405 projection_cell_face);
1406 std::vector<Point<dim>> q_points =
1407 fe_face_projection_values
1408 .get_quadrature_points();
1409 try
1410 {
1411 if (this->simulation_parameters.fem_parameters
1412 .velocity_order > 1)
1413 {
1414 FullMatrix<double> interpolation_matrix(
1415 local_face_dof_indices.size(),
1416 local_face_dof_indices.size());
1417 FullMatrix<double>
1418 inv_interpolation_matrix(
1419 local_face_dof_indices.size(),
1420 local_face_dof_indices.size());
1421
1422 // Define the interpolation matrix of the
1423 // surface cell
1424 for (unsigned int i = 0;
1425 i < local_face_dof_indices.size();
1426 ++i)
1427 {
1428 Point<dim> point_projection;
1429 particles[p].closest_surface_point(
1430 support_points
1431 [local_face_dof_indices[i]],
1432 point_projection,
1433 cell);
1434
1435 auto projected_point_unit =
1436 local_face_map
1437 .transform_real_to_unit_cell(
1438 projection_cell_face,
1439 point_projection);
1440
1441 for (unsigned int j = 0;
1442 j <
1443 local_face_dof_indices.size();
1444 ++j)
1445 {
1446 interpolation_matrix[i][j] = 0;
1447 if (
1448 this->fe
1449 ->face_system_to_component_index(
1450 j)
1451 .first ==
1452 this->fe
1453 ->face_system_to_component_index(
1454 i)
1455 .first)
1456 interpolation_matrix[i][j] +=
1457 local_face_fe.shape_value(
1458 j, projected_point_unit);
1459 }
1460 }
1461 inv_interpolation_matrix.invert(
1462 interpolation_matrix);
1463 // Define the value of the fluid stress
1464 // tensor on the surface cell at the DOF
1465 // support points location.
1466 for (unsigned int i = 0;
1467 i < local_face_dof_indices.size();
1468 ++i)
1469 {
1470 for (unsigned int j = 0;
1471 j <
1472 local_face_dof_indices.size();
1473 ++j)
1474 {
1475 local_face_viscous_stress_tensor
1476 [i] +=
1477 inv_interpolation_matrix[i][j] *
1478 local_face_viscous_stress_tensor_old
1479 [j];
1480 local_face_pressure_tensor[i] +=
1481 inv_interpolation_matrix[i][j] *
1482 local_face_pressure_tensor_old
1483 [j];
1484 }
1485 }
1486 }
1487 else
1488 {
1489 local_face_viscous_stress_tensor =
1490 local_face_viscous_stress_tensor_old;
1491 local_face_pressure_tensor =
1492 local_face_pressure_tensor_old;
1493 }
1494 }
1495 catch (...)
1496 {
1497 local_face_viscous_stress_tensor =
1498 local_face_viscous_stress_tensor_old;
1499 local_face_pressure_tensor =
1500 local_face_pressure_tensor_old;
1501 }
1502 for (unsigned int q = 0; q < n_q_points_face; q++)
1503 {
1504 // Evaluate the total surface
1505 // Redefined the normal at the quadrature
1506 // point since we don't control the
1507 // orientation of the cell.
1508 normal_vector =
1509 fe_face_projection_values.normal_vector(q);
1510 if (scalar_product(
1511 normal_vector,
1512 approximate_surface_cell_normal) < 0)
1513 normal_vector = -normal_vector;
1514
1515 fluid_viscous_stress = 0;
1516 fluid_pressure = 0;
1517
1518 double local_weight = 0;
1519 // Integrate
1520 for (unsigned int i = 0;
1521 i < local_face_dof_indices.size();
1522 ++i)
1523 {
1524 const unsigned int component_i =
1525 local_face_fe
1526 .system_to_component_index(i)
1527 .first;
1528 if (component_i == 0)
1529 {
1530 fluid_viscous_stress +=
1531 fe_face_projection_values
1532 .shape_value(i, q) *
1533 local_face_viscous_stress_tensor[i];
1534
1535 fluid_pressure +=
1536 fe_face_projection_values
1537 .shape_value(i, q) *
1538 local_face_pressure_tensor[i];
1539
1540 total_area[p] +=
1541 fe_face_projection_values.JxW(q) *
1542 fe_face_projection_values
1543 .shape_value(i, q);
1544 local_weight +=
1545 fe_face_projection_values
1546 .shape_value(i, q);
1547 }
1548 }
1549
1550 auto viscous_force =
1551 fluid_viscous_stress * normal_vector *
1552 fe_face_projection_values.JxW(q);
1553
1554 auto pressure_force =
1555 fluid_pressure * normal_vector *
1556 fe_face_projection_values.JxW(q);
1557
1558 auto force = viscous_force + pressure_force;
1559
1560 if (force != force)
1561 {
1562 // The force is nan; this happens when the
1563 // face projection is a line. This
1564 // generally happens when the surface on
1565 // which the force is evaluated is
1566 // perfectly flat and aligned with the
1567 // mesh. Since the area associated with
1568 // this face projection is zero, we set
1569 // the local force contribution to zero.
1570 force = 0;
1571 viscous_force = 0;
1572 pressure_force = 0;
1573 }
1574 if (force.norm() > 0)
1575 {
1576 nb_evaluation[p] += local_weight;
1577 }
1578
1579 if (force.norm() > 0)
1580 {
1581 nb_evaluation[p] += local_weight;
1582 }
1583
1584
1585 // Add the local contribution of this surface
1586 // cell.
1587
1588 particles[p].fluid_viscous_forces +=
1589 tensor_nd_to_3d(viscous_force);
1590
1591 particles[p].fluid_pressure_forces +=
1592 tensor_nd_to_3d(pressure_force);
1593
1594 auto distance =
1595 q_points[q] - particles[p].position;
1596 if constexpr (dim == 2)
1597 {
1598 particles[p].fluid_torque[0] += 0.;
1599 particles[p].fluid_torque[1] += 0.;
1600 particles[p].fluid_torque[2] +=
1601 distance[0] * force[1] -
1602 distance[1] * force[0];
1603 }
1604 else if constexpr (dim == 3)
1605 {
1606 particles[p].fluid_torque[0] +=
1607 distance[1] * force[2] -
1608 distance[2] * force[1];
1609 particles[p].fluid_torque[1] +=
1610 distance[2] * force[0] -
1611 distance[0] * force[2];
1612 particles[p].fluid_torque[2] +=
1613 distance[0] * force[1] -
1614 distance[1] * force[0];
1615 }
1616 }
1617 }
1618 }
1619 }
1620 }
1621 }
1622 }
1623 }
1624
1625 // Sums the force evaluation on each of the processor.
1626 for (unsigned int i = 0; i < particles.size(); ++i)
1627 {
1628 particles[i].fluid_viscous_forces =
1629 Utilities::MPI::sum(particles[i].fluid_viscous_forces,
1630 this->mpi_communicator) *
1631 fluid_density;
1632 particles[i].fluid_pressure_forces =
1633 Utilities::MPI::sum(particles[i].fluid_pressure_forces,
1634 this->mpi_communicator) *
1635 fluid_density;
1636 particles[i].fluid_forces =
1637 particles[i].fluid_viscous_forces + particles[i].fluid_pressure_forces;
1638 particles[i].fluid_torque =
1639 Utilities::MPI::sum(particles[i].fluid_torque, this->mpi_communicator) *
1640 fluid_density;
1641 total_area[i] =
1642 Utilities::MPI::sum(total_area[i], this->mpi_communicator);
1643 nb_evaluation[i] =
1644 Utilities::MPI::sum(nb_evaluation[i], this->mpi_communicator);
1645 }

1819template <int dim>
1820std::pair<double, double>
1821FluidDynamicsSharp<dim>::calculate_L2_error_particles()
1822{
1823 TimerOutput::Scope t(this->computing_timer, "error");
1824 QGauss<dim> quadrature_formula(this->number_quadrature_points + 1);
1825 FEValues<dim> fe_values(*this->mapping,
1826 *this->fe,
1827 quadrature_formula,
1828 update_values | update_gradients |
1829 update_quadrature_points | update_JxW_values);
1830 FEFaceValues<dim> fe_face_values(*this->mapping,
1831 *this->fe,
1832 *this->face_quadrature,
1833 update_values | update_gradients |
1834 update_quadrature_points |
1835 update_JxW_values);
1836
1837 const FEValuesExtractors::Vector velocities(0);
1838 const FEValuesExtractors::Scalar pressure(dim);
1839
1840 const unsigned int dofs_per_cell =
1841 this->fe->dofs_per_cell; // This gives you DOFs per cell
1842 std::vector<types::global_dof_index> local_dof_indices(
1843 dofs_per_cell); // Local connectivity
1844
1845 const unsigned int n_q_points = quadrature_formula.size();
1846 const unsigned int n_q_points_face = this->face_quadrature->size();
1847
1848 std::vector<Vector<double>> q_exactSol(n_q_points, Vector<double>(dim + 1));
1849
1850 std::vector<Tensor<1, dim>> local_velocity_values(n_q_points);
1851 std::vector<Tensor<1, dim>> local_face_velocity_values(n_q_points_face);
1852 std::vector<double> local_pressure_values(n_q_points);
1853 std::vector<double> div_phi_u(dofs_per_cell);
1854 std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
1855
1856 Function<dim> *l_exact_solution = this->exact_solution;
1857
1858 std::map<types::global_dof_index, Point<dim>> support_points =
1859 DoFTools::map_dofs_to_support_points(*this->mapping, this->dof_handler);
1860
1861 double l2errorU = 0.;
1862 double l2errorU_boundary = 0.;
1863 double l2errorP = 0.;
1864 double total_velocity_divergence = 0.;
1865 double pressure_integral = 0;
1866 double exact_pressure_integral = 0;
1867 double volume = 0;
1868
1869 // loop over elements
1870 typename DoFHandler<dim>::active_cell_iterator cell = this->dof_handler
1871 .begin_active(),
1872 endc = this->dof_handler.end();
1873
1874 // loop over elements to calculate average pressure
1875 for (; cell != endc; ++cell)
1876 {
1877 if (cell->is_locally_owned())
1878 {
1879 bool cell_is_cut;
1880 // std::ignore is used because we don't care about what particle cut
1881 // the cell or the number of particles that cut the cell.
1882 std::tie(cell_is_cut, std::ignore, std::ignore) = cut_cells_map[cell];
1883 bool cell_is_inside;
1884 std::tie(cell_is_inside, std::ignore) = cells_inside_map[cell];
1885 bool cell_is_overconstrained;
1886 std::tie(cell_is_overconstrained, std::ignore, std::ignore) =
1887 overconstrained_fluid_cell_map[cell];
1888
1889 if ((!cell_is_cut && !cell_is_inside) && !cell_is_overconstrained)
1890 {
1891 auto &evaluation_point = this->evaluation_point;
1892 fe_values.reinit(cell);
1893
1894 fe_values[pressure].get_function_values(evaluation_point,
1895 local_pressure_values);
1896 // Get the exact solution at all gauss points
1897 l_exact_solution->vector_value_list(
1898 fe_values.get_quadrature_points(), q_exactSol);
1899
1900
1901 // Retrieve the effective "connectivity matrix" for this element
1902 cell->get_dof_indices(local_dof_indices);
1903
1904 for (unsigned int q = 0; q < n_q_points; q++)
1905 {
1906 pressure_integral +=
1907 local_pressure_values[q] * fe_values.JxW(q);
1908 exact_pressure_integral +=
1909 q_exactSol[q][dim] * fe_values.JxW(q);
1910 volume += fe_values.JxW(q);
1911 }
1912 }
1913 }
1914 }
1915
1916 pressure_integral =
1917 Utilities::MPI::sum(pressure_integral, this->mpi_communicator);
1918 exact_pressure_integral =
1919 Utilities::MPI::sum(exact_pressure_integral, this->mpi_communicator);
1920 volume = Utilities::MPI::sum(volume, this->mpi_communicator);
1921
1922
1923 double average_pressure = pressure_integral / volume;
1924 double average_exact_pressure = exact_pressure_integral / volume;
1925 cell = this->dof_handler.begin_active(), endc = this->dof_handler.end();
1926
1927 for (; cell != endc; ++cell)
1928 {
1929 if (cell->is_locally_owned())
1930 {
1931 cell->get_dof_indices(local_dof_indices);
1932
1933 bool cell_is_cut;
1934 // std::ignore is used because we don't care about what particle cut
1935 // the cell or the number of particles that cut the cell.
1936 std::tie(cell_is_cut, std::ignore, std::ignore) = cut_cells_map[cell];
1937
1938 bool cell_is_overconstrained;
1939 std::tie(cell_is_overconstrained, std::ignore, std::ignore) =
1940 overconstrained_fluid_cell_map[cell];
1941
1942 bool cell_is_inside;
1943 std::tie(cell_is_inside, std::ignore) = cells_inside_map[cell];
1944 if (cell->at_boundary() &&
1945 this->check_existance_of_bc(
1946 BoundaryConditions::BoundaryType::function_weak))
1947 {
1948 for (const auto face : cell->face_indices())
1949 {
1950 if (cell->face(face)->at_boundary())
1951 {
1952 unsigned int boundary_id =
1953 cell->face(face)->boundary_id();
1954 if (this->simulation_parameters.boundary_conditions.type
1955 .at(boundary_id) ==
1956 BoundaryConditions::BoundaryType::function_weak)
1957 {
1958 NavierStokesFunctionDefined<dim> function_v(
1959 &this->simulation_parameters.boundary_conditions
1960 .navier_stokes_functions.at(boundary_id)
1961 ->u,
1962 &this->simulation_parameters.boundary_conditions
1963 .navier_stokes_functions.at(boundary_id)
1964 ->v,
1965 &this->simulation_parameters.boundary_conditions
1966 .navier_stokes_functions.at(boundary_id)
1967 ->w);
1968
1969 fe_face_values.reinit(cell, face);
1970 fe_face_values[velocities].get_function_values(
1971 this->present_solution, local_face_velocity_values);
1972 for (unsigned int q = 0; q < n_q_points_face; q++)
1973 {
1974 double u_x = local_face_velocity_values[q][0];
1975 double u_y = local_face_velocity_values[q][1];
1976
1977 double u_x_a = function_v.value(
1978 fe_face_values.quadrature_point(q), 0);
1979 double u_y_a = function_v.value(
1980 fe_face_values.quadrature_point(q), 1);
1981
1982 l2errorU_boundary +=
1983 ((u_x - u_x_a) * (u_x - u_x_a) +
1984 (u_y - u_y_a) * (u_y - u_y_a)) *
1985 fe_face_values.JxW(q);
1986 if constexpr (dim == 3)
1987 {
1988 double u_z = local_face_velocity_values[q][2];
1989 double u_z_a = function_v.value(
1990 fe_face_values.quadrature_point(q), 2);
1991 l2errorU_boundary += (u_z - u_z_a) *
1992 (u_z - u_z_a) *
1993 fe_face_values.JxW(q);
1994 }
1995 }
1996 }
1997 }
1998 }
1999 }
2000
2001 if (!cell_is_cut && !cell_is_overconstrained)
2002 {
2003 auto &evaluation_point = this->evaluation_point;
2004 auto &present_solution = this->present_solution;
2005 fe_values.reinit(cell);
2006 fe_values[velocities].get_function_values(present_solution,
2007 local_velocity_values);
2008 fe_values[pressure].get_function_values(present_solution,
2009 local_pressure_values);
2010 fe_values[velocities].get_function_gradients(
2011 evaluation_point, present_velocity_gradients);
2012
2013
2014 // Retrieve the effective "connectivity matrix" for this element
2015 cell->get_dof_indices(local_dof_indices);
2016
2017 // Get the exact solution at all gauss points
2018 l_exact_solution->vector_value_list(
2019 fe_values.get_quadrature_points(), q_exactSol);
2020 for (unsigned int q = 0; q < n_q_points; q++)
2021 {
2022 double present_velocity_divergence =
2023 trace(present_velocity_gradients[q]);
2024 double mass_source =
2025 this->simulation_parameters.source_term
2026 .navier_stokes_source->value(
2027 fe_values.get_quadrature_points()[q], dim);
2028
2029
2030 // Find the values of x and u_h (the finite element solution)
2031 // at the quadrature points
2032 double ux_sim = local_velocity_values[q][0];
2033 double ux_exact = q_exactSol[q][0];
2034
2035 double uy_sim = local_velocity_values[q][1];
2036 double uy_exact = q_exactSol[q][1];
2037 l2errorU += (ux_sim - ux_exact) * (ux_sim - ux_exact) *
2038 fe_values.JxW(q);
2039 l2errorU += (uy_sim - uy_exact) * (uy_sim - uy_exact) *
2040 fe_values.JxW(q);
2041
2042 if constexpr (dim == 3)
2043 {

2068 l2errorP = Utilities::MPI::sum(l2errorP, this->mpi_communicator);
2069 total_velocity_divergence =
2070 Utilities::MPI::sum(total_velocity_divergence, this->mpi_communicator);
