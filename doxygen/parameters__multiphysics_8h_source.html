51 /**
52 * @brief Different transformation function types for the signed distance:
53 * - tanh: hyperbolic tangent
54 * - piecewise_polynomial: 4th degree piecewise polynomial. The latter takes
55 * the following form:
56 *
57 * if \f$d < 0 \f$
58 * \f$\phi = 0.5 - 0.5(4d' + 6d'^2 + 4*d'^3 + d'^4)\f$
59 *
60 * else
61 * \f$\phi = 0.5 - 0.5(4d' - 6d'^2 + 4*d'^3 - d'^4)\f$
62 *
63 * with \f$d' = d/d_\mathrm{max}\f$, where \f$\phi\f$ is the phase
64 * fraction, \f$d\f$ the signed distance and \f$d_\mathrm{max}\f$ the
65 * maximum redistanciation distance. This transformation clamps the phase
66 * fraction to 0 or 1 when \f$d = \pm d_\mathrm{max}\f$.
67 */
68 enum class RedistanciationTransformationType : std::int8_t
69 {
70 tanh,
71 piecewise_polynomial
72 };

121 /**
122 * @brief Type of electromagnetic scaling to apply to the solution of the time-harmonic Maxwell solver after solving the linear system. This is relevant when the user wants to recover the physical solution in dimensional units from the dimensionless solution.
123 * - none: no scaling will be applied to the solution after solving the linear
124 * system, so the solution will be in dimensionless units.
125 * - electric_field: the solution will be scaled by the electric scaling
126 * factor given from the user input parameter. This factor needs to match the
127 * dimensionality of the problem.
128 * - magnetic_field: the solution will be scaled by the magnetic scaling
129 * factor given from the user input parameter. This factor needs to match the
130 * dimensionality of the problem.
131 * - power: the solution will be scaled by the power given from the user input
132 * parameter in waveguide inlets. This option is only relevant when there is
133 * at least one waveguide inlet in the problem. Note that it is the user's
134 * responsibility to ensure that if the problem also applies non-zero electric
135 * or magnetic field Dirichlet boundary conditions, those need to be scaled
136 * accordingly.
137 */
138 enum class ElectromagneticScalingType : std::int8_t
139 {
140 none,
141 electric_field,
142 magnetic_field,
143 power
144 };
