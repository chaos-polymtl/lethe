35 /**
36 * @brief Different projection-based interface sharpening types:
37 * - constant: the sharpening threshold is the same throughout the
38 * simulation,
39 * - adaptive: the sharpening threshold is determined by binary search, to
40 * ensure mass conservation of the monitored phase
41 */
42 enum class SharpeningType : std::int8_t
43 {
44 constant,
45 adaptive
46 };
47
48 /**
49 * @brief Different transformation function types for the signed distance:
50 * - tanh: hyperbolic tangent
51 * - piecewise_polynomial: 4th degree piecewise polynomial. The latter takes
52 * the following form:
53 *
54 * if \f$d < 0 \f$
55 * \f$\phi = 0.5 - 0.5(4d' + 6d'^2 + 4*d'^3 + d'^4)\f$
56 *
57 * else
58 * \f$\phi = 0.5 - 0.5(4d' - 6d'^2 + 4*d'^3 - d'^4)\f$
59 *
60 * with \f$d' = d/d_\mathrm{max}\f$, where \f$\phi\f$ is the phase
61 * fraction, \f$d\f$ the signed distance and \f$d_\mathrm{max}\f$ the
62 * maximum redistanciation distance. This transformation clamps the phase
63 * fraction to 0 or 1 when \f$d = \pm d_\mathrm{max}\f$.
64 */
65 enum class RedistanciationTransformationType : std::int8_t
66 {
67 tanh,
68 piecewise_polynomial
69 };
70
71 /**
72 * @brief Different phase fraction filtering types:
73 * - none: no filter wil be applied on the calculated phase fraction
74 * - tanh: the tanh filter function will be applied to the phase fraction,
75 * a \f$\beta\f$ parameter influencing the interface definition must be
76 * defined
77 */
78 enum class FilterType : std::int8_t
79 {
80 none,
81 clip,
82 tanh
