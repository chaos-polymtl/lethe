12template <int dim>
13void
14GLSNavierStokesVOFAssemblerCore<dim>::assemble_matrix(
15 const NavierStokesScratchData<dim> &scratch_data,
16 StabilizedMethodsTensorCopyData<dim> &copy_data)
17{
18 const std::vector<double> &viscosity_for_stabilization_vector =
19 scratch_data.dynamic_viscosity_for_stabilization;
20
21 // Loop and quadrature information
22 const auto &JxW_vec = scratch_data.JxW;
23 const unsigned int n_q_points = scratch_data.n_q_points;
24 const unsigned int n_dofs = scratch_data.n_dofs;
25 const double h = scratch_data.cell_size;
26
27 // Copy data elements
28 auto &strong_residual_vec = copy_data.strong_residual;
29 auto &strong_jacobian_vec = copy_data.strong_jacobian;
30 auto &local_matrix = copy_data.local_matrix;
31
32 // Time steps and inverse time steps which is used for stabilization constant
33 std::vector<double> time_steps_vector =
34 this->simulation_control->get_time_steps_vector();
35 const double dt = time_steps_vector[0];
36 const double sdt = 1. / dt;
37
38 Assert(scratch_data.properties_manager.density_is_constant(),
39 RequiresConstantDensity(
40 "GLSNavierStokesVOFAssemblerCore<dim>::assemble_matrix"));
41
42 // Loop over the quadrature points
43 for (unsigned int q = 0; q < n_q_points; ++q)
44 {
45 // Gather into local variables the relevant fields
46 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
47 const Tensor<2, dim> velocity_gradient =
48 scratch_data.velocity_gradients[q];
49 const Tensor<1, dim> velocity_laplacian =
50 scratch_data.velocity_laplacians[q];
51 const Tensor<3, dim> &velocity_hessian =
52 scratch_data.velocity_hessians[q];
53
54 // From hessian, calculate grad (div (u)) term needed for VOF problems
55 Tensor<1, dim> grad_div_velocity;
56 for (int d = 0; d < dim; ++d)
57 {
58 for (int k = 0; k < dim; ++k)
59 {
60 // hessian[c][i][j] is the (i,j)th component of the matrix of
61 // second derivatives of the cth vector component of the vector
62 // field at quadrature point q of the current cell
63 grad_div_velocity[d] += velocity_hessian[k][d][k];
64 }
65 }
66
67 const Tensor<1, dim> pressure_gradient =
68 scratch_data.pressure_gradients[q];
69
70 // Forcing term
71 Tensor<1, dim> force = scratch_data.force[q];
72
73 // Calculation of the magnitude of the velocity for the
74 // stabilization parameter
75 const double u_mag = std::max(velocity.norm(), 1e-12);
76
77 // Store JxW in local variable for faster access;
78 const double JxW = JxW_vec[q];
79
80 // Calculation of the equivalent properties at the quadrature point
81 double density_eq = scratch_data.density[q];
82 const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
83
84 // Calculation of the GLS stabilization parameter. The
85 // stabilization parameter used is different if the simulation
86 // is steady or unsteady. In the unsteady case it includes the
87 // value of the time-step
88 const double tau =
89 this->simulation_control->get_assembly_method() ==
90 Parameters::SimulationControl::TimeSteppingMethod::steady ?
91 calculate_navier_stokes_gls_tau_steady(
92 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h) :
93 calculate_navier_stokes_gls_tau_transient(
94 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h, sdt);
95
96 // Calculate the strong residual for GLS stabilization.
97 // According to the mathematical formulation, the viscosity jump term
98 // (-dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient))
99 // should be included. However, it is ill-posed for phase change cases
100 // where the solid is represented by a highly viscous fluid. Hence, we
101 // don't consider this term in the current formulation.
102 auto strong_residual = density_eq * velocity_gradient * velocity +
103 pressure_gradient -
104 dynamic_viscosity_eq * velocity_laplacian -
105 dynamic_viscosity_eq * grad_div_velocity -
106 density_eq * force + strong_residual_vec[q];
107 /*- dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient);*/
108
109
110 std::vector<Tensor<1, dim>> grad_phi_u_j_x_velocity(n_dofs);
111 std::vector<Tensor<1, dim>> velocity_gradient_x_phi_u_j(n_dofs);
112
113 // Pressure scaling factor
114 const double pressure_scaling_factor =
115 scratch_data.pressure_scaling_factor;
116
117 // We loop over the column first to prevent recalculation
118 // of the strong jacobian in the inner loop
119 for (unsigned int j = 0; j < n_dofs; ++j)
120 {
121 const auto &phi_u_j = scratch_data.phi_u[q][j];
122 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
123 const auto &laplacian_phi_u_j = scratch_data.laplacian_phi_u[q][j];
124
125 const auto &grad_phi_p_j =
126 scratch_data.grad_phi_p[q][j] * pressure_scaling_factor;
127
128 strong_jacobian_vec[q][j] +=
129 (density_eq * velocity_gradient * phi_u_j +
130 density_eq * grad_phi_u_j * velocity + grad_phi_p_j -
131 dynamic_viscosity_eq * laplacian_phi_u_j);
132
133 // Store these temporary products in auxiliary variables for speed
134 grad_phi_u_j_x_velocity[j] = grad_phi_u_j * velocity;
135 velocity_gradient_x_phi_u_j[j] = velocity_gradient * phi_u_j;
136 }
137
138 for (unsigned int i = 0; i < n_dofs; ++i)
139 {
140 const unsigned int component_i = scratch_data.components[i];
141
142 const auto &phi_u_i = scratch_data.phi_u[q][i];
143 const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
144 const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
145 const auto &phi_p_i = scratch_data.phi_p[q][i];
146 const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
147
148 // Store these temporary products in auxiliary variables for speed
149 const auto grad_phi_u_i_x_velocity = grad_phi_u_i * velocity;
150 const auto strong_residual_x_grad_phi_u_i =
151 strong_residual * grad_phi_u_i;
152
153 for (unsigned int j = 0; j < n_dofs; ++j)
154 {
155 const unsigned int component_j = scratch_data.components[j];
156
157 const auto &phi_u_j = scratch_data.phi_u[q][j];
158 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
159 // const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
160 const auto shear_rate_j = grad_phi_u_j + transpose(grad_phi_u_j);
161
162 const auto &phi_p_j =
163 scratch_data.phi_p[q][j] * pressure_scaling_factor;
164
165 const auto &strong_jac = strong_jacobian_vec[q][j];
166
167 double local_matrix_ij =
168 component_j == dim ? -div_phi_u_i * phi_p_j : 0;
169
170 if (component_i == dim)
171 {
172 const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
173
174 local_matrix_ij += phi_p_i * div_phi_u_j;
175
176 local_matrix_ij +=
177 tau / density_eq * (strong_jac * grad_phi_p_i);
178 }
179
180 if (component_i < dim && component_j < dim)
181 {
182 local_matrix_ij +=
183 dynamic_viscosity_eq *
184 scalar_product(shear_rate_j, grad_phi_u_i) +
185 density_eq * velocity_gradient_x_phi_u_j[j] * phi_u_i +
186 density_eq * grad_phi_u_j_x_velocity[j] * phi_u_i;
187 }
188
189 if (component_i < dim)
190 {
191 // Jacobian is currently incomplete
192 if (SUPG)
193 {
194 local_matrix_ij +=
195 tau * (strong_jac * grad_phi_u_i_x_velocity +
196 strong_residual_x_grad_phi_u_i * phi_u_j);
197 }
198 }
199 local_matrix_ij *= JxW;
200 local_matrix(i, j) += local_matrix_ij;
201 }
202 }
203 }
204}

207template <int dim>
208void
209GLSNavierStokesVOFAssemblerCore<dim>::assemble_rhs(
210 const NavierStokesScratchData<dim> &scratch_data,
211 StabilizedMethodsTensorCopyData<dim> &copy_data)
212{
213 const std::vector<double> &viscosity_for_stabilization_vector =
214 scratch_data.dynamic_viscosity_for_stabilization;
215 // Loop and quadrature information
216 const auto &JxW_vec = scratch_data.JxW;
217 const unsigned int n_q_points = scratch_data.n_q_points;
218 const unsigned int n_dofs = scratch_data.n_dofs;
219 const double h = scratch_data.cell_size;
220
221 // Copy data elements
222 auto &strong_residual_vec = copy_data.strong_residual;
223 auto &local_rhs = copy_data.local_rhs;
224
225 // Time steps and inverse time steps which is used for stabilization constant
226 std::vector<double> time_steps_vector =
227 this->simulation_control->get_time_steps_vector();
228 const double dt = time_steps_vector[0];
229 const double sdt = 1. / dt;
230
231 Assert(scratch_data.properties_manager.density_is_constant(),
232 RequiresConstantDensity(
233 "GLSNavierStokesVOFAssemblerCore<dim>::assemble_rhs"));
234
235 // Loop over the quadrature points
236 for (unsigned int q = 0; q < n_q_points; ++q)
237 {
238 // Velocity
239 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
240 const double velocity_divergence = scratch_data.velocity_divergences[q];
241 const Tensor<2, dim> velocity_gradient =
242 scratch_data.velocity_gradients[q];
243 const Tensor<1, dim> velocity_laplacian =
244 scratch_data.velocity_laplacians[q];
245
246
247 const Tensor<3, dim> &velocity_hessian =
248 scratch_data.velocity_hessians[q];
249 // From hessian, calculate grad (div (u)) term needed for VOF problems
250 Tensor<1, dim> grad_div_velocity;
251 for (int d = 0; d < dim; ++d)
252 {
253 for (int k = 0; k < dim; ++k)
254 {
255 // hessian[c][i][j] is the (i,j)th component of the matrix of
256 // second derivatives of the cth vector component of the vector
257 // field at quadrature point q of the current cell
258 grad_div_velocity[d] += velocity_hessian[k][d][k];
259 }
260 }
261
262 // Calculate shear rate
263 const Tensor<2, dim> shear_rate =
264 velocity_gradient + transpose(velocity_gradient);
265
266 // Pressure
267 const double pressure = scratch_data.pressure_values[q];
268 const Tensor<1, dim> pressure_gradient =
269 scratch_data.pressure_gradients[q];
270
271 // Forcing term
272 Tensor<1, dim> force = scratch_data.force[q];
273
274 // Calculation of the magnitude of the
275 // velocity for the stabilization parameter
276 const double u_mag = std::max(velocity.norm(), 1e-12);
277
278 // Store JxW in local variable for faster access;
279 const double JxW = JxW_vec[q];
280
281 // Calculation of the equivalent properties at the quadrature point
282 double density_eq = scratch_data.density[q];
283 const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
284
285 // Calculation of the GLS stabilization parameter. The
286 // stabilization parameter used is different if the simulation
287 // is steady or unsteady. In the unsteady case it includes the
288 // value of the time-step
289 const double tau =
290 this->simulation_control->get_assembly_method() ==
291 Parameters::SimulationControl::TimeSteppingMethod::steady ?
292 calculate_navier_stokes_gls_tau_steady(
293 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h) :
294 calculate_navier_stokes_gls_tau_transient(
295 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h, sdt);
296
297
298 // Calculate the strong residual for GLS stabilization.
299 // According to the mathematical formulation, the viscosity jump term
300 // (-dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient))
301 // should be included. However, it is ill-posed for phase change cases
302 // where the solid is represented by a highly viscous fluid. Hence, we
303 // don't consider this term in the current formulation.
304 auto strong_residual = density_eq * velocity_gradient * velocity +
305 pressure_gradient -
306 dynamic_viscosity_eq * velocity_laplacian -
307 dynamic_viscosity_eq * grad_div_velocity -
308 density_eq * force + strong_residual_vec[q];
309 /*- dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient);*/
310
311 // Assembly of the right-hand side
312 for (unsigned int i = 0; i < n_dofs; ++i)
313 {
314 const unsigned int component_i = scratch_data.components[i];
315
316 const auto phi_u_i = scratch_data.phi_u[q][i];
317 const auto grad_phi_u_i = scratch_data.grad_phi_u[q][i];
318 const auto phi_p_i = scratch_data.phi_p[q][i];
319 const auto grad_phi_p_i = scratch_data.grad_phi_p[q][i];
320 const auto div_phi_u_i = scratch_data.div_phi_u[q][i];
321
322
323 // Navier-Stokes Residual
324 // Momentum
325 local_rhs(i) +=
326 (-dynamic_viscosity_eq * scalar_product(shear_rate, grad_phi_u_i) -
327 density_eq * velocity_gradient * velocity * phi_u_i +
328 pressure * div_phi_u_i + density_eq * force * phi_u_i) *
329 JxW;
330
331 if (component_i == dim)
332 {
333 // Continuity
334 local_rhs(i) += -(velocity_divergence * phi_p_i) * JxW;
335
336 // PSPG GLS term
337 local_rhs(i) +=
338 -tau / density_eq * (strong_residual * grad_phi_p_i) * JxW;
339 }
340 // SUPG GLS term
341 if (SUPG)
342 {
343 local_rhs(i) +=
344 -tau * (strong_residual * (grad_phi_u_i * velocity)) * JxW;
345 }
346 }
347 }
348}

354template <int dim>
355void
356GLSNavierStokesVOFAssemblerBDF<dim>::assemble_matrix(
357 const NavierStokesScratchData<dim> &scratch_data,
358 StabilizedMethodsTensorCopyData<dim> &copy_data)
359{
360 // Loop and quadrature information
361 const auto &JxW = scratch_data.JxW;
362 const unsigned int n_q_points = scratch_data.n_q_points;
363 const unsigned int n_dofs = scratch_data.n_dofs;
364
365 // Copy data elements
366 auto &strong_residual = copy_data.strong_residual;
367 auto &strong_jacobian = copy_data.strong_jacobian;
368 auto &local_matrix = copy_data.local_matrix;
369
370 // Time stepping information
371 const auto method = this->simulation_control->get_assembly_method();
372 // Vector for the BDF coefficients
373 const Vector<double> &bdf_coefs =
374 this->simulation_control->get_bdf_coefficients();
375 std::vector<Tensor<1, dim>> velocity(1 +
376 number_of_previous_solutions(method));
377
378 // Loop over the quadrature points
379 for (unsigned int q = 0; q < n_q_points; ++q)
380 {
381 velocity[0] = scratch_data.velocity_values[q];
382 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
383 velocity[p + 1] = scratch_data.previous_velocity_values[p][q];
384
385 const double density = scratch_data.density[q];
386
387 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
388 ++p)
389 {
390 strong_residual[q] += density * bdf_coefs[p] * velocity[p];
391 }
392
393 for (unsigned int j = 0; j < n_dofs; ++j)
394 {
395 strong_jacobian[q][j] +=
396 density * bdf_coefs[0] * scratch_data.phi_u[q][j];
397 }
398
399
400 for (unsigned int i = 0; i < n_dofs; ++i)
401 {
402 const Tensor<1, dim> &phi_u_i = scratch_data.phi_u[q][i];
403 for (unsigned int j = 0; j < n_dofs; ++j)
404 {
405 const Tensor<1, dim> &phi_u_j = scratch_data.phi_u[q][j];
406
407 local_matrix(i, j) +=
408 phi_u_j * phi_u_i * density * bdf_coefs[0] * JxW[q];
409 }
410 }
411 }
412}

414template <int dim>
415void
416GLSNavierStokesVOFAssemblerBDF<dim>::assemble_rhs(
417 const NavierStokesScratchData<dim> &scratch_data,
418 StabilizedMethodsTensorCopyData<dim> &copy_data)
419{
420 // Loop and quadrature information
421 const auto &JxW = scratch_data.JxW;
422 const unsigned int n_q_points = scratch_data.n_q_points;
423 const unsigned int n_dofs = scratch_data.n_dofs;
424
425 // Copy data elements
426 auto &strong_residual = copy_data.strong_residual;
427 auto &local_rhs = copy_data.local_rhs;
428
429 // Time stepping information
430 const auto method = this->simulation_control->get_assembly_method();
431 // Vector for the BDF coefficients
432 const Vector<double> &bdf_coefs =
433 this->simulation_control->get_bdf_coefficients();
434 std::vector<Tensor<1, dim>> velocity(1 +
435 number_of_previous_solutions(method));
436
437 // Loop over the quadrature points
438 for (unsigned int q = 0; q < n_q_points; ++q)
439 {
440 velocity[0] = scratch_data.velocity_values[q];
441 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
442 velocity[p + 1] = scratch_data.previous_velocity_values[p][q];
443
444 const double density = scratch_data.density[q];
445
446 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
447 ++p)
448 {
449 strong_residual[q] += (density * bdf_coefs[p] * velocity[p]);
450 }
451
452
453 for (unsigned int i = 0; i < n_dofs; ++i)
454 {
455 const auto phi_u_i = scratch_data.phi_u[q][i];
456 double local_rhs_i = 0;
457 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
458 ++p)
459 {
460 local_rhs_i -= density * bdf_coefs[p] * (velocity[p] * phi_u_i);
461 }
462 local_rhs(i) += local_rhs_i * JxW[q];
463 }
464 }

471template <int dim>
472void
473PhaseChangeDarcyVOFAssembler<dim>::assemble_matrix(
474 const NavierStokesScratchData<dim> &scratch_data,
475 StabilizedMethodsTensorCopyData<dim> &copy_data)
476{
477 // Loop and quadrature information
478 const auto &JxW_vec = scratch_data.JxW;
479 const unsigned int n_q_points = scratch_data.n_q_points;
480 const unsigned int n_dofs = scratch_data.n_dofs;
481 const auto &velocities = scratch_data.velocity_values;
482 const auto &temperatures = scratch_data.temperature_values;
483 const auto &filtered_phase = scratch_data.filtered_phase_values;
484
485 auto &local_matrix = copy_data.local_matrix;
486 auto &strong_residual = copy_data.strong_residual;
487 auto &strong_jacobian = copy_data.strong_jacobian;
488
489 const unsigned int number_of_fluids = phase_change_parameters_vector.size();
490 std::vector<double> liquid_fractions;
491 std::vector<double> darcy_penalties;
492 liquid_fractions.reserve(number_of_fluids);
493 darcy_penalties.reserve(number_of_fluids);
494
495 // Loop over the quadrature points
496 for (unsigned int q = 0; q < n_q_points; ++q)
497 {
498 // Store JxW in local variable for faster access;
499 const double JxW = JxW_vec[q];
500 // Current temperature values
501 double current_temperature = temperatures[q];
502 // Loop to calculate the liquid fraction and Darcy permeability of each
503 // fluid. Calculated Darcy permeability coefficients depend on the
504 // temperature and material (fluid) properties. A min(max) approach is
505 // used to calculate the liquid fractions avoiding if conditions.
506 for (unsigned int p = 0; p < number_of_fluids; p++)
507 {
508 liquid_fractions.emplace_back(
509 (std::min(1.,
510 std::max((current_temperature -
511 phase_change_parameters_vector[p].T_solidus) /
512 (phase_change_parameters_vector[p].T_liquidus -
513 phase_change_parameters_vector[p].T_solidus),
514 0.))));
515 darcy_penalties.emplace_back(
516 phase_change_parameters_vector[p].penalty_l * liquid_fractions[p] +
517 (1. - liquid_fractions[p]) *
518 phase_change_parameters_vector[p].penalty_s);
519 }
520
521 // For a VOF two-fluid system, the global Darcy penalty coefficient
522 // takes into account the phase change parameters in both fluids
523 const double darcy_penalty =
524 ((1 - filtered_phase[q]) * darcy_penalties[0] +
525 filtered_phase[q] * darcy_penalties[1]);
526
527 strong_residual[q] += darcy_penalty * velocities[q];
528
529 for (unsigned int j = 0; j < n_dofs; ++j)
530 {
531 strong_jacobian[q][j] += darcy_penalty * scratch_data.phi_u[q][j];
532 }
533
534 for (unsigned int i = 0; i < n_dofs; ++i)
535 {
536 const auto &phi_u_i = scratch_data.phi_u[q][i];
537 for (unsigned int j = 0; j < n_dofs; ++j)
538 {
539 const auto &phi_u_j = scratch_data.phi_u[q][j];
540 local_matrix(i, j) += darcy_penalty * phi_u_i * phi_u_j * JxW;
541 }
542 }
543 liquid_fractions.clear();
544 darcy_penalties.clear();
545 }
546}

598 // takes into account the phase change parameters in both fluids
599 const double darcy_penalty =
600 ((1 - filtered_phase[q]) * darcy_penalties[0] +
601 filtered_phase[q] * darcy_penalties[1]);
602
603 strong_residual[q] += darcy_penalty * velocities[q];
604
605 // Assembly of the right-hand side
606 for (unsigned int i = 0; i < n_dofs; ++i)
607 {
608 const auto phi_u_i = scratch_data.phi_u[q][i];
609
610 // Laplacian on the velocity terms
611 local_rhs(i) -= darcy_penalty * velocities[q] * phi_u_i * JxW;
612 }
613 }
614}

702{
703 // Densities of phases
704 Assert(scratch_data.properties_manager.density_is_constant(),
705 RequiresConstantDensity(
706 "GLSNavierStokesVOFAssemblerMarangoni<dim>::assemble_rhs"));
707
708 // Loop and quadrature information
709 const auto &JxW = scratch_data.JxW;
710 const unsigned int n_q_points = scratch_data.n_q_points;
711 const unsigned int n_dofs = scratch_data.n_dofs;
712
713 // Copy data elements
714 auto &strong_residual = copy_data.strong_residual;
715 auto &local_rhs = copy_data.local_rhs;
716
717 // Loop over the quadrature points
718 for (unsigned int q = 0; q < n_q_points; ++q)
719 {
720 // Surface tension
721 const double surface_tension = scratch_data.surface_tension[q];
722 // Surface tension gradient with respect to the temperature
723 const double surface_tension_gradient =
724 scratch_data.surface_tension_gradient[q];
725
726 const double &curvature_value = scratch_data.curvature_values[q];
727
728 // Gather phase fraction gradient
729 const Tensor<1, dim> &phase_gradient_value_q =
730 scratch_data.phase_gradient_values[q];
731
732 // Gather filtered phase fraction gradient
733 const Tensor<1, dim> &filtered_phase_gradient_value_q =
734 scratch_data.filtered_phase_gradient_values[q];
735
736 const double phase_gradient_norm = phase_gradient_value_q.norm();
737
738 const double filtered_phase_gradient_norm =
739 filtered_phase_gradient_value_q.norm();
740
741 const Tensor<1, dim> normalized_phase_fraction_gradient =
742 phase_gradient_value_q / (phase_gradient_norm + DBL_MIN);
743
744 // Gather temperature gradient
745 const Tensor<1, dim> temperature_gradient =
746 scratch_data.temperature_gradients[q];
747
748 const double JxW_value = JxW[q];
749
750
751 const Tensor<1, dim> surface_tension_force =
752 -surface_tension * curvature_value *
753 normalized_phase_fraction_gradient * filtered_phase_gradient_norm;

756 -surface_tension_gradient *
757 (temperature_gradient -
758 normalized_phase_fraction_gradient *

760 filtered_phase_gradient_norm;

765 {
766 const auto phi_u_i = scratch_data.phi_u[q][i];
767 double local_rhs_i = 0;
768
769 local_rhs_i -= (marangoni_effect + surface_tension_force) * phi_u_i;
770 local_rhs(i) += local_rhs_i * JxW_value;
771 }
772 }

855template <int dim>
856void
857GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix(
858 const NavierStokesScratchData<dim> &scratch_data,
859 StabilizedMethodsTensorCopyData<dim> &copy_data)
860{
861 const std::vector<double> &viscosity_for_stabilization_vector =
862 scratch_data.dynamic_viscosity_for_stabilization;
863
864 // Loop and quadrature information
865 const auto &JxW_vec = scratch_data.JxW;
866 const unsigned int n_q_points = scratch_data.n_q_points;
867 const unsigned int n_dofs = scratch_data.n_dofs;
868 const double h = scratch_data.cell_size;
869
870 // Copy data elements
871 auto &strong_residual_vec = copy_data.strong_residual;
872 auto &strong_jacobian_vec = copy_data.strong_jacobian;
873 auto &local_matrix = copy_data.local_matrix;
874
875 // Time steps and inverse time steps which is used for stabilization constant
876 std::vector<double> time_steps_vector =
877 this->simulation_control->get_time_steps_vector();
878 const double dt = time_steps_vector[0];
879 const double sdt = 1. / dt;
880
881 Assert(
882 scratch_data.properties_manager.density_is_constant(),
883 RequiresConstantDensity(
884 "GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix"));
885
886 // Loop over the quadrature points
887 for (unsigned int q = 0; q < n_q_points; ++q)
888 {
889 // Gather into local variables the relevant fields
890 const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
891 const Tensor<2, dim> &velocity_gradient =
892 scratch_data.velocity_gradients[q];
893 const Tensor<1, dim> &velocity_laplacian =
894 scratch_data.velocity_laplacians[q];
895 const Tensor<3, dim> &velocity_hessian =
896 scratch_data.velocity_hessians[q];
897 const Tensor<1, dim> &pressure_gradient =
898 scratch_data.pressure_gradients[q];
899
900 // Calculate shear rate (at each q)
901 const Tensor<2, dim> shear_rate =
902 velocity_gradient + transpose(velocity_gradient);
903
904 // Calculate the shear rate magnitude
905 double shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
906 // Set the shear rate magnitude to 1e-12 if it is too close to zero,
907 // since the viscosity gradient is undefined for shear_rate_magnitude = 0
908 shear_rate_magnitude =
909 shear_rate_magnitude > 1e-12 ? shear_rate_magnitude : 1e-12;
910
911 // Calculate kinematic viscosity gradient
912 const Tensor<1, dim> kinematic_viscosity_gradient =
913 this->get_kinematic_viscosity_gradient(
914 velocity_gradient,
915 velocity_hessian,
916 shear_rate_magnitude,
917 scratch_data.grad_kinematic_viscosity_shear_rate[q]);
918
919 // Forcing term
920 Tensor<1, dim> force = scratch_data.force[q];
921
922 // Calculation of the magnitude of the velocity for the
923 // stabilization parameter
924 const double u_mag = std::max(velocity.norm(), 1e-12);
925
926 // Store JxW in local variable for faster access;
927 const double JxW = JxW_vec[q];
928
929 // Calculation of the equivalent properties at the quadrature point
930 double density_eq = scratch_data.density[q];
931 const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
932 const Tensor<1, dim> dynamic_viscosity_gradient =
933 density_eq * kinematic_viscosity_gradient;
934
935 // Calculation of the GLS stabilization parameter. The
936 // stabilization parameter used is different if the simulation
937 // is steady or unsteady. In the unsteady case it includes the
938 // value of the time-step
939 const double tau =
940 this->simulation_control->get_assembly_method() ==
941 Parameters::SimulationControl::TimeSteppingMethod::steady ?
942 calculate_navier_stokes_gls_tau_steady(
943 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h) :
944 calculate_navier_stokes_gls_tau_transient(
945 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h, sdt);
946
947
948 // Calculate the strong residual for GLS stabilization.
949 // According to the mathematical formulation, the viscosity jump term
950 // (-dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient))
951 // should be included. However, it is ill-posed for phase change cases
952 // where the solid is represented by a highly viscous fluid. Hence, we
953 // don't consider this term in the current formulation.
954 auto strong_residual = density_eq * velocity_gradient * velocity +
955 pressure_gradient -
956 shear_rate * dynamic_viscosity_gradient -
957 dynamic_viscosity_eq * velocity_laplacian -
958 density_eq * force + strong_residual_vec[q];
959 /*- dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient);*/
960
961 std::vector<Tensor<1, dim>> grad_phi_u_j_x_velocity(n_dofs);
962 std::vector<Tensor<1, dim>> velocity_gradient_x_phi_u_j(n_dofs);
963
964 // Pressure scaling factor
965 const double pressure_scaling_factor =
966 scratch_data.pressure_scaling_factor;
967
968 // We loop over the column first to prevent recalculation
969 // of the strong jacobian in the inner loop
970 for (unsigned int j = 0; j < n_dofs; ++j)
971 {
972 const auto &phi_u_j = scratch_data.phi_u[q][j];
973 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
974 const auto &laplacian_phi_u_j = scratch_data.laplacian_phi_u[q][j];
975
976 const auto &grad_phi_p_j =
977 scratch_data.grad_phi_p[q][j] * pressure_scaling_factor;
978
979 const auto &grad_phi_u_j_non_newtonian =
980 grad_phi_u_j + transpose(grad_phi_u_j);
981
982 strong_jacobian_vec[q][j] +=
983 density_eq * velocity_gradient * phi_u_j +
984 density_eq * grad_phi_u_j * velocity + grad_phi_p_j -
985 dynamic_viscosity_eq * laplacian_phi_u_j -
986 grad_phi_u_j_non_newtonian * dynamic_viscosity_gradient;
987
988 // Store these temporary products in auxiliary variables for speed
989 grad_phi_u_j_x_velocity[j] = grad_phi_u_j * velocity;
990 velocity_gradient_x_phi_u_j[j] = velocity_gradient * phi_u_j;
991 }
992
993 shear_rate_magnitude =
994 shear_rate_magnitude > 1e-3 ? shear_rate_magnitude : 1e-3;
995
996 for (unsigned int i = 0; i < n_dofs; ++i)
997 {
998 const auto &phi_u_i = scratch_data.phi_u[q][i];
999 const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
1000 const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
1001 const auto &phi_p_i = scratch_data.phi_p[q][i];
1002 const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
1003
1004 // Store these temporary products in auxiliary variables for speed
1005 const auto grad_phi_u_i_x_velocity = grad_phi_u_i * velocity;
1006 const auto strong_residual_x_grad_phi_u_i =
1007 strong_residual * grad_phi_u_i;
1008
1009 for (unsigned int j = 0; j < n_dofs; ++j)
1010 {
1011 const auto &phi_u_j = scratch_data.phi_u[q][j];
1012 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
1013 const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
1014
1015 const auto &grad_phi_u_j_non_newtonian =
1016 grad_phi_u_j + transpose(grad_phi_u_j);
1017
1018 const auto &phi_p_j =
1019 scratch_data.phi_p[q][j] * pressure_scaling_factor;
1020
1021 const auto &strong_jac = strong_jacobian_vec[q][j];
1022
1023 double local_matrix_ij =
1024 dynamic_viscosity_eq *
1025 scalar_product(grad_phi_u_j_non_newtonian, grad_phi_u_i) +
1026 0.5 * scratch_data.grad_kinematic_viscosity_shear_rate[q] /
1027 shear_rate_magnitude *
1028 scalar_product(grad_phi_u_j_non_newtonian, shear_rate) *
1029 scalar_product(shear_rate, grad_phi_u_i) +
1030 density_eq * velocity_gradient_x_phi_u_j[j] * 0.5 * phi_u_i +
1031 density_eq * grad_phi_u_j_x_velocity[j] * phi_u_i -
1032 div_phi_u_i * phi_p_j;
1033
1034 // Continuity
1035 local_matrix_ij += phi_p_i * div_phi_u_j;
1036
1037 // PSPG GLS term
1038 local_matrix_ij += tau * (strong_jac * grad_phi_p_i);
1039
1040 // The jacobian matrix for the SUPG formulation
1041 // currently does not include the jacobian of the stabilization
1042 // parameter tau. Our experience has shown that does not alter the
1043 // number of newton iteration for convergence, but greatly
1044 // simplifies assembly.
1045 if (SUPG)
1046 {
1047 local_matrix_ij +=
1048 tau * (strong_jac * grad_phi_u_i_x_velocity +
1049 strong_residual_x_grad_phi_u_i * phi_u_j);
1050 }
1051 local_matrix_ij *= JxW;
1052 local_matrix(i, j) += local_matrix_ij;
1053 }
1054 }
1055 }
1056}

1088 for (unsigned int q = 0; q < n_q_points; ++q)
1089 {
1090 // Velocity
1091 const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
1092 const double velocity_divergence = scratch_data.velocity_divergences[q];
1093 const Tensor<2, dim> &velocity_gradient =
1094 scratch_data.velocity_gradients[q];
1095 const Tensor<1, dim> &velocity_laplacian =
1096 scratch_data.velocity_laplacians[q];
1097 const Tensor<3, dim> &velocity_hessian =
1098 scratch_data.velocity_hessians[q];
1099
1100 // Calculate shear rate (at each q)
1101 const Tensor<2, dim> shear_rate =
1102 velocity_gradient + transpose(velocity_gradient);
1103
1104 // Calculate the shear rate magnitude
1105 double shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
1106
1107 shear_rate_magnitude =
1108 shear_rate_magnitude > 1e-12 ? shear_rate_magnitude : 1e-12;
1109
1110 // Calculate kinematic viscosity gradient
1111 const Tensor<1, dim> kinematic_viscosity_gradient =
1112 this->get_kinematic_viscosity_gradient(
1113 velocity_gradient,
1114 velocity_hessian,
1115 shear_rate_magnitude,
1116 scratch_data.grad_kinematic_viscosity_shear_rate[q]);
1117
1118 // Pressure
1119 const double pressure = scratch_data.pressure_values[q];
1120 const Tensor<1, dim> pressure_gradient =
1121 scratch_data.pressure_gradients[q];
1122
1123 // Forcing term
1124 Tensor<1, dim> force = scratch_data.force[q];
1125
1126 // Calculation of the magnitude of the
1127 // velocity for the stabilization parameter
1128 const double u_mag = std::max(velocity.norm(), 1e-12);
1129
1130 // Store JxW in local variable for faster access;

1132

1135 const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
1136 const Tensor<1, dim> dynamic_viscosity_gradient =
1137 density_eq * kinematic_viscosity_gradient;

1140 // stabilization parameter used is different if the simulation
1141 // is steady or unsteady. In the unsteady case it includes the
1142 // value of the time-step
1143 const double tau =
1144 this->simulation_control->get_assembly_method() ==
1145 Parameters::SimulationControl::TimeSteppingMethod::steady ?
1146 calculate_navier_stokes_gls_tau_steady(
1147 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h) :
1148 calculate_navier_stokes_gls_tau_transient(
1149 u_mag, viscosity_for_stabilization_vector[q] / density_eq, h, sdt);
1150
1151 // Calculate the strong residual for GLS stabilization.
1152 // According to the mathematical formulation, the viscosity jump term
1153 // (-dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient))
1154 // should be included. However, it is ill-posed for phase change cases
1155 // where the solid is represented by a highly viscous fluid. Hence, we
1156 // don't consider this term in the current formulation.
1157 auto strong_residual = density_eq * velocity_gradient * velocity +
1158 pressure_gradient -
1159 shear_rate * dynamic_viscosity_gradient -
1160 dynamic_viscosity_eq * velocity_laplacian -
1161 density_eq * force + strong_residual_vec[q];
1162 /*- dynamic_viscosity_jump * (shear_rate * filtered_phase_gradient);*/
1163
1164 // Assembly of the right-hand side
1165 for (unsigned int i = 0; i < n_dofs; ++i)
1166 {
1167 const auto &phi_u_i = scratch_data.phi_u[q][i];
1168 const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
1169 const auto &phi_p_i = scratch_data.phi_p[q][i];
1170 const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
1171 const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
1172
1173 // Navier-Stokes Residual
1174 local_rhs(i) +=
1175 (
1176 // Momentum
1177 -dynamic_viscosity_eq * scalar_product(shear_rate, grad_phi_u_i) -
1178 density_eq * velocity_gradient * velocity * phi_u_i +
1179 pressure * div_phi_u_i + density_eq * force * phi_u_i) *
1180 JxW;
1181
1182 // Continuity
1183 local_rhs(i) += -(velocity_divergence * phi_p_i) * JxW;
1184
1185 // PSPG GLS term
1186 local_rhs(i) += -tau * (strong_residual * grad_phi_p_i) * JxW;
