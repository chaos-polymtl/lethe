13template <int dim, typename PropertiesIndex>
14void
15Visualization<dim, PropertiesIndex>::build_patches(
16 dealii::Particles::ParticleHandler<dim> &particle_handler,
17 const std::vector<std::pair<std::string, int>> &properties)
18{
19 // We reserve 1 more name than the number of properties, since we will also be
20 // writing the ID of the particles
21 this->dataset_names.reserve(PropertiesIndex::n_properties + 1);
22
23 // Adding ID to properties vector for visualization
24 dataset_names.emplace_back("ID");
25
26 // Defining property field position by iterating over properties.
27 for (int field_position = 0; field_position < PropertiesIndex::n_properties;
28 ++field_position)
29 {
30 // Get the property field name
31 const std::string field_name = properties[field_position].first;
32
33 // Number of components of the corresponding property
34 const unsigned n_components = properties[field_position].second;
35
36 // Check to see if the property is a vector
37 // By default we assume that even 2D simulations have 3 components
38 // Since the velocity and the angular velocity are stored in 3D
39 // even for 2D simulations
40 if (n_components == 3)
41 {
42 // The property is a vector, thus we set that the components
43 // are part of a vector. Do not forget that since we added the ID
44 // of the particles, we need to shift the field_position by 1. Hence,
45 // we add 1 to field_position
46 vector_datasets.emplace_back(std::make_tuple(
47 field_position + 1,
48 field_position + n_components,
49 field_name,
50 DataComponentInterpretation::component_is_part_of_vector));
51 }
52 dataset_names.emplace_back(field_name);
53 }
54
55 // Building the patch data
56 patches.resize(particle_handler.n_locally_owned_particles());
57
58 typename dealii::Particles::ParticleHandler<dim>::particle_iterator particle =
59 particle_handler.begin();
60
61 // Looping over particle to get the properties from the particle_handler
62 for (unsigned int i = 0; particle != particle_handler.end(); ++particle, ++i)
63 {
64 // Particle location
65 patches[i].vertices[0] = particle->get_location();
66 patches[i].patch_index = i;
67 patches[i].data.reinit(PropertiesIndex::n_properties + 1, 1);
68
69 // ID and other properties
70 if (particle->has_properties())
71 {
72 auto particle_properties = particle->get_properties();
73
74 // Adding ID to patches
75 patches[i].data(0, 0) = particle->get_id();
76
77 // We need to offset the data we write by one since we have one more
78 // property due to the fact that we save the ID of the particles.
79 for (unsigned int property_index = 0;
80 property_index < PropertiesIndex::n_properties;
81 ++property_index)
82 patches[i].data(property_index + 1, 0) =
83 particle_properties[property_index];
84 }
85 }
86}

88template <int dim, typename PropertiesIndex>
89void
90Visualization<dim, PropertiesIndex>::print_xyz(
91 dealii::Particles::ParticleHandler<dim> &particle_handler,
92 const MPI_Comm &mpi_communicator,
93 const ConditionalOStream &pcout)
94{
95 const bool is_dem_mp =
96 std::is_same_v<PropertiesIndex, DEM::DEMMPProperties::PropertiesIndex>;
97 pcout << "id, type, dp, x, y, z";
98 if constexpr (is_dem_mp)
99 {
100 pcout << ", T";
101 }
102 pcout << " " << std::endl;
103 // Aggressively force synchronization of the header line
104 usleep(500);
105 MPI_Barrier(mpi_communicator);
106 usleep(500);
107 MPI_Barrier(mpi_communicator);
108
109 std::map<int, Particles::ParticleIterator<dim>> global_particles;
110 unsigned int current_id, current_id_max = 0;
111
112 // Mapping of all particles & find the max id on current processor
113 for (auto particle = particle_handler.begin();
114 particle != particle_handler.end();
115 ++particle)
116 {
117 current_id = particle->get_id();
118 current_id_max = std::max(current_id, current_id_max);
119
120 global_particles.insert({current_id, particle});
121 }
122
123 // Find global max particle index
124 unsigned int id_max = Utilities::MPI::max(current_id_max, mpi_communicator);
125
126 // Print particle info one by one in ascending order
127 for (unsigned int i = 0; i <= id_max; i++)
128 {
129 for (auto &iterator : global_particles)
130 {
131 unsigned int id = iterator.first;
132 if (id == i)
133 {
134 auto particle = iterator.second;
135 auto particle_properties = particle->get_properties();
136 auto particle_location = particle->get_location();
137
138 std::cout << std::fixed << std::setprecision(0) << id << " "
139 << std::setprecision(0)
140 << particle_properties[PropertiesIndex::type] << " "
141 << std::setprecision(5)
142 << particle_properties[PropertiesIndex::dp] << " "
143 << std::setprecision(4) << particle_location << " ";
144 if constexpr (is_dem_mp)
145 {
146 std::cout << std::fixed << std::setprecision(4)
147 << particle_properties[PropertiesIndex::T];
148 }
149 std::cout << std::endl;
150 }
151 }
152 usleep(500);
153 MPI_Barrier(mpi_communicator);
154 }
155}

157template <int dim, typename PropertiesIndex>
158void
159Visualization<dim, PropertiesIndex>::print_intermediate_format(
160 const Vector<float> &data_to_print,
161 const DoFHandler<dim> &background_dh,
162 const MPI_Comm &mpi_communicator)
163{
164 unsigned int n_mpi_processes(
165 Utilities::MPI::n_mpi_processes(mpi_communicator));
166 unsigned int this_mpi_process(
167 Utilities::MPI::this_mpi_process(mpi_communicator));
168
169 // Attach the data to the background mesh
170 DataOut<dim> data_out;
171 data_out.attach_dof_handler(background_dh);
172 data_out.add_data_vector(data_to_print,
173 "print_from_processor_" +
174 Utilities::int_to_string(this_mpi_process),
175 DataOut<dim>::type_cell_data);
176 data_out.build_patches();
177
178 std::stringstream out;
179
180 // Add data in deal.II intermediate format to the string stream buffer for
181 // each processor in order
182 for (unsigned int processor_number = 0; processor_number < n_mpi_processes;
183 ++processor_number)
184 {
185 usleep(100);
186 MPI_Barrier(mpi_communicator);
187 if (processor_number == this_mpi_process)
188 {
189 // Generate a string stream buffer to store the output
190 data_out.write_deal_II_intermediate(out);
191
192 // Print in terminal but remove part of the header since it
193 // contains some deal.II version information that may change
194 std::string line;
195 unsigned int counter = 0;
196 while (std::getline(out, line))
197 {
198 if (counter++ > 4)
199 std::cout << line << std::endl;
200 }
201 }
202 }
203}
