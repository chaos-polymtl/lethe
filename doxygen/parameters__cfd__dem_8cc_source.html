66 template <int dim>
67 void
68 VoidFractionParameters<dim>::parse_parameters(ParameterHandler &prm)
69 {
70 prm.enter_subsection("void fraction");
71 const std::string op = prm.get("mode");
72 if (op == "function")
73 mode = Parameters::VoidFractionMode::function;
74 else if (op == "pcm")
75 mode = Parameters::VoidFractionMode::pcm;
76 else if (op == "qcm")
77 mode = Parameters::VoidFractionMode::qcm;
78 else if (op == "spm")
79 mode = Parameters::VoidFractionMode::spm;
80 else
81 throw(std::runtime_error("Invalid void fraction calculation scheme"));
82 prm.enter_subsection("function");
83 void_fraction.parse_parameters(prm);
84 prm.leave_subsection();
85
86 read_dem = prm.get_bool("read dem");
87 dem_file_name = prm.get("dem file name");
88 l2_smoothing_length = prm.get_double("l2 smoothing length");
89 particle_refinement_factor = prm.get_integer("particle refinement factor");
90 qcm_sphere_diameter = prm.get_double("qcm sphere diameter");
91 qcm_sphere_equal_cell_volume = prm.get_bool("qcm sphere equal cell volume");
92 const std::string quadrature_rule_op = prm.get("quadrature rule");
93
94 if (quadrature_rule_op == "gauss")
95 quadrature_rule = Parameters::VoidFractionQuadratureRule::gauss;
96 else if (quadrature_rule_op == "gauss-lobatto")
97 quadrature_rule = Parameters::VoidFractionQuadratureRule::gauss_lobatto;
98 else
99 throw(std::runtime_error(
100 "Invalid quadrature rule for the void fraction calculation scheme. Options are 'gauss' or 'gauss-lobatto'"));
101
102 n_quadrature_points = prm.get_integer("n quadrature points");
103
104 project_particle_velocity = prm.get_bool("project particle velocity");
105
106 prm.leave_subsection();
107 }

194 CFDDEM::parse_parameters(ParameterHandler &prm)
195 {
196 prm.enter_subsection("cfd-dem");
197 grad_div = prm.get_bool("grad div");
198 void_fraction_time_derivative =
199 prm.get_bool("void fraction time derivative");
200 interpolated_void_fraction = prm.get_bool("interpolated void fraction");
201 drag_force = prm.get_bool("drag force");
202 buoyancy_force = prm.get_bool("buoyancy force");
203 shear_force = prm.get_bool("shear force");
204 pressure_force = prm.get_bool("pressure force");
205 saffman_lift_force = prm.get_bool("saffman lift force");
206 magnus_lift_force = prm.get_bool("magnus lift force");
207 rotational_viscous_torque = prm.get_bool("rotational viscous torque");
208 vortical_viscous_torque = prm.get_bool("vortical viscous torque");
209 coupling_frequency = prm.get_integer("coupling frequency");
210 cstar = prm.get_double("grad-div length scale");
211 implicit_stabilization = prm.get_bool("implicit stabilization");
212 particle_statistics = prm.get_bool("particle statistics");
213
214 const std::string op = prm.get("drag model");
215 if (op == "difelice")
216 drag_model = Parameters::DragModel::difelice;
217 else if (op == "rong")
218 drag_model = Parameters::DragModel::rong;
219 else if (op == "dallavalle")
220 drag_model = Parameters::DragModel::dallavalle;
221 else if (op == "kochhill")
222 drag_model = Parameters::DragModel::kochhill;
223 else if (op == "beetstra")
224 drag_model = Parameters::DragModel::beetstra;
225 else if (op == "gidaspow")
226 drag_model = Parameters::DragModel::gidaspow;
227 else
228 throw(std::runtime_error("Invalid drag model"));
229
230 const std::string op1 = prm.get("vans model");
231 if (op1 == "modelA")
232 vans_model = Parameters::VANSModel::modelA;
233 else if (op1 == "modelB")
234 vans_model = Parameters::VANSModel::modelB;
235 else
236 throw(std::runtime_error(
237 "Invalid vans model. Valid choices are modelA and modelB."));
238 prm.leave_subsection();
239 }
