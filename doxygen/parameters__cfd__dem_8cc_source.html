67 template <int dim>
68 void
69 VoidFractionParameters<dim>::parse_parameters(ParameterHandler &prm)
70 {
71 prm.enter_subsection("void fraction");
72 const std::string op = prm.get("mode");
73 if (op == "function")
74 mode = Parameters::VoidFractionMode::function;
75 else if (op == "pcm")
76 mode = Parameters::VoidFractionMode::pcm;
77 else if (op == "qcm")
78 mode = Parameters::VoidFractionMode::qcm;
79 else if (op == "spm")
80 mode = Parameters::VoidFractionMode::spm;
81 else
82 throw(std::runtime_error("Invalid void fraction calculation scheme"));
83 prm.enter_subsection("function");
84 void_fraction.parse_parameters(prm);
85 prm.leave_subsection();
86
87 read_dem = prm.get_bool("read dem");
88 dem_file_name = prm.get("dem file name");
89 l2_smoothing_length = prm.get_double("l2 smoothing length");
90 particle_refinement_factor = prm.get_integer("particle refinement factor");
91 qcm_sphere_diameter = prm.get_double("qcm sphere diameter");
92 qcm_sphere_equal_cell_volume = prm.get_bool("qcm sphere equal cell volume");
93 const std::string quadrature_rule_op = prm.get("quadrature rule");
94
95 if (quadrature_rule_op == "gauss")
96 quadrature_rule = Parameters::VoidFractionQuadratureRule::gauss;
97 else if (quadrature_rule_op == "gauss-lobatto")
98 quadrature_rule = Parameters::VoidFractionQuadratureRule::gauss_lobatto;
99 else
100 throw(std::runtime_error(
101 "Invalid quadrature rule for the void fraction calculation scheme. Options are 'gauss' or 'gauss-lobatto'"));
102
103 n_quadrature_points = prm.get_integer("n quadrature points");
104
105 project_particle_velocity = prm.get_bool("project particle velocity");
106
107 prm.leave_subsection();
108 }

207 CFDDEM::parse_parameters(ParameterHandler &prm)
208 {
209 prm.enter_subsection("cfd-dem");
210 grad_div = prm.get_bool("grad div");
211 void_fraction_time_derivative =
212 prm.get_bool("void fraction time derivative");
213 interpolated_void_fraction = prm.get_bool("interpolated void fraction");
214 drag_force = prm.get_bool("drag force");
215 buoyancy_force = prm.get_bool("buoyancy force");
216 shear_force = prm.get_bool("shear force");
217 pressure_force = prm.get_bool("pressure force");
218 saffman_lift_force = prm.get_bool("saffman lift force");
219 magnus_lift_force = prm.get_bool("magnus lift force");
220 rotational_viscous_torque = prm.get_bool("rotational viscous torque");
221 vortical_viscous_torque = prm.get_bool("vortical viscous torque");
222 coupling_frequency = prm.get_integer("coupling frequency");
223 cstar = prm.get_double("grad-div length scale");
224 implicit_stabilization = prm.get_bool("implicit stabilization");
225 particle_statistics = prm.get_bool("particle statistics");
226 project_particle_forces = prm.get_bool("project particle forces");
227
228 const std::string op = prm.get("drag model");
229 if (op == "difelice")
230 drag_model = Parameters::DragModel::difelice;
231 else if (op == "rong")
232 drag_model = Parameters::DragModel::rong;
233 else if (op == "dallavalle")
234 drag_model = Parameters::DragModel::dallavalle;
235 else if (op == "kochhill")
236 drag_model = Parameters::DragModel::kochhill;
237 else if (op == "beetstra")
238 drag_model = Parameters::DragModel::beetstra;
239 else if (op == "gidaspow")
240 drag_model = Parameters::DragModel::gidaspow;
241 else
242 AssertThrow(false, ExcMessage("Invalid drag model"));
243
244 const std::string drag_coupling_str = prm.get("drag coupling");
245 if (drag_coupling_str == "implicit")
246 drag_coupling = Parameters::DragCoupling::fully_implicit;
247 else if (drag_coupling_str == "explicit")
248 drag_coupling = Parameters::DragCoupling::fully_explicit;
249 else if (drag_coupling_str == "semi-implicit")
250 drag_coupling = Parameters::DragCoupling::semi_implicit;
251 else
252 AssertThrow(false, ExcMessage("Drag coupling formulation"));
253
254 const std::string op1 = prm.get("vans model");
255 if (op1 == "modelA")
256 vans_model = Parameters::VANSModel::modelA;
257 else if (op1 == "modelB")
258 vans_model = Parameters::VANSModel::modelB;
259 else
260 throw(std::runtime_error(
261 "Invalid vans model. Valid choices are modelA and modelB."));
262 prm.leave_subsection();
