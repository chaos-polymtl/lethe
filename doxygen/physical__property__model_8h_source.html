30/*
31 * Fields on which physical property can depend. All fields are assumed
32 * to be at time t+dt other than those for which a _p suffix is explicitly
33 * given. The suffix _p1 implies that the field is at time t (previous one step)
34 * and the suffix _p2 implies that the field is at time t_dt (previous two
35 * steps).
36 */
37enum field : int
38{
39 shear_rate,
40 temperature,
41 temperature_p1, // temperature at time t
42 temperature_p2, // temperature at time t-dt
43 pressure,
44 phase_order_cahn_hilliard,
45 phase_order_cahn_hilliard_filtered,
46 levelset,
47 tracer_concentration
48};

63/**
64 * @brief Abstract class that defines the interface for a physical property model
65 * Physical property model provides an abstract interface to calculate the
66 * value of a physical property or a vector of physical property value for
67 * given field value. By default, the interface does not require that all (or
68 * any) fields be specified. This is why a map is used to pass the dependent
69 * variable. To allow for the calculation of the appropriate jacobian matrix
70 * (when that is necessary) the interface also provides a jacobian function,
71 * which must provide the derivative with respect to the field specified as an
72 * argument.
73 */
74class PhysicalPropertyModel
75{
76public:
77 /**
78 * @brief Default constructor. Set the model_depends_on to false for all variables.
79 */
80 PhysicalPropertyModel()

84 model_depends_on[temperature_p1] = false;

86 model_depends_on[pressure] = false;
87 model_depends_on[phase_order_cahn_hilliard] = false;
88 model_depends_on[levelset] = false;

163 /**
164 * @brief numerical_jacobian Calculates the jacobian through a forward finite difference (Euler) approximation.
165 * This approach, although not preferable, is meant as a fall-back when
166 * calculating the jacobian manually is too difficult.
167 * @param fields_values The values of the various fields
168 * @param id Field id of the field with respect to which the jacobian should be calculated
169 * @return
170 */
171 inline double
172 numerical_jacobian(const std::map<field, double> &field_values,
173 const field id)
174 {
175 double f_x = this->value(field_values);
176 auto x_dx = field_values;
177 double dx = std::max(1e-6 * x_dx[id], 1e-8);
178 x_dx[id] = x_dx[id] + dx;
179 double f_xdx = this->value(x_dx);
180 return (f_xdx - f_x) / dx;
181 }
182
183 /**
184 * @brief vector_numerical_jacobian Calculates the vector of jacobian through forward finite difference (Euler) approximation.
185 * This approach, although not preferable, is meant as a fall-back when
186 * calculating the jacobian manually is too difficult.
187 * @param field_vectors
188 * @param field_id
189 * @param jacobian_vector
190 * @return
191 */
192 inline void
193 vector_numerical_jacobian(
194 const std::map<field, std::vector<double>> &field_vectors,
195 const field id,
196 std::vector<double> &jacobian_vector)
197 {
198 const unsigned int n_pts = jacobian_vector.size();
199
200 // Evaluate the properties using the values of the field vector
201 std::vector<double> f_x(n_pts);
202 vector_value(field_vectors, f_x);
203
204 // Make a copy of the field vector for the field we wil perturbate
205 std::map<field, std::vector<double>> perturbed_field_vectors =
206 field_vectors;
207 std::vector<double> &x = perturbed_field_vectors.at(id);
208 std::vector<double> dx(n_pts);
209
210 // Perturb the field by dx
211 for (unsigned int i = 0; i < n_pts; ++i)
212 {
213 dx[i] = std::max(1e-6 * x[i], 1e-8);
214 x[i] += dx[i];
215 }
216
217 // Evaluate the properties using the perturbed value of the field vector
218 std::vector<double> f_xdx(n_pts);
219 vector_value(perturbed_field_vectors, f_xdx);
220
221 // Fill jacobian
222 for (unsigned int i = 0; i < n_pts; ++i)
223 jacobian_vector[i] = (f_xdx[i] - f_x[i]) / dx[i];
224 }
225
226protected:
227 std::shared_ptr<SimulationControl> &
228 get_simulation_control()
229 {
230 AssertThrow(simulation_control, SimulationControlUsageByProperty());
231 return simulation_control;
232 }
233
234 // Map to indicate on which variables the model depends on
235 std::map<field, bool> model_depends_on;
236
237private:
238 // SimulationControl object. This can be used to set time-dependent
239 // physical properties or properties which depend on time derivatives.
240 // The SimulationControl must be provided to the physical property
241 // through the solver itself.
242 std::shared_ptr<SimulationControl> simulation_control;
243};
