19 /* @brief Converts an input particle handler with a given set of input properties
20  * into an output particle handler with a different set of output properties.
21  * This is used, notably, to convert DEM simulation particle_handlers into
22  * CFD-DEM particle_handlers.
23  *
24  * @tparam dim the number of dimensions
25  * @tparam input the properties index used for the ph_in
26  * @tparam output the properties index used for the ph_out
27  * @param triangulation The triangulation on which the ph_out is stored
28  * @param ph_in The input particle_handler that has properties indified by the input
29  * @param ph_out The output partcile_handler that has properties identified by the output
30  */
31 
32 template <int dim, typename input, typename output>
33 void
34 convert_particle_handler(
35  const parallel::distributed::Triangulation<dim> &triangulation,
36  const Particles::ParticleHandler<dim> &ph_in,
37  Particles::ParticleHandler<dim> &ph_out)
38 {
39  // Pre-allocate the vector of particle properties and location of ph_out
40  std::vector<std::vector<double>> ph_out_properties;
41  std::vector<Point<dim>> ph_out_points;
42  ph_out_properties.reserve(ph_in.n_locally_owned_particles());
43  ph_out_points.reserve(ph_in.n_locally_owned_particles());
44 
45  // We loop through all particles of the input particle handler and gather the
46  // properties and locations
47  for (auto particle = ph_in.begin(); particle != ph_in.end(); ++particle)
48  {
49  // Get and store particle location
50  auto particle_location = particle->get_location();
51  ph_out_points.emplace_back(particle_location);
52 
53  // Get the particle properties
54  auto particle_properties = particle->get_properties();
55  // Give a default value of zero to all properties. This will account for
56  // increasing number of properties
57  std::vector<double> ph_out_particle_properties(output::n_properties, 0.);
58  ph_out_particle_properties[output::type] =
59  particle_properties[input::type];
60  ph_out_particle_properties[output::dp] = particle_properties[input::dp];
61 
62  ph_out_particle_properties[output::v_x] = particle_properties[input::v_x];
63  ph_out_particle_properties[output::v_y] = particle_properties[input::v_y];
64  ph_out_particle_properties[output::v_z] = particle_properties[input::v_z];
65 
66  ph_out_particle_properties[output::omega_x] =
67  particle_properties[input::omega_x];
68  ph_out_particle_properties[output::omega_y] =
69  particle_properties[input::omega_y];
70  ph_out_particle_properties[output::omega_z] =
71  particle_properties[input::omega_z];
72 
73  ph_out_particle_properties[output::mass] =
74  particle_properties[input::mass];
75 
76  ph_out_properties.emplace_back(ph_out_particle_properties);
77  }
78 
79 
80  // We use the insert global functionalities to insert the particles in the
81  // output particle handler. This is not optimal since it will recalculate the
82  // reference location of every particle, but since this operation is only done
83  // once per simulation, this cost should be negligible.
84 
85  // Calculate global bounding box for global insertion
86  const auto my_bounding_box = GridTools::compute_mesh_predicate_bounding_box(
87  triangulation, IteratorFilters::LocallyOwnedCell());
88 #if DEAL_II_VERSION_GTE(9, 7, 0)
89  const auto global_bounding_boxes =
90  Utilities::MPI::all_gather(triangulation.get_mpi_communicator(),
91  my_bounding_box);
92 #else
93  const auto global_bounding_boxes =
94  Utilities::MPI::all_gather(triangulation.get_communicator(),
95  my_bounding_box);
96 #endif
97 
98  // Do the global insertion
99  ph_out.insert_global_particles(ph_out_points,
100  global_bounding_boxes,
101  ph_out_properties);
102 }
