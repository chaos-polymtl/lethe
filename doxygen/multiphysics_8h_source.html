17/**
18 * @brief Solution fields of the different physics that are used as an indicator
19 * for multiple purposes (e.g. adaptive mesh refinement, solid domain
20 * constraints).
21 */
22enum class Variable : unsigned int
23{ /// Velocity vector field from fluid dynamics
24 velocity = 0,
25 /// Pressure scalar field from fluid dynamics

33 /// Chemical potential scalar field from Cahn Hilliard
34 chemical_potential_cahn_hilliard = 5,

37};

39/**
40 * @brief Utility function used for parsing physics-based
41 * parameters
42 *
43 */
44inline PhysicsID
45get_physics_id(std::string physics_name)
46{
47 if (physics_name == "fluid dynamics")
48 return PhysicsID::fluid_dynamics;
49 else if (physics_name == "heat transfer")
50 return PhysicsID::heat_transfer;
51 else if (physics_name == "tracer")
52 return PhysicsID::tracer;
53 else if (physics_name == "VOF")
54 return PhysicsID::VOF;
55 else if (physics_name == "cahn hilliard")
56 return PhysicsID::cahn_hilliard;
57 else if (physics_name == "void fraction")
58 return PhysicsID::void_fraction;
59 else
60 AssertThrow(false,
61 dealii::StandardExceptions::ExcMessage(
62 "An unknown Physics name was requested"));
63}
