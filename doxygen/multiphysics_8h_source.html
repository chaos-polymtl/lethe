4/**
5 * @file multiphysics.h
6 * @brief Identifiers for multiphysics modules and their solution fields.
7 *
8 * This file defines the enumerations used to identify the different physics
9 * modules available in Lethe (fluid dynamics, heat transfer, VOF, etc.) and
10 * the solution fields they produce. It also provides a utility function to
11 * convert a physics name string into its corresponding PhysicsID.
12 */
13
14#ifndef lethe_multiphysics_h
15#define lethe_multiphysics_h

17/**
18 * @brief Unique identifiers for each physics module in Lethe.
19 *
20 * These identifiers are used to index and refer to the different physics
21 * solvers throughout the multiphysics framework.
22 */
23enum PhysicsID : unsigned int
24{
25 /// Fluid Dynamics (Velocity,Pressure)
26 fluid_dynamics = 0,
27 /// Heat transfer (Temperature)
28 heat_transfer = 1,
29 /// Passive scalar tracer (Tracer)
30 tracer = 2,
31 /// Conservative level-set / Volume-of-Fluid (Phase indicator / phase
32 /// fraction)
33 VOF = 3,
34 /// Cahn-Hilliard (Phase indicator, Potential)
35 cahn_hilliard = 4,
36 /// Void fraction
37 void_fraction = 5,
38 /// Time-harmonic electromagnetics (Electric field, Magnetic field)
39 electromagnetics = 6
40};
41
42/**
43 * @brief Solution fields of the different physics that are used as an indicator
44 * for multiple purposes (e.g. adaptive mesh refinement, solid domain
45 * constraints).
46 */
47enum class Variable : unsigned int
48{ /// Velocity vector field from fluid dynamics
49 velocity = 0,
50 /// Pressure scalar field from fluid dynamics
51 pressure = 1,
52 /// Phase fraction scalar field from VOF
53 phase = 2,
54 /// Temperature scalar field from heat transfer
55 temperature = 3,
56 /// Phase order scalar field from Cahn Hilliard
57 phase_cahn_hilliard = 4,
58 /// Chemical potential scalar field from Cahn Hilliard
59 chemical_potential_cahn_hilliard = 5,
60 /// Tracer scalar field
61 tracer = 6,
62 /// Electromagnetics field. This is a single field since mesh adaptation
63 /// is carried out on the entire solution and not on the electric or magnetic
64 /// field separately.
65 electomagnetics = 7
66};
67
68/**
69 * @brief Utility function used for parsing physics-based
70 * parameters
71 *
72 */
73inline PhysicsID
74get_physics_id(std::string physics_name)
75{
76 if (physics_name == "fluid dynamics")
77 return PhysicsID::fluid_dynamics;
78 else if (physics_name == "heat transfer")
79 return PhysicsID::heat_transfer;
80 else if (physics_name == "tracer")
81 return PhysicsID::tracer;
82 else if (physics_name == "VOF")
83 return PhysicsID::VOF;
84 else if (physics_name == "cahn hilliard")
85 return PhysicsID::cahn_hilliard;
86 else if (physics_name == "void fraction")
87 return PhysicsID::void_fraction;
88 else if (physics_name == "electromagnetics")
89 return PhysicsID::electromagnetics;
90 else
91 AssertThrow(false,
92 dealii::StandardExceptions::ExcMessage(
93 "An unknown Physics name was requested"));
94}
