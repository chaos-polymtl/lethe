24template <int dim>
25BoundaryCellsInformation<dim>::BoundaryCellsInformation() = default;

27template <int dim>
28void
29BoundaryCellsInformation<dim>::build(
30 const parallel::distributed::Triangulation<dim> &triangulation,
31 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
32 const std::vector<unsigned int> &outlet_boundaries,
33 const bool &check_diamond_cells,
34 const bool &expand_particle_wall_contact_search,
35 const ConditionalOStream &pcout)
36{
37 boundary_cells_with_faces.clear();
38 global_boundary_cells_with_faces.clear();
39 boundary_cells_with_lines.clear();
40 local_cells_with_boundary_lines.clear();
41 boundary_cells_information.clear();
42 global_boundary_cells_information.clear();
43 boundary_cells_with_points.clear();
44 boundary_cells_for_floating_walls.clear();
45
46 const double maximal_cell_diameter =
47 GridTools::maximal_cell_diameter(triangulation);
48
49 find_boundary_cells_information(triangulation, outlet_boundaries);
50
51 // Finding boundary cells with lines and points
52 find_particle_point_and_line_contact_cells(triangulation);
53
54 // Finding cells adjacent to floating walls
55 find_boundary_cells_for_floating_walls(triangulation,
56 floating_wall_properties,
57 maximal_cell_diameter);
58
59 // Add cells with boundary lines to boundary cells container
60 add_cells_with_boundary_lines_to_boundary_cells(triangulation,
61 outlet_boundaries,
62 check_diamond_cells,
63 pcout);
64
65 if (expand_particle_wall_contact_search)
66 {
67 if (display_pw_contact_expansion_warning)
68 {
69 pcout
70 << "Warning: expansion of particle-wall contact list is enabled. "
71 << std::endl
72 << "This feature should only be activated in geometries with concave boundaries. "
73 "(For example, for particles flow inside a cylinder or sphere). In geometries with "
74 "convex boundaries, this feature MUST NOT be activated."
75 << std::endl;
76 display_pw_contact_expansion_warning = false;
77 }
78
79 add_boundary_neighbors_of_boundary_cells(
80 triangulation,
81 outlet_boundaries,
82 boundary_cells_information,
83 global_boundary_cells_information);
84 }
85 else
86 {
87 if (display_pw_contact_expansion_warning)
88 {
89 pcout
90 << "Warning: expansion of particle-wall contact list is disabled. "
91 << std::endl
92 << "This feature is useful in geometries with concave boundaries. "
93 << std::endl;
94 display_pw_contact_expansion_warning = false;
95 }
96 }
97}

128// This function finds all the boundary cells and faces in the triangulation,
129// for each cell the boundary faces are specified and the normal vector as well
130// as a point on the boundary faces are obtained
131template <int dim>
132void
133BoundaryCellsInformation<dim>::find_boundary_cells_information(
134 const parallel::distributed::Triangulation<dim> &triangulation,
135 const std::vector<unsigned int> &outlet_boundaries)
136{
137 // Initializing output_vector and a search_vector (containing boundary_id and
138 // cell) to avoid replication of a boundary face. All the found boundary faces
139 // will be searched in this vector before being added to the output_vector. If
140 // they are not found in this search_vector they will be added to the
141 // output_vector as well as the search_vector
142 // std::map<int, boundary_cells_info_struct<dim>> output_vector;
143 std::vector<std::pair<int, typename Triangulation<dim>::active_cell_iterator>>
144 search_vector;
145
146 // Initialize a simple quadrature for on the system. This will be used to
147 // obtain a single sample point on the boundary faces
148 const FE_Q<dim> fe(1);
149 QGauss<dim - 1> face_quadrature_formula(1);
150 FEFaceValues<dim> fe_face_values(fe,
151 face_quadrature_formula,
152 update_values | update_quadrature_points |
153 update_normal_vectors);
154
155 // Iterating over the active cells in the triangulation
156 for (const auto &cell : triangulation.active_cell_iterators())
157 {
158 // Iterating over the faces of each cell
159 for (const auto &face : cell->face_iterators())
160 {
161 // We search to see if the boundary is defined as an outlet, periodic
162 // or not. If it is not defined as one of those, we proceed.
163 int face_id = cell->face_iterator_to_index(face);
164 bool is_outlet =
165 std::find(outlet_boundaries.begin(),
166 outlet_boundaries.end(),
167 face->boundary_id()) != outlet_boundaries.end();
168 bool is_periodic = cell->has_periodic_neighbor(face_id);
169
170 if (!is_outlet && !is_periodic)
171 {
172 // Check if the face is located at boundary
173 if (face->at_boundary())
174 {
175 fe_face_values.reinit(cell, face_id);
176
177 // Finding the normal vector of the boundary face
178 Tensor<1, dim> normal_vector =
179 -fe_face_values.normal_vector(0);
180
181 // Finding a point on the boundary face
182 Point<dim> quad_point = fe_face_values.quadrature_point(0);
183
184 // Get global index of the face
185 unsigned int global_face_id = cell->face_index(face_id);
186
187 // Storing these information into the
188 // boundary_cells_info_struct
189 boundary_cells_info_struct<dim> boundary_information;
190 boundary_information.cell = cell;
191 boundary_information.boundary_id = face->boundary_id();
192 boundary_information.global_face_id = global_face_id;
193 boundary_information.normal_vector = normal_vector;
194 boundary_information.point_on_face = quad_point;
195
196 // Searching to see if boundary with this information was
197 // already found or not. If this is a new
198 // boundary face, it will be added to the
199 // output_vector as well as the search_vector to avoid
200 // repetition
201 auto information_search_element =
202 std::make_pair(face_id, cell);
203 auto search_iterator = std::find(search_vector.begin(),
204 search_vector.end(),
205 information_search_element);
206 if (search_iterator == search_vector.end())
207 {
208 if (cell->is_locally_owned())
209 {
210 boundary_cells_information.insert(
211 {global_face_id, boundary_information});
212 boundary_cells_with_faces.push_back(cell);
213 search_vector.push_back(information_search_element);
214 }
215 global_boundary_cells_information.insert(
216 {global_face_id, boundary_information});
217 }
218 }
219 }
220 }
221 }
222}

225// This function is used to update the normal vector and the location of the
226// stored point of the boundary faces. It is used when the grid is moving.
227// Updated points and normal vectors are then used to update the particle-wall
228// contact list.
229template <int dim>
230void
231BoundaryCellsInformation<dim>::update_boundary_info_after_grid_motion(
232 typename DEM::dem_data_structures<dim>::boundary_points_and_normal_vectors
233 &updated_boundary_points_and_normal_vectors)
234{
235 // If there is no grid motion, exit the function
236 if (!DEMActionManager::get_action_manager()->check_grid_motion_enabled())
237 return;
238
239 // Initialize a simple quadrature for on the system. This will be used to
240 // obtain a single sample point on the boundary faces
241 const FE_Q<dim> fe(1);
242 QGauss<dim - 1> face_quadrature_formula(1);
243 FEFaceValues<dim> fe_face_values(fe,
244 face_quadrature_formula,
245 update_values | update_quadrature_points |
246 update_normal_vectors);
247 for (auto &cell : boundary_cells_with_faces)
248 {
249 if (cell->is_locally_owned())
250 {
251 // Iterating over the faces of each cell
252 for (const auto &face : cell->face_iterators())
253 {
254 // Check to see if the face is located at boundary
255 if (face->at_boundary())
256 {
257 int face_id = cell->face_iterator_to_index(face);
258 fe_face_values.reinit(cell, face_id);
259
260 // Finding the normal vector of the boundary face
261 Tensor<1, 3> normal_vector;
262
263 if constexpr (dim == 3)
264 normal_vector = -fe_face_values.normal_vector(0);
265
266 if constexpr (dim == 2)
267 normal_vector =
268 tensor_nd_to_3d(-fe_face_values.normal_vector(0));
269
270 // Finding a point on the boundary face
271 Point<3> quad_point;
272
273 if constexpr (dim == 3)
274 quad_point = fe_face_values.quadrature_point(0);
275
276 if constexpr (dim == 2)
277 quad_point =
278 point_nd_to_3d(fe_face_values.quadrature_point(0));
279
280 updated_boundary_points_and_normal_vectors[cell->face_index(
281 face_id)] = std::make_pair(normal_vector, quad_point);
282 }
283 }
284 }
285 }

288// This function returns all the cells at which particle-line and particle-point
289// contact search should be performed
290template <int dim>
291void
292BoundaryCellsInformation<dim>::find_particle_point_and_line_contact_cells(
293 const parallel::distributed::Triangulation<dim> &triangulation)
294{
295 // Here, we first loop through all the non-boundary faces of the local cells
296 // and find (add them to all cells_with_boundary_lines container) all boundary
297 // lines of these faces. Then the boundary lines which are the borders of two
298 // boundary faces should be removed from this container. Finally the remained
299 // elements of this container are added to boundary_cells_with_lines
300
301 // Boundary lines only exist in three-dimensional cases
302 if (dim == 3)
303 {
304 std::unordered_map<std::string,
305 std::unordered_map<types::global_dof_index,
306 std::pair<Point<dim>, Point<dim>>>>
307 all_cells_with_boundary_lines;
308
309 // Iterating over the active cells in the triangulation
310 for (const auto &cell : triangulation.active_cell_iterators())
311 {
312 if (cell->is_locally_owned())
313 {
314 // Looping through all the faces of the cell
315 for (const auto &face : cell->face_iterators())
316 {
317 // Check to see if the face is not located at boundary
318 if (!face->at_boundary())
319 {
320 // Adding all the boundary lines of these faces into the
321 // all_cells_with_boundary_lines container
322 for (unsigned int l = 0; l < face->n_lines(); ++l)
323 {
324 if (face->line(l)->at_boundary())
325 {
326 all_cells_with_boundary_lines[cell->id()
327 .to_string()]
328 .emplace(
329 face->line_index(l),
330 std::make_pair(face->line(l)->vertex(0),
331 face->line(l)->vertex(1)));
332 }
333 }
334 }
335 }
336 }
337 }
338
339 // Now iterating again over the cells, faces and lines and if the boundary
340 // line of the boundary face exists in the container, it is removed from
341 // the container
342 if (!all_cells_with_boundary_lines.empty())
343 {
344 for (const auto &cell : triangulation.active_cell_iterators())
345 {
346 if (cell->is_locally_owned())
347 {
348 for (const auto &face : cell->face_iterators())
349 {
350 // Check to see if the face is located at boundary
351 if (face->at_boundary())
352 {
353 for (unsigned int l = 0; l < face->n_lines(); ++l)
354 {
355 all_cells_with_boundary_lines[cell->id()
356 .to_string()]
357 .erase(face->line_index(l));
358 }
359 }
360 }
361 }
362 }
363 }
364
365 // Finally, adding the remained elements of the
366 // all_cells_with_boundary_lines container to boundary_cells_with_lines
367 if (!all_cells_with_boundary_lines.empty())
368 {
369 for (const auto &cell : triangulation.active_cell_iterators())
370 {
371 if (cell->is_locally_owned())
372 {
373 std::string cell_id_string = cell->id().to_string();
374 std::unordered_map<types::global_dof_index,
375 std::pair<Point<dim>, Point<dim>>>
376 &cell_boundary_lines =
377 all_cells_with_boundary_lines[cell_id_string];
378
379 if (!cell_boundary_lines.empty())
380 {
381 for (auto &map_iterator : cell_boundary_lines)
382 {
383 Point<3> vertex_one, vertex_two;
384
385 if constexpr (dim == 3)
386 {
387 vertex_one = map_iterator.second.first;
388 vertex_two = map_iterator.second.second;
389 }
390
391 if constexpr (dim == 2)
392 {
393 vertex_one =
394 point_nd_to_3d(map_iterator.second.first);
395 vertex_two =
396 point_nd_to_3d(map_iterator.second.second);
397 }
398 boundary_cells_with_lines.emplace(
399 cell_id_string,
400 cell_line_info<dim>{cell, vertex_one, vertex_two});
401
402 // Add the cell to local_cells_with_boundary_lines to
403 // be used in
404 // add_cells_with_boundary_lines_to_boundary_cells
405 local_cells_with_boundary_lines.push_back(cell);
406 }
407 }
408 }
409 }
410 }
411 }
412
413 // Finding boundary points. We need this container because at_boundary()
414 // function is not defined for vertex id (unsigned int) nor vertex position
415 // (Point<dim>). For line and face objects, at_boundary() function is
416 // available to help
417
418 // First getting a set of all boundary vertices
419 std::set<unsigned int> boundary_vertices;
420 for (const auto &face : triangulation.active_face_iterators())
421 {
422 if (face->at_boundary())
423 {
424 for (unsigned int v = 0; v < face->n_vertices(); ++v)
425 {
426 boundary_vertices.insert(face->vertex_index(v));
427 }
428 }
429 }
430
431 // If a cell does not have any boundary line, but has boundary vertex, it is
432 // added to boundary_cells_with_points container Iterating over the active
433 // cells in the triangulation
434 for (const auto &cell : triangulation.active_cell_iterators())
435 {
436 if (cell->is_locally_owned() && !cell->has_boundary_lines())
437 {
438 for (const auto &face : cell->face_iterators())
439 {
440 if (!face->at_boundary())
441 {
442 for (unsigned int v = 0; v < face->n_vertices(); ++v)
443 {
444 if (boundary_vertices.count(face->vertex_index(v)) > 0)
445 {
446 Point<3> vertex;
447
448 if constexpr (dim == 3)
449 vertex = face->vertex(v);
450
451 if constexpr (dim == 2)
452 vertex = point_nd_to_3d(face->vertex(v));
453
454 boundary_cells_with_points.emplace(
455 cell->id().to_string(),
456 cell_point_info<dim>{cell, vertex});
457 }
458 }
459 }
460 }
461 }
462 }

465template <int dim>
466void
467BoundaryCellsInformation<dim>::find_global_boundary_cells_with_faces(
468 const parallel::distributed::Triangulation<dim> &triangulation,
469 const std::vector<unsigned int> &outlet_boundaries)
470{
471 // Iterating over the active cells in the triangulation
472 for (const auto &cell : triangulation.active_cell_iterators())
473 {
474 // Iterating over the faces of each cell
475 for (const auto &face : cell->face_iterators())
476 {
477 // We search to see if the boundary is defined as an outlet or
478 // not. If it is not defined as an outlet we proceed.
479 if (std::find(outlet_boundaries.begin(),
480 outlet_boundaries.end(),
481 face->boundary_id()) == outlet_boundaries.end())
482 {
483 // Check to see if the face is located at boundary
484 if (face->at_boundary())
485 {
486 global_boundary_cells_with_faces.push_back(cell);
487 }
488 }
489 }
490 }

493template <int dim>
494void
495BoundaryCellsInformation<dim>::add_cells_with_boundary_lines_to_boundary_cells(
496 const parallel::distributed::Triangulation<dim> &triangulation,
497 const std::vector<unsigned int> &outlet_boundaries,
498 const bool &check_diamond_cells,
499 const ConditionalOStream &pcout)
500{
501 std::vector<typename Triangulation<dim>::active_cell_iterator>
502 boundary_neighbor_cells;
503
504 // First we have to store all the global (including local and non-local) cells
505 // with boundary faces in a container. Note that the boundary_cells_with_faces
506 // container only stores the local boundary cells
507 find_global_boundary_cells_with_faces(triangulation, outlet_boundaries);
508
509 if (!local_cells_with_boundary_lines.empty())
510 {
511 for (auto &cell_with_boundary_line : local_cells_with_boundary_lines)
512 {
513 // Each cell with boundary line must have two neighbor cells with
514 // boundary faces. We have to find these two neighbors first.
515 // Finding neighbors of the cell
516 std::vector<typename Triangulation<dim>::active_cell_iterator>
517 active_neighbors;
518 GridTools::get_active_neighbors<Triangulation<dim>>(
519 cell_with_boundary_line, active_neighbors);
520
521 // Looping through the neighbors and storing the ones with boundary
522 // faces
523 boundary_neighbor_cells.clear();
524 for (auto &neighbor : active_neighbors)
525 {
526 if (std::count(global_boundary_cells_with_faces.begin(),
527 global_boundary_cells_with_faces.end(),
528 neighbor))
529 {
530 boundary_neighbor_cells.push_back(neighbor);
531 }
532 }
533
534 // Find the information of every combination of two cells in
535 // global_boundary_cells_normals with counters
536 for (unsigned int counter_one = 0;
537 counter_one < boundary_neighbor_cells.size();
538 ++counter_one)
539 {
540 for (unsigned int counter_two = 0;
541 counter_two < boundary_neighbor_cells.size();
542 ++counter_two)
543 {
544 if (counter_one > counter_two)
545 {
546 // Get the pair of cells
547 typename Triangulation<dim>::active_cell_iterator
548 cell_one = boundary_neighbor_cells[counter_one];
549 typename Triangulation<dim>::active_cell_iterator
550 cell_two = boundary_neighbor_cells[counter_two];
551
552 // Loop on all faces of the 2 cells
553 for (const auto &face_one : cell_one->face_iterators())
554 {
555 for (const auto &face_two :
556 cell_two->face_iterators())
557 {
558 // Check if faces are located at boundaries
559 if (face_one->at_boundary() &&
560 face_two->at_boundary())
561 {
562 unsigned int face_id_one, face_id_two;
563 face_id_one =
564 cell_one->face_iterator_to_index(face_one);
565 face_id_two =
566 cell_two->face_iterator_to_index(face_two);
567
568 // Check if faces are on a wall boundary
569 // (not outlet nor periodic)
570 bool face_one_is_wall, face_two_is_wall;
571 face_one_is_wall =
572 (std::find(outlet_boundaries.begin(),
573 outlet_boundaries.end(),
574 face_one->boundary_id()) ==
575 outlet_boundaries.end() &&
576 !cell_one->has_periodic_neighbor(
577 face_id_one));
578 face_two_is_wall =
579 (std::find(outlet_boundaries.begin(),
580 outlet_boundaries.end(),
581 face_two->boundary_id()) ==
582 outlet_boundaries.end() &&
583 !cell_two->has_periodic_neighbor(
584 face_id_two));
585
586 if (face_one_is_wall && face_two_is_wall)
587 {
588 // Get the normal vector of the first cell
589 Tensor<1, dim> normal_vector_one,
590 normal_vector_two;
591 normal_vector_one =
592 global_boundary_cells_information
593 .at(cell_one->face_index(face_id_one))
594 .normal_vector;
595
596 normal_vector_two =
597 global_boundary_cells_information
598 .at(cell_two->face_index(face_id_two))
599 .normal_vector;
600
601 // Check to see if the dot product of the
602 // two normal vectors is larger than a
603 // specified threshold (cos(45) = 0.707)
604 if (normal_vector_one *
605 normal_vector_two >
606 0.707)
607 {
608 if (display_diamond_cells_warning)
609 {
610 pcout
611 << std::endl
612 << "Warning: There are diamond-shaped cells in the input triangulation. It is strongly recommended to use a different triangulation without such cells. It should be mentioned that these cells are not detected if you have grid motion"
613 << std::endl;
614 display_diamond_cells_warning =
615 false;
616 }
617
618 if (check_diamond_cells &&
619 cell_with_boundary_line
620 ->is_locally_owned())
621 {
622 // If this condition is true, add
623 // this cell to
624 // boundary_cells_information.
625 // Since the key of
626 // boundary_cells_information is
627 // the boundary face id, and this
628 // cell does not have a boundary
629 // face, we add it to the
630 // boundary_cells_information with
631 // a negative key
632 int imaginary_face_id =
633 -1 * cell_two->face_index(
634 face_id_two);
635
636 // Update cell to the cell with
637 // boundary line
638 boundary_cells_info_struct<dim>
639 imaginary_information =
640 global_boundary_cells_information
641 .at(cell_two->face_index(
642 face_id_two));
643 imaginary_information.cell =
644 cell_with_boundary_line;
645 imaginary_information
646 .global_face_id =
647 imaginary_face_id;
648
649 boundary_cells_information.insert(
650 {imaginary_face_id,
651 imaginary_information});
652 }
653 }
654 }
655 }
656 }
657 }
658 }
659 }
660 }
661 }
662 }

665// This function finds the triangulation cells adjacent to the floating walls
666template <int dim>
667void
668BoundaryCellsInformation<dim>::find_boundary_cells_for_floating_walls(
669 const parallel::distributed::Triangulation<dim> &triangulation,
670 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
671 const double maximum_cell_diameter)
672{
673 // Reading floating wall properties
674 std::vector<Point<dim>> point_on_wall =
675 floating_wall_properties.points_on_walls;
676 std::vector<Tensor<1, dim>> wall_normal_vector =
677 floating_wall_properties.floating_walls_normal_vectors;
678 unsigned int floating_wall_number =
679 floating_wall_properties.floating_walls_number;
680
681 // Looping through floating walls
682 for (unsigned int floating_wall_id = 0;
683 floating_wall_id < floating_wall_number;
684 ++floating_wall_id)
685 {
686 // Looping through cells
687 for (const auto &cell : triangulation.active_cell_iterators())
688 {
689 // If the cell is owned by the processor
690 if (cell->is_locally_owned())
691 {
692 // Looping over vertices of each cell
693 for (unsigned int vertex = 0; vertex < cell->n_vertices();
694 ++vertex)
695 {
696 // Finding vertex-floating wall distance
697 Tensor<1, dim> connecting_vector =
698 cell->vertex(vertex) - point_on_wall[floating_wall_id];
699 double vertex_wall_distance =
700 connecting_vector * wall_normal_vector[floating_wall_id];
701
702 // If the distance is less than the largest cell size,
703 // it should be added to the
704 // boundary_cells_for_floating_walls
705 if (abs(vertex_wall_distance) < maximum_cell_diameter)
706 {
707 boundary_cells_for_floating_walls[floating_wall_id]
708 .insert(cell);
709 }
710 }
711 }
712 }
713 }

716template <int dim>
717void
718BoundaryCellsInformation<dim>::add_boundary_neighbors_of_boundary_cells(
719 const parallel::distributed::Triangulation<dim> &triangulation,
720 const std::vector<unsigned int> &outlet_boundaries,
721 std::map<int, boundary_cells_info_struct<dim>> &boundary_cells_information,
722 const std::map<int, boundary_cells_info_struct<dim>>
723 &global_boundary_cells_information)
724{
725 // Create a vector of a set of adjacent cells of all the vertices
726 std::vector<std::set<typename Triangulation<dim>::active_cell_iterator>>
727 v_to_c = GridTools::vertex_to_cell_map(triangulation);
728
729 for (auto &&boundary_cells_info :
730 boundary_cells_information | boost::adaptors::map_values)
731 {
732 // The boundary cell is local by definition
733 // This loop will check for all the cell neighbors located at vertices of
734 // a boundary cell and will add the proper information in the boundary
735 // cell list if faces are not at outlet nor periodic boundary.
736
737 // Iterate over the vertices of each boundary cell
738 for (auto vertex_id : GeometryInfo<dim>::vertex_indices())
739 {
740 // Iterate over the neighbors of each boundary cell having vertex_id
741 for (const auto &neighbor :
742 v_to_c[boundary_cells_info.cell->vertex_index(vertex_id)])
743 {
744 // Iterate over the faces of the neighbor cell
745 for (const auto &face : neighbor->face_iterators())
746 {
747 // Check if the face of the neighbor is located at boundary
748 if (face->at_boundary())
749 {
750 // Check if face is on a wall boundary (not outlet nor
751 // periodic)
752 unsigned int face_id =
753 neighbor->face_iterator_to_index(face);
754 bool face_is_wall =
755 (std::find(outlet_boundaries.begin(),
756 outlet_boundaries.end(),
757 face->boundary_id()) ==
758 outlet_boundaries.end() &&
759 !neighbor->has_periodic_neighbor(face_id));
760
761 if (face_is_wall)
762 {
763 // Get the boundary neighbor cell info
764 boundary_cells_info_struct<dim>
765 boundary_neighbor_information =
766 global_boundary_cells_information.at(
767 neighbor->face_index(face_id));
768
769 // Add the main boundary cell with the information
770 // (point and normal vector) of the neighbor boundary
771 // cell to the boundary_cells_information container.
772 // Note that since we may already have an element with
773 // the key of face_id (key of the
774 // boundary_cells_information map) in the
775 // boundary_cells_information, we add the new element
776 // with a unique key to create a unique id in the map.
777 // This unique key is generated using Cantor pairing
778 // function: unique_key = 0.5 * (a + b) * (a + b + 1)
779 // + b where a and b are global boundary face ids of
780 // the main boundary cell and the neighbor boundary
781 // cell.
782 int imaginary_face_id =
783 -0.5 *
784 (boundary_cells_info.global_face_id +
785 boundary_neighbor_information.global_face_id) *
786 (boundary_cells_info.global_face_id +
787 boundary_neighbor_information.global_face_id +
788 1) +
789 boundary_neighbor_information.global_face_id;
790
791 // Create a cell info object which is a copy of all
792 // the boundary neighbor information applied to the
793 // boundary cell & store in map with imaginary key.
794 boundary_cells_info_struct<dim> boundary_information =
795 boundary_neighbor_information;
796 boundary_information.cell = boundary_cells_info.cell;
797
798 boundary_cells_information.insert(
799 {imaginary_face_id, boundary_information});
800 }
801 }
802 }
803 }
804 }
805 }
806}
