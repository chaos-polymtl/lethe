19/**
20 * @brief Calculation of the constant rolling resistance torque using the
21 * information obtained from the fine search and physical properties.
22 *
23 * The model works as follows: (particle i, wall j)
24 * M_r = - mu_r * R_i * |F_n| * omega_hat
25 * omega_hat = omega_i / |omega_i|
26 *
27 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
28 *
29 * @param[in] particle_radius Particle radius.
30 * @param[in] particle_one_properties Properties of particle.
31 * @param[in] effective_rolling_friction_coefficient Effective_rolling friction
32 * coefficient
33 * @param[in] normal_force_norm Norm of the normal force.
34 *
35 * @return rolling_resistance_torque with constant resistance
36 */
37template <typename PropertiesIndex>
38inline Tensor<1, 3>
39constant_rolling_torque(const double particle_radius,
40 const ArrayView<const double> &particle_properties,
41 const double effective_rolling_friction_coefficient,
42 const double normal_force_norm)
43
44{
45 // Getting the angular velocity of the particle
46 Tensor<1, 3> particle_angular_velocity{
47 {particle_properties[PropertiesIndex::omega_x],
48 particle_properties[PropertiesIndex::omega_y],
49 particle_properties[PropertiesIndex::omega_z]}};
50
51 // Calculation of particle-wall angular velocity (norm of the
52 // particle angular velocity)
53 const double omega_value = particle_angular_velocity.norm();
54 Tensor<1, 3> particle_wall_angular_velocity =
55 particle_angular_velocity / (omega_value + DBL_MIN);
56
57 // Calculation of rolling resistance torque
58 Tensor<1, 3> rolling_resistance_torque =
59 -effective_rolling_friction_coefficient * particle_radius *
60 normal_force_norm * particle_wall_angular_velocity;
61
62 return rolling_resistance_torque;
63}

65/**
66 * @brief Calculation of the viscous rolling resistance torque using the
67 * information obtained from the fine search and physical properties.
68 *
69 * The model works as follows: (particle i, wall j)
70 * M_r = - mu_r * R_i * |F_n| * |V_omega| * omega_hat
71 * omega_hat = omega_i / |omega_i|
72 * V_omega = omega_i Ã— (R_p * n_ji)
73 *
74 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
75 *
76 * @param[in] particle_radius Particle radius.
77 * @param[in] particle_properties Properties of particle.
78 * @param[in] effective_rolling_friction_coefficient Effective_rolling friction
79 * coefficient
80 * @param[in] normal_force_norm Norm of the normal force.
81 * @param[in] normal_unit_vector Normal unit vector.
82 *
83 * @return rolling_resistance_torque with viscous resistance
84 */
85
86template <typename PropertiesIndex>
87inline Tensor<1, 3>
88viscous_rolling_torque(const double particle_radius,
89 const ArrayView<const double> &particle_properties,
90 const double effective_rolling_friction_coefficient,
91 const double normal_force_norm,
92 const Tensor<1, 3> &normal_unit_vector)
93
94{
95 // Getting the angular velocity of particle
96 Tensor<1, 3> particle_angular_velocity{
97 {particle_properties[PropertiesIndex::omega_x],
98 particle_properties[PropertiesIndex::omega_y],
99 particle_properties[PropertiesIndex::omega_z]}};
100
101 // Calculation of particle-wall angular velocity (norm of the
102 // particle angular velocity)
103 const double omega_value = particle_angular_velocity.norm();
104 Tensor<1, 3> particle_wall_angular_velocity =
105 particle_angular_velocity / (omega_value + DBL_MIN);
106
107 Tensor<1, 3> v_omega = cross_product_3d(particle_angular_velocity,
108 particle_radius * normal_unit_vector);
109
110 // Calculation of rolling resistance torque
111 Tensor<1, 3> rolling_resistance_torque =
112 -effective_rolling_friction_coefficient * particle_radius *
113 normal_force_norm * v_omega.norm() * particle_wall_angular_velocity;
114
115 return rolling_resistance_torque;
116}

118/**
119 * @brief Calculation of the EPSD rolling resistance torque using the
120 * information obtained from the fine search and physical properties.
121 *
122 * @tparam dim An integer that denotes the dimension of the space in which
123 * the problem is solved.
124 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
125 *
126 * @param[in] particle_radius Particle radius.
127 * @param[in] particle_properties Properties of particle.
128 * @param[in] effective_rolling_viscous_damping_coefficient
129 * Effective rolling viscous damping
130 * @param[in] effective_rolling_friction_coefficient Effective_rolling friction
131 * coefficient
132 * @param[in] f_coefficient Model parameter for the viscous damping.
133 * @param[in] normal_force_norm Norm of the normal force.
134 * @param[in] dt DEM time step.
135 * @param[in] normal_spring_constant normal contact stiffness constant.
136 * @param[in] normal_unit_vector Normal unit vector between particles in
137 * contact.
138 * @param[in,out] cumulative_rolling_resistance_spring_torque Cumulative
139 * rolling resistance torque for the EPSD rolling resistance model.
140 *
141 * @return rolling_resistance_torque with EPSD resistance
142 */
143
144template <int dim, typename PropertiesIndex>
145inline Tensor<1, 3>
146epsd_rolling_torque(const double particle_radius,
147 const ArrayView<const double> &particle_properties,
148 const double effective_rolling_friction_coefficient,
149 const double effective_rolling_viscous_damping_coefficient,
150 const double f_coefficient,
151 const double normal_force_norm,
152 const double dt,
153 const double normal_spring_constant,
154 const Tensor<1, 3> &normal_unit_vector,
155 Tensor<1, 3> &cumulative_rolling_resistance_spring_torque)
156
157{
158 // Useful value used more than once:
159 // mu_r * R
160 const double mu_r_times_R =
161 effective_rolling_friction_coefficient * particle_radius;
162
163 // Getting the angular velocity of particle
164 Tensor<1, 3> particle_angular_velocity{
165 {particle_properties[PropertiesIndex::omega_x],
166 particle_properties[PropertiesIndex::omega_y],
167 particle_properties[PropertiesIndex::omega_z]}};
168
169 // Non-collinear component of the relative velocity.
170 const Tensor<1, 3> omega_perpendicular =
171 particle_angular_velocity -
172 scalar_product(particle_angular_velocity, normal_unit_vector) *
173 normal_unit_vector;
174
175 // Delta theta : incremental relative rotation between i and j
176 const Tensor<1, 3> delta_theta = dt * omega_perpendicular;
177
178 // Rolling stiffness
179 const double K_r = [&]() {
180 if constexpr (dim == 3)
181 return 2.25 * normal_spring_constant *
182 Utilities::fixed_power<2>(mu_r_times_R);
183 else
184 return 3. * normal_spring_constant *
185 Utilities::fixed_power<2>(mu_r_times_R);
186 }();
187
188 // Update the spring torque
189 cumulative_rolling_resistance_spring_torque -= K_r * delta_theta;
190
191 // Limiting spring torque
192 const double M_r_max = mu_r_times_R * normal_force_norm;
193
194 const double rolling_resistance_spring_torque_norm =
195 cumulative_rolling_resistance_spring_torque.norm();
196
197 // 1.4 * m_i * (R_i)^2
198 // Total inertia of particle i evaluated at its surface. Mass of the
199 // wall is considered infinite, thus I_i is equal to I_e. (Effective
200 // inertia)
201 const double I_e = 1.4 * particle_properties[PropertiesIndex::mass] *
202 Utilities::fixed_power<2>(particle_radius);
203
204 // C_r_crit = 2. * sqrt(I_r * K_r)
205 // C_r = eta_r * C_r_crit
206 const double C_r =
207 effective_rolling_viscous_damping_coefficient * 2. * sqrt(I_e * K_r);
208
209 // Similarly to the coulomb limit, the spring torque must be decrease to the
210 // limit value if it exceeds the limiting spring toque.
211 if (rolling_resistance_spring_torque_norm > M_r_max)
212 {
213 cumulative_rolling_resistance_spring_torque =
214 cumulative_rolling_resistance_spring_torque *
215 (M_r_max / rolling_resistance_spring_torque_norm);
216
217 // If the limiting spring torque is exceeded, the damping is multiplied by
218 // the f_coefficient which should be between 0 and 1.
219
220 return cumulative_rolling_resistance_spring_torque -
221 f_coefficient * C_r * omega_perpendicular;
222 }
223 // Minus sign has been verified
224 return cumulative_rolling_resistance_spring_torque -
225 C_r * omega_perpendicular;
226}
