9template <int dim,
10 typename pairs_structure,
11 typename candidates_structure,
12 ContactType contact_type>
13void
14update_fine_search_candidates(pairs_structure &pairs_in_contact,
15 candidates_structure &contact_candidates)
16{
17 for (auto pairs_in_contact_iterator = pairs_in_contact.begin();
18 pairs_in_contact_iterator != pairs_in_contact.end();)
19 {
20 // Get the current particle id from fine search history
21 auto particle_id = pairs_in_contact_iterator->first;
22
23 // Get the reference to all the object (particle/wall/face) contact
24 // candidates of the current particle from the new broad search
25 auto contact_candidate_element = contact_candidates.find(particle_id);
26
27 // Get adjacent object (particle/wall/face) content of the current
28 // particle (from the fine search history)
29 auto adjacent_pairs_content = &pairs_in_contact_iterator->second;
30
31 // Loop over all the adjacent objects
32 for (auto adjacent_map_iterator = adjacent_pairs_content->begin();
33 adjacent_map_iterator != adjacent_pairs_content->end();)
34 {
35 // Get the object (2nd particle/wall/face) id in the history list
36 auto object_id = adjacent_map_iterator->first;
37
38 // The handling of the update of the fine search candidate list is
39 // different when the contact is particle-particle compared to
40 // particle-wall
41 if constexpr (contact_type == ContactType::local_particle_particle ||
42 contact_type == ContactType::ghost_particle_particle ||
43 contact_type ==
44 ContactType::local_periodic_particle_particle ||
45 contact_type ==
46 ContactType::ghost_periodic_particle_particle ||
47 contact_type ==
48 ContactType::ghost_local_periodic_particle_particle)
49 {
50 if (contact_candidate_element != contact_candidates.end())
51 {
52 // Check if the adjacent particle from history is a particle
53 // candidate of the current particle and get the iterator to
54 // that object if so
55 auto search_iterator =
56 std::find(contact_candidate_element->second.begin(),
57 contact_candidate_element->second.end(),
58 object_id);
59
60 // Particle-particle pairs are removed from the list of the
61 // new broad search candidates or from the list of the history
62 // of the fine search candidate list depending on the last
63 // checks
64 if (search_iterator !=
65 contact_candidate_element->second.end())
66 {
67 // The current 2nd is a candidate to the current particle.
68 // The 2nd particle is then removed from the new broad
69 // search particle-particle contact candidates list but
70 // kept as history in the fine search list
71 contact_candidate_element->second.erase(search_iterator);
72
73 ++adjacent_map_iterator;
74
75 // To prevent searching through the contact_candidates a
76 // second time if there is a deletion in the first search,
77 // we continue here instead of use an if / else if / else
78 // statement
79 continue;
80 }
81 }
82
83 // Get the reference to all the 2nd particle contact
84 // candidates of the current particle from the new broad search
85 auto object_contact_candidates =
86 contact_candidates.find(object_id);
87
88 if (object_contact_candidates != contact_candidates.end())
89 {
90 // Check if the current particle is a particle candidate
91 // of the adjacent particle in history
92 auto search_object_to_particle_iterator =
93 std::find(object_contact_candidates->second.begin(),
94 object_contact_candidates->second.end(),
95 particle_id);
96
97 if (search_object_to_particle_iterator !=
98 object_contact_candidates->second.end())
99 {
100 if constexpr (contact_type ==
101 ContactType::local_particle_particle ||
102 contact_type ==
103 ContactType::
104 local_periodic_particle_particle)
105 {
106 // The current particle from history list is still a
107 // candidate to the 2nd particle and the contact is
108 // local The 2nd particle is then removed from the new
109 // broad search particle-particle contact candidates
110 // list but kept as history in the fine search list
111 object_contact_candidates->second.erase(
112 search_object_to_particle_iterator);
113 ++adjacent_map_iterator;
114 continue;
115 }
116
117 if constexpr (
118 contact_type == ContactType::ghost_particle_particle ||
119 contact_type ==
120 ContactType::ghost_periodic_particle_particle ||
121 contact_type ==
122 ContactType::ghost_local_periodic_particle_particle)
123 {
124 // The current particle from history list is still a
125 // candidate to the 2nd particle and the contact is
126 // not local The 2nd particle is then removed from the
127 // history in the fine search list particle-particle
128 // contact candidates list but kept in new broad
129 // search list
130 adjacent_map_iterator = adjacent_pairs_content->erase(
131 adjacent_map_iterator);
132 continue;
133 }
134 }
135 }
136
137 // Current particles are no longer contact candidates.
138 // The particle is then removed as a candidate for fine
139 // search contact candidates list
140 adjacent_map_iterator =
141 adjacent_pairs_content->erase(adjacent_map_iterator);
142 }
143
144 if constexpr (contact_type == ContactType::particle_wall ||
145 contact_type == ContactType::particle_floating_wall ||
146 contact_type == ContactType::particle_floating_mesh)
147 {
148 if (contact_candidate_element != contact_candidates.end())
149 {
150 // Check if the wall/face is a candidate of the current
151 // particle and get the iterator to that object if so
152 auto search_iterator =
153 contact_candidate_element->second.find(object_id);
154
155 // Particle-wall/face pair is removed from the list of the new
156 // broad search candidates or from the list of the history of
157 // the fine search candidate list depending on the last check
158 if (search_iterator !=
159 contact_candidate_element->second.end())
160 {
161 // Current wall/face (from new broad search) is a
162 // candidate to the current particle. The wall/face is
163 // then removed from the new broad search
164 // particle-wall/face contact candidates list but kept as
165 // history in the fine search list
166 contact_candidate_element->second.erase(search_iterator);
167 ++adjacent_map_iterator;
168 continue;
169 }
170 }
171
172 // Current wall/face (from new broad search) is no longer a
173 // candidate to the current particle. The wall/face is then
174 // removed as a candidate for fine search contact candidates
175 // list
176 adjacent_map_iterator =
177 adjacent_pairs_content->erase(adjacent_map_iterator);
178 }
179 }
180
181 // If there are still particle/wall/face in the adjacent_pairs_content
182 // then the pairs_in_contact_iterator remains in memory
183 if (adjacent_pairs_content->size() > 0)
184 ++pairs_in_contact_iterator;
185
186 // Otherwise it is deleted. This is necessary to prevent memory inflation.
187 // If this is not done, pairs_in_contact keeps on growing until it reaches
188 // the total number of particles in the simulation.
189 else
190 pairs_in_contact_iterator =
191 pairs_in_contact.erase(pairs_in_contact_iterator);
192 }
193}
