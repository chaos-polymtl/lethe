85 template <int dim>
86 void
87 FluidDynamicsVANS<dim>::read_dem()
88 {
89  std::string prefix =
90  this->cfd_dem_simulation_parameters.void_fraction->dem_file_name;
91 
92  // Load checkpoint controller
93  std::string checkpoint_controller_object_filename =
94  prefix + ".checkpoint_controller";
95  std::ifstream iss_checkpoint_controller_obj(
96  checkpoint_controller_object_filename);
97  boost::archive::text_iarchive ia_checkpoint_controller_obj(
98  iss_checkpoint_controller_obj, boost::archive::no_header);
99 
100  unsigned int checkpoint_id;
101  ia_checkpoint_controller_obj >> checkpoint_id;
102 
103  // New prefix for the remaining files
104  prefix = prefix + "_" + Utilities::int_to_string(checkpoint_id);
105 
106  // Gather particle serialization information
107  std::string particle_filename = prefix + ".particles";
108  std::ifstream input(particle_filename.c_str());
109  AssertThrow(input, ExcFileNotOpen(particle_filename));
110 
111  std::string buffer;
112  std::getline(input, buffer);
113  std::istringstream iss(buffer);
114  boost::archive::text_iarchive ia(iss, boost::archive::no_header);
115 
116  // Create a temporary particle_handler with DEM properties
117  Particles::ParticleHandler<dim> temporary_particle_handler(
118  *this->triangulation,
119  particle_mapping,
120  DEM::get_number_properties<DEM::DEMProperties::PropertiesIndex>());
121 
122 
123  ia >> temporary_particle_handler;
124 
125  const std::string filename = prefix + ".triangulation";
126  std::ifstream in(filename.c_str());
127  if (!in)
128  AssertThrow(false,
129  ExcMessage(
130  std::string(
131  "You are trying to restart a previous computation, "
132  "but the restart file <") +
133  filename + "> does not appear to exist!"));
134 
135  if (auto parallel_triangulation =
136  dynamic_cast<parallel::distributed::Triangulation<dim> *>(
137  &*this->triangulation))
138  {
139  try
140  {
141  this->triangulation->load(filename.c_str());
142 
143  // Deserialize particles have the triangulation has been read
144  temporary_particle_handler.deserialize();
145  }
146  catch (...)
147  {
148  AssertThrow(false,
149  ExcMessage("Cannot open snapshot mesh file or read the"
150  "triangulation stored there."));
151  }
152 
153  // Fill the existing particle handler using the temporary one
154  // This is done during the dynamic cast for the convert_particle_handler
155  // function which requires a pararallel::distributed::triangulation
156  convert_particle_handler<dim,
157  DEM::DEMProperties::PropertiesIndex,
158  DEM::CFDDEMProperties::PropertiesIndex>(
159  *parallel_triangulation, temporary_particle_handler, particle_handler);
160  }
161  else
162  {
163  throw std::runtime_error(
164  "VANS equations currently do not support "
165  "triangulations other than parallel::distributed");
166  }
167 }

206 template <int dim>
207 void
208 FluidDynamicsVANS<dim>::setup_assemblers()
209 {
210  this->assemblers.clear();
211  particle_fluid_assemblers.clear();
212 
213  if (this->check_existance_of_bc(
214  BoundaryConditions::BoundaryType::function_weak))
215  {
216  this->assemblers.push_back(
217  std::make_shared<WeakDirichletBoundaryCondition<dim>>(
218  this->simulation_control,
219  this->simulation_parameters.boundary_conditions));
220  }
221  if (this->check_existance_of_bc(
222  BoundaryConditions::BoundaryType::partial_slip))
223  {
224  this->assemblers.push_back(
225  std::make_shared<PartialSlipDirichletBoundaryCondition<dim>>(
226  this->simulation_control,
227  this->simulation_parameters.boundary_conditions));
228  }
229  if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::outlet))
230  {
231  this->assemblers.push_back(std::make_shared<OutletBoundaryCondition<dim>>(
232  this->simulation_control,
233  this->simulation_parameters.boundary_conditions));
234  }
235  if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::pressure))
236  {
237  this->assemblers.push_back(
238  std::make_shared<PressureBoundaryCondition<dim>>(
239  this->simulation_control,
240  this->simulation_parameters.boundary_conditions));
241  }
242 
243  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_force == true)
244  {
245  // Particle_Fluid Interactions Assembler
246  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
247  Parameters::DragModel::difelice)
248  {
249  // DiFelice Model drag Assembler
250  particle_fluid_assemblers.push_back(
251  std::make_shared<VANSAssemblerDiFelice<dim>>(
252  this->cfd_dem_simulation_parameters.cfd_dem));
253  }
254 
255  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
256  Parameters::DragModel::rong)
257  {
258  // Rong Model drag Assembler
259  particle_fluid_assemblers.push_back(
260  std::make_shared<VANSAssemblerRong<dim>>(
261  this->cfd_dem_simulation_parameters.cfd_dem));
262  }
263 
264  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
265  Parameters::DragModel::dallavalle)
266  {
267  // Dallavalle Model drag Assembler
268  particle_fluid_assemblers.push_back(
269  std::make_shared<VANSAssemblerDallavalle<dim>>(
270  this->cfd_dem_simulation_parameters.cfd_dem));
271  }
272 
273  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
274  Parameters::DragModel::kochhill)
275  {
276  // Koch and Hill Model drag Assembler
277  particle_fluid_assemblers.push_back(
278  std::make_shared<VANSAssemblerKochHill<dim>>(
279  this->cfd_dem_simulation_parameters.cfd_dem));
280  }
281  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
282  Parameters::DragModel::beetstra)
283  {
284  // Beetstra drag model assembler
285  particle_fluid_assemblers.push_back(
286  std::make_shared<VANSAssemblerBeetstra<dim>>(
287  this->cfd_dem_simulation_parameters.cfd_dem));
288  }
289  if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
290  Parameters::DragModel::gidaspow)
291  {
292  // Gidaspow Model drag Assembler
293  particle_fluid_assemblers.push_back(
294  std::make_shared<VANSAssemblerGidaspow<dim>>(
295  this->cfd_dem_simulation_parameters.cfd_dem));
296  }
297  }
298 
299  if (this->cfd_dem_simulation_parameters.cfd_dem.saffman_lift_force == true)
300  // Saffman Mei Lift Force Assembler
301  particle_fluid_assemblers.push_back(
302  std::make_shared<VANSAssemblerSaffmanMei<dim>>(
303  this->cfd_dem_simulation_parameters.dem_parameters
304  .lagrangian_physical_properties));
305 
306  if (this->cfd_dem_simulation_parameters.cfd_dem.magnus_lift_force == true)
307  // Magnus Lift Force Assembler
308  particle_fluid_assemblers.push_back(
309  std::make_shared<VANSAssemblerMagnus<dim>>(
310  this->cfd_dem_simulation_parameters.dem_parameters
311  .lagrangian_physical_properties));
312 
313  if (this->cfd_dem_simulation_parameters.cfd_dem.rotational_viscous_torque ==
314  true)
315  // Viscous Torque Assembler
316  particle_fluid_assemblers.push_back(
317  std::make_shared<VANSAssemblerViscousTorque<dim>>(
318  this->cfd_dem_simulation_parameters.dem_parameters
319  .lagrangian_physical_properties));
320 
321  if (this->cfd_dem_simulation_parameters.cfd_dem.vortical_viscous_torque ==
322  true)
323  // Vortical Torque Assembler
324  particle_fluid_assemblers.push_back(
325  std::make_shared<VANSAssemblerVorticalTorque<dim>>(
326  this->cfd_dem_simulation_parameters.dem_parameters
327  .lagrangian_physical_properties));
328 
329  if (this->cfd_dem_simulation_parameters.cfd_dem.buoyancy_force == true)
330  // Buoyancy Force Assembler
331  particle_fluid_assemblers.push_back(
332  std::make_shared<VANSAssemblerBuoyancy<dim>>(
333  this->cfd_dem_simulation_parameters.dem_parameters
334  .lagrangian_physical_properties));
335 
336  if (this->cfd_dem_simulation_parameters.cfd_dem.pressure_force == true)
337  // Pressure Force
338  particle_fluid_assemblers.push_back(
339  std::make_shared<VANSAssemblerPressureForce<dim>>(
340  this->cfd_dem_simulation_parameters.cfd_dem));
341 
342  if (this->cfd_dem_simulation_parameters.cfd_dem.shear_force == true)
343  // Shear Force
344  particle_fluid_assemblers.push_back(
345  std::make_shared<VANSAssemblerShearForce<dim>>(
346  this->cfd_dem_simulation_parameters.cfd_dem));
347 
348  // Time-stepping schemes
349  if (is_bdf(this->simulation_control->get_assembly_method()))
350  {
351  this->assemblers.push_back(std::make_shared<VANSAssemblerBDF<dim>>(
352  this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
353  }
354 
355  // Fluid_Particle Interactions Assembler
356  this->assemblers.push_back(std::make_shared<VANSAssemblerFPI<dim>>(
357  this->cfd_dem_simulation_parameters.cfd_dem));
358 
359  // The core assembler should always be the last assembler to be called
360  // in the stabilized formulation as to have all strong residual and
361  // jacobian stored. Core assembler
362  if (this->cfd_dem_simulation_parameters.cfd_dem.vans_model ==
363  Parameters::VANSModel::modelA)
364  this->assemblers.push_back(std::make_shared<VANSAssemblerCoreModelA<dim>>(
365  this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
366 
367  if (this->cfd_dem_simulation_parameters.cfd_dem.vans_model ==
368  Parameters::VANSModel::modelB)
369  this->assemblers.push_back(std::make_shared<VANSAssemblerCoreModelB<dim>>(
370  this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
371 }

377  {
378  TimerOutput::Scope t(this->computing_timer, "Assemble matrix");
379  this->system_matrix = 0;
380 
381  setup_assemblers();
382 
383  auto scratch_data = NavierStokesScratchData<dim>(

385  this->simulation_parameters.physical_properties_manager,

394 
395  scratch_data.enable_particle_fluid_interactions(

397  this->cfd_dem_simulation_parameters.cfd_dem.interpolated_void_fraction);
398 
399  WorkStream::run(
400  this->dof_handler.begin_active(),
401  this->dof_handler.end(),
402  *this,
403  &FluidDynamicsVANS::assemble_local_system_matrix,
404  &FluidDynamicsVANS::copy_local_matrix_to_global_matrix,
405  scratch_data,
406  StabilizedMethodsTensorCopyData<dim>(this->fe->n_dofs_per_cell(),
407  this->cell_quadrature->size()));
408  this->system_matrix.compress(VectorOperation::add);
409  }

414 FluidDynamicsVANS<dim>::assemble_local_system_matrix(

442  scratch_data.reinit_particle_fluid_interactions(
443  cell,
444  this->evaluation_point,
445  this->previous_solutions[0],
446  this->void_fraction_manager.void_fraction_locally_relevant,
447  particle_handler,
448  this->dof_handler,
449  void_fraction_manager.dof_handler);
450 
451  scratch_data.calculate_physical_properties();
452  copy_data.reset();

454  for (auto &pf_assembler : particle_fluid_assemblers)

458 
459  for (auto &assembler : this->assemblers)

601 template <int dim>
602 void
603 FluidDynamicsVANS<dim>::monitor_mass_conservation()
604 {
605  QGauss<dim> quadrature_formula(this->number_quadrature_points);
606 
607  FEValues<dim> fe_values(*this->mapping,
608  *this->fe,
609  quadrature_formula,
610  update_values | update_quadrature_points |
611  update_JxW_values | update_gradients |
612  update_hessians);
613 
614  FEValues<dim> fe_values_void_fraction(*this->mapping,
615  *this->void_fraction_manager.fe,
616  quadrature_formula,
617  update_values |
618  update_quadrature_points |
619  update_JxW_values | update_gradients);
620 
621  const FEValuesExtractors::Vector velocities(0);
622 
623  const unsigned int n_q_points = quadrature_formula.size();
624 
625  std::vector<double> present_void_fraction_values(n_q_points);
626  std::vector<Tensor<1, dim>> present_void_fraction_gradients(n_q_points);
627  // Values at previous time step for transient schemes for void
628  // fraction
629  std::vector<double> p1_void_fraction_values(n_q_points);
630  std::vector<double> p2_void_fraction_values(n_q_points);
631  std::vector<double> p3_void_fraction_values(n_q_points);
632 
633  std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
634  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
635 
636  // Values of the force function, the last component of which is the external
637  // mass source
638  std::vector<Vector<double>> rhs_force(n_q_points, Vector<double>(dim + 1));
639 
640  double continuity = 0;
641  double max_local_continuity = 0;
642  double local_mass_source = 0;
643 
644  std::vector<double> time_steps_vector =
645  this->simulation_control->get_time_steps_vector();
646 
647  const auto scheme = this->simulation_control->get_assembly_method();
648  const Vector<double> &bdf_coefs =
649  this->simulation_control->get_bdf_coefficients();
650 
651  for (const auto &cell : this->dof_handler.active_cell_iterators())
652  {
653  if (cell->is_locally_owned())
654  {
655  typename DoFHandler<dim>::active_cell_iterator void_fraction_cell(
656  &(*this->triangulation),
657  cell->level(),
658  cell->index(),
659  &this->void_fraction_manager.dof_handler);
660  fe_values_void_fraction.reinit(void_fraction_cell);
661 
662  // Gather void fraction (values, gradient)
663  fe_values_void_fraction.get_function_values(
664  void_fraction_manager.void_fraction_locally_relevant,
665  present_void_fraction_values);
666  fe_values_void_fraction.get_function_gradients(
667  void_fraction_manager.void_fraction_locally_relevant,
668  present_void_fraction_gradients);
669 
670  fe_values.reinit(cell);
671 
672  // Gather velocity (values and gradient)
673  auto &evaluation_point = this->evaluation_point;
674  fe_values[velocities].get_function_values(evaluation_point,
675  present_velocity_values);
676  fe_values[velocities].get_function_gradients(
677  evaluation_point, present_velocity_gradients);
678 
679  // Gather the external mass source at the quadrature point location
680  std::vector<Point<dim>> quadrature_points =
681  fe_values.get_quadrature_points();
682 
683  this->forcing_function->vector_value_list(quadrature_points,
684  rhs_force);
685 
686  // Gather the previous time steps depending on the number of
687  // stages of the time integration scheme for the void fraction
688 
689  if (scheme !=
690  Parameters::SimulationControl::TimeSteppingMethod::steady)
691  {
692  fe_values_void_fraction.get_function_values(
693  void_fraction_manager.previous_void_fraction[0],
694  p1_void_fraction_values);
695 
696  if (scheme ==
697  Parameters::SimulationControl::TimeSteppingMethod::bdf2)
698  fe_values_void_fraction.get_function_values(
699  void_fraction_manager.previous_void_fraction[1],
700  p2_void_fraction_values);
701 
702  if (scheme ==
703  Parameters::SimulationControl::TimeSteppingMethod::bdf3)
704  fe_values_void_fraction.get_function_values(
705  void_fraction_manager.previous_void_fraction[2],
706  p3_void_fraction_values);
707  }
708 
709  local_mass_source = 0;
710 
711  for (unsigned int q = 0; q < n_q_points; ++q)
712  {
713  // Calculate external mass source
714  const unsigned int component_mass =
715  this->fe->system_to_component_index(dim).first;
716  double external_mass_source = rhs_force[q](component_mass);
717 
718  // Calculate the divergence of the velocity
719  const double present_velocity_divergence =
720  trace(present_velocity_gradients[q]);
721 
722 
723  // Evaluation of global mass conservation
724  local_mass_source =
725  (present_velocity_values[q] *
726  present_void_fraction_gradients[q] +
727  present_void_fraction_values[q] * present_velocity_divergence -
728  external_mass_source) *
729  fe_values_void_fraction.JxW(q);
730 
731  if (scheme ==
732  Parameters::SimulationControl::TimeSteppingMethod::bdf1 ||
733  scheme == Parameters::SimulationControl::TimeSteppingMethod::
734  steady_bdf)
735  local_mass_source +=
736  (bdf_coefs[0] * present_void_fraction_values[q] +
737  bdf_coefs[1] * p1_void_fraction_values[q]) *
738  fe_values_void_fraction.JxW(q);
739 
740  if (scheme ==
741  Parameters::SimulationControl::TimeSteppingMethod::bdf2)
742  local_mass_source +=
743  (bdf_coefs[0] * present_void_fraction_values[q] +
744  bdf_coefs[1] * p1_void_fraction_values[q] +
745  bdf_coefs[2] * p2_void_fraction_values[q]) *
746  fe_values_void_fraction.JxW(q);
747 
748  if (scheme ==
749  Parameters::SimulationControl::TimeSteppingMethod::bdf3)
750  local_mass_source +=
751  (bdf_coefs[0] * present_void_fraction_values[q] +
752  bdf_coefs[1] * p1_void_fraction_values[q] +
753  bdf_coefs[2] * p2_void_fraction_values[q] +
754  bdf_coefs[3] * p3_void_fraction_values[q]) *
755  fe_values_void_fraction.JxW(q);
756 
757  continuity += local_mass_source;
758  }
759 
760  max_local_continuity =
761  std::max(max_local_continuity, abs(local_mass_source));
762  }
763  }
764 
765  continuity = Utilities::MPI::sum(continuity, this->mpi_communicator);
766  max_local_continuity =
767  Utilities::MPI::max(max_local_continuity, this->mpi_communicator);
768 
769 
770  this->pcout << std::setprecision(
771  this->simulation_control->get_log_precision());
772 
773  this->pcout << "Global continuity equation error: " << continuity << " s^-1"
774  << std::endl;
775  this->pcout << "Max local continuity error: " << max_local_continuity
776  << " s^-1" << std::endl;
777 }

786  this->cfd_dem_simulation_parameters.cfd_parameters.manifolds_parameters,
787  this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
788  .restart ||

790  this->cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions);

793  this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
794  .restart == false)
795  read_dem();
796 
797  this->setup_dofs();
798 
799  this->set_initial_condition(
800  this->cfd_dem_simulation_parameters.cfd_parameters.initial_condition->type,
801  this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
802  .restart);
803 
804  particle_handler.exchange_ghost_particles(true);
805 
806  while (this->simulation_control->integrate())
807  {
808  this->simulation_control->print_progression(this->pcout);
809 
810  // We allow the physics to update their boundary conditions
811  // according to their own parameters
812  this->update_boundary_conditions();

815  this->dynamic_flow_control();
816

836
