13// Constructor for class FluidDynamicsVANS
14template <int dim>
15FluidDynamicsVANS<dim>::FluidDynamicsVANS(
16 CFDDEMSimulationParameters<dim> &nsparam)
17 : FluidDynamicsMatrixBased<dim>(nsparam.cfd_parameters)
18 , cfd_dem_simulation_parameters(nsparam)
19 , particle_mapping(1)
20 , particle_handler(*this->triangulation,
21 particle_mapping,
22 DEM::CFDDEMProperties::n_properties)
23 , particle_projector(
24 &(*this->triangulation),
25 nsparam.void_fraction,
26 this->cfd_dem_simulation_parameters.cfd_parameters.linear_solver.at(
27 PhysicsID::void_fraction),
28 &particle_handler,
29 this->cfd_dem_simulation_parameters.cfd_parameters.fem_parameters
30 .void_fraction_order,
31 this->cfd_dem_simulation_parameters.cfd_parameters.mesh.simplex,
32 this->pcout)
33 , has_periodic_boundaries(false)
34{
35 unsigned int n_pbc = 0;
36 for (auto const &[id, type] :
37 cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions.type)
38 {
39 if (type == BoundaryConditions::BoundaryType::periodic)
40 {
41 if (n_pbc++ > 1)
42 {
43 throw std::runtime_error(
44 "GLS VANS solver does not support more than one periodic boundary condition.");
45 }
46 else
47 {
48 has_periodic_boundaries = true;
49 }
50 }
51 }
52}

82template <int dim>
83void
84FluidDynamicsVANS<dim>::read_dem()
85{
86 std::string prefix =
87 this->cfd_dem_simulation_parameters.void_fraction->dem_file_name;
88
89 // Load checkpoint controller
90 std::string checkpoint_controller_object_filename =
91 prefix + ".checkpoint_controller";
92 std::ifstream iss_checkpoint_controller_obj(
93 checkpoint_controller_object_filename);
94 boost::archive::text_iarchive ia_checkpoint_controller_obj(
95 iss_checkpoint_controller_obj, boost::archive::no_header);
96
97 unsigned int checkpoint_id;
98 ia_checkpoint_controller_obj >> checkpoint_id;
99
100 // New prefix for the remaining files
101 prefix = prefix + "_" + Utilities::int_to_string(checkpoint_id);
102
103 // Gather particle serialization information
104 std::string particle_filename = prefix + ".particles";
105 std::ifstream input(particle_filename.c_str());
106 AssertThrow(input, ExcFileNotOpen(particle_filename));
107
108 std::string buffer;
109 std::getline(input, buffer);
110 std::istringstream iss(buffer);
111 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
112
113 // Create a temporary particle_handler with DEM properties
114 Particles::ParticleHandler<dim> temporary_particle_handler(
115 *this->triangulation, particle_mapping, DEM::DEMProperties::n_properties);
116
117 ia >> temporary_particle_handler;
118
119 const std::string filename = prefix + ".triangulation";
120 std::ifstream in(filename.c_str());
121 if (!in)
122 AssertThrow(false,
123 ExcMessage(
124 std::string(
125 "You are trying to restart a previous computation, "
126 "but the restart file <") +
127 filename + "> does not appear to exist!"));
128
129 if (auto parallel_triangulation =
130 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
131 &*this->triangulation))
132 {
133 try
134 {
135 this->triangulation->load(filename.c_str());
136
137 // Deserialize particles have the triangulation has been read
138 temporary_particle_handler.deserialize();
139 }
140 catch (...)
141 {
142 AssertThrow(false,
143 ExcMessage("Cannot open snapshot mesh file or read the"
144 "triangulation stored there."));
145 }
146
147 // Fill the existing particle handler using the temporary one
148 // This is done during the dynamic cast for the convert_particle_handler
149 // function which requires a pararallel::distributed::triangulation
150 convert_particle_handler<dim,
151 DEM::DEMProperties::PropertiesIndex,
152 DEM::CFDDEMProperties::PropertiesIndex>(
153 *parallel_triangulation, temporary_particle_handler, particle_handler);
154 }
155 else
156 {
157 throw std::runtime_error(
158 "VANS equations currently do not support "
159 "triangulations other than parallel::distributed");
160 }
161}

200template <int dim>
201void
202FluidDynamicsVANS<dim>::setup_assemblers()
203{
204 this->assemblers.clear();
205 particle_fluid_assemblers.clear();
206
207 if (this->check_existance_of_bc(
208 BoundaryConditions::BoundaryType::function_weak))
209 {
210 this->assemblers.push_back(
211 std::make_shared<WeakDirichletBoundaryCondition<dim>>(
212 this->simulation_control,
213 this->simulation_parameters.boundary_conditions));
214 }
215 if (this->check_existance_of_bc(
216 BoundaryConditions::BoundaryType::partial_slip))
217 {
218 this->assemblers.push_back(
219 std::make_shared<PartialSlipDirichletBoundaryCondition<dim>>(
220 this->simulation_control,
221 this->simulation_parameters.boundary_conditions));
222 }
223 if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::outlet))
224 {
225 this->assemblers.push_back(std::make_shared<OutletBoundaryCondition<dim>>(
226 this->simulation_control,
227 this->simulation_parameters.boundary_conditions));
228 }
229 if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::pressure))
230 {
231 this->assemblers.push_back(
232 std::make_shared<PressureBoundaryCondition<dim>>(
233 this->simulation_control,
234 this->simulation_parameters.boundary_conditions));
235 }
236
237 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_force == true)
238 {
239 // Particle_Fluid Interactions Assembler
240 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
241 Parameters::DragModel::difelice)
242 {
243 // DiFelice Model drag Assembler
244 particle_fluid_assemblers.push_back(
245 std::make_shared<VANSAssemblerDiFelice<dim>>(
246 this->cfd_dem_simulation_parameters.cfd_dem));
247 }
248
249 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
250 Parameters::DragModel::rong)
251 {
252 // Rong Model drag Assembler
253 particle_fluid_assemblers.push_back(
254 std::make_shared<VANSAssemblerRong<dim>>(
255 this->cfd_dem_simulation_parameters.cfd_dem));
256 }
257
258 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
259 Parameters::DragModel::dallavalle)
260 {
261 // Dallavalle Model drag Assembler
262 particle_fluid_assemblers.push_back(
263 std::make_shared<VANSAssemblerDallavalle<dim>>(
264 this->cfd_dem_simulation_parameters.cfd_dem));
265 }
266
267 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
268 Parameters::DragModel::kochhill)
269 {
270 // Koch and Hill Model drag Assembler
271 particle_fluid_assemblers.push_back(
272 std::make_shared<VANSAssemblerKochHill<dim>>(
273 this->cfd_dem_simulation_parameters.cfd_dem));
274 }
275 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
276 Parameters::DragModel::beetstra)
277 {
278 // Beetstra drag model assembler
279 particle_fluid_assemblers.push_back(
280 std::make_shared<VANSAssemblerBeetstra<dim>>(
281 this->cfd_dem_simulation_parameters.cfd_dem));
282 }
283 if (this->cfd_dem_simulation_parameters.cfd_dem.drag_model ==
284 Parameters::DragModel::gidaspow)
285 {
286 // Gidaspow Model drag Assembler
287 particle_fluid_assemblers.push_back(
288 std::make_shared<VANSAssemblerGidaspow<dim>>(
289 this->cfd_dem_simulation_parameters.cfd_dem));
290 }
291 }
292
293 if (this->cfd_dem_simulation_parameters.cfd_dem.saffman_lift_force == true)
294 // Saffman Mei Lift Force Assembler
295 particle_fluid_assemblers.push_back(
296 std::make_shared<VANSAssemblerSaffmanMei<dim>>());
297
298 if (this->cfd_dem_simulation_parameters.cfd_dem.magnus_lift_force == true)
299 // Magnus Lift Force Assembler
300 particle_fluid_assemblers.push_back(
301 std::make_shared<VANSAssemblerMagnus<dim>>());
302
303 if (this->cfd_dem_simulation_parameters.cfd_dem.rotational_viscous_torque ==
304 true)
305 // Viscous Torque Assembler
306 particle_fluid_assemblers.push_back(
307 std::make_shared<VANSAssemblerViscousTorque<dim>>());
308
309 if (this->cfd_dem_simulation_parameters.cfd_dem.vortical_viscous_torque ==
310 true)
311 // Vortical Torque Assembler
312 particle_fluid_assemblers.push_back(
313 std::make_shared<VANSAssemblerVorticalTorque<dim>>());
314
315 if (this->cfd_dem_simulation_parameters.cfd_dem.buoyancy_force == true)
316 // Buoyancy Force Assembler
317 particle_fluid_assemblers.push_back(
318 std::make_shared<VANSAssemblerBuoyancy<dim>>(
319 this->cfd_dem_simulation_parameters.dem_parameters
320 .lagrangian_physical_properties.g));
321
322 if (this->cfd_dem_simulation_parameters.cfd_dem.pressure_force == true)
323 // Pressure Force
324 particle_fluid_assemblers.push_back(
325 std::make_shared<VANSAssemblerPressureForce<dim>>(
326 this->cfd_dem_simulation_parameters.cfd_dem));
327
328 if (this->cfd_dem_simulation_parameters.cfd_dem.shear_force == true)
329 // Shear Force
330 particle_fluid_assemblers.push_back(
331 std::make_shared<VANSAssemblerShearForce<dim>>(
332 this->cfd_dem_simulation_parameters.cfd_dem));
333
334 // Time-stepping schemes
335 if (is_bdf(this->simulation_control->get_assembly_method()))
336 {
337 this->assemblers.push_back(std::make_shared<VANSAssemblerBDF<dim>>(
338 this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
339 }
340
341 // Fluid_Particle Interactions Assembler
342 this->assemblers.push_back(std::make_shared<VANSAssemblerFPI<dim>>(
343 this->cfd_dem_simulation_parameters.cfd_dem));
344
345 // The core assembler should always be the last assembler to be called
346 // in the stabilized formulation as to have all strong residual and
347 // jacobian stored. Core assembler
348 if (this->cfd_dem_simulation_parameters.cfd_dem.vans_model ==
349 Parameters::VANSModel::modelA)
350 this->assemblers.push_back(std::make_shared<VANSAssemblerCoreModelA<dim>>(
351 this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
352
353 if (this->cfd_dem_simulation_parameters.cfd_dem.vans_model ==
354 Parameters::VANSModel::modelB)
355 this->assemblers.push_back(std::make_shared<VANSAssemblerCoreModelB<dim>>(
356 this->simulation_control, this->cfd_dem_simulation_parameters.cfd_dem));
357}

359template <int dim>
360void
361FluidDynamicsVANS<dim>::assemble_system_matrix()
362{
363 {
364 TimerOutput::Scope t(this->computing_timer, "Assemble matrix");
365 this->system_matrix = 0;
366
367 setup_assemblers();
368
369 auto scratch_data = NavierStokesScratchData<dim>(
370 this->simulation_control,
371 this->simulation_parameters.physical_properties_manager,
372 *this->fe,
373 *this->cell_quadrature,
374 *this->mapping,
375 *this->face_quadrature);
376
377 if (this->simulation_parameters.multiphysics.VOF)
378 {
379 const DoFHandler<dim> *dof_handler_vof =
380 this->multiphysics->get_dof_handler(PhysicsID::VOF);
381 scratch_data.enable_vof(
382 dof_handler_vof->get_fe(),
383 *this->cell_quadrature,
384 *this->mapping,
385 this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
386
387 if (this->simulation_parameters.multiphysics.vof_parameters
388 .surface_tension_force.enable)
389 {
390 const DoFHandler<dim>
391 &projected_phase_fraction_gradient_dof_handler =
392 this->multiphysics
393 ->get_projected_phase_fraction_gradient_dof_handler();
394 const DoFHandler<dim> &curvature_dof_handler =
395 this->multiphysics->get_curvature_dof_handler();
396 scratch_data.enable_projected_phase_fraction_gradient(
397 projected_phase_fraction_gradient_dof_handler.get_fe(),
398 *this->cell_quadrature,
399 *this->mapping);
400 scratch_data.enable_curvature(curvature_dof_handler.get_fe(),
401 *this->cell_quadrature,
402 *this->mapping);
403 }
404 }
405
406 scratch_data.enable_void_fraction(*particle_projector.fe,
407 *this->cell_quadrature,
408 *this->mapping);
409
410 scratch_data.enable_particle_fluid_interactions(
411 particle_handler.n_global_max_particles_per_cell(),
412 this->cfd_dem_simulation_parameters.cfd_dem.interpolated_void_fraction);
413
414 WorkStream::run(
415 this->dof_handler.begin_active(),
416 this->dof_handler.end(),
417 *this,
418 &FluidDynamicsVANS::assemble_local_system_matrix,
419 &FluidDynamicsVANS::copy_local_matrix_to_global_matrix,
420 scratch_data,
421 StabilizedMethodsTensorCopyData<dim>(this->fe->n_dofs_per_cell(),
422 this->cell_quadrature->size()));
423 this->system_matrix.compress(VectorOperation::add);
424 }
425}

427template <int dim>
428void
429FluidDynamicsVANS<dim>::assemble_local_system_matrix(
430 const typename DoFHandler<dim>::active_cell_iterator &cell,
431 NavierStokesScratchData<dim> &scratch_data,
432 StabilizedMethodsTensorCopyData<dim> &copy_data)
433{
434 copy_data.cell_is_local = cell->is_locally_owned();
435 if (!cell->is_locally_owned())
436 return;
437
438 scratch_data.reinit(
439 cell,
440 this->evaluation_point,
441 this->previous_solutions,
442 this->sdirk_vectors.sum_over_previous_stages,
443 this->forcing_function,
444 this->flow_control.get_beta(),
445 this->simulation_parameters.stabilization.pressure_scaling_factor);
446
447 std::optional<typename DoFHandler<dim>::active_cell_iterator> phase_cell;
448
449 if (this->simulation_parameters.multiphysics.VOF)
450 {
451 phase_cell.emplace(&(*(this->triangulation)),
452 cell->level(),
453 cell->index(),
454 this->multiphysics->get_dof_handler(PhysicsID::VOF));
455
456 scratch_data.reinit_vof(
457 *phase_cell,
458 *this->multiphysics->get_solution(PhysicsID::VOF),
459 *this->multiphysics->get_filtered_solution(PhysicsID::VOF),
460 *this->multiphysics->get_previous_solutions(PhysicsID::VOF));
461 }
462
463 typename DoFHandler<dim>::active_cell_iterator void_fraction_cell(
464 &(*(this->triangulation)),
465 cell->level(),
466 cell->index(),
467 &this->particle_projector.dof_handler);
468
469 scratch_data.reinit_void_fraction(
470 void_fraction_cell,
471 particle_projector.void_fraction_locally_relevant,
472 particle_projector.previous_void_fraction);
473
474 scratch_data.calculate_physical_properties();
475
476 if (this->simulation_parameters.multiphysics.VOF)
477 {
478 scratch_data.reinit_particle_fluid_interactions(
479 cell,
480 void_fraction_cell,
481 *phase_cell,
482 this->evaluation_point,
483 this->previous_solutions[0],
484 this->particle_projector.void_fraction_locally_relevant,
485 particle_handler,
486 *this->multiphysics->get_filtered_solution(PhysicsID::VOF));
487 }
488 else
489 {
490 scratch_data.reinit_particle_fluid_interactions(
491 cell,
492 void_fraction_cell,
493 this->evaluation_point,
494 this->previous_solutions[0],
495 this->particle_projector.void_fraction_locally_relevant,
496 particle_handler);
497 }
498
499 copy_data.reset();
500
501 for (auto &pf_assembler : particle_fluid_assemblers)
502 {
503 pf_assembler->calculate_particle_fluid_interactions(scratch_data);
504 }
505
506 for (auto &assembler : this->assemblers)
507 {
508 assembler->assemble_matrix(scratch_data, copy_data);
509 }
510
511 cell->get_dof_indices(copy_data.local_dof_indices);
512}

514template <int dim>
515void
516FluidDynamicsVANS<dim>::copy_local_matrix_to_global_matrix(
517 const StabilizedMethodsTensorCopyData<dim> &copy_data)
518{
519 if (!copy_data.cell_is_local)
520 return;
521
522 const AffineConstraints<double> &constraints_used = this->zero_constraints;
523 constraints_used.distribute_local_to_global(copy_data.local_matrix,
524 copy_data.local_dof_indices,
525 this->system_matrix);
526}

555 this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
556 }
557 scratch_data.enable_void_fraction(*particle_projector.fe,
558 *this->cell_quadrature,
559 *this->mapping);
560
561 scratch_data.enable_particle_fluid_interactions(
562 particle_handler.n_global_max_particles_per_cell(),
563 this->cfd_dem_simulation_parameters.cfd_dem.interpolated_void_fraction);
564

566 this->dof_handler.begin_active(),
567 this->dof_handler.end(),

573 this->cell_quadrature->size()));
574
575 this->system_rhs.compress(VectorOperation::add);
576
577 if (this->simulation_control->is_first_assembly())
578 this->simulation_control->provide_residual(this->system_rhs.l2_norm());
579}

668template <int dim>
669void
670FluidDynamicsVANS<dim>::copy_local_rhs_to_global_rhs(
671 const StabilizedMethodsTensorCopyData<dim> &copy_data)
672{
673 if (!copy_data.cell_is_local)
674 return;

711template <int dim>
712void
713FluidDynamicsVANS<dim>::monitor_mass_conservation()
714{
715 QGauss<dim> quadrature_formula(this->number_quadrature_points);
716
717 FEValues<dim> fe_values(*this->mapping,
718 *this->fe,
719 quadrature_formula,
720 update_values | update_quadrature_points |
721 update_JxW_values | update_gradients |
722 update_hessians);
723
724 FEValues<dim> fe_values_void_fraction(*this->mapping,
725 *this->particle_projector.fe,
726 quadrature_formula,
727 update_values |
728 update_quadrature_points |
729 update_JxW_values | update_gradients);
730
731 const FEValuesExtractors::Vector velocities(0);
732
733 const unsigned int n_q_points = quadrature_formula.size();
734
735 std::vector<double> present_void_fraction_values(n_q_points);
736 std::vector<Tensor<1, dim>> present_void_fraction_gradients(n_q_points);
737 // Values at previous time step for transient schemes for void
738 // fraction
739 std::vector<double> p1_void_fraction_values(n_q_points);
740 std::vector<double> p2_void_fraction_values(n_q_points);
741 std::vector<double> p3_void_fraction_values(n_q_points);
742
743 std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
744 std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
745
746 // Values of the force function, the last component of which is the external
747 // mass source
748 std::vector<Vector<double>> rhs_force(n_q_points, Vector<double>(dim + 1));
749
750 double continuity = 0;
751 double max_local_continuity = 0;
752 double local_mass_source = 0;
753
754 std::vector<double> time_steps_vector =
755 this->simulation_control->get_time_steps_vector();
756
757 const auto scheme = this->simulation_control->get_assembly_method();
758 const Vector<double> &bdf_coefs =
759 this->simulation_control->get_bdf_coefficients();
760
761 for (const auto &cell : this->dof_handler.active_cell_iterators())
762 {
763 if (cell->is_locally_owned())
764 {
765 typename DoFHandler<dim>::active_cell_iterator void_fraction_cell(
766 &(*this->triangulation),
767 cell->level(),
768 cell->index(),
769 &this->particle_projector.dof_handler);
770 fe_values_void_fraction.reinit(void_fraction_cell);
771
772 // Gather void fraction (values, gradient)
773 fe_values_void_fraction.get_function_values(
774 particle_projector.void_fraction_locally_relevant,
775 present_void_fraction_values);
776 fe_values_void_fraction.get_function_gradients(
777 particle_projector.void_fraction_locally_relevant,
778 present_void_fraction_gradients);
779
780 fe_values.reinit(cell);
781
782 // Gather velocity (values and gradient)
783 auto &evaluation_point = this->evaluation_point;
784 fe_values[velocities].get_function_values(evaluation_point,
785 present_velocity_values);
786 fe_values[velocities].get_function_gradients(
787 evaluation_point, present_velocity_gradients);
788
789 // Gather the external mass source at the quadrature point location
790 std::vector<Point<dim>> quadrature_points =
791 fe_values.get_quadrature_points();
792
793 this->forcing_function->vector_value_list(quadrature_points,
794 rhs_force);
795
796 // Gather the previous time steps depending on the number of
797 // stages of the time integration scheme for the void fraction
798
799 if (scheme !=
800 Parameters::SimulationControl::TimeSteppingMethod::steady)
801 {
802 fe_values_void_fraction.get_function_values(
803 particle_projector.previous_void_fraction[0],
804 p1_void_fraction_values);
805
806 if (scheme ==
807 Parameters::SimulationControl::TimeSteppingMethod::bdf2)
808 fe_values_void_fraction.get_function_values(
809 particle_projector.previous_void_fraction[1],
810 p2_void_fraction_values);
811
812 if (scheme ==
813 Parameters::SimulationControl::TimeSteppingMethod::bdf3)
814 fe_values_void_fraction.get_function_values(
815 particle_projector.previous_void_fraction[2],
816 p3_void_fraction_values);
817 }
818
819 local_mass_source = 0;
820
821 for (unsigned int q = 0; q < n_q_points; ++q)
822 {
823 // Calculate external mass source
824 const unsigned int component_mass =
825 this->fe->system_to_component_index(dim).first;
826 double external_mass_source = rhs_force[q](component_mass);
827
828 // Calculate the divergence of the velocity
829 const double present_velocity_divergence =
830 trace(present_velocity_gradients[q]);
831
832 // Evaluation of global mass conservation
833 local_mass_source =
834 (present_velocity_values[q] *
835 present_void_fraction_gradients[q] +
836 present_void_fraction_values[q] * present_velocity_divergence -
837 external_mass_source) *
838 fe_values_void_fraction.JxW(q);
839
840 if (scheme ==
841 Parameters::SimulationControl::TimeSteppingMethod::bdf1 ||
842 scheme == Parameters::SimulationControl::TimeSteppingMethod::
843 steady_bdf)
844 local_mass_source +=
845 (bdf_coefs[0] * present_void_fraction_values[q] +
846 bdf_coefs[1] * p1_void_fraction_values[q]) *
847 fe_values_void_fraction.JxW(q);
848
849 if (scheme ==
850 Parameters::SimulationControl::TimeSteppingMethod::bdf2)
851 local_mass_source +=
852 (bdf_coefs[0] * present_void_fraction_values[q] +
853 bdf_coefs[1] * p1_void_fraction_values[q] +
854 bdf_coefs[2] * p2_void_fraction_values[q]) *
855 fe_values_void_fraction.JxW(q);
856
857 if (scheme ==
858 Parameters::SimulationControl::TimeSteppingMethod::bdf3)
859 local_mass_source +=
860 (bdf_coefs[0] * present_void_fraction_values[q] +
861 bdf_coefs[1] * p1_void_fraction_values[q] +
862 bdf_coefs[2] * p2_void_fraction_values[q] +
863 bdf_coefs[3] * p3_void_fraction_values[q]) *
864 fe_values_void_fraction.JxW(q);
865
866 continuity += local_mass_source;
867 }
868
869 max_local_continuity =
870 std::max(max_local_continuity, abs(local_mass_source));
871 }
872 }
873
874 continuity = Utilities::MPI::sum(continuity, this->mpi_communicator);
875 max_local_continuity =
876 Utilities::MPI::max(max_local_continuity, this->mpi_communicator);
877
878 this->pcout << std::setprecision(
879 this->simulation_control->get_log_precision());
880
881 this->pcout << "Global continuity equation error: " << continuity << " s^-1"
882 << std::endl;
883 this->pcout << "Max local continuity error: " << max_local_continuity
884 << " s^-1" << std::endl;
885}

887template <int dim>
888void
889FluidDynamicsVANS<dim>::solve()
890{
891 read_mesh_and_manifolds(
892 *this->triangulation,
893 this->cfd_dem_simulation_parameters.cfd_parameters.mesh,
894 this->cfd_dem_simulation_parameters.cfd_parameters.manifolds_parameters,
895 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
896 .restart ||
897 this->cfd_dem_simulation_parameters.void_fraction->read_dem == true,
898 this->cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions);
899
900 if (this->cfd_dem_simulation_parameters.void_fraction->read_dem == true &&
901 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
902 .restart == false)
903 read_dem();
904
905 this->setup_dofs();
906
907 this->set_initial_condition(
908 this->cfd_dem_simulation_parameters.cfd_parameters.initial_condition->type,
909 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
910 .restart);
911
912 particle_handler.exchange_ghost_particles(true);
913
914 while (this->simulation_control->integrate())
915 {
916 this->simulation_control->print_progression(this->pcout);
917
918 // We allow the physics to update their boundary conditions
919 // according to their own parameters
920 this->update_boundary_conditions();
921 this->multiphysics->update_boundary_conditions();
922
923 this->dynamic_flow_control();
924
925 if (this->simulation_control->is_at_start())
926 {
927 vertices_cell_mapping();
928 particle_projector.initialize_void_fraction(
929 this->simulation_control->get_current_time());
930 this->iterate();
931 }
932 else
933 {
934 NavierStokesBase<dim, GlobalVectorType, IndexSet>::refine_mesh();
935 vertices_cell_mapping();
936 calculate_void_fraction(this->simulation_control->get_current_time());
937 this->iterate();
938 }
939
940 this->postprocess(false);
941 monitor_mass_conservation();
942 finish_time_step_fd();
943 }
944
945 this->finish_simulation();
946}
