10 template <int dim>
11 void
12 CahnHilliardAssemblerCore<dim>::assemble_matrix(
13  const CahnHilliardScratchData<dim> &scratch_data,
14  StabilizedMethodsCopyData &copy_data)
15 {
16  // Gather physical properties
17  const double epsilon = this->epsilon;
18  const double cell_size = scratch_data.cell_size;
19  const double xi =
20  this->cahn_hilliard_parameters.potential_smoothing_coefficient;
21 
22  // Loop and quadrature information
23  const auto &JxW_vec = scratch_data.JxW;
24  const unsigned int n_q_points = scratch_data.n_q_points;
25  const unsigned int n_dofs = scratch_data.n_dofs;
26 
27  // Copy data elements
28  auto &local_matrix = copy_data.local_matrix;
29 
30  for (unsigned int q = 0; q < n_q_points; ++q)
31  {
32  const double lambda =
33  3 * epsilon * scratch_data.surface_tension[q] / (2 * sqrt(2));
34 
35  const double mobility = scratch_data.mobility_cahn_hilliard[q];
36  const double mobility_gradient =
37  scratch_data.mobility_cahn_hilliard_gradient[q];
38 
39  // Gather into local variables the relevant fields
40  const Tensor<1, dim> velocity_field = scratch_data.velocity_values[q];
41 
42  // Store JxW in local variable for faster access;
43  const double JxW = JxW_vec[q];
44 
45  const double phase_order_value = scratch_data.phase_order_values[q];
46  const Tensor<1, dim> potential_gradient =
47  scratch_data.chemical_potential_gradients[q];
48 
49  for (unsigned int i = 0; i < n_dofs; ++i)
50  {
51  const double phi_phase_i = scratch_data.phi_phase[q][i];
52  const Tensor<1, dim> grad_phi_phase_i =
53  scratch_data.grad_phi_phase[q][i];
54 
55  const double phi_potential_i = scratch_data.phi_potential[q][i];
56  const Tensor<1, dim> grad_phi_potential_i =
57  scratch_data.grad_phi_potential[q][i];
58 
59 
60  for (unsigned int j = 0; j < n_dofs; ++j)
61  {
62  const double phi_phase_j = scratch_data.phi_phase[q][j];
63  const Tensor<1, dim> grad_phi_phase_j =
64  scratch_data.grad_phi_phase[q][j];
65 
66  const double phi_potential_j = scratch_data.phi_potential[q][j];
67  const Tensor<1, dim> grad_phi_potential_j =
68  scratch_data.grad_phi_potential[q][j];
69 
70  local_matrix(i, j) +=
71  // Phase order equation
72  (phi_phase_i * (velocity_field * grad_phi_phase_j) +
73  mobility * grad_phi_phase_i * grad_phi_potential_j +
74  mobility_gradient * grad_phi_phase_i * potential_gradient *
75  phi_phase_j
76  // Chemical potential equation
77  + phi_potential_i * phi_potential_j -
78  (lambda / (epsilon * epsilon)) * phi_potential_i *
79  (3 * phase_order_value * phase_order_value - 1.0) *
80  phi_phase_j -
81  lambda * grad_phi_potential_i * grad_phi_phase_j
82  // Chemical potential smoothing
83  + xi * cell_size * cell_size * grad_phi_potential_i *
84  grad_phi_potential_j) *
85  JxW;
86  }
87  }
88  }
89 }

102  const double xi =
103  this->cahn_hilliard_parameters.potential_smoothing_coefficient;
104 
105 
106  // Loop and quadrature information
107  const auto &JxW_vec = scratch_data.JxW;
108  const unsigned int n_q_points = scratch_data.n_q_points;
109  const unsigned int n_dofs = scratch_data.n_dofs;
110 
111  // Copy data elements
112  auto &local_rhs = copy_data.local_rhs;
113 
114  for (unsigned int q = 0; q < n_q_points; ++q)
115  {
116  const double lambda =
117  3 * epsilon * scratch_data.surface_tension[q] / (2 * sqrt(2));
118  const double mobility = scratch_data.mobility_cahn_hilliard[q];
119 
120 
121  // Gather into local variables the relevant fields
122  const Tensor<1, dim> velocity_field = scratch_data.velocity_values[q];
123 
124  // Store JxW in local variable for faster access;
125  const double JxW = JxW_vec[q];
126  const double phase_order_value = scratch_data.phase_order_values[q];
127  const Tensor<1, dim> phase_order_gradient =
128  scratch_data.phase_order_gradients[q];
129  const double potential_value = scratch_data.chemical_potential_values[q];
130  const Tensor<1, dim> potential_gradient =
131  scratch_data.chemical_potential_gradients[q];
132  const double source_phase_order = scratch_data.source_phase_order[q];
133  const double source_chemical_potential =
134  scratch_data.source_chemical_potential[q];
135 
136 
137  for (unsigned int i = 0; i < n_dofs; ++i)
138  {
139  const double phi_phase_i = scratch_data.phi_phase[q][i];
140  const Tensor<1, dim> grad_phi_phase_i =
141  scratch_data.grad_phi_phase[q][i];
142  const double phi_potential_i = scratch_data.phi_potential[q][i];
143  const Tensor<1, dim> grad_phi_potential_i =
144  scratch_data.grad_phi_potential[q][i];
145 
146  local_rhs(i) +=
147  // Phase order equation
148  (-phi_phase_i * (velocity_field * phase_order_gradient) -
149  mobility * grad_phi_phase_i * potential_gradient
150  // Chemical potential equation
151  - phi_potential_i * potential_value +
152  (lambda / (epsilon * epsilon)) * phi_potential_i *
153  (phase_order_value * phase_order_value - 1) * phase_order_value +
154  lambda * grad_phi_potential_i * phase_order_gradient
155  // Chemical potential smoothing
156  - xi * cell_size * cell_size * grad_phi_potential_i *
157  potential_gradient
158  // Source term
159  + source_phase_order * phi_phase_i +
160  source_chemical_potential * phi_potential_i) *
161  JxW;
162  }
163  }
164 }

169 template <int dim>
170 void
171 CahnHilliardAssemblerAngleOfContact<dim>::assemble_matrix(
172  const CahnHilliardScratchData<dim> &scratch_data,
173  StabilizedMethodsCopyData &copy_data)
174 {
175  if (!scratch_data.is_boundary_cell)
176  {
177  return;
178  }
179  const double epsilon = this->epsilon;
180 
181  auto &local_matrix = copy_data.local_matrix;
182 
183 
184  for (unsigned int f = 0; f < scratch_data.n_faces; f++)
185  {
186  // Check if the face is on a boundary
187  if (scratch_data.is_boundary_face[f])
188  {
189  if (this->boundary_conditions_cahn_hilliard.type.at(
190  scratch_data.boundary_face_id[f]) ==
191  BoundaryConditions::BoundaryType::cahn_hilliard_angle_of_contact)
192  {
193  const double angle_of_contact =
194  this->boundary_conditions_cahn_hilliard.angle_of_contact.at(
195  scratch_data.boundary_face_id[f]);
196  for (unsigned int q = 0; q < scratch_data.n_faces_q_points; ++q)
197  {
198  const Tensor<1, dim> face_phase_grad_value =
199  scratch_data.face_phase_grad_values[f][q];
200  const double lambda = 3 * epsilon *
201  scratch_data.surface_tension[q] /
202  (2 * sqrt(2));
203 
204  const double JxW_face = scratch_data.face_JxW[f][q];
205 
206  for (unsigned int i = 0; i < scratch_data.n_dofs; ++i)
207  {
208  const double phi_potential_i =
209  scratch_data.phi_potential[q][i];
210 
211  for (unsigned int j = 0; j < scratch_data.n_dofs; ++j)
212  {
213  const Tensor<1, dim> grad_phi_face_phase_j =
214  scratch_data.grad_phi_face_phase[f][q][j];
215 
216  local_matrix(i, j) +=
217  -lambda * -phi_potential_i * grad_phi_face_phase_j *
218  face_phase_grad_value *
219  std::cos(angle_of_contact * M_PI / 180.0) *
220  (1.0 / (face_phase_grad_value.norm() +
221  std::numeric_limits<double>::min())) *
222  JxW_face;
223  }
224  }
225  }
226  }
227  }
228  }

395 void
396 CahnHilliardAssemblerBDF<dim>::assemble_matrix(
397  const CahnHilliardScratchData<dim> &scratch_data,
398  StabilizedMethodsCopyData &copy_data)
399 {
400  // Loop and quadrature information
401  const auto &JxW = scratch_data.JxW;
402  const unsigned int n_q_points = scratch_data.n_q_points;
403  const unsigned int n_dofs = scratch_data.n_dofs;
404 
405  // Copy data elements
406  auto &local_matrix = copy_data.local_matrix;
407 
408  // Time stepping information
409  const auto method = this->simulation_control->get_assembly_method();
410  // Vector for the BDF coefficients
411  const Vector<double> &bdf_coefs =
412  this->simulation_control->get_bdf_coefficients();
413  std::vector<double> phase_order(1 + number_of_previous_solutions(method));
414 
415  // Loop over the quadrature points
416  for (unsigned int q = 0; q < n_q_points; ++q)
417  {
418  phase_order[0] = scratch_data.phase_order_values[q];
419  for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
420  phase_order[p + 1] = scratch_data.previous_phase_order_values[p][q];
421 
422  for (unsigned int i = 0; i < n_dofs; ++i)
423  {
424  const double phi_phase_i = scratch_data.phi_phase[q][i];
425  for (unsigned int j = 0; j < n_dofs; ++j)
426  {
427  const double phi_phase_j = scratch_data.phi_phase[q][j];
428 
429  local_matrix(i, j) +=
430  phi_phase_j * phi_phase_i * bdf_coefs[0] * JxW[q];
431  }
432  }
433  }
434 }
