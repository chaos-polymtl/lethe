12template <int dim, typename PropertiesIndex>
13void
14VelocityVerletIntegrator<dim, PropertiesIndex>::integrate_half_step_location(
15 Particles::ParticleHandler<dim> &particle_handler,
16 const Tensor<1, 3> &g,
17 const double dt,
18 const std::vector<Tensor<1, 3>> &torque,
19 const std::vector<Tensor<1, 3>> &force,
20 const std::vector<double> &MOI)
21{
22 Tensor<1, dim> particle_acceleration;
23
24 for (auto particle = particle_handler.begin();
25 particle != particle_handler.end();
26 ++particle)
27 {
28 // Get the total array view to the particle properties once to improve
29 // efficiency
30 auto particle_properties = particle->get_properties();
31 auto particle_position = particle->get_location();
32 types::particle_index particle_id = particle->get_local_index();
33
34 for (int d = 0; d < dim; ++d)
35 {
36 // Update acceleration
37 particle_acceleration[d] =
38 g[d] + (force[particle_id][d]) /
39 particle_properties[PropertiesIndex::mass];
40
41 // Half-step velocity
42 particle_properties[PropertiesIndex::v_x + d] +=
43 0.5 * particle_acceleration[d] * dt;
44
45 // Half-step angular velocity
46 particle_properties[PropertiesIndex::omega_x + d] +=
47 0.5 * (torque[particle_id][d] / MOI[particle_id]) * dt;
48
49 // Update particle position using half-step velocity
50 particle_position[d] +=
51 particle_properties[PropertiesIndex::v_x + d] * dt;
52 }
53 particle->set_location(particle_position);
54 }
55}

57template <int dim, typename PropertiesIndex>
58void
59VelocityVerletIntegrator<dim, PropertiesIndex>::integrate(
60 Particles::ParticleHandler<dim> &particle_handler,
61 const Tensor<1, 3> &g,
62 const double dt,
63 std::vector<Tensor<1, 3>> &torque,
64 std::vector<Tensor<1, 3>> &force,
65 const std::vector<double> &MOI)
66{
67 Point<3> particle_position;
68 const Tensor<1, 3> dt_g = g * dt;
69
70 for (auto &particle : particle_handler)
71 {
72 // Get the total array view to the particle properties once to improve
73 // efficiency
74 types::particle_index particle_id = particle.get_local_index();
75
76 auto particle_properties = particle.get_properties();
77 Tensor<1, 3> &particle_torque = torque[particle_id];
78 Tensor<1, 3> &particle_force = force[particle_id];
79
80 double dt_mass_inverse = dt / particle_properties[PropertiesIndex::mass];
81 double dt_MOI_inverse = dt / MOI[particle_id];
82
83 particle_position = [&] {
84 if constexpr (dim == 3)
85 {
86 return particle.get_location();
87 }
88 else
89 {
90 return (point_nd_to_3d(particle.get_location()));
91 }
92 }();
93
94 // Loop is manually unrolled for performance reason.
95 // for (int d = 0; d < 3; ++d)
96 {
97 // Particle velocity integration
98 particle_properties[PropertiesIndex::v_x] +=
99 dt_g[0] + particle_force[0] * dt_mass_inverse;
100 particle_properties[PropertiesIndex::v_y] +=
101 dt_g[1] + particle_force[1] * dt_mass_inverse;
102 particle_properties[PropertiesIndex::v_z] +=
103 dt_g[2] + particle_force[2] * dt_mass_inverse;
104
105
106 // Particle location integration
107 particle_position[0] += particle_properties[PropertiesIndex::v_x] * dt;
108 particle_position[1] += particle_properties[PropertiesIndex::v_y] * dt;
109 particle_position[2] += particle_properties[PropertiesIndex::v_z] * dt;
110
111 // Updating angular velocity
112 particle_properties[PropertiesIndex::omega_x] +=
113 particle_torque[0] * dt_MOI_inverse;
114 particle_properties[PropertiesIndex::omega_y] +=
115 particle_torque[1] * dt_MOI_inverse;
116 particle_properties[PropertiesIndex::omega_z] +=
117 particle_torque[2] * dt_MOI_inverse;
118 }
119
120 // Reinitialize force and torque of particle
121 particle_force = 0;
122 particle_torque = 0;
123
124 if constexpr (dim == 3)
125 particle.set_location(particle_position);
126
127 if constexpr (dim == 2)
128 {
129 Point<2> position_2d;
130 position_2d[0] = particle_position[0];
131 position_2d[1] = particle_position[1];
132 particle.set_location(position_2d);
133 }
134 }

137template <int dim, typename PropertiesIndex>
138void
139VelocityVerletIntegrator<dim, PropertiesIndex>::integrate(
140 Particles::ParticleHandler<dim> &particle_handler,
141 const Tensor<1, 3> &g,
142 const double dt,
143 std::vector<Tensor<1, 3>> &torque,
144 std::vector<Tensor<1, 3>> &force,
145 const std::vector<double> &MOI,
146 const parallel::distributed::Triangulation<dim> &triangulation,
147 AdaptiveSparseContacts<dim, PropertiesIndex> &sparse_contacts_object)
148{
149 auto *action_manager = DEMActionManager::get_action_manager();
150
151 bool use_default_function =
152 !action_manager->check_sparse_contacts_enabled() ||
153 action_manager->check_mobility_status_reset();
154
155 // Regular integration process if sparse contacts are not enabled or if this
156 // is the first iteration.
157 if (use_default_function)
158 {
159 integrate(particle_handler, g, dt, torque, force, MOI);
160 return;
161 }
162
163 // If there are advected particles, we use another function since the average
164 // velocity and acceleration of cells are computed for mobile cells
165 if (sparse_contacts_object.has_advected_particles())
166 {
167 integrate_with_advected_particles(particle_handler,
168 g,
169 dt,
170 torque,
171 force,
172 MOI,
173 triangulation,
174 sparse_contacts_object);
175 }
176 else
177 {
178 Point<3> particle_position;
179 const Tensor<1, 3> dt_g = g * dt;
180
181 // Get the map of mobility status of cells
182 auto &cell_mobility_status_map =
183 sparse_contacts_object.get_mobility_status();
184
185 for (auto &cell : triangulation.active_cell_iterators())
186 {
187 if (cell->is_locally_owned())
188 {
189 // Get the mobility status of the cell
190 unsigned int mobility_status =
191 cell_mobility_status_map.at(cell->active_cell_index());
192
193 // We loop over the particles, even if cell is not mobile, to
194 // reset the force and torques value of the particle with the
195 // particle id
196 auto particles_in_cell = particle_handler.particles_in_cell(cell);
197 unsigned int n_particles_in_cell =
198 particle_handler.n_particles_in_cell(cell);
199
200 if (n_particles_in_cell > 0)
201 {
202 if (mobility_status ==
203 AdaptiveSparseContacts<dim, PropertiesIndex>::mobile)
204 {
205 for (auto &particle : particles_in_cell)
206 {
207 types::particle_index particle_id =
208 particle.get_local_index();
209 auto particle_properties = particle.get_properties();
210
211 Tensor<1, 3> &particle_torque = torque[particle_id];
212 Tensor<1, 3> &particle_force = force[particle_id];
213
214 double dt_mass_inverse =
215 dt / particle_properties[PropertiesIndex::mass];
216 double dt_MOI_inverse = dt / MOI[particle_id];
217
218 particle_position = [&] {
219 if constexpr (dim == 3)
220 {
221 return particle.get_location();
222 }
223 else
224 {
225 return (
226 point_nd_to_3d(particle.get_location()));
227 }
228 }();
229
230 for (unsigned int d = 0; d < 3; ++d)
231 {
232 // Particle velocity integration
233 particle_properties[PropertiesIndex::v_x + d] +=
234 dt_g[d] + particle_force[d] * dt_mass_inverse;
235
236 // Particle location integration
237 particle_position[d] +=
238 particle_properties[PropertiesIndex::v_x + d] *
239 dt;
240
241 // Updating angular velocity
242 particle_properties[PropertiesIndex::omega_x +
243 d] +=
244 particle_torque[d] * dt_MOI_inverse;
245 }
246
247 // Reinitialize force and torque of particle
248 particle_force = 0.0;
249 particle_torque = 0.0;
250
251 // Update particle location
252 if constexpr (dim == 3)
253 particle.set_location(particle_position);
254
255 if constexpr (dim == 2)
256 {
257 Point<2> position_2d;
258 position_2d[0] = particle_position[0];
259 position_2d[1] = particle_position[1];
260 particle.set_location(position_2d);
261 }
262 }
263 }
264 else // Active and inactive cells
265 {
266 for (auto &particle : particles_in_cell)
267 {
268 types::particle_index particle_id =
269 particle.get_local_index();
270
271 // Reset forces
272 force[particle_id] = 0.0;
273 torque[particle_id] = 0.0;
274 }
275 }
276 }
277 }
278 }
279 }

282template <int dim, typename PropertiesIndex>
283void
284VelocityVerletIntegrator<dim, PropertiesIndex>::
285 integrate_with_advected_particles(
286 Particles::ParticleHandler<dim> &particle_handler,
287 const Tensor<1, 3> &g,
288 const double dt,
289 std::vector<Tensor<1, 3>> &torque,
290 std::vector<Tensor<1, 3>> &force,
291 const std::vector<double> &MOI,
292 const parallel::distributed::Triangulation<dim> &triangulation,
293 AdaptiveSparseContacts<dim, PropertiesIndex> &sparse_contacts_object)
294{
295 Point<3> particle_position;
296 const Tensor<1, 3> dt_g = g * dt;
297
298 // Get the map of average velocities and accelerations of cells
299 auto cell_velocities_accelerations_map =
300 sparse_contacts_object.get_velocities_accelerations();
301
302 // Get the map of mobility status of cells
303 auto cell_mobility_status_map = sparse_contacts_object.get_mobility_status();
304
305 for (auto &cell : triangulation.active_cell_iterators())
306 {
307 if (cell->is_locally_owned())
308 {
309 // Get the mobility status of the cell
310 unsigned int mobility_status =
311 cell_mobility_status_map.at(cell->active_cell_index());
312
313 // Get the average velocity and acceleration * dt of the cell
314 auto &[velocity_cell_average, acc_dt_cell_average] =
315 cell_velocities_accelerations_map[cell];
316
317 // We loop over the particles, even if cell is not mobile, to reset
318 // the force and torques value of the particle with the particle id
319 auto particles_in_cell = particle_handler.particles_in_cell(cell);
320 unsigned int n_particles_in_cell =
321 particle_handler.n_particles_in_cell(cell);
322
323 if (n_particles_in_cell > 0)
324 {
325 if (mobility_status ==
326 AdaptiveSparseContacts<dim, PropertiesIndex>::mobile)
327 {
328 // When mobile, the average velocity of the cell and the
329 // acceleration is updated at the same step that the particles
330 // are integrated of performance reasons
331 velocity_cell_average.clear();
332 acc_dt_cell_average.clear();
333
334 for (auto &particle : particles_in_cell)
335 {
336 types::particle_index particle_id =
337 particle.get_local_index();
338 auto particle_properties = particle.get_properties();
339
340 Tensor<1, 3> &particle_torque = torque[particle_id];
341 Tensor<1, 3> &particle_force = force[particle_id];
342
343 double dt_mass_inverse =
344 dt / particle_properties[PropertiesIndex::mass];
345 double dt_MOI_inverse = dt / MOI[particle_id];
346
347 particle_position = [&] {
348 if constexpr (dim == 3)
349 {
350 return particle.get_location();
351 }
352 else
353 {
354 return (point_nd_to_3d(particle.get_location()));
355 }
356 }();
357
358 // Calculate acceleration * dt and add velocity value for
359 // average acceleration computation
360 Tensor<1, 3> acc_dt_particle =
361 dt_g + particle_force * dt_mass_inverse;
362 acc_dt_cell_average += acc_dt_particle;
363
364 for (unsigned int d = 0; d < 3; ++d)
365 {
366 // Particle velocity integration
367 particle_properties[PropertiesIndex::v_x + d] +=
368 acc_dt_particle[d];
369
370 // Add velocity value for average velocity computation
371 velocity_cell_average[d] +=
372 particle_properties[PropertiesIndex::v_x + d];
373
374 // Particle location integration
375 particle_position[d] +=
376 particle_properties[PropertiesIndex::v_x + d] * dt;
377
378 // Updating angular velocity
379 particle_properties[PropertiesIndex::omega_x + d] +=
380 particle_torque[d] * dt_MOI_inverse;
381 }
382
383 // Reinitialize force and torque of particle
384 particle_force = 0.0;
385 particle_torque = 0.0;
386
387 // Update particle location
388 if constexpr (dim == 3)
389 particle.set_location(particle_position);
390
391 if constexpr (dim == 2)
392 {
393 Point<2> position_2d;
394 position_2d[0] = particle_position[0];
395 position_2d[1] = particle_position[1];
396 particle.set_location(position_2d);
397 }
398 }
399
400 // Compute average velocity and acceleration
401 acc_dt_cell_average /= n_particles_in_cell;
402 velocity_cell_average /= n_particles_in_cell;
403 }
404 else if (mobility_status ==
405 AdaptiveSparseContacts<dim,
406 PropertiesIndex>::advected ||
407 mobility_status ==
408 AdaptiveSparseContacts<dim, PropertiesIndex>::
409 advected_active)
410 {
411 for (auto &particle : particles_in_cell)
412 {
413 types::particle_index particle_id =
414 particle.get_local_index();
415
416 particle_position = [&] {
417 if constexpr (dim == 3)
418 {
419 return particle.get_location();
420 }
421 else
422 {
423 return (point_nd_to_3d(particle.get_location()));
424 }
425 }();
426
427 // Update the new average velocity of the cell : v + a*dt
428 velocity_cell_average += acc_dt_cell_average;
429
430 // Particle location integration with the average velocity
431 for (unsigned int d = 0; d < 3; ++d)
432 {
433 particle_position[d] += velocity_cell_average[d] * dt;
434 }
435
436 // Reset forces
437 force[particle_id] = 0.0;
438 torque[particle_id] = 0.0;
439
440 // Update particle location
441 if constexpr (dim == 3)
442 particle.set_location(particle_position);
443
444 if constexpr (dim == 2)
445 {
446 Point<2> position_2d;
447 position_2d[0] = particle_position[0];
448 position_2d[1] = particle_position[1];
449 particle.set_location(position_2d);
450 }
451 }
452 }
453 else // Static active and inactive cells (no particles
454 // integration)
455 {
456 for (auto &particle : particles_in_cell)
457 {
458 types::particle_index particle_id =
459 particle.get_local_index();
460
461 // Reset forces
462 force[particle_id] = 0.0;
463 torque[particle_id] = 0.0;
464 }
465 }
466 }
467 }
468 }
