41/**
42 * @brief Implementation of heat transfer as an auxiliary physics. The heat
43 * equation is weakly coupled to the velocity field. Equation solved:
44 * \f$ \rho C_p \cdot \left( \frac{dT}{dt} + u \cdot \nabla T \right) =
45 * k \cdot \nabla^2 T + \tau : \nabla u \f$
46 *
47 */
48
49template <int dim>
50class HeatTransfer : public AuxiliaryPhysics<dim, GlobalVectorType>
51{
52public:
53 /**
54 * @brief Constructor of the HeatTransfer object.
55 *
56 * @param multiphysics_interface Map of the auxiliary physics that will be
57 * solved on top of a computational fluid dynamic simulation.
58 *
59 * @param p_simulation_parameters Contain the simulation parameter file
60 * information.
61 *
62 * @param p_triangulation Contain the mesh information. In a
63 * parallel::DistributedTriangulationBase<dim> not every detail may be known
64 * on each processor. The mesh is distributed between the processors.
65 *
66 * @param p_simulation_control Object responsible for the control of
67 * steady-state and transient simulations. Contains all the information
68 * related to time stepping and the stopping criteria.
69 *
70 */
71 HeatTransfer(MultiphysicsInterface<dim> *multiphysics_interface,
72 const SimulationParameters<dim> &p_simulation_parameters,
73 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
74 p_triangulation,
75 std::shared_ptr<SimulationControl> p_simulation_control)
76 : AuxiliaryPhysics<dim, GlobalVectorType>(
77 p_simulation_parameters.non_linear_solver.at(PhysicsID::heat_transfer))
78 , multiphysics(multiphysics_interface)
79 , computing_timer(p_triangulation->get_mpi_communicator(),
80 this->pcout,
81 TimerOutput::summary,
82 TimerOutput::wall_times)
83 , simulation_parameters(p_simulation_parameters)
84 , triangulation(p_triangulation)
85 , simulation_control(p_simulation_control)
86 , dof_handler(*triangulation)
87 , thermal_conductivity_models(
88 p_simulation_parameters.physical_properties_manager
89 .get_thermal_conductivity_vector())
90
91 {
92 if (simulation_parameters.mesh.simplex)
93 {
94 // for simplex meshes
95 fe = std::make_shared<FE_SimplexP<dim>>(
96 simulation_parameters.fem_parameters.temperature_order);
97 temperature_mapping = std::make_shared<MappingFE<dim>>(*fe);
98 cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
99 face_quadrature =
100 std::make_shared<QGaussSimplex<dim - 1>>(fe->degree + 1);
101 }
102 else
103 {
104 // Usual case, for quad/hex meshes
105 fe = std::make_shared<FE_Q<dim>>(
106 simulation_parameters.fem_parameters.temperature_order);
107 temperature_mapping = std::make_shared<MappingQ<dim>>(fe->degree);
108 cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
109 face_quadrature = std::make_shared<QGauss<dim - 1>>(fe->degree + 1);
110 }
111
112 // Allocate solution transfer
113 solution_transfer =
114 std::make_shared<SolutionTransfer<dim, GlobalVectorType>>(dof_handler);
115
116 // Set size of previous solutions using BDF schemes information
117 previous_solutions.resize(maximum_number_of_previous_solutions());
118
119 // Prepare previous solutions transfer
120 previous_solutions_transfer.reserve(previous_solutions.size());
121 for (unsigned int i = 0; i < previous_solutions.size(); ++i)
122 {
123 previous_solutions_transfer.emplace_back(
124 SolutionTransfer<dim, GlobalVectorType>(this->dof_handler));
125 }
126
127 // Change the behavior of the timer for situations when you don't want
128 // outputs
129 if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
130 this->computing_timer.disable_output();
131
132 if (simulation_parameters.post_processing.calculate_average_temp_and_hf)
133 {
134 average_temperature =
135 std::make_shared<AverageScalar<dim>>(this->dof_handler);
136 }
137 }

395 /**
396 * @brief Verify consistency of the input parameters for boundary
397 * conditions to ensure that for every boundary condition within the
398 * triangulation, a boundary condition has been specified in the input file.
399 */
400 void
401 verify_consistency_of_boundary_conditions()
402 {
403 // Sanity check all of the boundary conditions of the triangulation to
404 // ensure that they have a type.
405 std::vector<types::boundary_id> boundary_ids_in_triangulation =
406 this->triangulation->get_boundary_ids();
407 for (auto const &boundary_id_in_tria : boundary_ids_in_triangulation)
408 {
409 AssertThrow(simulation_parameters.boundary_conditions_ht.type.find(
410 boundary_id_in_tria) !=
411 simulation_parameters.boundary_conditions_ht.type.end(),
412 HeatTransferBoundaryConditionMissing(boundary_id_in_tria));
413 }
414 }
