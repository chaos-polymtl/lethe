11template <int dim>
12void
13GLSIsothermalCompressibleNavierStokesAssemblerCore<dim>::assemble_matrix(
14 const NavierStokesScratchData<dim> &scratch_data,
15 StabilizedMethodsTensorCopyData<dim> &copy_data)
16{
17 // Physical properties
18 const std::vector<double> &density_vector = scratch_data.density;
19 const std::vector<double> &dynamic_viscosity_vector =
20 scratch_data.dynamic_viscosity;
21 const double density_psi = scratch_data.density_psi;
22
23 const std::vector<double> &viscosity_for_stabilization_vector =
24 scratch_data.dynamic_viscosity_for_stabilization;
25
26 // Loop and quadrature information
27 const auto &JxW_vec = scratch_data.JxW;
28 const unsigned int n_q_points = scratch_data.n_q_points;
29 const unsigned int n_dofs = scratch_data.n_dofs;
30 const double h = scratch_data.cell_size;
31
32 // Copy data elements
33 auto &strong_residual_vec = copy_data.strong_residual;
34 auto &strong_jacobian_vec = copy_data.strong_jacobian;
35 auto &local_matrix = copy_data.local_matrix;
36
37 // Time steps and inverse time steps which is used for stabilization constant
38 std::vector<double> time_steps_vector =
39 this->simulation_control->get_time_steps_vector();
40 const double dt = time_steps_vector[0];
41 const double sdt = 1. / dt;
42
43 // Pressure scaling factor
44 const double pressure_scaling_factor = scratch_data.pressure_scaling_factor;
45
46 // Loop over the quadrature points
47 for (unsigned int q = 0; q < n_q_points; ++q)
48 {
49 // Gather into local variables the relevant fields
50 const double density = density_vector[q];
51 const double dynamic_viscosity = dynamic_viscosity_vector[q];
52 const double dynamic_viscosity_for_stabilization =
53 viscosity_for_stabilization_vector[q];
54
55
56 const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
57 const Tensor<2, dim> &velocity_gradient =
58 scratch_data.velocity_gradients[q];
59 const Tensor<1, dim> &velocity_laplacian =
60 scratch_data.velocity_laplacians[q];
61
62 const Tensor<1, dim> &pressure_gradient =
63 scratch_data.pressure_gradients[q];
64
65 // Forcing term
66 const Tensor<1, dim> &force = scratch_data.force[q];
67 double mass_source = scratch_data.mass_source[q];
68
69 // Calculation of the magnitude of the velocity for the
70 // stabilization parameter
71 const double u_mag = std::max(velocity.norm(), 1e-12);
72
73 // Store JxW in local variable for faster access;
74 const double JxW = JxW_vec[q];
75
76 // Calculation of the GLS stabilization parameter. The
77 // stabilization parameter used is different if the simulation
78 // is steady or unsteady. In the unsteady case it includes the
79 // value of the time-step
80 const double tau =
81 this->simulation_control->get_assembly_method() ==
82 Parameters::SimulationControl::TimeSteppingMethod::steady ?
83 calculate_navier_stokes_gls_tau_steady(
84 u_mag, dynamic_viscosity_for_stabilization / density, h) :
85 calculate_navier_stokes_gls_tau_transient(
86 u_mag, dynamic_viscosity_for_stabilization / density, h, sdt);
87
88 // Calculate the strong residual for GLS stabilization
89 auto strong_residual = density * velocity_gradient * velocity +
90 pressure_gradient -
91 dynamic_viscosity * velocity_laplacian - force +
92 mass_source * velocity + strong_residual_vec[q];
93
94 std::vector<Tensor<1, dim>> grad_phi_u_j_x_velocity(n_dofs);
95 std::vector<Tensor<1, dim>> velocity_gradient_x_phi_u_j(n_dofs);
96
97 // We loop over the column first to prevent recalculation
98 // of the strong jacobian in the inner loop
99 for (unsigned int j = 0; j < n_dofs; ++j)
100 {
101 const auto &phi_u_j = scratch_data.phi_u[q][j];
102 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
103 const auto &laplacian_phi_u_j = scratch_data.laplacian_phi_u[q][j];
104
105 const auto &grad_phi_p_j =
106 pressure_scaling_factor * scratch_data.grad_phi_p[q][j];
107
108 strong_jacobian_vec[q][j] +=
109 (density * velocity_gradient * phi_u_j + grad_phi_u_j * velocity +
110 grad_phi_p_j - dynamic_viscosity * laplacian_phi_u_j +
111 mass_source * phi_u_j);
112
113 // Store these temporary products in auxiliary variables for speed
114 grad_phi_u_j_x_velocity[j] = grad_phi_u_j * velocity;
115 velocity_gradient_x_phi_u_j[j] = velocity_gradient * phi_u_j;
116 }
117
118 for (unsigned int i = 0; i < n_dofs; ++i)
119 {
120 const unsigned int component_i = scratch_data.components[i];
121
122 const auto &phi_u_i = scratch_data.phi_u[q][i];
123 const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
124 const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
125 const auto &phi_p_i = scratch_data.phi_p[q][i];
126 const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
127
128 // Store these temporary products in auxiliary variables for speed
129 const auto grad_phi_u_i_x_velocity = grad_phi_u_i * velocity;
130 const auto strong_residual_x_grad_phi_u_i =
131 strong_residual * grad_phi_u_i;
132
133 for (unsigned int j = 0; j < n_dofs; ++j)
134 {
135 const unsigned int component_j = scratch_data.components[j];
136
137 const auto &phi_u_j = scratch_data.phi_u[q][j];
138
139 const auto &phi_p_j =
140 pressure_scaling_factor * scratch_data.phi_p[q][j];
141 const auto &grad_phi_p_j =
142 pressure_scaling_factor * scratch_data.grad_phi_p[q][j];
143
144 const auto &strong_jac = strong_jacobian_vec[q][j];
145
146 // $$\nabla p$$ from the momentum equation
147 double local_matrix_ij =
148 component_j == dim ? -div_phi_u_i * phi_p_j : 0;
149
150 // continuity equation
151 if (component_i == dim)
152 {
153 const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
154
155 local_matrix_ij +=
156 // $$\psi \mathbf{u} \nabla p$$
157 phi_p_i * density_psi * phi_u_j * pressure_gradient +
158 phi_p_i * density_psi * velocity * grad_phi_p_j +
159
160 // $$\rho \nabla \cdot \mathbf{u}$$
161 phi_p_i * density * div_phi_u_j;
162
163 // PSPG GLS term
164 local_matrix_ij += tau * (strong_jac * grad_phi_p_i);
165 }
166
167 // momentum equation
168 if (component_i < dim && component_j < dim)
169 {
170 local_matrix_ij +=
171 // $$\rho (\mathbf{u} \cdot \nabla) \mathbf{u}$$
172 phi_u_i * density * velocity_gradient_x_phi_u_j[j] +
173 phi_u_i * density * grad_phi_u_j_x_velocity[j];
174
175 if (component_i == component_j)
176 {
177 const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
178
179 local_matrix_ij +=
180 // weak form of $$\mu \nabla^{2} \mathbf{u}$$
181 dynamic_viscosity * (grad_phi_u_j[component_j] *
182 grad_phi_u_i[component_i]) +
183 // volume source
184 phi_u_i * mass_source * phi_u_j;
185 }
186 }
187
188 // The jacobian matrix for the SUPG formulation
189 // currently does not include the jacobian of the
190 // stabilization parameter tau. Our experience has shown that
191 // does not alter the number of newton iteration for
192 // convergence, but greatly simplifies assembly.
193 local_matrix_ij +=
194 tau * (strong_jac * (grad_phi_u_i_x_velocity) +
195 strong_residual_x_grad_phi_u_i * phi_u_j);
196
197 local_matrix_ij *= JxW;
198 local_matrix(i, j) += local_matrix_ij;
199 }
200 }
201 }
202}

204template <int dim>
205void
206GLSIsothermalCompressibleNavierStokesAssemblerCore<dim>::assemble_rhs(
207 const NavierStokesScratchData<dim> &scratch_data,
208 StabilizedMethodsTensorCopyData<dim> &copy_data)
209{
210 // Physical properties
211 const std::vector<double> &density_vector = scratch_data.density;
212 const std::vector<double> &dynamic_viscosity_vector =
213 scratch_data.dynamic_viscosity;
214
215 const std::vector<double> &viscosity_for_stabilization_vector =
216 scratch_data.dynamic_viscosity_for_stabilization;
217
218 const double density_psi = scratch_data.density_psi;
219
220 // Loop and quadrature information
221 const auto &JxW_vec = scratch_data.JxW;
222 const unsigned int n_q_points = scratch_data.n_q_points;
223 const unsigned int n_dofs = scratch_data.n_dofs;
224 const double h = scratch_data.cell_size;
225
226 // Copy data elements
227 auto &strong_residual_vec = copy_data.strong_residual;
228 auto &local_rhs = copy_data.local_rhs;
229
230 // Time steps and inverse time steps which is used for stabilization constant
231 std::vector<double> time_steps_vector =
232 this->simulation_control->get_time_steps_vector();
233 const double dt = time_steps_vector[0];
234 const double sdt = 1. / dt;
235
236 // Loop over the quadrature points
237 for (unsigned int q = 0; q < n_q_points; ++q)
238 {
239 // Physical properties
240 const double density = density_vector[q];
241 const double dynamic_viscosity = dynamic_viscosity_vector[q];
242
243 // Velocity
244 const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
245 const double velocity_divergence = scratch_data.velocity_divergences[q];
246 const Tensor<2, dim> &velocity_gradient =
247 scratch_data.velocity_gradients[q];
248 const Tensor<1, dim> &velocity_laplacian =
249 scratch_data.velocity_laplacians[q];
250
251 // Pressure
252 const double pressure = scratch_data.pressure_values[q];
253 const Tensor<1, dim> &pressure_gradient =
254 scratch_data.pressure_gradients[q];
255
256 // Forcing term
257 const Tensor<1, dim> &force = scratch_data.force[q];
258 double mass_source = scratch_data.mass_source[q];
259 // Calculation of the magnitude of the
260 // velocity for the stabilization parameter
261 const double u_mag = std::max(velocity.norm(), 1e-12);
262
263 // Store JxW in local variable for faster access;
264 const double JxW = JxW_vec[q];
265
266 // Calculation of the GLS stabilization parameter. The
267 // stabilization parameter used is different if the simulation
268 // is steady or unsteady. In the unsteady case it includes the
269 // value of the time-step
270 const double tau =
271 this->simulation_control->get_assembly_method() ==
272 Parameters::SimulationControl::TimeSteppingMethod::steady ?
273 calculate_navier_stokes_gls_tau_steady(
274 u_mag, viscosity_for_stabilization_vector[q] / density, h) :
275 calculate_navier_stokes_gls_tau_transient(
276 u_mag, viscosity_for_stabilization_vector[q] / density, h, sdt);
277
278 // Calculate the strong residual for GLS stabilization
279 auto strong_residual = density * velocity_gradient * velocity +
280 pressure_gradient -
281 dynamic_viscosity * velocity_laplacian - force +
282 mass_source * velocity + strong_residual_vec[q];
283
284 // Assembly of the right-hand side
285 for (unsigned int i = 0; i < n_dofs; ++i)
286 {
287 const auto &phi_u_i = scratch_data.phi_u[q][i];
288 const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
289 const auto &phi_p_i = scratch_data.phi_p[q][i];
290 const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
291 const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
292
293 double local_rhs_i = 0;
294
295 // Navier-Stokes Residual
296 local_rhs_i +=
297 (
298 // Continuity
299 -phi_p_i * density_psi * velocity * pressure_gradient -
300 phi_p_i * density * velocity_divergence - phi_p_i * mass_source -
301 // Momentum
302 phi_u_i * density * velocity_gradient * velocity +
303 div_phi_u_i * pressure -
304 dynamic_viscosity *
305 scalar_product(velocity_gradient, grad_phi_u_i) +
306 phi_u_i * force - mass_source * velocity * phi_u_i) *
307 JxW;
308
309 // PSPG GLS term
310 local_rhs_i +=
311 -tau / density * (strong_residual * grad_phi_p_i) * JxW;
312
313 // SUPG GLS term
314 local_rhs_i +=
315 -tau * (strong_residual * (grad_phi_u_i * velocity)) * JxW;
316
317 local_rhs(i) += local_rhs_i;
318 }
319 }
320}

325template <int dim>
326void
327GLSIsothermalCompressibleNavierStokesAssemblerBDF<dim>::assemble_matrix(
328 const NavierStokesScratchData<dim> &scratch_data,
329 StabilizedMethodsTensorCopyData<dim> &copy_data)
330{
331 // Physical properties
332 const std::vector<double> &density_vector = scratch_data.density;
333 const double density_psi = scratch_data.density_psi;
334
335 // Loop and quadrature information
336 const auto &JxW = scratch_data.JxW;
337 const unsigned int n_q_points = scratch_data.n_q_points;
338 const unsigned int n_dofs = scratch_data.n_dofs;
339
340 // Copy data elements
341 auto &strong_residual = copy_data.strong_residual;
342 auto &strong_jacobian = copy_data.strong_jacobian;
343 auto &local_matrix = copy_data.local_matrix;
344
345 // Time stepping information
346 const auto method = this->simulation_control->get_assembly_method();
347 // Vector for the BDF coefficients
348 const Vector<double> &bdf_coefs =
349 this->simulation_control->get_bdf_coefficients();
350 std::vector<double> pressure(1 + number_of_previous_solutions(method));
351 std::vector<Tensor<1, dim>> velocity(1 +
352 number_of_previous_solutions(method));
353
354 // Loop over the quadrature points
355 for (unsigned int q = 0; q < n_q_points; ++q)
356 {
357 // Physical properties
358 const double density = density_vector[q];
359
360 velocity[0] = scratch_data.velocity_values[q];
361 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
362 velocity[p + 1] = scratch_data.previous_velocity_values[p][q];
363
364 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
365 ++p)
366 strong_residual[q] += bdf_coefs[p] * density * velocity[p];
367
368 for (unsigned int j = 0; j < n_dofs; ++j)
369 strong_jacobian[q][j] +=
370 bdf_coefs[0] * density * scratch_data.phi_u[q][j];
371
372 for (unsigned int i = 0; i < n_dofs; ++i)
373 {
374 const double phi_p_i = scratch_data.phi_p[q][i];
375 const Tensor<1, dim> &phi_u_i = scratch_data.phi_u[q][i];
376 for (unsigned int j = 0; j < n_dofs; ++j)
377 {
378 const double &phi_p_j = scratch_data.phi_p[q][j];
379 const Tensor<1, dim> &phi_u_j = scratch_data.phi_u[q][j];
380
381 local_matrix(i, j) +=
382 (phi_p_i * density_psi * phi_p_j * bdf_coefs[0] +
383 phi_u_i * density * phi_u_j * bdf_coefs[0]) *
384 JxW[q];
385 }
386 }
387 }

390template <int dim>
391void
392GLSIsothermalCompressibleNavierStokesAssemblerBDF<dim>::assemble_rhs(
393 const NavierStokesScratchData<dim> &scratch_data,
394 StabilizedMethodsTensorCopyData<dim> &copy_data)
395{
396 // Physical properties
397 const std::vector<double> &density_vector = scratch_data.density;
398 const double density_psi = scratch_data.density_psi;
399
400 // Loop and quadrature information
401 const auto &JxW = scratch_data.JxW;
402 const unsigned int n_q_points = scratch_data.n_q_points;
403 const unsigned int n_dofs = scratch_data.n_dofs;
404
405 // Copy data elements
406 auto &strong_residual = copy_data.strong_residual;
407 auto &local_rhs = copy_data.local_rhs;
408
409 // Time stepping information
410 const auto method = this->simulation_control->get_assembly_method();
411 // Vector for the BDF coefficients
412 const Vector<double> &bdf_coefs =
413 this->simulation_control->get_bdf_coefficients();
414 std::vector<double> pressure(1 + number_of_previous_solutions(method));
415 std::vector<Tensor<1, dim>> velocity(1 +
416 number_of_previous_solutions(method));
417
418 // Loop over the quadrature points
419 for (unsigned int q = 0; q < n_q_points; ++q)
420 {
421 // Physical properties
422 const double density = density_vector[q];
423
424 pressure[0] = scratch_data.pressure_values[q];
425 velocity[0] = scratch_data.velocity_values[q];
426 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
427 {
428 pressure[p + 1] = scratch_data.previous_pressure_values[p][q];
429 velocity[p + 1] = scratch_data.previous_velocity_values[p][q];
430 }
431
432 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
433 ++p)
434 strong_residual[q] += (bdf_coefs[p] * density * velocity[p]);
435
436
437 for (unsigned int i = 0; i < n_dofs; ++i)
438 {
439 const double phi_p_i = scratch_data.phi_p[q][i];
440 const Tensor<1, dim> &phi_u_i = scratch_data.phi_u[q][i];
441 double local_rhs_i = 0;
442 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
443 ++p)
444 {
445 local_rhs_i -=
446 bdf_coefs[p] * (phi_p_i * density_psi * pressure[p] +
447 phi_u_i * density * velocity[p]);
448 }
449 local_rhs(i) += local_rhs_i * JxW[q];
450 }
451 }
452}
