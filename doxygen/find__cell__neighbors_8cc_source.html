8template <int dim, bool reciprocal>
9void
10find_cell_neighbors(
11 const parallel::distributed::Triangulation<dim> &triangulation,
12 typename dem_data_structures<dim>::cells_neighbor_list
13 &cells_local_neighbor_list,
14 typename dem_data_structures<dim>::cells_neighbor_list
15 &cells_ghost_neighbor_list)
16{
17 // The output vectors of the function are cells_local_neighbor_list and
18 // cells_ghost_neighbor_list. The first contains all the local neighbors cells
19 // of all local cells; while the second contains all the ghost cells of all
20 // local cells. They are two vectors with size of the number of active cells.
21 // The first elements of all vectors are the main cells
22 typename dem_data_structures<dim>::cell_vector local_neighbor_vector;
23 typename dem_data_structures<dim>::cell_vector ghost_neighbor_vector;
24
25 // This vector is used to avoid repetition of adjacent cells. For instance if
26 // cell B is recognized as the neighbor of cell A, cell A will not be added to
27 // the neighbor list of cell B again. This is done using the total_cell_list
28 // vector
29 typename dem_data_structures<dim>::cell_set total_cell_list;
30
31 // For each cell, the cell vertices are found and used to find adjacent cells.
32 // The reason is to find the cells located on the corners of the main cell.
33 auto v_to_c = GridTools::vertex_to_cell_map(triangulation);
34
35 // Looping over cells
36 for (const auto &cell : triangulation.active_cell_iterators())
37 {
38 // If the cell is owned by the processor
39 if (cell->is_locally_owned())
40 {
41 // The first element of each vector is the cell itself.
42 local_neighbor_vector.push_back(cell);
43
44 if constexpr (!reciprocal)
45 total_cell_list.insert(cell);
46
47 for (unsigned int vertex = 0; vertex < cell->n_vertices(); ++vertex)
48 {
49 for (const auto &neighbor : v_to_c[cell->vertex_index(vertex)])
50 {
51 if (neighbor->is_locally_owned())
52 {
53 auto search_iterator = total_cell_list.find(neighbor);
54 auto local_search_iterator =
55 std::find(local_neighbor_vector.begin(),
56 local_neighbor_vector.end(),
57 neighbor);
58
59 // If the cell (neighbor) is a local cell and not present
60 // in the total_cell_list vector, it will be added as the
61 // neighbor of the main cell
62 // ("cell") and also to the total_cell_list to avoid
63 // repetition for next cells.
64 if (search_iterator == total_cell_list.end() &&
65 local_search_iterator == local_neighbor_vector.end())
66 {
67 local_neighbor_vector.push_back(neighbor);
68 }
69
70 // If the neighbor cell is a ghost, it should be added in
71 // the ghost_neighbor_vector container
72 }
73 else if (neighbor->is_ghost())
74 {
75 auto ghost_search_iterator =
76 std::find(ghost_neighbor_vector.begin(),
77 ghost_neighbor_vector.end(),
78 neighbor);
79 if (ghost_search_iterator == ghost_neighbor_vector.end())
80 {
81 if (ghost_neighbor_vector.empty())
82 {
83 ghost_neighbor_vector.push_back(cell);
84 }
85
86 ghost_neighbor_vector.push_back(neighbor);
87 }
88 }
89 }
90 }
91 }
92 if (!local_neighbor_vector.empty())
93 cells_local_neighbor_list.push_back(local_neighbor_vector);
94 if (!ghost_neighbor_vector.empty())
95 cells_ghost_neighbor_list.push_back(ghost_neighbor_vector);
96
97 local_neighbor_vector.clear();
98 ghost_neighbor_vector.clear();
99 }
100}

102template <int dim>
103void
104find_cell_periodic_neighbors(
105 const parallel::distributed::Triangulation<dim> &triangulation,
106 const typename dem_data_structures<dim>::periodic_boundaries_cells_info
107 &periodic_boundaries_cells_information,
108 typename dem_data_structures<dim>::cells_neighbor_list
109 &cells_local_periodic_neighbor_list,
110 typename dem_data_structures<dim>::cells_neighbor_list
111 &cells_ghost_periodic_neighbor_list,
112 typename dem_data_structures<dim>::cells_neighbor_list
113 &cells_ghost_local_periodic_neighbor_list)
114{
115 typename dem_data_structures<dim>::cell_vector local_periodic_neighbor_vector;
116 typename dem_data_structures<dim>::cell_vector ghost_periodic_neighbor_vector;
117 typename dem_data_structures<dim>::cell_vector
118 ghost_local_periodic_neighbor_vector;
119
120 typename dem_data_structures<dim>::cell_set total_cell_list;
121 typename dem_data_structures<dim>::cell_set total_ghost_cell_list;
122
123 // For each cell, the cell vertices are found and used to find adjacent cells.
124 // The reason is to find the cells located on the corners of the main cell.
125 auto v_to_c = GridTools::vertex_to_cell_map(triangulation);
126
127 // A map of coinciding vertices labeled by an arbitrary element from them
128 std::map<unsigned int, std::vector<unsigned int>> coinciding_vertex_groups;
129
130 // Map of a vertex to the label of a group of coinciding vertices
131 std::map<unsigned int, unsigned int> vertex_to_coinciding_vertex_group;
132
133 // Collect for a given triangulation all locally relevant vertices that
134 // coincide due to periodicity.
135 GridTools::collect_coinciding_vertices(triangulation,
136 coinciding_vertex_groups,
137 vertex_to_coinciding_vertex_group);
138
139 // Looping over cells struct at periodic boundaries 0
140 for (const auto &pb_cell_struct : periodic_boundaries_cells_information)
141 {
142 auto &cell = pb_cell_struct.second.cell;
143
144 // If the cell is owned by the processor
145 if (cell->is_locally_owned())
146 {
147 // The first element of each vector is the cell itself.
148 local_periodic_neighbor_vector.push_back(cell);
149
150 // Store every main cell
151 total_cell_list.insert(cell);
152
153 // Empty list of periodic cell neighbor
154 typename dem_data_structures<dim>::cell_vector periodic_neighbor_list;
155
156 // Get the periodic neighbor of the cell
157 get_periodic_neighbor_list<dim>(cell,
158 coinciding_vertex_groups,
159 vertex_to_coinciding_vertex_group,
160 v_to_c,
161 periodic_neighbor_list);
162
163 for (const auto &periodic_neighbor : periodic_neighbor_list)
164 {
165 if (periodic_neighbor->is_locally_owned())
166 {
167 // Check if the neighbor cell has been processed as a
168 // main cell
169 auto search_iterator =
170 total_cell_list.find(periodic_neighbor);
171
172 // Check if the neighbor cell is in already in the
173 // local_periodic_neighbor_vector
174 // Note from Gabo: I don't understand how this could happened
175 // since we are looping over cell on boundary 1.
176 // I think the only case would be if periodic_neighbor_list
177 // has duplicate cell. If this is the case, it is weird that
178 // get_periodic_neighbor_list returns a list with duplicates.
179 auto local_search_iterator =
180 std::find(local_periodic_neighbor_vector.begin(),
181 local_periodic_neighbor_vector.end(),
182 periodic_neighbor);
183
184 // If the cell neighbor is a local cell and not present
185 // in the total_cell_list vector, it will be added as the
186 // neighbor of the main cell.
187 if (search_iterator == total_cell_list.end() &&
188 local_search_iterator ==
189 local_periodic_neighbor_vector.end())
190 {
191 local_periodic_neighbor_vector.push_back(
192 periodic_neighbor);
193 }
194 }
195 else if (periodic_neighbor->is_ghost())
196 {
197 // If the cell neighbor is a ghost, it should be added in
198 // the ghost_periodic_neighbor_vector container
199 auto ghost_search_iterator =
200 std::find(ghost_periodic_neighbor_vector.begin(),
201 ghost_periodic_neighbor_vector.end(),
202 periodic_neighbor);
203 if (ghost_search_iterator ==
204 ghost_periodic_neighbor_vector.end())
205 {
206 if (ghost_periodic_neighbor_vector.empty())
207 {
208 ghost_periodic_neighbor_vector.push_back(cell);
209 }
210
211 ghost_periodic_neighbor_vector.push_back(
212 periodic_neighbor);
213 }
214 }
215 }
216 if (!local_periodic_neighbor_vector.empty())
217 cells_local_periodic_neighbor_list.push_back(
218 local_periodic_neighbor_vector);
219 if (!ghost_periodic_neighbor_vector.empty())
220 cells_ghost_periodic_neighbor_list.push_back(
221 ghost_periodic_neighbor_vector);
222 local_periodic_neighbor_vector.clear();
223 ghost_periodic_neighbor_vector.clear();
224 }
225 else if (cell->is_ghost())
226 {
227 // Since periodic cells are mapped on one side only (cells on pb 0
228 // with cells on pb 1), we need a 3rd container for ghost-local
229 // contacts for force calculation. Here we store local neighbors of
230 // ghost cells.
231
232 // The first element of each vector is the cell itself
233 ghost_local_periodic_neighbor_vector.push_back(cell);
234 total_ghost_cell_list.insert(cell);
235
236 // Empty list of periodic cell neighbor
237 typename dem_data_structures<dim>::cell_vector periodic_neighbor_list;
238
239 // Get the periodic neighbor of the cell
240 get_periodic_neighbor_list<dim>(cell,
241 coinciding_vertex_groups,
242 vertex_to_coinciding_vertex_group,
243 v_to_c,
244 periodic_neighbor_list);
245
246 for (const auto &periodic_neighbor : periodic_neighbor_list)
247 {
248 if (periodic_neighbor->is_locally_owned())
249 {
250 auto search_iterator =
251 total_ghost_cell_list.find(periodic_neighbor);
252
253 auto local_search_iterator =
254 std::find(ghost_local_periodic_neighbor_vector.begin(),
255 ghost_local_periodic_neighbor_vector.end(),
256 periodic_neighbor);
257
258 if (search_iterator == total_ghost_cell_list.end() &&
259 local_search_iterator ==
260 ghost_local_periodic_neighbor_vector.end())
261 {
262 ghost_local_periodic_neighbor_vector.push_back(
263 periodic_neighbor);
264 }
265 }
266 }
267 if (!ghost_local_periodic_neighbor_vector.empty())
268 cells_ghost_local_periodic_neighbor_list.push_back(
269 ghost_local_periodic_neighbor_vector);
270 ghost_local_periodic_neighbor_vector.clear();
271 }
272 }
273}

275// This function finds the full neighbor list (with repetition) of all the
276// active cells in the triangulation. Because particle-floating mesh
277// contacts need all the particles located in ALL the neighbor cells of
278// the main cell to search for possible collisions with the floating mesh.
279template <int dim>
280void
281find_full_cell_neighbors(
282 const parallel::distributed::Triangulation<dim> &triangulation,
283 typename dem_data_structures<dim>::cells_total_neighbor_list
284 &cells_total_neighbor_list)
285{
286 auto v_to_c = GridTools::vertex_to_cell_map(triangulation);
287
288 // Looping over cells
289 for (const auto &cell : triangulation.active_cell_iterators())
290 {
291 // If the cell is owned by the processor
292 if (cell->is_locally_owned())
293 {
294 std::vector<typename Triangulation<dim>::active_cell_iterator>
295 full_neighbor_vector(0);
296
297 full_neighbor_vector.push_back(cell);
298
299 for (unsigned int vertex = 0; vertex < cell->n_vertices(); ++vertex)
300 {
301 for (const auto &neighbor : v_to_c[cell->vertex_index(vertex)])
302 {
303 if (neighbor->is_locally_owned())
304 {
305 auto search_iterator =
306 std::find(full_neighbor_vector.begin(),
307 full_neighbor_vector.end(),
308 neighbor);
309
310 if (search_iterator == full_neighbor_vector.end())
311 {
312 full_neighbor_vector.push_back(neighbor);
313 }
314 }
315 }
316 }
317
318 cells_total_neighbor_list.insert(
319 {cell->global_active_cell_index(), full_neighbor_vector});
320 }
321 }
322}

324template <int dim>
325void
326get_periodic_neighbor_list(
327 const typename Triangulation<dim>::active_cell_iterator &cell,
328 const std::map<unsigned int, std::vector<unsigned int>>
329 &coinciding_vertex_groups,
330 const std::map<unsigned int, unsigned int> &vertex_to_coinciding_vertex_group,
331 const std::vector<std::set<typename Triangulation<dim>::active_cell_iterator>>
332 &v_to_c,
333 typename dem_data_structures<dim>::cell_vector &periodic_neighbor_list)
334{
335 // Loop over all vertices of the cell (periodic and non periodic)
336 for (unsigned int vertex = 0; vertex < cell->n_vertices(); ++vertex)
337 {
338 // Get global id of vertex
339 unsigned int vertex_id = cell->vertex_index(vertex);
340
341 // Check if vertex is at periodic boundary, there should be a key if so
342 if (vertex_to_coinciding_vertex_group.find(vertex_id) !=
343 vertex_to_coinciding_vertex_group.end())
344 {
345 // Get the coinciding vertex key to the current vertex
346 unsigned int coinciding_vertex_key =
347 vertex_to_coinciding_vertex_group.at(vertex_id);
348
349 // Store the neighbor cells in list
350 for (auto coinciding_vertex :
351 coinciding_vertex_groups.at(coinciding_vertex_key))
352 {
353 // Skip the current vertex since we want only cells linked
354 // to the periodic vertices
355 if (coinciding_vertex != vertex_id)
356 {
357 // Loop over all periodic neighbor
358 for (const auto &neighbor_id : v_to_c[coinciding_vertex])
359 {
360 periodic_neighbor_list.push_back(neighbor_id);
361 }
362 }
363 }
364 }
365 }
