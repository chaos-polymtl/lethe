21/**
22 * This class corresponds to a manipulator of the particles crossing periodic
23 * cells in DEM. It maps cell information of the pairs of periodic cells
24 * and manipulate the particles' location when they cross periodic boundaries,
25 * i.e. when a particle cross a periodic boundary (0 or 1) its location is
26 * modified with the offset between faces of periodic cells on periodic
27 * boundaries.
28 *
29 * @note Currently the code only supports periodic boundaries that are
30 * parallel and aligned with an axis of the domain (e.g., x axis).
31 */
32
33template <int dim>
34class PeriodicBoundariesManipulator
35{
36public:
37 PeriodicBoundariesManipulator();
38
39 /**
40 * @brief Sets the periodic boundaries parameters.
41 *
42 * @param[in] periodic_directions Map of perpendicular axes of each PB pair,
43 * keyed by the principal periodic boundary id (periodic id 0). The keys are
44 * the principal periodic boundary ids.
45 */
46 void
47 set_periodic_boundaries_information(
48 const std::map<types::boundary_id, unsigned int> &periodic_directions)
49 {
50 // If this function is reached and the map is not empty
51 if (periodic_directions.empty())
52 return;
53
54 periodic_boundaries_enabled = true;
55
56 // Communicate to the action manager that there are periodic boundaries
57 DEMActionManager::get_action_manager()->set_periodic_boundaries_enabled();
58
59 this->directions = periodic_directions;
60
61 // Initialize offset map
62 this->periodic_offsets.clear();
63 }
64
65 /**
66 * @brief Execute the mapping of the cells on periodic boundaries and store
67 * information in periodic_boundaries_cells_information.
68 *
69 * @param[in] triangulation Triangulation.
70 * @param[out] periodic_boundaries_cells_information Map (multimap) of
71 * information of the pair of cells on periodic boundaries.
72 */
73 void
74 map_periodic_cells(
75 const parallel::distributed::Triangulation<dim> &triangulation,
76 typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
77 &periodic_boundaries_cells_information);
78
79 /**
80 * @brief Moves particles crossing periodic boundaries (any side)
81 * particle_handler doesn't allow automated particle displacement since it is
82 * not linked to triangulation and its periodic mapping.
83 *
84 * @param[in,out] particle_handler Particle handler of particles located in
85 * boundary cells.
86 * @param[in] periodic_boundaries_cells_information Map of information of the
87 * pair of cells on periodic boundaries.
88 *
89 * @return Flag if at least one particle has been moved to the other periodic
90 * cell.
91 */
92 bool
93 execute_particles_displacement(
94 const Particles::ParticleHandler<dim> &particle_handler,
95 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
96 &periodic_boundaries_cells_information);
97
98 /**
99 * @brief Return the periodic offset distance for a specific pb0 boundary ID.
100 * All pairs of cells on that pair of periodic boundaries are assumed to have
101 * the same offset. If no periodic offset has been identified for the
102 * boundary id, a zero tensor is inserted as a default value.
103 *
104 * @param[in] boundary_id ID of the boundary to query.
105 *
106 * @return Offset distance between periodic boundaries.
107 */
108 inline const Tensor<1, dim> &
109 get_periodic_offset_distance(const types::boundary_id boundary_id)
110 {
111 // If no offset was identified during looping over the cell, a zero
112 // tensor is returned as a default offset.
113 if (!periodic_offsets.contains(boundary_id))
114 {
115 Tensor<1, dim> zero_tensor;
116 periodic_offsets.insert(std::make_pair(boundary_id, zero_tensor));
117 }
118 return periodic_offsets.at(boundary_id);
119 }
120
121 /**
122 * @brief Return the directions of the periodic boundary pairs, keyed by the
123 * principal periodic boundary id (periodic id 0). The keys are the principal
124 * periodic boundary ids.
125 *
126 * @return Map of periodic directions keyed by principal periodic boundary id.
127 */
128 inline const std::map<types::boundary_id, unsigned int> &
129 get_periodic_directions() const
130 {
131 return directions;
132 }
133
134 /**
135 * @brief Return the combined periodic offsets
136 *
137 * @return Combined periodic offsets.
138 */
139 inline const std::vector<Tensor<1, dim>> &
140 get_combined_periodic_offsets() const
141 {
142 return combined_periodic_offsets;
143 }
144
145private:
146 /**
147 * @brief Get boundary information related to a face at a periodic
148 * boundary and store it in a periodic_boundaries_cells_info_struct
149 * object.
150 *
151 * @param[in] cell Current cell on boundary.
152 * @param[in] face_id Face located on boundary.
153 * @param[out] boundaries_information Reference to the object with periodic
154 * boundary information.
155 */
156 void
157 get_periodic_boundaries_info(
158 typename Triangulation<dim>::cell_iterator cell,
159 const unsigned int face_id,
160 periodic_boundaries_cells_info_struct<dim> &boundaries_information);
161
162 /**
163 * @brief Check if particle is outside the cell, if so, modify the
164 * location of the particle using the relevant periodic offset for the
165 * boundary.
166 *
167 * @param[in] boundaries_cells_content Reference to the object with periodic
168 * boundary information.
169 * @param[in] particles_in_pb0_cell If the particles are linked to a cell on
170 * the periodic boundary 0 (true) or the periodic boundary 1 (false).
171 * @param[in,out] particles_in_cell Iterator to the particles in cell.
172 * @param[out] particle_has_been_moved Flag the particle has been moved to
173 * the other periodic cell.
174 */
175 void
176 check_and_move_particles(
177 const periodic_boundaries_cells_info_struct<dim> &boundaries_cells_content,
178 const bool &particles_in_pb0_cell,
179 typename Particles::ParticleHandler<dim>::particle_iterator_range
180 &particles_in_cell,
181 bool &particle_has_been_moved);
182
183 /**
184 * @brief Compute the combined periodic offsets and store them in
185 * combined_periodic_offsets.
186 */
187 void
188 compute_combined_periodic_offsets();
189
190 /**
191 * @brief Flag for periodic boundary conditions in simulation. Useful to
192 * exit function when there are no periodic boundaries.
193 */
194 bool periodic_boundaries_enabled;
195
196 /**
197 * @brief Direction of the periodic boundaries, it is the perpendicular axis
198 * of the periodic boundaries. Keys of this map are the principal periodic
199 * boundary ids (periodic id 0).
200 */
201 std::map<types::boundary_id, unsigned int> directions;
202
203 /**
204 * @brief Map storing offset distance between periodic boundaries, keyed by
205 * the boundary ID (pb0). It is calculated from the first pair of cells on
206 * periodic boundaries, so all pairs of cells on a given peridodic boundary
207 * are assumed to have the same offset.
208 */
209 std::unordered_map<types::boundary_id, Tensor<1, dim>> periodic_offsets;
210
211 /**
212 * @brief Storage for all 9 (2D) or 27 (3D) precomputed periodic translation
213 * vectors. Calculated from periodic_offsets.
214 */
215 std::vector<Tensor<1, dim>> combined_periodic_offsets;
216};
