369/**
370 * @brief Compute outer product for the multicomponent case
371 *
372 * @param[in] grad Rank-1 tensor
373 * @param[in] normal Rank-1 tensor
374 *
375 * @return Rank-0 tensor
376 *
377 */
378template <int n_components, int dim, typename Number>
379Tensor<1, n_components, Number>
380contract(const Tensor<1, n_components, Tensor<1, dim, Number>> &grad,
381 const Tensor<1, dim, Number> &normal)
382{
383 Tensor<1, n_components, Number> result;
384
385 for (int r = 0; r < n_components; ++r)
386 result[r] = grad[r] * normal;
387
388 return result;
389}

448/**
449 * @brief Compute scalar product
450 *
451 * @param[in, out] v_gradient Rank-1 tensor within rank-1 tensor where result is
452 * stored
453 * @param[in] u_gradient Rank-2 tensor
454 * @param[in] factor Scalar factor
455 */
456template <int dim, int dim_, typename Number>
457inline DEAL_II_ALWAYS_INLINE void
458symm_scalar_product_add(Tensor<1, dim_, Tensor<1, dim, Number>> &v_gradient,
459 const Tensor<2, dim, Number> &u_gradient,
460 const Number &factor)
461{
462 for (unsigned int d = 0; d < dim; ++d)
463 v_gradient[d][d] += u_gradient[d][d] * factor;
464
465 for (unsigned int e = 0; e < dim; ++e)
466 for (unsigned int d = e + 1; d < dim; ++d)
467 {
468 const auto tmp = (u_gradient[d][e] + u_gradient[e][d]) * (factor * 0.5);
469 v_gradient[d][e] += tmp;
470 v_gradient[e][d] += tmp;
471 }
472}
473
474/**
475 * @brief Compute scalar product
476 *
477 * @param[in, out] v_gradient Rank-2 tensor where result is stored
478 * @param[in] u_gradient Rank-2 tensor
479 * @param[in] factor Scalar factor
480 */
481template <int dim, typename Number>
482inline DEAL_II_ALWAYS_INLINE void
483symm_scalar_product_add(Tensor<2, dim, Number> &v_gradient,
484 const Tensor<2, dim, Number> &u_gradient,
485 const Number &factor)
486{
487 for (unsigned int d = 0; d < dim; ++d)
488 v_gradient[d][d] += u_gradient[d][d] * factor;
489
490 for (unsigned int e = 0; e < dim; ++e)
491 for (unsigned int d = e + 1; d < dim; ++d)
492 {
493 const auto tmp = (u_gradient[d][e] + u_gradient[e][d]) * (factor * 0.5);
494 v_gradient[d][e] += tmp;
495 v_gradient[e][d] += tmp;
496 }
497}

499/**
500 * @brief Compute scalar product
501 *
502 * @param[in, out] v_gradient Rank-2 tensor where result is stored
503 * @param[in] u_gradient Rank-2 tensor
504 * @param[in] factor Scalar factor
505 */
506template <typename Number>
507inline DEAL_II_ALWAYS_INLINE void
508symm_scalar_product_add(Tensor<1, 1, Number> &v_gradient,
509 const Tensor<1, 1, Number> &u_gradient,
510 const Number &factor)
511{
512 v_gradient[0] += u_gradient[0] * factor;
513}
