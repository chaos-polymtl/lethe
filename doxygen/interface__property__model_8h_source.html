12/**
13 * @brief Abstract class that defines the interface for a interface property model.
14 * Interface property model provides an abstract interface to calculate the
15 * value of an interface property or a vector of interface property value for
16 * given field value. By default, the interface does not require that all (or
17 * any) fields be specified. This is why a map is used to pass the dependent
18 * variable. To allow for the calculation of the appropriate jacobian matrix
19 * (when that is necessary) the interface also provides a jacobian function,
20 * which must provide the derivative with respect to the field specified as an
21 * argument.
22 */
23class InterfacePropertyModel
24{
25public:
26 /**
27 * @brief class Default constructor. Set the model_depends_on to false for all variables.
28 */
29 InterfacePropertyModel()
30 {
31 model_depends_on[shear_rate] = false;
32 model_depends_on[temperature] = false;
33 model_depends_on[temperature_p1] = false;
34 model_depends_on[pressure] = false;
35 model_depends_on[phase_order_cahn_hilliard] = false;
36 }

95 /**
96 * @brief numerical_jacobian Calculates the jacobian through a forward finite difference (Euler) approximation.
97 * This approach, although not preferable, is meant as a fall-back when
98 * calculating the jacobian manually is too difficult.
99 * @param fields_values The values of the various fields
100 * @param id Field id of the field with respect to which the jacobian should be calculated
101 * @return
102 */
103 inline double
104 numerical_jacobian(const std::map<field, double> &field_values,
105 const field id)
106 {
107 double f_x = this->value(field_values);
108 auto x_dx = field_values;
109 double dx = std::max(1e-6 * x_dx[id], 1e-8);
110 x_dx[id] = x_dx[id] + dx;
111 double f_xdx = this->value(x_dx);
112 return (f_xdx - f_x) / dx;
113 }
114
115 /**
116 * @brief vector_numerical_jacobian Calculates the vector of jacobian through forward finite difference (Euler) approximation.
117 * This approach, although not preferable, is meant as a fall-back when
118 * calculating the jacobian manually is too difficult.
119 * @param field_vectors
120 * @param field_id
121 * @param jacobian_vector
122 * @return
123 */
124 inline void
125 vector_numerical_jacobian(
126 const std::map<field, std::vector<double>> &field_vectors,
127 const field id,
128 std::vector<double> &jacobian_vector)
129 {
130 const unsigned int n_pts = jacobian_vector.size();
131
132 // Evaluate the properties using the values of the field vector
133 std::vector<double> f_x(n_pts);
134 vector_value(field_vectors, f_x);
135
136 // Make a copy of the field vector for the field we will perturbate
137 std::map<field, std::vector<double>> perturbed_field_vectors =
138 field_vectors;
139 std::vector<double> &x = perturbed_field_vectors.at(id);
140 std::vector<double> dx(n_pts);
141
142 // Perturb the field by dx
143 for (unsigned int i = 0; i < n_pts; ++i)
144 {
145 dx[i] = std::max(1e-6 * x[i], 1e-8);
146 x[i] += dx[i];
147 }
148
149 // Evaluate the properties using the perturbed value of the field vector
150 std::vector<double> f_xdx(n_pts);
151 vector_value(perturbed_field_vectors, f_xdx);
152
153 // Fill jacobian
154 for (unsigned int i = 0; i < n_pts; ++i)
155 jacobian_vector[i] = (f_xdx[i] - f_x[i]) / dx[i];
156 }
157
158protected:
159 // Map to indicate on which variables the model depends on
160 std::map<field, bool> model_depends_on;
161};
