100/**
101 * @brief Diffusivity that depends on the tanh of the level set
102 */
103class TanhLevelsetTracerDiffusivity : public TracerDiffusivityModel
104{
105public:
106 /**
107 * @brief Constructor of the level set-dependent diffusivity model.
108 *
109 * @param[in] p_tracer_diffusivity_outside Diffusivity outside the solid
110 * @param[in] p_tracer_diffusivity_inside Diffusivity inside the solid
111 * @param[in] p_thickness Thickness of the tanh function used to smooth the
112 * property jump
113 */
114 TanhLevelsetTracerDiffusivity(const double p_tracer_diffusivity_outside,
115 const double p_tracer_diffusivity_inside,
116 const double p_thickness)
117 : tracer_diffusivity_outside(p_tracer_diffusivity_outside)
118 , tracer_diffusivity_inside(p_tracer_diffusivity_inside)
119 , thickness(p_thickness)
120 , delta_diffusivity(tracer_diffusivity_outside - tracer_diffusivity_inside)
121 {
122 this->model_depends_on[field::levelset] = true;
123 }
124
125 /**
126 * @brief Compute the diffusivity.
127 *
128 * @param[in] field_values Values of the various fields on which the property
129 * may depend. In this case, the diffusivity depends on the level set.
130 * The map stores a single value per field.
131 *
132 * @return Value of the diffusivity computed with the @p field_values.
133 */
134 double
135 value(const std::map<field, double> &field_values) override
136 {
137 Assert(field_values.find(field::levelset) != field_values.end(),
138 PhysicialPropertyModelFieldUndefined("TanhLevelsetTracerDiffusivity",
139 "levelset"));
140 double levelset = field_values.at(field::levelset);
141
142 return tracer_diffusivity_inside +
143 delta_diffusivity * (0.5 + 0.5 * tanh(levelset / thickness));
144 }
145
146 /**
147 * @brief Compute a vector of diffusivity.
148 *
149 * @param[in] field_vectors Vectors of the fields on which the diffusivity
150 * may depend. In this case, the diffusivity depends on the level set. The map
151 * stores a vector of values per field.
152 *
153 * @param[out] property_vector Vectors of computed diffusivities.
154 */
155 void
156 vector_value(const std::map<field, std::vector<double>> &field_vectors,
157 std::vector<double> &property_vector) override
158 {
159 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
160 PhysicialPropertyModelFieldUndefined("TanhLevelsetTracerDiffusivity",
161 "levelset"));
162
163 const std::vector<double> &levelset_vec = field_vectors.at(field::levelset);
164
165 const unsigned int n_values = property_vector.size();
166
167 Assert(n_values == levelset_vec.size(),
168 SizeOfFields(n_values, levelset_vec.size()));
169
170 for (unsigned int i = 0; i < n_values; ++i)
171 {
172 const double levelset = levelset_vec[i];
173 property_vector[i] =
174 tracer_diffusivity_inside +
175 delta_diffusivity * (0.5 + 0.5 * tanh(levelset / thickness));
176 }
177 }
178
179 /**
180 * @brief Compute the jacobian (the partial derivative) of the diffusivity
181 * with respect to a specified field.
182 *
183 * @param[in] field_values Values of the various fields on which the specific
184 * heat may depend. The map stores a single value per field.
185 *
186 * @param[in] id Indicator of the field with respect to which the jacobian
187 * should be computed.
188 *
189 * @return Value of the derivative of the diffusivity with respect to the
190 * specified field.
191 */
192 double
193 jacobian(const std::map<field, double> &field_values, field id) override
194 {
195 if (id == field::levelset)
196 {
197 Assert(field_values.find(field::levelset) != field_values.end(),
198 PhysicialPropertyModelFieldUndefined(
199 "TanhLevelsetTracerDiffusivity", "levelset"));
200 return numerical_jacobian(field_values, field::levelset);
201 }
202 else
203 return 0;
204 };
205
206 /**
207 * @brief Compute the derivative of the diffusivity with respect to a field.
208 *
209 * @param[in] field_vectors Vector of values of the fields used to evaluate
210 * the diffusivity. The map stores a vector of values per field.
211 *
212 * @param[in] id Identifier of the field with respect to which a derivative
213 * should be computed.
214 *
215 * @param[out] jacobian Vector of computed derivative values of the
216 * diffusivity with respect to the field of the specified @p id.
217 */
218 void
219 vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
220 const field id,
221 std::vector<double> &jacobian_vector) override
222 {
223 if (id == field::levelset)
224 {
225 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
226 PhysicialPropertyModelFieldUndefined(
227 "TanhLevelsetTracerDiffusivity", "levelset"));
228 vector_numerical_jacobian(field_vectors, id, jacobian_vector);
229 }
230 else
231 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0.);
232 };
233
234
235private:
236 const double tracer_diffusivity_outside;
237 const double tracer_diffusivity_inside;
238 const double thickness;
239 const double delta_diffusivity;
240};

242/**
243 * @brief Diffusivity that depends on the Gaussian of the level set
244 */
245class GaussianLevelsetTracerDiffusivity : public TracerDiffusivityModel
246{
247public:
248 /**
249 * @brief Constructor of the Gaussian level set-dependent diffusivity model.
250 *
251 * @param[in] p_tracer_diffusivity_interface Diffusivity at the fluid-solid
252 * interface the solid
253 * @param[in] p_tracer_diffusivity_bulk Diffusivity in the bulk of the phases
254 * @param[in] p_thickness Thickness of the Gaussian function used to smooth
255 * the property jump
256 */
257 GaussianLevelsetTracerDiffusivity(const double p_tracer_diffusivity_interface,
258 const double p_tracer_diffusivity_bulk,
259 const double p_thickness)
260 : tracer_diffusivity_interface(p_tracer_diffusivity_interface)
261 , tracer_diffusivity_bulk(p_tracer_diffusivity_bulk)
262 , delta_diffusivity(tracer_diffusivity_interface - tracer_diffusivity_bulk)
263 , squared_thickness(pow(p_thickness, 2))
264 {
265 this->model_depends_on[field::levelset] = true;
266 }
267
268 /**
269 * @brief Compute the diffusivity.
270 *
271 * @param[in] field_values Values of the various fields on which the property
272 * may depend. In this case, the diffusivity depends on the level set.
273 * The map stores a single value per field.
274 *
275 * @return Value of the diffusivity computed with the @p field_values.
276 */
277 double
278 value(const std::map<field, double> &field_values) override
279 {
280 Assert(field_values.find(field::levelset) != field_values.end(),
281 PhysicialPropertyModelFieldUndefined(
282 "GaussianLevelsetTracerDiffusivity", "levelset"));
283 double levelset = field_values.at(field::levelset);
284
285 return tracer_diffusivity_bulk +
286 delta_diffusivity * exp(-(pow(levelset, 2)) / squared_thickness);
287 }
288
289 /**
290 * @brief Compute a vector of diffusivity.
291 *
292 * @param[in] field_vectors Vectors of the fields on which the diffusivity
293 * may depend. In this case, the diffusivity depends on the level set. The map
294 * stores a vector of values per field.
295 *
296 * @param[out] property_vector Vectors of computed diffusivities.
297 */
298 void
299 vector_value(const std::map<field, std::vector<double>> &field_vectors,
300 std::vector<double> &property_vector) override
301 {
302 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
303 PhysicialPropertyModelFieldUndefined(
304 "GaussianLevelsetTracerDiffusivity", "levelset"));
305
306 const std::vector<double> &levelset_vec = field_vectors.at(field::levelset);
307
308 const unsigned int n_values = property_vector.size();
309
310 Assert(n_values == levelset_vec.size(),
311 SizeOfFields(n_values, levelset_vec.size()));
312
313 for (unsigned int i = 0; i < n_values; ++i)
314 {
315 const double levelset = levelset_vec[i];
316 property_vector[i] =
317 tracer_diffusivity_bulk +
318 delta_diffusivity * exp(-(pow(levelset, 2)) / squared_thickness);
319 ;
320 }
321 }
322
323 /**
324 * @brief Compute the jacobian (the partial derivative) of the diffusivity
325 * with respect to a specified field.
326 *
327 * @param[in] field_values Values of the various fields on which the specific
328 * heat may depend. The map stores a single value per field.
329 *
330 * @param[in] id Indicator of the field with respect to which the jacobian
331 * should be computed.
332 *
333 * @return Value of the derivative of the diffusivity with respect to the
334 * specified field.
335 */
336 double
337 jacobian(const std::map<field, double> &field_values, field id) override
338 {
339 if (id == field::levelset)
340 {
341 Assert(field_values.find(field::levelset) != field_values.end(),
342 PhysicialPropertyModelFieldUndefined(
343 "GaussianLevelsetTracerDiffusivity", "levelset"));
344
345 const double levelset = field_values.at(field::levelset);
346 const double exponential =
347 std::exp(-(levelset * levelset) / squared_thickness);
348
349 return delta_diffusivity * exponential *
350 (-2.0 * levelset / squared_thickness);
351 }
352 else
353 return 0.0;
354 }
355
356 /**
357 * @brief Compute the derivative of the diffusivity with respect to a field.
358 *
359 * @param[in] field_vectors Vector of values of the fields used to evaluate
360 * the diffusivity. The map stores a vector of values per field.
361 *
362 * @param[in] id Identifier of the field with respect to which a derivative
363 * should be computed.
364 *
365 * @param[out] jacobian Vector of computed derivative values of the
366 * diffusivity with respect to the field of the specified @p id.
367 */
368 void
369 vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
370 const field id,
371 std::vector<double> &jacobian_vector) override
372 {
373 if (id == field::levelset)
374 {
375 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
376 PhysicialPropertyModelFieldUndefined(
377 "GaussianLevelsetTracerDiffusivity", "levelset"));
378
379 const std::vector<double> &levelset_vec =
380 field_vectors.at(field::levelset);
381 const unsigned int n = jacobian_vector.size();
382
383 for (unsigned int i = 0; i < n; ++i)
384 {
385 const double levelset = levelset_vec[i];
386 const double exponential =
387 std::exp(-pow(levelset, 2) / squared_thickness);
388 jacobian_vector[i] = delta_diffusivity * exponential *
389 (-2.0 * levelset / squared_thickness);
390 }
391 }
392 else
393 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0.0);
394 }
395
396private:
397 const double tracer_diffusivity_interface;
398 const double tracer_diffusivity_bulk;
399 const double delta_diffusivity;
400 const double squared_thickness;
401};
