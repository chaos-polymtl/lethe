10/**
11 * @brief The SimulationControl class is responsible for the control of steady-state and transient
12 * simulations carried out with Lethe. This base class is a pure virtual class
13 * and cannot be instantiated. However, it stores the core variables which are
14 * necessary for its serialization (write and read)
15 **/
16
17class SimulationControl
18{
19protected:
20 // Time stepping method used for the simulation
21 Parameters::SimulationControl::TimeSteppingMethod method;
22
23 // Current method used for the present assembly
24 // This is used to differentiate the substeps of SDIRK methods or
25 // to start-up BDF simulations
26 Parameters::SimulationControl::TimeSteppingMethod assembly_method;
27
28 // Time of the current iteration being solved for
29 double current_time;
30
31 // Time of the previous iteration
32 double previous_time;
33
34 // Time step linking the previous iteration and the current time
35 double time_step;
36
37 // Initial time step given in the parameter
38 double initial_time_step;
39
40 // Simulation end time
41 double end_time;
42
43 // Boolean to keep the time step for the last iteration regardless of the end
44 // time specify. Both for fixed time step and adaptive time step.
45 bool time_step_independent_of_end_time;
46
47 // Time step vector. This vector accumulates the time steps of the previous
48 // iterations. This is required for multiple steps methods such as the bdfs.
49 std::vector<double> time_step_vector;
50
51 // Iteration. Iterations start at 0, but the first actual iteration is
52 // iteration 1.
53 unsigned int iteration_number;
54
55 // Number of mesh adaptation iteration
56 unsigned int number_mesh_adapt;
57
58 // Courant-Friedrich-Levy (CFL) condition
59 // Since the simulation control is unaware of the information propagation
60 // mechanism (for instance the velocity), the current CFL must be set by the
61 // solver itself

63
64 // Maximal CFL condition. This is used to control adaptative time stepping. In
65 // the case of constant time stepping or steady-state simulations, this
66 // parameter remains unused.
67 double max_CFL;
68
69 // Current value of the norm of the rhs
70 double residual;
71
72 // Residual at which the simulation stops when adjoint time-stepping is used
73 double stop_tolerance;
74
75 // Number of time steps stored
76 // BDF methods require a number of previous time steps. This number is known a
77 // priori and depends on the method used. We do not keep all the time steps to
78 // prevent the accumulation within a large vector.
79 static const unsigned int n_previous_time_steps = 4;
80
81 // Output iteration frequency
82 // Controls the output of the simulation results when the output is controlled
83 // by the iteration number.
84 unsigned int output_iteration_frequency;
85
86 // Log iteration frequency
87 // Controls the frequency at which status of the simulation is written to
88 // the terminal

110 bool output_boundaries;
111
112 // Indicator to tell if this is the first assembly of a step
113 bool first_assembly;
114
115 // The method use to start high order bdf scheme

117
118 // The time scaling used to do small time-steps at the startup of the
119 // simulation
120 double startup_timestep_scaling;

482 std::vector<double>
483 get_simulation_times() const
484 {
485 // Create a vector of the previous times
486 std::vector<double> times(n_previous_time_steps + 1);
487 for (unsigned int i = 0; i < n_previous_time_steps + 1; ++i)
488 {
489 times[i] = current_time;
490 for (unsigned int j = 0; j < i; ++j)
491 times[i] -= time_step_vector[j];
492 }

538class SimulationControlTransient : public SimulationControl
539{
540protected:
541 // Enable adaptative time stepping
542 bool adapt;
543
544 // Time step scaling for adaptative time stepping
545 double adaptative_time_step_scaling;
546
547 // Max time step
548 double max_dt;
549
550 // Time last output
551 double time_last_output;
552
553 // Specific output frequency for time output control
554 double output_time_frequency;
555
556 // Specific output times for time output control
557 std::vector<double> output_times_vector;
558
559 // Counter to move between output times given in previous vector
560 unsigned int output_times_counter;
561
562 // Variable to check whether we still have specific times to check in output
563 // times vector
564 bool no_more_output_times;
565
566 // Time interval for output of transient iterations either with time output
567 // control or iterations control
568 std::vector<double> output_time_interval;
569
570 // Output control type: iteration or type
571 Parameters::SimulationControl::OutputControl output_control;
572
573 /**
574 * @brief Calculates the next value of the time step. If adaptation
575 * is enabled, the time step is calculated in order to ensure
576 * that the CFL condition is bound by the maximal CFL value.
577 * The new time step is equal to adaptative_time_step_scaling * the previous
578 * time step. If the time_step_independent_of_end_time is set to false, the
579 * time step is asjusted to meet exactly the end time; the default is to not
580 * modify the time step.
581 */
582 virtual double
583 calculate_time_step() override;
584
585public:
586 SimulationControlTransient(const Parameters::SimulationControl &param);
587
588 virtual void
589 print_progression(const ConditionalOStream &pcout) override;
590
591 /** @brief Indicate if the simulation uses adaptive time stepping or not.
592 *
593 * @return adapt A boolean indicating if the simulation has adaptive time stepping
594 */
595 bool
596 is_adaptive_time_stepping() const override
597 {
598 return adapt;
599 }
600
601 /**
602 * @brief Proceeds with the simulation until the end condition is reached
603 */
604 virtual bool
605 integrate() override;
606
607 /**
608 * @brief Ends the simulation when the end time is reached
609 */
610 virtual bool
611 is_at_end() override;
612
613 /**
614 * @brief Output iterations are calculated based on the value of the output time
615 * or the output interval and the output frequency within the interval.
616 */
617 virtual bool
618 is_output_iteration() override;
619
620 /**
621 * @brief Save the simulation control information from the checkpoint file and
622 * updates the time step vector, the CFL value, the time, the iteration number
623 * and the index of the output times vector if time control is used.
624 *
625 * @param prefix The prefix of the checkpoint of the simulation
626 */
627 void
628 save(const std::string &prefix) override;
629
630 /**
631 * @brief Reads the simulation control information from the checkpoint file and
632 * updates the time step vector, the CFL value, the time, the iteration number
633 * and the index of the output times vector if time control is used. Allows to
634 * change time step if adaptive time stepping is disabled.
635 *
636 * @param prefix The prefix of the checkpoint of the simulation
637 */
638 void
639 read(const std::string &prefix) override;
640};
