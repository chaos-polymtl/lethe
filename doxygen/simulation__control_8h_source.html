8 * This file defines the simulation control hierarchy used in Lethe to manage
9 * steady-state and transient simulations. The SimulationControl base class and
10 * its derived classes handle time-stepping methods (including BDF and SDIRK),
11 * adaptive time stepping, output control, and simulation progression logic.
12 */

22/**
23 * @brief Base class for controlling steady-state and transient simulations in Lethe
24 *
25 * The SimulationControl class is responsible for managing the progression of
26 * simulations carried out with Lethe. This is a pure virtual base class that
27 * cannot be instantiated directly. It stores core variables necessary for
28 * time-stepping control, output management, and serialization.
29 *
30 * The class supports multiple time-stepping methods including:
31 * - Backward Differentiation Formula (BDF) methods: BDF1, BDF2, BDF3
32 * - Singly Diagonally Implicit Runge-Kutta (SDIRK) methods
33 * - Steady-state methods
34 *
35 * Derived classes implement specific simulation control strategies for
36 * transient, steady-state, adjoint, and specialized simulations (e.g., DEM, ray
37 * tracing).
38 *
39 * @note This class handles serialization through save() and read() methods
40 * to enable checkpointing and restart capabilities.
41 */
42class SimulationControl
43{
44protected:
45 /// Time stepping method used for the simulation (BDF, SDIRK, steady, etc.)
46 Parameters::SimulationControl::TimeSteppingMethod method;
47
48 /**
49 * @brief Current method used for the present assembly
50 *
51 * This is used to differentiate the substeps of SDIRK methods or
52 * to start-up BDF simulations. For example, BDF2 and BDF3 schemes
53 * may start with lower-order methods before transitioning to the
54 * full-order scheme.
55 */
56 Parameters::SimulationControl::TimeSteppingMethod assembly_method;
57
58 /// Time of the current iteration being solved for
59 double current_time;
60

73 /**
74 * @brief Boolean to keep the time step for the last iteration regardless of the end time
75 *
76 * When true, the last time step is not adjusted to match the end time
77 * exactly. Applies to both fixed time step and adaptive time step
78 * simulations.
79 */
80 bool time_step_independent_of_end_time;

82 /**
83 * @brief Time step vector storing previous time steps
84 *
85 * This vector accumulates the time steps of the previous iterations.

87 * The vector size is limited by n_previous_time_steps to prevent
88 * accumulation within a large vector.
89 */
90 std::vector<double> time_step_vector;
91
92 /**
93 * @brief Iteration number
94 *
95 * Iterations start at 0, but the first actual iteration is iteration 1.

97 unsigned int iteration_number;

99 /// Number of mesh adaptation iterations performed
100 unsigned int number_mesh_adapt;
101
102 /**
103 * @brief Courant-Friedrich-Levy (CFL) condition value

106 * mechanism (for instance the velocity), the current CFL must be set by the
107 * solver itself using set_CFL().

109 double CFL;
110
111 /**
112 * @brief Maximal CFL condition for adaptive time stepping
113 *
114 * This is used to control adaptive time stepping. In the case of constant

119 /// Current value of the norm of the right-hand side residual

122 /**

129 /**
130 * @brief Number of time steps stored

132 * BDF methods require a number of previous time steps. This number is known a

138 static constexpr unsigned int n_previous_time_steps = 4;

140 /**
141 * @brief Output iteration frequency

148 /**
149 * @brief Log iteration frequency

678 /**
679 * @brief Get simulation times for current and previous time steps
680 *
681 * Creates a vector containing the current time and previous times based on
682 * the time step history. The times are calculated by subtracting the
683 * accumulated time steps from the current time.
684 *
685 * @return Vector of simulation times (current and previous)
686 */
687 std::vector<double>
688 get_simulation_times() const
689 {
690 // Create a vector of the previous times
691 std::vector<double> times(n_previous_time_steps + 1);
692 for (unsigned int i = 0; i < n_previous_time_steps + 1; ++i)
693 {
694 times[i] = current_time;
695 for (unsigned int j = 0; j < i; ++j)
696 times[i] -= time_step_vector[j];
697 }
698 return times;
699 }

746 virtual void

766/**
767 * @brief Simulation control for transient simulations
768 *
769 * This class extends SimulationControl to handle transient (time-dependent)
770 * simulations. It implements adaptive time stepping based on CFL conditions,
771 * time-based output control, and ensures the simulation progresses until the
772 * end time is reached.
773 *
774 * Key features:
775 * - Adaptive time stepping with CFL-based control
776 * - Time-based or iteration-based output control
777 * - Specific output times capability
778 * - Maximum time step limiting
779 */
780class SimulationControlTransient : public SimulationControl
781{
782protected:
783 /// Enable adaptive time stepping
784 bool adapt;
785
786 /**
787 * @brief Time step scaling factor for adaptive time stepping
788 *
789 * Used to scale the time step based on the ratio of max_CFL to current CFL
790 * when adaptive time stepping is enabled.
791 */
792 double adaptative_time_step_scaling;
793
794 /// Maximum allowed time step
795 double max_dt;
796
797 /// Time of the last output
798 double time_last_output;
799
800 /// Output frequency for time-based output control
801 double output_time_frequency;
802
803 /**
804 * @brief Specific output times for time output control
805 *
806 * Vector containing specific times at which output should be generated.
807 */
808 std::vector<double> output_times_vector;
809
810 /// Counter to move between output times given in output_times_vector
811 unsigned int output_times_counter;
812
813 /**
814 * @brief Flag indicating whether there are more output times to check
815 *
816 * Set to true when all specific output times in output_times_vector have been
817 * reached.
818 */
819 bool no_more_output_times;
820
821 /**
822 * @brief Flag to override the time-step with the set value upon restart
823 */
824 bool override_time_step_on_restart;
825
826 /**
827 * @brief Time interval for output of transient iterations
828 *
829 * Used with either time output control or iteration control to define
830 * intervals for output generation.
831 */
832 std::vector<double> output_time_interval;
833
834 /// Output control type: iteration-based or time-based
835 Parameters::SimulationControl::OutputControl output_control;
836
837 /**
838 * @brief Calculate the next value of the time step
839 *
840 * If adaptation is enabled, the time step is calculated to ensure that the
841 * CFL condition is bound by the maximal CFL value. The new time step is
842 * scaled by adaptative_time_step_scaling. If the
843 * time_step_independent_of_end_time is set to false, the time step is
844 * adjusted to meet exactly the end time; the default is to not modify the
845 * time step.
846 *
847 * @return The calculated time step value
848 */
849 virtual double
850 calculate_time_step() override;

852public:

909 /**
910 * @brief Save the simulation control information to a checkpoint file
