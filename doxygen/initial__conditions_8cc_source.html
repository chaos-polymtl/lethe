123 Patterns::Double(),

126

129 prm.leave_subsection();

133 prm.leave_subsection();

137 prm.declare_entry(
138 "smoothing type",
139 "none",
140 Patterns::Selection("none|diffusive|geometric"),
141 "Apply a projection step with diffusion to smooth the VOF initial condition");
142
143 prm.declare_entry(
144 "diffusion factor",
145 "1",
146 Patterns::Double(),
147 "Factor applied to the diffusion term in the projection step");
148 prm.leave_subsection();

152 prm.leave_subsection();
153
154 prm.enter_subsection("average velocity profile");
155 prm.declare_entry(
156 "checkpoint folder",
157 "./",
158 Patterns::FileName(),
159 "the path leading to the checkpointed average velocity profile");

161 "restart",

171 template <int dim>
172 void
173 InitialConditions<dim>::parse_parameters(ParameterHandler &prm)
174 {
175 prm.enter_subsection("initial conditions");
176 {
177 const std::string op = prm.get("type");
178 if (op == "L2projection")
179 type = FluidDynamicsInitialConditionType::L2projection;
180 else if (op == "viscous")
181 type = FluidDynamicsInitialConditionType::viscous;
182 else if (op == "nodal")
183 type = FluidDynamicsInitialConditionType::nodal;
184 else if (op == "ramp")
185 type = FluidDynamicsInitialConditionType::ramp;
186 else if (op == "average_velocity_profile")
187 type = FluidDynamicsInitialConditionType::average_velocity_profile;
188
189 kinematic_viscosity = prm.get_double("kinematic viscosity");
190 prm.enter_subsection("uvwp");
191 uvwp.parse_parameters(prm);
192 prm.leave_subsection();
193
194 prm.enter_subsection("temperature");
195 temperature.parse_parameters(prm);
196 prm.leave_subsection();
197
198 prm.enter_subsection("tracer");
199 tracer.parse_parameters(prm);
200 prm.leave_subsection();
201
202 prm.enter_subsection("VOF");
203 {
204 VOF.parse_parameters(prm);
205 const std::string op = prm.get("smoothing type");
206 if (op == "none")
207 vof_initial_condition_smoothing = VOFInitialConditionType::none;
208 else if (op == "diffusive")
209 vof_initial_condition_smoothing = VOFInitialConditionType::diffusive;
210 else if (op == "geometric")
211 vof_initial_condition_smoothing = VOFInitialConditionType::geometric;
212 else
213 throw(
214 std::runtime_error("Unknown VOF initial condition smoothing type"));
215
216 projection_step_diffusion_factor = prm.get_double("diffusion factor");
217 }
218 prm.leave_subsection();
219
220 ramp.parse_parameters(prm);
221
222 prm.enter_subsection("cahn hilliard");
223 cahn_hilliard.parse_parameters(prm);
224 prm.leave_subsection();
225
226 prm.enter_subsection("average velocity profile");
227 average_velocity_folder = prm.get("checkpoint folder");
228 average_velocity_file_name = prm.get("checkpoint file name");
229
230 prm.leave_subsection();
231 }
232 prm.leave_subsection();
233 }
