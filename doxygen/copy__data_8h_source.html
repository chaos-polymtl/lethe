16/**
17 * @brief Class responsible for storing the information calculated using the assembly of regular (meaning
18 * non-stabilized) equations. It is also used to
19 * initialize, zero (reset) and store the cell_matrix, the cell_rhs and the
20 * dof indices associated with the dofs of the cell.
21 **/
22
23class CopyData
24{
25public:
26 /**
27 * @brief Constructor. Allocates the memory for the cell_matrix, cell_rhs and the local_dof_indices
28 *
29 * @param n_dofs Number of degrees of freedom per cell in the problem
30 */
31 CopyData(const unsigned int n_dofs)
32 : local_matrix(n_dofs, n_dofs)
33 , local_rhs(n_dofs)
34 , local_dof_indices(n_dofs){};
35
36 /**
37 * @brief Resets the cell_matrix and the cell_rhs to zero
38 */
39 void
40 reset()
41 {
42 local_matrix = 0;
43 local_rhs = 0;
44 }
45
46 FullMatrix<double> local_matrix;
47 Vector<double> local_rhs;
48 std::vector<types::global_dof_index> local_dof_indices;
49
50 // Boolean used to indicate if the cell being assembled is local or not
51 // This information is used to indicate to the copy_local_to_global function
52 // if it should indeed copy or not.
53 bool cell_is_local;
54};

57/**
58 * @brief Class responsible for storing the information calculated using the assembly of stabilized
59 * scalar equations. Like the CopyData class, this class is used to initialize,
60 * zero (reset) and store the cell_matrix and the cell_rhs.
61 * Contrary to the regular CopyData class, this class
62 * also stores the strong_residual and the strong_jacobian of the equation being
63 * assembled. This is useful for equations that implement residual-based
64 * stabilization such as SUPG. This class is specialized for single component
65 * equations because the strong jacobian is stored using a Vector<double>
66 **/
67class StabilizedMethodsCopyData
68{
69public:
70 /**
71 * @brief Constructor. Allocates the memory for the cell_matrix, cell_rhs
72 * and dof-indices using the number of dofs and the strong_residual using the
73 * number of quadrature points and, the strong_jacobian using both
74 *
75 * @param n_dofs Number of degrees of freedom per cell in the problem
76 *
77 * @param n_q_points Number of quadrature points
78 */
79 StabilizedMethodsCopyData(const unsigned int n_dofs,
80 const unsigned int n_q_points)
81 : local_matrix(n_dofs, n_dofs)
82 , local_rhs(n_dofs)
83 , local_dof_indices(n_dofs)
84 , strong_residual(n_q_points)
85 , strong_jacobian(n_q_points, Vector<double>(n_dofs)){};
86
87
88 /**
89 * @brief Resets the cell_matrix, cell_rhs, strong_residual
90 * and strong_jacobian to zero
91 */
92 void
93 reset()
94 {
95 local_matrix = 0;
96 local_rhs = 0;
97
98 strong_residual = 0;
99 for (unsigned int i = 0; i < strong_jacobian.size(); ++i)
100 {
101 strong_jacobian[i] = 0;
102 }
103 }
104
105 FullMatrix<double> local_matrix;
106 Vector<double> local_rhs;
107 std::vector<types::global_dof_index> local_dof_indices;
108 Vector<double> strong_residual;
109 std::vector<Vector<double>> strong_jacobian;
110
111 // Boolean used to indicate if the cell being assembled is local or not
112 // This information is used to indicate to the copy_local_to_global function
113 // if it should indeed copy or not.
114 bool cell_is_local;
115};

153/**
154 * @brief Class responsible for storing the information calculated using the assembly of stabilized
155 * Tensor<1,dim> equations. Like the CopyData class, this class is used to
156 * initialize, zero (reset) and store the cell_matrix and the cell_rhs. Contrary
157 * to the regular CopyData class, this class also stores the strong_residual and
158 * the strong_jacobian of the equation being assembled. This is useful for
159 * equations that implement residual-based stabilization such as SUPG. This
160 * class is specialized for Tensor<1,dim> equations because the strong
161 * jacobian is stored using a Tensor<1,dim>
162 **/
163
164
165template <int dim>
166class StabilizedMethodsTensorCopyData
167{
168public:
169 /**
170 * @brief Constructor. Allocates the memory for the cell_matrix and cell_rhs
171 * using the number of dofs and the strong_residual using the number of
172 * quadrature points and, the strong_jacobian using both
173 *
174 * @param n_dofs Number of degrees of freedom per cell in the problem
175 *
176 * @param n_q_points Number of quadrature points
177 */
178 StabilizedMethodsTensorCopyData(const unsigned int n_dofs,
179 const unsigned int n_q_points)
180 : local_matrix(n_dofs, n_dofs)
181 , local_rhs(n_dofs)
182 , local_dof_indices(n_dofs)
183 , strong_residual(n_q_points)
184 , strong_jacobian(n_q_points, std::vector<Tensor<1, dim>>(n_dofs)){};
185
186 /**
187 * @brief Resets the cell_matrix, cell_rhs, strong_residual
188 * and strong_jacobian to zero
189 */
190 void
191 reset()
192 {
193 local_matrix = 0;
194 local_rhs = 0;
195
196 for (unsigned int q = 0; q < strong_jacobian.size(); ++q)
197 {
198 strong_residual[q] = 0;
199
200 for (unsigned int i = 0; i < strong_jacobian[q].size(); ++i)
201 strong_jacobian[q][i] = 0;
202 }
203 }
204
205 FullMatrix<double> local_matrix;
206 Vector<double> local_rhs;
207 std::vector<types::global_dof_index> local_dof_indices;
208 std::vector<Tensor<1, dim>> strong_residual;
209 std::vector<std::vector<Tensor<1, dim>>> strong_jacobian;
210
211
212 // Boolean used to indicate if the cell being assembled is local or not
213 // This information is used to indicate to the copy_local_to_global function
214 // if it should indeed copy or not.
215 bool cell_is_local;
216 bool cell_is_cut;
217};
