29 *
30 */

106/**
107 * @brief Variant handler of the three output structs
108 * (OutputStructPostprocessor, OutputStructSolution and OutputStructCellVector).
109 * This is used to allow the output of both postprocessors with their respective
110 * DoF handlers, and the straightforward cell data struct in a single vector of
111 * OutputStruct.
112 */
113template <int dim, typename VectorType>
114using OutputStruct = std::variant<OutputStructPostprocessor<dim, VectorType>,
115 OutputStructSolution<dim, VectorType>,
116 OutputStructCellVector>;
