190  /**
191  * @brief Calculates an approximation of the gradient of the kinematic viscosity
192  * @param velocity_gradient The velocity gradient tensor on the quadrature point
193  @param velocity_hessians The velocity hessian tensor on the quadrature point
194  @param non_newtonian_viscosity The viscosity at which the gradient is calculated
195  @param d_gamma_dot Th difference in the shear rate magnitude to approximate the
196  viscosity variation with a slight change in the shear_rate magnitude
197  */
198  inline Tensor<1, dim>
199  get_kinematic_viscosity_gradient(
200  const Tensor<2, dim> &velocity_gradient,
201  const Tensor<3, dim> &velocity_hessians,
202  const double shear_rate_magnitude,
203  const double grad_kinematic_viscosity_shear_rate) const
204  {
205  // Calculates an approximation of the shear rate magnitude gradient using
206  // the derived form, since it does not change with rheological models
207  Tensor<1, dim> grad_shear_rate;
208  for (unsigned int d = 0; d < dim; ++d)
209  {
210  if (dim == 2)
211  {
212  for (unsigned int k = 0; k < dim; ++k)
213  {
214  grad_shear_rate[d] +=
215  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
216  shear_rate_magnitude;
217  }
218  grad_shear_rate[d] +=
219  (velocity_gradient[0][1] + velocity_gradient[1][0]) *
220  (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
221  shear_rate_magnitude;
222  }
223  else
224  {
225  for (unsigned int k = 0; k < dim; ++k)
226  {
227  grad_shear_rate[d] +=
228  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
229  shear_rate_magnitude +
230  (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
231  velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
232  (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
233  velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
234  shear_rate_magnitude;
235  }
236  }
237  }
238  return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
239  };

652  * \f$-\beta_D \mathbf{u} \f$ to the right hand-side of the Navier-Stokes
