6std::shared_ptr<RheologicalModel>
7RheologicalModel::model_cast(const Parameters::Material &material_properties)
8{
9 if (material_properties.rheological_model ==
10 Parameters::Material::RheologicalModel::newtonian)
11 return std::make_shared<Newtonian>(material_properties.kinematic_viscosity);
12 else if (material_properties.rheological_model ==
13 Parameters::Material::RheologicalModel::powerlaw)
14 return std::make_shared<PowerLaw>(
15 material_properties.non_newtonian_parameters.powerlaw_parameters.K,
16 material_properties.non_newtonian_parameters.powerlaw_parameters.n,
17 material_properties.non_newtonian_parameters.powerlaw_parameters
18 .shear_rate_min);
19 else if (material_properties.rheological_model ==
20 Parameters::Material::RheologicalModel::carreau)
21 return std::make_shared<Carreau>(
22 material_properties.non_newtonian_parameters.carreau_parameters
23 .kinematic_viscosity_0,
24 material_properties.non_newtonian_parameters.carreau_parameters
25 .kinematic_viscosity_inf,
26 material_properties.non_newtonian_parameters.carreau_parameters.lambda,
27 material_properties.non_newtonian_parameters.carreau_parameters.a,
28 material_properties.non_newtonian_parameters.carreau_parameters.n);
29
30 else // (fluid_properties.rheological_model ==
31 // Parameters::Fluid::RheologicalModel::phase_change)
32 return std::make_shared<PhaseChangeRheology>(
33 material_properties.phase_change_parameters);
34}

65double
66PowerLaw::jacobian(const std::map<field, double> &field_values, const field id)
67{
68 const double shear_rate_magnitude = field_values.at(field::shear_rate);
69 if (id == field::shear_rate)
70 {
71 Assert(field_values.find(field::shear_rate) != field_values.end(),
72 PhysicialPropertyModelFieldUndefined("PowerLaw", "shear_rate"));
73 return calculate_derivative(shear_rate_magnitude);
74 }
75 else
76 return 0;
