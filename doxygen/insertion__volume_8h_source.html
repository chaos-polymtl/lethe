11template <int dim, typename PropertiesIndex>
12class InsertionVolume : public Insertion<dim, PropertiesIndex>
13{
14public:
15 /**
16 * @brief The constructor investigates if the insertion box is large enough
17 * to handle the desired number of particles with the specified insertion
18 * parameters.
19 * If the insertion box is not adequately large, the number of
20 * inserted particles at each insertion step is updated. It also finds the
21 * insertion points in each direction (number_of_particles_x_direction,
22 * number_of_particles_y_direction and number_of_particles_z_direction).
23 *
24 * @param[in] size_distribution_object_container Contains all distribution for
25 * each particle type
26 * @param[in] triangulation Triangulation to access the cells in which the
27 * particles are inserted
28 * @param[in] dem_parameters DEM parameters declared in the .prm file
29 * @param[in] maximum_particle_diameter Maximum particle diameter based on
30 * values defined in the parameter handler
31 */
32 InsertionVolume(
33 const std::vector<std::shared_ptr<Distribution>>
34 &size_distribution_object_container,
35 const parallel::distributed::Triangulation<dim> &triangulation,
36 const DEMSolverParameters<dim> &dem_parameters,
37 const double maximum_particle_diameter);
38
39 /**
40 * @brief Carries out the volume insertion of particles.
41 *
42 * @param[in] particle_handler The particle handler of particles which are
43 * being inserted

45 * particles are inserted
46 * @param[in] dem_parameters DEM parameters declared in the .prm file
47 *
48 */

157
