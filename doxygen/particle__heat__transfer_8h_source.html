139/**
140 * @brief Calculate the total thermal conductance for particle-particle
141 * or particle-wall contacts.
142 * The resistance_macrocontact, resistance_solid_macrogap and
143 * resistance_gas_macrogap differ from particle-particle contacts to
144 * particle-wall contacts.
145 *
146 * @tparam contact_type Type of contact. Only particle-particle and
147 * particle-floating-mesh contacts are accepted for now.
148 *
149 * @param[in] radius_one Radius of particle one.
150 * @param[in] radius_two Radius of particle two for particle-particle contacts.
151 * Unused for particle-wall contacts.
152 * @param[in] effective_youngs_modulus Effective young's modulus of the contact
153 * pair.
154 * @param[in] effective_real_youngs_modulus Effective real young's modulus of
155 * the contact pair.
156 * @param[in] equivalent_surface_roughness Equivalent surface roughness of the
157 * contact pair.
158 * @param[in] equivalent_surface_slope Equivalent surface slope of the contact
159 * pair.
160 * @param[in] effective_microhardness Effective microhardness of the contact
161 * pair.
162 * @param[in] thermal_conductivity_one Thermal conductivity of particle one.
163 * @param[in] thermal_conductivity_two Thermal conductivity of particle two or
164 * of wall.
165 * @param[in] thermal_conductivity_gas Thermal conductivity of the interstitial
166 * gas.
167 * @param[in] gas_parameter_m Gas parameter.
168 * @param[in] normal_overlap Normal overlap of the contact pair.
169 * @param[in] normal_force_norm Norm of the normal contact force of the contact
170 * pair.
171 * @param[out] thermal_conductance Total thermal conductance for the contact
172 * pair.
173 */
174template <ContactType contact_type>
175void
176calculate_contact_thermal_conductance(
177 const double radius_one,
178 [[maybe_unused]] const double radius_two,
179 const double effective_youngs_modulus,
180 const double effective_real_youngs_modulus,
181 const double equivalent_surface_roughness,
182 const double equivalent_surface_slope,
183 const double effective_microhardness,
184 const double thermal_conductivity_one,
185 const double thermal_conductivity_two,
186 const double thermal_conductivity_gas,
187 const double gas_parameter_m,
188 const double normal_overlap,
189 const double normal_force_norm,
190 double &thermal_conductance);
