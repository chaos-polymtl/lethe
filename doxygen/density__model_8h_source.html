202 /**
203  * @brief Isothermal ideal gas density.
204  *
205  * Isothermal ideal gas density assumes that the fluid's density varies
206  * according the following state equation:
207  *
208  * \f$\rho = \rho_\text{ref} + \psi p = \rho_\text{ref} + \frac{1}{R T} \ p\f$
209  *
210  * where \f$\rho_\text{ref}\f$ is the density of the fluid at the reference
211  * state, \f$\psi = \frac{1}{R T}\f$ is the compressibility factor derived from
212  * the ideal gas law with \f$R= \frac{R_u}{M}\f$ the specific gas constant
213  * (universal gas constant (\f$R_u\f$) divided by the molar mass of the gas
214  * (\f$M\f$)) and \f$T\f$ the temperature of the gas, finally, \f$p\f$ is the
215  * differential pressure between the reference state and the current state.
216  * The model is used for weakly compressible flows when temperature
217  * fluctuations' influence on density can be neglected.
218  */
219 class DensityIsothermalIdealGas : public DensityModel
220 {
221 public:
222  /**
223  * @brief Constructor of the isothermal ideal gas density model.
224  *
225  * @param[in] p_density_ref Value of the density of the gas at reference
226  * state.
227  *
228  * @param[in] p_R Value of the specific gas constant \f$\left(R=\frac{R_u}{M}
229  * \right)\f$ with \f$R_u\f$ the universal gas constant and \f$M\f$ the molar
230  * mass of the gas.
231  *
232  * @param[in] p_T Value of the temperature of the gas at reference state.
233  */
234  DensityIsothermalIdealGas(const double p_density_ref,
235  const double p_R,
236  const double p_T)
237  : density_ref(p_density_ref)
238  , psi(1. / (p_R * p_T))
239  {
240  this->model_depends_on[field::pressure] = true;
241  }
242 
243  /**
244  * @brief Compute the density of the isothermal ideal gas.
245  *
246  * @param[in] field_values Values of the various fields on which the property
247  * may depend. In this case, the density depends on pressure. The map stores a
248  * single value per field.
249  *
250  * @return Value of the density computed with the @p field_values.
251  */
252  double
253  value(const std::map<field, double> &field_values) override
254  {
255  AssertThrow(field_values.find(field::pressure) != field_values.end(),
256  PhysicialPropertyModelFieldUndefined(
257  "DensityIsothermalIdealGas", "pressure"));
258  const double pressure = field_values.at(field::pressure);
259  return density_ref + psi * pressure;
260  }
261 
262  /**
263  * @brief Compute a vector of density values for an isothermal ideal gas.
264  *
265  * @param[in] field_vectors Vectors of the fields on which the density may
266  * depend. In this case, the density depends on pressure. The map stores a
267  * vector of values per field.
268  *
269  * @param[out] property_vector Vectors of computed density values.
270  */
271  void
272  vector_value(const std::map<field, std::vector<double>> &field_vectors,
273  std::vector<double> &property_vector) override
274  {
275  AssertThrow(field_vectors.find(field::pressure) != field_vectors.end(),
276  PhysicialPropertyModelFieldUndefined(
277  "DensityIsothermalIdealGas", "pressure"));
278  const std::vector<double> &pressure = field_vectors.at(field::pressure);
279  for (unsigned int i = 0; i < property_vector.size(); ++i)
280  property_vector[i] = density_ref + psi * pressure[i];
281  }

284  * @brief Compute the jacobian (the partial derivative) of the density with

286  *
287  * @param[in] field_values Values of the various fields on which the density
288  * may depend. The map stores a single value per field.
289  *

291  * should be computed.
292  *

296  double
