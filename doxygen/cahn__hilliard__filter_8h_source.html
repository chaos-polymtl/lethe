62/**
63 * @brief Calculate a filtered phase fraction for Cahn-Hilliard simulations.
64 * In this case, a simple clamping is performed on the phase fraction parameter
65 * for it to remain in the [-1,1] interval.
66 */
67class CahnHilliardClipFilter : public CahnHilliardFilterBase
68{
69public:
70 CahnHilliardClipFilter()
71 {}
72
73 /**
74 * @brief Calculate the value of the filtered phase fraction.
75 * @param[in] unfiltered_phase Value of the phase fraction before applying the
76 * filter.
77 * @return Value of the phase fraction after applying the filter.
78 */
79 virtual double
80 filter_phase(const double &unfiltered_phase) override
81 {
82 return (std::abs(unfiltered_phase) < 1) ? unfiltered_phase :
83 sgn(unfiltered_phase);
84 }
85};
86
87/**
88 * @brief Calculate a filtered phase fraction for Cahn-Hilliard simulations.
89 *
90 * The filtered phase is defined as \f$\phi_f = \tanh(\beta \phi)\f$ with
91 * \f$ \phi \f$ the unfiltered phase field.
92 */
93class CahnHilliardTanhFilter : public CahnHilliardFilterBase
94{
95public:
96 CahnHilliardTanhFilter(const double beta)
97 : beta(beta)
98 {}
99
100 /**
101 * @brief Calculate the value of the filtered phase fraction.
102 * @param[in] unfiltered_phase Value of the phase fraction before applying the
103 * filter.
104 * @return Value of the phase fraction after applying the filter.
105 */
106 virtual double
107 filter_phase(const double &unfiltered_phase) override
108 {
109 return tanh(beta * unfiltered_phase);
110 }
111
112private:
113 /**
114 * User-defined parameter that influences how sharp the filtering is.
115 */
116 const double beta;
117};
