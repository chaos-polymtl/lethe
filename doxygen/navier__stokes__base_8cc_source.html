542 // Do an iteration with the NavierStokes Solver
543 // Handles the fact that we may or may not be at a first
544 // iteration with the solver and sets the initial condition
545 template <int dim, typename VectorType, typename DofsType>
546 void
547 NavierStokesBase<dim, VectorType, DofsType>::iterate()
548 {
549  auto &present_solution = this->present_solution;
550 
551  if (simulation_parameters.multiphysics.fluid_dynamics)
552  {
553  // Solve and percolate the auxiliary physics that should be treated BEFORE
554  // the fluid dynamics
555  multiphysics->solve(false,
556  simulation_parameters.simulation_control.method);
557  multiphysics->percolate_time_vectors(false);
558 
559  if (simulation_parameters.non_linear_solver.at(PhysicsID::fluid_dynamics)
560  .verbosity != Parameters::Verbosity::quiet ||
561  simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
562  .verbosity != Parameters::Verbosity::quiet)
563  announce_string(this->pcout, "Fluid Dynamics");
564  PhysicsSolver<VectorType>::solve_non_linear_system(false);
565 
566  // If the physics need to be solved after the physics, the matrix free
567  // solver requires to update the value here. This is due to the different
568  // type of vectors.
569  if (this->multiphysics->get_active_physics().size() > 1)
570  this->update_solutions_for_multiphysics();
571 
572  // Solve and percolate the auxiliary physics that should be treated AFTER
573  // the fluid dynamics
574  multiphysics->solve(true,
575  simulation_parameters.simulation_control.method);
576  // Dear future Bruno, percolating auxiliary physics before fluid dynamics
577  // is necessary because of the checkpointing mechanism. You spent an
578  // evening debugging this, trust me.
579  multiphysics->percolate_time_vectors(true);
580  }
581  else
582  {
583  // Fluid dynamics is not to be solved, but rather specified. Update
584  // condition and move on.
585 
586  // Solve and percolate the auxiliary physics that should be treated
587  // BEFORE the fluid dynamics
588  multiphysics->solve(false,
589  simulation_parameters.simulation_control.method);
