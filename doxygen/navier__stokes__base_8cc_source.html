46template <int dim, typename VectorType, typename DofsType>
47NavierStokesBase<dim, VectorType, DofsType>::NavierStokesBase(
48 SimulationParameters<dim> &p_nsparam)
49 : PhysicsSolver<VectorType>(
50 p_nsparam.non_linear_solver.at(PhysicsID::fluid_dynamics))
51 , mpi_communicator(MPI_COMM_WORLD)
52 , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
53 , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
54 , dof_handler()
55 , computing_timer(this->mpi_communicator,
56 this->pcout,
57 TimerOutput::summary,
58 TimerOutput::wall_times)
59 , simulation_parameters(p_nsparam)
60 , flow_control(simulation_parameters.flow_control)
61 , velocity_fem_degree(p_nsparam.fem_parameters.velocity_order)
62 , pressure_fem_degree(p_nsparam.fem_parameters.pressure_order)
63 , number_quadrature_points(p_nsparam.fem_parameters.velocity_order + 1)
64 , mesh_controller(p_nsparam.mesh_adaptation.maximum_number_elements)
65{
66 if (simulation_parameters.mesh.simplex)
67 {
68 // for simplex meshes
69 const FE_SimplexP<dim> velocity_fe(
70 p_nsparam.fem_parameters.velocity_order);
71 const FE_SimplexP<dim> pressure_fe(
72 p_nsparam.fem_parameters.pressure_order);
73 fe = std::make_shared<FESystem<dim>>(velocity_fe, dim, pressure_fe, 1);
74
75 AssertThrow(
76 !p_nsparam.fem_parameters.enable_bubble_function_velocity ||
77 !p_nsparam.fem_parameters.enable_bubble_function_pressure,
78 ExcMessage(
79 "Bubble enrichment functions are not compatible with simplex meshes."));
80
81 mapping = std::make_shared<MappingFE<dim>>(velocity_fe);
82 cell_quadrature =
83 std::make_shared<QGaussSimplex<dim>>(number_quadrature_points);
84 face_quadrature =
85 std::make_shared<QGaussSimplex<dim - 1>>(number_quadrature_points);
86 triangulation =
87 std::make_shared<parallel::fullydistributed::Triangulation<dim>>(
88 this->mpi_communicator);
89 dof_handler.clear();
90 dof_handler.reinit(*this->triangulation);
91 }
92 else
93 {
94 // Usual case, for quad/hex meshes
95 std::shared_ptr<FiniteElement<dim>> fe_u;
96 std::shared_ptr<FiniteElement<dim>> fe_p;
97
98 if (p_nsparam.fem_parameters.enable_bubble_function_velocity)
99 fe_u = std::make_shared<FE_Q_Bubbles<dim>>(
100 p_nsparam.fem_parameters.velocity_order);
101 else
102 fe_u =
103 std::make_shared<FE_Q<dim>>(p_nsparam.fem_parameters.velocity_order);
104
105 if (p_nsparam.fem_parameters.enable_bubble_function_pressure)
106 fe_p = std::make_shared<FE_Q_Bubbles<dim>>(
107 p_nsparam.fem_parameters.pressure_order);
108 else
109 fe_p =
110 std::make_shared<FE_Q<dim>>(p_nsparam.fem_parameters.pressure_order);
111
112 fe = std::make_shared<FESystem<dim>>(*fe_u, dim, *fe_p, 1);
113 mapping = std::make_shared<MappingQ<dim>>(velocity_fem_degree);
114 cell_quadrature = std::make_shared<QGauss<dim>>(number_quadrature_points);
115 face_quadrature =
116 std::make_shared<QGauss<dim - 1>>(number_quadrature_points);
117 triangulation =
118 std::make_shared<parallel::distributed::Triangulation<dim>>(
119 this->mpi_communicator,
120 Triangulation<dim>::limit_level_difference_at_vertices,
121 (p_nsparam.linear_solver.at(fluid_dynamics).preconditioner ==
122 Parameters::LinearSolver::PreconditionerType::lsmg) ?
123 parallel::distributed::Triangulation<
124 dim>::construct_multigrid_hierarchy :
125 parallel::distributed::Triangulation<dim>::default_setting);
126 dof_handler.clear();
127 dof_handler.reinit(*this->triangulation);
128 }
129
130 this->pcout.set_condition(
131 Utilities::MPI::this_mpi_process(this->mpi_communicator) == 0);
132
133 // Check if the output directory exists
134 std::string output_dir_name =
135 simulation_parameters.simulation_control.output_folder;
136 struct stat buffer;
137
138 // If output directory does not exist, create it
139 if (Utilities::MPI::this_mpi_process(this->mpi_communicator) == 0)
140 {
141 if (stat(output_dir_name.c_str(), &buffer) != 0)
142 {
143 create_output_folder(output_dir_name);
144 }
145 }
146
147 if (simulation_parameters.simulation_control.method ==
148 Parameters::SimulationControl::TimeSteppingMethod::steady)
149 simulation_control = std::make_shared<SimulationControlSteady>(
150 simulation_parameters.simulation_control);
151 else if (simulation_parameters.simulation_control.method ==
152 Parameters::SimulationControl::TimeSteppingMethod::steady_bdf)
153 {
154 simulation_control = std::make_shared<SimulationControlAdjointSteady>(
155 simulation_parameters.simulation_control);
156 }
157 else
158 {
159 simulation_control = std::make_shared<SimulationControlTransient>(
160 simulation_parameters.simulation_control);
161 }
162
163 // Provide the simulation control object to the physical property manager
164 simulation_parameters.physical_properties_manager.provide_simulation_control(
165 simulation_control);
166
167 multiphysics = std::make_shared<MultiphysicsInterface<dim>>(
168 simulation_parameters, triangulation, simulation_control, this->pcout);
169
170
171 if (this->simulation_control->is_sdirk())
172 {
173 // Pre-allocate memory for the previous k_j solutions using the number of
174 // stages of the method
175 sdirk_vectors.previous_k_j_solutions.resize(
176 SimulationControl::get_number_of_stages(
177 this->simulation_control->get_assembly_method()));
178 }
179 // Pre-allocate memory for the previous solutions using the information
180 // of the BDF schemes
181 previous_solutions.resize(maximum_number_of_previous_solutions());
182
183 // Change the behavior of the timer for situations when you don't want
184 // outputs
185 if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
186 this->computing_timer.disable_output();
187
188 // Get the exact solution from the parser
189 exact_solution = &simulation_parameters.analytical_solution->uvwp;
190
191 // If there is a forcing function, get it from the parser
192 forcing_function = simulation_parameters.source_term.navier_stokes_source;
193
194 if (this->simulation_parameters.post_processing
195 .calculate_average_velocities ||
196 this->simulation_parameters.initial_condition->type ==
197 Parameters::FluidDynamicsInitialConditionType::average_velocity_profile)
198 average_velocities =
199 std::make_shared<AverageVelocities<dim, VectorType, DofsType>>(
200 dof_handler);
201
202 this->pcout << "Running on "
203 << Utilities::MPI::n_mpi_processes(this->mpi_communicator)
204 << " MPI rank(s)..." << std::endl;
205
206 this->pcout << std::setprecision(
207 simulation_parameters.simulation_control.log_precision);
208}

580template <int dim, typename VectorType, typename DofsType>
581void
582NavierStokesBase<dim, VectorType, DofsType>::multi_stage_preresolution(
583 unsigned int stage,
584 Parameters::SimulationControl::TimeSteppingMethod method)
585{
586 // Copy the reference to some of the vectors to enhance readability below
587 auto &previous_k_j_solutions = this->sdirk_vectors.previous_k_j_solutions;
588 auto &sum_over_previous_stages = this->sdirk_vectors.sum_over_previous_stages;
589 auto &local_sum_over_previous_stages =
590 this->sdirk_vectors.local_sum_over_previous_stages;
591 auto &locally_owned_for_calculation =
592 this->sdirk_vectors.locally_owned_for_calculation;
593
594 // If a SDIRK method is selected, we need to solve as many
595 // nonlinear systems as the number of stages.
596
597 // At each stage, the value of the (a_ij), (b_i) and (c_i)
598 // coefficients are differents
599
600 SDIRKTable table = sdirk_table(method);
601 SDIRKStageData stage_data(table, stage + 1);
602 const double a_ii = stage_data.a_ij[stage];
603
604 // At each stage, we need to recompute sum(a_{ij} * k_j)
605 // = sum_over_previous_stages
606 local_sum_over_previous_stages = 0;
607
608 if (stage > 0)
609 {
610 // At the first stage, the sum_over_previous_stages is set to
611 // 0. But for the next stages, we need to update this
612 // sum_over_previous_stages
613 for (unsigned int p = 0; p < stage; ++p)
614 {
615 locally_owned_for_calculation = previous_k_j_solutions[p];
616 local_sum_over_previous_stages.add(stage_data.a_ij[p] / a_ii,
617 locally_owned_for_calculation);
618 }
619 }
620
621 sum_over_previous_stages = local_sum_over_previous_stages;
622}

628 Parameters::SimulationControl::TimeSteppingMethod method,

630{
631 // Copy the reference to some of the vectors to enhance readability below
632 auto &present_solution = this->present_solution;
633 auto &previous_solutions = this->previous_solutions;
634 auto &local_evaluation_point = this->local_evaluation_point;
635 auto &local_sum_over_previous_stages =
636 this->sdirk_vectors.local_sum_over_previous_stages;
637 auto &locally_owned_for_calculation =
638 this->sdirk_vectors.locally_owned_for_calculation;
639 auto &previous_k_j_solutions = this->sdirk_vectors.previous_k_j_solutions;
640 auto &sum_bi_ki = this->sdirk_vectors.sum_bi_ki;
641 auto &local_sum_bi_ki = this->sdirk_vectors.local_sum_bi_ki;
642
643 SDIRKTable table = sdirk_table(method);
644 SDIRKStageData stage_data(table, stage + 1);
645 const double a_ii = stage_data.a_ij[stage];
646
647 // Once we have solved the nonlinear system for the velocity, we
648 // want to store the value of the coefficient k_i for the final
649 // sum b_i*k_i with k_i = (u*_{i} - u_{n})/(time_step*a_ii) -
650 // sum_over_previous_stages
651 locally_owned_for_calculation = previous_solutions[0];
652 local_evaluation_point = present_solution;
653 local_evaluation_point.add(-1.0, locally_owned_for_calculation);
654 local_evaluation_point *= 1.0 / (time_step * a_ii);
655 local_evaluation_point.add(-1.0, local_sum_over_previous_stages);
656
657 // We store the value of the present_k_i_solution in the
658 // previous_k_j_solutions vector.
659 previous_k_j_solutions[stage] = local_evaluation_point;
660
661 // We update the sum of b_i*k_i
662 const double b_i = stage_data.b_i;
663 local_sum_bi_ki.add(b_i, local_evaluation_point);
664 sum_bi_ki = local_sum_bi_ki;
665}
666
667template <int dim, typename VectorType, typename DofsType>
668void
669NavierStokesBase<dim, VectorType, DofsType>::update_multi_stage_solution(
670 double time_step)
671{
672 // Copy the reference to some of the vectors to enhance readability below
673 auto &present_solution = this->present_solution;
674 auto &previous_solutions = this->previous_solutions;
675 auto &local_evaluation_point = this->local_evaluation_point;
676 auto &sum_bi_ki = this->sdirk_vectors.sum_bi_ki;
677 auto &local_sum_bi_ki = this->sdirk_vectors.local_sum_bi_ki;
678
679 // At each time iteration, we update the value of present_solution
680 local_sum_bi_ki = sum_bi_ki;
681 local_evaluation_point = previous_solutions[0];
682 local_evaluation_point.add(time_step, local_sum_bi_ki);
683 present_solution = local_evaluation_point;
684}

686// Do an iteration with the NavierStokes Solver
687// Handles the fact that we may or may not be at a first
688// iteration with the solver and sets the initial condition
689template <int dim, typename VectorType, typename DofsType>
690void
691NavierStokesBase<dim, VectorType, DofsType>::iterate()
692{
693 // The locally_relevant_dofs are used for the assembly,
694 // while the non-locally_relevant_dofs are used for the calculations on the
695 // vectors (addition, multiplication, etc.).
696 const auto method = this->simulation_control->get_assembly_method();
697 const auto time_step = this->simulation_control->get_time_step();
698 const unsigned int n_stages = SimulationControl::get_number_of_stages(method);
699
700 auto &present_solution = this->present_solution;
701
702 // For a multi-stages method, at each time iteration, we reset the value of
703 // sum_bi_ki
704 if (n_stages > 1)
705 this->sdirk_vectors.local_sum_bi_ki = 0;
706
707 for (unsigned int stage = 0; stage < n_stages; ++stage)
708 {
709 if (simulation_parameters.multiphysics.fluid_dynamics)
710 {
711 // Solve and percolate the auxiliary physics that should be treated
712 // BEFORE the fluid dynamics
713 multiphysics->solve(false,
714 simulation_parameters.simulation_control.method);
715 multiphysics->percolate_time_vectors(false);
716
717 if (simulation_parameters.non_linear_solver
718 .at(PhysicsID::fluid_dynamics)
719 .verbosity != Parameters::Verbosity::quiet ||
720 simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
721 .verbosity != Parameters::Verbosity::quiet)
722 announce_string(this->pcout, "Fluid Dynamics");
723
724 if (n_stages > 1)
725 {
726 // For SDIRK, some variables need to be updated before solving the
727 // non-linear system.
728 multi_stage_preresolution(stage, method);
729 }
730
731 PhysicsSolver<VectorType>::solve_non_linear_system(false);
732
733 if (n_stages > 1)
734 {
735 // For SDIRK, some variables need to be updated after solving the
736 // non-linear system.
737 multi_stage_postresolution(stage, method, time_step);
738 }
739
740
741 // If the physics need to be solved after the physics, the matrix free
742 // solver requires to update the value here. This is due to the
743 // different type of vectors.
744 if (this->multiphysics->get_active_physics().size() > 1)
745 {
746 this->update_solutions_for_multiphysics();
747 }
748
749 // Solve and percolate the auxiliary physics that should be treated
750 // AFTER the fluid dynamics
751 multiphysics->solve(true,
752 simulation_parameters.simulation_control.method);
753 // Dear future Bruno, percolating auxiliary physics before fluid
754 // dynamics is necessary because of the checkpointing mechanism. You
755 // spent an evening debugging this, trust me.
756 multiphysics->percolate_time_vectors(true);
757 }
758 else
759 {
760 // Fluid dynamics is not to be solved, but rather specified. Update
761 // condition and move on.
762
763 // Solve and percolate the auxiliary physics that should be treated
764 // BEFORE the fluid dynamics
765 multiphysics->solve(false,
766 simulation_parameters.simulation_control.method);
767 multiphysics->percolate_time_vectors(false);
768
769 if (this->simulation_parameters.initial_condition->type ==
770 Parameters::FluidDynamicsInitialConditionType::
771 average_velocity_profile)
772 {
773 // We get the solution via the average solution
774 this->local_evaluation_point =
775 this->average_velocities->get_average_velocities();
776 present_solution = this->local_evaluation_point;
777 }
778 else
779 {
780 // We get the solution via an initial condition
781 this->simulation_parameters.initial_condition->uvwp.set_time(
782 this->simulation_control->get_current_time());
783 set_initial_condition_fd(
784 this->simulation_parameters.initial_condition->type);
785 }
786
787 // Solve and percolate the auxiliary physics that should be treated
788 // AFTER the fluid dynamics
789 multiphysics->solve(true,
790 simulation_parameters.simulation_control.method);
791 multiphysics->percolate_time_vectors(true);
792 }
793 }
794 if (n_stages > 1)
795 {
796 // For SDIRK, the solution is not directly computed with the non-linear
797 // solver. Once we looped over all the stages, we can finally update the
798 // present_solution vector.
799 update_multi_stage_solution(time_step);
800 }
801}

1091NavierStokesBase<dim, VectorType, DofsType>::refine_mesh_kelly()
1092{
1093 if (dynamic_cast<parallel::distributed::Triangulation<dim> *>(
1094 this->triangulation.get()) == nullptr)
1095 return;
1096
1097 auto &tria = *dynamic_cast<parallel::distributed::Triangulation<dim> *>(
1098 this->triangulation.get());
1099
1100 // Time monitoring
1101 TimerOutput::Scope t(this->computing_timer, "Refine");
1102
1103 Vector<float> estimated_error_per_cell(tria.n_active_cells());
1104 const FEValuesExtractors::Vector velocity(0);
1105 const FEValuesExtractors::Scalar pressure(dim);
1106 auto &present_solution = this->present_solution;
1107 VectorType locally_relevant_solution;
1108 locally_relevant_solution.reinit(this->locally_owned_dofs,
1109 this->locally_relevant_dofs,
1110 this->mpi_communicator);
1111 locally_relevant_solution = this->present_solution;
1112 locally_relevant_solution.update_ghost_values();
1113
1114 // Global flags
1115 // Their dimension is consistent with the dimension returned by
1116 // save_refine_flags(), in order to be able to use load_refine_flags()
1117 std::vector<bool> global_refine_flags(dim * tria.n_active_cells(), false);
1118 std::vector<bool> global_coarsen_flags(dim * tria.n_active_cells(), false);
1119
1120 bool first_variable(true);
1121 const double coarsening_factor = mesh_controller.calculate_coarsening_factor(
1122 this->triangulation->n_global_active_cells());
1123
1124 unsigned int maximal_number_of_elements =
1125 this->simulation_parameters.mesh_adaptation.maximum_number_elements;
1126 // Override the maximal number of elements if the controller is used. The
1127 // controller will find a coarsening_factor that respects the user-defined
1128 // maximal_number_of_elements.
1129 if (this->simulation_parameters.mesh_adaptation.mesh_controller_is_enabled)
1130 maximal_number_of_elements = INT_MAX;
1131
1132 for (const std::pair<const Variable, Parameters::MultipleAdaptationParameters>
1133 &ivar : this->simulation_parameters.mesh_adaptation.variables)
1134 {
1135 double ivar_coarsening_factor = ivar.second.coarsening_fraction;
1136 if (this->simulation_parameters.mesh_adaptation
1137 .mesh_controller_is_enabled)
1138 ivar_coarsening_factor = coarsening_factor;
1139
1140 if (ivar.first == Variable::pressure)
1141 {
1142 KellyErrorEstimator<dim>::estimate(
1143 *this->get_mapping(),
1144 this->dof_handler,
1145 *this->face_quadrature,
1146 typename std::map<types::boundary_id,
1147 const Function<dim, double> *>(),
1148 locally_relevant_solution,
1149 estimated_error_per_cell,
1150 this->fe->component_mask(pressure));
1151 }
1152 else if (ivar.first == Variable::velocity)
1153 {
1154 KellyErrorEstimator<dim>::estimate(
1155 *this->get_mapping(),
1156 this->dof_handler,
1157 *this->face_quadrature,
1158 typename std::map<types::boundary_id,
1159 const Function<dim, double> *>(),
1160 locally_relevant_solution,
1161 estimated_error_per_cell,
1162 this->fe->component_mask(velocity));
1163 }
1164 else
1165 {
1166 // refine_mesh on an auxiliary physic parameter
1167 multiphysics->compute_kelly(ivar, estimated_error_per_cell);
1168 }
1169
1170 if (this->simulation_parameters.mesh_adaptation.fractionType ==
1171 Parameters::MeshAdaptation::FractionType::number)
1172 parallel::distributed::GridRefinement::refine_and_coarsen_fixed_number(
1173 tria,
1174 estimated_error_per_cell,
1175 ivar.second.refinement_fraction,
1176 ivar_coarsening_factor,
1177 maximal_number_of_elements);
1178
1179 else if (this->simulation_parameters.mesh_adaptation.fractionType ==
1180 Parameters::MeshAdaptation::FractionType::fraction)
1181 parallel::distributed::GridRefinement::
1182 refine_and_coarsen_fixed_fraction(tria,
1183 estimated_error_per_cell,
1184 ivar.second.refinement_fraction,
1185 ivar_coarsening_factor);
1186
1187 // Remove the flags if the cell is at the boundary and is set as do not
1188 // touch in the parameter file
1189 if (this->simulation_parameters.mesh_adaptation
1190 .is_boundary_refinement_fixed)
1191 for (const auto &cell : tria.active_cell_iterators())
1192 if (cell->is_locally_owned())
1193 {
1194 if (cell->at_boundary())
1195 for (const auto &face : cell->face_iterators())
1196 {
1197 if (std::find(this->simulation_parameters.mesh_adaptation
1198 .boundaries_to_fix.begin(),
1199 this->simulation_parameters.mesh_adaptation
1200 .boundaries_to_fix.end(),
1201 face->boundary_id()) !=
1202 this->simulation_parameters.mesh_adaptation
1203 .boundaries_to_fix.end())
1204 {
1205 cell->clear_refine_flag();
1206 cell->clear_coarsen_flag();
1207 }
1208 }
1209 }
1210
1211 std::vector<bool> current_refine_flags;
1212 std::vector<bool> current_coarsen_flags;
1213
1214 tria.save_refine_flags(current_refine_flags);
1215 tria.save_coarsen_flags(current_coarsen_flags);
1216
1217 // Fill global flags
1218 if (first_variable)
1219 {
1220 // special case of the first refinement variable
1221 global_refine_flags = current_refine_flags;
1222 global_coarsen_flags = current_coarsen_flags;
1223
1224 first_variable = false;
1225 }
1226 else
1227 {
1228 // for subsequent refinement variables
1229 for (std::vector<bool>::size_type i = 0;
1230 i != global_refine_flags.size();
1231 ++i)
1232 {
1233 // refine if at least refinement flag on one variable
1234 global_refine_flags[i] =
1235 global_refine_flags[i] || current_refine_flags[i];
1236 }
1237
1238 // for subsequent coarsen variables
1239 for (std::vector<bool>::size_type i = 0;
1240 i != global_coarsen_flags.size();
1241 ++i)
1242 {
1243 // coarsen if refinement flag on all variables
1244 global_coarsen_flags[i] =
1245 global_coarsen_flags[i] && current_coarsen_flags[i];
1246 }
1247 }
1248
1249 // Clear flags in the triangulation
1250 for (const auto &cell : tria.active_cell_iterators())
1251 {
1252 cell->clear_coarsen_flag();
1253 cell->clear_refine_flag();
1254 }
1255 }
1256
1257 // Load global flags
1258 tria.load_refine_flags(global_refine_flags);
1259 tria.load_coarsen_flags(global_coarsen_flags);
1260
1261 if (tria.n_levels() >
1262 this->simulation_parameters.mesh_adaptation.maximum_refinement_level)
1263 for (typename Triangulation<dim>::active_cell_iterator cell =

1274 this->simulation_parameters.mesh_adaptation.minimum_refinement_level);

1300
1301 multiphysics->prepare_for_mesh_adaptation();
1302 if (this->simulation_parameters.post_processing
1303 .calculate_average_velocities ||
1304 this->simulation_parameters.initial_condition->type ==

1373template <int dim, typename VectorType, typename DofsType>
1374void
1375NavierStokesBase<dim, VectorType, DofsType>::transfer_solution(
1376 SolutionTransfer<dim, VectorType> &solution_transfer,
1377 std::vector<SolutionTransfer<dim, VectorType>> &previous_solutions_transfer)
1378{
1379 // Set up the vectors for the transfer
1380 VectorType tmp = init_temporary_vector();
1381
1382 // Interpolate the solution at time and previous time
1383 solution_transfer.interpolate(tmp);
1384
1385 // Distribute constraints
1386 auto &nonzero_constraints = this->nonzero_constraints;
1387 nonzero_constraints.distribute(tmp);
1388
1389 // Fix on the new mesh
1390 present_solution = tmp;
1391
1392 // Set up the vectors for the transfer
1393 for (unsigned int i = 0; i < previous_solutions.size(); ++i)
1394 {
1395 VectorType tmp_previous_solution = init_temporary_vector();
1396 previous_solutions_transfer[i].interpolate(tmp_previous_solution);
1397 nonzero_constraints.distribute(tmp_previous_solution);
1398 previous_solutions[i] = tmp_previous_solution;
1399 }
1400
1401 multiphysics->post_mesh_adaptation();
1402 if (this->simulation_parameters.post_processing
1403 .calculate_average_velocities ||
1404 this->simulation_parameters.initial_condition->type ==
1405 Parameters::FluidDynamicsInitialConditionType::average_velocity_profile)
1406 average_velocities->post_mesh_adaptation();
1407
1408 // Only needed if other physics apart from fluid dynamics are enabled.
1409 if (this->multiphysics->get_active_physics().size() > 1)
1410 this->update_multiphysics_time_average_solution();

2369 std::ifstream in(filename.c_str());
2370 if (!in)
2371 AssertThrow(false,
2372 ExcMessage(
2373 std::string(
2374 "You are trying to restart a previous computation, "
2375 "but the restart file <") +
2376 filename + "> does not appear to exist!"));

2379 {
2380 if (auto tria = dynamic_cast<parallel::distributed::Triangulation<dim> *>(
2381 this->triangulation.get()))
2382 {
2383 tria->load(filename.c_str());

2395 enable_dynamic_zero_constraints_fd();

2397 // BB note: There is an issue right now that will prevent this code from
2398 // running in debug mode with Trilinos vectors Deal.II vectors require that
2399 // the vectors used in the checkpointing mechanism have their relevant dofs
2400 // whereas Trilinos vectors do not allow for this. Right now this code works
2401 // well in release mode for both vector types, but will not work in debug mode
2402 // for Trilinos vectors because of an assertion. A workaround will be
2403 // implemented in a near future
2404

2413 distributed_previous_solutions.reserve(previous_solutions.size());

2415 {
2416 distributed_previous_solutions.emplace_back(
2417 VectorType(locally_owned_dofs,
2418 this->locally_relevant_dofs,
2419 this->mpi_communicator));

2429 this->average_velocities->read(checkpoint_file_prefix);
2430

2461 this->simulation_parameters.initial_condition->type ==
2462 Parameters::FluidDynamicsInitialConditionType::average_velocity_profile)

2468template <int dim, typename VectorType, typename DofsType>
2469void
2470NavierStokesBase<dim, VectorType, DofsType>::establish_solid_domain(
2471 const bool non_zero_constraints)
2472{
2473 // If there are no solid regions, there is no work to be done and we can
2474 // return.
2475 if (simulation_parameters.physical_properties_manager
2476 .get_number_of_solids() == 0)
2477 return;
2478
2479 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
2480 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
2481
2482 // We will need to identify which pressure degrees of freedom are connected to
2483 // fluid region. For these, we won't establish a zero pressure constraint.
2484 std::unordered_set<types::global_dof_index> dofs_are_connected_to_fluid;
2485
2486 // Loop through all cells to identify which cells are solid. This first step
2487 // is used to 1) constraint the velocity degree of freedom to be zero in the
2488 // solid region and 2) to identify which pressure degrees of freedom are
2489 // connected to fluid cells
2490 for (const auto &cell : dof_handler.active_cell_iterators())
2491 {
2492 if (cell->is_locally_owned() || cell->is_ghost())
2493 {
2494 cell->get_dof_indices(local_dof_indices);
2495 // If the material_id is higher than 0, the region is a solid region.
2496 // Constrain the velocity DOFs to be zero.
2497 if (cell->material_id() > 0)
2498 {
2499 constrain_solid_cell_velocity_dofs(non_zero_constraints,
2500 local_dof_indices,
2501 this->zero_constraints);
2502 }
2503 else
2504 {
2505 // Cell is a fluid cell and as such all the pressure DOFs of that
2506 // cell are connected to the fluid. This will be used later on to
2507 // identify which pressure cells to constrain.
2508 flag_dofs_connected_to_fluid(local_dof_indices,
2509 dofs_are_connected_to_fluid);
2510 }
2511 }
2512 }
2513
2514 // All pressure DOFs that are not connected to a fluid cell are constrained
2515 // to ensure that the system matrix has adequate conditioning.
2516 for (const auto &cell : dof_handler.active_cell_iterators())
2517 {
2518 if (cell->is_locally_owned() || cell->is_ghost())
2519 {
2520 cell->get_dof_indices(local_dof_indices);
2521 // If the material_id is > 0, the region is a solid region
2522 if (cell->material_id() > 0)
2523 {
2524 // First check if the cell is connected to a fluid cell by
2525 // checking if one of the DOF of the cell is connected to a fluid
2526 // cell.
2527 bool connected_to_fluid =
2528 check_cell_is_connected_to_fluid(dofs_are_connected_to_fluid,
2529 local_dof_indices);
2530
2531 // All the pressure DOFs with the cell are not connected to the
2532 // fluid. Consequently, we fix a constraint on these pressure
2533 // DOFs.
2534 if (!connected_to_fluid)
2535 {
2536 constrain_pressure(non_zero_constraints,
2537 local_dof_indices,
2538 this->zero_constraints);
2539 }
2540 }

2599void
2600NavierStokesBase<dim, VectorType, DofsType>::
2601 constrain_stasis_with_temperature_vof(const DoFHandler<dim> *dof_handler_vof,
2602 const DoFHandler<dim> *dof_handler_ht)
2603{
2604 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
2605 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
2606
2607 // Get domain restriction with plane information
2608 bool restrain_domain_with_plane =
2609 this->simulation_parameters.constrain_solid_domain
2610 .enable_domain_restriction_with_plane;
2611 Point<dim> plane_point(
2612 this->simulation_parameters.constrain_solid_domain.restriction_plane_point);
2613 Tensor<1, dim> plane_normal_vector(
2614 this->simulation_parameters.constrain_solid_domain
2615 .restriction_plane_normal_vector);
2616
2617 // Get filtered phase fraction solution
2618 const auto filtered_phase_fraction_solution =
2619 *this->multiphysics->get_filtered_solution(PhysicsID::VOF);
2620 std::vector<double> local_filtered_phase_fraction_values(
2621 this->cell_quadrature->size());
2622
2623 // Get temperature solution
2624 const auto temperature_solution =
2625 *this->multiphysics->get_solution(PhysicsID::heat_transfer);
2626 std::vector<double> local_temperature_values(this->cell_quadrature->size());
2627
2628 // Loop over structs containing fluid id, temperature and phase fraction range
2629 // information, and flag containers for DOFs.
2630 for (StasisConstraintWithTemperature &stasis_constraint_struct :
2631 this->stasis_constraint_structs)
2632 {
2633 for (const auto &cell : dof_handler.active_cell_iterators())
2634 {
2635 if (cell->is_locally_owned() || cell->is_ghost())
2636 {
2637 cell->get_dof_indices(local_dof_indices);
2638
2639 // If a restriction plane is defined, check if the cell is in the
2640 // valid domain.
2641 if (!restrain_domain_with_plane ||
2642 cell_in_constraining_domain(cell,
2643 plane_point,
2644 plane_normal_vector))
2645 {
2646 bool cell_is_in_right_fluid = true;

2648 cell,
2649 dof_handler_vof,
2650 filtered_phase_fraction_solution,
2651 local_filtered_phase_fraction_values);

2654 // one filtered phase fraction value is outside the tolerated
2655 // range, the cell is perceived as being in the wrong fluid.
2656 for (const double &filtered_phase_fraction :
2657 local_filtered_phase_fraction_values)

2666 }
2667 }

2670 // will be applied on the cell's DOFs; we skip to the next
2671 // cell.
2672 if (!cell_is_in_right_fluid)
2673 continue;
2674
2675 get_cell_temperature_values(cell,
2676 dof_handler_ht,
2677 temperature_solution,
2678 local_temperature_values);
2679 identify_cell_and_constrain_velocity(
2680 local_dof_indices,
2681 local_temperature_values,
2682 stasis_constraint_struct);
2683 }
2684 }
2685 }
