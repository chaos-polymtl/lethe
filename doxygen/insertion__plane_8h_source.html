9/**
10 * @brief Insertion of particles using cells cut by a plane.
11 * Particle insertion using cells cut by a plane. Locally owned cells that are
12 * cut by the plane are flagged. From those flagged cells, we insert a particle
13 * at their centroids if they are individually empty (contain no particle). This
14 * way, no significant overlap is occurring on the insertion of new particle
15 * which can occur with other insertion method.
16 */
17template <int dim, typename PropertiesIndex>
18class InsertionPlane : public Insertion<dim, PropertiesIndex>
19{
20public:
21 /**
22 * The plane insertion class insert particles using a plane
23 * define by a point and a normal vector. This method of insertion can be
24 * useful when dealing with a domain close to be fully filled with particle.
25 * In this situation, other insertion method have a high risk to create a big
26 * overlap between particles on insertion. The plane insertion method mitigate
27 * this risk by insertion at the center of empty cells.
28 *
29 * @param size_distribution_object_container Contains all distribution for each
30 * particle type
31 * @param triangulation Triangulation to access the cells in which the
32 * particles are inserted
33 * @param dem_parameters DEM parameters declared in the .prm file
34 */
35 InsertionPlane(const std::vector<std::shared_ptr<Distribution>>
36 &size_distribution_object_container,
37 const parallel::distributed::Triangulation<dim> &triangulation,
38 const DEMSolverParameters<dim> &dem_parameters);
39
40 /**
41 * Carries out the insertion of particles.
42 * @param particle_handler The particle handler of particles which are being
43 * inserted.
44 * @param triangulation Triangulation to access the cells in which the
45 * particles are inserted.
46 * @param dem_parameters DEM parameters declared in the .prm file.
47 */
48 virtual void
49 insert(Particles::ParticleHandler<dim> &particle_handler,
50 const parallel::distributed::Triangulation<dim> &triangulation,
51 const DEMSolverParameters<dim> &dem_parameters) override;
52
