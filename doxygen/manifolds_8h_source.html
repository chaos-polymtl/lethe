83  * @brief BoundaryPostprocessor Post-processor class used to attach the boundary
84  * id to the faces when outputting the surfaces within the domain.
85  */
86 template <int dim>
87 class BoundaryPostprocessor : public DataPostprocessorScalar<dim>
88 {
89 public:
90  BoundaryPostprocessor()
91  : DataPostprocessorScalar<dim>("boundary_id", update_quadrature_points)
92  {}
93  virtual void
94  evaluate_vector_field(
95  const DataPostprocessorInputs::Vector<dim> &input_data,
96  std::vector<Vector<double>> &computed_quantities) const override
97  {
98  auto current_cell = input_data.template get_cell<dim>();
99 
100  for (unsigned int p = 0; p < input_data.evaluation_points.size(); ++p)
101  {
102  unsigned int boundary_id = 0;
103  double min_distance = DBL_MAX;
104 
105  for (const auto face : current_cell->face_indices())
106  {
107  if (current_cell->face(face)->at_boundary())
108  {
109  for (const auto vertex :
110  current_cell->face(face)->vertex_indices())
111  {
112  double distance = input_data.evaluation_points[p].distance(
113  current_cell->face(face)->vertex(vertex));
114  if (distance < min_distance)
115  {
116  min_distance = distance;
117 
118  boundary_id = current_cell->face(face)->boundary_id();
119  }
120  }
121  }
122  }
123  computed_quantities[p][0] = boundary_id;
124  }
125  };
