68/**
69 * @brief BoundaryPostprocessor Post-processor class used to attach the boundary
70 * id to the faces when outputting the surfaces within the domain.
71 */
72template <int dim>
73class BoundaryPostprocessor : public DataPostprocessorScalar<dim>
74{
75public:
76 BoundaryPostprocessor()
77 : DataPostprocessorScalar<dim>("boundary_id", update_quadrature_points)
78 {}
79 virtual void
80 evaluate_vector_field(
81 const DataPostprocessorInputs::Vector<dim> &input_data,
82 std::vector<Vector<double>> &computed_quantities) const override
83 {
84 auto current_cell = input_data.template get_cell<dim>();
85
86 for (unsigned int p = 0; p < input_data.evaluation_points.size(); ++p)
87 {
88 unsigned int boundary_id = 0;
89 double min_distance = DBL_MAX;
90
91 for (const auto face : current_cell->face_indices())
92 {
93 if (current_cell->face(face)->at_boundary())
94 {
95 for (const auto vertex :
96 current_cell->face(face)->vertex_indices())
97 {
98 double distance = input_data.evaluation_points[p].distance(
99 current_cell->face(face)->vertex(vertex));
100 if (distance < min_distance)
101 {
102 min_distance = distance;
103
104 boundary_id = current_cell->face(face)->boundary_id();
105 }
106 }
107 }
108 }
109 computed_quantities[p][0] = boundary_id;
110 }
111 };
112

114 evaluate_scalar_field(
