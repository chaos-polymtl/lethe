13template <int dim, int spacedim>
14GridUniformChannelWithMeshedSquarePrism<dim, spacedim>::
15 GridUniformChannelWithMeshedSquarePrism(const std::string &grid_arguments)
16{
17 if constexpr (dim == 1 || spacedim == 1)
18 {
19 AssertThrow(
20 false,
21 ExcMessage(
22 "GridUniformChannelWithMeshedSquarePrism is only supported for <2,2> and <3,3> <dim,spacedim> specializations."));
23 return;
24 }
25 else if constexpr (dim == 2 && spacedim == 3)
26 {
27 AssertThrow(
28 false,
29 ExcMessage(
30 "GridUniformChannelWithMeshedSquarePrism is only supported for <2,2> and <3,3> <dim,spacedim> specializations."));
31 return;
32 }
33
34 const std::vector<std::string> arguments =
35 Utilities::split_string_list(grid_arguments, ':');
36
37 if (arguments.size() < 5)
38 {
39 AssertThrow(
40 false,
41 ExcMessage(
42 "Mandatory parameters are (bottom left point : top right point : center of the obstacle : inner half-side : outer half-side). The points should be given as x,y and the half-sides should be single numbers. The optional parameters are (rotation_deg : pad_bottom : pad_top : pad_left : pad_right : height : n_slices : colorize )."));
43 }
44
45 // Parse bottom_left point
46 try
47 {
48 Tensor<1, 2> bottom_left_coords = value_string_to_tensor<2>(arguments[0]);
49 this->bottom_left =
50 (dim == 2) ?
51 Point<dim>(bottom_left_coords[0], bottom_left_coords[1]) :
52 Point<dim>(bottom_left_coords[0], bottom_left_coords[1], 0.0);
53 }
54 catch (const std::exception &e)
55 {
56 AssertThrow(
57 false,
58 ExcMessage(
59 "The bottom left point should have 2 components coordinates (x,y format) because the channel is extruded in the z direction."));
60 }
61
62 // Parse top_right point
63 try
64 {
65 Tensor<1, 2> top_right_coords = value_string_to_tensor<2>(arguments[1]);
66 this->top_right =
67 (dim == 2) ? Point<dim>(top_right_coords[0], top_right_coords[1]) :
68 Point<dim>(top_right_coords[0], top_right_coords[1], 0.0);
69 }
70 catch (const std::exception &e)
71 {
72 AssertThrow(
73 false,
74 ExcMessage(
75 "The top right point should have 2 components coordinates (x,y format) because the channel is extruded in the z direction."));
76 }
77
78
79 // Parse center point
80 try
81 {
82 Tensor<1, 2> center_coords = value_string_to_tensor<2>(arguments[2]);
83 this->center = (dim == 2) ?
84 Point<dim>(center_coords[0], center_coords[1]) :
85 Point<dim>(center_coords[0], center_coords[1], 0.0);
86 }
87 catch (const std::exception &e)
88 {
89 AssertThrow(
90 false,
91 ExcMessage(
92 "The center point should have 2 components coordinates (x,y format) because the channel is extruded in the z direction."));
93 }
94
95
96 // Parse half-sides of inner and outer squares
97 this->inner_half_side = Utilities::string_to_double(arguments[3]);
98 this->outer_half_side = Utilities::string_to_double(arguments[4]);
99
100 AssertThrow(
101 inner_half_side > 0.0,
102 ExcMessage(
103 " The inner_half_side must be greater than 0 to have an object in the channel."));
104
105 AssertThrow(
106 outer_half_side >= 1.5 * inner_half_side - 1e-12 * inner_half_side,
107 ExcMessage(
108 "The outer half-side must be at least 1.5 times the inner half-side, if not the elements in the transition region will be too distorted."));
109
110
111 // Check if the outer transition square fits in the channel
112 AssertThrow((center[0] - outer_half_side >= bottom_left[0]) &&
113 (center[0] + outer_half_side <= top_right[0]) &&
114 (center[1] - outer_half_side >= bottom_left[1]) &&
115 (center[1] + outer_half_side <= top_right[1]),
116 ExcMessage(
117 "The outer transition square does not fit in the channel."));
118
119 // Parse optional parameters if provided
120 this->inner_rotation_angle =
121 (arguments.size() > 5) ? Utilities::string_to_double(arguments[5]) : 0.0;
122
123 AssertThrow(
124 (inner_rotation_angle >= 0.0) && (inner_rotation_angle < 90.0),
125 ExcMessage("The rotation angle needs to be in the range [0, 90) degrees."));
126
127 inner_rotation_angle = inner_rotation_angle * std::numbers::pi / 180.0;
128
129 this->pad_bottom =
130 (arguments.size() > 6) ? Utilities::string_to_int(arguments[6]) : 0;
131 this->pad_top =
132 (arguments.size() > 7) ? Utilities::string_to_int(arguments[7]) : 0;
133 this->pad_left =
134 (arguments.size() > 8) ? Utilities::string_to_int(arguments[8]) : 0;
135 this->pad_right =
136 (arguments.size() > 9) ? Utilities::string_to_int(arguments[9]) : 0;
137
138 // Padding is required in each direction where the transition square does not
139 // reach the channel boundary (i.e., there is a gap to fill with rectangles).
140 AssertThrow(!((pad_bottom == 0) && (std::abs((center[1] - outer_half_side) -
141 bottom_left[1]) > 1e-12)),
142 ExcMessage("Bottom padding is required because the transition "
143 "region does not reach the bottom channel boundary."));
144 AssertThrow(!((pad_top == 0) && (std::abs((center[1] + outer_half_side) -
145 top_right[1]) > 1e-12)),
146 ExcMessage("Top padding is required because the transition "
147 "region does not reach the top channel boundary."));
148 AssertThrow(!((pad_left == 0) && (std::abs((center[0] - outer_half_side) -
149 bottom_left[0]) > 1e-12)),
150 ExcMessage("Left padding is required because the transition "
151 "region does not reach the left channel boundary."));
152 AssertThrow(!((pad_right == 0) && (std::abs((center[0] + outer_half_side) -
153 top_right[0]) > 1e-12)),
154 ExcMessage("Right padding is required because the transition "
155 "region does not reach the right channel boundary."));
156
157 this->height =
158 (arguments.size() > 10) ? Utilities::string_to_double(arguments[10]) : 1.0;
159 this->n_slices =
160 (arguments.size() > 11) ? Utilities::string_to_int(arguments[11]) : 2;
161 this->colorize =
162 (arguments.size() > 12 && arguments[12] == "true") ? true : false;
163}

165template <int dim, int spacedim>
166void
167GridUniformChannelWithMeshedSquarePrism<dim, spacedim>::
168 generate_2d_channel_mesh(Triangulation<2> &triangulation,
169 const Point<2> &bottom_left,
170 const Point<2> &top_right,
171 const Point<2> &center,
172 const double inner_half_side,
173 const double outer_half_side,
174 const double inner_rotation_angle,
175 const unsigned int pad_bottom,
176 const unsigned int pad_top,
177 const unsigned int pad_left,
178 const unsigned int pad_right,
179 const bool colorize)
180{
181 const types::material_id fluid_id = 0;
182 const types::material_id obstacle_id = 1;
183 const double tol_inner = 1e-12 * inner_half_side;
184
185 Triangulation<2> obstacle_tria;
186 // Start with a circular obstacle mesh, which helps define the edges and
187 // vertices links. The circular shape is then transformed to the square
188 // obstacle by moving the vertices. The outer ring of the circular mesh is
189 // used to define the transition region and is attached to the outer square by
190 // vertex projection, the inner vertices are shaped to form the inner square.
191 // To ensure that the inner square vertices can be easily identified even
192 // when rotated, the initial circular mesh is generated using twice the
193 // inner_half_side as radius and every vertex can be identified using a
194 // distance criterion.
195 GridGenerator::hyper_ball_balanced(obstacle_tria,
196 center,
197 2 * inner_half_side);
198
199 // We apply a transformation to shape inner vertices to form the inner square
200 GridTools::transform(
201 [&](const Point<2> &p) {
202 const double dist = p.distance(center);
203
204 // If the vertex is the one in the center, we keep it fixed.
205 if (dist <= tol_inner)
206 return p;
207
208 // If the vertex is within sqrt(2)*inner_half_side, it cannot be on the
209 // outer ring as we defined it to be 2*inner_half_side, so it is an inner
210 // vertex. We move it to the inner square by scaling the vector from the
211 // center to the vertex. The second condition checks if the vertex is on
212 // the diagonal (within a tolerance) in which case we need to scale it to
213 // the corner instead of the face of the inner square.
214 if ((dist < std::numbers::sqrt2 * inner_half_side + tol_inner) &&
215 (std::abs(std::abs(p[0] - center[0]) - std::abs(p[1] - center[1])) <
216 tol_inner))
217 return center +
218 std::numbers::sqrt2 * inner_half_side * (p - center) / dist;
219 // This is for non corner inner vertices, we scale them to the face of the
220 // inner square.
221 if ((dist < std::numbers::sqrt2 * inner_half_side + tol_inner) &&
222 (std::abs(std::abs(p[0] - center[0]) - std::abs(p[1] - center[1])) >=
223 tol_inner))
224 return center + (inner_half_side / dist) * (p - center);
225 return p;
226 },
227 obstacle_tria);
228
229 // We apply the rotation to the whole mesh if a rotation angle is specified.
230 if (inner_rotation_angle != 0.0)
231 {
232 const double cos_a = std::cos(inner_rotation_angle);
233 const double sin_a = std::sin(inner_rotation_angle);
234 GridTools::transform(
235 [&](const Point<2> &p) {
236 const double x_rel = p[0] - center[0];
237 const double y_rel = p[1] - center[1];
238 return Point<2>(center[0] + cos_a * x_rel - sin_a * y_rel,
239 center[1] + sin_a * x_rel + cos_a * y_rel);
240 },
241 obstacle_tria);
242 }
243
244 // Here we pre-compute the 8 projected positions of vertices that we want for
245 // the outer square. The outer ring of hyper_ball_balanced has vertices at
246 // angles k*pi/4. After rotation they sit at inner_rotation_angle + k*pi/4.
247 // The projection maps each direction onto the outer square along the same
248 // line. To avoid having voids in the corners we choose the subset of the 4
249 // target points that are closest to corners and change the target projection
250 // to attach them there instead.
251 std::array<Point<2>, 8> outer_targets;
252 for (unsigned int k = 0; k < 8; ++k)
253 {
254 double theta = inner_rotation_angle + k * std::numbers::pi / 4.0;
255
256 // If the rotation angle is between pi/18 and 4*pi/9, we want to project
257 // the vertices at the square middle points to the corners.
258 if (inner_rotation_angle >= std::numbers::pi / 18.0 - tol_inner &&
259 inner_rotation_angle < 4 * std::numbers::pi / 9.0 - tol_inner)
260 {
261 // The even k are associated to the vertices at the square middle
262 // points.
263 if (k % 2 == 0)
264 {
265 theta = (k + 1) * std::numbers::pi / 4.0;
266 }
267 }
268 // If the rotation angle is between 0 and pi/18 or between 4*pi/9 and
269 // pi/2, we want to project the vertices at the square corners to the
270 // corners.
271 else if (inner_rotation_angle >= 0.0 - tol_inner &&
272 inner_rotation_angle < std::numbers::pi / 18.0 - tol_inner)
273 {
274 // The odd k are associated to the vertices at the square corners.
275 if (k % 2 == 1)
276 {
277 theta = k * std::numbers::pi / 4.0;
278 }
279 }
280 else if (inner_rotation_angle >= 4 * std::numbers::pi / 9.0 - tol_inner &&
281 inner_rotation_angle < std::numbers::pi / 2.0 - tol_inner)
282 {
283 if (k % 2 == 1)
284 {
285 theta = (k + 2) * std::numbers::pi / 4.0;
286 }
287 }
288
289 const double cx = std::cos(theta);
290 const double cy = std::sin(theta);
291 const double scale =
292 outer_half_side / std::max(std::abs(cx), std::abs(cy));
293 outer_targets[k] = center + Point<2>(scale * cx, scale * cy);
294 }
295
296 // Attach each outer ring vertex to its analytical target (closest by angle).
297 // Using angle difference avoids the Euclidean distance ambiguity and ensures
298 // exact vertex matching with the padding meshes.
299 GridTools::transform(
300 [&](const Point<2> &p) {
301 const double dist = p.distance(center);
302
303 // Check if the vertex is an inside vertex. The outer vertex still lies on
304 // the circle defined by the hyper_ball_balanced, so we can use the same
305 // distance criterion as before to identify them.
306 if (dist <= std::numbers::sqrt2 * inner_half_side + tol_inner)
307 return p;
308
309 double min_angle = -1.0;
310 Point<2> closest = outer_targets[0];
311 // We compute the unitary vector from the center to the vertex and do the
312 // dot product with the unitary vector from the center to each target. The
313 // one with the highest cosine is the closest by angle and we project to
314 // that one (when the orientation is superposed, the cosine is 1, when it
315 // is orthogonal, the cosine is 0, and when it is opposite, the cosine is
316 // -1).
317 const Tensor<1, 2> p_normalize = (p - center) / (p - center).norm();
318 for (const Point<2> &candidate : outer_targets)
319 {
320 const Tensor<1, 2> target_normalize =
321 (candidate - center) / (candidate - center).norm();
322 const double cos_angle = p_normalize * target_normalize;
323
324 if (cos_angle > min_angle)
325 {
326 min_angle = cos_angle;
327 closest = candidate;
328 }
329 }
330 return closest;
331 },
332 obstacle_tria);
333
334 // Collect the projected x/y coordinates per outer-square face.
335 // Vertices landing exactly on a corner (|cos θ| == |sin θ|) are skipped
336 // because they already coincide with the corner of the padding rectangle.
337 // This is used to get the padding meshes to have vertices at the exact same
338 // positions so that merge_triangulations can merge them without needing to
339 // introduce new vertices.
340 std::vector<double> top_x, bottom_x, left_y, right_y;
341 for (unsigned int k = 0; k < 8; ++k)
342 {
343 const Point<2> &q = outer_targets[k];
344 const double theta = std::atan2(q[1] - center[1], q[0] - center[0]);
345 const double abs_cx = std::abs(std::cos(theta));
346 const double abs_cy = std::abs(std::sin(theta));
347
348 // If the vertex lands on the corner, we do not need to add a breakpoint
349 // for the padding meshes because they already have a vertex there.
350 if (std::abs(abs_cx - abs_cy) < 1e-12)
351 continue;
352
353 // If cos θ is larger than sin θ, then the vertex is either on the left or
354 // right face.
355 if (abs_cx > abs_cy)
356 {
357 if (std::cos(theta) > 0.0)
358 right_y.push_back(q[1]);
359 else
360 left_y.push_back(q[1]);
361 }
362 // If sin θ is larger than cos θ, then the vertex is either on the top or
363 // bottom face.
364 else
365 {
366 if (std::sin(theta) > 0.0)
367 top_x.push_back(q[0]);
368 else
369 bottom_x.push_back(q[0]);
370 }
371 }
372 std::ranges::sort(top_x);
373 std::ranges::sort(bottom_x);
374 std::ranges::sort(left_y);
375 std::ranges::sort(right_y);
376
377 // Build a vector of step sizes from sorted breakpoints + outer bounds.
378 // This lets subdivided_hyper_rectangle place vertices exactly at the
379 // projected positions so merge_triangulations sees coincident nodes.
380 // Example:
381 // inner = {x1, x2}, lo = a, hi = b
382 // coords = {a, x1, x2, b}
383 // steps = {x1-a, x2-x1, b-x2}
384 auto make_steps = [](const std::vector<double> &inner_coords,
385 const double lo,
386 const double hi) {
387 // Create the coordinates list
388 std::vector<double> coords = {lo};
389 for (const double c : inner_coords)
390 coords.push_back(c);
391 coords.push_back(hi);
392
393 // Create the steps list
394 std::vector<double> steps;
395 for (std::size_t i = 1; i < coords.size(); ++i)
396 steps.push_back(coords[i] - coords[i - 1]);
397 return steps;
398 };
399
400 // Similar as above, this lambda function creates a vector of uniform step
401 // sizes given the number of steps n and the total length to cover.
402 auto uniform = [](const unsigned int n, const double length) {
403 return std::vector<double>(n, length / static_cast<double>(n));
404 };
405
406 // Bottom pad: columns split at bottom_x projected coords
407 Triangulation<2> pad_bottom_tria;
408 if (pad_bottom > 0)
409 GridGenerator::subdivided_hyper_rectangle(
410 pad_bottom_tria,
411 {make_steps(bottom_x,
412 center[0] - outer_half_side,
413 center[0] + outer_half_side),
414 uniform(pad_bottom, center[1] - outer_half_side - bottom_left[1])},
415 Point<2>(center[0] - outer_half_side, bottom_left[1]),
416 Point<2>(center[0] + outer_half_side, center[1] - outer_half_side));
417
418 // Top pad: columns split at top_x projected coords
419 Triangulation<2> pad_top_tria;
420 if (pad_top > 0)
421 GridGenerator::subdivided_hyper_rectangle(
422 pad_top_tria,
423 {make_steps(top_x,
424 center[0] - outer_half_side,
425 center[0] + outer_half_side),
426 uniform(pad_top, top_right[1] - (center[1] + outer_half_side))},
427 Point<2>(center[0] - outer_half_side, center[1] + outer_half_side),
428 Point<2>(center[0] + outer_half_side, top_right[1]));
429
430 // Left pad: rows split at left_y projected coords
431 Triangulation<2> pad_left_tria;
432 if (pad_left > 0)
433 GridGenerator::subdivided_hyper_rectangle(
434 pad_left_tria,
435 {uniform(pad_left, center[0] - outer_half_side - bottom_left[0]),
436 make_steps(left_y,
437 center[1] - outer_half_side,
438 center[1] + outer_half_side)},
439 Point<2>(bottom_left[0], center[1] - outer_half_side),
440 Point<2>(center[0] - outer_half_side, center[1] + outer_half_side));
441
442 // Right pad: rows split at right_y projected coords
443 Triangulation<2> pad_right_tria;
444 if (pad_right > 0)
445 GridGenerator::subdivided_hyper_rectangle(
446 pad_right_tria,
447 {uniform(pad_right, top_right[0] - (center[0] + outer_half_side)),
448 make_steps(right_y,
449 center[1] - outer_half_side,
450 center[1] + outer_half_side)},
451 Point<2>(center[0] + outer_half_side, center[1] - outer_half_side),
452 Point<2>(top_right[0], center[1] + outer_half_side));
453
454 // Corner pads never touch the obstacle tria boundary, so no special split
455 // needed
456 Triangulation<2> pad_bottom_left_corner_tria;
457 if (pad_bottom > 0 && pad_left > 0)
458 GridGenerator::subdivided_hyper_rectangle(
459 pad_bottom_left_corner_tria,
460 {pad_left, pad_bottom},
461 bottom_left,
462 Point<2>(center[0] - outer_half_side, center[1] - outer_half_side));
463
464 Triangulation<2> pad_bottom_right_corner_tria;
465 if (pad_bottom > 0 && pad_right > 0)
466 GridGenerator::subdivided_hyper_rectangle(
467 pad_bottom_right_corner_tria,
468 {pad_right, pad_bottom},
469 Point<2>(center[0] + outer_half_side, bottom_left[1]),
470 Point<2>(top_right[0], center[1] - outer_half_side));
471
472 Triangulation<2> pad_top_left_corner_tria;
473 if (pad_top > 0 && pad_left > 0)
474 GridGenerator::subdivided_hyper_rectangle(
475 pad_top_left_corner_tria,
476 {pad_left, pad_top},
477 Point<2>(bottom_left[0], center[1] + outer_half_side),
478 Point<2>(center[0] - outer_half_side, top_right[1]));
479
480 Triangulation<2> pad_top_right_corner_tria;
481 if (pad_top > 0 && pad_right > 0)
482 GridGenerator::subdivided_hyper_rectangle(
483 pad_top_right_corner_tria,
484 {pad_right, pad_top},
485 Point<2>(center[0] + outer_half_side, center[1] + outer_half_side),
486 top_right);
487
488 // Merge only non-empty triangulations
489 std::vector<const Triangulation<2> *> trias = {&obstacle_tria};
490 if (pad_bottom > 0)
491 trias.push_back(&pad_bottom_tria);
492 if (pad_top > 0)
493 trias.push_back(&pad_top_tria);
494 if (pad_left > 0)
495 trias.push_back(&pad_left_tria);
496 if (pad_right > 0)
497 trias.push_back(&pad_right_tria);
498 if (pad_bottom > 0 && pad_left > 0)
499 trias.push_back(&pad_bottom_left_corner_tria);
500 if (pad_bottom > 0 && pad_right > 0)
501 trias.push_back(&pad_bottom_right_corner_tria);
502 if (pad_top > 0 && pad_left > 0)
503 trias.push_back(&pad_top_left_corner_tria);
504 if (pad_top > 0 && pad_right > 0)
505 trias.push_back(&pad_top_right_corner_tria);
506 GridGenerator::merge_triangulations(trias, triangulation);
507
508 triangulation.reset_all_manifolds();
509 triangulation.set_all_manifold_ids(0);
510
511 // Here we mark obstacle cells with the obstacle material ID. This is done in
512 // the unrotated frame by rotating the cell center back to the axis-aligned
513 // frame temporarily and checking if it is within the inner square dimensions.
514 for (const auto &cell : triangulation.active_cell_iterators())
515 {
516 const Point<2> c = cell->center();
517
518 // Translate to obstacle-centered coordinates
519 const double dx = c[0] - center[0];
520 const double dy = c[1] - center[1];
521
522 // Rotate point back to axis-aligned frame
523 const double cos_a = std::cos(inner_rotation_angle);
524 const double sin_a = std::sin(inner_rotation_angle);
525
526 const double x_local = cos_a * dx + sin_a * dy;
527 const double y_local = -sin_a * dx + cos_a * dy;
528
529 // Axis-aligned square test
530 const bool inside_obstacle =
531 (std::abs(x_local) <= inner_half_side + tol_inner) &&
532 (std::abs(y_local) <= inner_half_side + tol_inner);
533
534 cell->set_material_id(inside_obstacle ? obstacle_id : fluid_id);
535 }
536
537 // Assign boundary IDs following the subdivided_hyper_rectangle convention:
538 // 0: left (-x), 1: right (+x), 2: bottom (-y), 3: top (+y)
539 if (colorize)
540 {
541 const double tol_x = 1e-10 * (top_right[0] - bottom_left[0]);
542 const double tol_y = 1e-10 * (top_right[1] - bottom_left[1]);
543 for (const auto &face : triangulation.active_face_iterators())
544 {
545 if (!face->at_boundary())
546 continue;
547 const Point<2> fc = face->center();
548 if (std::abs(fc[0] - bottom_left[0]) < tol_x)
549 face->set_boundary_id(0);
550 else if (std::abs(fc[0] - top_right[0]) < tol_x)
551 face->set_boundary_id(1);
552 else if (std::abs(fc[1] - bottom_left[1]) < tol_y)
553 face->set_boundary_id(2);
554 else if (std::abs(fc[1] - top_right[1]) < tol_y)
555 face->set_boundary_id(3);
556 }
557 }
558}

560template <>
561void
562GridUniformChannelWithMeshedSquarePrism<2, 2>::make_grid(
563 Triangulation<2, 2> &triangulation)
564{
565 generate_2d_channel_mesh(triangulation,
566 bottom_left,
567 top_right,
568 center,
569 inner_half_side,
570 outer_half_side,
571 inner_rotation_angle,
572 pad_bottom,
573 pad_top,
574 pad_left,
575 pad_right,
576 colorize);
577
578 // Attach manifold for id 0 required by deal.II
579 TransfiniteInterpolationManifold<2> tfi_manifold;
580 tfi_manifold.initialize(triangulation);
581 triangulation.set_manifold(0, tfi_manifold);
582}

585template <>
586void
587GridUniformChannelWithMeshedSquarePrism<3, 3>::make_grid(
588 Triangulation<3, 3> &triangulation)
589{
590 // Generate the 2D cross-section (geometry + material IDs + boundary IDs)
591 Triangulation<2> tria_2D;
592 generate_2d_channel_mesh(tria_2D,
593 Point<2>(bottom_left[0], bottom_left[1]),
