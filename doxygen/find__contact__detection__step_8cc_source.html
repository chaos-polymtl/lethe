9template <int dim, typename PropertiesIndex>
10void
11find_particle_contact_detection_step(
12 Particles::ParticleHandler<dim> &particle_handler,
13 const double dt,
14 const double smallest_contact_search_criterion,
15 MPI_Comm &mpi_communicator,
16 std::vector<double> &displacement,
17 const bool parallel_update)
18{
19 // Get the action manager
20 auto *action_manager = DEMActionManager::get_action_manager();
21 // If something else has already triggered contact search,
22 // no need to do it again
23 if (action_manager->check_contact_search())
24 return;
25
26 double max_displacement = 0.;
27 bool contact_detection_step = false;
28
29 // Updating displacement
30 for (auto &particle : particle_handler)
31 {
32 auto particle_properties = particle.get_properties();
33 auto &particle_displacement = displacement[particle.get_local_index()];
34
35 // Finding displacement of each particle during last step
36 particle_displacement +=
37 dt * sqrt(particle_properties[PropertiesIndex::v_x] *
38 particle_properties[PropertiesIndex::v_x] +
39 particle_properties[PropertiesIndex::v_y] *
40 particle_properties[PropertiesIndex::v_y] +
41 particle_properties[PropertiesIndex::v_z] *
42 particle_properties[PropertiesIndex::v_z]);
43
44 // Updating maximum displacement of particles
45 max_displacement = std::max(max_displacement, particle_displacement);
46 }
47
48 // If the maximum displacement of particles exceeds criterion, this step
49 // is a contact detection step
50 contact_detection_step = max_displacement > smallest_contact_search_criterion;
51
52 // Broadcasting contact detection step value to other processors
53 if (parallel_update)
54 {
55 contact_detection_step =
56 Utilities::MPI::logical_or(contact_detection_step, mpi_communicator);
57 if (contact_detection_step)
58 action_manager->contact_detection_step();
59 }
60}

117template <int dim>
118void
119find_floating_mesh_mapping_step(
120 const double smallest_contact_search_criterion,
121 std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> solids)
122{
123 // Get the action manager
124 auto *action_manager = DEMActionManager::get_action_manager();
125
126 // If there is no solid object, no need to do anything
127 if (!action_manager->check_solid_objects_enabled())
128 return;
129
130 bool floating_mesh_requires_map = false;
131
132 for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
133 {
134 double displacement =
135 solids[i_solid]->get_max_displacement_since_mapped();
136 floating_mesh_requires_map =
137 floating_mesh_requires_map ||
138 displacement > smallest_contact_search_criterion;
139 }
140
141 if (floating_mesh_requires_map)
142 action_manager->solid_objects_search_step();
143}
