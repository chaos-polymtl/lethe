34 /**
35  * @brief Calculates the area of intersection between a circular (2D) particle and a circle
36  *
37  * @param r_particle Radius of the particle
38  *
39  * @param r_circle Radius of the circle
40  *
41  * @param neighbor_distance Distance between the particle and the circle
42  */
43 inline double
44 particle_circle_intersection_2d(double r_particle,
45  double r_circle,
46  double neighbor_distance)
47 {
48  return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
49  cos((pow(neighbor_distance, 2) +
50  pow(r_particle, 2) - pow(r_circle, 2)) /
51  (2 * neighbor_distance * r_particle))) +
52  Utilities::fixed_power<2, double>(r_circle) *
53  Utilities::fixed_power<-1, double>(
54  cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
55  pow(r_circle, 2)) /
56  (2 * neighbor_distance * r_circle))) -
57  0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
58  (neighbor_distance + r_particle - r_circle) *
59  (neighbor_distance - r_particle + r_circle) *
60  (neighbor_distance + r_particle + r_circle));
61 }

63 /**
64  * @brief Calculates the volume of intersection between a spherical (3D) particle and a sphere
65  *
66  * @param r_particle Radius of the particle
67  *
68  * @param r_sphere Radius of the sphere
69  *
70  * @param neighbor_distance Distance between the particle and the sphere
71  */
72 
73 inline double
74 particle_sphere_intersection_3d(double r_particle,
75  double r_sphere,
76  double neighbor_distance)
77 {
78  return M_PI *
79  Utilities::fixed_power<2, double>(r_sphere + r_particle -
80  neighbor_distance) *
81  (Utilities::fixed_power<2, double>(neighbor_distance) +
82  (2 * neighbor_distance * r_particle) -
83  (3 * Utilities::fixed_power<2, double>(r_particle)) +
84  (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
85  (3 * Utilities::fixed_power<2, double>(r_sphere))) /
86  (12 * neighbor_distance);
87 }

133  /**
134  * @brief This function calculates and returns the periodic offset distance of the domain which is needed
135  * for the periodic boundary conditions using the QCM or SPM for void fraction
136  * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
137  * periodic boundaries and all particle location shifted by this distance is
138  * according to this periodic boundary.
139  *
140  * @param boundary_id The id of one of the periodic boundaries
141  *
142  * @return The periodic offset distance
143  */
144  inline Tensor<1, dim>
145  get_periodic_offset_distance(unsigned int boundary_id) const
146  {
147  Tensor<1, dim> offset;
148 
149  // Iterating over the active cells in the triangulation
150  for (const auto &cell : (*this->triangulation).active_cell_iterators())
151  {
152  if (cell->is_locally_owned() || cell->is_ghost())
153  {
154  if (cell->at_boundary())
155  {
156  // Iterating over cell faces
157  for (unsigned int face_id = 0; face_id < cell->n_faces();
158  ++face_id)
159  {
160  unsigned int face_boundary_id =
161  cell->face(face_id)->boundary_id();
162 
163  // Check if face is on the boundary, if so, get
164  // the periodic offset distance for one pair of periodic
165  // faces only since periodic boundaries are aligned with the
166  // direction and only axis are currently allowed
167  if (face_boundary_id == boundary_id)
168  {
169  Point<dim> face_center = cell->face(face_id)->center();
170  auto periodic_cell = cell->periodic_neighbor(face_id);
171  unsigned int periodic_face_id =
172  cell->periodic_neighbor_face_no(face_id);
173  Point<dim> periodic_face_center =
174  periodic_cell->face(periodic_face_id)->center();
175 
176  offset = periodic_face_center - face_center;
177 
178  return offset;
179  }
180  }
181  }
182  }
183  }
184 
185  // A zero tensor is returned in case no cells are found on the periodic
186  // boundaries on this processor. This processor won't handle particle in
187  // cells at periodic boundaries, so it won't affect any computation.
188  return offset;
189  }
