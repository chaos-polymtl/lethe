546 std::string
547 get_last_value_of_parameter(const std::string &file_name,
548  const std::string &parameter_name)
549 {
550  std::string return_value;
551 
552  std::ifstream x_file(file_name);
553  AssertThrow(x_file.fail() == false, ExcIO());
554 
555  while (x_file)
556  {
557  // Get one line and then match a regex to it that matches the parameter
558  // we are looking for. Before we do that, strip spaces from the front
559  // and back of the line:
560  std::string line;
561  std::getline(x_file, line);
562 
563  while ((line.size() > 0) && (line[0] == ' ' || line[0] == '\t'))
564  line.erase(0, 1);
565  while ((line.size() > 0) &&
566  (line[line.size() - 1] == ' ' || line[line.size() - 1] == '\t'))
567  line.erase(line.size() - 1, std::string::npos);
568 
569  std::match_results<std::string::const_iterator> matches;
570  const std::string regex =
571  "set[ \t]+" + parameter_name + "[ \t]*=[ \t]*(.*)";
572  if (std::regex_match(line, matches, std::regex(regex)))
573  {
574  // Since the line as a whole matched, the 'matches' variable needs
575  // to contain two entries: [0] denotes the whole string, and [1] the
576  // one that was matched by the '(.*)' expression.
577  Assert(matches.size() == 2, dealii::ExcInternalError());
578  return_value = std::string(matches[1].first, matches[1].second);
579  }
580  }
581 
582  return return_value;
583 }

630 unsigned int
631 get_dimension(const std::string &file_name)
632 {
633  const std::string dimension =
634  get_last_value_of_parameter(file_name, "dimension");
635 
636  if (dimension.size() > 0)
637  {
638  // Extracted from ASPECT
639  // A common problem is that people have .prm files that were generated
640  // on Windows, but then run this on Linux where the line endings are
641  // different. This is pernicious because it means that the conversion
642  // of a string such as "2\r" to an integer fails, but if we print
643  // this string, it comes out completely garbled because it contains
644  // a carriage-return without a newline -- so the error message looks
645  // like this:
646  //
647  // >. While reading the dimension from the input file, Lethe found
648  // a string that can not be converted to an integer: <2
649  //
650  // Note how the end of the error message overwrites the beginning
651  // of the line.
652  //
653  // To avoid this kind of error, specifically test up front that the
654  // text in question does not contain '\r' characters. If we are on
655  // linux, then this kind of character would means that the line endings
656  // are wrong. On the other hand, if we are on windows, then the
657  // getline command we have used in finding 'dimension' would have
658  // filtered it out. So its presence points to a problem.
659 
660  AssertThrow(
661  dimension.find('\r') == std::string::npos,
662  dealii::ExcMessage(
663  "It appears that your input file uses Windows-style "
664  "line endings ('\\r\\n') but you are running on a system where "
665  "the C++ run time environment expects input files to have "
666  "Unix-style line endings ('\\n'). You need to convert your "
667  "input file to use the correct line endings before running "
668  "ASPECT with it."));
669  try
670  {
671  return dealii::Utilities::string_to_int(dimension);
672  }
673  catch (...)
674  {
675  AssertThrow(false,
676  dealii::ExcMessage(
677  "While reading the dimension from the input file, "
678  "Lethe found a string that can not be converted to "
679  "an integer: <" +
680  dimension + ">."));
681  return 0; // we should never get here.
682  }
683  }
684  else
685  {
686  AssertThrow(
687  false,
688  dealii::ExcMessage(
689  "While reading the dimension from the input file, "
690  "Lethe found a value that is neither 2 or 3. Since August 2023, "
691  "Lethe requires that the user explicitly specify the dimension of the problem within the parameter file. This can be achieved by adding set dimension = 2 or set dimension = 3 within the parameter file"));
692  }
693 }
