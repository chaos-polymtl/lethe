12/**
13 * @brief Computes the effective concentration above the threshold and with a
14 * smooth transition
15 *
16 * This function prevents NaN in the reaction rate computations when the
17 * reaction order is lower than 1 by introducing a small threshold value, to be
18 * set above 0.
19 *
20 * @param[in] C Concentration value
21 * @param[in] threshold Threshold value above which effective concentration will
22 * be
23 */
24inline double
25calculate_effective_concentration(const double C, const double threshold)
26{
27 return std::sqrt(C * C + threshold * threshold);
28}
29
30/**
31 * @brief Computes the derivative of effective concentration in regards to the
32 * input concentration
33 *
34 * @param[in] C Concentration value
35 * @param[in] threshold Threshold value above which effective concentration will
36 * be
37 */
38inline double
39calculate_d_effective_concentration_dC(const double C, const double threshold)
40{
41 const double Ce = calculate_effective_concentration(C, threshold);
42 return (Ce > 0.0) ? (C / Ce) : 0.0;
43}

45/**
46 * @brief Abstract class to calculate the tracer reaction prefactor.

48 * This model computes the prefactor

50 * such that the full reaction term is assembled as
51 * \f$ r = -k \, C \f$.

54 * This parent class could have children classes that are not based on the power
55 * law.
56 */

58{
59public:
60 /**

63 *
64 * @param material_properties Parameters for a material.
65 */
66 static std::shared_ptr<TracerReactionPrefactorModel>
67 model_cast(const Parameters::Material &material_properties);
68};

122/**
123 * @brief Constant tracer reaction prefactor.
124 *
125 * This model computes the prefactor as
126 * \f[ k = \alpha C_\text{eff}^{n-1}, \f]
127 * with a smooth floor \f[C_\text{eff} = \sqrt{C^2 + \epsilon^2} \f]
128 * where \f$ \alpha \f$ (the reaction constant) and the reaction order \f$ n \f$
129 * are provided.
130 */
131class ConstantTracerReactionPrefactor : public TracerReactionPrefactorModel
132{
133public:
134 /**
135 * @brief Constructor.
136 *
137 * @param[in] p_tracer_reaction_constant The base reaction constant \f$ \alpha
138 * \f$.
139 * @param[in] p_tracer_reaction_order The reaction order \f$ n \f$.
140 * @param[in] p_tracer_reaction_threshold The threshold value used to smoothly
141 * floor the concentration.
142 */
143 ConstantTracerReactionPrefactor(const double p_tracer_reaction_constant,
144 const double p_tracer_reaction_order,
145 const double p_tracer_reaction_threshold)
146 : tracer_reaction_constant(p_tracer_reaction_constant)
147 , tracer_reaction_order(p_tracer_reaction_order)
148 , tracer_reaction_threshold(p_tracer_reaction_threshold)
149 {
150 this->model_depends_on[field::tracer_concentration] = true;
151 }
152
153 /**
154 * @brief Compute the tracer reaction prefactor.
155 *
156 * @param[in] fields_value Map of field values.
157 * @return The computed prefactor value.
158 */
159 double
160 value(const std::map<field, double> &fields_value) override
161 {
162 Assert(fields_value.find(field::tracer_concentration) != fields_value.end(),
163 PhysicialPropertyModelFieldUndefined(
164 "ConstantTracerReactionPrefactor", "tracer_concentration"));
165
166 const double Ceff = calculate_effective_concentration(
167 fields_value.at(field::tracer_concentration), tracer_reaction_threshold);
168 return tracer_reaction_constant *
169 std::pow(Ceff, tracer_reaction_order - 1.);
170 }
171
172 /**
173 * @brief Compute the vector of tracer reaction prefactor values.
174 *
175 * @param[in] field_vectors Map of field vectors.
176 * @param property_vector Vector to be filled with computed prefactor values.
177 */
178 void
179 vector_value(const std::map<field, std::vector<double>> &field_vectors,
180 std::vector<double> &property_vector) override
181 {
182 Assert(
183 field_vectors.find(field::tracer_concentration) != field_vectors.end(),
184 PhysicialPropertyModelFieldUndefined("ConstantTracerReactionPrefactor",
185 "tracer_concentration"));
186 const auto &concentration_vector =
187 field_vectors.at(field::tracer_concentration);
188 for (size_t i = 0; i < property_vector.size(); ++i)
189 {
190 const double Ceff =
191 calculate_effective_concentration(concentration_vector[i],
192 tracer_reaction_threshold);
193 property_vector[i] =
194 tracer_reaction_constant * std::pow(Ceff, tracer_reaction_order - 1.);
195 }
196 }
197
198 /**
199 * @brief Compute the jacobian (partial derivative) of the prefactor with respect to a field.
200 *
201 * @param[in] field_values Map of field values.
202 * @param[in] id The field identifier with respect to which the derivative is
203 * computed.
204 * @return The computed derivative value.
205 */
206 double
207 jacobian(const std::map<field, double> &field_values, field id) override
208 {
209 Assert(field_values.find(field::tracer_concentration) != field_values.end(),
210 PhysicialPropertyModelFieldUndefined(
211 "ConstantTracerReactionPrefactor", "tracer_concentration"));
212 if (id == field::tracer_concentration)
213 {
214 const double C = field_values.at(field::tracer_concentration);
215 const double Ceff =
216 calculate_effective_concentration(C, tracer_reaction_threshold);
217 const double dCe =
218 calculate_d_effective_concentration_dC(C, tracer_reaction_threshold);
219
220 return tracer_reaction_constant * (tracer_reaction_order - 1.) *
221 std::pow(Ceff, tracer_reaction_order - 2.) * dCe;
222 }
223 return 0.0;
224 }
225
226 /**
227 * @brief Compute the vector jacobian (partial derivatives) of the prefactor with respect to a field.
228 *
229 * @param[in] field_vectors Map of field vectors.
230 * @param[in] id The field identifier with respect to which the derivative is
231 * computed.
232 * @param[out] jacobian_vector Vector to be filled with computed derivative
233 * values.
234 */
235 void
236 vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
237 const field id,
238 std::vector<double> &jacobian_vector) override
239 {
240 Assert(
241 field_vectors.find(field::tracer_concentration) != field_vectors.end(),
242 PhysicialPropertyModelFieldUndefined("ConstantTracerReactionPrefactor",
243 "tracer_concentration"));
244 if (id != field::tracer_concentration)
245 {
246 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0.0);
247 return;
248 }
249 const auto &concentration_vector =
250 field_vectors.at(field::tracer_concentration);
251 for (size_t i = 0; i < jacobian_vector.size(); ++i)
252 {
253 const double Ceff =
254 calculate_effective_concentration(concentration_vector[i],
255 tracer_reaction_threshold);
256 const double dCe =
257 calculate_d_effective_concentration_dC(concentration_vector[i],
258 tracer_reaction_threshold);
259 jacobian_vector[i] = tracer_reaction_constant *
260 (tracer_reaction_order - 1.) *
261 std::pow(Ceff, tracer_reaction_order - 2.) * dCe;
262 }
263 };
264
265private:
266 const double tracer_reaction_constant;
267 const double tracer_reaction_order;
268 const double tracer_reaction_threshold;

271/**
272 * @brief Tanh level-set-dependent tracer reaction prefactor.
273 *
274 * This model computes a prefactor that depends on the level-set function,
275 * smoothly transitioning between a reaction constant \‍(\alpha\‍) inside and
276 * outside the solid.
277 */
278class TanhLevelsetTracerReactionPrefactor : public TracerReactionPrefactorModel
279{
280public:
281 /**
282 * @brief Constructor.
283 *
284 * @param[in] p_tracer_reaction_constant_outside Reaction constant \‍(\alpha\‍)
285 * outside the solid.
286 * @param[in] p_tracer_reaction_constant_inside Reaction constant \‍(\alpha\‍)
287 * inside the solid.
288 * @param[in] p_thickness Thickness for the tanh smoothing function.
289 * @param[in] p_tracer_reaction_order The reaction order.
290 * @param[in] p_tracer_reaction_threshold The threshold value used to smoothly
291 * floor the concentration.
292 */
293 TanhLevelsetTracerReactionPrefactor(
294 const double p_tracer_reaction_constant_outside,
295 const double p_tracer_reaction_constant_inside,
296 const double p_thickness,
297 const double p_tracer_reaction_order,
298 const double p_tracer_reaction_threshold)
299 : tracer_reaction_constant_outside(p_tracer_reaction_constant_outside)
300 , tracer_reaction_constant_inside(p_tracer_reaction_constant_inside)
301 , thickness(p_thickness)
302 , delta_reaction_constant(tracer_reaction_constant_outside -
303 tracer_reaction_constant_inside)
304 , tracer_reaction_order(p_tracer_reaction_order)
305 , tracer_reaction_threshold(p_tracer_reaction_threshold)
306 {
307 this->model_depends_on[field::levelset] = true;
308 this->model_depends_on[field::tracer_concentration] = true;
309 }

311 /**
312 * @brief Compute the tracer reaction prefactor.
313 *
314 * @param[in] field_values Map of field values.
315 * @return The computed prefactor value.
316 */
317 double
318 value(const std::map<field, double> &field_values) override
319 {
320 Assert(field_values.find(field::levelset) != field_values.end(),
321 PhysicialPropertyModelFieldUndefined(
322 "TanhLevelsetTracerReactionPrefactor", "levelset"));
323 Assert(field_values.find(field::tracer_concentration) != field_values.end(),
324 PhysicialPropertyModelFieldUndefined(
325 "TanhLevelsetTracerReactionPrefactor", "tracer_concentration"));
326 const double levelset = field_values.at(field::levelset);
327 const double Ceff = calculate_effective_concentration(
328 field_values.at(field::tracer_concentration), tracer_reaction_threshold);
329 const double k =
330 tracer_reaction_constant_inside +
331 delta_reaction_constant * (0.5 + 0.5 * std::tanh(levelset / thickness));
332 return k * std::pow(Ceff, tracer_reaction_order - 1.);
333 }

335 /**
336 * @brief Compute a vector of tracer reaction prefactor values.
337 *
338 * @param[in] field_vectors Map of field vectors.
339 * @param[out] property_vector Vector to be filled with computed prefactor
340 * values.
341 */
342 void
343 vector_value(const std::map<field, std::vector<double>> &field_vectors,
344 std::vector<double> &property_vector) override
345 {
346 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
347 PhysicialPropertyModelFieldUndefined(
348 "TanhLevelsetTracerReactionPrefactor", "levelset"));
349 Assert(field_vectors.find(field::tracer_concentration) !=
350 field_vectors.end(),
351 PhysicialPropertyModelFieldUndefined(
352 "TanhLevelsetTracerReactionPrefactor", "tracer_concentration"));
353
354 const std::vector<double> &levelset_vec = field_vectors.at(field::levelset);
355 const std::vector<double> &concentration_vector =

357
358 const unsigned int n_values = levelset_vec.size();
359 Assert(n_values == concentration_vector.size(),
360 SizeOfFields(n_values, concentration_vector.size()));
361 for (unsigned int i = 0; i < n_values; ++i)
362 {

364 delta_reaction_constant *
365 (0.5 + 0.5 * std::tanh(levelset_vec[i] / thickness));
366 const double Ceff =
367 calculate_effective_concentration(concentration_vector[i],
368 tracer_reaction_threshold);

370 }

373 /**
374 * @brief Compute the jacobian (partial derivative) of the prefactor with respect to a specified field.
375 *
376 * @param[in] field_values Map of field values.
377 * @param[in] id The field identifier with respect to which the derivative is
378 * computed.
379 * @return The computed derivative value.
380 */
381 double
382 jacobian(const std::map<field, double> &field_values, field id) override
383 {
384 Assert(field_values.find(field::levelset) != field_values.end(),
385 PhysicialPropertyModelFieldUndefined(
386 "TanhLevelsetTracerReactionPrefactor", "levelset"));
387 Assert(field_values.find(field::tracer_concentration) != field_values.end(),
388 PhysicialPropertyModelFieldUndefined(
389 "TanhLevelsetTracerReactionPrefactor", "tracer_concentration"));
390 const double levelset = field_values.at(field::levelset);
391 const double concentration_val =
392 field_values.at(field::tracer_concentration);
393 const double Ceff =
394 calculate_effective_concentration(concentration_val,
395 tracer_reaction_threshold);
396 if (id == field::levelset)
397 {
398 // dk/dlambda for tanh profile: k * 0.5 * (1 - tanh^2(lambda/sigma)) *
399 // (1/sigma)
400 const double tanh = std::tanh(levelset / thickness);
401 const double dkdlambda =
402 delta_reaction_constant * 0.5 * (1.0 - tanh * tanh) / thickness;
403 return dkdlambda * std::pow(Ceff, tracer_reaction_order - 1.);
404 }
405 else if (id == field::tracer_concentration)
406 {
407 const double k = tracer_reaction_constant_inside +
408 delta_reaction_constant *
409 (0.5 + 0.5 * std::tanh(levelset / thickness));
410 const double dCe =
411 calculate_d_effective_concentration_dC(concentration_val,
412 tracer_reaction_threshold);
413 return k * (tracer_reaction_order - 1.) *
414 std::pow(Ceff, tracer_reaction_order - 2.) * dCe;
415 }
416 else
417 return 0.0;
418 }

445 if (id == field::levelset)

457 }
458 }
459 else if (id == field::tracer_concentration)
460 {
461 for (unsigned int i = 0; i < n_values; ++i)
462 {
463 const double k =
464 tracer_reaction_constant_inside +
465 delta_reaction_constant *
466 (0.5 + 0.5 * std::tanh(levelset_vec[i] / thickness));
467 const double Ceff =
468 calculate_effective_concentration(concentration_vec[i],
469 tracer_reaction_threshold);
470 const double dCe =
471 calculate_d_effective_concentration_dC(concentration_vec[i],

474 std::pow(Ceff, tracer_reaction_order - 2.) *
475 dCe;
476 }
477 }
478 else
479 {
480 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0.0);
481 }
482 }

486 const double tracer_reaction_constant_inside;
487 const double thickness;
488 const double delta_reaction_constant;
489 const double tracer_reaction_order;
490 const double tracer_reaction_threshold;

531 /**
532 * @brief Compute the reaction constant.
533 *
534 * @param[in] field_values Values of the various fields on which the property
535 * may depend. In this case, the reaction constant depends on the level set.
536 * The map stores a single value per field.
537 *
538 * @return Value of the reaction constant computed with the @p field_values.

540 double
541 value(const std::map<field, double> &field_values) override

602 /**
603 * @brief Compute the jacobian (the partial derivative) of the reaction constant
604 * with respect to a specified field.
605 *
606 * @param[in] field_values Values of the various fields on which the specific
607 * heat may depend. The map stores a single value per field.
608 *
609 * @param[in] id Indicator of the field with respect to which the jacobian
610 * should be computed.
611 *
612 * @return Value of the derivative of the reaction constant with respect to the
613 * specified field.
614 */
615 double
616 jacobian(const std::map<field, double> &field_values, field id) override
617 {
618 Assert(field_values.find(field::levelset) != field_values.end(),
619 PhysicialPropertyModelFieldUndefined(
620 "GaussianLevelsetTracerReactionPrefactor", "levelset"));
621 Assert(field_values.find(field::tracer_concentration) != field_values.end(),
622 PhysicialPropertyModelFieldUndefined(
623 "GaussianLevelsetTracerReactionPrefactor",
624 "tracer_concentration"));
625
626 const double levelset_val = field_values.at(field::levelset);
627 const double concentration_val =
628 field_values.at(field::tracer_concentration);
629 const double Ceff =
630 calculate_effective_concentration(concentration_val,
631 tracer_reaction_threshold);
632 if (id == field::levelset)
633 {
634 // dk/dlambda = k * exp(-lambda^2/sigma^2) * (-2*lambda/sigma^2)
635 const double exponential =
636 std::exp(-levelset_val * levelset_val / squared_thickness);
637 const double dkdlambda = delta_reaction_constant * exponential *
638 (-2.0 * levelset_val / squared_thickness);
639 return dkdlambda * std::pow(Ceff, tracer_reaction_order - 1.);
640 }
641 else if (id == field::tracer_concentration)
642 {
643 const double k =
644 tracer_reaction_constant_bulk +
645 delta_reaction_constant *
646 std::exp(-levelset_val * levelset_val / squared_thickness);
647 const double dCe =
648 calculate_d_effective_concentration_dC(concentration_val,
649 tracer_reaction_threshold);
650 return k * (tracer_reaction_order - 1.) *
651 std::pow(Ceff, tracer_reaction_order - 2.) * dCe;
652 }
653 else
654 return 0.0;
655 }

672 std::vector<double> &jacobian_vector) override
673 {
674 Assert(field_vectors.find(field::levelset) != field_vectors.end(),
675 PhysicialPropertyModelFieldUndefined(
676 "GaussianLevelsetTracerReactionPrefactor", "levelset"));
677 Assert(field_vectors.find(field::tracer_concentration) !=
678 field_vectors.end(),
679 PhysicialPropertyModelFieldUndefined(
680 "GaussianLevelsetTracerReactionPrefactor",
681 "tracer_concentration"));
682 const std::vector<double> &levelset_vec = field_vectors.at(field::levelset);
683 const std::vector<double> &concentration_vec =
684 field_vectors.at(field::tracer_concentration);
685 const unsigned int n_values = levelset_vec.size();
686 if (id == field::levelset)
687 {
688 for (unsigned int i = 0; i < n_values; ++i)
689 {
690 const double exponential =
691 std::exp(-levelset_vec[i] * levelset_vec[i] / squared_thickness);
692 const double dkdlambda =
693 delta_reaction_constant * exponential *
694 (-2.0 * levelset_vec[i] / squared_thickness);
695 const double Ceff =
696 calculate_effective_concentration(concentration_vec[i],
697 tracer_reaction_threshold);
698 jacobian_vector[i] =
699 dkdlambda * std::pow(Ceff, tracer_reaction_order - 1.);
700 }
701 }
702 else if (id == field::tracer_concentration)
703 {
704 for (unsigned int i = 0; i < n_values; ++i)
705 {
706 const double k = tracer_reaction_constant_bulk +
707 delta_reaction_constant *
708 std::exp(-levelset_vec[i] * levelset_vec[i] /
709 squared_thickness);
710 const double Ceff =
711 calculate_effective_concentration(concentration_vec[i],
712 tracer_reaction_threshold);
713 const double dCe =
714 calculate_d_effective_concentration_dC(concentration_vec[i],
715 tracer_reaction_threshold);
716 jacobian_vector[i] = k * (tracer_reaction_order - 1.) *
717 std::pow(Ceff, tracer_reaction_order - 2.) *
718 dCe;
719 }
720 }
721 else
722 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0.0);
723 }
