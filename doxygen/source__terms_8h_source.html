17/**
18 * The source term class provides an interface for all common
19 * Element required for the introduction of a source term in equations
20 * All equation-specific source term should derive
21 * from the base class but also call it's declare_parameters and
22 * parse_parameters routine. This allows specialized classes to focus on their
23 * specificity and forget about other non-specific elements that are generic to
24 * the calculation of analytical solutions
25 **/
26
27namespace SourceTerms
28{
29 template <int dim>
30 class SourceTerm
31 {
32 public:
33 SourceTerm()
34 {
35 navier_stokes_source =
36 std::make_shared<Functions::ParsedFunction<dim>>(dim + 1);
37 heat_transfer_source =
38 std::make_shared<Functions::ParsedFunction<dim>>(1);
39 tracer_source = std::make_shared<Functions::ParsedFunction<dim>>(1);
40 cahn_hilliard_source =
41 std::make_shared<Functions::ParsedFunction<dim>>(2);
42 }
43
44 virtual void
45 declare_parameters(ParameterHandler &prm);
46 virtual void
47 parse_parameters(ParameterHandler &prm);
48
49 /// Enable the Navier-Stokes source term
50 bool enable;
51
52 /// Velocity-pressure components
53 std::shared_ptr<Functions::ParsedFunction<dim>> navier_stokes_source;
54
55 /// Heat transfer source
56 std::shared_ptr<Functions::ParsedFunction<dim>> heat_transfer_source;
57
58 /// Tracer source
59 std::shared_ptr<Functions::ParsedFunction<dim>> tracer_source;
60
61 /// Cahn-Hilliard source
62 std::shared_ptr<Functions::ParsedFunction<dim>> cahn_hilliard_source;
63 };

65 template <int dim>
66 void
67 SourceTerm<dim>::declare_parameters(ParameterHandler &prm)
68 {
69 prm.enter_subsection("source term");
70
71 prm.enter_subsection("fluid dynamics");
72 prm.declare_entry("enable",
73 "true",
74 Patterns::Bool(),
75 "Enable the usage of a source term for the fluid solver");
76 navier_stokes_source->declare_parameters(prm, dim + 1);
77 prm.leave_subsection();
78
79 prm.enter_subsection("heat transfer");
80 heat_transfer_source->declare_parameters(prm);
81 prm.leave_subsection();
82
83 prm.enter_subsection("tracer");
84 tracer_source->declare_parameters(prm);
85 prm.leave_subsection();
86
87 prm.enter_subsection("cahn hilliard");
88 cahn_hilliard_source->declare_parameters(prm, 2);
89 prm.leave_subsection();
90
91 prm.leave_subsection();
92 }

94 template <int dim>
95 void
96 SourceTerm<dim>::parse_parameters(ParameterHandler &prm)
97 {
98 prm.enter_subsection("source term");
99
100 prm.enter_subsection("fluid dynamics");

102 navier_stokes_source->parse_parameters(prm);
103 prm.leave_subsection();
104
105 prm.enter_subsection("heat transfer");
106 heat_transfer_source->parse_parameters(prm);
107 prm.leave_subsection();

110 tracer_source->parse_parameters(prm);

112
113 prm.enter_subsection("cahn hilliard");
114 cahn_hilliard_source->parse_parameters(prm);
115 prm.leave_subsection();
116
117 prm.leave_subsection();
118 }
