146/**
147 * @brief Phase change specific heat model.
148 *
149 * The phase change specific heat \f$(c^{*}_\mathrm{p})\f$ model takes into
150 * account the phase change of a material by considering the latent heat into
151 * the specific heat over a phase change temperature interval determined by
152 * \f$\left[T_\mathrm{s},T_\mathrm{l}\right]\f$ where \f$T_\mathrm{s}\f$ is the
153 * solidus temperature and \f$T_\mathrm{l}\f$ is the liquidus temperature.
154 *
155 * The specific heat is evaluated as:
156 * \f$ c^{*}_\mathrm{p}(T) =
157 * \begin{cases}
158 * C_\mathrm{p,s} & \text{if} \; T<T_\mathrm{s}\\
159 * \frac{C_\mathrm{p,s}+C_\mathrm{p,l}}{2}+\frac{h_\mathrm{l}}
160 * {T_\mathrm{l}-T_\mathrm{s}} & \text{if} \;
161 * T\in[T_\mathrm{s},T_\mathrm{l}]\\ C_\mathrm{p,l} & \text{if} \;
162 * T>T_\mathrm{l} \end{cases} \f$
163 *
164 * where \f$C_\mathrm{p,s}\f$ and \f$C_\mathrm{p,l}\f$ are the solid and liquid
165 * specific heat, respectively, and \f$h_\mathrm{l}\f$ is the latent enthalpy
166 * (enthalpy related to the phase change).
167 */
168class PhaseChangeSpecificHeat : public SpecificHeatModel
169{
170public:
171 /**
172 * @brief Constructor of the phase change specific heat model.
173 *
174 * @param[in] p_phase_change_parameters Set of parameters of the phase change
175 * model.
176 */
177 PhaseChangeSpecificHeat(
178 const Parameters::PhaseChange p_phase_change_parameters)
179 : param(p_phase_change_parameters)
180 {
181 this->model_depends_on[field::temperature] = true;
182 this->model_depends_on[field::temperature_p1] = true;
183 this->model_depends_on[field::temperature_p2] = true;
184 }
185
186 /**
187 * @brief Compute the specific heat of the isothermal ideal gas.
188 *
189 * @param[in] field_values Values of the various fields on which the property
190 * may depend. In this case, the specific heat depends on temperature.
191 * The map stores a single value per field.
192 *
193 * @return Value of the specific heat computed with the @p field_values.
194 */
195 double
196 value(const std::map<field, double> &field_values) override
197 {
198 Assert(field_values.find(field::temperature) != field_values.end(),
199 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
200 "temperature"));
201 Assert(field_values.find(field::temperature_p1) != field_values.end(),
202 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
203 "temperature_p1"));
204 Assert(field_values.find(field::temperature_p2) != field_values.end(),
205 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
206 "temperature_p2"));
207 double temperature = field_values.at(field::temperature);
208 double temperature_p1 = field_values.at(field::temperature_p1);
209 double temperature_p2 = field_values.at(field::temperature_p2);
210
211
212 // Gather information required from the simulation control to have the time
213 // history
214 std::shared_ptr<SimulationControl> simulation_control =
215 get_simulation_control();
216
217 // Time stepping information
218 auto method = simulation_control->get_assembly_method();
219 // Vector for the BDF coefficients
220 const Vector<double> &bdf_coefs =
221 simulation_control->get_bdf_coefficients();
222
223 switch (method)
224 {
225 case Parameters::SimulationControl::TimeSteppingMethod::bdf1:
226 {
227 // Clipping to ensure a minimal temperature difference
228 if (temperature > temperature_p1)
229 temperature = std::max(temperature, temperature_p1 + 1e-6);
230 else
231 temperature_p1 = std::max(temperature_p1, temperature + 1e-6);
232
233 const double enthalpy_current = enthalpy(temperature);
234 const double enthalpy_p1 = enthalpy(temperature_p1);
235 const double dH =
236 bdf_coefs[0] * enthalpy_current + bdf_coefs[1] * enthalpy_p1;
237 const double dT =
238 bdf_coefs[0] * temperature + bdf_coefs[1] * temperature_p1;
239 return dH / dT;
240 }
241
242 case Parameters::SimulationControl::TimeSteppingMethod::bdf2:
243 {
244 // Clipping to ensure a minimal temperature difference
245 if (temperature > temperature_p2)
246 temperature = std::max(temperature, temperature_p2 + 1e-6);
247 else
248 temperature_p2 = std::max(temperature_p2, temperature + 1e-6);
249
250 const double enthalpy_current = enthalpy(temperature);
251 const double enthalpy_p1 = enthalpy(temperature_p1);
252 const double enthalpy_p2 = enthalpy(temperature_p2);
253 const double dH = bdf_coefs[0] * enthalpy_current +
254 bdf_coefs[1] * enthalpy_p1 +
255 bdf_coefs[2] * enthalpy_p2;
256 const double dT = bdf_coefs[0] * temperature +
257 bdf_coefs[1] * temperature_p1 +
258 bdf_coefs[2] * temperature_p2;
259 return dH / dT;
260 }
261
262 default:
263 throw(std::runtime_error(
264 "Other time integration schemes are not supported by the phase change model"));
265 }
266 }
267
268 /**
269 * @brief Compute a vector of specific heat values for an isothermal ideal gas.
270 *
271 * @param[in] field_vectors Vectors of the fields on which the specific heat
272 * may depend. In this case, the specific heat depends on temperature. The map
273 * stores a vector of values per field.
274 *
275 * @param[out] property_vector Vectors of computed specific heat values.
276 */
277 void
278 vector_value(const std::map<field, std::vector<double>> &field_vectors,
279 std::vector<double> &property_vector) override
280 {
281 Assert(field_vectors.find(field::temperature) != field_vectors.end(),
282 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
283 "temperature"));
284 Assert(field_vectors.find(field::temperature_p1) != field_vectors.end(),
285 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
286 "temperature_p1"));
287 Assert(field_vectors.find(field::temperature_p2) != field_vectors.end(),
288 PhysicialPropertyModelFieldUndefined("PhaseChangeSpecificHeat",
289 "temperature_p2"));
290 const std::vector<double> &temperature_vec =
291 field_vectors.at(field::temperature);
292 const std::vector<double> &p1_temperature_vec =
293 field_vectors.at(field::temperature_p1);
294 const std::vector<double> &p2_temperature_vec =
295 field_vectors.at(field::temperature_p2);
296
297 const unsigned int n_values = temperature_vec.size();
298
299 Assert(n_values == p2_temperature_vec.size(),
300 SizeOfFields(n_values, p2_temperature_vec.size()));
301 Assert(n_values == p1_temperature_vec.size(),
302 SizeOfFields(n_values, p1_temperature_vec.size()));
303 Assert(n_values == property_vector.size(),
304 SizeOfFields(n_values, property_vector.size()));
305
306 // Gather information required from the simulation control to have the time
307 // history
308 std::shared_ptr<SimulationControl> simulation_control =
309 get_simulation_control();
310 std::vector<double> time_steps_vector =
311 simulation_control->get_time_steps_vector();
312
313 auto method = simulation_control->get_assembly_method();
314 const Vector<double> &bdf_coefs =
315 simulation_control->get_bdf_coefficients();
316
317 for (unsigned int i = 0; i < n_values; ++i)
318 {
319 double temperature = temperature_vec[i];
320 double temperature_p1 = p1_temperature_vec[i];
321 double temperature_p2 = p2_temperature_vec[i];
322
323 switch (method)
324 {
325 case Parameters::SimulationControl::TimeSteppingMethod::bdf1:
326 {
327 // Clipping to ensure a minimal temperature difference
328 if (temperature > temperature_p1)
329 temperature = std::max(temperature, temperature_p1 + 1e-6);
330 else
331 temperature_p1 = std::max(temperature_p1, temperature + 1e-6);
332
333 const double enthalpy_current = enthalpy(temperature);
334 const double enthalpy_p1 = enthalpy(temperature_p1);
335 const double dH =
336 bdf_coefs[0] * enthalpy_current + bdf_coefs[1] * enthalpy_p1;
337 const double dT =
338 bdf_coefs[0] * temperature + bdf_coefs[1] * temperature_p1;
339
340 // The division does not need a tolerance to avoid a division by
341 // 0 since the temperature is clipped
342 property_vector[i] = dH / dT;
343
344 break;
345 }
346
347 case Parameters::SimulationControl::TimeSteppingMethod::bdf2:
348 {
349 // Clipping to ensure a minimal temperature difference
350 if (temperature > temperature_p2)
351 temperature = std::max(temperature, temperature_p2 + 1e-6);
352 else
353 temperature_p2 = std::max(temperature_p2, temperature + 1e-6);
354
355 const double enthalpy_current = enthalpy(temperature);
356 const double enthalpy_p1 = enthalpy(temperature_p1);
357 const double enthalpy_p2 = enthalpy(temperature_p2);
358 const double dH = bdf_coefs[0] * enthalpy_current +
359 bdf_coefs[1] * enthalpy_p1 +
360 bdf_coefs[2] * enthalpy_p2;
361 const double dT = bdf_coefs[0] * temperature +
362 bdf_coefs[1] * temperature_p1 +
363 bdf_coefs[2] * temperature_p2;
364 property_vector[i] = dH / (dT + 1e-8);
365 break;
366 }
367
368 default:
369 throw(std::runtime_error(
370 "Other time integration schemes are not supported by the phase change model"));
371 }
372 }
373 }

426 /**
427 * @brief Compute the enthalpy of a phase change material for a temperature
428 * \f$T\f$.
429 *
430 * The enthalpy \f$(H)\f$ is defined:
431 *
432 * - In pure liquid \f$(T>T_\mathrm{l})\f$ as:\n
433 * \f$H(T) = c_\mathrm{p,s} T_\mathrm{s} + 0.5(c_\mathrm{p,s}+c_\mathrm{p,l})
434 * (T_\mathrm{l}-T_\mathrm{s}) + h_\mathrm{l} + c_\mathrm{p,l}
435 * (T-T_\mathrm{l})\f$\n\n
436 *
437 * - In liquid-solid mixture \f$(T_\mathrm{s} \leq T \leq T_\mathrm{l})\f$
438 * as:\n
439 *\f$H(T) = c_\mathrm{p,s} T_\mathrm{s} + 0.5(c_\mathrm{p,s}+c_\mathrm{p,l})
440 * (T_\mathrm{l}-T_\mathrm{s}) + \alpha_\mathrm{l} h_\mathrm{l}\f$\n
441 * where \f$\alpha_\mathrm{l} = \frac{T-T_\mathrm{s}}{T_\mathrm{l}-
442 * T_\mathrm{s}}\f$ is the liquid fraction.\n\n
443 *
444 * - In pure solid \f$(T<T_\mathrm{s})\f$ as:\n
445 * \f$H(T) = c_\mathrm{p,s} T\f$
446 *
447 * @param[in] T temperature at which the enthalpy is computed.
448 *
449 * @return Value of the enthalpy evaluated at \f$T\f$.
450 */
451
452 inline double
453 enthalpy(const double T)
454 {
455 if (T > param.T_liquidus)
456 return (param.cp_s * param.T_solidus +
457 0.5 * (param.cp_l + param.cp_s) *
458 (param.T_liquidus - param.T_solidus) +
459 param.latent_enthalpy + (T - param.T_liquidus) * param.cp_l);
460 else if (T > param.T_solidus)
461 {
462 liquid_fraction = calculate_liquid_fraction(T, param);
463 return (param.cp_s * param.T_solidus +
464 0.5 * (param.cp_l + param.cp_s) * (T - param.T_solidus) +
465 param.latent_enthalpy * liquid_fraction);
466 }
467 else
468 return param.cp_s * T;
469 }
470
471private:
472 const Parameters::PhaseChange param;
473 double liquid_fraction;
474};
