6template <int dim>
7void
8VOFCurvatureProjection<dim>::assemble_system_matrix_and_rhs()
9{
10 // Reinitialize system matrix and right-hand side (rhs)
11 this->system_matrix = 0;
12 this->system_rhs = 0;
13
14 // Get phase gradient L2 projection DoFHandler
15 const DoFHandler<dim> &dof_handler_phase_fraction_gradient =
16 this->subequations_interface.get_dof_handler(
17 VOFSubequationsID::phase_gradient_projection);
18
19 // Initialize FEValues for curvature and phase fraction gradient projection
20 FEValues<dim> fe_values_curvature_projection(*this->mapping,
21 *this->fe,
22 *this->cell_quadrature,
23 update_values |
24 update_gradients |
25 update_JxW_values);
26 FEValues<dim> fe_values_phase_gradient_projection(
27 *this->mapping,
28 dof_handler_phase_fraction_gradient.get_fe(),
29 *this->cell_quadrature,
30 update_values);
31
32 // Initialize size of arrays
33 const unsigned int n_q_points =
34 fe_values_curvature_projection.get_quadrature().size();
35 const unsigned int n_dofs_per_cell =
36 fe_values_curvature_projection.get_fe().n_dofs_per_cell();
37
38 // Initialize local matrix and rhs
39 FullMatrix<double> local_matrix(n_dofs_per_cell, n_dofs_per_cell);
40 Vector<double> local_rhs(n_dofs_per_cell);
41
42 // Initialize the local dof indices array
43 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
44
45 // Extractor for phase fraction gradient vector
46 FEValuesExtractors::Vector phase_fraction_gradients(0);
47
48 // Initialize projected phase fraction gradient solution array
49 std::vector<Tensor<1, dim>> present_phase_gradient_projection_values(
50 n_q_points);
51
52 // Initialize shape function arrays
53 std::vector<double> phi(n_dofs_per_cell);
54 std::vector<Tensor<1, dim>> grad_phi(n_dofs_per_cell);
55
56 // Get the diffusion factor
57 const double diffusion_factor =
58 this->simulation_parameters.multiphysics.vof_parameters
59 .surface_tension_force.curvature_diffusion_factor;
60
61 // Loop over phase gradient projection cells
62 for (const auto &cell : this->dof_handler->active_cell_iterators())
63 {
64 if (cell->is_locally_owned())
65 {
66 // Reinitialize local linear system
67 local_matrix = 0;
68 local_rhs = 0;
69
70 // Get phase gradient projection cell iterator
71 typename DoFHandler<dim>::active_cell_iterator
72 phase_gradient_projection_cell(
73 &(*this->triangulation),
74 cell->level(),
75 cell->index(),
76 &dof_handler_phase_fraction_gradient);
77
78 // Reinitialize FEValues with corresponding cells
79 fe_values_curvature_projection.reinit(cell);
80 fe_values_phase_gradient_projection.reinit(
81 phase_gradient_projection_cell);
82
83 // Get vector of Jacobi determinant times the quadrature weights
84 std::vector<double> JxW_vec =
85 fe_values_curvature_projection.get_JxW_values();
86
87 // Compute cell size
88 auto &fe_curvature_projection =
89 fe_values_curvature_projection.get_fe();
90 const double h =
91 compute_cell_diameter<dim>(compute_cell_measure_with_JxW(JxW_vec),
92 fe_curvature_projection.degree);
93
94 // Get projected fraction gradient values
95 fe_values_phase_gradient_projection[phase_fraction_gradients]
96 .get_function_values(
97 this->subequations_interface.get_solution(
98 VOFSubequationsID::phase_gradient_projection),
99 present_phase_gradient_projection_values);
100
101 // Set tolerance to avoid division by zero
102 const double tolerance = 1e-12;
103
104 // Loop over quadrature points
105 for (unsigned int q = 0; q < n_q_points; ++q)
106 {
107 // Shape functions
108 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
109 {
110 phi[k] = fe_values_curvature_projection.shape_value(k, q);
111 grad_phi[k] = fe_values_curvature_projection.shape_grad(k, q);
112 }
113
114 // Compute projected phase gradient norm
115 const Tensor<1, dim> projected_vof_phase_gradient =
116 present_phase_gradient_projection_values[q];
117 const double projected_vof_phase_gradient_norm =
118 projected_vof_phase_gradient.norm() + tolerance;
119
120 // Local linear system assembly
121 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
122 {
123 // Assemble local matrix
124 for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
125 {
126 local_matrix(i, j) +=
127 (phi[i] * phi[j] +
128 h * h * diffusion_factor *
129 scalar_product(grad_phi[i], grad_phi[j])) *
130 JxW_vec[q];
131 }
132 // Assemble local right-hand side (rhs)
133 local_rhs(i) += grad_phi[i] *
134 (projected_vof_phase_gradient /
135 projected_vof_phase_gradient_norm) *
136 JxW_vec[q];
137 }
138 }
139
140 // Distribute the local contributions to the global system
141 cell->get_dof_indices(local_dof_indices);
142 this->constraints.distribute_local_to_global(local_matrix,
143 local_rhs,
144 local_dof_indices,
145 this->system_matrix,
146 this->system_rhs);
147 }
148 }
149 this->system_matrix.compress(VectorOperation::add);
150 this->system_rhs.compress(VectorOperation::add);
151}
