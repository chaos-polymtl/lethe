27/**
28 * @brief Manage the numerous of contact detection search operations in the DEM.
29 *
30 * This class mostly calls proper functions in regards the type of contacts for
31 * the contact detection and updates of data containers.
32 *
33 * @tparam dim An integer that denotes the number of spatial dimensions.
34 * @tparam PropertiesIndex Index of the properties used within the ParticleHandler.
35 */
36template <int dim, typename PropertiesIndex>
37class DEMContactManager
38{
39public:
40 /**
41 * @brief Execute functions to find the lists of cell neighbors of the active
42 * cells in the triangulation.
43 *
44 * For contacts of particles, the neighbor search excludes the repetitions of
45 * neighbors: B is neighbor of A but A will not be neighbor of B. For contacts
46 * of particles with floating meshes, the neighbor search includes all
47 * neighbors. These floating mesh contacts searches need all the particles
48 * located in all the neighbor cells of the main cell to detect
49 * collisions with the floating mesh.
50 *
51 * @param[in] triangulation The triangulation to access the information of the
52 * cells.
53 * @param[in] periodic_boundaries_cells_information Information of periodic
54 * cells used if periodic boundaries are enabled (next parameter).
55 */
56 void
57 execute_cell_neighbors_search(
58 const parallel::distributed::Triangulation<dim> &triangulation,
59 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
60 periodic_boundaries_cells_information);
61
62 /**
63 * @brief Execute functions to clear and update the neighbor lists of all the
64 * active cells in the triangulation.
65 *
66 * @param[in] triangulation The triangulation to access the information of the
67 * cells.
68 * @param[in] periodic_boundaries_cells_information Information of periodic
69 * cells used if periodic boundaries are enabled (next parameter).
70 */
71 void
72 update_cell_neighbors(
73 const parallel::distributed::Triangulation<dim> &triangulation,
74 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
75 periodic_boundaries_cells_information)
76 {
77 cells_local_neighbor_list.clear();
78 cells_ghost_neighbor_list.clear();
79 cells_local_periodic_neighbor_list.clear();
80 cells_ghost_periodic_neighbor_list.clear();
81 cells_ghost_local_periodic_neighbor_list.clear();
82 total_neighbor_list.clear();
83
84 execute_cell_neighbors_search(triangulation,
85 periodic_boundaries_cells_information);
86 }
87
88 /**
89 * @brief Execute functions to update the contact data containers with the new
90 * contact pairs.
91 *
92 * Call proper functions to remove contact repetitions and to add new contact
93 * pairs to the contact containers when particles are exchanged between
94 * processors after the fine search.
95 * Contact pairs are local particle-particle, ghost particle-particle,
96 * particle-wall, particle-floating wall contacts and particle-floating mesh
97 * contacts.
98 */
99 void
100 update_contacts();
101
102 /**
103 * @breif Execute functions to update the particle iterators in local-local
104 * contact containers.
105 *
106 * This is essential since sort_particles_into_subdomains_and_cells() and
107 * exchange_ghost_particles() functions change the particle iterators
108 * everytime they are called.
109 *
110 * @param[in] particle_handler Storage of particles and their accessor
111 * functions.
112 */
113 void
114 update_local_particles_in_cells(
115 const Particles::ParticleHandler<dim> &particle_handler);
116
117 /**
118 * @brief Execute the particle-particle broad searches with adaptive sparse
119 * contacts.
120 *
121 * Calls proper functions to find the candidates of local and ghost
122 * particle-particle contact pairs and the periodic particle-particle contacts
123 * if required. These contact pairs will be used in the fine search step to
124 * investigate if they are in contact.
125 * It checks if the adaptive sparse contacts is enabled and use proper
126 * functions.
127 *
128 * @param[in,out] particle_handler Storage of particles and their accessor
129 * functions.
130 * @param[in] sparse_particle_contact_object Allow to check the mobility
131 * status of cells.
132 */
133 void
134 execute_particle_particle_broad_search(
135 dealii::Particles::ParticleHandler<dim> &particle_handler,
136 const AdaptiveSparseContacts<dim, PropertiesIndex>
137 &sparse_particle_contact_object);
138
139 /**
140 * @brief Execute the particle-wall broad searches with adaptive sparse
141 * contacts.
142 *
143 * Calls proper functions to find the candidates of particle-wall contact
144 * pairs and the particle-floating wall or particle-floating mesh if required.
145 * These contact pairs will be used in the fine search step to investigate if
146 * they are in contact.
147 * It checks if the adaptive sparse contacts is enabled and use proper
148 * functions.
149 *
150 * @param[in] particle_handler Storage of particles and their accessor
151 * functions.
152 * @param[in] boundary_cells_object Information of the boundary cells and
153 * faces.
154 * @param[in] solid_surfaces_mesh_info Mapping of solid surfaces meshes.
155 * @param[in] floating_wall Properties of the floating walls.
156 * @param[in] simulation_time Current simulation time.
157 * @param[in] sparse_particle_contact_object Allow to check the mobility
158 * status of cells.
159 */
160 void
161 execute_particle_wall_broad_search(
162 const Particles::ParticleHandler<dim> &particle_handler,
163 BoundaryCellsInformation<dim> &boundary_cell_object,
164 const typename dem_data_structures<dim>::solid_surfaces_mesh_information
165 solid_surfaces_mesh_info,
166 const Parameters::Lagrangian::FloatingWalls<dim> &floating_walls,
167 const double simulation_time,
168 const AdaptiveSparseContacts<dim, PropertiesIndex>
169 &sparse_particle_contact_object);
170
171 /**
172 * @brief Execute the particle-particles fine searches.
173 *
174 * Executes functions that update the particle contacts pairs containers and
175 * compute the contact information of the collision pairs.
176 *
177 * @param[in] neighborhood_threshold Threshold value of contact detection.
178 */
179 void
180 execute_particle_particle_fine_search(const double neighborhood_threshold);
181
182 /**
183 * @brief Execute the particle-wall fine searches.
184 *
185 * Executes functions that update the particle-wall contacts pairs containers
186 * and compute the contact information of the collision particle-wall.
187 *
188 * @param[in] floating_wall Properties of the floating walls.
189 * @param[in] simulation_time Current simulation time.
190 * @param[in] neighborhood_threshold Threshold value of contact detection.
191 */
192 void
193 execute_particle_wall_fine_search(
194 const Parameters::Lagrangian::FloatingWalls<dim> &floating_walls,
195 const double simulation_time,
196 const double neighborhood_threshold);
197
198 /**
199 * @brief Set the constant periodic offset for the periodic boundaries. If
