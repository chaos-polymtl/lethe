23template <int dim, typename PropertiesIndex>
24void
25ExplicitEulerIntegrator<dim, PropertiesIndex>::integrate(
26 Particles::ParticleHandler<dim> &particle_handler,
27 const Tensor<1, 3> &g,
28 const double dt,
29 std::vector<Tensor<1, 3>> &torque,
30 std::vector<Tensor<1, 3>> &force,
31 const std::vector<double> &MOI)
32{
33 for (auto particle = particle_handler.begin();
34 particle != particle_handler.end();
35 ++particle)
36 {
37 // Get the total array view to the particle properties and location once
38 // to improve efficiency
39 types::particle_index particle_id = particle->get_local_index();
40
41 auto particle_properties = particle->get_properties();
42 Tensor<1, 3> &particle_torque = torque[particle_id];
43 Tensor<1, 3> &particle_force = force[particle_id];
44 Point<3> particle_position;
45 double mass_inverse = 1 / particle_properties[PropertiesIndex::mass];
46 double MOI_inverse = 1 / MOI[particle_id];
47
48
49 if constexpr (dim == 3)
50 particle_position = particle->get_location();
51
52 if constexpr (dim == 2)
53 particle_position = point_nd_to_3d(particle->get_location());
54
55 Tensor<1, 3> acceleration;
56 for (int d = 0; d < 3; ++d)
57 {
58 acceleration[d] = g[d] + (particle_force[d]) * mass_inverse;
59
60 // Velocity integration:
61 particle_properties[PropertiesIndex::v_x + d] += dt * acceleration[d];
62
63 // Position integration
64 particle_position[d] +=
65 dt * particle_properties[PropertiesIndex::v_x + d];
66
67 particle_properties[PropertiesIndex::omega_x + d] +=
68 dt * (particle_torque[d] * MOI_inverse);
69 }
70
71 // Reinitialize force
72 particle_force = 0;
73
74 // Reinitialize torque
75 particle_torque = 0;
76
77 if constexpr (dim == 3)
78 particle->set_location(particle_position);
79
80 if constexpr (dim == 2)
81 {
82 Point<2> position_2d;
83 position_2d[0] = particle_position[0];
84 position_2d[1] = particle_position[1];
85 particle->set_location(position_2d);
86 }
87 }
88}

90// Explicit Euler not implemented for adaptive sparse contacts
91template <int dim, typename PropertiesIndex>
92void
93ExplicitEulerIntegrator<dim, PropertiesIndex>::integrate(
94 Particles::ParticleHandler<dim> &particle_handler,
95 const Tensor<1, 3> &g,
96 const double dt,
97 std::vector<Tensor<1, 3>> &torque,
98 std::vector<Tensor<1, 3>> &force,
99 const std::vector<double> &MOI,
100 const parallel::distributed::Triangulation<dim> & /* triangulation */,
101 AdaptiveSparseContacts<dim, PropertiesIndex> & /* sparse_contacts_object */)
102{
103 auto *action_manager = DEMActionManager::get_action_manager();
104
105 bool use_default_function =
106 !action_manager->check_sparse_contacts_enabled() ||
107 action_manager->check_mobility_status_reset();
108
109 if (use_default_function)
110 {
111 integrate(particle_handler, g, dt, torque, force, MOI);
112 return;
113 }
114
115 throw std::runtime_error(
116 "Adaptive sparse contacts are not supported with explicit Euler integrator, use Velocity Verlet integrator.");
117}
