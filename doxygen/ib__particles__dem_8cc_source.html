38template <int dim>
39void
40IBParticlesDEM<dim>::update_contact_candidates()
41{
42 particles_contact_candidates.resize(dem_particles.size());
43
44 double radius_factor = parameters->contact_search_radius_factor;
45
46 for (auto &particle_one : dem_particles)
47 {
48 const Point<dim> particle_one_location = particle_one.position;
49 for (auto &particle_two : dem_particles)
50 {
51 if (particle_one.particle_id < particle_two.particle_id)
52 {
53 const Point<dim> particle_two_location = particle_two.position;
54 double distance =
55 (particle_one_location - particle_two_location).norm();
56 if (typeid(*particle_one.shape) == typeid(Sphere<dim>) &&
57 typeid(*particle_two.shape) == typeid(Sphere<dim>))
58 {
59 distance =
60 (particle_one_location - particle_two_location).norm();
61 }
62 else if (typeid(*particle_one.shape) == typeid(Sphere<dim>) &&
63 typeid(*particle_two.shape) != typeid(Sphere<dim>))
64 {
65 distance = particle_two.shape->value(particle_one_location);
66 }
67 else if (typeid(*particle_one.shape) != typeid(Sphere<dim>) &&
68 typeid(*particle_two.shape) == typeid(Sphere<dim>))
69 {
70 distance = particle_one.shape->value(particle_two_location);
71 }
72 else
73 {
74 distance = ((particle_one.shape->get_rotation_matrix() *
75 point_nd_to_3d(
76 particle_one.shape->bounding_box_center) +
77 point_nd_to_3d(particle_one.position)) -
78 (particle_two.shape->get_rotation_matrix() *
79 point_nd_to_3d(
80 particle_two.shape->bounding_box_center) +
81 point_nd_to_3d(particle_two.position)))
82 .norm();
83 }
84
85 if (distance <
86 (particle_one.shape->bounding_box_half_length.norm() +
87 particle_two.shape->bounding_box_half_length.norm()) *
88 radius_factor)
89 {
90 (particles_contact_candidates[particle_one.particle_id])
91 .insert(particle_two.particle_id);
92 }
93 }
94 }
95 }

100template <int dim>
101void
102IBParticlesDEM<dim>::calculate_pp_contact_force(
103 const double dt_dem,
104 std::vector<Tensor<1, 3>> &contact_force,
105 std::vector<Tensor<1, 3>> &contact_torque)
106{
107 using dealii::numbers::PI;
108
109 for (auto &particle_one : dem_particles)
110 {
111 std::set<unsigned int>::iterator particle_contact_candidates_id;
112 for (particle_contact_candidates_id =
113 particles_contact_candidates[particle_one.particle_id].begin();
114 particle_contact_candidates_id !=
115 particles_contact_candidates[particle_one.particle_id].end();
116 ++particle_contact_candidates_id)
117 {
118 const unsigned int particle_contact_id =
119 *particle_contact_candidates_id;
120
121 auto &particle_two = dem_particles[particle_contact_id];
122 if (particle_one.particle_id < particle_two.particle_id)
123 {
124 ContactInfo contact_info;
125 // Check if there is already information on the contact of
126 // these to particles. If not initialize it in the contact map
127 // with 0 values.
128 try
129 {
130 contact_info = pp_contact_map[particle_one.particle_id]
131 [particle_two.particle_id];
132 }
133 catch (...)
134 {
135 for (int d = 0; d < dim; ++d)
136 {
137 contact_info.tangential_displacement[d] = 0.;
138 contact_info.rolling_resistance_spring_torque[d] = 0.;
139 }
140 pp_contact_map[particle_one.particle_id]
141 [particle_two.particle_id] = contact_info;
142 }
143
144 // Calculation of normal overlap
145 double normal_overlap;
146 Point<dim> contact_point;
147 Tensor<1, dim> normal;
148 std::vector<Point<dim>> contact_points_candidate;
149
150 auto iterator =
151 previous_wall_contact_point[particle_one.particle_id].find(
152 particle_two.particle_id);
153 if (iterator !=
154 previous_wall_contact_point[particle_one.particle_id].end())
155 {
156 contact_points_candidate.push_back(iterator->second);
157 }
158 else
159 {
160 contact_points_candidate.push_back(
161 particle_one.position +
162 (particle_two.position - particle_one.position) *
163 (particle_one.radius /
164 (particle_one.radius + particle_two.radius)));
165 }
166 // The normal vector goes from particle one to particle two.
167 if (typeid(*particle_one.shape) == typeid(Sphere<dim>) &&
168 typeid(*particle_two.shape) == typeid(Sphere<dim>))
169 {
170 auto contact_state = particle_one.shape->distance_to_shape(
171 *particle_two.shape, contact_points_candidate);
172 normal_overlap = -std::get<double>(contact_state);
173 contact_point = std::get<Point<dim>>(contact_state);
174 normal = std::get<Tensor<1, dim>>(contact_state);
175 }
176 else if (typeid(*particle_one.shape) == typeid(Sphere<dim>) &&
177 typeid(*particle_two.shape) != typeid(Sphere<dim>))
178 {
179 auto contact_state = particle_one.shape->distance_to_shape(
180 *particle_two.shape, contact_points_candidate);
181 normal_overlap = -std::get<double>(contact_state);
182 contact_point = std::get<Point<dim>>(contact_state);
183 normal = std::get<Tensor<1, dim>>(contact_state);
184 }
185 else if (typeid(*particle_one.shape) != typeid(Sphere<dim>) &&
186 typeid(*particle_two.shape) == typeid(Sphere<dim>))
187 {
188 auto contact_state = particle_two.shape->distance_to_shape(
189 *particle_one.shape, contact_points_candidate);
190 normal_overlap = -std::get<double>(contact_state);
191 contact_point = std::get<Point<dim>>(contact_state);
192 normal = -std::get<Tensor<1, dim>>(
193 contact_state); // Flip the normal since it returns the
194 // normal using the second particle;
195 }
196 else
197 {
198 auto contact_state = particle_one.shape->distance_to_shape(
199 *particle_two.shape, contact_points_candidate);
200 normal_overlap = -std::get<double>(contact_state);
201 contact_point = std::get<Point<dim>>(contact_state);
202 normal = std::get<Tensor<1, dim>>(contact_state);
203 }
204 previous_wall_contact_point[particle_one.particle_id]
205 [particle_two.particle_id] =
206 contact_point;
207 Point<3> contact_point_3d = point_nd_to_3d(contact_point);
208 Tensor<1, 3> contact_normal = tensor_nd_to_3d(normal);
209
210
211 if (normal_overlap > 0)
212 // This means that the adjacent particles are in contact
213 {
214 double contact_radius_particle_one;
215 double contact_radius_particle_two;
216 if (parameters->approximate_radius_for_contact)
217 {
218 contact_radius_particle_one = particle_one.radius;
219 contact_radius_particle_two = particle_two.radius;
220 }
221 else
222 {
223 contact_radius_particle_one =
224 particle_one.shape->local_curvature_radius(
225 contact_point);
226 contact_radius_particle_two =
227 particle_two.shape->local_curvature_radius(
228 contact_point);
229 }
230 Tensor<1, 3> normal_force;
231 Tensor<1, 3> tangential_force;
232 Tensor<1, 3> particle_one_tangential_torque;
233 Tensor<1, 3> particle_two_tangential_torque;
234 Tensor<1, 3> rolling_resistance_torque;
235 Tensor<1, 3> particle_one_velocity_3d =
236 tensor_nd_to_3d(particle_one.velocity);
237 Tensor<1, 3> particle_two_velocity_3d =
238 tensor_nd_to_3d(particle_two.velocity);
239
240 ObjectProperties particle_one_properties;
241 particle_one_properties.object_friction_coefficient =
242 particle_one.friction_coefficient;
243 particle_one_properties.object_mass = particle_one.mass;
244 particle_one_properties.object_poisson_ratio =
245 particle_one.poisson_ratio;
246 particle_one_properties.object_radius =
247 contact_radius_particle_one;
248 particle_one_properties.object_restitution_coefficient =
249 particle_one.restitution_coefficient;
250 particle_one_properties.object_rolling_friction_coefficient =
251 particle_one.rolling_friction_coefficient;
252 particle_one_properties.object_youngs_modulus =
253 particle_one.youngs_modulus;
254 ObjectProperties particle_two_properties;
255 particle_two_properties.object_friction_coefficient =
256 particle_two.friction_coefficient;
257 particle_two_properties.object_mass = particle_two.mass;
258 particle_two_properties.object_poisson_ratio =
259 particle_two.poisson_ratio;
260 particle_two_properties.object_radius =
261 contact_radius_particle_two;
262 particle_two_properties.object_restitution_coefficient =
263 particle_two.restitution_coefficient;
264 particle_two_properties.object_rolling_friction_coefficient =
265 particle_two.rolling_friction_coefficient;
266 particle_two_properties.object_youngs_modulus =
267 particle_two.youngs_modulus;
268
269 calculate_force_model(normal_overlap,
270 contact_info,
271 contact_point_3d,
272 contact_normal,
273 normal_force,
274 tangential_force,
275 rolling_resistance_torque,
276 particle_one.position,
277 particle_one_velocity_3d,
278 particle_one.omega,
279 particle_one_properties,
280 particle_two.position,
281 particle_two_velocity_3d,
282 particle_two.omega,
283 particle_two_properties,
284 dt_dem);
285
286
287 contact_force[particle_one.particle_id] -=
288 (normal_force + tangential_force);
289 contact_force[particle_two.particle_id] +=
290 (normal_force + tangential_force);
291
292 contact_torque[particle_one.particle_id] =
293 contact_torque[particle_one.particle_id] -
294 particle_one_tangential_torque + rolling_resistance_torque +
295 cross_product_3d((contact_point_3d -
296 point_nd_to_3d(particle_one.position)),
297 -(normal_force + tangential_force));
298 contact_torque[particle_two.particle_id] =
299 contact_torque[particle_two.particle_id] -
300 particle_two_tangential_torque - rolling_resistance_torque +
301 cross_product_3d((contact_point_3d -
302 point_nd_to_3d(particle_two.position)),
303 (normal_force + tangential_force));
304 ;
305 }
306
307 else
308 {
309 // if the adjacent pair is not in contact anymore
310 for (int d = 0; d < dim; ++d)
311 {
312 contact_info.tangential_displacement[d] = 0.;
313 contact_info.rolling_resistance_spring_torque[d] = 0.;
314 }
315 pp_contact_map[particle_one.particle_id].erase(
316 particle_two.particle_id);
317 }
318 }
319 }
320 }

323template <int dim>
324void
325IBParticlesDEM<dim>::calculate_pp_lubrication_force(
326 const double /*dt_dem*/,
327 const double h_max,
328 const double h_min,
329 const double mu,
330 std::vector<Tensor<1, 3>> &lubrication_force,
331 std::vector<Tensor<1, 3>> & /*lubrication_torque*/)
332{
333 using dealii::numbers::PI;
334
335 // loop over all particles to find pair of close particles
336 for (auto &particle_one : dem_particles)
337 {
338 for (auto particle_contact_candidates_id =
339 particles_contact_candidates[particle_one.particle_id].begin();
340 particle_contact_candidates_id !=
341 particles_contact_candidates[particle_one.particle_id].end();
342 ++particle_contact_candidates_id)
343 {
344 const auto &particle_contact_id = *particle_contact_candidates_id;
345 auto &particle_two = dem_particles[particle_contact_id];
346 if (particle_one.particle_id < particle_two.particle_id)
347 {
348 const Point<dim> particle_one_location = particle_one.position;
349 const Point<dim> particle_two_location = particle_two.position;
350 Tensor<1, 3> radial_vector;
351 double radial_velocity;
352 Tensor<1, 3> f_lub;
353
354 // Calculation of normal overlap
355 double gap =
356 particle_one_location.distance(particle_two_location) -
357 (particle_one.radius + particle_two.radius);
358 radial_vector =
359 tensor_nd_to_3d(particle_one.position - particle_two.position);
360 if (gap > 0 and gap < h_max)
361 // This means that the adjacent particles are very close but
362 // not in contact
363 {
364 // Limit the smallest gap calculated
365 if (gap < h_min)
366 {
367 gap = h_min;
368 }
369
370 // Evaluate the force
371 radial_velocity =
372 scalar_product(-radial_vector, particle_one.velocity) +
373 scalar_product(radial_vector, particle_two.velocity);
374 f_lub =
375 3 / 2 * PI * mu *
376 (particle_one.radius * 2 * particle_two.radius * 2 /
377 (particle_one.radius * 2 + particle_two.radius * 2)) *
378 (particle_one.radius * 2 * particle_two.radius * 2 /
379 (particle_one.radius * 2 + particle_two.radius * 2)) /
380 gap * radial_velocity * radial_vector /
381 radial_vector.norm() -
382 3 / 2 * PI * mu *
383 (particle_one.radius * 2 * particle_two.radius * 2 /
384 (particle_one.radius * 2 + particle_two.radius * 2)) *
385 (particle_one.radius * 2 * particle_two.radius * 2 /
386 (particle_one.radius * 2 + particle_two.radius * 2)) /
387 h_max * radial_velocity * radial_vector /
388 radial_vector.norm();
389 ;
390 }
391
392 lubrication_force[particle_one.particle_id] = f_lub;
393 lubrication_force[particle_two.particle_id] = -f_lub;
394 }
395 }
396 }

399template <int dim>
400void
401IBParticlesDEM<dim>::update_particles_boundary_contact(
402 std::vector<IBParticle<dim>> &particles,
403 const DoFHandler<dim> &dof_handler,
404 const Quadrature<dim - 1> &face_quadrature_formula,
405 const Mapping<dim> &mapping)
406{
407 const FESystem<dim, dim> fe = dof_handler.get_fe();
408 std::vector<std::map<unsigned int, double>> best_contact_candidate(
409 particles.size());
410 for (unsigned int p_i = 0; p_i < particles.size(); ++p_i)
411 {
412 // Clear the last boundary cell candidates.
413 boundary_cells[p_i].clear();
414
415 // Find the new cells that are at a boundary and in proximity of the
416 // particle.
417 auto cells_at_boundary = LetheGridTools::find_boundary_cells_in_sphere(
418 dof_handler,
419 particles[p_i].position,
420 particles[p_i].radius * parameters->contact_search_radius_factor);
421
422
423 // Loop over the cells at the boundary.
424 for (unsigned int i = 0; i < cells_at_boundary.size(); ++i)
425 {
426 unsigned int n_face_q_points = face_quadrature_formula.size();
427 FEFaceValues<dim> fe_face_values(mapping,
428 fe,
429 face_quadrature_formula,
430 update_values |
431 update_quadrature_points |
432 update_normal_vectors);
433 // Loop over the faces of the cell at the boundary.
434 for (int face_id = 0;
435 face_id < int(GeometryInfo<dim>::faces_per_cell);
436 ++face_id)
437 {
438 // Find the face at the boundary
439 if (cells_at_boundary[i]->face(face_id)->at_boundary())
440 {
441 fe_face_values.reinit(cells_at_boundary[i], face_id);
442 // Loop over the quadrature point of the face at the boundary
443 // to store information about the location and normals of the
444 // boundary.
445 for (unsigned int f_q_point = 0; f_q_point < n_face_q_points;
446 ++f_q_point)
447 {
448 Tensor<1, dim> normal_vector =
449 -fe_face_values.normal_vector(f_q_point);
450 BoundaryCellsInfo boundary_information;
451 boundary_information.normal_vector = normal_vector;
452 boundary_information.point_on_boundary =
453 fe_face_values.quadrature_point(f_q_point);
454 boundary_information.boundary_index =
455 cells_at_boundary[i]->face(face_id)->boundary_id();
456 auto iterator = best_contact_candidate[p_i].find(
457 boundary_information.boundary_index);
458
459 double level_set = particles[p_i].get_levelset(
460 boundary_information.point_on_boundary);
461 if (iterator != best_contact_candidate[p_i].end())
462 {
463 if (level_set <
464 best_contact_candidate[p_i][boundary_information
465 .boundary_index])
466 {
467 boundary_cells[p_i][boundary_information
468 .boundary_index] =
469 boundary_information;
470 best_contact_candidate[p_i][boundary_information
471 .boundary_index] =
472 level_set;
473 }
474 }
475 else
476 {
477 boundary_cells[p_i]
478 [boundary_information.boundary_index] =
479 boundary_information;
480 best_contact_candidate[p_i][boundary_information
481 .boundary_index] =
482 level_set;
483 }
484 }
485 }
486 }
487 }
488
489 // Regroup the information of all the processor
490 auto global_boundary_cell =
491 Utilities::MPI::all_gather(this->mpi_communicator, boundary_cells[p_i]);
492 boundary_cells[p_i].clear();
493 for (unsigned int i = 0; i < global_boundary_cell.size(); ++i)
494 {
495 boundary_cells[p_i].insert(global_boundary_cell[i].begin(),
496 global_boundary_cell[i].end());
497 }
498 }
499}

502template <int dim>
503void
504IBParticlesDEM<dim>::calculate_pw_contact_force(
505 const double dt_dem,
506 std::vector<Tensor<1, 3>> &contact_force,
507 std::vector<Tensor<1, 3>> &contact_torque)
508{
509 ObjectProperties wall_properties;
510 wall_properties.object_friction_coefficient =
511 parameters->wall_friction_coefficient;
512 wall_properties.object_poisson_ratio = parameters->wall_poisson_ratio;
513 wall_properties.object_restitution_coefficient =
514 parameters->wall_restitution_coefficient;
515 wall_properties.object_rolling_friction_coefficient =
516 parameters->wall_rolling_friction_coefficient;
517 wall_properties.object_youngs_modulus = parameters->wall_youngs_modulus;
518
519 using dealii::numbers::PI;
520
521 // Loop over the particles
522 for (auto &particle : dem_particles)
523 {
524 if (particle.integrate_motion)
525 {
526 // Defines map with default values.
527 unsigned int boundary_index = 0;
528 std::map<unsigned int, DefaultDBL_MAX> best_dist;
529 std::map<unsigned int, DefaultUINT_MAX> best_indices;
530 std::map<unsigned int, BoundaryCellsInfo> best_cells;
531 // Loop over the point and normal identified as
532 // potential contact candidate.
533 for (auto boundary_cell_iter =
534 boundary_cells[particle.particle_id].begin();
535 boundary_cell_iter != boundary_cells[particle.particle_id].end();
536 boundary_cell_iter++)
537 {
538 // Find the best candidate (the closest point) for each
539 // different wall.
540 double dist = particle.get_levelset(
541 boundary_cell_iter->second.point_on_boundary);
542
543 // Check if the distance is smaller than the best distance.
544 // If it is the first time this boundary is encountered, the
545 // distance is compared to the default value of the map, which
546 // is DBL_MAX
547 if (dist <
548 best_dist[boundary_cell_iter->second.boundary_index].value)
549 {
550 best_dist[boundary_cell_iter->second.boundary_index].value =
551 dist;
552 best_indices[boundary_cell_iter->second.boundary_index]
553 .value = boundary_index;
554 best_cells[boundary_cell_iter->second.boundary_index] =
555 boundary_cell_iter->second;
556 }
557 boundary_index += 1;
558 }
559
560 // Add all the floating walls as contact candidates. Their indices
561 // start from 1M (defined in the definition of:
562 // lowest_floating_wall_indices). This prevents a floating wall
563 // from sharing the same indices as a normal boundary of the
564 // domain in the contact candidates list.
565 for (unsigned int i = 0;
566 i < floating_walls_parameters->points_on_walls.size();
567 ++i)
568 {
569 if (floating_walls_parameters->time_start[i] < cfd_time &&
570 floating_walls_parameters->time_end[i] > cfd_time)
571 {
572 BoundaryCellsInfo floating_wall_cell_info;
573 floating_wall_cell_info.point_on_boundary =
574 floating_walls_parameters->points_on_walls[i];
575 floating_wall_cell_info.normal_vector =
576 floating_walls_parameters->floating_walls_normal_vectors[i];
577 best_cells[lowest_floating_wall_indices + i + 1] =
578 floating_wall_cell_info;
579 }
580 }
581
582 // Do the particle wall contact calculation with the best
583 // candidate.
584 for (auto best_index_of_face = best_cells.begin();
585 best_index_of_face != best_cells.end();
586 ++best_index_of_face)
587 {
588 auto &boundary_cell = best_index_of_face->second;
589
590 auto boundary_cell_information = boundary_cell;
591 ContactInfo contact_info;
592
593 // Check if there is already information on the contact
594 // between this particle and this boundary contact point. If
595 // not initialize the contact history with 0 values.
596 try
597 {
598 contact_info = pw_contact_map[particle.particle_id]
599 [boundary_cell.boundary_index];
600 }
601 catch (...)
602 {
603 for (int d = 0; d < dim; ++d)
604 {
605 contact_info.tangential_displacement[d] = 0;
606 contact_info.tangential_relative_velocity[d] = 0;
607 }
608
609 contact_info.normal_overlap = 0;
610 contact_info.normal_relative_velocity = 0;
611
612 pw_contact_map[particle.particle_id]
613 [boundary_cell.boundary_index] = contact_info;
614 }
615
616 Tensor<1, 3> normal =
617 tensor_nd_to_3d(boundary_cell_information.normal_vector);
618 auto point_on_boundary =
619 boundary_cell_information.point_on_boundary;
620
621 Point<3> particle_position_3d = point_nd_to_3d(particle.position);
622 Tensor<1, 3> particle_velocity_3d =
623 tensor_nd_to_3d(particle.velocity);
624
625 // Finding the projected vector on the normal vector of the
626 // boundary. Here we have used the private function
627 // find_projection. Using this projected vector, the
628 // particle-wall distance is calculated
629
630 // Create a Rotation matrix from normal to z axis to
631 // initialize the plane for contact with the right
632 // orientation;
633 Tensor<1, 3> rotation_axis;
634 double angle = 0;
635 if constexpr (dim == 2)
636 {
637 rotation_axis[2] = 1;
638 angle = std::atan2(-normal[0], normal[1]);
639 }
640 else
641 {
642 angle =
643 std::acos(scalar_product(Tensor<1, 3>({0, 0, 1}), normal) /
644 normal.norm());
645 if (abs(scalar_product(Tensor<1, 3>({0, 0, 1}), normal)) !=
646 normal.norm())
647 {
648 rotation_axis =
649 cross_product_3d(Tensor<1, 3>({0, 0, 1}), normal);
650 rotation_axis = rotation_axis / rotation_axis.norm();
651 }
652 else
653 {
654 rotation_axis = Tensor<1, 3>({0, 1, 0});
655 }
656 }
657 Tensor<2, 3> rotation_matrix =
658 Physics::Transformations::Rotations::rotation_matrix_3d(
659 rotation_axis, angle);
660 Tensor<1, 3> orientation =
661 particle.shape->rotation_matrix_to_xyz_angles(rotation_matrix);
662
663 std::shared_ptr<Shape<dim>> contact_plane =
664 std::make_shared<Plane<dim>>(point_on_boundary, orientation);
665 contact_plane->set_position(point_on_boundary);
666 contact_plane->set_orientation(orientation);
667
668
669 std::vector<Point<dim>> contact_point_candidate;
670 auto iterator =
671 previous_wall_contact_point[particle.particle_id].find(
672 boundary_cell.boundary_index);
673 if (iterator !=
674 previous_wall_contact_point[particle.particle_id].end())
675 {
676 contact_point_candidate.push_back(iterator->second);
677 }
678 else
679 {
680 contact_point_candidate.push_back(point_on_boundary);
681 }
682 // Use the last contact point as an initial guess if the level
683 // set is smaller than the wall initial guess. Find the normal
684 // overlap
685 auto contact_state =
686 particle.shape->distance_to_shape(*contact_plane,
687 contact_point_candidate);
688 double normal_overlap = -2 * std::get<double>(contact_state);
689 Point<3> contact_point =
690 point_nd_to_3d(std::get<Point<dim>>(contact_state));
691 // The normal vector goes from particle one to particle two.
692 // Here, particle two is the wall. As such, we flip the
693 // normal. Here, particle two is the wall. As such, we flip
694 // the normal.
695 Tensor<1, 3> contact_normal = -normal;
696
697
698 // Keep the last contact point as an initial guess for the
699 // next contact point.
700 previous_wall_contact_point[particle.particle_id]
701 [boundary_cell.boundary_index] =
702 std::get<Point<dim>>(contact_state);
703
704 if (normal_overlap > 0)
705 {
706 // Evaluates the curvature radius at the
707 // contact point.
708 double contact_radius_particle_one;
709
710 if (parameters->approximate_radius_for_contact)
711 {
712 contact_radius_particle_one = particle.radius;
713 }
714 else
715 {
716 contact_radius_particle_one =
717 particle.shape->local_curvature_radius(
718 std::get<Point<dim>>(contact_state));
719 }
720 // Do the calculation to evaluate the particle wall
721 // contact force.
722 contact_info.normal_overlap = normal_overlap;
723
724 Tensor<1, 3> normal_force;
725 Tensor<1, 3> tangential_force;
726 Tensor<1, 3> tangential_torque;
727 Tensor<1, 3> rolling_resistance_torque;
728 Tensor<1, 3> wall_velocity;
729 Tensor<1, 3> wall_angular_velocity;
730 Point<dim> wall_center_of_rotation;
731
732 pw_contact_map[particle.particle_id][boundary_index]
733 .normal_overlap = normal_overlap;
734 pw_contact_map[particle.particle_id][boundary_index]
735 .normal_vector = normal;
736
737 ObjectProperties particle_properties;
738 particle_properties.object_friction_coefficient =
739 particle.friction_coefficient;
740 particle_properties.object_mass = particle.mass;
741 particle_properties.object_poisson_ratio =
742 particle.poisson_ratio;
743 particle_properties.object_radius =
744 contact_radius_particle_one;
745 particle_properties.object_restitution_coefficient =
746 particle.restitution_coefficient;
747 particle_properties.object_rolling_friction_coefficient =
748 particle.rolling_friction_coefficient;
749 particle_properties.object_youngs_modulus =
750 particle.youngs_modulus;
751
752 // Wall mass is 100 times higher than
753 // the particles that contact it.
754 wall_properties.object_mass = particle.mass * 100;
755 // Wall curvature radius is 100 times higher than
756 // the particle that contacts it.
757 wall_properties.object_radius = particle.radius * 100;
758
759
760 calculate_force_model(normal_overlap,
761 contact_info,
762 contact_point,
763 contact_normal,
764 normal_force,
765 tangential_force,
766 rolling_resistance_torque,
767 particle.position,
768 particle_velocity_3d,
769 particle.omega,
770 particle_properties,
771 wall_center_of_rotation,
772 wall_velocity,
773 wall_angular_velocity,
774 wall_properties,
775 dt_dem);
776
777 // Updating the force of particles in the particle handler
778 contact_force[particle.particle_id] -=
779 (normal_force + tangential_force);
780 // Updating the torque acting on particles
781 contact_torque[particle.particle_id] +=
782 tangential_torque + rolling_resistance_torque +
783 cross_product_3d((contact_point - particle_position_3d),
784 -(normal_force + tangential_force));
785 }
786 else
787 {
788 // Set to 0 the tangential displacement if the particle is not
789 // in contact with the wall anymore.
790 for (int d = 0; d < dim; ++d)
791 {
792 contact_info.tangential_displacement[d] = 0.;
793 contact_info.rolling_resistance_spring_torque[d] = 0.;
794 }
795 }
796 }
797 }
798 }

802template <int dim>
803void
804IBParticlesDEM<dim>::calculate_pw_lubrication_force(
805 const double /*dt_dem*/,
806 const double h_max,
807 const double h_min,
808 const double mu,
809 std::vector<Tensor<1, 3>> &lubrication_force,
810 std::vector<Tensor<1, 3>> & /*lubrication_torque*/)
811{
812 using dealii::numbers::PI;
813
814 // Loop over the particles
815 for (auto &particle : dem_particles)
816 {
817 if (particle.integrate_motion)
818 {
819 // Defines map with default values.
820 unsigned int boundary_index = 0;
821 std::map<unsigned int, DefaultDBL_MAX> best_dist;
822 std::map<unsigned int, DefaultUINT_MAX> best_indices;
823 std::map<unsigned int, BoundaryCellsInfo> best_cells;
824 // For each particle loop over the point and normal identified as
825 // potential contact candidate.
826 for (auto boundary_cell_iter =
827 boundary_cells[particle.particle_id].begin();
828 boundary_cell_iter != boundary_cells[particle.particle_id].end();
829 boundary_cell_iter++)
830 {
831 // Find the best candidate (the closest point) for each
832 // different wall.
833
834 double dist = particle.get_levelset(
835 boundary_cell_iter->second.point_on_boundary);
836
837 // Check if the distance is smaller than the best distance.
838 // If it is the first time this boundary is encountered, the
839 // distance is compared to the default value of the map, which
840 // is DBL_MAX
841 if (dist <
842 best_dist[boundary_cell_iter->second.boundary_index].value)
843 {
844 best_dist[boundary_cell_iter->second.boundary_index].value =
845 dist;
846 best_indices[boundary_cell_iter->second.boundary_index]
847 .value = boundary_index;
848 best_cells[boundary_cell_iter->second.boundary_index] =
849 boundary_cell_iter->second;
850 }
851 boundary_index += 1;
852 }
853
854
855 // Add all the floating wall has contact candidate. Their indices
856 // start from 1M (define in the definition of:
857 // lowest_floating_wall_indices). This prevents a floating wall
858 // from sharing the same indices as a normal boundary of the
859 // domain in the contact candidates list.
860 for (unsigned int i = 0;
861 i < floating_walls_parameters->points_on_walls.size();
862 ++i)
863 {
864 if (floating_walls_parameters->time_start[i] < cfd_time &&
865 floating_walls_parameters->time_end[i] > cfd_time)
866 {
867 BoundaryCellsInfo floating_wall_cell_info;
868 floating_wall_cell_info.point_on_boundary =
869 floating_walls_parameters->points_on_walls[i];
870 floating_wall_cell_info.normal_vector =
871 floating_walls_parameters->floating_walls_normal_vectors[i];
872 best_cells[lowest_floating_wall_indices + i + 1] =
873 floating_wall_cell_info;
874 }
875 }
876
877 // Do the particle wall contact calculation with the best
878 // candidate.
879 for (auto best_index_of_face = best_cells.begin();
880 best_index_of_face != best_cells.end();
881 ++best_index_of_face)
882 {
883 auto &boundary_cell = best_index_of_face->second;
884
885 auto boundary_cell_information = boundary_cell;
886 Tensor<1, 3> normal =
887 tensor_nd_to_3d(boundary_cell_information.normal_vector);
888 auto point_on_boundary =
889 boundary_cell_information.point_on_boundary;
890
891 // Calculation of gap
892 Point<3> particle_position_3d = point_nd_to_3d(particle.position);
893 Point<3> point_on_boundary_3d = point_nd_to_3d(point_on_boundary);
894
895 Tensor<1, 3> point_to_particle_vector =
896 particle_position_3d - point_on_boundary_3d;
897
898 // Finding the projected vector on the normal vector of the
899 // boundary. Here we have used the private function
900 // find_projection. Using this projected vector, the
901 // particle-wall distance is calculated.
902 Tensor<1, 3> projected_vector =
903 ((point_to_particle_vector * normal) / (normal.norm_square())) *
904 normal;
905
906 double gap = (projected_vector.norm()) - particle.radius;
907
908 Tensor<1, 3> radial_vector;
909 double radial_velocity;
910 Tensor<1, 3> f_lub;
911
912 radial_vector = projected_vector;
913 if (gap > 0 and gap < h_max)
914 // This means that the particles is very close to the wall
915 // but not in contact
916 {
917 // Limit the smallest gap calculated
918 if (gap < h_min)
919 {
920 gap = h_min;
921 }
922 // Evaluate the force
923 radial_velocity =
924 scalar_product(-radial_vector, particle.velocity);
925 f_lub = 3 / 2 * PI * mu * (particle.radius) *
926 (particle.radius) / gap * radial_velocity *
927 radial_vector / radial_vector.norm() -
928 3 / 2 * PI * mu * (particle.radius) *
929 (particle.radius) / h_max * radial_velocity *
930 radial_vector / radial_vector.norm();
931 }
932 lubrication_force[particle.particle_id] = f_lub;
933 }
934 }
935 }

938
939template <int dim>
940void
941IBParticlesDEM<dim>::calculate_force_model(
942 const double normal_overlap,
943 ContactInfo &contact_info,
944 Point<3> &contact_point,
945 Tensor<1, 3> &contact_normal,
946 Tensor<1, 3> &normal_force,
947 Tensor<1, 3> &tangential_force,
948 Tensor<1, 3> &rolling_resistance_torque,
949 Point<dim> &particle_one_position,
950 Tensor<1, 3> &particle_one_velocity,
951 Tensor<1, 3> &particle_one_omega,
952 const ObjectProperties particle_one_properties,
953 Point<dim> &particle_two_position,
954 Tensor<1, 3> &particle_two_velocity,
955 Tensor<1, 3> &particle_two_omega,
956 const ObjectProperties particle_two_properties,
957 const double dt)
958{
959 double effective_youngs_modulus;
960 double effective_shear_modulus;
961 double effective_coefficient_of_restitution;
962 double effective_coefficient_of_friction;
963 double effective_coefficient_of_rolling_friction;
964 double model_parameter_beta;
965
966
967 // Define local variables which will be used within the contact calculation
968 // Namely: normal and tangential contact forces, tangential and rolling
969 // torques, normal unit vector of the contact and contact relative velocity
970 // in the normal direction
971 Tensor<1, 3> normal_unit_vector;
972 double normal_relative_velocity_value;
973 Tensor<1, 3> tangential_relative_velocity;
974
975 // DEM::PropertiesIndex::type is the first (0) property of particles in the
976 // DEM solver. For the IB particles, the first property is ID. For force and
977 // torque calculations, we need pairwise properties (such as effective
978 // Young's modulus, effective coefficient of restitution, etc.) We rewrite
979 // these pairwise properties by using the ID of IB particles (using
980 // DEM::PropertiesIndex::type) and use them in force calculations.
981
982 effective_youngs_modulus =
983 (particle_one_properties.object_youngs_modulus *
984 particle_two_properties.object_youngs_modulus) /
985 ((particle_two_properties.object_youngs_modulus *
986 (1.0 - particle_one_properties.object_poisson_ratio *
987 particle_one_properties.object_poisson_ratio)) +
988 (particle_one_properties.object_youngs_modulus *
989 (1.0 - particle_two_properties.object_poisson_ratio *
990 particle_two_properties.object_poisson_ratio)) +
991 DBL_MIN);
992
993 effective_shear_modulus =
994 (particle_one_properties.object_youngs_modulus *
995 particle_one_properties.object_youngs_modulus) /
996 (2.0 * ((particle_two_properties.object_youngs_modulus *
997 (2.0 - particle_one_properties.object_poisson_ratio) *
998 (1.0 + particle_two_properties.object_poisson_ratio)) +
999 (particle_one_properties.object_youngs_modulus *
1000 (2.0 - particle_two_properties.object_poisson_ratio) *
1001 (1.0 + particle_two_properties.object_poisson_ratio))) +
1002 DBL_MIN);
1003
1004 effective_coefficient_of_restitution =
1005 harmonic_mean(particle_one_properties.object_restitution_coefficient,
1006 particle_two_properties.object_restitution_coefficient);
1007 effective_coefficient_of_friction =
1008 harmonic_mean(particle_one_properties.object_friction_coefficient,
1009 particle_two_properties.object_friction_coefficient);
1010 effective_coefficient_of_rolling_friction =
1011 harmonic_mean(particle_one_properties.object_rolling_friction_coefficient,
1012 particle_two_properties.object_rolling_friction_coefficient);
1013
1014 const double restitution_coefficient_particle_log =
1015 std::log(effective_coefficient_of_restitution);
1016
1017 model_parameter_beta = restitution_coefficient_particle_log /
1018 sqrt(restitution_coefficient_particle_log *
1019 restitution_coefficient_particle_log +
1020 9.8696);
1021
1022 // Since the normal overlap is already calculated we update
1023 // this element of the container here. The rest of information
1024 // are updated using the following function
1025
1026 ////////////// Contact info ////////////////
1027
1028 // Calculation of the normal unit contact vector
1029 normal_unit_vector = contact_normal;
1030
1031 // Defining relative contact velocity
1032 Tensor<1, 3> contact_relative_velocity;
1033
1034 // Assigning velocities and angular velocities of particles
1035 Tensor<1, 3> velocity_of_particle_one_at_contact_point =
1036 particle_one_velocity +
1037 cross_product_3d(particle_one_omega,
1038 (contact_point - point_nd_to_3d(particle_one_position)));
1039
1040 Tensor<1, 3> velocity_of_particle_two_at_contact_point =
1041 particle_two_velocity +
1042 cross_product_3d(particle_two_omega,
1043 (contact_point - point_nd_to_3d(particle_two_position)));
1044 contact_relative_velocity = (velocity_of_particle_one_at_contact_point -
1045 velocity_of_particle_two_at_contact_point);
1046
1047 // Calculation of normal relative velocity. Note that in the
1048 // following line the product acts as inner product since both
1049 // sides are vectors, while in the second line the product is
1050 // scalar and vector product
1051 normal_relative_velocity_value =
1052 contact_relative_velocity * normal_unit_vector;
1053
1054 // Calculation of tangential relative velocity
1055 // v_rt = v_ij - (v_ijn_ij)*n_ij
1056 tangential_relative_velocity =
1057 contact_relative_velocity -
1058 (normal_relative_velocity_value * normal_unit_vector);
1059
1060 // Calculation of new tangential_displacement, since this value is
1061 // history-dependent it needs the value at previous time-step
1062 // This variable is the main reason that we have iterations over
1063 // two different vectors : tangential_displacement of the particles
1064 // which were already in contact needs to be
1065 // modified using its history, while the tangential_displacements of
1066 // new particles are equal to zero
1067 // delta_t_new = delta_t_old + v_rt*dt
1068 contact_info.tangential_relative_velocity.push_back(
1069 tangential_relative_velocity);
1070 contact_info.tangential_displacement += tangential_relative_velocity * dt;
1071
1072 ///////////// Hertz contact force model ////////////////
1073 // Calculation of effective radius and mass
1074 double effective_mass =
1075 (particle_one_properties.object_mass *
1076 particle_two_properties.object_mass) /
1077 (particle_one_properties.object_mass + particle_two_properties.object_mass);
1078 double effective_radius = (particle_one_properties.object_radius *
1079 particle_two_properties.object_radius) /
1080 ((particle_one_properties.object_radius +
1081 particle_two_properties.object_radius));
1082
1083 const double radius_times_overlap_sqrt =
1084 sqrt(effective_radius * normal_overlap);
1085 const double model_parameter_sn =
1086 2 * effective_youngs_modulus * radius_times_overlap_sqrt;
1087
1088 // Calculation of normal and tangential spring and dashpot constants
1089 // using particle properties
1090 double normal_spring_constant = 0.66665 * model_parameter_sn;
1091 double normal_damping_constant =
1092 -1.8257 * model_parameter_beta * sqrt(model_parameter_sn * effective_mass);
1093 double tangential_spring_constant =
1094 8.0 * effective_shear_modulus * radius_times_overlap_sqrt + DBL_MIN;
1095
1096 // Calculation of normal force using spring and dashpot normal forces
1097 normal_force =
1098 ((normal_spring_constant * normal_overlap) * normal_unit_vector) +
1099 ((normal_damping_constant * normal_relative_velocity_value) *
1100 normal_unit_vector);
1101
1102 // Calculation of tangential force using spring and dashpot tangential
1103 // forces. Since we need dashpot tangential force in the gross sliding
1104 // again, we define it as a separate variable
1105 tangential_force =
1106 tangential_spring_constant * contact_info.tangential_displacement;
1107
1108 double coulomb_threshold =
1109 effective_coefficient_of_friction * normal_force.norm();
1110
1111 // Check for gross sliding
1112 if (tangential_force.norm() > coulomb_threshold)
1113 {
1114 // Gross sliding occurs and the tangential displacement and tangential
1115 // force are limited to Coulomb's criterion
1116 tangential_force =
1117 coulomb_threshold *
1118 (tangential_force / (tangential_force.norm() + DBL_MIN));
1119 }
1120
1121 // For calculation of rolling resistance torque, we need to obtain
1122 // omega_ij using rotational velocities of particles one and two
1123 Tensor<1, 3> omega_ij = particle_one_omega - particle_two_omega;
1124 Tensor<1, 3> omega_ij_direction = omega_ij / (omega_ij.norm() + DBL_MIN);
1125
1126 // Calculation of the linear rolling resistance torque
1127 rolling_resistance_torque =
1128 (-effective_coefficient_of_rolling_friction * effective_radius *
1129 normal_force.norm() * omega_ij_direction);
1130}

1134template <int dim>
1135void
1136IBParticlesDEM<dim>::integrate_particles_motion(const double dt,
1137 const double h_max,
1138 const double h_min,
1139 const double rho,
1140 const double mu)
1141{
1142 // Initialize local containers and physical variables
1143 using dealii::numbers::PI;
1144 double dt_dem = dt / parameters->coupling_frequency;
1145
1146 std::vector<Tensor<1, 3>> contact_force(dem_particles.size());
1147 std::vector<Tensor<1, 3>> contact_wall_force(dem_particles.size());
1148 std::vector<Tensor<1, 3>> contact_torque(dem_particles.size());
1149 std::vector<Tensor<1, 3>> contact_wall_torque(dem_particles.size());
1150 std::vector<Tensor<1, 3>> current_fluid_force(dem_particles.size());
1151 std::vector<Tensor<1, 3>> current_fluid_torque(dem_particles.size());
1152 std::vector<Tensor<1, 3>> lubrication_force(dem_particles.size());
1153 std::vector<Tensor<1, 3>> lubrication_torque(dem_particles.size());
1154 std::vector<Tensor<1, 3>> lubrication_wall_force(dem_particles.size());
1155 std::vector<Tensor<1, 3>> lubrication_wall_torque(dem_particles.size());
1156
1157 std::vector<Tensor<1, 3>> velocity(dem_particles.size());
1158 std::vector<Point<dim>> position(dem_particles.size());
1159
1160 // Local time for the dem step.
1161 double t = 0;
1162 // The gravitational acceleration.
1163 Tensor<1, 3> g;
1164 this->parameters->f_gravity->set_time(cfd_time + dt);
1165 // The gravitational force on the particle.
1166 Tensor<1, 3> gravity;
1167 // Reset the previous particle contact point list
1168 previous_wall_contact_point.clear();
1169 previous_particle_particle_contact_point.clear();
1170
1171 // Initialize the particles
1172 for (unsigned int p_i = 0; p_i < dem_particles.size(); ++p_i)
1173 {
1174 dem_particles[p_i].position = dem_particles[p_i].previous_positions[0];
1175 dem_particles[p_i].velocity = dem_particles[p_i].previous_velocity[0];
1176 dem_particles[p_i].omega = dem_particles[p_i].previous_omega[0];
1177 dem_particles[p_i].orientation =
1178 dem_particles[p_i].previous_orientation[0];
1179 dem_particles[p_i].impulsion = 0;
1180 dem_particles[p_i].omega_impulsion = 0;
1181 dem_particles[p_i].contact_impulsion = 0;
1182 dem_particles[p_i].omega_contact_impulsion = 0;
1183 // Initialized the gravity at the particle position.
1184 g[0] = this->parameters->f_gravity->value(dem_particles[p_i].position, 0);
1185 g[1] = this->parameters->f_gravity->value(dem_particles[p_i].position, 1);
1186 if (dim == 3)
1187 g[2] =
1188 this->parameters->f_gravity->value(dem_particles[p_i].position, 2);
1189 dem_particles[p_i].set_position(dem_particles[p_i].position);
1190 dem_particles[p_i].set_orientation(dem_particles[p_i].orientation);
1191 }
1192
1193 // Integrate with the sub_time_step
1194 while (t + 0.5 * dt_dem < dt)
1195 {
1196 // Initialize vector to contain the RK 4 step
1197 std::vector<Tensor<1, 3>> last_velocity;
1198 std::vector<Point<dim>> last_position;
1199 std::vector<Tensor<1, 3>> last_omega;
1200 std::vector<Tensor<2, 3>> last_orientation_matrix;
1201 auto last_pp_contact_info = pp_contact_map;
1202 auto last_pw_contact_info = pw_contact_map;
1203
1204 last_velocity = std::vector<Tensor<1, 3>>(dem_particles.size());
1205 last_position = std::vector<Point<dim>>(dem_particles.size());
1206 last_omega = std::vector<Tensor<1, 3>>(dem_particles.size());
1207 last_orientation_matrix = std::vector<Tensor<2, 3>>(dem_particles.size());
1208
1209 std::vector<std::vector<Tensor<1, 3>>> k_velocity =
1210 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1211 std::vector<Tensor<1, 3>>(4));
1212 std::vector<std::vector<Tensor<1, dim>>> k_position =
1213 std::vector<std::vector<Tensor<1, dim>>>(
1214 dem_particles.size(), std::vector<Tensor<1, dim>>(4));
1215 std::vector<std::vector<Tensor<1, 3>>> k_omega =
1216 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1217 std::vector<Tensor<1, 3>>(4));
1218 std::vector<std::vector<Tensor<1, 3>>> k_orientation =
1219 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1220 std::vector<Tensor<1, 3>>(4));
1221 std::vector<std::vector<Tensor<1, 3>>> k_impulsion =
1222 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1223 std::vector<Tensor<1, 3>>(4));
1224 std::vector<std::vector<Tensor<1, 3>>> k_omega_impulsion =
1225 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1226 std::vector<Tensor<1, 3>>(4));
1227 std::vector<std::vector<Tensor<1, 3>>> k_contact_impulsion =
1228 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1229 std::vector<Tensor<1, 3>>(4));
1230 std::vector<std::vector<Tensor<1, 3>>> k_omega_contact_impulsion =
1231 std::vector<std::vector<Tensor<1, 3>>>(dem_particles.size(),
1232 std::vector<Tensor<1, 3>>(4));
1233
1234 // Solve each of the 4 step of the RK4 method
1235 for (unsigned int step = 0; step < 4; ++step)
1236 {
1237 std::fill(current_fluid_force.begin(), current_fluid_force.end(), 0);
1238 std::fill(current_fluid_torque.begin(),
1239 current_fluid_torque.end(),
1240 0);
1241 std::fill(contact_torque.begin(), contact_torque.end(), 0);
1242 std::fill(contact_force.begin(), contact_force.end(), 0);
1243 std::fill(contact_wall_force.begin(), contact_wall_force.end(), 0);
1244 std::fill(contact_wall_torque.begin(), contact_wall_torque.end(), 0);
1245 std::fill(lubrication_force.begin(), lubrication_force.end(), 0);
1246 std::fill(lubrication_torque.begin(), lubrication_torque.end(), 0);
1247 std::fill(lubrication_wall_force.begin(),
1248 lubrication_wall_force.end(),
1249 0);
1250 std::fill(lubrication_wall_torque.begin(),
1251 lubrication_wall_torque.end(),
1252 0);
1253
1254 // define local time of the rk step
1255 double local_dt = dt_dem * 0.5;
1256 if (step == 3)
1257 {
1258 local_dt = local_dt * 2;
1259 }
1260 // Calculate particle-particle and particle-wall contact force
1261
1262 calculate_pp_contact_force(local_dt, contact_force, contact_torque);
1263 calculate_pw_contact_force(local_dt,
1264 contact_wall_force,
1265 contact_wall_torque);
1266
1267 if (parameters->enable_lubrication_force)
1268 {
1269 calculate_pp_lubrication_force(local_dt,
1270 h_max,
1271 h_min,
1272 mu,
1273 lubrication_force,
1274 lubrication_torque);
1275 calculate_pw_lubrication_force(local_dt,
1276 h_max,
1277 h_min,
1278 mu,
