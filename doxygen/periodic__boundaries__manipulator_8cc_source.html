23template <int dim>
24void
25PeriodicBoundariesManipulator<dim>::get_periodic_boundaries_info(
26 typename Triangulation<dim>::cell_iterator cell,
27 const unsigned int face_id,
28 periodic_boundaries_cells_info_struct<dim> &boundaries_information)
29{
30 // Initialize a simple quadrature for the system. This will be used to
31 // obtain a single sample point on the boundaries faces.
32 const FE_Q<dim> fe(1);
33 QGauss<dim - 1> face_quadrature_formula(1);
34 FEFaceValues<dim> fe_face_values(fe,
35 face_quadrature_formula,
36 update_values | update_quadrature_points |
37 update_normal_vectors);
38 Tensor<1, dim> normal_vector;
39 Point<dim> quad_point;
40
41 // Store information of the cell
42 boundaries_information.cell = cell;
43 boundaries_information.boundary_id = cell->face(face_id)->boundary_id();
44
45 // Store information of the periodic cell
46 boundaries_information.periodic_cell = cell->periodic_neighbor(face_id);
47
48 // Find the normal vector of the boundary face and point
49 fe_face_values.reinit(cell, face_id);
50 boundaries_information.normal_vector = fe_face_values.normal_vector(0);
51 boundaries_information.point_on_face = fe_face_values.quadrature_point(0);
52
53 // Find the normal vector of the periodic boundary face and point with the
54 // reinit FE face values
55 fe_face_values.reinit(boundaries_information.periodic_cell,
56 cell->periodic_neighbor_face_no(face_id));
57 normal_vector = fe_face_values.normal_vector(0);
58 quad_point = fe_face_values.quadrature_point(0);
59 boundaries_information.periodic_normal_vector = normal_vector;
60 boundaries_information.point_on_periodic_face = quad_point;
61}

63template <int dim>
64void
65PeriodicBoundariesManipulator<dim>::map_periodic_cells(
66 const parallel::distributed::Triangulation<dim> &triangulation,
67 typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
68 &periodic_boundaries_cells_information)
69{
70 if (!periodic_boundaries_enabled)
71 return;
72
73 periodic_boundaries_cells_information.clear();
74
75 // Iterating over the active cells in the triangulation
76 for (const auto &cell : triangulation.active_cell_iterators())
77 {
78 if (cell->is_locally_owned() || cell->is_ghost())
79 {
80 if (cell->at_boundary())
81 {
82 // Iterating over cell faces
83 for (const auto &face : cell->face_iterators())
84 {
85 // Check if face is on the periodic boundary 0
86 // Pairs of periodic cells are stored once.
87 unsigned int face_boundary_id = face->boundary_id();
88 if (face_boundary_id == periodic_boundary_0)
89 {
90 // Save boundaries information related to the cell on
91 // the periodic boundary 0
92 // Information about both boundaries are stored in
93 // periodic_boundary_cells_info_struct
94 periodic_boundaries_cells_info_struct<dim>
95 boundaries_information;
96 unsigned int face_id = cell->face_iterator_to_index(face);
97
98 get_periodic_boundaries_info(cell,
99 face_id,
100 boundaries_information);
101
102 // Store boundaries information in map with cell id at
103 // periodic boundary 0 as key
104 periodic_boundaries_cells_information.insert(
105 {boundaries_information.cell
106 ->global_active_cell_index(),
107 boundaries_information});
108 }
109 }
110 }
111 }
112 }
113
114 // Store constant periodic offset on each processor (if container not empty)
115 // Information of faces on periodic boundaries is from the first pair of
116 // periodic cells since all pairs of faces have the same offset.
117 if (!periodic_boundaries_cells_information.empty())
118 {
119 periodic_boundaries_cells_info_struct<dim> first_cells_content =
120 periodic_boundaries_cells_information.begin()->second;
121 constant_periodic_offset[direction] =
122 first_cells_content.point_on_periodic_face[direction] -
123 first_cells_content.point_on_face[direction];
124 }
125}

127template <int dim>
128void
129PeriodicBoundariesManipulator<dim>::check_and_move_particles(
130 const periodic_boundaries_cells_info_struct<dim> &boundaries_cells_content,
131 const bool &particles_in_pb0_cell,
132 typename Particles::ParticleHandler<dim>::particle_iterator_range
133 &particles_in_cell,
134 bool &particle_has_been_moved)
135{
136 for (auto particle = particles_in_cell.begin();
137 particle != particles_in_cell.end();
138 ++particle)
139 {
140 // Get the current particle location
141 Point<dim> particle_position = particle->get_location();
142
143 // Initialize points and normal vector related of the cell that contains
144 // the particle
145 Point<dim> point_on_face;
146 Tensor<1, dim> normal_vector, distance_between_faces;
147
148 if (particles_in_pb0_cell)
149 {
150 point_on_face = boundaries_cells_content.point_on_face;
151 normal_vector = boundaries_cells_content.normal_vector;
152 distance_between_faces = constant_periodic_offset;
153 }
154 else
155 {
156 point_on_face = boundaries_cells_content.point_on_periodic_face;
157 normal_vector = boundaries_cells_content.periodic_normal_vector;
158 distance_between_faces = -constant_periodic_offset;
159 }
160
161 // Calculate distance between particle position and the face of
162 // boundary cell d = n•(pt_particle - pt_face)
163 double distance_with_face =
164 scalar_product(particle_position - point_on_face, normal_vector);
165
166 // If distance >= 0, particle is outside of cell (or on face).
167 // If so, particle location is modified to get moved into the periodic
168 // cell.
169 if (distance_with_face >= 0.0)
170 {
171 // Move particle outside the current cell to the periodic cell.
172 particle_position += distance_between_faces;
173 particle->set_location(particle_position);
174
175 // Update flag to indicate that particle has been moved
176 particle_has_been_moved = true;
177 }
178 }
179}

181template <int dim>
182bool
183PeriodicBoundariesManipulator<dim>::execute_particles_displacement(
184 const Particles::ParticleHandler<dim> &particle_handler,
185 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
186 &periodic_boundaries_cells_information)
187{
188 if (!periodic_boundaries_enabled)
189 return false;
190
191 bool particle_has_been_moved = false;
192
193 if (!periodic_boundaries_cells_information.empty())
194 {
195 for (auto boundaries_cells_information_iterator =
196 periodic_boundaries_cells_information.begin();
197 boundaries_cells_information_iterator !=
198 periodic_boundaries_cells_information.end();
199 ++boundaries_cells_information_iterator)
200 {
201 // Get the cell and periodic cell from map
202 auto boundaries_cells_content =
203 boundaries_cells_information_iterator->second;
204 auto cell = boundaries_cells_content.cell;
205 auto periodic_cell = boundaries_cells_content.periodic_cell;
206
207 // Check and execute displacement of particles crossing a periodic
208 // wall.
209 // Case when particle goes from periodic boundary 0 to 1
210 if (cell->is_locally_owned())
211 {
212 typename Particles::ParticleHandler<dim>::particle_iterator_range
213 particles_in_cell = particle_handler.particles_in_cell(cell);
214 const bool particles_exist_in_main_cell =
215 !particles_in_cell.empty();
216
217 if (particles_exist_in_main_cell)
218 {
219 bool particles_in_pb0_cell = true;
220 check_and_move_particles(boundaries_cells_content,
221 particles_in_pb0_cell,
222 particles_in_cell,
223 particle_has_been_moved);
224 }
225 }
226
227 // Case when particle goes from periodic boundary 1 to 0
228 if (periodic_cell->is_locally_owned())
229 {
230 typename Particles::ParticleHandler<dim>::particle_iterator_range
231 particles_in_periodic_cell =
232 particle_handler.particles_in_cell(periodic_cell);
233 const bool particles_exist_in_periodic_cell =
234 !particles_in_periodic_cell.empty();
235
236 if (particles_exist_in_periodic_cell)
237 {
238 bool particles_in_pb0_cell = false;
239 check_and_move_particles(boundaries_cells_content,
240 particles_in_pb0_cell,
241 particles_in_periodic_cell,
242 particle_has_been_moved);
243 }
244 }
245 }
246 }
247
248 return particle_has_been_moved;
249}
