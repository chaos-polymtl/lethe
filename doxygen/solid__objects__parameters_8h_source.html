4/**
5 * @file solid_objects_parameters.h
6 * @brief Parameter classes for solid objects used in Nitsche immersed boundary
7 * and DEM simulations.
8 *
9 * This file defines the parameter structures for configuring solid objects
10 * within Lethe simulations. It provides three levels of solid object
11 * management:
12 * - NitscheObject: individual solid for the Nitsche immersed boundary method.
13 * - Nitsche: container managing multiple NitscheObject instances.
14 * - RigidSolidObject: individual rigid solid for DEM simulations.
15 * - DEMSolidObjects: container managing multiple RigidSolidObject instances
16 * as surfaces and volumes.
17 */

245 const std::vector<double> temp =
246 convert_string_to_vector<double>(prm, "center of rotation");

250 Utilities::int_to_string(dim) +
251 " dimensional point."));
252
253 for (unsigned int i = 0; i < dim; ++i)
254 {
255 center_of_rotation[i] = temp.at(i);
256 }
257

259 calculate_torque_on_solid = prm.get_bool("calculate torque on solid");
260 force_output_name = prm.get("solid force name");
261 torque_output_name = prm.get("solid torque name");
262 }
263 prm.leave_subsection();

345 {

421 template <int dim>
422 void
423 RigidSolidObject<dim>::declare_parameters(ParameterHandler &prm,
424 unsigned int id)
425 {
426 // Use ParsedFunction<dim> during parameter declaration. We need to do this
427 // to use the declare_parameters function, which is not possible with
428 // std::make_shared<Function<dim>> type. Please refer to the comment before
429 // the translational_velocity and angular_velocity attributes declaration.
430 auto translational_velocity_parsed =
431 std::make_shared<Functions::ParsedFunction<dim>>(dim);
432 auto angular_velocity_parsed =
433 std::make_shared<Functions::ParsedFunction<dim>>(3);
434 auto solid_temperature_parsed =
435 std::make_shared<Functions::ParsedFunction<dim>>(1);
436
437 prm.enter_subsection("solid object " + Utilities::int_to_string(id, 1));
438 {
439 solid_mesh.declare_parameters(prm);
440
441 prm.enter_subsection("translational velocity");
442 translational_velocity_parsed->declare_parameters(prm, dim);
443 prm.leave_subsection();
444
445 prm.enter_subsection("angular velocity");
446 angular_velocity_parsed->declare_parameters(prm, 3);
447 prm.leave_subsection();
448
449 if constexpr (dim == 2)
450 {
451 prm.declare_entry("center of rotation",
452 "0., 0.",
453 Patterns::List(Patterns::Double()),
454 "Solid object center of rotation");
455 }
456 if constexpr (dim == 3)
457 {
458 prm.declare_entry("center of rotation",
459 "0., 0., 0.",
460 Patterns::List(Patterns::Double()),
461 "Solid object center of rotation");
462 }
463
464 prm.declare_entry("thermal boundary type",
465 "adiabatic",
466 Patterns::Selection("adiabatic|isothermal"),
467 "Choosing thermal boundary type"
468 "Choices are <adiabatic|isothermal>.");
469
470 // Isothermal boundary
471 prm.enter_subsection("temperature");
472 solid_temperature_parsed->declare_parameters(prm, 1);
473 prm.leave_subsection();
474
475 prm.declare_entry("output solid object",
476 "true",
477 Patterns::Bool(),
478 "Controls the generation of output files");
479 }
480 prm.leave_subsection();
481
482 // Cast to std::shared_ptr<Function<dim>> after parameter declaration.
483 translational_velocity = translational_velocity_parsed;
484 angular_velocity = angular_velocity_parsed;
485 solid_temperature = solid_temperature_parsed;
486 }

488 template <int dim>
489 void
490 RigidSolidObject<dim>::parse_parameters(ParameterHandler &prm,
491 unsigned int id)
492 {
493 // Use ParsedFunction<dim> during parameter declaration. We need to do this
494 // to use the parse_parameters function, which is not possible with
495 // std::make_shared<Function<dim>> type. Please refer to the comment before
496 // the translational_velocity and angular_velocity attributes declaration.
497 auto translational_velocity_parsed =
498 std::make_shared<Functions::ParsedFunction<dim>>(dim);
499 auto angular_velocity_parsed =
500 std::make_shared<Functions::ParsedFunction<dim>>(3);
501 auto solid_temperature_parsed =
502 std::make_shared<Functions::ParsedFunction<dim>>(1);
503
504 prm.enter_subsection("solid object " + Utilities::int_to_string(id, 1));
505 {
506 solid_mesh.parse_parameters(prm);
507 prm.enter_subsection("translational velocity");
508 translational_velocity_parsed->parse_parameters(prm);
509 prm.leave_subsection();
510
511 prm.enter_subsection("angular velocity");
512 angular_velocity_parsed->parse_parameters(prm);
513 prm.leave_subsection();
514
515 const std::vector<double> temp =
516 convert_string_to_vector<double>(prm, "center of rotation");
517
518 AssertThrow(temp.size() == dim,
519 ExcMessage("Invalid center of rotation. This should be a " +
520 Utilities::int_to_string(dim) +
521 " dimensional point."));
522
523 for (unsigned int i = 0; i < dim; ++i)
524 {
525 center_of_rotation[i] = temp.at(i);
526 }
527
528 output_bool = prm.get_bool("output solid object");
529
530 const std::string thermal_type = prm.get("thermal boundary type");
531 if (thermal_type == "adiabatic")
532 thermal_boundary_type = ThermalBoundaryType::adiabatic;
533 else if (thermal_type == "isothermal")
534 thermal_boundary_type = ThermalBoundaryType::isothermal;
535 else
536 {
537 throw(std::runtime_error("Invalid thermal boundary type"));
538 }
539
540 // Isothermal boundary
541 prm.enter_subsection("temperature");
542 solid_temperature_parsed->parse_parameters(prm);
543 prm.leave_subsection();
544 }
545 prm.leave_subsection();
546 translational_velocity = translational_velocity_parsed;
547 angular_velocity = angular_velocity_parsed;
548 solid_temperature = solid_temperature_parsed;
549 }
