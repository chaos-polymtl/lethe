32  Parameters::Lagrangian::RollingResistanceMethod::viscous_resistance)

38  dem_parameters.forces_torques.calculate_force_torque;
39  this->center_mass_container = dem_parameters.forces_torques.point_center_mass;

46 void
47 ParticleWallDMTForce<dim>::calculate_particle_wall_contact_force(
48  typename DEM::dem_data_structures<dim>::particle_wall_in_contact
49  &particle_wall_pairs_in_contact,
50  const double dt,
51  std::vector<Tensor<1, 3>> &torque,
52  std::vector<Tensor<1, 3>> &force)
53 {
54  constexpr double M_2PI = 6.283185307179586; // 2. * M_PI
55 
56  ParticleWallContactForce<dim>::force_on_walls =
57  ParticleWallContactForce<dim>::initialize();
58  ParticleWallContactForce<dim>::torque_on_walls =
59  ParticleWallContactForce<dim>::initialize();
60 
61  // Set the force_calculation_threshold_distance. This is useful for non-
62  // contact cohesive force models such as the DMT force model.
63  const double force_calculation_threshold_distance =
64  set_dmt_cut_off_distance();
65 
66  // Looping over particle_wall_pairs_in_contact, which means looping over all
67  // the active particles with iterator particle_wall_pairs_in_contact_iterator
68  for (auto &&pairs_in_contact_content :
69  particle_wall_pairs_in_contact | boost::adaptors::map_values)
70  {
71  // Now an iterator (particle_wall_contact_information_iterator) on each
72  // element of the particle_wall_pairs_in_contact vector is defined. This
73  // iterator iterates over a map which contains the required information
74  // for calculation of the contact force for each particle
75  for (auto &&contact_information :
76  pairs_in_contact_content | boost::adaptors::map_values)
77  {
78  // Defining the total force of contact, properties of particle as
79  // local parameters
80  auto particle = contact_information.particle;
81  auto particle_properties = particle->get_properties();
82 
83  auto normal_vector = contact_information.normal_vector;
84  auto point_on_boundary = contact_information.point_on_boundary;
85 
86  Point<3> particle_location_3d = [&] {
87  if constexpr (dim == 3)
88  {
89  return particle->get_location();
90  }
91  if constexpr (dim == 2)
92  {
93  return (point_nd_to_3d(particle->get_location()));
94  }
95  }();
96  ;
97 
98  // A vector (point_to_particle_vector) is defined which connects the
99  // center of particle to the point_on_boundary. This vector will then
100  // be projected on the normal vector of the boundary to obtain the
101  // particle-wall distance
102  Tensor<1, 3> point_to_particle_vector =
103  particle_location_3d - point_on_boundary;
104 
105  // Finding the projected vector on the normal vector of the boundary.
106  // Here we have used the private function find_projection. Using this
107  // projected vector, the particle-wall distance is calculated
108  Tensor<1, 3> projected_vector =
109  this->find_projection(point_to_particle_vector, normal_vector);
110 
111  double normal_overlap =
112  ((particle_properties[DEM::PropertiesIndex::dp]) * 0.5) -
113  (projected_vector.norm());
114 
115  // Minimal delta_star. We know a force has to be computed.
116  if (normal_overlap > force_calculation_threshold_distance)
117  {
118  // i is the particle, j is the wall.
119  // we need to put a minus sign in front of the
120  // normal_vector to respect the convention (i ->
121  // j)
122  const unsigned int particle_type =
123  particle_properties[DEM::PropertiesIndex::type];
124  const double effective_radius =
125  0.5 * particle_properties[DEM::PropertiesIndex::dp];
126  const double effective_surface_energy =
127  this->effective_surface_energy[particle_type];
128  const double effective_hamaker_constant =
129  this->effective_hamaker_constant[particle_type];
130 
131  const double F_po =
132  M_2PI * effective_radius * effective_surface_energy;
133 
134  const double delta_0 = -std::sqrt(effective_hamaker_constant *
135  effective_radius / (6. * F_po));
136  // Cohesive force. This will need to be added to the
137  // first vector inside the tuple.
138  Tensor<1, 3> cohesive_force;
139 
140  // This tuple (forces and torques) contains four
141  // elements which are: 1, normal force, 2,
142  // tangential force, 3, tangential torque and 4,
143  // rolling resistance torque, respectively
144  std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
145  forces_and_torques;
146 
147  forces_and_torques = std::make_tuple(Tensor<1, 3>(),
148  Tensor<1, 3>(),
149  Tensor<1, 3>(),
150  Tensor<1, 3>());
151  // Contact + constant cohesive force.
152  if (normal_overlap > 0.)
153  {
154  // i is the particle, j is the wall.
155  // we need to put a minus sign infront of the
156  // normal_vector to respect the convention (i ->
157  // j)
158  cohesive_force = -F_po * -normal_vector;
159 
160  contact_information.normal_overlap = normal_overlap;
161  this->update_contact_information(contact_information,
162  particle_location_3d,
163  particle_properties,
164  dt);
165 
166  // This tuple (forces and torques) contains
167  // four elements which are: 1, normal force,
168  // 2, tangential force, 3, tangential torque
169  // and 4, rolling resistance torque,
170  // respectively
171  forces_and_torques =
172  this->calculate_nonlinear_contact_force_and_torque(
173  contact_information, particle_properties);
174  }
175  // No contact, but still in the constant zone for the cohesive
176  // force.
177  else if (normal_overlap > delta_0)
178  {
179  // i is the particle, j is the wall.
180  // we need to put a minus sign infront of the
181  // normal_vector to respect the convention (i ->
182  // j)
183  cohesive_force = -F_po * -normal_vector;
184 
185  // No contact means there is no tangential
186  // overlap.
187  for (int d = 0; d < dim; ++d)
188  {
189  contact_information.tangential_overlap[d] = 0.;
190  }
191  }
192  else // between delta_* and delta_0.
193  {
194  // i is the particle, j is the wall.
195  // we need to put a minus sign infront of the
196  // normal_vector to respect the convention (i ->
197  // j)
198  const double F_cohesion =
199  effective_hamaker_constant * effective_radius /
200  (6. * Utilities::fixed_power<2>(normal_overlap));
201 
202  // No contact means there is not tangential
203  // overlap.
204  for (int d = 0; d < dim; ++d)
205  {
206  contact_information.tangential_overlap[d] = 0.;
207  }
208  cohesive_force = -F_cohesion * -normal_vector;
209  }
210 
211  // Get particle's torque and force
212  types::particle_index particle_id = particle->get_local_index();
213 
214  Tensor<1, 3> &particle_torque = torque[particle_id];
215  Tensor<1, 3> &particle_force = force[particle_id];
216 
217  // Added the cohesive term
218  std::get<0>(forces_and_torques) += cohesive_force;
219 
220  // Apply the calculated forces and torques on the particle
221  this->apply_force_and_torque(forces_and_torques,
222  particle_torque,
223  particle_force,
224  point_on_boundary,
225  contact_information.boundary_id);
226  }
227  else // No cohesive force
228  {
229  // No contact means there is not tangential
230  // overlap.
231  for (int d = 0; d < dim; ++d)
232  {
233  contact_information.tangential_overlap[d] = 0.;
234  }
235  }
236  }
237  }
238 }

243 ParticleWallDMTForce<dim>::calculate_particle_floating_wall_contact_force(
244  typename DEM::dem_data_structures<dim>::particle_floating_mesh_in_contact
245  &particle_floating_mesh_in_contact,
246  const double dt,
247  std::vector<Tensor<1, 3>> &torque,
248  std::vector<Tensor<1, 3>> &force,
249  const std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> &solids)
250 {
251  constexpr double M_2PI = 6.283185307179586; // 2. * M_PI
252 
253  std::vector<Particles::ParticleIterator<dim>> particle_locations;
254  std::vector<Point<dim>> triangle(this->vertices_per_triangle);
255 
256  // Set the force_calculation_threshold_distance. This is useful for non-
257  // contact cohesive force models such as the DMT force model.
258  const double force_calculation_threshold_distance =
259  set_dmt_cut_off_distance();
260 
261  for (unsigned int solid_counter = 0; solid_counter < solids.size();
262  ++solid_counter)
263  {
264  // Get translational and rotational velocities and
265  // center of rotation
266  Tensor<1, 3> translational_velocity =
267  solids[solid_counter]->get_translational_velocity();
268  Tensor<1, 3> angular_velocity =
269  solids[solid_counter]->get_angular_velocity();
270  Point<3> center_of_rotation =
271  solids[solid_counter]->get_center_of_rotation();
272 
273  auto &particle_floating_mesh_contact_pair =
274  particle_floating_mesh_in_contact[solid_counter];
275 
276  for (auto &[cut_cell, map_info] : particle_floating_mesh_contact_pair)
277  {
278  if (!map_info.empty())
279  {
280  // Clear the particle locations vector for the new cut cell
281  particle_locations.clear();
282  const unsigned int n_particles = map_info.size();
283 
284  // Gather all the particles locations in a vector
285  for (auto &&contact_info : map_info | boost::adaptors::map_values)
286  {
287  particle_locations.push_back(contact_info.particle);
288  }
289 
290  // Build triangle vector
291  for (unsigned int vertex = 0;
292  vertex < this->vertices_per_triangle;
293  ++vertex)
294  {
295  // Find vertex-floating wall distance
296  triangle[vertex] = cut_cell->vertex(vertex);
297  }
298 
299  // Call find_particle_triangle_projection to get the
300  // distance and projection of particles on the triangle
301  // (floating mesh cell)
302  auto particle_triangle_information =
303  LetheGridTools::find_particle_triangle_projection(
304  triangle, particle_locations, n_particles);
305 
306  const std::vector<bool> pass_distance_check =
307  std::get<0>(particle_triangle_information);
308  const std::vector<Point<3>> projection_points =
309  std::get<1>(particle_triangle_information);
310  const std::vector<Tensor<1, 3>> normal_vectors =
311  std::get<2>(particle_triangle_information);
312 
313  unsigned int particle_counter = 0;
314 
315  for (auto &&contact_info : map_info | boost::adaptors::map_values)
316  {
317  // If particle passes the distance check
318  if (pass_distance_check[particle_counter])
319  {
320  // Define the total force of contact, properties of
321  // particle as local parameters
322  auto &particle = contact_info.particle;
323  auto particle_properties = particle->get_properties();
324 
325  const Point<3> &projection_point =
326  projection_points[particle_counter];
327 
328  Point<3> particle_location_3d;
329 
330  if constexpr (dim == 3)
331  particle_location_3d = particle->get_location();
332 
333  if constexpr (dim == 2)
334  particle_location_3d =
335  point_nd_to_3d(particle->get_location());
336 
337  const double particle_triangle_distance =
338  particle_location_3d.distance(projection_point);
339 
340  // Find normal overlap
341  double normal_overlap =
342  ((particle_properties[DEM::PropertiesIndex::dp]) *
343  0.5) -
344  particle_triangle_distance;
345 
346  // We check if a force need to be computed.
347  if (normal_overlap > force_calculation_threshold_distance)
348  {
349  // i is the particle, j is the wall.
350  // we need to put a minus sign in front of the
351  // normal_vector to respect the convention (i ->
352  // j)
353  const unsigned int particle_type =
354  particle_properties[DEM::PropertiesIndex::type];
355  const double effective_radius =
356  0.5 * particle_properties[DEM::PropertiesIndex::dp];
357  const double effective_surface_energy =
358  this->effective_surface_energy[particle_type];
359  const double effective_hamaker_constant =
360  this->effective_hamaker_constant[particle_type];
361 
362  const double F_po =
363  M_2PI * effective_radius * effective_surface_energy;
364 
365  const double delta_0 =
366  -std::sqrt(effective_hamaker_constant *
367  effective_radius / (6. * F_po));
368 
369  contact_info.normal_overlap = normal_overlap;
370 
371  contact_info.normal_vector =
372  normal_vectors[particle_counter];
373 
374  contact_info.point_on_boundary = projection_point;
375 
376  contact_info.boundary_id = solid_counter;
377 
378  this
379  ->update_particle_floating_wall_contact_information(
380  contact_info,
381  particle_properties,
382  dt,
383  translational_velocity,
384  angular_velocity,
385  center_of_rotation.distance(
386  particle_location_3d));
387

389  // first vector inside the tuple.

393  // elements which are: 1, normal force, 2,

410  // i is the particle, j is the wall.
411  // we need to put a minus sign infront of the
412  // normal_vector to respect the convention (i ->
413  // j)
414  cohesive_force =
415  -F_po * -normal_vectors[particle_counter];
416 
417  // This tuple (forces and torques) contains
418  // four elements which are: 1, normal force,
419  // 2, tangential force, 3, tangential torque

421  // respectively
422  forces_and_torques =

429  else if (normal_overlap > delta_0)
430  {
431  // i is the particle, j is the wall.
432  // we need to put a minus sign infront of the
433  // normal_vector to respect the convention (i ->
434  // j)
435  cohesive_force =
436  -F_po * -normal_vectors[particle_counter];
437 
438  // No contact means there is no tangential
439  // overlap.
440  for (int d = 0; d < dim; ++d)
441  {
442  contact_info.tangential_overlap[d] = 0.;
