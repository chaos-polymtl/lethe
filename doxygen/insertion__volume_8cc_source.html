32 // The main insertion function. Insert_global_function is utilized to insert
33 // the particles
34 template <int dim>
35 void
36 InsertionVolume<dim>::insert(
37  Particles::ParticleHandler<dim> &particle_handler,
38  const parallel::distributed::Triangulation<dim> &triangulation,
39  const DEMSolverParameters<dim> &dem_parameters)
40 {
41  if (particles_of_each_type_remaining == 0 &&
42  current_inserting_particle_type !=
43  dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
44  {
45  particles_of_each_type_remaining =
46  dem_parameters.lagrangian_physical_properties.number.at(
47  ++current_inserting_particle_type);
48  }
49 
50  // Check to see if the remaining un-inserted particles are equal to zero or
51  // not
52  if (particles_of_each_type_remaining != 0)
53  {
54  if (this->removing_particles_in_region)
55  {
56  if (this->mark_for_update)
57  {
58  this->find_cells_in_removing_box(triangulation);
59  this->mark_for_update = false;
60  }
61  this->remove_particles_in_box(particle_handler);
62  }
63 
64  MPI_Comm communicator = triangulation.get_communicator();
65  ConditionalOStream pcout(
66  std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
67 
68  auto this_mpi_process = Utilities::MPI::this_mpi_process(communicator);
69  auto n_mpi_process = Utilities::MPI::n_mpi_processes(communicator);
70 
71 
72  this->calculate_insertion_domain_maximum_particle_number(dem_parameters,
73  pcout);
74  // The inserted_this_step value is the minimum of
75  // particles_of_each_type_remaining and inserted_this_step
76  this->inserted_this_step =
77  std::min(particles_of_each_type_remaining, this->inserted_this_step);
78 
79  // Obtaining global bounding boxes
80  const auto my_bounding_box =
81  GridTools::compute_mesh_predicate_bounding_box(
82  triangulation, IteratorFilters::LocallyOwnedCell());
83  const auto global_bounding_boxes =
84  Utilities::MPI::all_gather(communicator, my_bounding_box);
85 
86  // Distributing particles between processors
87  this->inserted_this_step_this_proc =
88  floor(this->inserted_this_step / n_mpi_process);
89  if (this_mpi_process == (n_mpi_process - 1))
90  this->inserted_this_step_this_proc =
91  this->inserted_this_step -
92  (n_mpi_process - 1) * floor(this->inserted_this_step / n_mpi_process);
93 
94  // Call random number generator
95  std::vector<double> random_number_vector;
96  random_number_vector.reserve(this->inserted_this_step_this_proc);
97  this->create_random_number_container(
98  random_number_vector,
99  dem_parameters.insertion_info.insertion_maximum_offset,
100  dem_parameters.insertion_info.seed_for_insertion);
101 
102  Point<dim> insertion_location;
103  std::vector<Point<dim>> insertion_points_on_proc;
104  insertion_points_on_proc.reserve(this->inserted_this_step_this_proc);
105 
106  // Find the first and the last particle id for each process
107  // The number of particles on the last process is different
108  unsigned int first_id;
109  unsigned int last_id;
110  if (this_mpi_process == (n_mpi_process - 1))
111  {
112  first_id =
113  this->inserted_this_step - this->inserted_this_step_this_proc;
114  last_id = this->inserted_this_step;
115  }
116  // For the processes 1 : n-1
117  else
118  {
119  first_id = this_mpi_process * this->inserted_this_step_this_proc;
120  last_id = (this_mpi_process + 1) * this->inserted_this_step_this_proc;
121  }
122 
123  // Looping through the particles on each process and finding their
124  // insertion location
125  unsigned int particle_counter = 0;
126  for (unsigned int id = first_id; id < last_id; ++id, ++particle_counter)
127  {
128  find_insertion_location_volume(
129  insertion_location,
130  id,
131  random_number_vector[particle_counter],
132  random_number_vector[this->inserted_this_step - particle_counter -
133  1],
134  dem_parameters.insertion_info);
135  insertion_points_on_proc.push_back(insertion_location);
136  }
137 
138  std::vector<std::vector<double>> particle_properties;
139 
140  // Assigning inserted particles properties using
141  // assign_particle_properties function
142  this->assign_particle_properties(dem_parameters,
143  this->inserted_this_step_this_proc,
144  current_inserting_particle_type,
145  particle_properties);
146 
147  // Insert the particles using the points and assigned properties
148  particle_handler.insert_global_particles(insertion_points_on_proc,
149  global_bounding_boxes,
150  particle_properties);
151 
152  // Updating remaining particles
153  particles_of_each_type_remaining -= this->inserted_this_step;
154 
155  this->print_insertion_info(this->inserted_this_step,
156  particles_of_each_type_remaining,
157  current_inserting_particle_type,
158  pcout);
159  }
160 }

166 InsertionVolume<dim>::create_random_number_container(
167  std::vector<double> &random_container,
168  const double maximum_range,
169  const int seed_for_insertion)
170 {
171  for (unsigned int i = 0; i < this->inserted_this_step; ++i)
172  {

174  random_container.push_back((((double)rand()) / ((double)RAND_MAX)) *

179 // This function assigns the insertion points of the inserted particles
180 template <>
181 void
182 InsertionVolume<2>::find_insertion_location_volume(
183  Point<2> &insertion_location,
184  const unsigned int id,
185  const double random_number1,
186  const double random_number2,
187  const Parameters::Lagrangian::InsertionInfo &insertion_information)
188 {
189  std::vector<int> insertion_index;
190  insertion_index.resize(2);
191 
192  unsigned int axis_0, axis_1;
193  int number_of_particles_0;
194 
195  // First direction (axis) to have particles inserted
196  axis_0 = insertion_information.direction_sequence.at(0);
197  number_of_particles_0 = this->number_of_particles_directions[axis_0];
198  insertion_index[axis_0] = id % number_of_particles_0;
199  insertion_location[axis_0] =
200  this->axis_min[axis_0] + ((insertion_index[axis_0] + 0.5) *
201  insertion_information.distance_threshold -
202  random_number1) *
203  this->maximum_diameter;
204 
205  // Second direction (axis) to have particles inserted
206  axis_1 = insertion_information.direction_sequence.at(1);
207  insertion_index[axis_1] = static_cast<int>(id / number_of_particles_0);
208  insertion_location[axis_1] =
209  this->axis_min[axis_1] + ((insertion_index[axis_1] + 0.5) *
210  insertion_information.distance_threshold -
211  random_number2) *
212  this->maximum_diameter;
213 }
