10// The constructor of the volume insertion class. In the constructor, we
11// investigate if the insertion box is adequately large to handle the desired
12// number of inserted particles. The number of insertion points in each
13// direction (number_of_particles_x_direction, number_of_particles_y_direction
14// and number_of_particles_z_direction) are also obtained
15template <int dim, typename PropertiesIndex>
16InsertionVolume<dim, PropertiesIndex>::InsertionVolume(
17 const std::vector<std::shared_ptr<Distribution>>
18 &size_distribution_object_container,
19 const parallel::distributed::Triangulation<dim> &triangulation,
20 const DEMSolverParameters<dim> &dem_parameters,
21 const double maximum_particle_diameter)
22 : Insertion<dim, PropertiesIndex>(size_distribution_object_container,
23 triangulation,
24 dem_parameters)
25 , particles_of_each_type_remaining(
26 dem_parameters.lagrangian_physical_properties.number.at(0))
27 , acceptance_fct(dem_parameters.insertion_info.insertion_acceptance_fct)
28 , maximum_diameter(maximum_particle_diameter)
29{
30 // Initializing the current inserting particle type
31 current_inserting_particle_type = 0;
32
33 // We need to fill the vector filted_box_index.
34 // To do this, we loop of every insertion point inside the box considering
35 // that the acceptance_fct accepts every point.
36 // For each point, we check if it respects the condition. If so, we insert
37 // the index associated with this point inside the vector.
38 set_filtered_index(dem_parameters.insertion_info);
39}

42template <int dim, typename PropertiesIndex>
43void
44InsertionVolume<dim, PropertiesIndex>::insert(
45 Particles::ParticleHandler<dim> &particle_handler,
46 const parallel::distributed::Triangulation<dim> &triangulation,
47 const DEMSolverParameters<dim> &dem_parameters)
48{
49 if (particles_of_each_type_remaining == 0 &&
50 current_inserting_particle_type !=
51 dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
52 {
53 particles_of_each_type_remaining =
54 dem_parameters.lagrangian_physical_properties.number.at(
55 ++current_inserting_particle_type);
56 }
57
58 // Check if the remaining number of uninserted particles is equal to zero
59 if (particles_of_each_type_remaining != 0)
60 {
61 // Remove the particle if the remove box region feature is activated
62 if (this->removing_particles_in_region)
63 {
64 if (this->mark_for_update)
65 {
66 this->find_cells_in_removing_box(triangulation);
67 this->mark_for_update = false;
68 }
69 this->remove_particles_in_box(particle_handler);
70 }
71
72 // Get the MPI communicator and the parallel cout
73 MPI_Comm communicator = triangulation.get_mpi_communicator();
74 ConditionalOStream pcout(
75 std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
76
77 // Compute the number of particle to insert during this step.
78 unsigned int inserted_this_step;
79 adjust_insertion_count_by_insertion_box_capacity(
80 dem_parameters.insertion_info, pcout, inserted_this_step);
81
82 // Get global bounding boxes
83 const auto my_bounding_box =
84 GridTools::compute_mesh_predicate_bounding_box(
85 triangulation, IteratorFilters::LocallyOwnedCell());
86 const auto global_bounding_boxes =
87 Utilities::MPI::all_gather(communicator, my_bounding_box);
88
89 // Call the random number generator for the offsets
90 std::vector<double> random_number_vector;
91 random_number_vector.reserve(inserted_this_step);
92 create_random_number_container(
93 random_number_vector,
94 inserted_this_step,
95 dem_parameters.insertion_info.insertion_maximum_offset,
96 dem_parameters.insertion_info.seed_for_insertion);
97
98 Point<dim> insertion_location;
99 std::vector<Point<dim>> insertion_points_on_proc;
100 insertion_points_on_proc.reserve(filted_box_index.size());
101
102 // Looping through the particles on each process and finding their
103 // insertion location
104 unsigned int particle_counter = 0;
105 for (unsigned int global_index = first_index_this_proc;
106 global_index < last_index_this_proc &&
107 global_index < inserted_this_step;
108 ++global_index, ++particle_counter)
109 {
110 find_insertion_location(insertion_location,
111 filted_box_index.at(particle_counter),
112 random_number_vector.at(particle_counter),
113 random_number_vector.at(inserted_this_step -
114 particle_counter - 1),
115 dem_parameters.insertion_info);
116
117 insertion_points_on_proc.push_back(insertion_location);
118 }
119
120 std::vector<std::vector<double>> particle_properties;
121
122 // Assigning inserted particles properties using
123 // assign_particle_properties function
124 this->assign_particle_properties(dem_parameters,
125 particle_counter,
126 current_inserting_particle_type,
127 insertion_points_on_proc,
128 particle_properties);
129
130 // Insert the particles using the points and assigned properties
131 particle_handler.insert_global_particles(insertion_points_on_proc,
132 global_bounding_boxes,
133 particle_properties);
134
135 // Updating remaining particles
136 particles_of_each_type_remaining -= inserted_this_step;
137
138 this->print_insertion_info(inserted_this_step,
139 particles_of_each_type_remaining,
140 current_inserting_particle_type,
141 pcout);
142 }
143}

145// This function assigns the insertion points of the inserted particles
146template <int dim, typename PropertiesIndex>
147void
148InsertionVolume<dim, PropertiesIndex>::find_insertion_location(
149 Point<dim> &insertion_location,
150 const unsigned int id,
151 const double random_number1,
152 const double random_number2,
153 const Parameters::Lagrangian::InsertionInfo<dim> &insertion_information)
154{
155 std::vector<int> insertion_index;
156 insertion_index.resize(dim);
157
158 // First direction (axis) to have particles inserted
159 unsigned int axis_0 = insertion_information.direction_sequence[0];
160 unsigned int number_of_particles_0 = number_of_particles_directions[axis_0];
161 insertion_index[axis_0] = id % number_of_particles_0;
162 insertion_location[axis_0] =
163 axis_min[axis_0] + ((insertion_index[axis_0] + 0.5) *
164 insertion_information.distance_threshold -
165 random_number1) *
166 maximum_diameter;
167
168 // Second direction (axis) to have particles inserted
169 unsigned int axis_1 = insertion_information.direction_sequence.at(1);
170
171 if constexpr (dim == 2)
172 {
173 insertion_index[axis_1] = static_cast<int>(id / number_of_particles_0);
174 insertion_location[axis_1] =
175 axis_min[axis_1] + ((insertion_index[axis_1] + 0.5) *
176 insertion_information.distance_threshold -
177 random_number2) *
178 maximum_diameter;
179 }
180 else
181 {
182 unsigned int number_of_particles_1 =
183 number_of_particles_directions[axis_1];
184 insertion_index[axis_1] =
185 static_cast<int>(id % (number_of_particles_0 * number_of_particles_1)) /
186 (number_of_particles_0);
187 insertion_location[axis_1] =
188 axis_min[axis_1] + ((insertion_index[axis_1] + 0.5) *
189 insertion_information.distance_threshold -
190 random_number2) *
191 maximum_diameter;
192
193 // Third direction (axis) to have particles inserted
194 unsigned int axis_2 = insertion_information.direction_sequence[2];
195 insertion_index[axis_2] =
196 static_cast<int>(id / (number_of_particles_0 * number_of_particles_1));
197 insertion_location[axis_2] =
198 axis_min[axis_2] + ((insertion_index[axis_2] + 0.5) *
199 insertion_information.distance_threshold -
200 random_number1) *
201 maximum_diameter;
202 }
203}

206template <int dim, typename PropertiesIndex>
207void
208InsertionVolume<dim, PropertiesIndex>::set_filtered_index(
209 const InsertionInfo<dim> &insertion_information)
210{
211 // If you don't insert particles in this simulation, we don't need to build
212 // the container.
213 if (insertion_information.inserted_this_step == 0)
214 return;
215
216 // Check if the insertion directions are valid (no repetition)
217 unsigned int axis_sum = 0;
218 if constexpr (dim == 2)
219 {
220 axis_sum = insertion_information.direction_sequence[0] +
221 insertion_information.direction_sequence[1];
222
223 AssertThrow(
224 axis_sum == 1,
225 ExcMessage("Invalid insertion directions: 2 directions are the same "));
226 }
227 if constexpr (dim == 3)
228 {
229 axis_sum = insertion_information.direction_sequence[0] +
230 insertion_information.direction_sequence[1] +
231 insertion_information.direction_sequence[2];
232
233 AssertThrow(
234 axis_sum == 3,
235 ExcMessage(
236 "Invalid insertion directions: at least 2 directions are the same "));
237 }
238
239 // This variable is used to compute the maximum number of particles
240 // that can fit in the chosen insertion box before the acceptance function
241 // is applied.
242 unsigned int maximum_number_of_points = 1;
243
244 number_of_particles_directions.resize(dim);
245 axis_min.resize(dim);
246 axis_max.resize(dim);
247 std::vector<unsigned int> axis_list = {
248 insertion_information.direction_sequence[0],
249 insertion_information.direction_sequence[1]};
250
251 if constexpr (dim == 3)
252 {
253 axis_list.push_back(insertion_information.direction_sequence[2]);
254 }
255
256 // Assigning the minimum and maximum positions of the insertion box in respect
257 // to the axis order.
258 for (const unsigned int axis : axis_list)
259 {
260 switch (axis)
261 {
262 case 0:
263 axis_min[0] = insertion_information.insertion_box_point_1(0);
264 axis_max[0] = insertion_information.insertion_box_point_2(0);
265 break;
266 case 1:
267 axis_min[1] = insertion_information.insertion_box_point_1(1);
268 axis_max[1] = insertion_information.insertion_box_point_2(1);
269 break;
270 case 2:
271 axis_min[2] = insertion_information.insertion_box_point_1(2);
272 axis_max[2] = insertion_information.insertion_box_point_2(2);
273 break;
274 default:
275 AssertThrow(false,
276 ExcMessage("Insertion direction must be 0, 1 or 2"));
277 }
278
279 // Assign the maximum number of insertion points according to the
280 // direction and calculate the total number of points that fit in the box
281 // (maximum_number_of_points = max_x * max_y * max_z)
282 const unsigned int number_of_points =
283 (axis_max[axis] - axis_min[axis]) /
284 (insertion_information.distance_threshold * maximum_diameter);
285
286 number_of_particles_directions[axis] = number_of_points;
287
288 maximum_number_of_points *= number_of_points;
289 }
290
291 // We split the number of points evenly on every process.
292 // The last process needs to be adjusted to match the maximum number of points
293 // in the box.
294 MPI_Comm communicator = MPI_COMM_WORLD;
295 auto this_mpi_process = Utilities::MPI::this_mpi_process(communicator);
296 auto n_mpi_process = Utilities::MPI::n_mpi_processes(communicator);
297
298 unsigned int n_points_this_proc =
299 floor(maximum_number_of_points / n_mpi_process);
300 if (this_mpi_process == (n_mpi_process - 1))
301 n_points_this_proc =
302 maximum_number_of_points -
303 (n_mpi_process - 1) * floor(maximum_number_of_points / n_mpi_process);
304
305 // First and last index of the unfiltered points that were assigned to this
306 // process.
307 unsigned int first_index_unfiltered;
308 unsigned int last_index_unfiltered;
309 if (this_mpi_process == (n_mpi_process - 1)) // For the last process
310 {
311 first_index_unfiltered = maximum_number_of_points - n_points_this_proc;
312 last_index_unfiltered = maximum_number_of_points;
313 }
314 else // For processes 1 to n-1
315 {
316 first_index_unfiltered = this_mpi_process * n_points_this_proc;
317 last_index_unfiltered = (this_mpi_process + 1) * n_points_this_proc;
318 }
319
320 // Now, we know that the indexes before applying the acceptance function
321 // will go from first_index_unfiltered to last_index_unfiltered.
322 // We count the number of insertion points that respect the acceptance
323 // function and store the index of those points
324 Point<dim> insertion_location;
325 for (unsigned int index = first_index_unfiltered;
326 index < last_index_unfiltered;
327 ++index)
328 {
329 // Create the insertion point, associated with the current index, with
330 // no offset
331 find_insertion_location(
332 insertion_location, index, 0., 0., insertion_information);
333
334 // If the point respects the acceptance function, we insert the index in
335 // the vector.
336 if (acceptance_fct->value(insertion_location) > 0.)
337 filted_box_index.push_back(index);
338 }
339
340 // Each process needs to know what is its first and last filtered indexes.
341 // To do so, every process needs to know how many insertion point were kept
342 // valid by the previous process.
343 // The numeration used for the filtered indexes follows the same principal as
344 // the one used before filtering. The only difference is that we skip the
345 // rejected points.
346 std::vector<unsigned int> starting_index_on_every_proc(n_mpi_process);
347 starting_index_on_every_proc.resize(n_mpi_process);
348
349 // The number of insertion points assigned to this process that respect the
350 // acceptance function.
351 const unsigned int n_valid_insertion_point_this_proc =
352 filted_box_index.size();
353
354 // Every process sends to process 0 the number of valid insertion points it
355 // has.
356 auto number_of_insertion_point_per_core =
357 Utilities::MPI::gather(communicator, n_valid_insertion_point_this_proc, 0);
358
359 if (this_mpi_process == 0)
360 {
361 starting_index_on_every_proc[0] = 0;
362 for (unsigned int i = 1; i < n_mpi_process; ++i)
363 starting_index_on_every_proc[i] =
364 starting_index_on_every_proc[i - 1] +
365 number_of_insertion_point_per_core[i - 1];
366
367 number_of_valid_insertion_point_global =
368 std::accumulate(number_of_insertion_point_per_core.begin(),
369 number_of_insertion_point_per_core.end(),
370 0);
371 }
372
373 // This scatters the number of particles to insert
374 first_index_this_proc =
375 Utilities::MPI::scatter(communicator, starting_index_on_every_proc, 0);
376
377 // We find the last index with the size of the vector.
378 last_index_this_proc =
379 first_index_this_proc + n_valid_insertion_point_this_proc;
380
381 // We need this for the adjust_insertion_count_by_insertion_box_capacity
382 // function
383 number_of_valid_insertion_point_global =
384 Utilities::MPI::broadcast(communicator,
385 number_of_valid_insertion_point_global,
386 0);
387}

389template <int dim, typename PropertiesIndex>
390void
391InsertionVolume<dim, PropertiesIndex>::
392 adjust_insertion_count_by_insertion_box_capacity(
393 const InsertionInfo<dim> &insertion_information,
394 const ConditionalOStream &pcout,
395 unsigned int &inserted_this_step)
396{
397 // If the inserted number of particles at this step exceeds the maximum
398 // number, a warning is printed
399 if (insertion_information.inserted_this_step >
400 number_of_valid_insertion_point_global)
401 {
402 pcout << "Warning: the requested number of particles for insertion ("
403 << insertion_information.inserted_this_step
404 << ") is higher than maximum expected number of particles ("
405 << number_of_valid_insertion_point_global << ")" << std::endl;
406
407 // Updating the number of inserted particles at each step
408 inserted_this_step = number_of_valid_insertion_point_global;
409 }
410 else
411 {
412 inserted_this_step = insertion_information.inserted_this_step;
413 }
414
415 // The inserted_this_step value is the minimum of
416 // particles_of_each_type_remaining and inserted_this_step
417 inserted_this_step =
418 std::min(particles_of_each_type_remaining, inserted_this_step);
419}
