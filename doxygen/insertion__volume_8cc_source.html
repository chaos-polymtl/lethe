34// The main insertion function. Insert_global_function is utilized to insert
35// the particles
36template <int dim, typename PropertiesIndex>
37void
38InsertionVolume<dim, PropertiesIndex>::insert(
39 Particles::ParticleHandler<dim> &particle_handler,
40 const parallel::distributed::Triangulation<dim> &triangulation,
41 const DEMSolverParameters<dim> &dem_parameters)
42{
43 if (particles_of_each_type_remaining == 0 &&
44 current_inserting_particle_type !=
45 dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
46 {
47 particles_of_each_type_remaining =
48 dem_parameters.lagrangian_physical_properties.number.at(
49 ++current_inserting_particle_type);
50 }
51
52 // Check to see if the remaining un-inserted particles are equal to zero or
53 // not
54 if (particles_of_each_type_remaining != 0)
55 {
56 if (this->removing_particles_in_region)
57 {
58 if (this->mark_for_update)
59 {
60 this->find_cells_in_removing_box(triangulation);
61 this->mark_for_update = false;
62 }
63 this->remove_particles_in_box(particle_handler);
64 }
65
66 MPI_Comm communicator = triangulation.get_mpi_communicator();
67 ConditionalOStream pcout(
68 std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
69
70 auto this_mpi_process = Utilities::MPI::this_mpi_process(communicator);
71 auto n_mpi_process = Utilities::MPI::n_mpi_processes(communicator);
72
73
74 this->calculate_insertion_domain_maximum_particle_number(dem_parameters,
75 pcout);
76 // The inserted_this_step value is the minimum of
77 // particles_of_each_type_remaining and inserted_this_step
78 this->inserted_this_step =
79 std::min(particles_of_each_type_remaining, this->inserted_this_step);
80
81 // Obtaining global bounding boxes
82 const auto my_bounding_box =
83 GridTools::compute_mesh_predicate_bounding_box(
84 triangulation, IteratorFilters::LocallyOwnedCell());
85 const auto global_bounding_boxes =
86 Utilities::MPI::all_gather(communicator, my_bounding_box);
87
88 // Distributing particles between processors
89 this->inserted_this_step_this_proc = static_cast<unsigned int>(
90 floor(this->inserted_this_step / n_mpi_process));
91 if (this_mpi_process == (n_mpi_process - 1))
92 this->inserted_this_step_this_proc = static_cast<unsigned int>(
93 this->inserted_this_step -
94 (n_mpi_process - 1) *
95 floor(this->inserted_this_step / n_mpi_process));
96
97 // Call random number generator
98 std::vector<double> random_number_vector;
99 random_number_vector.reserve(this->inserted_this_step_this_proc);
100 create_random_number_container(
101 random_number_vector,
102 this->inserted_this_step,
103 dem_parameters.insertion_info.insertion_maximum_offset,
104 dem_parameters.insertion_info.seed_for_insertion);
105
106 Point<dim> insertion_location;
107 std::vector<Point<dim>> insertion_points_on_proc;
108 insertion_points_on_proc.reserve(this->inserted_this_step_this_proc);
109
110 // Find the first and the last particle id for each process
111 // The number of particles on the last process is different
112 unsigned int first_id;
113 unsigned int last_id;
114 if (this_mpi_process == (n_mpi_process - 1))
115 {
116 first_id =
117 this->inserted_this_step - this->inserted_this_step_this_proc;
118 last_id = this->inserted_this_step;
119 }
120 // For the processes 1 : n-1
121 else
122 {
123 first_id = this_mpi_process * this->inserted_this_step_this_proc;
124 last_id = (this_mpi_process + 1) * this->inserted_this_step_this_proc;
125 }
126
127 // Looping through the particles on each process and finding their
128 // insertion location
129 unsigned int particle_counter = 0;
130 for (unsigned int id = first_id; id < last_id; ++id, ++particle_counter)
131 {
132 find_insertion_location_volume(
133 insertion_location,
134 id,
135 random_number_vector[particle_counter],
136 random_number_vector[this->inserted_this_step - particle_counter -
137 1],
138 dem_parameters.insertion_info);
139 insertion_points_on_proc.push_back(insertion_location);
140 }
141
142 std::vector<std::vector<double>> particle_properties;
143
144 // Assigning inserted particles properties using
145 // assign_particle_properties function
146 this->assign_particle_properties(dem_parameters,
147 this->inserted_this_step_this_proc,
148 current_inserting_particle_type,
149 insertion_points_on_proc,
150 particle_properties);
151
152 // Insert the particles using the points and assigned properties
153 particle_handler.insert_global_particles(insertion_points_on_proc,
154 global_bounding_boxes,
155 particle_properties);
156
157 // Updating remaining particles
158 particles_of_each_type_remaining -= this->inserted_this_step;
159
160 this->print_insertion_info(this->inserted_this_step,
161 particles_of_each_type_remaining,
162 current_inserting_particle_type,
163 pcout);
164 }
165}

167// This function assigns the insertion points of the inserted particles
168template <int dim, typename PropertiesIndex>
169void
170InsertionVolume<dim, PropertiesIndex>::find_insertion_location_volume(
171 Point<dim> &insertion_location,
172 const unsigned int id,
173 const double random_number1,
174 const double random_number2,
175 const Parameters::Lagrangian::InsertionInfo<dim> &insertion_information)
176{
177 std::vector<int> insertion_index;
178 insertion_index.resize(dim);
179
180 unsigned int axis_0, axis_1;
181 int number_of_particles_0, number_of_particles_1;
182
183 // First direction (axis) to have particles inserted
184 axis_0 = insertion_information.direction_sequence.at(0);
185 number_of_particles_0 = this->number_of_particles_directions[axis_0];
186 insertion_index[axis_0] = id % number_of_particles_0;
187 insertion_location[axis_0] =
188 this->axis_min[axis_0] + ((insertion_index[axis_0] + 0.5) *
189 insertion_information.distance_threshold -
190 random_number1) *
191 this->maximum_diameter;
192
193 // Second direction (axis) to have particles inserted
194 axis_1 = insertion_information.direction_sequence.at(1);
195
196 if constexpr (dim == 2)
197 {
198 insertion_index[axis_1] = static_cast<int>(id / number_of_particles_0);
199 insertion_location[axis_1] =
200 this->axis_min[axis_1] + ((insertion_index[axis_1] + 0.5) *
201 insertion_information.distance_threshold -
202 random_number2) *
203 this->maximum_diameter;
204 }
205 else
206 {
207 number_of_particles_1 = this->number_of_particles_directions[axis_1];
208 insertion_index[axis_1] =
209 static_cast<int>(id % (number_of_particles_0 * number_of_particles_1)) /
210 (number_of_particles_0);
211 insertion_location[axis_1] =
212 this->axis_min[axis_1] + ((insertion_index[axis_1] + 0.5) *
213 insertion_information.distance_threshold -
214 random_number2) *
215 this->maximum_diameter;
216
217 // Third direction (axis) to have particles inserted
218 unsigned int axis_2;
219 axis_2 = insertion_information.direction_sequence.at(2);
220 insertion_index[axis_2] =
221 static_cast<int>(id / (number_of_particles_0 * number_of_particles_1));
222 insertion_location[axis_2] =
223 this->axis_min[axis_2] + ((insertion_index[axis_2] + 0.5) *
224 insertion_information.distance_threshold -
225 random_number1) *
226 this->maximum_diameter;
227 }
228}
