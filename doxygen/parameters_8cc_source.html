211 void
212 SimulationControl::parse_parameters(ParameterHandler &prm)
213 {
214 prm.enter_subsection("simulation control");
215 {
216 const std::string sv = prm.get("method");
217 if (sv == "steady")
218 method = TimeSteppingMethod::steady;
219 else if (sv == "steady_bdf")
220 method = TimeSteppingMethod::steady_bdf;
221 else if (sv == "bdf1")
222 method = TimeSteppingMethod::bdf1;
223 else if (sv == "bdf2")
224 method = TimeSteppingMethod::bdf2;
225 else if (sv == "bdf3")
226 method = TimeSteppingMethod::bdf3;
227 else if (sv == "sdirk22")
228 method = TimeSteppingMethod::sdirk22;
229 else if (sv == "sdirk33")
230 method = TimeSteppingMethod::sdirk33;
231 else if (sv == "sdirk43")
232 method = TimeSteppingMethod::sdirk43;
233 else
234 {
235 AssertThrow(false, ExcMessage("Invalid time stepping scheme"));
236 }
237 const std::string bdf_startup_string = prm.get("bdf startup method");
238 if (bdf_startup_string == "multiple step bdf")
239 bdf_startup_method = BDFStartupMethods::multiple_step_bdf;
240 else if (bdf_startup_string == "initial solution")
241 bdf_startup_method = BDFStartupMethods::initial_solution;
242 else
243 {
244 AssertThrow(false, ExcMessage("Invalid bdf startup scheme"));
245 }
246
247 const std::string osv = prm.get("output control");
248 if (osv == "iteration")
249 output_control = OutputControl::iteration;
250 else if (osv == "time")
251 output_control = OutputControl::time;
252 else
253 {
254 AssertThrow(false, ExcMessage("Invalid output control scheme"));
255 }
256 dt = prm.get_double("time step");
257 time_end = prm.get_double("time end");
258 adapt = prm.get_bool("adapt");
259 time_step_independent_of_end_time =
260 prm.get_bool("time step independent of end time");
261 maxCFL = prm.get_double("max cfl");
262 max_dt = prm.get_double("max time step");
263 stop_tolerance = prm.get_double("stop tolerance");
264 adaptative_time_step_scaling =
265 prm.get_double("adaptative time step scaling");
266 startup_timestep_scaling = prm.get_double("startup time scaling");
267 number_mesh_adaptation = prm.get_integer("number mesh adapt");
268 override_time_step_on_restart =
269 prm.get_bool("override time step on restart");
270
271 output_folder = prm.get("output path");
272 output_name = prm.get("output name");
273 std::erase(output_name, '/');
274 output_iteration_frequency = prm.get_integer("output frequency");
275 output_time_frequency = prm.get_double("output time frequency");
276 output_times_vector =
277 convert_string_to_vector<double>(prm, "output times");
278 output_time_interval =
279 convert_string_to_vector<double>(prm, "output time interval");
280 output_boundaries = prm.get_bool("output boundaries");
281
282 subdivision = prm.get_integer("subdivision");
283 group_files = prm.get_integer("group files");
284 log_frequency = prm.get_integer("log frequency");
285 log_precision = prm.get_integer("log precision");
286 }
287 prm.leave_subsection();
288 } // namespace Parameters

567 const Dimensionality &dimensions)
568 {
569 prm.enter_subsection("isothermal_ideal_gas");
570 {
571 // Isothermal ideal gas equation of state parameters
572 // The reference state density of the gas (rho_{ref}) is in M L^-3
573 density_ref = prm.get_double("density_ref");
574 density_ref *= dimensions.density_scaling;
575
576 // The specific gas constant (R) is in L^2 T^-2 theta^-1
577 R = prm.get_double("R");
578 R *= dimensions.specific_gas_constant_scaling;

581 T = prm.get_double("T");
582 T *= 1. / dimensions.temperature;
583 }

617 void
618 SurfaceTensionParameters::parse_parameters(
619 const ParameterHandler &prm,
620 const Parameters::Dimensionality &dimensions)
621 {
622 surface_tension_coefficient = prm.get_double("surface tension coefficient");
623 surface_tension_coefficient *= dimensions.surface_tension_scaling;
624 T_0 = prm.get_double("reference state temperature");
625 T_0 *= 1. / dimensions.temperature;
626 surface_tension_gradient =
627 prm.get_double("temperature-driven surface tension gradient");
628 surface_tension_gradient *= dimensions.surface_tension_gradient_scaling;
629 T_solidus = prm.get_double("solidus temperature");
630 T_solidus *= 1. / dimensions.temperature;
631 T_liquidus = prm.get_double("liquidus temperature");
632 T_liquidus *= 1. / dimensions.temperature;
633 Assert(T_liquidus > T_solidus,
634 PhaseChangeIntervalError(T_liquidus, T_solidus));
635 }

1275 {

1277 Utilities::int_to_string(id, 1));
1278 {
1279 // String that will be used to parse the models
1280 std::string op;
1281
1282 //---------------------------------------------------
1283 // Density
1284 //---------------------------------------------------
1285 op = prm.get("density model");
1286 if (op == "constant")
1287 {
1288 density_model = DensityModel::constant;
1289 }
1290 else if (op == "isothermal_ideal_gas")
1291 {
1292 density_model = DensityModel::isothermal_ideal_gas;
1293 }
1294 density = prm.get_double("density");
1295 // Density is in M L^-3, rescale
1296 density *= dimensions.density_scaling;
1297 isothermal_ideal_gas_density_parameters.parse_parameters(prm, dimensions);
1298
1299 //---------------------------------------------------
1300 // Kinematic viscosity and Rheology
1301 //---------------------------------------------------
1302 op = prm.get("rheological model");
1303 if (op == "power-law")
1304 {
1305 rheological_model = RheologicalModel::powerlaw;
1306 }
1307 else if (op == "carreau")
1308 {
1309 rheological_model = RheologicalModel::carreau;
1310 }
1311 else if (op == "newtonian")
1312 {
1313 rheological_model = RheologicalModel::newtonian;
1314 }
1315 else if (op == "phase_change")
1316 {
1317 rheological_model = RheologicalModel::phase_change;
1318 }
1319
1320 kinematic_viscosity = prm.get_double("kinematic viscosity");
1321 // Kinematic viscosity is in L^2 T^-1, rescale
1322 kinematic_viscosity *= dimensions.viscosity_scaling;
1323 non_newtonian_parameters.parse_parameters(prm, dimensions);
1324
1325 //--------------
1326 // Specific heat
1327 //--------------
1328 op = prm.get("specific heat model");
1329 if (op == "constant")
1330 specific_heat_model = SpecificHeatModel::constant;
1331 else if (op == "phase_change")
1332 specific_heat_model = SpecificHeatModel::phase_change;
1333 specific_heat = prm.get_double("specific heat");
1334
1335 // specific heat is in L^2 T^-2 theta^-1
1336 specific_heat *= dimensions.specific_heat_scaling;
1337
1338
1339 //----------------------
1340 // Thermal conductivity
1341 //----------------------
1342 op = prm.get("thermal conductivity model");
1343 if (op == "constant")
1344 thermal_conductivity_model = ThermalConductivityModel::constant;
1345 else if (op == "linear")
1346 thermal_conductivity_model = ThermalConductivityModel::linear;
1347 else if (op == "phase_change")
1348 thermal_conductivity_model = ThermalConductivityModel::phase_change;
1349
1350 thermal_conductivity = prm.get_double("thermal conductivity");
1351 // thermal conductivity is in M L T^-3 theta^-1
1352 thermal_conductivity *= dimensions.thermal_conductivity_scaling;
1353
1354
1355 // Linear conductivity model parameters
1356 k_A0 = prm.get_double("k_A0");
1357 // k_A0 is in M L T^-3 theta^-1
1358 k_A0 *= dimensions.thermal_conductivity_scaling;
1359
1360 k_A1 = prm.get_double("k_A1");
1361 // k_A1 is in M L T^-3 theta ^-2
1362 k_A1 *= dimensions.thermal_conductivity_scaling * dimensions.temperature;
1363
1364
1365 //------------------
1366 // Thermal expansion
1367 //------------------
1368 op = prm.get("thermal expansion model");
1369 if (op == "constant")
1370 thermal_expansion_model = ThermalExpansionModel::constant;
1371 else if (op == "phase_change")
1372 thermal_expansion_model = ThermalExpansionModel::phase_change;
1373
1374 thermal_expansion = prm.get_double("thermal expansion");
1375 // thermal expansion is in theta^-1
1376 thermal_expansion *= dimensions.temperature;
1377
1378 //-------------------
1379 // Tracer diffusivity
1380 //-------------------
1381 op = prm.get("tracer diffusivity model");
1382 if (op == "immersed solid tanh")
1383 tracer_diffusivity_model =
1384 TracerDiffusivityModel::immersed_boundary_tanh;
1385 else if (op == "immersed solid gaussian")
1386 tracer_diffusivity_model =
1387 TracerDiffusivityModel::immersed_boundary_gaussian;
1388 else
1389 tracer_diffusivity_model = TracerDiffusivityModel::constant;
1390 tracer_diffusivity = prm.get_double("tracer diffusivity");
1391 // Diffusivity is in L^2 T^-1
1392 tracer_diffusivity *= dimensions.diffusivity_scaling;
1393
1394 //-------------------
1395 // Tracer reaction constant
1396 //-------------------
1397 op = prm.get("tracer reaction constant model");
1398 if (op == "none")
1399 tracer_reaction_prefactor_model = TracerReactionPrefactorModel::none;
1400 else if (op == "immersed solid tanh")
1401 tracer_reaction_prefactor_model =
1402 TracerReactionPrefactorModel::immersed_boundary_tanh;
1403 else if (op == "immersed solid gaussian")
1404 tracer_reaction_prefactor_model =
1405 TracerReactionPrefactorModel::immersed_boundary_gaussian;
1406 else
1407 tracer_reaction_prefactor_model =
1408 TracerReactionPrefactorModel::constant;
1409 tracer_reaction_constant = prm.get_double("tracer reaction constant");
1410 tracer_reaction_order = prm.get_double("tracer reaction order");
1411 tracer_reaction_threshold = prm.get_double("tracer reaction threshold");
1412
1413 // Parsing of the immersed solids models parameters
1414 immersed_solid_tanh_parameters.parse_parameters(prm, dimensions);
1415 immersed_solid_gaussian_parameters.parse_parameters(prm, dimensions);
1416
1417 //--------------------------------
1418 // Phase change properties
1419 //--------------------------------
1420 phase_change_parameters.parse_parameters(prm, dimensions);
1421 }
