205 void
206 SimulationControl::parse_parameters(ParameterHandler &prm)
207 {
208 prm.enter_subsection("simulation control");
209 {
210 const std::string sv = prm.get("method");
211 if (sv == "steady")
212 method = TimeSteppingMethod::steady;
213 else if (sv == "steady_bdf")
214 method = TimeSteppingMethod::steady_bdf;
215 else if (sv == "bdf1")
216 method = TimeSteppingMethod::bdf1;
217 else if (sv == "bdf2")
218 method = TimeSteppingMethod::bdf2;
219 else if (sv == "bdf3")
220 method = TimeSteppingMethod::bdf3;
221 else if (sv == "sdirk22")
222 method = TimeSteppingMethod::sdirk22;
223 else if (sv == "sdirk33")
224 method = TimeSteppingMethod::sdirk33;
225 else if (sv == "sdirk43")
226 method = TimeSteppingMethod::sdirk43;
227 else
228 {
229 std::runtime_error("Invalid time stepping scheme");
230 }
231 const std::string bdf_startup_string = prm.get("bdf startup method");
232 if (bdf_startup_string == "multiple step bdf")
233 bdf_startup_method = BDFStartupMethods::multiple_step_bdf;
234 else if (bdf_startup_string == "initial solution")
235 bdf_startup_method = BDFStartupMethods::initial_solution;
236 else
237 {
238 std::runtime_error("Invalid bdf startup scheme");
239 }
240
241 const std::string osv = prm.get("output control");
242 if (osv == "iteration")
243 output_control = OutputControl::iteration;
244 else if (osv == "time")
245 output_control = OutputControl::time;
246 else
247 {
248 std::runtime_error("Invalid output control scheme");
249 }
250 dt = prm.get_double("time step");
251 time_end = prm.get_double("time end");
252 adapt = prm.get_bool("adapt");
253 time_step_independent_of_end_time =
254 prm.get_bool("time step independent of end time");
255 maxCFL = prm.get_double("max cfl");
256 max_dt = prm.get_double("max time step");
257 stop_tolerance = prm.get_double("stop tolerance");
258 adaptative_time_step_scaling =
259 prm.get_double("adaptative time step scaling");
260 startup_timestep_scaling = prm.get_double("startup time scaling");
261 number_mesh_adaptation = prm.get_integer("number mesh adapt");
262
263 output_folder = prm.get("output path");
264 output_name = prm.get("output name");
265 output_name.erase(std::remove(output_name.begin(),
266 output_name.end(),
267 '/'),
268 output_name.end());
269 output_iteration_frequency = prm.get_integer("output frequency");
270 output_time_frequency = prm.get_double("output time frequency");
271 output_times_vector =
272 convert_string_to_vector<double>(prm, "output times");
273 output_time_interval =
274 convert_string_to_vector<double>(prm, "output time interval");
275 output_boundaries = prm.get_bool("output boundaries");
276
277 subdivision = prm.get_integer("subdivision");
278 group_files = prm.get_integer("group files");
279 log_frequency = prm.get_integer("log frequency");
280 log_precision = prm.get_integer("log precision");
281 }
282 prm.leave_subsection();
283 } // namespace Parameters

507 ParameterHandler &prm,
508 const Dimensionality dimensions)
509 {
510 prm.enter_subsection("isothermal_ideal_gas");
511 {
512 // Isothermal ideal gas equation of state parameters
513 // The reference state density of the gas (rho_{ref}) is in M L^-3
514 density_ref = prm.get_double("density_ref");
515 density_ref *= dimensions.density_scaling;
516

522 T = prm.get_double("T");

524 }

558 void
559 SurfaceTensionParameters::parse_parameters(
560 ParameterHandler &prm,
561 const Parameters::Dimensionality dimensions)
562 {
563 surface_tension_coefficient = prm.get_double("surface tension coefficient");
564 surface_tension_coefficient *= dimensions.surface_tension_scaling;
565 T_0 = prm.get_double("reference state temperature");
566 T_0 *= 1. / dimensions.temperature;
567 surface_tension_gradient =
568 prm.get_double("temperature-driven surface tension gradient");
569 surface_tension_gradient *= dimensions.surface_tension_gradient_scaling;
570 T_solidus = prm.get_double("solidus temperature");
571 T_solidus *= 1. / dimensions.temperature;
572 T_liquidus = prm.get_double("liquidus temperature");
573 T_liquidus *= 1. / dimensions.temperature;
574 Assert(T_liquidus > T_solidus,
575 PhaseChangeIntervalError(T_liquidus, T_solidus));
576 }
