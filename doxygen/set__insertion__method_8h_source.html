14/**
15 * @brief Set up the particle insertion method for the simulation.
16 * Creates and configures the appropriate insertion object based on the
17 * simulation parameters (e.g., volume insertion, plane insertion, etc.).
18 *
19 * @param size_distribution_object_container Contains all distribution for each
20 * particle type.
21 * @param triangulation Triangulation to access the cells in which the
22 * particles will be inserted.
23 * @param dem_parameters DEM parameters declared in the .prm file.
24 * @param maximum_particle_diameter Maximum particle diameter based on values
25 * defined in the parameter handler.
26 *
27 * @return Shared pointer to the configured insertion object
28 */
29template <int dim, typename PropertiesIndex>
30std::shared_ptr<Insertion<dim, PropertiesIndex>>
31set_insertion_type(std::vector<std::shared_ptr<Distribution>>
32 &size_distribution_object_container,
33 parallel::distributed::Triangulation<dim> &triangulation,
34 DEMSolverParameters<dim> &dem_parameters,
35 const double &maximum_particle_diameter)
36{
37 using namespace Parameters::Lagrangian;
38 typename InsertionInfo<dim>::InsertionMethod insertion_method =
39 dem_parameters.insertion_info.insertion_method;
40
41 switch (insertion_method)
42 {
43 case InsertionInfo<dim>::InsertionMethod::file:
44 return std::make_shared<InsertionFile<dim, PropertiesIndex>>(
45 size_distribution_object_container, triangulation, dem_parameters);
46 case InsertionInfo<dim>::InsertionMethod::list:
47 return std::make_shared<InsertionList<dim, PropertiesIndex>>(
48 size_distribution_object_container, triangulation, dem_parameters);
49 case InsertionInfo<dim>::InsertionMethod::plane:
50 return std::make_shared<InsertionPlane<dim, PropertiesIndex>>(
51 size_distribution_object_container, triangulation, dem_parameters);
52 case InsertionInfo<dim>::InsertionMethod::volume:
53 return std::make_shared<InsertionVolume<dim, PropertiesIndex>>(
54 size_distribution_object_container,
55 triangulation,
56 dem_parameters,
57 maximum_particle_diameter);
58 default:
59 AssertThrow(false, ExcMessage("Invalid insertion method."));
60 }
61}
