9template <int dim, typename PropertiesIndex>
10Insertion<dim, PropertiesIndex>::Insertion(
11 const std::vector<std::shared_ptr<Distribution>>
12 &size_distribution_object_container,
13 const parallel::distributed::Triangulation<dim> &triangulation,
14 const DEMSolverParameters<dim> &dem_parameters)
15 : removing_particles_in_region(
16 dem_parameters.insertion_info.removing_particles_in_region)
17{
18 distributions_objects = size_distribution_object_container;
19
20 // Boost signal for load balancing
21 this->mark_for_update = true;
22 this->change_to_triangulation = triangulation.signals.any_change.connect(
23 [&] { this->mark_for_update = true; });
24
25 for (unsigned int i = 0; i < dim; ++i)
26 {
27 if (dem_parameters.insertion_info.clear_box_point_1[i] <=
28 dem_parameters.insertion_info.clear_box_point_2[i])
29 {
30 p_min[i] = dem_parameters.insertion_info.clear_box_point_1[i];
31 p_max[i] = dem_parameters.insertion_info.clear_box_point_2[i];
32 }
33 else
34 {
35 p_min[i] = dem_parameters.insertion_info.clear_box_point_2[i];
36 p_max[i] = dem_parameters.insertion_info.clear_box_point_1[i];
37 }
38 }
39}

59// Carries out assigning the properties of inserted particles. The output vector
60// is used in insert_global_particles as input argument
61template <int dim, typename PropertiesIndex>
62void
63Insertion<dim, PropertiesIndex>::assign_particle_properties(
64 const DEMSolverParameters<dim> &dem_parameters,
65 const unsigned int &inserted_this_step_this_proc,
66 const unsigned int &current_inserting_particle_type,
67 const std::vector<Point<dim>> &insertion_points,
68 std::vector<std::vector<double>> &particle_properties)
69{
70 // Clearing and resizing particle_properties
71 particle_properties.reserve(inserted_this_step_this_proc);
72
73 // Getting properties as local parameters
74 // TODO: MAYBE CHANGE THE INPUT TO PHYSICAL PROPERTIES DIRECTLY
75 auto physical_properties = dem_parameters.lagrangian_physical_properties;
76
77 distributions_objects[current_inserting_particle_type]
78 ->particle_size_sampling(inserted_this_step_this_proc);
79
80 // A loop is defined over the number of particles which are going to be
81 // inserted at this step
82 for (unsigned int particle_counter = 0;
83 particle_counter < inserted_this_step_this_proc;
84 ++particle_counter)
85 {
86 double type = current_inserting_particle_type;
87 // We make sure that the diameter is positive
88 double diameter =
89 std::abs(distributions_objects[current_inserting_particle_type]
90 ->particle_sizes[particle_counter]);
91 double density =
92 physical_properties.density_particle[current_inserting_particle_type];
93 double vel_x = dem_parameters.insertion_info.initial_vel[0];
94 double vel_y = dem_parameters.insertion_info.initial_vel[1];
95 double vel_z = dem_parameters.insertion_info.initial_vel[2];
96 double omega_x = dem_parameters.insertion_info.initial_omega[0];
97 double omega_y = dem_parameters.insertion_info.initial_omega[1];
98 double omega_z = dem_parameters.insertion_info.initial_omega[2];
99 double mass = density * 4. / 3. * M_PI *
100 Utilities::fixed_power<3, double>(diameter * 0.5);
101
102 std::vector<double> properties_of_one_particle{
103 type, diameter, mass, vel_x, vel_y, vel_z, omega_x, omega_y, omega_z};
104
105 if constexpr (std::is_same_v<PropertiesIndex,
106 DEM::DEMMPProperties::PropertiesIndex>)
107 {
108 double T =
109 dem_parameters.insertion_info.initial_temperature_function->value(
110 insertion_points[particle_counter]);
111 double specific_heat =
112 physical_properties
113 .specific_heat_particle[current_inserting_particle_type];
114 properties_of_one_particle.push_back(T);
115 properties_of_one_particle.push_back(specific_heat);
116 }
117
118 if constexpr (std::is_same_v<PropertiesIndex,
119 DEM::CFDDEMProperties::PropertiesIndex>)
120 {
121 // Push back all zero variables for the CFD-DEM coupling properties
122 for (unsigned int i = properties_of_one_particle.size();
123 i < PropertiesIndex::n_properties;
124 ++i)
125 properties_of_one_particle.push_back(0.);
126 }
127
128 particle_properties.push_back(properties_of_one_particle);
129 properties_of_one_particle.clear();
130 }
131}

133template <int dim, typename PropertiesIndex>
134void
135Insertion<dim, PropertiesIndex>::
136 calculate_insertion_domain_maximum_particle_number(
137 const DEMSolverParameters<dim> &dem_parameters,
138 const ConditionalOStream &pcout)
139{
140 // Getting properties as local parameters
141 const auto insertion_information = dem_parameters.insertion_info;
142
143 // Checking if the insertion directions are valid (no repetition)
144 int axis_sum = 0;
145 if constexpr (dim == 2)
146 {
147 axis_sum = insertion_information.direction_sequence[0] +
148 insertion_information.direction_sequence[1];
149
150 AssertThrow(
151 axis_sum == 1,
152 ExcMessage("Invalid insertion directions: 2 directions are the same "));
153 }
154 if constexpr (dim == 3)
155 {
156 axis_sum = insertion_information.direction_sequence[0] +
157 insertion_information.direction_sequence[1] +
158 insertion_information.direction_sequence[2];
159
160 AssertThrow(
161 axis_sum == 3,
162 ExcMessage(
163 "Invalid insertion directions: at least 2 directions are the same "));
164 }
165
166
167 // This variable is used for calculation of the maximum number of particles
168 // that can fit in the chosen insertion box
169 int maximum_particle_number = 1;
170
171 number_of_particles_directions.resize(dim);
172 axis_min.resize(dim);
173 axis_max.resize(dim);
174
175 std::vector<unsigned int> axis_list = {
176 insertion_information.direction_sequence[0],
177 insertion_information.direction_sequence[1]};
178
179 if constexpr (dim == 3)
180 {
181 axis_list.push_back(insertion_information.direction_sequence[2]);
182 }
183
184 // Assigning the minimum and maximum positions of the insertion box in respect
185 // to the axis order
186 for (unsigned int axis : axis_list)
187 {
188 switch (axis)
189 {
190 case 0:
191 axis_min[0] = insertion_information.insertion_box_point_1(0);
192 axis_max[0] = insertion_information.insertion_box_point_2(0);
193 break;
194 case 1:
195 axis_min[1] = insertion_information.insertion_box_point_1(1);
196 axis_max[1] = insertion_information.insertion_box_point_2(1);
197 break;
198 case 2:
199 axis_min[2] = insertion_information.insertion_box_point_1(2);
200 axis_max[2] = insertion_information.insertion_box_point_2(2);
201 break;
202 default:
203 AssertThrow(false,
204 ExcMessage("Insertion direction must be 0, 1 or 2"));
205 }
206
207 // Assign max number of particles according to the direction and calculate
208 // the total max number (maximum_particle_number = max_x * max_y * max_z)
209 int number_of_particles = static_cast<int>(
210 (axis_max[axis] - axis_min[axis]) /
211 (insertion_information.distance_threshold * this->maximum_diameter));
212 number_of_particles_directions[axis] = number_of_particles;
213 maximum_particle_number *= number_of_particles;
214 }
215
216 // If the inserted number of particles at this step exceeds the maximum
217 // number, a warning is printed
218 if (insertion_information.inserted_this_step > maximum_particle_number)
219 {
220 pcout << "Warning: the requested number of particles for insertion ("
221 << insertion_information.inserted_this_step
222 << ") is higher than maximum expected number of particles ("
223 << maximum_particle_number << ")" << std::endl;
224
225 // Updating the number of inserted particles at each step
226 inserted_this_step = maximum_particle_number;
227 }
228 else
229 {
230 inserted_this_step = insertion_information.inserted_this_step;
231 }
232}

235template <int dim, typename PropertiesIndex>
236void
237Insertion<dim, PropertiesIndex>::find_cells_in_removing_box(
238 const parallel::distributed::Triangulation<dim> &triangulation)
239{
240 // Clearing the containers
241 in_removal_box.clear();
242 edge_of_removal_box.clear();
243 bool partially_inside, completely_inside;
244 // Looping through cells
245 for (const auto &cell : triangulation.active_cell_iterators())
246 {
247 if (cell->is_locally_owned())
248 {
249 partially_inside = false;
250 completely_inside = true;
251 // Loop through its vertices
252 for (unsigned int vertex_id = 0; vertex_id < cell->n_vertices();
253 ++vertex_id)
254 {
255 Point<3> vertex = point_nd_to_3d(cell->vertex(vertex_id));
256
257 // Check if the n_th vertex is in the box
258 if (p_min[0] <= vertex[0] && vertex[0] <= p_max[0] &&
259 p_min[1] <= vertex[1] && vertex[1] <= p_max[1] &&
260 p_min[2] <= vertex[2] && vertex[2] <= p_max[2])
261 {
262 partially_inside = true;
263 }
264 else
265 {
266 completely_inside = false;
267 }
268 }
269 // If the cell is completely inside the clearing box, both bool will
270 // be true. We need to add the cell iterator to the
271 // in_removal_box container.
272
273 // If the cell is partially inside, at least one of the "if" will have
274 // failed, thus the second bool will be false but the first one true.
275 // We need to add the cell iterator to the edge_of_removal_box
276 // container.
277
278 // If the cell is outside, both bool will be false, and we do nothing.
279 if (completely_inside)
280 in_removal_box.insert(cell);
281 else if (!completely_inside && partially_inside)
282 edge_of_removal_box.insert(cell);
283 }
284 }
285}

287template <int dim, typename PropertiesIndex>
288void
289Insertion<dim, PropertiesIndex>::remove_particles_in_box(
290 Particles::ParticleHandler<dim> &particle_handler)
291{
292 std::vector<
293 typename dealii::Particles::ParticleHandler<dim>::particle_iterator>
294 to_remove_iterators;
295
296 // Reserve to the maximum number of particle on this processor.
297 to_remove_iterators.reserve(particle_handler.n_locally_owned_particles());
298
299 // Loop over the first container
300 for (const auto &cell_in_box : in_removal_box)
301 {
302 // Check if this cell has particles
303 auto particles_in_cell = particle_handler.particles_in_cell(cell_in_box);
304
305 if (!particles_in_cell.empty())
306 {
307 // Loop over the particle in the cell
308 for (auto particle_in_cell = particles_in_cell.begin();
309 particle_in_cell != particles_in_cell.end();
310 ++particle_in_cell)
311 {
312 // Since we know the cell is fully inside the box, we can
313 // remove every particles in it.
314
315 to_remove_iterators.push_back(particle_in_cell);
316 }
317 }
318 }
319
320 // Loop over the second container
321 for (auto cell_edge_of_removal_box = edge_of_removal_box.begin();
322 cell_edge_of_removal_box != edge_of_removal_box.end();
323 ++cell_edge_of_removal_box)
324 {
325 // Check if this cell has particle
326 auto particles_in_cell =
327 particle_handler.particles_in_cell(*cell_edge_of_removal_box);
328 const bool particles_exist_in_cell = !particles_in_cell.empty();
329
330 if (particles_exist_in_cell)
331 {
332 // Loop over the particles in the cell
333 for (auto particle_in_cell = particles_in_cell.begin();
334 particle_in_cell != particles_in_cell.end();
335 ++particle_in_cell)
336 {
337 // We need to check if the particle is inside since the cell
338 // is at the edge.
339 Point<3> particle_position =
340 point_nd_to_3d(particle_in_cell->get_location());
341 if (p_min[0] <= particle_position[0] &&
342 p_max[0] >= particle_position[0] &&
343 p_min[1] <= particle_position[1] &&
344 p_max[1] >= particle_position[1] &&
345 p_min[2] <= particle_position[2] &&
346 p_max[2] >= particle_position[2])
347 {
348 to_remove_iterators.push_back(particle_in_cell);
349 }
350 }
351 }
352 }
353 particle_handler.remove_particles(to_remove_iterators);
354}
