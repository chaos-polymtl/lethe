28/**
29 * @brief Evaluate the value of a function at a batch of points to obtain a vectorized array of numbers
30 *
31 * @tparam dim An integer that denotes the number of spatial dimensions.
32 * @tparam Number Abstract type for number across the class (i.e., double).
33 * @param function Function to evaluate.
34 * @param p_vectorized Batch of points to evaluate function at.
35 * @return VectorizedArray<Number> Batch of evaluated values.
36 */
37template <int dim, typename Number>
38VectorizedArray<Number>
39evaluate_function(const Function<dim> &function,
40 const Point<dim, VectorizedArray<Number>> &p_vectorized)
41{
42 VectorizedArray<Number> result;
43 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
44 {
45 Point<dim> p;
46 for (unsigned int d = 0; d < dim; ++d)
47 p[d] = p_vectorized[d][v];
48 result[v] = function.value(p);
49 }
50 return result;
51}

53/**
54 * @brief Evaluate the gradient of a function at a batch of points to obtain a tensor of vectorized arrays
55 *
56 * @tparam dim An integer that denotes the number of spatial dimensions.
57 * @tparam Number Abstract type for number across the class (i.e., double).
58 * @param function Function to evaluate.
59 * @param p_vectorized Batch of points to evaluate function at.
60 * @return Tensor<1, components, VectorizedArray<Number>> Batch of evaluated gradients.
61 */
62template <int dim, typename Number>
63Tensor<1, dim, VectorizedArray<Number>>
64evaluate_function_gradient(
65 const Function<dim> &function,
66 const Point<dim, VectorizedArray<Number>> &p_vectorized)
67{
68 Tensor<1, dim, VectorizedArray<Number>> result;
69 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
70 {
71 Point<dim> p;
72 for (unsigned int d = 0; d < dim; ++d)
73 p[d] = p_vectorized[d][v];
74
75 Tensor<1, dim> gradient = function.gradient(p);
76 for (unsigned int d = 0; d < dim; ++d)
77 result[d][v] = gradient[d];
78 }
79 return result;
80}

82/**
83 * @brief Evaluate the value of a function at a batch of points to obtain a tensor of vectorized arrays
84 *
85 * @tparam dim An integer that denotes the number of spatial dimensions.
86 * @tparam Number Abstract type for number across the class (i.e., double).
87 * @tparam components Number of solution components.
88 * @param function Function to evaluate.
89 * @param p_vectorized Batch of points to evaluate function at.
90 * @return Tensor<1, components, VectorizedArray<Number>> Batch of evaluated values.
91 */
92template <int dim, typename Number, int components>
93Tensor<1, components, VectorizedArray<Number>>
94evaluate_function(const Function<dim> &function,
95 const Point<dim, VectorizedArray<Number>> &p_vectorized)
96{
97 Tensor<1, components, VectorizedArray<Number>> result;
98 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
99 {
100 Point<dim> p;
101 for (unsigned int d = 0; d < dim; ++d)
102 p[d] = p_vectorized[d][v];
103 for (unsigned int d = 0; d < components; ++d)
104 result[d][v] = function.value(p, d);
105 }
106 return result;
107}
