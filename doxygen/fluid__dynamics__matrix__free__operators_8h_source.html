51/**
52 * @brief Evaluate the value of a function at a batch of points to obtain a vectorized array of numbers
53 *
54 * @tparam dim An integer that denotes the number of spatial dimensions.
55 * @tparam Number Abstract type for number across the class (i.e., double).
56 * @param function Function to evaluate.
57 * @param p_vectorized Batch of points to evaluate function at.
58 * @return VectorizedArray<Number> Batch of evaluated values.
59 */
60template <int dim, typename Number>
61VectorizedArray<Number>
62evaluate_function(const Function<dim> &function,
63 const Point<dim, VectorizedArray<Number>> &p_vectorized)
64{
65 VectorizedArray<Number> result;
66 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
67 {
68 Point<dim> p;
69 for (int d = 0; d < dim; ++d)
70 p[d] = p_vectorized[d][v];
71 result[v] = function.value(p);
72 }
73 return result;
74}

76/**
77 * @brief Evaluate the gradient of a function at a batch of points to obtain a tensor of vectorized arrays
78 *
79 * @tparam dim An integer that denotes the number of spatial dimensions.
80 * @tparam Number Abstract type for number across the class (i.e., double).
81 * @param function Function to evaluate.
82 * @param p_vectorized Batch of points to evaluate function at.
83 * @return Tensor<1, components, VectorizedArray<Number>> Batch of evaluated gradients.
84 */
85template <int dim, typename Number>
86Tensor<1, dim, VectorizedArray<Number>>
87evaluate_function_gradient(
88 const Function<dim> &function,
89 const Point<dim, VectorizedArray<Number>> &p_vectorized)
90{
91 Tensor<1, dim, VectorizedArray<Number>> result;
92 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
93 {
94 Point<dim> p;
95 for (int d = 0; d < dim; ++d)
96 p[d] = p_vectorized[d][v];
97
98 Tensor<1, dim> gradient = function.gradient(p);
99 for (int d = 0; d < dim; ++d)
100 result[d][v] = gradient[d];
101 }
102 return result;
103}

105/**
106 * @brief Evaluate the value of a function at a batch of points to obtain a tensor of vectorized arrays
107 *
108 * @tparam dim An integer that denotes the number of spatial dimensions.
109 * @tparam Number Abstract type for number across the class (i.e., double).
110 * @tparam components Number of solution components.
111 * @param function Function to evaluate.
112 * @param p_vectorized Batch of points to evaluate function at.
113 * @return Tensor<1, components, VectorizedArray<Number>> Batch of evaluated values.
114 */
115template <int dim, typename Number, int components>
116Tensor<1, components, VectorizedArray<Number>>
117evaluate_function(const Function<dim> &function,
118 const Point<dim, VectorizedArray<Number>> &p_vectorized)
119{
120 Tensor<1, components, VectorizedArray<Number>> result;
121 for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
122 {
123 Point<dim> p;
124 for (int d = 0; d < dim; ++d)
125 p[d] = p_vectorized[d][v];
126 for (unsigned int d = 0; d < components; ++d)
127 result[d][v] = function.value(p, d);
128 }
129 return result;
130}
