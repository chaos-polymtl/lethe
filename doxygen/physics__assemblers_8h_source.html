7/**
8 * @brief Base class for assembler objects of physics and subequations.
9 *
10 * @tparam ScratchDataType Type of scratch data object used for linear system
11 * assembly.
12 *
13 * @tparam CopyDataType Type of copy data object used for transferring from
14 * local system assembly to global system.
15 */
16template <typename ScratchDataType, typename CopyDataType>
17class PhysicsAssemblerBase
18{
19public:
20 /**
21 * @brief Assemble the matrix.
22 *
23 * @param[in] scratch_data Scratch data containing the information required
24 * for system assembly.
25 * It is important to note that the scratch data has to have been re-inited
26 * before calling for matrix assembly.
27 *
28 * @param[in,out] copy_data Destination where the local_matrix is copied to.
29 */
30 virtual void
31 assemble_matrix(const ScratchDataType &scratch_data,
32 CopyDataType &copy_data) = 0;
33
34 /**
35 * @brief Assemble the right-hand side (rhs).
36 *
37 * @param[in] scratch_data Scratch data containing the information required
38 * for system assembly.
39 * It is important to note that the scratch data has to have been re-inited
40 * before calling for rhs assembly.
41 *
42 * @param[in,out] copy_data Destination where the local_rhs is copied to.
43 */
44 virtual void
45 assemble_rhs(const ScratchDataType &scratch_data,
46 CopyDataType &copy_data) = 0;
47};
