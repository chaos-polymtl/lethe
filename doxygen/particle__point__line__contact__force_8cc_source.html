15// In this function, the particle-point and particle-line contact forces are
16// calculated and the particle handler is updated based on this force
17template <int dim, typename PropertiesIndex>
18void
19ParticlePointLineForce<dim, PropertiesIndex>::
20 calculate_particle_point_contact_force(
21 const typename DEM::dem_data_structures<dim>::particle_point_in_contact
22 *particle_point_pairs_in_contact,
23 const Parameters::Lagrangian::LagrangianPhysicalProperties
24 &physical_properties,
25 std::vector<Tensor<1, 3>> &force)
26
27{
28 // Looping over particle_point_line_pairs_in_contact
29 for (auto pairs_in_contact_iterator =
30 particle_point_pairs_in_contact->begin();
31 pairs_in_contact_iterator != particle_point_pairs_in_contact->end();
32 ++pairs_in_contact_iterator)
33 {
34 // Getting the required information for calculation of contact force from
35 // the map
36 auto contact_information = &pairs_in_contact_iterator->second;
37
38 // Defining particle and properties of particle as local parameters
39 auto particle = contact_information->particle;
40 auto particle_properties = particle->get_properties();
41 Point<3> particle_location_3d;
42
43 if constexpr (dim == 3)
44 particle_location_3d = particle->get_location();
45
46 if constexpr (dim == 2)
47 particle_location_3d = point_nd_to_3d(particle->get_location());
48
49 const Point<3> point = contact_information->point;
50 double normal_overlap = ((particle_properties[PropertiesIndex::dp]) / 2) -
51 point.distance(particle_location_3d);
52
53 if (normal_overlap > 0)
54 {
55 // Calculation of normal vector, particle velocity and contact
56 // relative velocity
57 Tensor<1, 3> point_to_particle_vector = particle_location_3d - point;
58 Tensor<1, 3> normal_vector =
59 point_to_particle_vector / point_to_particle_vector.norm();
60
61 Tensor<1, 3> particle_velocity;
62 particle_velocity[0] = particle_properties[PropertiesIndex::v_x];
63 particle_velocity[1] = particle_properties[PropertiesIndex::v_y];
64 particle_velocity[2] = particle_properties[PropertiesIndex::v_z];
65
66 Tensor<1, 3> particle_omega;
67 particle_omega[0] = particle_properties[PropertiesIndex::omega_x];
68 particle_omega[1] = particle_properties[PropertiesIndex::omega_y];
69 particle_omega[2] = particle_properties[PropertiesIndex::omega_z];
70
71 // Defining relative contact velocity
72 Tensor<1, 3> contact_relative_velocity =
73 particle_velocity +
74 cross_product_3d((((particle_properties[PropertiesIndex::dp]) / 2) *
75 particle_omega),
76 normal_vector);
77
78 // Calculation of normal relative velocity
79 double normal_relative_velocity =
80 contact_relative_velocity * normal_vector;
81
82 // Calculation of effective Young's modulus of the contact
83 double effective_youngs_modulus =
84 physical_properties.youngs_modulus_wall /
85 (2 * (1 - pow(physical_properties.poisson_ratio_wall, 2)));
86
87 // Calculation of model parameters (beta and sn). These values
88 // are used to consider non-linear relation of the contact force to
89 // the normal overlap
90 double model_parameter_beta =
91 log(physical_properties.restitution_coefficient_wall) /
92 sqrt(pow(log(physical_properties.restitution_coefficient_wall), 2) +
93 9.8696);
94 double model_parameter_sn =
95 2 * effective_youngs_modulus *
96 sqrt(particle_properties[PropertiesIndex::dp] * normal_overlap);
97
98 // Calculation of normal spring and dashpot constants
99 // using particle and wall properties
100 double normal_spring_constant =
101 1.3333 * effective_youngs_modulus *
102 sqrt(particle_properties[PropertiesIndex::dp] / 2 * normal_overlap);
103 double normal_damping_constant =
104 -1.8257 * model_parameter_beta *
105 sqrt(model_parameter_sn *
106 particle_properties[PropertiesIndex::mass]);
107
108 // Calculation of normal force using spring and dashpot normal forces
109 Tensor<1, 3> spring_normal_force =
110 (normal_spring_constant * normal_overlap) * normal_vector;
111
112 Tensor<1, 3> dashpot_normal_force =
113 (normal_damping_constant * normal_relative_velocity) *
114 normal_vector;
115
116 // In particle-point and particle-line contacts, the tangential force
117 // is meaningless. So the total force is equal to the normal force
118 Tensor<1, 3> total_force = spring_normal_force - dashpot_normal_force;
119
120 // Getting force
121 Tensor<1, 3> &particle_force = force[particle->get_local_index()];
122
123 // Updating the body force of particles in the particle handler
124 particle_force += total_force;
125 }
126 }
127}

254template <int dim, typename PropertiesIndex>
255Point<3>
256ParticlePointLineForce<dim, PropertiesIndex>::find_projection_point(
257 const Point<3> &point_p,
258 const Point<3> &point_a,
259 const Point<3> &point_b)
260{
261 Tensor<1, 3> vector_ab = point_b - point_a;
262 Tensor<1, 3> vector_ap = point_p - point_a;
263
264 Point<3> projection =
265 point_a + ((vector_ap * vector_ab) / (vector_ab * vector_ab)) * vector_ab;
266
267 return projection;
268}
