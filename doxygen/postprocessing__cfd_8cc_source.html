39 template <int dim, typename VectorType>
40 std::pair<double, double>
41 calculate_pressure_drop(const DoFHandler<dim> &dof_handler,
42  const Mapping<dim> &mapping,
43  const VectorType &evaluation_point,
44  const Quadrature<dim> &cell_quadrature_formula,
45  const Quadrature<dim - 1> &face_quadrature_formula,
46  const unsigned int inlet_boundary_id,
47  const unsigned int outlet_boundary_id)
48 {
49  FESystem<dim, dim> fe = dof_handler.get_fe();
50  FEValues<dim> fe_values(mapping,
51  fe,
52  cell_quadrature_formula,
53  update_values | update_quadrature_points |
54  update_JxW_values | update_gradients |
55  update_hessians);
56  FEFaceValues<dim> fe_face_values(fe,
57  face_quadrature_formula,
58  update_values | update_quadrature_points |
59  update_normal_vectors | update_JxW_values);
60 
61  const FEValuesExtractors::Vector velocities(0);
62  const FEValuesExtractors::Scalar pressure(dim);
63 
64  const unsigned int face_n_q_points = face_quadrature_formula.size();
65 
66  std::vector<Tensor<1, dim>> velocity_values(face_n_q_points);
67  std::vector<double> pressure_values(face_n_q_points);
68 
69  double static_pressure_outlet_boundary = 0.;
70  double static_pressure_inlet_boundary = 0.;
71  double static_pressure_drop = 0.;
72 
73  double dynamic_pressure_outlet_boundary = 0.;
74  double dynamic_pressure_inlet_boundary = 0.;
75  double dynamic_pressure_drop = 0.;
76 
77  double outlet_surface = 0.;
78  double inlet_surface = 0.;
79 
80  double temp_surface = 0.;
81  double temp_static_pressure_at_boundary = 0.;
82  double temp_dynamic_pressure_at_boundary = 0.;
83 
84  for (const auto &cell : dof_handler.active_cell_iterators())
85  {
86  if (cell->is_locally_owned())
87  {
88  for (unsigned int face_id = 0;
89  face_id < GeometryInfo<dim>::faces_per_cell;
90  face_id++)
91  {
92  if (cell->face(face_id)->at_boundary())
93  {
94  unsigned int boundary_id = cell->face(face_id)->boundary_id();
95  if (boundary_id != inlet_boundary_id &&
96  boundary_id != outlet_boundary_id)
97  continue;
98  fe_face_values.reinit(cell, face_id);
99  fe_face_values[velocities].get_function_values(
100  evaluation_point, velocity_values);
101  fe_face_values[pressure].get_function_values(evaluation_point,
102  pressure_values);
103  temp_surface = 0.;
104  temp_static_pressure_at_boundary = 0.;
105  temp_dynamic_pressure_at_boundary = 0.;
106  for (unsigned int q = 0; q < face_n_q_points; q++)
107  {
108  temp_surface += fe_face_values.JxW(q);
109  // Integration of velocity^2 and
110  // pressure, since the pressure in Lethe has units of
111  // Length^2/Time^2
112  temp_static_pressure_at_boundary +=
113  fe_face_values.JxW(q) * pressure_values[q];
114  temp_dynamic_pressure_at_boundary +=
115  0.5 * fe_face_values.JxW(q) * velocity_values[q] *
116  velocity_values[q];
117  }
118  if (boundary_id == inlet_boundary_id)
119  {
120  inlet_surface += temp_surface;
121  dynamic_pressure_inlet_boundary +=
122  temp_dynamic_pressure_at_boundary;
123  static_pressure_inlet_boundary +=
124  temp_static_pressure_at_boundary;
125  }
126  if (boundary_id == outlet_boundary_id)
127  {
128  outlet_surface += temp_surface;
129  dynamic_pressure_outlet_boundary +=
130  temp_dynamic_pressure_at_boundary;
131  static_pressure_outlet_boundary +=
132  temp_static_pressure_at_boundary;
133  }
134  }
135  }
136  }
137  }
138 
139  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
140  static_pressure_inlet_boundary =
141  Utilities::MPI::sum(static_pressure_inlet_boundary, mpi_communicator);
142  static_pressure_outlet_boundary =
143  Utilities::MPI::sum(static_pressure_outlet_boundary, mpi_communicator);
144 
145  dynamic_pressure_inlet_boundary =
146  Utilities::MPI::sum(dynamic_pressure_inlet_boundary, mpi_communicator);
147  dynamic_pressure_outlet_boundary =
148  Utilities::MPI::sum(dynamic_pressure_outlet_boundary, mpi_communicator);
149 
150  inlet_surface = Utilities::MPI::sum(inlet_surface, mpi_communicator);
151  outlet_surface = Utilities::MPI::sum(outlet_surface, mpi_communicator);
152 
153  static_pressure_outlet_boundary =
154  static_pressure_outlet_boundary / outlet_surface;
155  static_pressure_inlet_boundary =
156  static_pressure_inlet_boundary / inlet_surface;
157  static_pressure_drop =
158  static_pressure_inlet_boundary - static_pressure_outlet_boundary;
159 
160  dynamic_pressure_outlet_boundary =
161  dynamic_pressure_outlet_boundary / outlet_surface;
162  dynamic_pressure_inlet_boundary =
163  dynamic_pressure_inlet_boundary / inlet_surface;
164  dynamic_pressure_drop =
165  dynamic_pressure_inlet_boundary - dynamic_pressure_outlet_boundary;
166 
167  return {static_pressure_drop, static_pressure_drop + dynamic_pressure_drop};
168 }

232 template <int dim, typename VectorType>
233 double
234 calculate_CFL(const DoFHandler<dim> &dof_handler,
235  const VectorType &evaluation_point,
236  const double time_step,
237  const Quadrature<dim> &quadrature_formula,
238  const Mapping<dim> &mapping)
239 {
240  FESystem<dim, dim> fe = dof_handler.get_fe();
241  FEValues<dim> fe_values(mapping,
242  fe,
243  quadrature_formula,
244  update_values | update_quadrature_points |
245  update_JxW_values);
246 
247  const FEValuesExtractors::Vector velocities(0);
248  const unsigned int n_q_points = quadrature_formula.size();
249 
250 
251  std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
252 
253  // Element size
254  double h;
255 
256  // Element degree
257  double degree = double(fe.degree);
258 
259  // CFL
260  double CFL = 0;
261 
262  for (const auto &cell : dof_handler.active_cell_iterators())
263  {
264  if (cell->is_locally_owned())
265  {
266  fe_values.reinit(cell);
267 
268  // Compute cell diameter
269  double cell_measure =
270  compute_cell_measure_with_JxW(fe_values.get_JxW_values());
271  h = compute_cell_diameter<dim>(cell_measure, degree);
272 
273  fe_values[velocities].get_function_values(evaluation_point,
274  present_velocity_values);
275  for (unsigned int q = 0; q < n_q_points; ++q)
276  {
277  const double localCFL =
278  present_velocity_values[q].norm() / h * time_step;
279  CFL = std::max(CFL, localCFL);
280  }
281  }
282  }
283  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
284  CFL = Utilities::MPI::max(CFL, mpi_communicator);
285  return (CFL);
286 }

337 template <int dim, typename VectorType>
338 double
339 calculate_enstrophy(const DoFHandler<dim> &dof_handler,
340  const VectorType &evaluation_point,
341  const Quadrature<dim> &quadrature_formula,
342  const Mapping<dim> &mapping)
343 {
344  const FESystem<dim, dim> fe = dof_handler.get_fe();
345  FEValues<dim> fe_values(mapping,
346  fe,
347  quadrature_formula,
348  update_values | update_gradients |
349  update_quadrature_points | update_JxW_values);
350 
351  const FEValuesExtractors::Vector velocities(0);
352 
353  const unsigned int n_q_points = quadrature_formula.size();
354 
355  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
356  double en = 0.0;
357  double domain_volume =
358  GridTools::volume(dof_handler.get_triangulation(), mapping);
359 
360  for (const auto &cell : dof_handler.active_cell_iterators())
361  {
362  if (cell->is_locally_owned())
363  {
364  fe_values.reinit(cell);
365 
366  fe_values[velocities].get_function_gradients(
367  evaluation_point, present_velocity_gradients);
368 
369  for (unsigned int q = 0; q < n_q_points; q++)
370  {
371  // Find the values of gradient of ux and uy (the finite element
372  // solution) at the quadrature points
373  double ux_y = present_velocity_gradients[q][0][1];
374  double uy_x = present_velocity_gradients[q][1][0];
375 
376  if (dim == 2)
377  {
378  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
379  domain_volume;
380  }
381  else
382  {
383  double uz_y = present_velocity_gradients[q][2][1];
384  double uy_z = present_velocity_gradients[q][1][2];
385  double ux_z = present_velocity_gradients[q][0][2];
386  double uz_x = present_velocity_gradients[q][2][0];
387  en += 0.5 * (uz_y - uy_z) * (uz_y - uy_z) * fe_values.JxW(q) /
388  domain_volume;
389  en += 0.5 * (ux_z - uz_x) * (ux_z - uz_x) * fe_values.JxW(q) /
390  domain_volume;
391  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
392  domain_volume;
393  }
394  }
395  }
396  }
397  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
398  en = Utilities::MPI::sum(en, mpi_communicator);
399  return (en);
400 }

446 template <int dim, typename VectorType>
447 double
448 calculate_pressure_power(const DoFHandler<dim> &dof_handler,
449  const VectorType &evaluation_point,
450  const Quadrature<dim> &quadrature_formula,
451  const Mapping<dim> &mapping)
452 {
453  const FESystem<dim, dim> fe = dof_handler.get_fe();
454  FEValues<dim> fe_values(mapping,
455  fe,
456  quadrature_formula,
457  update_values | update_gradients | update_JxW_values);
458 
459  const FEValuesExtractors::Vector velocities(0);
460  const FEValuesExtractors::Scalar pressure(dim);
461 
462  const unsigned int n_q_points = quadrature_formula.size();
463 
464  std::vector<Tensor<1, dim>> velocity(n_q_points);
465  std::vector<Tensor<1, dim>> pressure_gradients(n_q_points);
466 
467  double pressure_power = 0.0;
468 
469  for (const auto &cell : dof_handler.active_cell_iterators())
470  {
471  if (cell->is_locally_owned())
472  {
473  fe_values.reinit(cell);
474 
475  fe_values[pressure].get_function_gradients(evaluation_point,
476  pressure_gradients);
477 
478  fe_values[velocities].get_function_values(evaluation_point, velocity);
479 
480  for (unsigned int q = 0; q < n_q_points; q++)
481  {
482  pressure_power +=
483  velocity[q] * pressure_gradients[q] * fe_values.JxW(q);
484  }
485  }
486  }
487 
488  pressure_power /= GridTools::volume(dof_handler.get_triangulation(), mapping);
489 
490  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
491  pressure_power = Utilities::MPI::sum(pressure_power, mpi_communicator);
492 
493  return pressure_power;
494 }

541 template <int dim, typename VectorType>
542 double
543 calculate_viscous_dissipation(
544  const DoFHandler<dim> &dof_handler,
545  const VectorType &evaluation_point,
546  const Quadrature<dim> &quadrature_formula,
547  const Mapping<dim> &mapping,
548  const PhysicalPropertiesManager &properties_manager)
549 {
550  const FESystem<dim, dim> fe = dof_handler.get_fe();
551  FEValues<dim> fe_values(mapping,
552  fe,
553  quadrature_formula,
554  update_gradients | update_JxW_values);
555 
556  const FEValuesExtractors::Vector velocity(0);
557 
558  const unsigned int n_q_points = quadrature_formula.size();
559 
560  std::vector<Tensor<2, dim>> velocity_gradients(n_q_points);
561 
562  const auto rheological_model = properties_manager.get_rheology();
563 
564  double viscous_dissipation = 0;
565 
566  for (const auto &cell : dof_handler.active_cell_iterators())
567  {
568  if (cell->is_locally_owned())
569  {
570  fe_values.reinit(cell);
571 
572  fe_values[velocity].get_function_gradients(evaluation_point,
573  velocity_gradients);
574 
575  for (unsigned int q = 0; q < n_q_points; q++)
576  {
577  Tensor<2, dim> shear_rate =
578  velocity_gradients[q] + transpose(velocity_gradients[q]);
579 
580  const double shear_rate_magnitude =
581  calculate_shear_rate_magnitude(shear_rate);
582 
583  std::map<field, double> field_values;
584  field_values[field::shear_rate] = shear_rate_magnitude;
585 
586  const double kinematic_viscosity =
587  rheological_model->value(field_values);
588 
589  // Equation is t_ij djui (Eq 11.2-1 of BSL 2nd edition)
590  viscous_dissipation +=
591  kinematic_viscosity *
592  scalar_product(shear_rate, transpose(velocity_gradients[q])) *
593  fe_values.JxW(q);
594  }
595  }
596  }
597 
598  viscous_dissipation /=
599  GridTools::volume(dof_handler.get_triangulation(), mapping);
600 
601  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
602  viscous_dissipation =
603  Utilities::MPI::sum(viscous_dissipation, mpi_communicator);
604  return viscous_dissipation;
605 }

759  const VectorType &evaluation_point,
760  const Quadrature<dim> &quadrature_formula,
761  const Mapping<dim> &mapping,
762  const PhysicalPropertiesManager &properties_manager)
763 {
764  double integral_viscosity_x_shear_rate = 0;
765  double integral_shear_rate = 0;
766  double shear_rate_magnitude;
767  Tensor<2, dim> shear_rate;
768  double kinematic_viscosity;
769  const auto rheological_model = properties_manager.get_rheology();
770 
771  const FESystem<dim, dim> fe = dof_handler.get_fe();
772  FEValues<dim> fe_values(mapping,
773  fe,
774  quadrature_formula,
775  update_gradients | update_JxW_values);
776 
777  const FEValuesExtractors::Vector velocity(0);
778 
779  const unsigned int n_q_points = quadrature_formula.size();
780

782 
783  for (const auto &cell : dof_handler.active_cell_iterators())

785  if (cell->is_locally_owned())
786  {
787  fe_values.reinit(cell);
788 
789  fe_values[velocity].get_function_gradients(evaluation_point,

795  velocity_gradients[q] + transpose(velocity_gradients[q]);

811

813 
814  integral_viscosity_x_shear_rate +=
815  kinematic_viscosity * shear_rate_x_velocity_gradient *
816  fe_values.JxW(q);
817  integral_shear_rate +=
818  shear_rate_x_velocity_gradient * fe_values.JxW(q);
819  }
820  }
821  }

826  Utilities::MPI::sum(integral_shear_rate, mpi_communicator);
