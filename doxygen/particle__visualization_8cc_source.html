33template <int dim>
34void
35ParticleVisualization<dim>::build_visualization_files()
36{
37 // Export grid at wall of the reactor grid
38 DataOutFaces<dim> grid_out_faces;
39 grid_out_faces.attach_dof_handler(empty_dof_handler);
40 grid_out_faces.build_patches();
41 std::string s = visualization_filename;
42 std::string filename = s.substr(0, s.find('.')) + "_grid_";
43 write_boundaries_vtu<dim>(
44 grid_out_faces, "./", 0, 0, MPI_COMM_WORLD, filename, 0);
45
46 // Clear memory consuming used objects
47 grid_out_faces.clear();
48 empty_dof_handler.clear();
49
50 // Insert and move the particle positions prior generate vtu/pvtu files
51 std::vector<Point<dim>> position(1);
52 for (unsigned int i = 0; i < particle_counts.size(); i++)
53 {
54 // Put one position in a vector (mandatory to insert particle without
55 // using triangulation iterator)
56 position[0] = particle_positions[i];
57
58 // If first position, insert it in particle handler, otherwise the
59 // position is moved
60 if (i == 0)
61 particle_handler.insert_particles(position);
62 else
63 particle_handler.set_particle_positions(position, false);
64
65 // Set properties (counts for all detectors) to the particle position
66 particle_handler.begin()->set_properties(particle_counts[i]);
67
68 // Generate files
69 output_particles(i);
70 }
71}

73template <int dim>
74void
75ParticleVisualization<dim>::output_particles(unsigned int it)
76{
77 // Create DataOut for particles
78 Particles::DataOut<dim> particle_output;
79
80 // Set detector ids
81 unsigned int n_detectors = particle_counts[0].size();
82 std::vector<std::string> detector_ids = {"detector_counts_0"};
83 std::vector<DataComponentInterpretation::DataComponentInterpretation>
84 data_component_interpretation(
85 n_detectors, DataComponentInterpretation::component_is_scalar);
86 if (n_detectors > 1)
87 {
88 for (unsigned int i = 1; i < n_detectors; i++)
89 detector_ids.push_back("detector_counts_" + std::to_string(i));
90 }
91
92 particle_output.build_patches(particle_handler,
93 detector_ids,
94 data_component_interpretation);
95
96 std::string s = visualization_filename;
97 std::string filename = s.substr(0, s.find('.')) + "_";
98 particle_output.write_vtu_with_pvtu_record("./",
99 filename,
100 it,
101 MPI_COMM_WORLD);
102}
