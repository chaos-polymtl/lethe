15template <int dim>
16bool
17MortarManagerBase<dim>::is_mesh_aligned() const
18{
19 AssertThrow(dim != 1, ExcInternalError());
20
21 constexpr double tolerance = 1e-8;
22 const double delta_0 = 2 * numbers::PI / n_subdivisions[0];
23
24 return std::abs(rotation_angle / delta_0 -
25 std::round(rotation_angle / delta_0)) < tolerance;
26}

28template <int dim>
29unsigned int
30MortarManagerBase<dim>::get_n_total_mortars() const
31{
32 if constexpr (dim == 1)
33 return 1;
34
35 unsigned int n_total_subdivisions = n_subdivisions[0];
36
37 // In 3D, besides the subdivisions in the plane perpendicular to the rotation
38 // axis, we also need to account for the number of subdivisions along the
39 // rotation axis direction
40 if constexpr (dim == 3)
41 n_total_subdivisions *= n_subdivisions[1];
42
43 if (this->is_mesh_aligned()) // aligned
44 {
45 return n_total_subdivisions;
46 }
47 else // inside/outside
48 {
49 return 2 * n_total_subdivisions;
50 }
51}

70template <int dim>
71std::vector<unsigned int>
72MortarManagerBase<dim>::get_mortar_indices(const Point<dim> &face_center,
73 const bool is_inner) const
74{
75 if constexpr (dim == 1)
76 return std::vector<unsigned int>{0};
77
78 // Mesh alignment type and cell indexes
79 const auto [type, id_in_plane, id_out_plane] =
80 get_config(face_center, is_inner);
81
82 if (type == 0) // aligned
83 {
84 std::vector<unsigned int> indices;
85
86 const unsigned int index = id_in_plane;
87
88 AssertIndexRange(index, n_subdivisions[0]);
89
90 indices.emplace_back(index + n_subdivisions[0] * id_out_plane);
91
92 return indices;
93 }
94 else if (type == 1) // inside
95 {
96 std::vector<unsigned int> indices;
97
98 for (unsigned int q = 0; q < 2; ++q)
99 {
100 const unsigned int index =
101 (id_in_plane * 2 + 1 + q) % (n_subdivisions[0] * 2);
102
103 AssertIndexRange(index, n_subdivisions[0] * 2);
104
105 indices.emplace_back(index + 2 * n_subdivisions[0] * id_out_plane);
106 }
107
108 return indices;
109 }
110 else // outside
111 {
112 std::vector<unsigned int> indices;
113
114 for (unsigned int q = 0; q < 2; ++q)
115 {
116 const unsigned int index = id_in_plane * 2 + q;
117
118 AssertIndexRange(index, n_subdivisions[0] * 2);
119
120 indices.emplace_back(index + 2 * n_subdivisions[0] * id_out_plane);
121 }
122
123 return indices;
124 }
125}

147template <int dim>
148std::vector<Point<dim>>
149MortarManagerBase<dim>::get_points(const Point<dim> &face_center,
150 const bool is_inner) const
151{
152 if constexpr (dim == 1)
153 return std::vector<Point<dim>>{face_center};
154
155 // Mesh alignment type and cell index
156 const auto [type, id_in_plane, id_out_plane] =
157 get_config(face_center, is_inner);
158 // Angle variation within each cell
159 const double delta_0 = 2 * numbers::PI / n_subdivisions[0];
160 double delta_1 = 1.0;
161
162 if constexpr (dim == 3)
163 delta_1 = radius[1] / n_subdivisions[1];
164
165 if (type == 0) // aligned
166 {
167 std::vector<Point<dim>> points;
168
169 for (unsigned int q = 0; q < n_quadrature_points; ++q)
170 {
171 const auto x =
172 from_1D((id_in_plane + quadrature.point(q)[0]) * delta_0);
173
174 if constexpr (dim == 3)
175 points.emplace_back(
176 x[0], x[1], (id_out_plane + quadrature.point(q)[1]) * delta_1);
177 else
178 points.emplace_back(x);
179 }
180
181 return points;
182 }
183 else // Point at the inner boundary lies somewhere in the face of the outer
184 // boundary cell
185 {
186 // rad_0: first cell vertex (fixed)
187 // rad_1: shifted vertex
188 // rad_2: last cell vertex (fixed)
189 double rad_0, rad_1, rad_2;
190 // Minimum rotation angle
191 double rot_min =
192 rotation_angle - std::floor(rotation_angle / delta_0) * delta_0;
193
194 if (type == 2) // outside
195 {
196 rad_0 = id_in_plane * delta_0;
197 rad_1 = id_in_plane * delta_0 + rot_min;
198 rad_2 = (id_in_plane + 1) * delta_0;
199 }
200 else // inside
201 {
202 rad_0 = id_in_plane * delta_0 + rot_min;
203 rad_1 = (id_in_plane + 1) * delta_0;
204 rad_2 = (id_in_plane + 1) * delta_0 + rot_min;
205 }
206
207 std::vector<Point<dim>> points;
208
209 for (unsigned int q = 0; q < n_quadrature_points; ++q)
210 {
211 const auto x =
212 from_1D(rad_0 + quadrature.point(q)[0] * (rad_1 - rad_0));
213
214 if constexpr (dim == 3)
215 points.emplace_back(
216 x[0], x[1], (id_out_plane + quadrature.point(q)[1]) * delta_1);
217 else
218 points.emplace_back(x);
219 }
220
221 for (unsigned int q = 0; q < n_quadrature_points; ++q)
222 {
223 const auto x =
224 from_1D(rad_1 + quadrature.point(q)[0] * (rad_2 - rad_1));
225
226 if constexpr (dim == 3)
227 points.emplace_back(
228 x[0], x[1], (id_out_plane + quadrature.point(q)[1]) * delta_1);
229 else
230 points.emplace_back(x);
231 }
232
233 return points;
234 }
235}

237template <int dim>
238std::vector<Point<std::max(1, dim - 1)>>
239MortarManagerBase<dim>::get_points_ref(const Point<dim> &face_center,
240 const bool is_inner) const
241{
242 if (dim == 1)
243 return std::vector<Point<std::max(1, dim - 1)>>{
244 Point<std::max(1, dim - 1)>()};
245
246 const auto [type, _, __] = get_config(face_center, is_inner);
247
248 const double delta_0 = 2 * numbers::PI / n_subdivisions[0];
249
250 if (type == 0) // aligned
251 {
252 std::vector<Point<std::max(1, dim - 1)>> points;
253 points.reserve(n_quadrature_points);
254
255 for (unsigned int q = 0; q < n_quadrature_points; ++q)
256 points.emplace_back(quadrature.point(q));
257
258 return points;
259 }
260 else // inside/outside
261 {
262 double rad_0, rad_1, rad_2;
263
264 double rot_min =
265 (rotation_angle - std::floor(rotation_angle / delta_0) * delta_0) /
266 delta_0;
267
268 if (type == 2) // outside
269 {
270 rad_0 = 0.0;
271 rad_1 = rot_min;
272 rad_2 = 1.0;
273 }
274 else // inside
275 {
276 rad_0 = 0.0;
277 rad_1 = 1.0 - rot_min;
278 rad_2 = 1.0;
279 }
280
281 std::vector<Point<std::max(1, dim - 1)>> points;
282 points.reserve(2 * n_quadrature_points);
283
284 for (unsigned int q = 0; q < n_quadrature_points; ++q)
285 {
286 const double x = rad_0 + quadrature.point(q)[0] * (rad_1 - rad_0);
287
288 if (dim == 2)
289 points.emplace_back(x);
290 else
291 points.emplace_back(x, quadrature.point(q)[1]);
292 }
293
294 for (unsigned int q = 0; q < n_quadrature_points; ++q)
295 {
296 const double x = rad_1 + quadrature.point(q)[0] * (rad_2 - rad_1);
297
298 if (dim == 2)
299 points.emplace_back(x);
300 else
301 points.emplace_back(x, quadrature.point(q)[1]);
302 }
303
304 return points;
305 }

308template <int dim>
309std::vector<double>
310MortarManagerBase<dim>::get_weights(const Point<dim> &face_center,
311 const bool is_inner) const
312{
313 if (dim == 1)
314 return std::vector<double>{1.0};
315
316 // Mesh alignment type and cell index
317 const auto [type, id_in_plane, _] = get_config(face_center, is_inner);
318 // Angle variation within each cell
319 const double delta_0 = 2 * numbers::PI / n_subdivisions[0];
320 double delta_1 = 1.0;
321
322 if (dim == 3)
323 delta_1 = radius[1] / n_subdivisions[1];
324
325 if (type == 0) // aligned
326 {
327 std::vector<double> weights;
328 weights.reserve(n_quadrature_points);
329
330 for (unsigned int q = 0; q < n_quadrature_points; ++q)
331 weights.emplace_back(radius[0] * quadrature.weight(q) * delta_0 *
332 delta_1);
333
334 return weights;
335 }
336 else // inside/outside
337 {
338 double rad_0, rad_1, rad_2;
339
340 double rot_min =
341 rotation_angle - std::floor(rotation_angle / delta_0) * delta_0;
342
343 if (type == 2) // outside
344 {
345 rad_0 = id_in_plane * delta_0;
346 rad_1 = id_in_plane * delta_0 + rot_min;
347 rad_2 = (id_in_plane + 1) * delta_0;
348 }
349 else // inside
350 {
351 rad_0 = id_in_plane * delta_0 + rot_min;
352 rad_1 = (id_in_plane + 1) * delta_0;
353 rad_2 = (id_in_plane + 1) * delta_0 + rot_min;
354 }
355
356 std::vector<double> weights;
357 weights.reserve(2 * n_quadrature_points);
358
359 for (unsigned int q = 0; q < n_quadrature_points; ++q)
360 weights.emplace_back(radius[0] * quadrature.weight(q) *
361 (rad_1 - rad_0) * delta_1);
362
363 for (unsigned int q = 0; q < n_quadrature_points; ++q)
364 weights.emplace_back(radius[0] * quadrature.weight(q) *
365 (rad_2 - rad_1) * delta_1);
366
367 return weights;
368 }

389template <int dim>
390std::tuple<unsigned int, unsigned int, unsigned int>
391MortarManagerBase<dim>::get_config(const Point<dim> &face_center,
392 const bool is_inner) const
393{
394 const auto angle_cell_center = to_1D(face_center);
395
396 // Angular variation in each cell
397 const double delta_0 = 2 * numbers::PI / n_subdivisions[0];
398 // Minimum rotation angle
399 const double rot_min =
400 rotation_angle - std::floor(rotation_angle / delta_0) * delta_0;
401
402 AssertThrow(rot_min <= delta_0, ExcInternalError());
403
404 // Point position in the cell
405 const double segment = (angle_cell_center - delta_0 / 2) / delta_0;
406 // Point position after rotation
407 const double segment_rot =
408 (angle_cell_center - delta_0 / 2 - rot_min) / delta_0;
409 // Cell index in the direction of the rotation axis
410 unsigned int id_out_plane = 0;
411
412 if constexpr (dim == 3)
413 {
414 const double delta_1 = radius[1] / n_subdivisions[1];
415 id_out_plane = static_cast<unsigned int>(
416 std::round((face_center[2] - delta_1 / 2) / delta_1));
417 }
418
419 if (this->is_mesh_aligned())
420 {
421 // Case 1: mesh is aligned
422 return {0, static_cast<unsigned int>(std::round(segment)), id_out_plane};
423 }
424 else
425 {
426 // Case 2: mesh is not aligned
427 if (!is_inner)
428 // outer (fixed) domain
429 return {2,
430 static_cast<unsigned int>(std::round(segment)),
431 id_out_plane};
432 else
433 // inner (rotated) domain
434 return {1,
435 (static_cast<unsigned int>(std::round(segment_rot)) +
436 2 * n_subdivisions[0]) %
437 (2 * n_subdivisions[0]),
438 id_out_plane};
439 }
440}

445#if DEAL_II_VERSION_GTE(9, 7, 0)
446template <int dim>
447std::tuple<std::vector<unsigned int>, std::vector<double>, double>
448compute_n_subdivisions_and_radius(
449 const Triangulation<dim> &triangulation,
450 const Mapping<dim> &mapping,
451 const Parameters::Mortar<dim> &mortar_parameters)
452{
453 // Number of subdivisions per process
454 unsigned int n_subdivisions_local = 0;
455 // Number of subdivisions in the radial direction per process
456 unsigned int n_subdivisions_plane_local = 0;
457 // Tolerance for rotor radius computation
458 constexpr double tolerance = 1e-4;
459 // Min and max values for rotor radius computation
460 double radius_min = std::numeric_limits<double>::max();
461 double radius_max = 0;
462 // Minimum rotation angle in initial mesh configuration
463 double pre_rotation_min = std::numeric_limits<double>::max();
464
465 // Coordinate of the reference cell for computation of the number of
466 // subdivisions in the radial direction. Used in 3D case
467 double coord_ref_local = std::numeric_limits<double>::max();
468 // Rotation axis direction. Used in 3D case
469 unsigned int direction = 0;
470 // Min and max vertex coordinates for length computation in the axial
471 // direction. Used in 3D case
472 double vertex_min = std::numeric_limits<double>::max();
473 double vertex_max = 0;
474
475 // Verify if rotation axis is a unit vector in x, y, or z
476 if constexpr (dim == 3)
477 {
478 // First check if the vector has a unit norm
479 bool is_unit_axis =
480 mortar_parameters.rotation_axis.norm() == 1 ? true : false;
481
482 // Now check if the vector represents the x, y, or z directions
483 // specifically (we assume those are the only options for now)
484 for (int i = 0; i < dim; i++)
485 if (mortar_parameters.rotation_axis[i] != 0. &&
486 mortar_parameters.rotation_axis[i] != 1.)
487 is_unit_axis = false;
488
489 AssertThrow(
490 is_unit_axis,
491 ExcMessage(
492 " The rotation axis must be a unit vector in x, y, or z direction."));
493
494 // Find the direction of the rotation axis
495 for (int d = 0; d < dim; d++)
496 if (mortar_parameters.rotation_axis[d] != 0.0)
497 direction = d;
498
499 // To compute the number of subdivisions in the radial direction, we will
500 // use a reference cell. Its coordinate in the direction of the rotation
501 // axis is stored; then, all the cells at the same plane will be counted
502 // as an "in-plane" (radial direction) subdivision
503 for (const auto &cell : triangulation.active_cell_iterators())
504 {
505 if (cell->is_locally_owned())
506 {
507 for (const auto face_no : cell->face_indices())
508 {
509 const auto face = cell->face(face_no);
510
511 if (face->at_boundary() &&
512 face->boundary_id() ==
513 mortar_parameters.rotor_boundary_id)
514 coord_ref_local =
515 std::min(coord_ref_local, cell->center()[direction]);
516 }
517 }
518 }
519 }
520
521 // Coordinate of reference cell at all processes
522 const double coord_ref =
523 Utilities::MPI::min(coord_ref_local, triangulation.get_mpi_communicator());
524
525 // Check number of faces and vertices at the rotor-stator interface
526 for (const auto &cell : triangulation.active_cell_iterators())
527 {
528 if (cell->is_locally_owned())
529 {
530 for (const auto face_no : cell->face_indices())
531 {
532 const auto face = cell->face(face_no);
533
534 if (face->at_boundary())
535 {
536 if (face->boundary_id() ==
537 mortar_parameters.rotor_boundary_id)
538 {
539 n_subdivisions_local++;
540
541 // Store the number of subdivisions in the radial
542 // direction
543 if constexpr (dim == 3)
544 {
545 // Check if the current cell is contained in the same
546 // plane as the reference cell
547 if (std::abs(cell->center()[direction] - coord_ref) <
548 tolerance)
549 n_subdivisions_plane_local++;
550 }
551 else
552 {
553 n_subdivisions_plane_local++;
554 }
555
556 const auto vertices = mapping.get_vertices(cell, face_no);
557
558 for (unsigned int vertex_index = 0;
559 vertex_index < face->n_vertices();
560 vertex_index++)
561 {
562 const auto v = vertices[vertex_index];
563 double radius_current =
564 v.distance(mortar_parameters.center_of_rotation);
565
566 // In 3D, the interface radius to be computed is
567 // actually with respect to the rotation axis. For
568 // this computation, we use the relation:
569 // radius_current = ||VC x d||/||d||, where VC is the
570 // current vertex minus the center of rotation, and d
571 // is the rotation axis
572 if constexpr (dim == 3)
573 {
574 vertex_min = std::min(vertex_min, v[direction]);
575 vertex_max = std::max(vertex_max, v[direction]);
576
577 const auto aux = cross_product_3d(
578 (v - mortar_parameters.center_of_rotation),
579 mortar_parameters.rotation_axis /
580 mortar_parameters.rotation_axis.norm());
581 const double num = aux.norm();
582 const double den =
583 mortar_parameters.rotation_axis.norm();
584 radius_current = num / den;
585 }
586
587 radius_min = std::min(radius_min, radius_current);
588 radius_max = std::max(radius_max, radius_current);
589 }
590 }
591 // Obtain the minimum initial rotation angle based on the
592 // stator interface
593 else if (face->boundary_id() ==
594 mortar_parameters.stator_boundary_id)
595 {
596 const auto vertices = mapping.get_vertices(cell, face_no);
597
598 for (unsigned int vertex_index = 0;
599 vertex_index < face->n_vertices();
600 vertex_index++)
601 {
602 const auto v = vertices[vertex_index];
603
604 pre_rotation_min = std::min(
605 pre_rotation_min,
606 point_to_angle(mortar_parameters.center_of_rotation,
607 v));
608 }
609 }
610 }
611 }
612 }
613 }
614
615 // Total number of faces
616 const unsigned int n_subdivisions =
617 Utilities::MPI::sum(n_subdivisions_local,
618 triangulation.get_mpi_communicator());
619
620 // Total number of faces at the radial direction
621 const unsigned int n_subdivisions_plane =
622 Utilities::MPI::sum(n_subdivisions_plane_local,
623 triangulation.get_mpi_communicator());
624
625 // Min and max values over all processes
626 radius_min =
627 Utilities::MPI::min(radius_min, triangulation.get_mpi_communicator());
628 radius_max =
629 Utilities::MPI::max(radius_max, triangulation.get_mpi_communicator());
630
631 pre_rotation_min =
632 Utilities::MPI::min(pre_rotation_min, triangulation.get_mpi_communicator());
633
634 vertex_min =
635 Utilities::MPI::min(vertex_min, triangulation.get_mpi_communicator());
636 vertex_max =
637 Utilities::MPI::max(vertex_max, triangulation.get_mpi_communicator());
638
639 AssertThrow(
640 std::abs(radius_max - radius_min) < tolerance,
641 ExcMessage(
642 "The computed radius of the rotor mesh has a variation greater than "
643 "the tolerance across the rotor domain, meaning that the prescribed "
644 "center of rotation and the rotor geometry are not in accordance."));
645
646 // Final radius value
647 const double radius = radius_min;
648 // Length along the axial direction
649 const double length_rot_axis = dim == 3 ? vertex_max - vertex_min : 0;
650
651 return {{n_subdivisions_plane, n_subdivisions / n_subdivisions_plane},
652 {radius, length_rot_axis},
653 pre_rotation_min};
654}
655#else

657std::tuple<std::vector<unsigned int>, std::vector<double>, double>
658compute_n_subdivisions_and_radius(const Triangulation<dim> &,

660 const Parameters::Mortar<dim> &)

662 AssertThrow(false,
663 ExcMessage(
664 "The mortar coupling requires deal.II 9.7 or more recent."));
665
666 return {{1, 1}, {1.0, 0}, 0.0};

716#if DEAL_II_VERSION_GTE(9, 7, 0)
717template <int dim, typename Number>
718CouplingOperator<dim, Number>::CouplingOperator(
719 const Mapping<dim> &mapping,
720 const DoFHandler<dim> &dof_handler,
721 const AffineConstraints<Number> &constraints,
722 const std::shared_ptr<CouplingEvaluationBase<dim, Number>> evaluator,
723 const std::shared_ptr<MortarManagerBase<dim>> mortar_manager,
724 const unsigned int bid_m,
725 const unsigned int bid_p,
726 const double sip_factor)
727 : mapping(mapping)
728 , dof_handler(dof_handler)
729 , bid_m(bid_m)
730 , bid_p(bid_p)
731 , evaluator(evaluator)
732 , mortar_manager(mortar_manager)
733{
734 this->q_data_size = evaluator->data_size();
735 this->relevant_dof_indices = evaluator->get_relevant_dof_indices();
736 this->n_dofs_per_cell = this->relevant_dof_indices.size();
737
738 data.penalty_factor =
739 compute_penalty_factor(dof_handler.get_fe().degree, sip_factor);
740
741 // Number of cells
742 const unsigned int n_sub_cells = mortar_manager->get_n_total_mortars();
743
744 std::vector<types::global_dof_index> is_local_cell;
745 std::vector<types::global_dof_index> is_ghost_cell;
746
747# ifdef DEBUG
748 std::vector<double> vec_local_cells(n_sub_cells * 2, 0.0);
749 std::vector<double> vec_ghost_cells(n_sub_cells * 2, 0.0);
750# endif
751
752 for (const auto &cell : dof_handler.active_cell_iterators())
753 if (cell->is_locally_owned())
754 for (const auto face_no : cell->face_indices())
755 if ((cell->face(face_no)->boundary_id() == bid_m) ||
756 (cell->face(face_no)->boundary_id() == bid_p))
757 {
758 const auto face = cell->face(face_no);
759
760 const auto center = get_face_center(cell, face);
761
762 // Indices of mortars on face of cell.
763 const auto indices = mortar_manager->get_mortar_indices(
764 center, cell->face(face_no)->boundary_id() == bid_m);
765
766 const auto local_dofs = this->get_dof_indices(cell);
767
768 // Loop over the mortar indices at the rotor-stator
769 // interface. The logic of local (rotor) and ghost (stator) is the
770 // same as in the previous loop.
771 for (unsigned int ii = 0; ii < indices.size(); ++ii)
772 {
773 const unsigned int i = indices[ii];
774 unsigned int id_local = 0, id_ghost = 0;
775
776 if (face->boundary_id() == bid_m)
777 {
778 id_local = i;
779 id_ghost = i + n_sub_cells;
780 }
781 else if (face->boundary_id() == bid_p)
782 {
783 id_local = i + n_sub_cells;
784 id_ghost = i;
785 }
786
787# ifdef DEBUG
788 vec_local_cells[id_local] += 1.0;
789 vec_ghost_cells[id_ghost] += 1.0;
790# endif
791 is_local_cell.emplace_back(id_local);
792 is_ghost_cell.emplace_back(id_ghost);
793
794 for (const auto l_dof : local_dofs)
795 dof_indices.emplace_back(l_dof);
796 }
797
798 // Weights of quadrature points
799 const auto weights =
800 mortar_manager->get_weights(get_face_center(cell, face),
801 cell->face(face_no)->boundary_id() ==
802 bid_m);
803 data.all_weights.insert(data.all_weights.end(),
804 weights.begin(),
805 weights.end());
806
807 // Normals of quadrature points
808 if constexpr (dim == 3)
809 {
810 const auto points = mortar_manager->get_points(
811 get_face_center(cell, face),
812 cell->face(face_no)->boundary_id() == bid_m);
813 std::vector<Point<dim, Number>> points_ref(points.size());
814 mapping.transform_points_real_to_unit_cell(cell,
815 points,
816 points_ref);
817 all_points_ref.insert(all_points_ref.end(),
818 points_ref.begin(),
819 points_ref.end());
820
821 std::vector<Point<dim - 1>> quad;
822
823 for (const auto p : points_ref)
824 {
825 Point<dim - 1> temp;
826 for (int i = 0, j = 0; i < dim; ++i)
827 if ((face_no / 2) != static_cast<unsigned int>(i))
828 temp[j++] = p[i];
829
830 if ((dim == 3) && ((face_no / 2) == 1))
831 std::swap(temp[0], temp[1]);
832
833 quad.emplace_back(temp);
834 }
835
836 FEFaceValues<dim> fe_face_values(mapping,
837 cell->get_fe(),
838 quad,
839 update_normal_vectors);
840
841 fe_face_values.reinit(cell, face_no);
842
843 data.all_normals.insert(
844 data.all_normals.end(),
845 fe_face_values.get_normal_vectors().begin(),
846 fe_face_values.get_normal_vectors().end());
847 }
848 else
849 {
850 auto normals = mortar_manager->get_normals(
851 get_face_center(cell, face),
852 cell->face(face_no)->boundary_id() == bid_m);
853 if (face->boundary_id() == bid_p)
854 for (auto &normal : normals)
855 normal *= -1.0;
856 data.all_normals.insert(data.all_normals.end(),
857 normals.begin(),
858 normals.end());
859
860 if constexpr (dim == 1)
861 {
862 if (face_no == 0)
863 all_points_ref.emplace_back(0.0);
864 else if (face_no == 1)
865 all_points_ref.emplace_back(1.0);
866 else
867 AssertThrow(false, ExcNotImplemented());
868 }
869 else if constexpr (dim == 2)
870 {
871 auto points = mortar_manager->get_points_ref(
872 get_face_center(cell, face),
873 cell->face(face_no)->boundary_id() == bid_m);
874
875 const bool flip =
876 (face->vertex(0)[0] * face->vertex(1)[1] -
877 face->vertex(0)[1] * face->vertex(1)[0]) < 0.0;
878
879 if (flip)
880 for (auto &p : points)
881 p[0] = 1.0 - p[0];
882
883 if (face_no / 2 == 0)
884 {
885 for (auto &p : points)
886 all_points_ref.emplace_back(face_no % 2, p[0]);
887 }
888 else if (face_no / 2 == 1)
889 {
890 for (auto &p : points)
891 all_points_ref.emplace_back(p[0], face_no % 2);
892 }
893 else
894 {
895 AssertThrow(false, ExcNotImplemented());
896 }
897 }
898 else if constexpr (dim == 3)
899 {
900 AssertThrow(false, ExcNotImplemented()); // TODO
901 }
902 else
903 AssertThrow(false, ExcNotImplemented());
904 }
905
906 // Penalty parameter
907 const Number penalty_parameter = compute_penalty_parameter(cell);
908
909 // Store penalty parameter for all quadrature points
910 for (unsigned int i = 0; i < mortar_manager->get_n_points(); ++i)
911 data.all_penalty_parameter.emplace_back(penalty_parameter);
912 }
913
914# ifdef DEBUG
915 Utilities::MPI::sum(vec_local_cells,
916 dof_handler.get_mpi_communicator(),
917 vec_local_cells);
918 Utilities::MPI::sum(vec_ghost_cells,
919 dof_handler.get_mpi_communicator(),
920 vec_ghost_cells);
921
922 std::set<unsigned int> vec_local_cells_0;
923 std::set<unsigned int> vec_local_cells_2;
924
925 if (Utilities::MPI::this_mpi_process(dof_handler.get_mpi_communicator()) == 0)
926 {
927 for (unsigned int i = 0; i < vec_local_cells.size(); ++i)
928 {
929 if (vec_local_cells[i] == 0)
930 vec_local_cells_0.insert(i);
931 if (vec_local_cells[i] > 1)
932 vec_local_cells_2.insert(i);
933 }
934
935 std::set<unsigned int> vec_ghost_cells_0;
936 std::set<unsigned int> vec_ghost_cells_2;
937
938 for (unsigned int i = 0; i < vec_ghost_cells.size(); ++i)
939 {
940 if (vec_ghost_cells[i] == 0)
941 vec_ghost_cells_0.insert(i);
942 if (vec_ghost_cells[i] > 1)
943 vec_ghost_cells_2.insert(i);
944 }
945
946 if (!(vec_local_cells_0.empty() && vec_local_cells_2.empty() &&
947 vec_ghost_cells_0.empty() && vec_ghost_cells_2.empty()))
948 {
949 std::cout << "CouplingOperator mortar matching failed:" << std::endl;
950
951 if (!vec_local_cells_0.empty())
952 {
953 std::cout << " - some local cells are not owned: ";
954 for (const auto &i : vec_local_cells_0)
955 std::cout << i << " ";
956 std::cout << std::endl;
957 }
958 if (!vec_local_cells_2.empty())
959 {
960 std::cout << " - some local cells are owned multiple times: ";
961 for (const auto &i : vec_local_cells_2)
962 std::cout << i << " ";
963 std::cout << std::endl;
964 }
965 if (!vec_ghost_cells_0.empty())
966 {
967 std::cout << " - some ghost cells are not owned: ";
968 for (const auto &i : vec_ghost_cells_0)
969 std::cout << i << " ";
970 std::cout << std::endl;
971 }
972 if (!vec_ghost_cells_2.empty())
973 {
974 std::cout << " - some ghost cells are owned multiple times: ";
975 for (const auto &i : vec_ghost_cells_2)
976 std::cout << i << " ";
977 std::cout << std::endl;
978 }
979
980 MPI_Barrier(dof_handler.get_mpi_communicator());
981
982 AssertThrow(false, ExcInternalError());
983 }
984 }
985# endif
986
987 // Setup communication
988 partitioner.reinit(is_local_cell,
989 is_ghost_cell,
990 dof_handler.get_mpi_communicator());
991
992 // Finalized penalty parameters
993 const unsigned n_q_points =
994 mortar_manager->get_n_points() / mortar_manager->get_n_mortars();
995 std::vector<Number> all_penalty_parameter_ghost(
996 data.all_penalty_parameter.size());
997 partitioner.export_to_ghosted_array<Number, 0>(data.all_penalty_parameter,
998 all_penalty_parameter_ghost,
999 n_q_points);
1000 for (unsigned int i = 0; i < data.all_penalty_parameter.size(); ++i)
1001 data.all_penalty_parameter[i] =
1002 std::max(data.all_penalty_parameter[i], all_penalty_parameter_ghost[i]);
1003
1004 // Finalize DoF indices and update constraints
1005 dof_indices_ghost.resize(dof_indices.size());
1006 partitioner.export_to_ghosted_array<types::global_dof_index, 0>(
1007 dof_indices, dof_indices_ghost, n_dofs_per_cell);
1008
1009 {
1010 auto locally_owned_dofs = constraints.get_locally_owned_indices();
1011 auto constraints_to_make_consistent = constraints.get_local_lines();
1012
1013
1014 for (unsigned int i = 0; i < dof_indices.size(); ++i)
1015 {
1016 constraints_to_make_consistent.add_index(dof_indices[i]);
1017 constraints_to_make_consistent.add_index(dof_indices_ghost[i]);
1018 }
1019
1020 constraints_extended.reinit(locally_owned_dofs,
1021 constraints_to_make_consistent);
1022 constraints_extended.merge(
1023 constraints,
1024 AffineConstraints<Number>::MergeConflictBehavior::no_conflicts_allowed,
1025 true);
1026
1027 constraints_extended.make_consistent_in_parallel(
1028 locally_owned_dofs,
1029 constraints_to_make_consistent,
1030 dof_handler.get_mpi_communicator());
1031
1032 partitioner_extended = std::make_shared<const Utilities::MPI::Partitioner>(
1033 locally_owned_dofs,
1034 constraints_extended.get_local_lines(),
1035 dof_handler.get_mpi_communicator());
1036 }
1037}

1077template <int dim, typename Number>
1078Number
1079CouplingOperator<dim, Number>::compute_penalty_parameter(
1080 const typename Triangulation<dim>::cell_iterator &cell) const
1081{
1082 const unsigned int degree = dof_handler.get_fe().degree;
1083
1084 FE_Nothing<dim> fe_nothing;
1085
1086 dealii::QGauss<dim> quadrature(degree + 1);
1087 dealii::FEValues<dim> fe_values(mapping,
1088 fe_nothing,
1089 quadrature,
1090 dealii::update_JxW_values);
1091
1092 dealii::QGauss<dim - 1> face_quadrature(degree + 1);
1093 dealii::FEFaceValues<dim> fe_face_values(mapping,
1094 fe_nothing,
1095 face_quadrature,
1096 dealii::update_JxW_values);
1097
1098 fe_values.reinit(cell);
1099
1100 Number volume = 0;
1101 for (unsigned int q = 0; q < quadrature.size(); ++q)
1102 volume += fe_values.JxW(q);
1103
1104 Number surface_area = 0;
1105 for (const auto f : cell->face_indices())
1106 {
1107 fe_face_values.reinit(cell, f);
1108
1109 const Number factor =
1110 (cell->at_boundary(f) && !cell->has_periodic_neighbor(f) &&
1111 (cell->face(f)->boundary_id() != bid_m &&
1112 cell->face(f)->boundary_id() != bid_p)) ?
1113 1. :
1114 0.5;
1115
1116 for (unsigned int q = 0; q < face_quadrature.size(); ++q)
1117 surface_area += fe_face_values.JxW(q) * factor;
1118 }
1119
1120 return surface_area / volume;
1121}

1274void
1275CouplingOperator<dim, Number>::add_diagonal_entries(VectorType &diagonal) const
1276{

1278
1279 if constexpr (std::is_same_v<VectorType, TrilinosWrappers::MPI::Vector>)
1280 {
1281 diagonal_internal.reinit(
1282 this->partitioner_extended->locally_owned_range(),
1283 this->partitioner_extended->get_mpi_communicator());
1284 }
1285 else
1286 diagonal_internal.reinit(this->partitioner_extended);
1287
1288 unsigned int ptr_q = 0;
1289
1290 Vector<Number> buffer, diagonal_local;
1291 std::vector<Number> all_values_local, all_values_ghost;
1292

1294 if (cell->is_locally_owned())
1295 for (const auto &face : cell->face_iterators())
1296 if ((face->boundary_id() == bid_m) || (face->boundary_id() == bid_p))
1297 {
1298 // Quadrature points at the current face. Note: we process
1299 // all mortars of the face together here.
1300 const unsigned int n_q_points = mortar_manager->get_n_points();
1301
1302 evaluator->local_reinit(
1303 cell,
1304 ArrayView<const Point<dim, Number>>(all_points_ref.data() + ptr_q,
1305 n_q_points));
1306
1307 buffer.reinit(n_dofs_per_cell);
1308 diagonal_local.reinit(n_dofs_per_cell);
1309 all_values_local.resize(n_q_points * q_data_size);
1310 all_values_ghost.resize(n_q_points * q_data_size);
1311
1312 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
1313 {
1314 // Create i-th basis vector
1315 for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
1316 buffer[j] = static_cast<Number>(i == j);
1317
1318 // Interpolate i-th basis vector to the quadrature points
1319 evaluator->local_evaluate(
1320 data, buffer, ptr_q, 1, all_values_local.data());
1321
1322 buffer.reinit(n_dofs_per_cell);
1323
1324 // integrate the coupling terms of the mortar using the
1325 // interpolated information from the cell
1326 evaluator->local_integrate(data,
1327 buffer,
1328 ptr_q,
1329 1,
1330 all_values_local.data(),
1331 all_values_ghost.data());
1332
1333 diagonal_local[i] = buffer[i];
1334 }
1335
1336 const auto local_dofs = this->get_dof_indices(cell);
1337 constraints_extended.distribute_local_to_global(diagonal_local,
1338 local_dofs,
1339 diagonal_internal);
1340
1341 ptr_q += n_q_points;
1342 }
1343
1344 diagonal_internal.compress(VectorOperation::add);
1345 diagonal.add(1.0, diagonal_internal);
1346}

1458 n_dofs_per_cell);

1461 // the mortar using the interpolated information from the

1464 {

1467 evaluator->local_integrate(

1469 buffer,
1470 ptr_q,
1471 n_dofs_per_cell,
1472 all_values_local.data() +
1473 (ptr_q * n_dofs_per_cell + i) * q_data_size,
1474 nullptr);
1475 else // ghost cell
1476 evaluator->local_integrate(
1477 data,
1478 buffer,
1479 ptr_q,
1480 n_dofs_per_cell,
1481 nullptr,
1482 all_values_ghost.data() +
1483 (ptr_q * n_dofs_per_cell + i) * q_data_size);
1484
1485 // Copy data from buffer to cell matrix

1487 cell_matrix[j][i] = buffer[j];
1488 }
1489
1490 // Vector of local dof indices from the cell in the negative
1491 // ('mortar') side
1492 std::vector<types::global_dof_index> local_dof_indices(
1493 dof_indices.begin() + ptr_dofs,
1494 dof_indices.begin() + ptr_dofs + n_dofs_per_cell);

1496 if (b == 0) // local cell -> local-local block
1497 {
1498 constraints_extended.distribute_local_to_global(
1499 cell_matrix, local_dof_indices, system_matrix);
1500 }
1501 else // ghost cell -> local-ghost block
1502 {
1503 std::vector<types::global_dof_index>
1504 local_dof_indices_ghost(dof_indices_ghost.begin() +
1505 ptr_dofs,
1506 dof_indices_ghost.begin() +
1507 ptr_dofs + n_dofs_per_cell);
1508
1509 constraints_extended.distribute_local_to_global(
1510 cell_matrix,
1511 local_dof_indices,
1512 local_dof_indices_ghost,
1513 system_matrix);
1514 }
1515 }
1516
1517 ptr_dofs += n_dofs_per_cell;
1518
1519 ptr_q += n_q_points;
1520 }
1521 }
1522
1523 AssertDimension(ptr_q, data.all_normals.size());
1524 AssertDimension(ptr_dofs, dof_indices.size());

1617template <int dim, int n_components, typename Number>
1618void
1619CouplingEvaluationSIPG<dim, n_components, Number>::local_integrate(
1620 const CouplingEvaluationData<dim, Number> &data,
1621 Vector<Number> &buffer,
1622 const unsigned int ptr_q,
1623 const unsigned int q_stride,
1624 Number *all_values_m,
1625 Number *all_values_p) const
1626{
1627 for (const auto q : this->phi_m.quadrature_point_indices())
1628 {
1629 const unsigned int q_index = ptr_q + q;
1630 // Initialize buffer for both 'mortar' and 'non-mortar' sides of the
1631 // interface
1632 BufferRW<Number> buffer_m(all_values_m, q * 2 * n_components * q_stride);
1633 BufferRW<Number> buffer_p(all_values_p, q * 2 * n_components * q_stride);
1634 // Read shape functions values and gradients stored in the buffer
1635 const auto value_m = buffer_m.template read<value_type>();
1636 const auto value_p = buffer_p.template read<value_type>();
1637 const auto normal_gradient_m = buffer_m.template read<value_type>();
1638 const auto normal_gradient_p = buffer_p.template read<value_type>();
1639
1640 const auto JxW = data.all_weights[q_index];
1641 const auto penalty_parameter = data.all_penalty_parameter[q_index];
1642 const auto normal = data.all_normals[q_index];
1643
1644 // The expression for the jump on the mortar interface is
1645 // jump(u) = u_m * normal_m + u_p * normal_p. Since we are accessing only
1646 // the value of normal_m, we use a minus sign here because normal_p = -
1647 // normal_m
1648 const auto value_jump = outer((value_m - value_p), normal);
1649
1650 // The expression for the average on the mortar interface is
1651 // avg(u).n = (u_m.normal_m + u_p.normal_p) * 0.5. For the same reason
1652 // above, we include the negative sign here
1653 const auto gradient_normal_avg =
1654 (normal_gradient_m - normal_gradient_p) * 0.5;
1655
1656 // SIPG penalty parameter
1657 const double sigma = penalty_parameter * data.penalty_factor;
1658
1659 // - (n avg(v), jump(u))
1660 this->phi_m.submit_gradient(-value_jump * 0.5 * JxW, q);
1661
1662 // + (jump(v), jump(u) - avg(u) n)
1663 this->phi_m.submit_value(
1664 (contract(value_jump, normal) * sigma - gradient_normal_avg) * JxW, q);
1665 }
1666 // Multiply previous terms by respective test functions values/gradients
1667 this->phi_m.test_and_sum(buffer,
1668 EvaluationFlags::values |
1669 EvaluationFlags::gradients);
1670}

1772template <int dim, typename Number>
1773void
1774NavierStokesCouplingEvaluation<dim, Number>::local_integrate(
1775 const CouplingEvaluationData<dim, Number> &data,
1776 Vector<Number> &buffer,
1777 const unsigned int ptr_q,
1778 const unsigned int q_stride,
1779 Number *all_values_m,
1780 Number *all_values_p) const
1781{
1782 for (const auto q : this->phi_u_m.quadrature_point_indices())
1783 {
1784 const unsigned int q_index = ptr_q + q;
1785
1786 // Initialize buffer for both local and ghost sides
1787 BufferRW<Number> buffer_m(all_values_m, q * 4 * dim * q_stride);
1788 BufferRW<Number> buffer_p(all_values_p, q * 4 * dim * q_stride);
1789
1790 // Read shape functions values and gradients stored in the buffer
1791 const auto u_value_m = buffer_m.template read<u_value_type>();
1792 const auto u_value_p = buffer_p.template read<u_value_type>();
1793 const auto u_grad_normal_m = buffer_m.template read<u_value_type>();
1794 const auto u_grad_normal_p = buffer_p.template read<u_value_type>();
1795 const auto p_value_normal_m = buffer_m.template read<u_value_type>();
1796 const auto p_value_normal_p = buffer_p.template read<u_value_type>();
1797
1798 const auto JxW = data.all_weights[q_index];
1799 const auto penalty_parameter = data.all_penalty_parameter[q_index];
1800 const auto normal = data.all_normals[q_index];
1801
1802 // jump(u) = u_m - u_p
1803 const auto u_value_jump = u_value_m - u_value_p;
1804
1805 // The expression for the average on the mortar interface is
1806 // avg(u).n = (u_m.normal_m + u_p.normal_p) * 0.5. Since we are
1807 // accessing only the value of normal_m, we use a minus sign here because
1808 // normal_p = - normal_m
1809 const auto u_grad_avg = (u_grad_normal_m - u_grad_normal_p) * 0.5;
1810
1811 // {{p}} = (p_m + p_p)/2
1812 const auto p_value_avg = (p_value_normal_m - p_value_normal_p) * 0.5;
1813
1814 typename FEPointIntegratorU::value_type u_grad_result = {};
1815 typename FEPointIntegratorU::value_type u_value_result = {};
1816 typename FEPointIntegratorP::value_type p_value_result = {};
1817
1818 // SIPG penalty parameter
1819 const double sigma = penalty_parameter * data.penalty_factor;
1820
1821 /* Contributions from viscous term */
1822 // - (n avg(v), jump(u))
1823 u_grad_result -= u_value_jump;
1824 // - (jump(v), avg(u) n)
1825 u_value_result -= u_grad_avg * this->kinematic_viscosity;
1826 // + (jump(v), jump(u))
1827 u_value_result += sigma * this->kinematic_viscosity * u_value_jump;
1828
1829 /* Contribution from pressure gradient term */
1830 // + (jump(v), avg(p) n)
1831 u_value_result += p_value_avg;
1832
1833 /* Contribution from velocity divergence term */
1834 // - (avg(q), jump(u) n)
1835 p_value_result -= 0.5 * contract(u_value_jump, normal);
1836
1837 // - (n avg(v), /2 jump(u))
1838 phi_u_m.submit_gradient(outer(u_grad_result, normal) *
1839 this->kinematic_viscosity * 0.5 * JxW,
1840 q);
1841 phi_u_m.submit_value(u_value_result * JxW, q);
1842 phi_p_m.submit_value(p_value_result * JxW, q);
1843 }
1844
1845 AssertDimension(buffer.size(),
1846 fe_sub_u.n_dofs_per_cell() + fe_sub_p.n_dofs_per_cell());
1847
1848 ArrayView<Number> buffer_u(buffer.data() + 0, fe_sub_u.n_dofs_per_cell());
1849 ArrayView<Number> buffer_p(buffer.data() + fe_sub_u.n_dofs_per_cell(),
1850 fe_sub_p.n_dofs_per_cell());
1851
1852 this->phi_u_m.test_and_sum(buffer_u,
1853 EvaluationFlags::values |
1854 EvaluationFlags::gradients);
1855 this->phi_p_m.test_and_sum(buffer_p, EvaluationFlags::values);
1856}
