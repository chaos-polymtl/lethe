11/**
12 * @brief Manage removing repetitions and adding new contact pairs to the
13 * contact containers when particles are exchanged between processors. If the
14 * contact pair (in adjacent particles containers) does not exist in the output
15 * of the new (current step) broad search, it is removed from the contact pair
16 * (adjacent containers), since it means that the contact is being handled by
17 * another processor. If the pair exists in the output of the new broad search,
18 * it is removed from the output of the broad search, as the contact is already
19 * being processed. This process is performed for many types of particle-object
20 * pairs : local-local particle-particle pairs, local-ghost particle-particle
21 * pairs and particle-wall pairs, particle-floating wall pairs and
22 * particle-floating mesh contacts pairs. Note : contact_type is important for
23 * local or ghost particle-particle contact only
24 *
25 * @tparam pairs_structure Adjacent particle-object pairs container type.
26 * @tparam candidates_structure Particle-object contact pairs container type.
27 * @tparam contact_type Label of contact type to apply proper manipulation of
28 * contact removal in containers.
29 *
30 * @param[in,out] pairs_in_contact adjacent particle-object pairs from fine
31 * search at previous time step.
32 * @param[in,out] contact_candidates particle-object contact pairs from board
33 * search at current time step.
34 */

36 typename pairs_structure,
37 typename candidates_structure,
38 ContactType contact_type>

40update_fine_search_candidates(pairs_structure &pairs_in_contact,
41 candidates_structure &contact_candidates);
