6template <int dim, int spacedim>
7CylinderGrid<dim, spacedim>::CylinderGrid(const std::string &grid_type,
8 const std::string &grid_arguments)
9{
10 if constexpr (!(dim == 3 && spacedim == 3))
11 {
12 AssertThrow(
13 false,
14 ExcMessage(
15 "Custom cylinder mesh is only supported in 3D space with 3D elements."));
16 return;
17 }
18
19 this->grid_arguments = grid_arguments;
20 if (grid_type == "cylinder_classic")
21 this->cylinder_type = CylinderType::classic;
22 else if (grid_type == "cylinder_balanced")
23 this->cylinder_type = CylinderType::balanced;
24 else if (grid_type == "cylinder_squared")
25 this->cylinder_type = CylinderType::squared;
26 else if (grid_type == "cylinder_regularized")
27 this->cylinder_type = CylinderType::regularized;
28 else
29 AssertThrow(
30 false,
31 ExcMessage(
32 "Unsupported custom cylinder, the choices are: cylinder_classic|cylinder_balanced|cylinder_squared|cylinder_regularized"));
33
34
35 // Separate arguments of the string
36 std::vector<std::string> arguments;
37 std::stringstream s_stream(grid_arguments);
38 while (s_stream.good())
39 {
40 std::string substr;
41 getline(s_stream, substr, ':');
42 arguments.push_back(substr);
43 }
44
45 // Arguments declaration
46 if (arguments.size() != 3)
47 {
48 AssertThrow(
49 false,
50 ExcMessage(
51 "Mandatory cylinder parameters are (x subdivisions: radius : half height)"));
52 }
53 else
54 {
55 std::vector<double> arguments_double =
56 Utilities::string_to_double(arguments);
57 this->subdivisions = static_cast<int>(arguments_double[0]);
58 this->radius = arguments_double[1];
59 this->half_height = arguments_double[2];
60 }
61}

64template <>
65void
66CylinderGrid<3, 3>::make_grid(Triangulation<3, 3> &triangulation)
67{
68 if (cylinder_type == CylinderType::classic)
69 {
70 // Create a subdivided cylinder from deal.ii
71 GridGenerator::subdivided_cylinder(triangulation,
72 subdivisions,
73 radius,
74 half_height);
75 }
76 else
77 {
78 // Create a temporary 2D mesh
79 Triangulation<2, 2> temporary_triangulation;
80
81 // Create a spherical manifold for 2D mesh
82 Point<2> center(0.0, 0.0);
83 const SphericalManifold<2, 2> m0(center);
84
85 if (cylinder_type == CylinderType::regularized ||
86 cylinder_type == CylinderType::squared)
87 {
88 // Create a square mesh
89 double real_radius = radius * std::sin(M_PI_4);
90 GridGenerator::hyper_cube(temporary_triangulation,
91 -real_radius,
92 real_radius,
93 true);
94
95 // Assign boundary 0 to perimeter as for cylinder
96 for (const auto &cell :
97 temporary_triangulation.active_cell_iterators())
98 {
99 if (cell->is_locally_owned())
100 {
101 // Looping through all the faces of the cell
102 for (const auto &face : cell->face_iterators())
103 {
104 // Check to see if the face is located at boundary
105 if (face->at_boundary())
106 {
107 face->set_boundary_id(0);
108 }
109 }
110 }
111 }
112 }
113 else if (cylinder_type == CylinderType::balanced)
114 {
115 GridGenerator::hyper_ball_balanced(temporary_triangulation,
116 center,
117 radius);
118 }
119
120 temporary_triangulation.reset_all_manifolds();
121 temporary_triangulation.set_all_manifold_ids_on_boundary(0);
122 temporary_triangulation.set_manifold(0, m0);
123
124 if (cylinder_type == CylinderType::regularized)
125 {
126 // Pre-refinement to reduce mesh size at corners before
127 // reinitialization
128 temporary_triangulation.refine_global(2);
129 GridTools::regularize_corner_cells(temporary_triangulation);
130
131 // Flatten the triangulation
132 Triangulation<2, 2> flat_temporary_triangulation;
133 flat_temporary_triangulation.copy_triangulation(
134 temporary_triangulation);
135 temporary_triangulation.clear();
136 GridGenerator::flatten_triangulation(flat_temporary_triangulation,
137 temporary_triangulation);
138 }
139
140 // Extrude the 2D temporary mesh to 3D cylinder
141 GridGenerator::extrude_triangulation(temporary_triangulation,
142 subdivisions + 1,
143 2.0 * half_height,
144 triangulation,
145 true);
146
147 // Rotate mesh in x-axis and set the (0,0,0) at the barycenter
148 // to be comparable to dealii cylinder meshes
149 Tensor<1, 3> axis_vector({0.0, 1.0, 0.0});
150 GridTools::rotate(axis_vector, M_PI_2, triangulation);
151 Tensor<1, 3> shift_vector({-half_height, 0.0, 0.0});
152 GridTools::shift(shift_vector, triangulation);
153
154 // Force the manifold id to be zero in the case of the balanced
155 // cylinder
156 if (cylinder_type == CylinderType::balanced)
157 triangulation.reset_manifold(1);
158
159 // Add a cylindrical manifold on the final unrefined mesh
160 const CylindricalManifold<3, 3> m1(0);
161 triangulation.set_manifold(0, m1);
162 }
163}
