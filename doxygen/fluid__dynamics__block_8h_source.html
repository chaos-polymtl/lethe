272template <class BSPreconditioner>
273void
274BlockSchurPreconditioner<BSPreconditioner>::vmult(
275 GlobalBlockVectorType &dst,
276 const GlobalBlockVectorType &src) const
277{
278 // TimerOutput computing_timer(std::cout,
279 // TimerOutput::summary,
280 // TimerOutput::wall_times);
281
282 GlobalVectorType utmp(src.block(0));
283 {
284 // computing_timer.enter_section("Pressure");
285 SolverControl solver_control(
286 linear_solver_parameters.max_iterations,
287 std::max(
288 1e-3 * src.block(1).l2_norm(),
289 // linear_solver_parameters.relative_residual*src.block(0).l2_norm(),
290 linear_solver_parameters.minimum_residual));
291 SolverCG<GlobalVectorType> cg(solver_control);
292
293 dst.block(1) = 0.0;
294 cg.solve(pressure_mass_matrix,
295 dst.block(1),
296 src.block(1),
297 *pmass_preconditioner);
298 dst.block(1) *= -(kinematic_viscosity + gamma);
299 // computing_timer.exit_section("Pressure");
300 }
301
302 {
303 // computing_timer.enter_section("Operations");
304 stokes_matrix.block(0, 1).vmult(utmp, dst.block(1));
305 utmp *= -1.0;
306 utmp += src.block(0);
307 // computing_timer.exit_section("Operations");
308 }
309 {
310 // computing_timer.enter_section("A Matrix");
311 SolverControl solver_control(
312 linear_solver_parameters.max_iterations,
313 std::max(1e-1 * src.block(0).l2_norm(),
314 linear_solver_parameters.minimum_residual));
315
316
317
318 // TrilinosWrappers::SolverBicgstab solver(solver_control);
319#ifndef LETHE_USE_LDV
320 TrilinosWrappers::SolverGMRES solver(solver_control);
321#else
322 SolverGMRES<GlobalVectorType> solver(solver_control);
323#endif
324
325 // A_inverse.solve(stokes_matrix.block(0, 0),dst.block(0), utmp,
326 // mp_preconditioner);
327 solver.solve(stokes_matrix.block(0, 0),
328 dst.block(0),
329 utmp,
330 *amat_preconditioner);
331 // computing_timer.exit_section("A Matrix");
332 }
333}

356template <class PreconditionerMp>
357BlockDiagPreconditioner<PreconditionerMp>::BlockDiagPreconditioner(
358 const TrilinosWrappers::BlockSparseMatrix &S,
359 const PreconditionerMp &Mppreconditioner,
360 SolverControl &A_parameters)
361 : stokes_matrix(S)
362 , mp_preconditioner(Mppreconditioner)
363 , A_inverse(A_parameters)
364{
365 TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(0,
366 1e-12,
367 1,
368 0);
369 amat_preconditioner.initialize(stokes_matrix.block(0, 0),
370 preconditionerOptions);
371
372 TrilinosWrappers::PreconditionILU::AdditionalData pmat_preconditionerOptions(
373 0, 1e-12, 1, 0);
374 TrilinosWrappers::PreconditionILU pmat_preconditioner;
375 pmat_preconditioner.initialize(stokes_matrix.block(1, 1),
376 pmat_preconditionerOptions);
377}

379template <class PreconditionerMp>
380void
381BlockDiagPreconditioner<PreconditionerMp>::vmult(
382 GlobalBlockVectorType &dst,
383 const GlobalBlockVectorType &src) const
384{
385 {
386 SolverControl solver_control(100000, 1e-12);
387 TrilinosWrappers::SolverCG cg(solver_control);
388
389 cg.solve(stokes_matrix.block(1, 1),
390 dst.block(1),
391 src.block(1),
392 pmat_preconditioner);
393 }
394 {
395 SolverControl solver_control(10000, 1e-12);
396
397 TrilinosWrappers::SolverGMRES solver(solver_control);
398 solver.solve(stokes_matrix.block(0, 0),
399 dst.block(0),
400 src.block(0),
401 amat_preconditioner);
402 }
403}
