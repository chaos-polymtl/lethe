6void
7PhysicalPropertiesManager::establish_fields_required_by_model(
8 PhysicalPropertyModel &model)
9{
10 // Loop through the map. The use of or (||) is there to ensure
11 // that if a field is already required, it won't be erased.
12 for (auto &f : required_fields)
13 {
14 f.second = f.second || model.depends_on(f.first);
15 }
16}

30void
31PhysicalPropertiesManager::initialize(
32 Parameters::PhysicalProperties physical_properties)
33{
34 // Keep a copy of the physical properties used to build the physical property
35 // manager
36 physical_properties_parameters = physical_properties;
37
38 is_initialized = true;
39
40 number_of_fluids = physical_properties.number_of_fluids;
41 number_of_solids = physical_properties.number_of_solids;
42 number_of_material_interactions =
43 physical_properties.number_of_material_interactions;
44 fluid_fluid_interactions_with_material_interaction_ids =
45 physical_properties.fluid_fluid_interactions_with_material_interaction_ids;
46 fluid_solid_interactions_with_material_interaction_ids =
47 physical_properties.fluid_solid_interactions_with_material_interaction_ids;
48 reference_temperature = physical_properties.reference_temperature;
49
50 non_newtonian_flow = false;
51 phase_change = false;
52 constant_density = true;
53 constant_surface_tension = true;
54
55 required_fields[field::temperature] = false;
56 required_fields[field::temperature_p1] = false;
57 required_fields[field::temperature_p2] = false;
58 required_fields[field::shear_rate] = false;
59 required_fields[field::pressure] = false;
60 required_fields[field::phase_order_cahn_hilliard] = false;
61 required_fields[field::levelset] = false;
62 required_fields[field::tracer_concentration] = false;
63
64 // For each fluid, declare the physical properties
65 for (unsigned int f = 0; f < number_of_fluids; ++f)
66 {
67 density.push_back(
68 DensityModel::model_cast(physical_properties.fluids[f]));
69 establish_fields_required_by_model(*density[f]);
70
71 // Store an indicator for the density to indicate if it is not constant
72 // This indicator is used elsewhere in the code to throw assertions
73 // if non-constant density is not implemented in a post-processing utility
74 if (!density.back()->is_constant_density_model())
75 constant_density = false;
76
77 specific_heat.push_back(
78 SpecificHeatModel::model_cast(physical_properties.fluids[f]));
79 establish_fields_required_by_model(*specific_heat[f]);
80
81 // Store an indicator that a phase change model is present
82 phase_change =
83 phase_change || physical_properties.fluids[f].specific_heat_model ==
84 Parameters::Material::SpecificHeatModel::phase_change;
85
86 thermal_conductivity.push_back(
87 ThermalConductivityModel::model_cast(physical_properties.fluids[f]));
88 establish_fields_required_by_model(*thermal_conductivity[f]);
89
90 rheology.push_back(
91 RheologicalModel::model_cast(physical_properties.fluids[f]));
92 this->establish_fields_required_by_model(*rheology[f]);
93 rheology[f]->set_dynamic_viscosity(density[f]->get_density_ref());
94
95 tracer_diffusivity.push_back(
96 TracerDiffusivityModel::model_cast(physical_properties.fluids[f]));
97 establish_fields_required_by_model(*tracer_diffusivity[f]);
98
99 tracer_reaction_prefactor.push_back(
100 TracerReactionPrefactorModel::model_cast(
101 physical_properties.fluids[f]));
102 establish_fields_required_by_model(*tracer_reaction_prefactor[f]);
103
104 tracer_reaction_order.push_back(
105 physical_properties.fluids[f].tracer_reaction_order);
106
107 thermal_expansion.push_back(
108 ThermalExpansionModel::model_cast(physical_properties.fluids[f]));
109 establish_fields_required_by_model(*thermal_expansion[f]);
110
111 phase_change_parameters.push_back(
112 physical_properties.fluids[f].phase_change_parameters);
113
114 if (rheology.back()->is_non_newtonian_rheological_model())
115 non_newtonian_flow = true;
116 }
117
118 // For each solid, append the physical properties
119 for (unsigned int s = 0; s < number_of_solids; ++s)
120 {
121 density.push_back(
122 DensityModel::model_cast(physical_properties.solids[s]));
123 establish_fields_required_by_model(*density[s + number_of_fluids]);
124
125 // Store an indicator for the density to indicate if it is not constant
126 // This indicator is used elsewhere in the code to throw assertions
127 // if non-constant density is not implemented in a post-processing utility
128 if (!density.back()->is_constant_density_model())
129 constant_density = false;
130
131 specific_heat.push_back(
132 SpecificHeatModel::model_cast(physical_properties.solids[s]));
133 establish_fields_required_by_model(*specific_heat[s + number_of_fluids]);
134
135 thermal_conductivity.push_back(
136 ThermalConductivityModel::model_cast(physical_properties.solids[s]));
137 establish_fields_required_by_model(
138 *thermal_conductivity[s + number_of_fluids]);
139
140 rheology.push_back(
141 RheologicalModel::model_cast(physical_properties.solids[s]));
142 this->establish_fields_required_by_model(*rheology[s + number_of_fluids]);
143 rheology.back()->set_dynamic_viscosity(density.back()->get_density_ref());
144
145 tracer_diffusivity.push_back(
146 TracerDiffusivityModel::model_cast(physical_properties.solids[s]));
147 establish_fields_required_by_model(
148 *tracer_diffusivity[s + number_of_fluids]);
149
150 tracer_reaction_prefactor.push_back(
151 TracerReactionPrefactorModel::model_cast(
152 physical_properties.solids[s]));
153 establish_fields_required_by_model(
154 *tracer_reaction_prefactor[s + number_of_fluids]);
155
156 tracer_reaction_order.push_back(
157 physical_properties.fluids[s].tracer_reaction_order);
158
159 thermal_expansion.push_back(
160 ThermalExpansionModel::model_cast(physical_properties.solids[s]));
161 establish_fields_required_by_model(
162 *thermal_expansion[s + number_of_fluids]);
163 }
164
165 // For each pair of interacting materials (fluid-fluid or fluid-solid), append
166 // physical properties
167 for (unsigned int i = 0; i < number_of_material_interactions; ++i)
168 {
169 surface_tension.push_back(SurfaceTensionModel::model_cast(
170 physical_properties.material_interactions[i]));
171 establish_fields_required_by_model(*surface_tension[i]);
172 if (!surface_tension.back()->is_constant_surface_tension_model())
173 constant_surface_tension = false;
174
175 mobility_cahn_hilliard.push_back(MobilityCahnHilliardModel::model_cast(
176 physical_properties.material_interactions[i]));
177 establish_fields_required_by_model(*mobility_cahn_hilliard[i]);
178 }
179}
