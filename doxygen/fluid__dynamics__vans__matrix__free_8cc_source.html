40template <int dim>
41void
42MFNavierStokesVANSPreconditionGMG<dim>::initialize(
43 const std::shared_ptr<SimulationControl> &simulation_control,
44 FlowControl<dim> &flow_control,
45 const VectorType &present_solution,
46 const VectorType &time_derivative_previous_solutions,
47 const ParticleProjector<dim> &void_fraction_manager)
48{
49 if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
50 .preconditioner == Parameters::LinearSolver::PreconditionerType::lsmg)
51 {
52 AssertThrow(false, ExcNotImplemented());
53 }
54 else if (this->simulation_parameters.linear_solver
55 .at(PhysicsID::fluid_dynamics)
56 .preconditioner ==
57 Parameters::LinearSolver::PreconditionerType::gcmg)
58 {
59 const unsigned int min_level = this->minlevel;
60 const unsigned int max_level = this->maxlevel;
61
62 this->void_fraction_dof_handlers.resize(min_level, max_level);
63
64 for (unsigned int l = min_level; l <= max_level; l++)
65 {
66 this->void_fraction_dof_handlers[l].reinit(
67 this->dof_handlers[l].get_triangulation());
68 this->void_fraction_dof_handlers[l].distribute_dofs(
69 void_fraction_manager.dof_handler.get_fe());
70 }
71
72 this->transfers_void_fraction.resize(min_level, max_level);
73
74 for (unsigned int l = min_level; l < max_level; l++)
75 {
76 this->transfers_void_fraction[l + 1].reinit(
77 this->void_fraction_dof_handlers[l + 1],
78 this->void_fraction_dof_handlers[l],
79 {},
80 {});
81 }
82
83 this->mg_transfer_gc_void_fraction =
84 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
85 this->transfers_void_fraction);
86
87#if DEAL_II_VERSION_GTE(9, 7, 0)
88 this->mg_transfer_gc_void_fraction->build(
89 void_fraction_manager.dof_handler, [&](const auto l, auto &vec) {
90 vec.reinit(
91 this->void_fraction_dof_handlers[l].locally_owned_dofs(),
92 DoFTools::extract_locally_active_dofs(
93 this->void_fraction_dof_handlers[l]),
94 this->void_fraction_dof_handlers[l].get_mpi_communicator());
95 });
96#endif
97
98 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
99 mg_void_fraction_solution(this->minlevel, this->maxlevel);
100
101
102 // A deal.II vector is required here, so we take the deal.II vector
103 // solution from the void fraction manager instead of the trilinos vector
104 // one.
105 this->mg_transfer_gc_void_fraction->interpolate_to_mg(
106 void_fraction_manager.dof_handler,
107 mg_void_fraction_solution,
108 void_fraction_manager.void_fraction_solution);
109
110 for (unsigned int l = min_level; l <= max_level; l++)
111 {
112 mg_void_fraction_solution[l].update_ghost_values();
113
114 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
115 &(*this->mg_operators[l])))
116 mf_operator->compute_void_fraction(
117 mg_void_fraction_solution[l],
118 this->void_fraction_dof_handlers[l]);
119 }
120 }
121
122 MFNavierStokesPreconditionGMG<dim>::initialize(
123 simulation_control,
124 flow_control,
125 present_solution,
126 time_derivative_previous_solutions);
127}

129template <int dim>
130FluidDynamicsVANSMatrixFree<dim>::FluidDynamicsVANSMatrixFree(
131 CFDDEMSimulationParameters<dim> &param)
132 : FluidDynamicsMatrixFree<dim>(param.cfd_parameters)
133 , cfd_dem_simulation_parameters(param)
134 , particle_mapping(1)
135 , particle_handler(*this->triangulation,
136 particle_mapping,
137 DEM::CFDDEMProperties::n_properties)
138 , particle_projector(
139 &(*this->triangulation),
140 param.void_fraction,
141 this->cfd_dem_simulation_parameters.cfd_parameters.linear_solver.at(
142 PhysicsID::void_fraction),
143 &particle_handler,
144 this->cfd_dem_simulation_parameters.cfd_parameters.fem_parameters
145 .void_fraction_order,
146 this->cfd_dem_simulation_parameters.cfd_parameters.mesh.simplex,
147 this->pcout)
148 , has_periodic_boundaries(false)
149{
150 unsigned int n_pbc = 0;
151 for (auto const &[id, type] :
152 cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions.type)
153 {
154 if (type == BoundaryConditions::BoundaryType::periodic)
155 {
156 if (n_pbc++ > 1)
157 {
158 throw std::runtime_error(
159 "GLS VANS solver does not support more than one periodic boundary condition.");
160 }
161 else
162 {
163 has_periodic_boundaries = true;
164 }
165 }
166 }
167
168 // The default MatrixFree solver sets a system_operator. We override the
169 // Navier-Stokes operator with the volume-averaged Navier-Stokes operator.
170 this->system_operator = std::make_shared<VANSOperator<dim, double>>(
171 cfd_dem_simulation_parameters.cfd_dem);
172}

185template <int dim>
186void
187FluidDynamicsVANSMatrixFree<dim>::read_dem()
188{
189 std::string prefix =
190 this->cfd_dem_simulation_parameters.void_fraction->dem_file_name;
191
192 // Load checkpoint controller
193 std::string checkpoint_controller_object_filename =
194 prefix + ".checkpoint_controller";
195 std::ifstream iss_checkpoint_controller_obj(
196 checkpoint_controller_object_filename);
197 boost::archive::text_iarchive ia_checkpoint_controller_obj(
198 iss_checkpoint_controller_obj, boost::archive::no_header);
199
200 unsigned int checkpoint_id;
201 ia_checkpoint_controller_obj >> checkpoint_id;
202
203 // New prefix for the remaining files
204 prefix = prefix + "_" + Utilities::int_to_string(checkpoint_id);
205
206 // Gather particle serialization information
207 std::string particle_filename = prefix + ".particles";
208 std::ifstream input(particle_filename.c_str());
209 AssertThrow(input, ExcFileNotOpen(particle_filename));
210
211 std::string buffer;
212 std::getline(input, buffer);
213 std::istringstream iss(buffer);
214 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
215
216 // Create a temporary particle_handler with DEM properties
217 Particles::ParticleHandler<dim> temporary_particle_handler(
218 *this->triangulation, particle_mapping, DEM::DEMProperties::n_properties);
219
220 ia >> temporary_particle_handler;
221
222 const std::string filename = prefix + ".triangulation";
223 std::ifstream in(filename.c_str());
224 if (!in)
225 AssertThrow(false,
226 ExcMessage(
227 std::string(
228 "You are trying to restart a previous computation, "
229 "but the restart file <") +
230 filename + "> does not appear to exist!"));
231
232 if (auto parallel_triangulation =
233 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
234 &*this->triangulation))
235 {
236 try
237 {
238 this->triangulation->load(filename.c_str());
239
240 // Deserialize particles have the triangulation has been read
241 temporary_particle_handler.deserialize();
242 }
243 catch (...)
244 {
245 AssertThrow(false,
246 ExcMessage("Cannot open snapshot mesh file or read the"
247 "triangulation stored there."));
248 }
249
250 // Fill the existing particle handler using the temporary one
251 // This is done during the dynamic cast for the convert_particle_handler
252 // function which requires a pararallel::distributed::triangulation
253 convert_particle_handler<dim,
254 DEM::DEMProperties::PropertiesIndex,
255 DEM::CFDDEMProperties::PropertiesIndex>(
256 *parallel_triangulation, temporary_particle_handler, particle_handler);
257 }
258 else
259 {
260 throw std::runtime_error(
261 "The VANS application currently does not support "
262 "triangulations other than parallel::distributed");
263 }
264}

293template <int dim>
294void
295FluidDynamicsVANSMatrixFree<dim>::solve()
296{
297 this->computing_timer.enter_subsection("Read mesh, manifolds and particles");
298
299 read_mesh_and_manifolds(
300 *this->triangulation,
301 this->cfd_dem_simulation_parameters.cfd_parameters.mesh,
302 this->cfd_dem_simulation_parameters.cfd_parameters.manifolds_parameters,
303 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
304 .restart ||
305 this->cfd_dem_simulation_parameters.void_fraction->read_dem == true,
306 this->cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions);
307
308 if (this->cfd_dem_simulation_parameters.void_fraction->read_dem == true &&
309 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
310 .restart == false)
311 read_dem();
312
313 this->computing_timer.leave_subsection("Read mesh, manifolds and particles");
314
315 this->setup_dofs();
316
317 particle_projector.calculate_void_fraction(
318 this->simulation_control->get_current_time());
319
320 this->set_initial_condition(
321 this->simulation_parameters.initial_condition->type,
322 this->simulation_parameters.restart_parameters.restart);
323
324 // Only needed if other physics apart from fluid dynamics are enabled.
325 if (this->multiphysics->get_active_physics().size() > 1)
326 this->update_multiphysics_time_average_solution();
327
328 while (this->simulation_control->integrate())
329 {
330 if (this->forcing_function)
331 this->forcing_function->set_time(
332 this->simulation_control->get_current_time());
333
334 this->update_boundary_conditions();
335 this->multiphysics->update_boundary_conditions();
336
337 this->simulation_control->print_progression(this->pcout);
338 this->dynamic_flow_control();
339
340 if (!this->simulation_control->is_at_start())
341 {
342 this->refine_mesh();
343 }
344
345 if (is_bdf(this->simulation_control->get_assembly_method()))
346 {
347 this->computing_timer.enter_subsection(
348 "Calculate time derivative previous solutions");
349
350 this->calculate_time_derivative_previous_solutions();
351 this->time_derivative_previous_solutions.update_ghost_values();
352 this->system_operator->evaluate_time_derivative_previous_solutions(
353 this->time_derivative_previous_solutions);
354
355 this->computing_timer.leave_subsection(
356 "Calculate time derivative previous solutions");
357
358 if (this->simulation_parameters.flow_control.enable_flow_control)
359 this->system_operator->update_beta_force(
360 this->flow_control.get_beta());
361 }
362
363 // Calculate the void fraction and evaluate it within the matrix-free
364 // operator
365 {
366 TimerOutput::Scope t(this->computing_timer, "Calculate void fraction");
367 particle_projector.calculate_void_fraction(
368 this->simulation_control->get_current_time());
369
370 // The base matrix-free operator is not aware of the void fraction. We
371 // must do a cast here to ensure that the operator is of the right type
372 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
373 this->system_operator.get()))
374 mf_operator->compute_void_fraction(
375 particle_projector.void_fraction_solution,
376 particle_projector.dof_handler);
377 }
378
379 this->iterate();
380 this->postprocess(false);
381 this->finish_time_step();
382
383 if (this->simulation_parameters.timer.type ==
384 Parameters::Timer::Type::iteration)
385 this->print_mg_setup_times();
386 }
387
388 if (this->simulation_parameters.timer.type == Parameters::Timer::Type::end)
389 this->print_mg_setup_times();
390
391 this->finish_simulation();
392}
