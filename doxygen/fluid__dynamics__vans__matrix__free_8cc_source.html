37template <int dim>
38void
39MFNavierStokesVANSPreconditionGMG<dim>::initialize(
40 const std::shared_ptr<SimulationControl> &simulation_control,
41 FlowControl<dim> &flow_control,
42 const VectorType &present_solution,
43 const VectorType &time_derivative_previous_solutions,
44 const VectorType &time_derivative_void_fraction,
45 const ParticleProjector<dim> &particle_projector)
46{
47 if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
48 .preconditioner == Parameters::LinearSolver::PreconditionerType::lsmg)
49 {
50 AssertThrow(false, ExcNotImplemented());
51 }
52 else if (this->simulation_parameters.linear_solver
53 .at(PhysicsID::fluid_dynamics)
54 .preconditioner ==
55 Parameters::LinearSolver::PreconditionerType::gcmg)
56 {
57 const unsigned int min_level = this->minlevel;
58 const unsigned int max_level = this->maxlevel;
59
60 // Resize all of the dof handlers related to every field for the
61 // particle-fluid coupling
62 this->void_fraction_dof_handlers.resize(min_level, max_level);
63 this->pf_force_dof_handlers.resize(min_level, max_level);
64 this->pf_drag_dof_handlers.resize(min_level, max_level);
65 this->particle_velocity_dof_handlers.resize(min_level, max_level);
66 this->momentum_transfer_coefficient_dof_handlers.resize(min_level,
67 max_level);
68
69
70 for (unsigned int l = min_level; l <= max_level; l++)
71 {
72 // Void fraction
73 this->void_fraction_dof_handlers[l].reinit(
74 this->dof_handlers[l].get_triangulation());
75 this->void_fraction_dof_handlers[l].distribute_dofs(
76 particle_projector.dof_handler.get_fe());
77
78 // Particle-fluid force
79 this->pf_force_dof_handlers[l].reinit(
80 this->dof_handlers[l].get_triangulation());
81 this->pf_force_dof_handlers[l].distribute_dofs(
82 particle_projector.fluid_force_on_particles_two_way_coupling
83 .dof_handler.get_fe());
84
85 // Particle-fluid drag
86 this->pf_drag_dof_handlers[l].reinit(
87 this->dof_handlers[l].get_triangulation());
88 this->pf_drag_dof_handlers[l].distribute_dofs(
89 particle_projector.fluid_drag_on_particles.dof_handler.get_fe());
90
91 // Particle velocity field
92 this->particle_velocity_dof_handlers[l].reinit(
93 this->dof_handlers[l].get_triangulation());
94 this->particle_velocity_dof_handlers[l].distribute_dofs(
95 particle_projector.particle_velocity.dof_handler.get_fe());
96
97 // Momentum exchange coefficient
98 this->momentum_transfer_coefficient_dof_handlers[l].reinit(
99 this->dof_handlers[l].get_triangulation());
100 this->momentum_transfer_coefficient_dof_handlers[l].distribute_dofs(
101 particle_projector.momentum_transfer_coefficient.dof_handler
102 .get_fe());
103 }
104
105 this->transfers_void_fraction.resize(min_level, max_level);
106 this->transfers_pf_force.resize(min_level, max_level);
107 this->transfers_pf_drag.resize(min_level, max_level);
108 this->transfers_particle_velocity.resize(min_level, max_level);
109 this->transfers_momentum_transfer_coefficient.resize(min_level,
110 max_level);
111
112
113 for (unsigned int l = min_level; l < max_level; l++)
114 {
115 this->transfers_void_fraction[l + 1].reinit(
116 this->void_fraction_dof_handlers[l + 1],
117 this->void_fraction_dof_handlers[l],
118 {},
119 {});
120
121 this->transfers_pf_force[l + 1].reinit(
122 this->pf_force_dof_handlers[l + 1],
123 this->pf_force_dof_handlers[l],
124 {},
125 {});
126
127 this->transfers_pf_drag[l + 1].reinit(
128 this->pf_drag_dof_handlers[l + 1],
129 this->pf_drag_dof_handlers[l],
130 {},
131 {});
132
133 this->transfers_particle_velocity[l + 1].reinit(
134 this->particle_velocity_dof_handlers[l + 1],
135 this->particle_velocity_dof_handlers[l],
136 {},
137 {});
138
139 this->transfers_momentum_transfer_coefficient[l + 1].reinit(
140 this->momentum_transfer_coefficient_dof_handlers[l + 1],
141 this->momentum_transfer_coefficient_dof_handlers[l],
142 {},
143 {});
144 }
145
146 // Make the transfer operators for every field we will transfer
147 this->mg_transfer_gc_void_fraction =
148 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
149 this->transfers_void_fraction);
150
151 this->mg_transfer_gc_pf_force =
152 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
153 this->transfers_pf_force);
154
155 this->mg_transfer_gc_pf_drag =
156 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
157 this->transfers_pf_drag);
158
159 this->mg_transfer_gc_particle_velocity =
160 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
161 this->transfers_particle_velocity);
162
163 this->mg_transfer_gc_momentum_transfer_coefficient =
164 std::make_shared<MFNavierStokesVANSPreconditionGMG::GCTransferType>(
165 this->transfers_momentum_transfer_coefficient);
166
167
168 // Build transfer operator for every field
169 this->mg_transfer_gc_void_fraction->build(
170 particle_projector.dof_handler, [&](const auto l, auto &vec) {
171 vec.reinit(
172 this->void_fraction_dof_handlers[l].locally_owned_dofs(),
173 DoFTools::extract_locally_active_dofs(
174 this->void_fraction_dof_handlers[l]),
175 this->void_fraction_dof_handlers[l].get_mpi_communicator());
176 });
177
178 this->mg_transfer_gc_pf_force->build(
179 particle_projector.fluid_force_on_particles_two_way_coupling
180 .dof_handler,
181 [&](const auto l, auto &vec) {
182 vec.reinit(this->pf_force_dof_handlers[l].locally_owned_dofs(),
183 DoFTools::extract_locally_active_dofs(
184 this->pf_force_dof_handlers[l]),
185 this->pf_force_dof_handlers[l].get_mpi_communicator());
186 });
187
188 this->mg_transfer_gc_pf_force->build(
189 particle_projector.fluid_drag_on_particles.dof_handler,
190 [&](const auto l, auto &vec) {
191 vec.reinit(this->pf_drag_dof_handlers[l].locally_owned_dofs(),
192 DoFTools::extract_locally_active_dofs(
193 this->pf_drag_dof_handlers[l]),
194 this->pf_drag_dof_handlers[l].get_mpi_communicator());
195 });
196
197 this->mg_transfer_gc_particle_velocity->build(
198 particle_projector.particle_velocity.dof_handler,
199 [&](const auto l, auto &vec) {
200 vec.reinit(
201 this->particle_velocity_dof_handlers[l].locally_owned_dofs(),
202 DoFTools::extract_locally_active_dofs(
203 this->particle_velocity_dof_handlers[l]),
204 this->particle_velocity_dof_handlers[l].get_mpi_communicator());
205 });
206
207 this->mg_transfer_gc_momentum_transfer_coefficient->build(
208 particle_projector.momentum_transfer_coefficient.dof_handler,
209 [&](const auto l, auto &vec) {
210 vec.reinit(this->momentum_transfer_coefficient_dof_handlers[l]
211 .locally_owned_dofs(),
212 DoFTools::extract_locally_active_dofs(
213 this->momentum_transfer_coefficient_dof_handlers[l]),
214 this->momentum_transfer_coefficient_dof_handlers[l]
215 .get_mpi_communicator());
216 });
217
218 // Create the MG Level Object for every field
219 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
220 mg_void_fraction_solution(this->minlevel, this->maxlevel);
221
222 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
223 mg_time_derivative_void_fraction(this->minlevel, this->maxlevel);
224
225 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
226 mg_pf_forces_solution(this->minlevel, this->maxlevel);
227
228 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
229 mg_pf_drag_solution(this->minlevel, this->maxlevel);
230
231 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
232 mg_particle_velocity_solution(this->minlevel, this->maxlevel);
233
234 MGLevelObject<MFNavierStokesVANSPreconditionGMG::MGVectorType>
235 mg_momentum_transfer_coefficient_solution(this->minlevel,
236 this->maxlevel);
237
238 // A deal.II vector is required here, so we take the deal.II vector
239 // solution from the particle projector instead of the Trillinos vector
240 // one. This is done for every field.
241
242 // Void fraction
243 this->mg_transfer_gc_void_fraction->interpolate_to_mg(
244 particle_projector.dof_handler,
245 mg_void_fraction_solution,
246 particle_projector.void_fraction_solution);
247
248 this->mg_transfer_gc_void_fraction->interpolate_to_mg(
249 particle_projector.dof_handler,
250 mg_time_derivative_void_fraction,
251 time_derivative_void_fraction);
252
253 particle_projector.fluid_force_on_particles_two_way_coupling
254 .particle_field_solution.update_ghost_values();
255
256 // Particle-fluid force
257 this->mg_transfer_gc_pf_force->interpolate_to_mg(
258 particle_projector.fluid_force_on_particles_two_way_coupling
259 .dof_handler,
260 mg_pf_forces_solution,
261 particle_projector.fluid_force_on_particles_two_way_coupling
262 .particle_field_solution);
263
264 // Particle-fluid drag
265 particle_projector.fluid_drag_on_particles.particle_field_solution
266 .update_ghost_values();
267
268 this->mg_transfer_gc_pf_force->interpolate_to_mg(
269 particle_projector.fluid_drag_on_particles.dof_handler,
270 mg_pf_drag_solution,
271 particle_projector.fluid_drag_on_particles.particle_field_solution);
272
273 // Particle-velocity
274 particle_projector.particle_velocity.particle_field_solution
275 .update_ghost_values();
276
277 this->mg_transfer_gc_pf_force->interpolate_to_mg(
278 particle_projector.particle_velocity.dof_handler,
279 mg_particle_velocity_solution,
280 particle_projector.particle_velocity.particle_field_solution);
281
282 // Momentum transfer coefficient
283 particle_projector.momentum_transfer_coefficient.particle_field_solution
284 .update_ghost_values();
285
286 this->mg_transfer_gc_momentum_transfer_coefficient->interpolate_to_mg(
287 particle_projector.momentum_transfer_coefficient.dof_handler,
288 mg_momentum_transfer_coefficient_solution,
289 particle_projector.momentum_transfer_coefficient
290 .particle_field_solution);
291
292
293 for (unsigned int l = min_level; l <= max_level; l++)
294 {
295 mg_void_fraction_solution[l].update_ghost_values();
296 mg_time_derivative_void_fraction[l].update_ghost_values();
297 mg_pf_forces_solution[l].update_ghost_values();
298 mg_pf_drag_solution[l].update_ghost_values();
299 mg_particle_velocity_solution[l].update_ghost_values();
300 mg_momentum_transfer_coefficient_solution[l].update_ghost_values();
301
302 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
303 &(*this->mg_operators[l])))
304 {
305 mf_operator->compute_void_fraction(
306 this->void_fraction_dof_handlers[l],
307 mg_void_fraction_solution[l],
308 mg_time_derivative_void_fraction[l]);
309
310 mf_operator->compute_particle_fluid_interaction(
311 this->pf_force_dof_handlers[l],
312 mg_pf_forces_solution[l],
313 this->pf_drag_dof_handlers[l],
314 mg_pf_drag_solution[l],
315 this->particle_velocity_dof_handlers[l],
316 mg_particle_velocity_solution[l],
317 this->momentum_transfer_coefficient_dof_handlers[l],
318 mg_momentum_transfer_coefficient_solution[l]);
319 }
320 }
321 }
322
323 MFNavierStokesPreconditionGMG<dim>::initialize(
324 simulation_control,
325 flow_control,
326 present_solution,
327 time_derivative_previous_solutions);
328}

330template <int dim>
331FluidDynamicsVANSMatrixFree<dim>::FluidDynamicsVANSMatrixFree(
332 CFDDEMSimulationParameters<dim> &param)
333 : FluidDynamicsMatrixFree<dim>(param.cfd_parameters)
334 , cfd_dem_simulation_parameters(param)
335 , particle_mapping(1)
336 , particle_handler(*this->triangulation,
337 particle_mapping,
338 DEM::CFDDEMProperties::n_properties)
339 , particle_projector(
340 &(*this->triangulation),
341 param.void_fraction,
342 this->cfd_dem_simulation_parameters.cfd_parameters.linear_solver.at(
343 PhysicsID::void_fraction),
344 &particle_handler,
345 this->cfd_dem_simulation_parameters.cfd_parameters.fem_parameters
346 .void_fraction_order,
347 this->cfd_dem_simulation_parameters.cfd_parameters.mesh.simplex,
348 this->pcout)
349 , has_periodic_boundaries(false)
350{
351 unsigned int n_pbc = 0;
352 for (auto const &[id, type] :
353 cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions.type)
354 {
355 if (type == BoundaryConditions::BoundaryType::periodic)
356 {
357 if (n_pbc++ > 1)
358 {
359 throw std::runtime_error(
360 "GLS VANS solver does not support more than one periodic boundary condition.");
361 }
362 else
363 {
364 has_periodic_boundaries = true;
365 }
366 }
367 }
368
369 // The default MatrixFree solver sets a system_operator. We override the
370 // Navier-Stokes operator with the volume-averaged Navier-Stokes operator.
371 this->system_operator = std::make_shared<VANSOperator<dim, double>>(
372 cfd_dem_simulation_parameters.cfd_dem);
373}

392template <int dim>
393void
394FluidDynamicsVANSMatrixFree<dim>::read_dem()
395{
396 std::string prefix =
397 this->cfd_dem_simulation_parameters.void_fraction->dem_file_name;
398
399 // Load checkpoint controller
400 std::string checkpoint_controller_object_filename =
401 prefix + ".checkpoint_controller";
402 std::ifstream iss_checkpoint_controller_obj(
403 checkpoint_controller_object_filename);
404 boost::archive::text_iarchive ia_checkpoint_controller_obj(
405 iss_checkpoint_controller_obj, boost::archive::no_header);
406
407 unsigned int checkpoint_id;
408 ia_checkpoint_controller_obj >> checkpoint_id;
409
410 // New prefix for the remaining files
411 prefix = prefix + "_" + Utilities::int_to_string(checkpoint_id);
412
413 // Gather particle serialization information
414 std::string particle_filename = prefix + ".particles";
415 std::ifstream input(particle_filename.c_str());
416 AssertThrow(input, ExcFileNotOpen(particle_filename));
417
418 std::string buffer;
419 std::getline(input, buffer);
420 std::istringstream iss(buffer);
421 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
422
423 // Create a temporary particle_handler with DEM properties
424 Particles::ParticleHandler<dim> temporary_particle_handler(
425 *this->triangulation, particle_mapping, DEM::DEMProperties::n_properties);
426
427 ia >> temporary_particle_handler;
428
429 const std::string filename = prefix + ".triangulation";
430 std::ifstream in(filename.c_str());
431 if (!in)
432 AssertThrow(false,
433 ExcMessage(
434 std::string(
435 "You are trying to restart a previous computation, "
436 "but the restart file <") +
437 filename + "> does not appear to exist!"));
438
439 if (auto parallel_triangulation =
440 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
441 &*this->triangulation))
442 {
443 try
444 {
445 this->triangulation->load(filename.c_str());
446
447 // Deserialize particles have the triangulation has been read
448 temporary_particle_handler.deserialize();
449 }
450 catch (...)
451 {
452 AssertThrow(false,
453 ExcMessage("Cannot open snapshot mesh file or read the"
454 "triangulation stored there."));
455 }
456
457 // Fill the existing particle handler using the temporary one
458 // This is done during the dynamic cast for the convert_particle_handler
459 // function which requires a pararallel::distributed::triangulation
460 convert_particle_handler<dim,
461 DEM::DEMProperties::PropertiesIndex,
462 DEM::CFDDEMProperties::PropertiesIndex>(
463 *parallel_triangulation, temporary_particle_handler, particle_handler);
464
465 // Exchange the ghost particles so that the particle handler is ready to
466 // be used to calculate the initial conditions of the void fraction or
467 // anything that requires the ghost particles.
468 particle_handler.exchange_ghost_particles(true);
469 }
470 else
471 {
472 throw std::runtime_error(
473 "The VANS application currently does not support "
474 "triangulations other than parallel::distributed");
475 }
476}

478template <int dim>
479void
480FluidDynamicsVANSMatrixFree<dim>::assemble_system_rhs()
481{
482 // First we update the particle-fluid coupling terms.
483 // The base matrix-free operator is not aware of the various VANS
484 // coupling terms. We must do a cast here to ensure that the operator is
485 // of the right type.
486 if (auto mf_operator =
487 dynamic_cast<VANSOperator<dim, double> *>(this->system_operator.get()))
488 {
489 TimerOutput::Scope t(this->computing_timer,
490 "Prepare MF operator for VANS");
491
492 // If the coupling is implicit, we need to carry out the projection of
493 // the particle information onto the CFD mesh at every RHS evaluation
494 // since the momentum exchange term evolves.
495 if (cfd_dem_simulation_parameters.cfd_dem.drag_coupling ==
496 Parameters::DragCoupling::fully_implicit)
497 particle_projector.calculate_particle_fluid_forces_projection(
498 this->cfd_dem_simulation_parameters.cfd_dem,
499 *this->dof_handler,
500 this->evaluation_point,
501 *this->previous_solutions,
502 this->cfd_dem_simulation_parameters.dem_parameters
503 .lagrangian_physical_properties.g,
504 NavierStokesScratchData<dim>(
505 this->simulation_control,
506 this->simulation_parameters.physical_properties_manager,
507 *this->fe,
508 *this->cell_quadrature,
509 *this->mapping,
510 *this->face_quadrature));
511
512 mf_operator->compute_particle_fluid_interaction(
513 particle_projector.fluid_force_on_particles_two_way_coupling
514 .dof_handler,
515 particle_projector.fluid_force_on_particles_two_way_coupling
516 .particle_field_solution,
517 particle_projector.fluid_drag_on_particles.dof_handler,
518 particle_projector.fluid_drag_on_particles.particle_field_solution,
519 particle_projector.particle_velocity.dof_handler,
520 particle_projector.particle_velocity.particle_field_solution,
521 particle_projector.momentum_transfer_coefficient.dof_handler,
522 particle_projector.momentum_transfer_coefficient
523 .particle_field_solution);
524 }
525
526 TimerOutput::Scope t(this->computing_timer, "Assemble RHS");
527
528
529 // Update the precomputed values needed for the evaluation of the residual.
530 // This is needed, otherwise the line-search mechanism used in the Newton
531 // method might fail even though the Newton step should have been accepted
532 // due to a wrong evaluation of the residual and, consequently, a wrong
533 // evaluation of the step length.
534 this->evaluation_point.update_ghost_values();
535 this->system_operator->evaluate_non_linear_term_and_calculate_tau(
536 this->evaluation_point);
537
538 this->system_operator->evaluate_residual(this->system_rhs,
539 this->evaluation_point);
540
541 // If mortar is enabled, add RHS entries
542 if (this->simulation_parameters.mortar_parameters.enable)
543 {
544 this->system_operator->mortar_coupling_operator_mf->vmult_add(
545 this->system_rhs, this->evaluation_point);
546 this->system_rhs.compress(VectorOperation::add);
547 }
548
549 this->system_rhs *= -1.0;
550
551 // Provide residual to simulation control for stopping criterion when using
552 // steady bdf
553 if (this->simulation_control->is_first_assembly())
554 this->simulation_control->provide_residual(this->system_rhs.l2_norm());
555}

557template <int dim>
558void
559FluidDynamicsVANSMatrixFree<dim>::finish_time_step_fd()
560{
561 // Void fraction percolation must be done before the time step is finished to

564
565 FluidDynamicsMatrixFree<dim>::finish_time_step();

568template <int dim>
569std::vector<OutputStruct<dim, LinearAlgebra::distributed::Vector<double>>>
570FluidDynamicsVANSMatrixFree<dim>::gather_output_hook()
571{
572 std::vector<std::string> name = {"void_fraction"};
573 std::vector<DataComponentInterpretation::DataComponentInterpretation>
574 component_interpretation = {
575 DataComponentInterpretation::component_is_scalar};
576 OutputStructSolution<dim, LinearAlgebra::distributed::Vector<double>>
577 void_fraction_struct(particle_projector.dof_handler,
578 particle_projector.void_fraction_solution,
579 name,
580 component_interpretation);
581
582 std::vector<std::string> force_names(dim, "fluid_drag_on_particles");
583 std::vector<DataComponentInterpretation::DataComponentInterpretation>
584 force_component_interpretation(
585 dim, DataComponentInterpretation::component_is_part_of_vector);
586
587 OutputStructSolution<dim, LinearAlgebra::distributed::Vector<double>>
588 particle_fluid_drag_struct(
589 particle_projector.fluid_drag_on_particles.dof_handler,
590 particle_projector.fluid_drag_on_particles.particle_field_solution,
591 force_names,
592 force_component_interpretation);
593
594 std::vector<std::string> particle_velocity_names(dim, "particle_velocity");
595 std::vector<DataComponentInterpretation::DataComponentInterpretation>
596 particle_velocity_component_interpretation(
597 dim, DataComponentInterpretation::component_is_part_of_vector);
598
599 OutputStructSolution<dim, LinearAlgebra::distributed::Vector<double>>
600 particle_velocity_struct(
601 particle_projector.particle_velocity.dof_handler,
602 particle_projector.particle_velocity.particle_field_solution,
603 particle_velocity_names,
604 particle_velocity_component_interpretation);
605
606 std::vector<std::string> mtc_name(1, "momentum_transfer_coefficient");
607 std::vector<DataComponentInterpretation::DataComponentInterpretation>
608 mtc_component_interpretation(
609 1, DataComponentInterpretation::component_is_scalar);
610
611 OutputStructSolution<dim, LinearAlgebra::distributed::Vector<double>>
612 mtc_struct(
613 particle_projector.momentum_transfer_coefficient.dof_handler,
614 particle_projector.momentum_transfer_coefficient.particle_field_solution,
615 mtc_name,
616 mtc_component_interpretation);
617
618
619 return {void_fraction_struct,
620 particle_fluid_drag_struct,
621 particle_velocity_struct,

625template <int dim>
626void
627FluidDynamicsVANSMatrixFree<dim>::evaluate_time_derivative_void_fraction()
628{
629 this->time_derivative_void_fraction = 0;
630
631 // If the void fraction time derivative is disabled, we just leave it at zero.
632 if (cfd_dem_simulation_parameters.cfd_dem.void_fraction_time_derivative ==
633 false)
634 {
635 // Even though it is zero, we still need to update its ghost values.
636 time_derivative_void_fraction.update_ghost_values();
637 return;
638 }
639 // Time stepping information
640 const auto method = this->simulation_control->get_assembly_method();
641 // Vector for the BDF coefficients
642 const Vector<double> &bdf_coefs =
643 this->simulation_control->get_bdf_coefficients();
644
645 this->time_derivative_void_fraction.add(
646 bdf_coefs[0], this->particle_projector.void_fraction_solution);
647 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
648 {
649 this->time_derivative_void_fraction.add(
650 bdf_coefs[p + 1],
651 this->particle_projector.void_fraction_previous_solution[p]);
652 }
653
654 // After the void fraction time derivative is calculated, we update its ghost
655 // values.
656 time_derivative_void_fraction.update_ghost_values();
657}

659template <int dim>
660void
661FluidDynamicsVANSMatrixFree<dim>::solve()
662{
663 this->computing_timer.enter_subsection("Read mesh, manifolds and particles");
664
665 read_mesh_and_manifolds(
666 *this->triangulation,
667 this->cfd_dem_simulation_parameters.cfd_parameters.mesh,
668 this->cfd_dem_simulation_parameters.cfd_parameters.manifolds_parameters,
669 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
670 .restart ||
671 this->cfd_dem_simulation_parameters.void_fraction->read_dem == true,
672 this->cfd_dem_simulation_parameters.cfd_parameters.boundary_conditions);
673
674 if (this->cfd_dem_simulation_parameters.void_fraction->read_dem == true &&
675 this->cfd_dem_simulation_parameters.cfd_parameters.restart_parameters
676 .restart == false)
677 read_dem();
678
679 this->computing_timer.leave_subsection("Read mesh, manifolds and particles");
680
681 this->setup_dofs();
682
683 particle_projector.initialize_void_fraction(
684 this->simulation_control->get_current_time());
685
686 this->set_initial_condition(
687 this->simulation_parameters.initial_condition->type,
688 this->simulation_parameters.restart_parameters.restart);
689
690 // Only needed if other physics apart from fluid dynamics are enabled.
691 if (this->multiphysics->get_active_physics().size() > 1)
692 this->update_multiphysics_time_average_solution();
693
694 while (this->simulation_control->integrate())
695 {
696 if (this->forcing_function)
697 this->forcing_function->set_time(
698 this->simulation_control->get_current_time());
699
700 this->update_boundary_conditions();
701 this->multiphysics->update_boundary_conditions();
702
703 this->simulation_control->print_progression(this->pcout);
704 this->dynamic_flow_control();
705
706 if (!this->simulation_control->is_at_start())
707 {
708 this->refine_mesh();
709 }
710
711 if (time_stepping_is_bdf(this->simulation_control->get_assembly_method()))
712 {
713 this->computing_timer.enter_subsection(
714 "Calculate time derivative previous solutions");
715
716 this->calculate_time_derivative_previous_solutions();
717 this->time_derivative_previous_solutions.update_ghost_values();
718 this->system_operator->evaluate_time_derivative_previous_solutions(
719 this->time_derivative_previous_solutions);
720
721 this->computing_timer.leave_subsection(
722 "Calculate time derivative previous solutions");
723
724 if (this->simulation_parameters.flow_control.enable_flow_control)
725 this->system_operator->update_beta_force(
726 this->flow_control.get_beta());
727 }
728
729 {
730 TimerOutput::Scope t(this->computing_timer,
731 "Calculate particle-fluid projection");
732
733 // Calculate the new void fraction
734 particle_projector.calculate_void_fraction(
735 this->simulation_control->get_current_time());
736
737 // and its derivative
738 evaluate_time_derivative_void_fraction();
739
740 particle_projector.calculate_particle_fluid_forces_projection(
741 this->cfd_dem_simulation_parameters.cfd_dem,
742 *this->dof_handler,
743 *this->present_solution,
744 *this->previous_solutions,
745 cfd_dem_simulation_parameters.dem_parameters
746 .lagrangian_physical_properties.g,
747 NavierStokesScratchData<dim>(
748 this->simulation_control,
749 this->simulation_parameters.physical_properties_manager,
750 *this->fe,
751 *this->cell_quadrature,
752 *this->mapping,
753 *this->face_quadrature));
754
755 // The base matrix-free operator is not aware of the various VANS
756 // coupling term. We must do a cast here to ensure that the operator is
757 // of the right type.
758 if (auto mf_operator = dynamic_cast<VANSOperator<dim, double> *>(
759 this->system_operator.get()))
760 {
761 TimerOutput::Scope t(this->computing_timer,
762 "Prepare MF operator for VANS");
763
764 mf_operator->compute_void_fraction(
765 particle_projector.dof_handler,
766 particle_projector.void_fraction_solution,
767 this->time_derivative_void_fraction);
768
769 mf_operator->compute_particle_fluid_interaction(
770 particle_projector.fluid_force_on_particles_two_way_coupling
771 .dof_handler,
772 particle_projector.fluid_force_on_particles_two_way_coupling
773 .particle_field_solution,
774 particle_projector.fluid_drag_on_particles.dof_handler,
775 particle_projector.fluid_drag_on_particles
776 .particle_field_solution,
777 particle_projector.particle_velocity.dof_handler,
778 particle_projector.particle_velocity.particle_field_solution,
779 particle_projector.momentum_transfer_coefficient.dof_handler,
780 particle_projector.momentum_transfer_coefficient
781 .particle_field_solution);
782 }
783 }
784
785 this->iterate();
786 this->postprocess(false);
787 this->finish_time_step();
788
789 if (this->simulation_parameters.timer.type ==
790 Parameters::Timer::Type::iteration)
791 this->print_mg_setup_times();
792 }
793 if (this->simulation_parameters.timer.type == Parameters::Timer::Type::end)
794 this->print_mg_setup_times();
795
796 this->finish_simulation();
797}
