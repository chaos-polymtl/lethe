40 VOFSubequationsID::curvature_projection,
41 p_simulation_parameters,
42 ((p_simulation_parameters.multiphysics.vof_parameters
43 .surface_tension_force.verbosity != Parameters::Verbosity::quiet) ||
44 ((p_simulation_parameters.multiphysics.vof_parameters
45 .regularization_method.algebraic_interface_reinitialization
46 .enable) &&
47 (p_simulation_parameters.multiphysics.vof_parameters
48 .regularization_method.verbosity !=
49 Parameters::Verbosity::quiet))) ?
50 Parameters::Verbosity::verbose :
51 Parameters::Verbosity::quiet, // Set to verbose if surface tension
52 // verbosity is enabled or if algebraic
53 // interface reinitialization is enabled
54 // and set to verbose
55 p_pcout,
56 p_triangulation,
57 p_subequations_interface)
58 {
59 if (this->simulation_parameters.mesh.simplex)
60 {
61 // For simplex meshes
62 this->fe = std::make_shared<FE_SimplexP<dim>>(
63 this->simulation_parameters.fem_parameters.VOF_order);
64 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
65 this->cell_quadrature =
66 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
67 }
68 else
69 {
70 // Usual case, for quad/hex meshes
71 this->fe = std::make_shared<FE_Q<dim>>(
72 this->simulation_parameters.fem_parameters.VOF_order);
73 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
74 this->cell_quadrature =
75 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
76 }
77 }
