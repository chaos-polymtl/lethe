13/**
14 * @brief Calculates the time-averaged velocities and pressure \f$(\langle u
15 * \rangle, \langle v \rangle, \langle w \rangle, \langle p \rangle)\f$ and the
16 * independent components of the Reynolds stresses tensor \f$(\langle u'u'
17 * \rangle, \langle v'v' \rangle, \langle w'w' \rangle, \langle u'v' \rangle,
18 * \langle v'w' \rangle, \langle w'u' \rangle)\f$.
19 *
20 * @tparam dim An integer that denotes the dimension of the space in which
21 * the flow is solved.
22 *
23 * @tparam VectorType The vector type used for the solvers (Trilinos (normal or block)
24 * or deal.II vectors).
25 *
26 * @tparam DofsType The type of storage of indices which differs according to vector.
27 *
28 */

39 *
40 */

170 /**
171 * @brief Sanitize the average velocity object after a checkpoint has been read by
172 * resetting all of the locally_owned vectors using the locally_relevant
173 * vectors. This is necessary because only the locally_relevant_vectors are
174 * saved, but the calculation routines expect that the content of the
175 * locally_owned vectors match that of the locally_relevant vectors.
176 */
177 void
178 sanitize_after_restart()
179 {
180 sum_velocity_dt = sum_velocity_dt_with_ghost_cells;
181 sum_reynolds_normal_stress_dt = sum_rns_dt_with_ghost_cells;
182 sum_reynolds_shear_stress_dt = sum_rss_dt_with_ghost_cells;
183 }
184
185 /**
186 * @brief Reinitialize the values of sum_velocities_dt_with_ghost_cells vector to 0 and the has_started_averaging flag to false. If the initial time for average velocities parameter is greater than the simulation time, the checkpointed time average velocities are ignored after a restart. This allows users to restart a simulation with a different averaging start time.
187 */
