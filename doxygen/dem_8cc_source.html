56template <int dim, typename PropertiesIndex>
57void
58DEMSolver<dim, PropertiesIndex>::setup_parameters()
59{
60 // Print simulation starting information
61 pcout << std::endl;
62 std::stringstream ss;
63 ss << "Running on " << n_mpi_processes << " rank(s)";
64 announce_string(pcout, ss.str(), '*');
65
66 // Check if the output directory exists
67 std::string output_dir_name = parameters.simulation_control.output_folder;
68 struct stat buffer;
69
70 // If output directory does not exist, create it
71 if (this_mpi_process == 0)
72 {
73 if (stat(output_dir_name.c_str(), &buffer) != 0)
74 {
75 create_output_folder(output_dir_name);
76 }
77 }
78
79 // Get the pointer of the only instance of the action manager
80 action_manager = DEMActionManager::get_action_manager();
81
82 // Change the behavior of the timer for situations when you don't want outputs
83 if (parameters.timer.type == Parameters::Timer::Type::none)
84 computing_timer.disable_output();
85
86 // Set the simulation control as transient DEM
87 simulation_control = std::make_shared<SimulationControlTransientDEM>(
88 parameters.simulation_control);
89
90 // Setup load balancing parameters and attach the correct functions to the
91 // signals inside the triangulation
92 load_balancing.set_parameters(parameters.model_parameters);
93 load_balancing.copy_references(simulation_control,
94 triangulation,
95 particle_handler,
96 sparse_contacts_object);
97 load_balancing.connect_weight_signals();
98
99 // Set the adaptive sparse contacts parameters
100 if (parameters.model_parameters.sparse_particle_contacts)
101 {
102 sparse_contacts_object.set_parameters(
103 parameters.model_parameters.granular_temperature_threshold,
104 parameters.model_parameters.solid_fraction_threshold,
105 parameters.model_parameters.advect_particles);
106 }
107
108 // Set the distribution type and initialize the neighborhood threshold
109 setup_distribution_type();
110
111 if (this_mpi_process == 0)
112 input_parameter_inspection(parameters,
113 pcout,
114 size_distribution_object_container);
115
116 // Set the grid motion type
117 grid_motion_object =
118 std::make_shared<GridMotion<dim, dim>>(parameters.grid_motion,
119 simulation_control->get_time_step());
120
121 // Set up the solid objects
122 setup_solid_objects();
123
124 // Check if there are periodic boundaries
125 for (unsigned int i_bc = 0;
126 i_bc < parameters.boundary_conditions.bc_types.size();
127 ++i_bc)
128 {
129 if (parameters.boundary_conditions.bc_types[i_bc] ==
130 Parameters::Lagrangian::BCDEM::BoundaryType::periodic)
131 {
132 periodic_boundaries_object.set_periodic_boundaries_information(
133 parameters.boundary_conditions.periodic_boundary_0,
134 parameters.boundary_conditions.periodic_direction);
135 break;
136 }
137 }
138
139 // Assign gravity/acceleration
140 g = parameters.lagrangian_physical_properties.g;
141
142 // If this is a start simulation
143 if (parameters.restart.restart)
144 {
145 action_manager->restart_simulation();
146
147 if (parameters.model_parameters.load_balance_method !=
148 Parameters::Lagrangian::ModelParameters<dim>::LoadBalanceMethod::none)
149 action_manager->load_balance_step();
150 }
151
152 // Disable position integration
153 disable_position_integration =
154 parameters.model_parameters.disable_position_integration;
155}

157template <int dim, typename PropertiesIndex>
158void
159DEMSolver<dim, PropertiesIndex>::setup_distribution_type()
160{
161 // Use namespace and alias to make the code more readable
162 using namespace Parameters::Lagrangian;
163 LagrangianPhysicalProperties &lpp = parameters.lagrangian_physical_properties;
164
165 maximum_particle_diameter = 0;
166 for (unsigned int particle_type = 0; particle_type < lpp.particle_type_number;
167 particle_type++)
168 {
169 switch (lpp.distribution_type.at(particle_type))
170 {
171 case SizeDistributionType::uniform:
172 size_distribution_object_container[particle_type] =
173 std::make_shared<UniformDistribution>(
174 lpp.particle_average_diameter.at(particle_type));
175 break;
176 case SizeDistributionType::normal:
177 size_distribution_object_container[particle_type] =
178 std::make_shared<NormalDistribution>(
179 lpp.particle_average_diameter.at(particle_type),
180 lpp.particle_size_std.at(particle_type),
181 lpp.seed_for_distributions[particle_type] + this_mpi_process);
182 break;
183 case SizeDistributionType::custom:
184 size_distribution_object_container[particle_type] =
185 std::make_shared<CustomDistribution>(
186 lpp.particle_custom_diameter.at(particle_type),
187 lpp.particle_custom_probability.at(particle_type),
188 lpp.seed_for_distributions[particle_type] + this_mpi_process);
189 break;
190 }
191
192 maximum_particle_diameter = std::max(
193 maximum_particle_diameter,
194 size_distribution_object_container[particle_type]->find_max_diameter());
195 }
196
197 neighborhood_threshold_squared =
198 std::pow(parameters.model_parameters.neighborhood_threshold *
199 maximum_particle_diameter,
200 2);
201}

203template <int dim, typename PropertiesIndex>
204void
205DEMSolver<dim, PropertiesIndex>::setup_solid_objects()
206{
207 // Set up solid objects and carry them in vectors
208 for (unsigned int i_solid = 0;
209 i_solid < parameters.solid_objects->number_solid_surfaces;
210 ++i_solid)
211 {
212 solid_surfaces.push_back(std::make_shared<SerialSolid<dim - 1, dim>>(
213 this->parameters.solid_objects->solid_surfaces[i_solid], i_solid));
214 }
215
216 for (unsigned int i_solid = 0;
217 i_solid < parameters.solid_objects->number_solid_volumes;
218 ++i_solid)
219 {
220 solid_volumes.push_back(std::make_shared<SerialSolid<dim, dim>>(
221 this->parameters.solid_objects->solid_volumes[i_solid], i_solid));
222 }
223
224 // Resize the mesh info containers
225 solid_surfaces_mesh_info.resize(solid_surfaces.size());
226 solid_volumes_mesh_info.resize(solid_volumes.size());
227
228 // Simulation has solid objects and resize the container
229 if ((solid_surfaces.size() + solid_volumes.size()) > 0)
230 {
231 action_manager->set_solid_objects_enabled();
232 }
233}

254template <int dim, typename PropertiesIndex>
255std::function<void()>
256DEMSolver<dim, PropertiesIndex>::set_contact_search_iteration_function()
257{
258 using namespace Parameters::Lagrangian;
259 typename ModelParameters<dim>::ContactDetectionMethod
260 contact_detection_method =
261 parameters.model_parameters.contact_detection_method;
262
263 switch (contact_detection_method)
264 {
265 case ModelParameters<dim>::ContactDetectionMethod::constant:
266 return [&] { check_contact_search_iteration_constant(); };
267 case ModelParameters<dim>::ContactDetectionMethod::dynamic:
268 return [&] { check_contact_search_iteration_dynamic(); };
269 default:
270 throw(std::runtime_error("Invalid contact detection method."));
271 }
272}

333template <int dim, typename PropertiesIndex>
334void
335DEMSolver<dim, PropertiesIndex>::setup_triangulation_dependent_parameters()
336{
337 // Find the smallest contact search frequency criterion between (smallest
338 // cell size - largest particle radius) and (security factor * (blob
339 // diameter - 1) * largest particle radius). This value is used in
340 // find_contact_detection_frequency function
341 smallest_contact_search_criterion =
342 std::min((GridTools::minimal_cell_diameter(triangulation) -
343 maximum_particle_diameter * 0.5),
344 (parameters.model_parameters.dynamic_contact_search_factor *
345 (parameters.model_parameters.neighborhood_threshold - 1) *
346 maximum_particle_diameter * 0.5));
347
348 // Find the smallest cell size and use this as the floating mesh mapping
349 // criterion. The edge case comes when the cell are completely square/cubic.
350 // In that case, every sides of a cell are 2^-0.5 or 3^-0.5 times the
351 // cell_diameter. We want to refresh the mapping each time the solid-objet
352 // pass through a cell or there will be late contact detection. Thus, we use
353 // this value.
354 smallest_solid_object_mapping_criterion = [&] {
355 if constexpr (dim == 2) // 2^-0.5 * D_c,min
356 return 0.70710678118 * GridTools::minimal_cell_diameter(triangulation);
357 if constexpr (dim == 3) // 3^-0.5 * D_c,min
358 return 0.57735026919 * GridTools::minimal_cell_diameter(triangulation);
359 }();
360
361 // Setup background dof
362 setup_background_dofs();
363
364 // Set up the periodic boundaries (if PBC enabled)
365 periodic_boundaries_object.map_periodic_cells(
366 triangulation, periodic_boundaries_cells_information);
367
368 // Set the periodic offset to contact managers and particles contact forces
369 // for periodic contact detection (if PBC enabled)
370 contact_manager.set_periodic_offset(
371 periodic_boundaries_object.get_periodic_offset_distance());
372 particle_particle_contact_force_object->set_periodic_offset(
373 periodic_boundaries_object.get_periodic_offset_distance());
374
375 // Set up the local and ghost cells (if ASC enabled)
376 sparse_contacts_object.update_local_and_ghost_cell_set(background_dh);
377}

379template <int dim, typename PropertiesIndex>
380void
381DEMSolver<dim, PropertiesIndex>::setup_background_dofs()
382{
383 FE_Q<dim> background_fe(1);
384 background_dh.distribute_dofs(background_fe);
385
386 // Periodic nodes must be mapped otherwise the disabling of contacts will not
387 // propagate the mobility status to the periodic nodes during iterations.
388 // Identification of periodic nodes is done with the background constraints.
389 // Those constraints are not used for any matrix assembly in DEM solver, this
390 // approach comes from CFD-DEM coupling where void fraction constraints are
391 // used to achieve the periodic node mapping.
392 if (action_manager->check_periodic_boundaries_enabled() &&
393 action_manager->check_sparse_contacts_enabled())
394 {
395 IndexSet locally_relevant_dofs =
396 DoFTools::extract_locally_relevant_dofs(background_dh);
397
398 background_constraints.clear();
399 background_constraints.reinit(background_dh.locally_owned_dofs(),
400 locally_relevant_dofs);
401
402 DoFTools::make_periodicity_constraints(
403 background_dh,
404 parameters.boundary_conditions.periodic_boundary_0,
405 parameters.boundary_conditions.periodic_boundary_1,
406 parameters.boundary_conditions.periodic_direction,
407 background_constraints);
408
409 background_constraints.close();
410
411 sparse_contacts_object.map_periodic_nodes(background_constraints);
412 }

415template <int dim, typename PropertiesIndex>
416void
417DEMSolver<dim, PropertiesIndex>::load_balance()
418{
419 load_balancing.check_load_balance_iteration();
420
421 if (!action_manager->check_load_balance())
422 return;
423
424 TimerOutput::Scope t(this->computing_timer, "Load balancing");
425 // Prepare particle handler for the adaptation of the triangulation to the
426 // load
427 particle_handler.prepare_for_coarsening_and_refinement();
428
429 pcout << "-->Repartitionning triangulation" << std::endl;
430 triangulation.repartition();
431
432 // Unpack the particle handler after the mesh has been repartitioned
433 particle_handler.unpack_after_coarsening_and_refinement();
434
435 // If PBC are enabled, update the periodic cells
436 periodic_boundaries_object.map_periodic_cells(
437 triangulation, periodic_boundaries_cells_information);
438
439 // If ASC is enabled, update the local and ghost cell set
440 sparse_contacts_object.update_local_and_ghost_cell_set(background_dh);
441
442 // Update neighbors of cells after load balance
443 contact_manager.update_cell_neighbors(triangulation,
444 periodic_boundaries_cells_information);
445
446 boundary_cell_object.build(
447 triangulation,
448 parameters.floating_walls,
449 parameters.boundary_conditions.outlet_boundaries,
450 parameters.mesh.check_for_diamond_cells,
451 parameters.mesh.expand_particle_wall_contact_search,
452 pcout);
453
454 // Update the boundary information (if grid motion)
455 boundary_cell_object.update_boundary_info_after_grid_motion(
456 updated_boundary_points_and_normal_vectors);
457
458 const auto average_minimum_maximum_cells =
459 Utilities::MPI::min_max_avg(triangulation.n_active_cells(),
460 mpi_communicator);
461
462 const auto average_minimum_maximum_particles =
463 Utilities::MPI::min_max_avg(particle_handler.n_locally_owned_particles(),
464 mpi_communicator);
465
466 pcout << "Load balance finished " << std::endl;
467 pcout
468 << "Average, minimum and maximum number of particles on the processors are "
469 << average_minimum_maximum_particles.avg << " , "
470 << average_minimum_maximum_particles.min << " and "
471 << average_minimum_maximum_particles.max << std::endl;
472 pcout << "Minimum and maximum number of cells owned by the processors are "
473 << average_minimum_maximum_cells.min << " and "
474 << average_minimum_maximum_cells.max << std::endl;
475
476 setup_background_dofs();
477}

589template <int dim, typename PropertiesIndex>
590void
591DEMSolver<dim, PropertiesIndex>::update_temperature_solid_objects()
592{
593 // Previous time must be used here instead of current time, which is the time
594 // for which we are doing calculations. The solid object is moved and its
595 // temperature is updated before calculating the contact outcomes for the
596 // current time step, so its properties should correspond to the
597 // previous time step.
598 if constexpr (std::is_same_v<PropertiesIndex,
599 DEM::DEMMPProperties::PropertiesIndex>)

601 if (!action_manager->check_solid_objects_enabled())
602 return;
603
604 for (auto &solid_object : solid_surfaces)
605 solid_object->update_solid_temperature(
606 simulation_control->get_previous_time());
607
608 for (auto &solid_object : solid_volumes)
609 solid_object->update_solid_temperature(
610 simulation_control->get_previous_time());
611 }

614template <int dim, typename PropertiesIndex>
615void
616DEMSolver<dim, PropertiesIndex>::finish_simulation()
617{
618 // Timer output
619 if (parameters.timer.type == Parameters::Timer::Type::end)
620 this->computing_timer.print_summary();
621
622 // Testing
623 if (parameters.test.enabled)
624 {
625 switch (parameters.test.test_type)
626 {
627 case Parameters::Testing::TestType::particles:
628 {
629 visualization_object.print_xyz(particle_handler,
630 mpi_communicator,
631 pcout);
632 break;
633 }
634 case Parameters::Testing::TestType::mobility_status:
635 {
636 // Get mobility status vector sorted by cell id
637 Vector<float> mobility_status(triangulation.n_active_cells());
638 sparse_contacts_object.get_mobility_status_vector(
639 mobility_status);
640
641 // Output mobility status vector
642 visualization_object.print_intermediate_format(mobility_status,
643 background_dh,
644 mpi_communicator);
645 break;
646 }
647 case Parameters::Testing::TestType::subdomain:
648 {
649 // Get mobility status vector sorted by cell id
650 Vector<float> subdomain(triangulation.n_active_cells());
651 for (unsigned int i = 0; i < subdomain.size(); ++i)
652 subdomain(i) = triangulation.locally_owned_subdomain();
653
654 // Output subdomain vector
655 visualization_object.print_intermediate_format(subdomain,
656 background_dh,
657 mpi_communicator);
658 break;
659 }
660 }
661 }
662
663 // Outputting force and torques over boundary
664 if (parameters.forces_torques.calculate_force_torque)
665 {
666 write_forces_torques_output_results(
667 parameters.forces_torques.force_torque_output_name,
668 parameters.forces_torques.output_frequency,
669 triangulation.get_boundary_ids(),
670 simulation_control->get_time_step(),
671 forces_boundary_information,
672 torques_boundary_information);
673 }

676template <int dim, typename PropertiesIndex>
677void
678DEMSolver<dim, PropertiesIndex>::write_output_results()
679{
680 TimerOutput::Scope t(this->computing_timer, "Output VTU");
681
682 const std::string folder = parameters.simulation_control.output_folder;
683 const std::string particles_solution_name =
684 parameters.simulation_control.output_name;
685 const unsigned int iter = simulation_control->get_step_number();
686 const double time = simulation_control->get_current_time();
687 const unsigned int group_files = parameters.simulation_control.group_files;
688
689 // Write particles
690 Visualization<dim, PropertiesIndex> particle_data_out;
691 particle_data_out.build_patches(particle_handler,
692 properties_class.get_properties_name());
693
694 write_vtu_and_pvd<0, dim>(particles_pvdhandler,
695 particle_data_out,
696 folder,
697 particles_solution_name,
698 time,
699 iter,
700 group_files,
701 mpi_communicator);
702
703 if (simulation_control->get_output_boundaries())
704 {
705 DataOutFaces<dim> data_out_faces;
706
707 // Setup background dofs to initiate right sized boundary_id vector
708 Vector<float> boundary_id(background_dh.n_dofs());
709
710 // Attach the boundary_id to data_out_faces object
711 BoundaryPostprocessor<dim> boundary;
712 data_out_faces.attach_dof_handler(background_dh);
713 data_out_faces.add_data_vector(boundary_id, boundary);

769 contact_list.max =

776 simulation_control->get_log_frequency();
777
778 // Calculate statistics on the particles

786 PropertiesIndex,
787 DEM::dem_statistic_variable::rotational_kinetic_energy>(particle_handler,
788 mpi_communicator);
789 statistics velocity =
790 calculate_granular_statistics<dim,
791 PropertiesIndex,
792 DEM::dem_statistic_variable::velocity>(
793 particle_handler, mpi_communicator);
794 statistics omega =
795 calculate_granular_statistics<dim,
796 PropertiesIndex,
797 DEM::dem_statistic_variable::omega>(
798 particle_handler, mpi_communicator);
799
800 if (this_mpi_process == 0)
801 {
802 TableHandler report;

804 report.declare_column("Variable");
805 report.declare_column("Min");
806 report.declare_column("Max");
807 report.declare_column("Average");
808 report.declare_column("Total");

811 report);
812 add_statistics_to_table_handler("Velocity magnitude", velocity, report);
813 add_statistics_to_table_handler("Angular velocity magnitude",
814 omega,
815 report);
816 add_statistics_to_table_handler("Translational kinetic energy",
817 translational_kinetic_energy,
818 report);

820 rotational_kinetic_energy,
821 report);

823
824
825 report.set_scientific("Min", true);
826 report.set_scientific("Max", true);
827 report.set_scientific("Average", true);
828 report.set_scientific("Total", true);
829
830 report.write_text(std::cout, dealii::TableHandler::org_mode_table);
831 }
832
833 // Timer output

840template <int dim, typename PropertiesIndex>
841inline void
842DEMSolver<dim, PropertiesIndex>::sort_particles_into_subdomains_and_cells()
843{
844 particle_handler.sort_particles_into_subdomains_and_cells();
845
846 // Exchange ghost particles
847 particle_handler.exchange_ghost_particles(true);
848
849 // Resize the displacement, force and torque containers only if the particles
850 // have changed subdomains
851 if (action_manager->check_resize_containers())
852 {
853 unsigned int number_of_particles =
854 particle_handler.get_max_local_particle_index();
855 // Resize displacement container
856 displacement.resize(number_of_particles);
857 // Resize outcome containers
858 contact_outcome.resize_interaction_containers(number_of_particles);
859 MOI.resize(number_of_particles);
860
861 // Updating moment of inertia container
862 for (auto &particle : particle_handler)
863 {
864 auto particle_properties = particle.get_properties();
865 MOI[particle.get_local_index()] =
866 0.1 * particle_properties[PropertiesIndex::mass] *
867 particle_properties[PropertiesIndex::dp] *
868 particle_properties[PropertiesIndex::dp];
869 }
870 }
871
872 // Always reset the displacement values since we are doing a search detection
873 std::fill(displacement.begin(), displacement.end(), 0.);
874}

876template <int dim, typename PropertiesIndex>
877void
878DEMSolver<dim, PropertiesIndex>::solve()
879{
880 // Set up the parameters
881 setup_parameters();
882
883 // Reading mesh
884 read_mesh(parameters.mesh,
885 action_manager->check_restart_simulation(),
886 pcout,
887 triangulation,
888 parameters.boundary_conditions);
889
890 // Set up functions and pointers according to parameters
891 setup_functions_and_pointers();
892
893 // Read checkpoint if needed
894 read_checkpoint(computing_timer,
895 parameters,
896 simulation_control,
897 particles_pvdhandler,
898 grid_pvdhandler,
899 triangulation,
900 particle_handler,
901 insertion_object,
902 solid_surfaces,
903 checkpoint_controller);
904
905 // Set up the various parameters that need the triangulation
906 setup_triangulation_dependent_parameters();
907
908 // Build the mapping of the cell neighbors
909 contact_manager.execute_cell_neighbors_search(
910 triangulation, periodic_boundaries_cells_information);
911
912 // Find boundary cells with faces
913 boundary_cell_object.build(
914 triangulation,
915 parameters.floating_walls,
916 parameters.boundary_conditions.outlet_boundaries,
917 parameters.mesh.check_for_diamond_cells,
918 parameters.mesh.expand_particle_wall_contact_search,
919 pcout);
920
921 // DEM engine iterator
922 while (simulation_control->integrate())
923 {
924 simulation_control->print_progression(pcout);
925 if (simulation_control->is_verbose_iteration())
926 report_statistics();
927
928 // Move grid and update the boundary information (if grid motion)
929 grid_motion_object->move_grid(triangulation);
930 boundary_cell_object.update_boundary_info_after_grid_motion(
931 updated_boundary_points_and_normal_vectors);
932
933 // Insert particle if needed
934 insert_particles();
935
936 // Load balancing (if load balancing enabled and if needed)
937 load_balance();
938
939 // Check for contact search according to the contact detection method
940 contact_detection_iteration_check_function();
941
942 // Check if solid object needs to be mapped with background mesh
943 // (if solid object)
944 find_floating_mesh_mapping_step(smallest_solid_object_mapping_criterion,
945 this->solid_surfaces);
946
947 // Map solid objects if the action was triggered (if solid object)
948 if (action_manager->check_solid_object_search())
949 {
950 // Store information about floating mesh/background mesh
951 // intersection
952 for (unsigned int i_solid = 0; i_solid < solid_surfaces.size();
953 ++i_solid)
954 {
955 solid_surfaces_mesh_info[i_solid] =
956 solid_surfaces[i_solid]->map_solid_in_background_triangulation(
957 triangulation);
958 }
959
960 for (unsigned int i_solid = 0; i_solid < solid_volumes.size();
961 ++i_solid)
962 {
963 solid_volumes_mesh_info[i_solid] =
964 solid_volumes[i_solid]->map_solid_in_background_triangulation(
965 triangulation);
966 }
967 }
968
969 // Execute contact search if the action was triggered
970 if (action_manager->check_contact_search())
971 {
972 // Particles displacement if passing through a periodic boundary
973 // (if PBC enabled)
974 periodic_boundaries_object.execute_particles_displacement(
975 particle_handler, periodic_boundaries_cells_information);
976
977 sort_particles_into_subdomains_and_cells();
978
979 // Compute cell mobility (if ASC enabled)
980 sparse_contacts_object.identify_mobility_status(
981 background_dh,
982 particle_handler,
983 triangulation.n_active_cells(),
984 mpi_communicator);
985
986 // Execute broad search by filling containers of particle-particle
987 // contact pair candidates and containers of particle-wall
988 // contact pair candidates
989 contact_manager.execute_particle_particle_broad_search(
990 particle_handler, sparse_contacts_object);
991
992 contact_manager.execute_particle_wall_broad_search(
993 particle_handler,
994 boundary_cell_object,
995 solid_surfaces_mesh_info,
996 parameters.floating_walls,
997 simulation_control->get_current_time(),
998 sparse_contacts_object);
999
1000 // Update contacts, remove replicates and add new contact pairs
1001 // to the contact containers when particles are exchanged between
1002 // processors
1003 contact_manager.update_contacts();
1004
1005 // Updates the iterators to particles in local-local contact
1006 // containers
1007 contact_manager.update_local_particles_in_cells(particle_handler);
1008
1009 // Execute fine search by updating particle-particle contact
1010 // containers according to the neighborhood threshold
1011 contact_manager.execute_particle_particle_fine_search(
1012 neighborhood_threshold_squared);
1013
1014 // Execute fine search by updating particle-wall contact
1015 // containers according to the neighborhood threshold
1016 contact_manager.execute_particle_wall_fine_search(
1017 parameters.floating_walls,
1018 simulation_control->get_current_time(),
1019 neighborhood_threshold_squared);
1020
1021 // Updating number of contact builds
1022 contact_build_number++;
1023 }
1024 else
1025 {
1026 particle_handler.update_ghost_particles();
1027 }
1028
1029 // Particle-particle contact force
1030 particle_particle_contact_force_object
1031 ->calculate_particle_particle_contact(
1032 contact_manager.get_local_adjacent_particles(),
1033 contact_manager.get_ghost_adjacent_particles(),
1034 contact_manager.get_local_local_periodic_adjacent_particles(),
1035 contact_manager.get_local_ghost_periodic_adjacent_particles(),
1036 contact_manager.get_ghost_local_periodic_adjacent_particles(),
1037 simulation_control->get_time_step(),
1038 contact_outcome);
1039
1040 // Update the boundary points and vectors (if grid motion)
1041 // We have to update the positions of the points on boundary faces and
1042 // their normal vectors here. The update_contacts deletes the
1043 // particle-wall contact candidate if it exists in the contact list.
1044 // As a result, when we update the points on boundary faces and their
1045 // normal vectors, update_contacts deletes it from the output of broad
1046 // search and they are not updated in the contact force calculations.
1047 grid_motion_object
1048 ->update_boundary_points_and_normal_vectors_in_contact_list(
1049 contact_manager.get_particle_wall_in_contact(),
1050 updated_boundary_points_and_normal_vectors);
1051
1052 // Move solid objects (if solid object)
1053 move_solid_objects();
1054
1055 // Update solid objects temperatures
1056 update_temperature_solid_objects();
1057
1058 // Particle-wall contact force
1059 particle_wall_contact_force();
1060
1061 // Integration of temperature for multiphysic DEM
1062 if constexpr (std::is_same_v<PropertiesIndex,
1063 DEM::DEMMPProperties::PropertiesIndex>)
1064 {
1065 integrate_temperature<dim, PropertiesIndex>(
1066 particle_handler,
1067 simulation_control->get_time_step(),
1068 contact_outcome.heat_transfer_rate,
1069 std::vector<double>(force.size()));
1070 }
1071
1072 // Integration of force and velocity for new location of particles
1073 // The half step is calculated at the first iteration
1074
1075 if (!disable_position_integration)
1076 {
1077 if (simulation_control->get_step_number() == 0)
1078 {
1079 integrator_object->integrate_half_step_location(
1080 particle_handler,
1081 g,
1082 simulation_control->get_time_step(),
1083 torque,
1084 force,
1085 MOI);
1086 }
1087 else
1088 {
1089 integrator_object->integrate(particle_handler,
1090 g,
1091 simulation_control->get_time_step(),
1092 torque,
1093 force,
1094 MOI,
1095 triangulation,
1096 sparse_contacts_object);
1097 }
1098 }
1099
1100 // Visualization
1101 if (simulation_control->is_output_iteration())
1102 write_output_results();
1103
1104 // Log the contact statistics if the parameter is enabled
1105 if (parameters.post_processing.particle_wall_collision_statistics)
1106 {
1107 log_collision_data<dim, PropertiesIndex>(
1108 parameters,
1109 contact_manager.get_particle_wall_in_contact(),
1110 simulation_control->get_current_time(),
1111 ongoing_collision_log,
1112 collision_event_log);
1113 }
1114
1115 // Calculation of forces and torques if needed
1116 if (parameters.forces_torques.calculate_force_torque)
1117 {
1118 if ((this_mpi_process == 0) &&
1119 (simulation_control->get_step_number() %
1120 parameters.forces_torques.output_frequency ==
1121 0) &&
1122 (parameters.forces_torques.force_torque_verbosity ==
1123 Parameters::Verbosity::verbose))
1124 {
1125 write_forces_torques_output_locally(
1126 forces_boundary_information[simulation_control
1127 ->get_step_number()],
1128 torques_boundary_information[simulation_control
1129 ->get_step_number()]);
1130 }
1131 }
1132
1133 // Post-processing if needed
1134 post_process_results();
1135
1136 // Write checkpoint if needed
1137 if (checkpoint_controller.is_checkpoint_time_step(
1138 simulation_control->get_step_number()))
1139 {
1140 write_checkpoint(computing_timer,
1141 parameters,
1142 simulation_control,
1143 particles_pvdhandler,
1144 grid_pvdhandler,
1145 triangulation,
1146 particle_handler,
1147 insertion_object,
1148 solid_surfaces,
1149 pcout,
1150 mpi_communicator,
1151 checkpoint_controller);
1152 }
1153
1154 // Reset all trigger flags
1155 action_manager->reset_triggers();
1156 }
1157
1158 // Write particle-wall collision statistics file if enabled
1159 if (parameters.post_processing.particle_wall_collision_statistics)
1160 write_collision_stats(parameters, collision_event_log, mpi_communicator);
1161
1162 finish_simulation();
1163}
