137 * @deprecated This function is to be deprecated when the non-linear solvers
138 * have been refactored to call for rhs and matrix assembly seperately.

140
141 /*
142 Modified version of assemble_matrix_and_rhs to include the presence of
143 extra steps. For more detail see the same function in the
144 fluid_dynamics_matrix_based.h solver.
145 */
146
147 virtual void
148 assemble_matrix_and_rhs()
149 {
150 if (some_particles_are_coupled)
151 {
152 if (this->get_current_newton_iteration() != 0)
153 {
154 force_on_ib();
155 }
156 integrate_particles();
157 update_precalculations_for_ib();
158 }
159 // Do the cut cell mapping only if it is the first Newton iteration or
160 // if the explicit position integration is false.
161 if (this->get_current_newton_iteration() == 0 ||
162 this->simulation_parameters.particlesParameters
163 ->explicit_position_integration_calculation == false)
164 {
165 if (all_spheres)
166 optimized_generate_cut_cells_map();
167 else
168 generate_cut_cells_map();
169 }
170 this->FluidDynamicsMatrixBased<dim>::assemble_system_matrix();
171
172 sharp_edge();
173 }

461 /**
462 * @brief
463 * A override of the get_current_residual to take into account the particles
464 * coupling residual.
465 */
466 double
467 get_current_residual() override
468 {
469 double scalling = this->simulation_parameters.non_linear_solver
470 .at(PhysicsID::fluid_dynamics)
471 .tolerance /
472 this->simulation_parameters.particlesParameters
473 ->particle_nonlinear_tolerance;
474 return std::max(this->system_rhs.l2_norm(), particle_residual * scalling);
475 }
