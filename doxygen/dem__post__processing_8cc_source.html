10 template <int dim, typename PropertiesIndex, dem_statistic_variable var>
11 statistics
12 calculate_granular_statistics(
13 const Particles::ParticleHandler<dim> &particle_handler,
14 const MPI_Comm &mpi_communicator)
15 {
16 double total_variable = 0;
17 double max_variable = DBL_MIN;
18 double min_variable = DBL_MAX;
19
20
21 for (auto &particle : particle_handler)
22 {
23 // Get the properties of the particle
24 auto particle_properties = particle.get_properties();
25
26 double variable = 0;
27
28 if constexpr (var ==
29 dem_statistic_variable::translational_kinetic_energy)
30 {
31 // Put particle velocity in a tensor
32 Tensor<1, dim> velocity;
33 for (int d = 0; d < dim; ++d)
34 velocity[d] = particle_properties[PropertiesIndex::v_x + d];
35
36 // Kinetic energy is 0.5*m*u^2
37 variable = 0.5 * particle_properties[PropertiesIndex::mass] *
38 velocity.norm_square();
39 }
40 if constexpr (var == dem_statistic_variable::rotational_kinetic_energy)
41 {
42 // Put angular velocity in a tensor
43 Tensor<1, 3> omega;
44 if constexpr (dim == 2)
45 {
46 omega[2] = particle_properties[PropertiesIndex::omega_z];
47 }
48 if constexpr (dim == 3)
49 {
50 for (int d = 0; d < dim; ++d)
51 omega[d] = particle_properties[PropertiesIndex::omega_x + d];
52 }
53
54 // Kinetic energy is 0.1*m*d^2 * w^2
55 variable = 0.1 * particle_properties[PropertiesIndex::mass] *
56 Utilities::fixed_power<2>(
57 particle_properties[PropertiesIndex::dp]) *
58 omega.norm_square();
59 }
60
61 if constexpr (var == dem_statistic_variable::velocity)
62 {
63 // Put particle velocity in a tensor
64 Tensor<1, dim> velocity;
65 for (int d = 0; d < dim; ++d)
66 velocity[d] = particle_properties[PropertiesIndex::v_x + d];
67
68 variable = velocity.norm();
69 }
70
71 if constexpr (var == dem_statistic_variable::omega)
72 {
73 // Put angular velocity in a tensor
74 Tensor<1, 3> omega;
75 if constexpr (dim == 2)
76 {
77 omega[2] = particle_properties[PropertiesIndex::omega_z];
78 }
79 if constexpr (dim == 3)
80 {
81 for (int d = 0; d < dim; ++d)
82 omega[d] = particle_properties[PropertiesIndex::omega_x + d];
83 }
84
85 variable = omega.norm();
86 }
87
88 total_variable += variable;
89
90 max_variable = std::max(variable, max_variable);
91 min_variable = std::min(variable, min_variable);
92 }
93
94 total_variable = Utilities::MPI::sum(total_variable, mpi_communicator);
95 max_variable = Utilities::MPI::max(max_variable, mpi_communicator);
96 min_variable = Utilities::MPI::min(min_variable, mpi_communicator);
97
98 statistics stats;
99 stats.total = total_variable;
100 stats.max = max_variable;
101 stats.min = min_variable;
102 stats.average = total_variable / particle_handler.n_global_particles();
103
104 return stats;
105 }
