16NormalDistribution::NormalDistribution(
17 const double &d_average,
18 const double &d_standard_deviation,
19 const unsigned int &prn_seed,
20 const double min_cutoff,
21 const double max_cutoff,
22 const DistributionWeightingType &distribution_weighting_type)
23 : Distribution(distribution_weighting_type)
24 , gen(prn_seed)
25{
26 // We need the average and the standard deviation to be based on number to do
27 // our sampling.
28 if (this->weighting_type == DistributionWeightingType::number_based)
29 {
30 diameter_average = d_average;
31 standard_deviation = d_standard_deviation;
32 }
33 // If the user specified the distribution as volume-based, we need to convert
34 // the average and standard deviation to be number-based. To do so, we need to
35 // solve a nonlinear system.
36 // https://mfix.netl.doe.gov/doc/mfix-exa/guide/latest/references/size_distributions.html#normal-distribution
37 else if (this->weighting_type == DistributionWeightingType::volume_based)
38 {
39 // Either there is an error with the Newton-Raphson solver or the problem
40 // is too stiff due to the order of the polynomial we are trying to solve
41 // for. Because of this, volume-based distribution is not supported for
42 // now. MFIX-exa uses a homotopy method to solve the system.
43
44 // Volume-based parameters
45 const double mv = d_average; // volume-based mean
46 const double mv2 = mv * mv; // volume-based mean squared
47
48 const double sv = d_standard_deviation; // volume-based standard deviation
49 const double sv2 = sv * sv; // volume-based standard deviation squared
50
51 // Matrix and RHS
52 LAPACKFullMatrix<double> system_matrix;
53 Vector<double> system_rhs;
54
55 double residual_l2_norm = 1.;
56 unsigned int iteration_counter = 0;
57
58 // Initial estimates
59 // mean based on number
60 double mu_n = d_average;
61 // standard deviation based on number
62 double sigma_n = d_standard_deviation;
63
64 while (residual_l2_norm > 1.e-12 && iteration_counter < 100)
65 {
66 system_matrix.reinit(2);
67 system_rhs.reinit(2);
68
69 const double mn = mu_n; // mean
70 const double mn2 = mn * mn; // mean squared
71 const double mn3 = mn2 * mn; // mean cubed
72 const double mn4 = mn2 * mn2; // mean ^4
73 const double mn5 = mn4 * mn; // mean ^5
74
75 const double sn = sigma_n; // standard deviation
76 const double sn2 = sn * sn; // standard deviation squared
77 const double sn3 = sn2 * sn; // standard deviation cubed
78 const double sn4 = sn2 * sn2; // standard deviation ^4
79
80 // Jacobian
81 const double dR1_sigma_n =
82 6. * (2. * (sn3 + sn * mn2) - mv * sn * mn);
83
84 const double dR1_mu_n =
85 12. * sn2 * mn + 4 * mn3 - 3. * mv * (sn2 + mn2);
86
87 const double dR2_sigma_n =
88 12. * sn3 * (5. * mn - 2. * mv) +
89 2. * sn * (10. * mn3 - 12. * mn2 * mv + 3. * mn * mv2) -
90 6. * sv2 * sn * mn;
91
92 const double dR2_mu_n = 15. * sn4 +
93 sn2 * (30. * mn2 - 24. * mn * mv + 3. * mv2) +
94 5. * mn4 + 3. * mn2 * mv2 - 8. * mn3 * mv -
95 sv2 * (3. * sn2 + 3. * mn2);
96
97 system_matrix.set(0, 0, dR1_sigma_n);
98 system_matrix.set(0, 1, dR1_mu_n);
99 system_matrix.set(1, 0, dR2_sigma_n);
100 system_matrix.set(1, 1, dR2_mu_n);
101
102 // Resisual RHS
103 const double R1 =
104 3. * sn4 + 6. * sn2 * mn2 + mn4 - mv * (3. * sn2 * mn + mn3);
105
106 const double R2 = 3. * sn4 * (5. * mn - 2. * mv) +
107 sn2 * (10. * mn3 - 12. * mn2 * mv + 3. * mn * mv2) +
108 mn5 + mn3 * mv2 - 2. * mn4 * mv -
109 sv2 * (3. * sn2 * mn + mn3);
110
111 // Add the minus sign for Newton-Raphson
112 system_rhs[0] = -R1;
113 system_rhs[1] = -R2;
114
115 // Solve for the update
116 system_matrix.compute_lu_factorization();
117 system_matrix.solve(system_rhs);
118
119 // Apply the delta
120 sigma_n += system_rhs[0];
121 mu_n += system_rhs[1];
122
123 // While loop conditions
124 residual_l2_norm = system_rhs.l2_norm();
125 iteration_counter++;
126 }
127 AssertThrow(
128 iteration_counter < 100.,
129 ExcMessage("The numbered weighted mean and standard deviation of "
130 "the normal distribution has not been found from the volume "
131 "weighted values provided in the parameter file. To solve "
132 "this issue, define the distribution as number-based."));
133
134 diameter_average = mu_n;
135 standard_deviation = sigma_n;
136 }
137
138 if (min_cutoff < 0.)
139 {
140 // 2.5 -> approx 99% of all diameters are bigger
141 this->dia_min_cutoff = diameter_average - 2.5 * standard_deviation;
142 AssertThrow(this->dia_min_cutoff > 0.,
143 ExcMessage(
144 "The \"standard deviation\" parameter is too "
145 "high relative to the \"average diameter\" "
146 "parameter. This would results with frequent negative "
147 "diameter values. To fix this problem, please change "
148 "the value of those two parameters or define a "
149 "\"minimum diameter cutoff\" bigger than \"0.\""));
150 }
151 else
152 this->dia_min_cutoff = min_cutoff;
153
154 if (max_cutoff < 0.)
155 // 2.5 -> approx 99% of all diameters are smaller
156 this->dia_max_cutoff = diameter_average + 2.5 * standard_deviation;
157 else
158 this->dia_max_cutoff = max_cutoff;
159
160 AssertThrow(this->dia_min_cutoff < this->dia_max_cutoff,
161 ExcMessage("The \"minimum diameter cutoff\" parameter needs to"
162 " be smaller than the \"maximum diameter cutoff\"."));
163}

165void
166NormalDistribution::particle_size_sampling(
167 const unsigned int &number_of_particles)
168{
169 this->particle_sizes.clear();
170 this->particle_sizes.reserve(number_of_particles);
171
172 std::normal_distribution<> distribution{diameter_average, standard_deviation};
173
174 unsigned int n_created_diameter = 0;
175 while (n_created_diameter < number_of_particles)
176 {
177 const double temp_diameter = distribution(gen);
178 if (temp_diameter > this->dia_min_cutoff &&
179 temp_diameter < this->dia_max_cutoff)
180 {
181 n_created_diameter++;
182 this->particle_sizes.push_back(temp_diameter);
183 }
184 }
185}

199void
200NormalDistribution::print_psd_declaration_string(
201 const unsigned int particle_type,
202 const ConditionalOStream &pcout)
203{
204 const double z_min =
205 (this->dia_min_cutoff - diameter_average) / standard_deviation;
206 const double z_max =
207 (this->dia_max_cutoff - diameter_average) / standard_deviation;
208
209 // Quantile
210 const double q_min = 0.5 * (1.0 + std::erf(z_min / std::numbers::sqrt2));
211 const double q_max = 0.5 * (1.0 + std::erf(z_max / std::numbers::sqrt2));
212
213 pcout << "The particle size distribution of particle type " << particle_type
214 << " is normal." << std::endl;
215
216 if (q_min > 0.25 || q_max < 0.75)
217 pcout
218 << "Warning: The minimal and maximal cutoffs of the distribution are at "
219 << q_min * 100. << " and " << q_max * 100. << " respectively."
220 << std::endl;
221 else
222 pcout << "The minimal and maximal cutoffs of the distribution are at "
223 << q_min * 100. << " and " << q_max * 100. << " respectively."
224 << std::endl;
225}

227LogNormalDistribution::LogNormalDistribution(
228 const double &d_average,
229 const double &d_standard_deviation,
230 const unsigned int &prn_seed,
231 double min_cutoff,
232 double max_cutoff,
233 const Parameters::Lagrangian::DistributionWeightingType
234 &distribution_weighting_type)
235 : Distribution(distribution_weighting_type)
236 , sigma_ln(std::sqrt(std::log(
237 1. + Utilities::fixed_power<2>(d_standard_deviation / d_average))))
238 , gen(prn_seed)
239{
240 // We need the number-based average and standard deviation to do
241 // our sample. The standard deviation is always the same.
242 if (this->weighting_type == DistributionWeightingType::number_based)
243 {
244 mu_ln = std::log(d_average) - 0.5 * Utilities::fixed_power<2>(sigma_ln);
245 }
246 // If the user specified the distribution as volume-based, we need to convert
247 // the average and standard deviation to be number-based.
248 // https://mfix.netl.doe.gov/doc/mfix-exa/guide/latest/references/size_distributions.html#log-normal-distribution
249 else if (this->weighting_type == DistributionWeightingType::volume_based)
250 {
251 mu_ln = std::log(d_average) - 3.5 * Utilities::fixed_power<2>(sigma_ln);
252 }
253
254 if (min_cutoff < 0.)
255 {
256 // approx 99% of all diameters are bigger
257 this->dia_min_cutoff = std::exp(mu_ln - 2.5 * sigma_ln);
258 AssertThrow(this->dia_min_cutoff > 0.,
259 ExcMessage(
260 "The \"standard deviation\" parameter is to "
261 "high relative to the \"average diameter\" "
262 "parameter. This would results with frequent negative "
263 "diameter values. To fix this problem, please change "
264 "the value of those two parameters or define a "
265 "\"minimum diameter cutoff\" bigger than \"0.\""));
266 }
267 else
268 this->dia_min_cutoff = min_cutoff;
269
270 if (max_cutoff < 0.)
271 // approx 99% of all diameters are bigger
272 this->dia_max_cutoff = std::exp(mu_ln + 2.5 * sigma_ln);
273 else
274 this->dia_max_cutoff = max_cutoff;
275
276 AssertThrow(this->dia_min_cutoff < this->dia_max_cutoff,
277 ExcMessage(
278 "The \"minimum diameter cutoff\" parameter need to be smaller "
279 "than the \"maximum diameter cutoff\"."));
280}

381CustomDistribution::CustomDistribution(
382 const std::vector<double> &d_list,
383 const std::vector<double> &d_probabilities,
384 const unsigned int &prn_seed,
385 // const double min_cutoff,
386 // const double max_cutoff,
387 const Parameters::Lagrangian::DistributionWeightingType
388 &distribution_weighting_type)
389 : Distribution(distribution_weighting_type)
390 , diameter_custom_values(d_list)
391 , gen(prn_seed)
392{
393 AssertThrow(distribution_weighting_type ==
394 DistributionWeightingType::volume_based,
395 ExcMessage(
396 "Custom distribution only supports volume-based "
397 "distribution. Please set the 'distribution weighting basis' "
398 "parameter to `volume`."));
399
400 std::vector<double> cumulative_probability_vector, n_i_vector;
401 double n_tot = 0.;
402
403 cumulative_probability_vector.reserve(d_probabilities.size());
404 n_i_vector.reserve(d_probabilities.size());
405
406 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
407 {
408 n_i_vector.push_back(d_probabilities[i] / Utilities::fixed_power<3>(
409 diameter_custom_values[i]));
410 n_tot += n_i_vector[i];
411 }
412
413 double i_probability;
414 double cumulative_value = 0.;
415
416 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
417 {
418 i_probability = n_i_vector[i] / n_tot;
419 cumulative_value += i_probability;
420 cumulative_probability_vector.push_back(cumulative_value);
421 }
422
423 diameter_custom_cumul_prob = cumulative_probability_vector;
424}

426void
427CustomDistribution::particle_size_sampling(
428 const unsigned int &number_of_particles)
429{
430 this->particle_sizes.clear();
431 this->particle_sizes.reserve(number_of_particles);
432
433 std::uniform_real_distribution<> dis(0.0, diameter_custom_cumul_prob.back());
434
435 for (unsigned int i = 0; i < number_of_particles; ++i)
436 {
437 // Search to find the appropriate diameter index
438 auto it = std::ranges::upper_bound(diameter_custom_cumul_prob, dis(gen));
439
440 // if dis(gen) returns exactly the maximum value of the cumulative
441 // distribution vector
442 if (it == diameter_custom_cumul_prob.end())
443 {
444 it = it - 1; // Move back to the last valid element
445 }
446
447 unsigned int index =
448 static_cast<unsigned int>(it - diameter_custom_cumul_prob.begin());
449
450 this->particle_sizes.push_back(diameter_custom_values[index]);
451 }
452}
