10NormalDistribution::NormalDistribution(const double &d_average,
11 const double &d_standard_deviation,
12 const unsigned int &prn_seed,
13 const double &min_cutoff,
14 const double &max_cutoff)
15 : diameter_average(d_average)
16 , standard_deviation(d_standard_deviation)
17 , gen(prn_seed)
18{
19 if (min_cutoff < 0.)
20 {
21 // 2.5 -> approx 99% of all diameters are bigger
22 dia_min_cutoff = diameter_average - 2.5 * standard_deviation;
23 std::cout << dia_min_cutoff << std::endl;
24 AssertThrow(dia_min_cutoff > 0.,
25 ExcMessage(
26 "The \"standard deviation\" parameter is to "
27 "high relative to the \"average diameter\" "
28 "parameter. This would results with frequent negative "
29 "diameter values. To fix this problem, please change "
30 "the value of those two parameters or define a "
31 "\"minimum diameter cutoff\" bigger than \"0.\""));
32 }
33 else
34 dia_min_cutoff = min_cutoff;
35
36 if (max_cutoff < 0.)
37 // 2.5 -> approx 99% of all diameters are smaller
38 dia_max_cutoff = diameter_average + 2.5 * standard_deviation;
39 else
40 dia_max_cutoff = max_cutoff;
41
42 AssertThrow(dia_min_cutoff < dia_max_cutoff,
43 ExcMessage("The \"minimum diameter cutoff\" parameter need to"
44 " be smaller than the \"maximum diameter cutoff\"."));
45}

47void
48NormalDistribution::particle_size_sampling(
49 const unsigned int &number_of_particles)
50{
51 this->particle_sizes.clear();
52 this->particle_sizes.reserve(number_of_particles);
53
54 std::normal_distribution<> distribution{diameter_average, standard_deviation};
55
56 unsigned int n_created_diameter = 0;
57 while (n_created_diameter < number_of_particles)
58 {
59 const double temp_diameter = distribution(gen);
60 if (temp_diameter > dia_min_cutoff && temp_diameter < dia_max_cutoff)
61 {
62 n_created_diameter++;
63 this->particle_sizes.push_back(temp_diameter);
64 }
65 }
66}

80void
81NormalDistribution::print_psd_declaration_string(
82 const unsigned int particle_type,
83 const ConditionalOStream &pcout)
84{
85 const double z_min = (dia_min_cutoff - diameter_average) / standard_deviation;
86 const double z_max = (dia_max_cutoff - diameter_average) / standard_deviation;
87
88 const double q_min = 0.5 * (1.0 + std::erf(z_min / std::numbers::sqrt2));
89 const double q_max = 0.5 * (1.0 + std::erf(z_max / std::numbers::sqrt2));
90
91 pcout << "The particle size distribution of particle type " << particle_type
92 << " is normal." << std::endl;
93
94 if (q_min > 0.25 || q_max < 0.75)
95 pcout
96 << "Warning: The minimal and maximal cutoffs of the distribution are at "
97 << q_min * 100. << " and " << q_max * 100. << " respectively."
98 << std::endl;
99 else
100 pcout << "The minimal and maximal cutoffs of the distribution are at "
101 << q_min * 100. << " and " << q_max * 100. << " respectively."
102 << std::endl;
103}

105LogNormalDistribution::LogNormalDistribution(const double &d_average,
106 const double &d_standard_deviation,
107 const unsigned int &prn_seed,
108 const double min_cutoff,
109 const double max_cutoff)
110 : sigma_ln(std::sqrt(std::log(
111 1. + Utilities::fixed_power<2>(d_standard_deviation / d_average))))
112 , mu_ln(std::log(d_average) -
113 0.5 * Utilities::fixed_power<2>(d_standard_deviation))
114 , gen(prn_seed)
115{
116 if (min_cutoff < 0.)
117 {
118 // approx 99% of all diameters are bigger
119 dia_min_cutoff = std::exp(mu_ln - 2.5 * sigma_ln);
120 AssertThrow(dia_min_cutoff > 0.,
121 ExcMessage(
122 "The \"standard deviation\" parameter is to "
123 "high relative to the \"average diameter\" "
124 "parameter. This would results with frequent negative "
125 "diameter values. To fix this problem, please change "
126 "the value of those two parameters or define a "
127 "\"minimum diameter cutoff\" bigger than \"0.\""));
128 }
129 else
130 dia_min_cutoff = min_cutoff;
131
132 if (max_cutoff < 0.)
133 // approx 99% of all diameters are bigger
134 dia_max_cutoff = std::exp(mu_ln + 2.5 * sigma_ln);
135 else
136 dia_max_cutoff = max_cutoff;
137
138 AssertThrow(dia_min_cutoff < dia_max_cutoff,
139 ExcMessage(
140 "The \"minimum diameter cutoff\" parameter need to be smaller "
141 "than the \"maximum diameter cutoff\"."));
142}

239CustomDistribution::CustomDistribution(
240 const std::vector<double> &d_list,
241 const std::vector<double> &d_probabilities,
242 const unsigned int &prn_seed)
243 : diameter_custom_values(d_list)
244 , gen(prn_seed)
245{
246 std::vector<double> cumulative_probability_vector, n_i_vector;
247 double n_tot = 0.;
248
249 cumulative_probability_vector.reserve(d_probabilities.size());
250 n_i_vector.reserve(d_probabilities.size());
251
252 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
253 {
254 n_i_vector.push_back(d_probabilities[i] / Utilities::fixed_power<3>(
255 diameter_custom_values[i]));
256 n_tot += n_i_vector[i];
257 }
258
259 double i_probability;
260 double cumulative_value = 0.;
261
262 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
263 {
264 i_probability = n_i_vector[i] / n_tot;
265 cumulative_value += i_probability;
266 cumulative_probability_vector.push_back(cumulative_value);
267 }
268
269 diameter_custom_cumul_prob = cumulative_probability_vector;
270}

272void
273CustomDistribution::particle_size_sampling(
274 const unsigned int &number_of_particles)
275{
276 this->particle_sizes.clear();
277 this->particle_sizes.reserve(number_of_particles);
278
279 std::uniform_real_distribution<> dis(0.0, diameter_custom_cumul_prob.back());
280
281 for (unsigned int i = 0; i < number_of_particles; ++i)
282 {
283 // Search to find the appropriate diameter index
284 auto it = std::ranges::upper_bound(diameter_custom_cumul_prob, dis(gen));
285
286 // if dis(gen) returns exactly the maximum value of the cumulative
287 // distribution vector
288 if (it == diameter_custom_cumul_prob.end())
289 {
290 it = it - 1; // Move back to the last valid element
291 }
292
293 unsigned int index =
294 static_cast<unsigned int>(it - diameter_custom_cumul_prob.begin());
295
296 this->particle_sizes.push_back(diameter_custom_values[index]);
297 }
298}
