74CustomDistribution::CustomDistribution(
75 const std::vector<double> &d_list,
76 const std::vector<double> &d_probabilities,
77 const unsigned int &prn_seed)
78 : diameter_custom_values(d_list)
79 , gen(prn_seed)
80{
81 std::vector<double> cumulative_probability_vector, n_i_vector;
82 double n_tot = 0.;
83
84 cumulative_probability_vector.reserve(d_probabilities.size());
85 n_i_vector.reserve(d_probabilities.size());
86
87 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
88 {
89 n_i_vector.push_back(d_probabilities[i] / Utilities::fixed_power<3>(
90 diameter_custom_values[i]));
91 n_tot += n_i_vector[i];
92 }
93
94 double i_probability;
95 double cumulative_value = 0.;
96
97 for (unsigned int i = 0; i < diameter_custom_values.size(); ++i)
98 {
99 i_probability = n_i_vector[i] / n_tot;
100 cumulative_value += i_probability;
101 cumulative_probability_vector.push_back(cumulative_value);
102 }
103
104 diameter_custom_cumul_prob = cumulative_probability_vector;
105}

107void
108CustomDistribution::particle_size_sampling(const unsigned int &particle_number)
109{
110 this->particle_sizes.clear();
111 this->particle_sizes.reserve(particle_number);
112
113 std::uniform_real_distribution<> dis(0.0, diameter_custom_cumul_prob.back());
114
115 for (unsigned int i = 0; i < particle_number; ++i)
116 {
117 // Search to find the appropriate diameter index
118 auto it = std::upper_bound(diameter_custom_cumul_prob.begin(),
119 diameter_custom_cumul_prob.end(),
120 dis(gen));
121
122 // if dis(gen) returns exactly the maximum value of the cumulative
123 // distribution vector
124 if (it == diameter_custom_cumul_prob.end())
125 {
126 it = it - 1; // Move back to the last valid element
127 }
128
129 unsigned int index =
130 static_cast<unsigned int>(it - diameter_custom_cumul_prob.begin());
131
132 this->particle_sizes.push_back(diameter_custom_values[index]);
133 }
134}
