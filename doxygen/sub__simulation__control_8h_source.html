11/**
12 * @file sub_simulation_control.h
13 * @brief Controls smaller simulation blocks within simulations.
14 *
15 * Lethe simulations are controlled using the SimulationControl class and its
16 * derivative. However, some models contain a sub model with it's own
17 * time-stepping. This is the case for the CFD-DEM modules which needs to do a
18 * number of DEM iterations per CFD time-step. Whenever this is required, a sub
19 * simulation control is used. This file contains classes which are used for sub
20 * simulation control. The sub simulation control do not all depend from a
21 * base class because they are all very different and don't share a common
22 * architecture.
23 *
24 */
25
26/**
27 * @brief Controls DEM sub iteration logic.
28 *
29 * The SubSimulationControlDEM class is responsible for managing the progression
30 * of the DEM component of a CFD-DEM simulation.
31 *
32 * This class supports two approaches for the DEM iteration:
33 * - Fixed number of iterations: In this case, a fixed number of DEM iterations
34 * is carried out for each CFD time step. If the CFD uses adaptive time
35 * stepping, then the DEM simulation time step may change dynamically during the
36 * simulation.
37 * - Fixed fraction of Rayleigh characteristic time: In this case, the DEM time
38 * step is fixed to a fraction of the Rayleigh characteristic time step. This
39 * means that the number of DEM iterations per CFD time step is adjusted
40 * dynamically.
41 *
42 */
43
44class SubSimulationControlDEM
45{
46public:
47 /**
48 * @brief Type of iteration method for DEM sub simulation control
49 *
50 * Declares the types of sub simulation control that are available for CFD-DEM
51 * simulations.
52 */
53
54 enum class DEMSubIterationLogic
55 {
56 /// Fixed number of DEM iterations per CFD time step.
57 fixed_number_of_iterations = 0,
58 /// DEM time step is fixed at a constant fraction of the Rayleigh time step.
59 fixed_fraction_of_rayleigh_time_step = 1
60 };
61
62 /**
63 * @brief Construct a SubSimulationControlDEM object
64 *
65 * Depending on the iteration logic, the constructor calculates the total
66 * number of DEM iterations and the DEM time step. When using a fixed number
67 * of iterations, the time step is derived from the time interval and the
68 * coupling frequency. When using a fixed fraction of the Rayleigh
69 * characteristic time, the time step is set to the specified fraction and the
70 * number of iterations is derived accordingly.
71 *
72 * @param[in] iteration_logic The logic used to determine the DEM sub
73 * iteration time step and number of iterations.
74 * @param[in] time_interval The time interval over which the DEM sub
75 * iterations are to be carried out. This corresponds to the CFD time step.
76 * @param[in] coupling_frequency The number of DEM iterations per CFD time
77 * step. Only used when @p iteration_logic is
78 * DEMSubIterationLogic::fixed_number_of_iterations.
79 * @param[in] rayleigh_characteristic_time The Rayleigh characteristic time
80 * step of the particles.
81 * @param[in] fraction_of_rayleigh_characteristic_time The target fraction of
82 * the Rayleigh characteristic time to use as the DEM time step. Only used
83 * when @p iteration_logic is
84 * DEMSubIterationLogic::fixed_fraction_of_rayleigh_time_step.
85 */
86 SubSimulationControlDEM(
87 const DEMSubIterationLogic iteration_logic,
88 const double time_interval,
89 const unsigned int coupling_frequency,
90 const double rayleigh_characteristic_time,
91 const double fraction_of_rayleigh_characteristic_time);
92
93
94 /**
95 * @brief Advance the DEM sub simulation by one iteration
96 *
97 * Increments the iteration counter by one and checks whether additional
98 * iterations are required to complete the DEM sub simulation within the
99 * current CFD time step. As long as this method returns true, the DEM
100 * simulation should continue iterating.
101 *
102 * @return true if more iterations are needed, false if the sub simulation is
103 * complete.
104 */
105 bool
106 iterate();
107
108 /**
109 * @brief Get the current DEM sub iteration number
110 *
111 * @return The current iteration number within the DEM sub simulation.
112 */
113 unsigned int
114 get_iteration() const
115 {
116 return iteration_number;
117 }
118
119 /**
120 * @brief Get the DEM sub simulation time step
121 *
122 * @return The time step used for the DEM sub iterations.
123 */
124 double
125 get_time_step() const
126 {
127 return time_step;
128 }
129
130private:
131 /// Logic used for the DEM iterations
132 const DEMSubIterationLogic iteration_logic;
133
134 /// Time interval over which the time integration is to be carried out.
135 const double time_interval;
136
137 /// Coupling frequency. This is only used when
138 /// iteration_logic=fixed_number_of_iterations
139 const unsigned int coupling_frequency;
140
141 /// Rayleigh characteristic time for the particles
142 const double rayleigh_characteristic_time;
143
144 /// Targeted fraction of the Rayleigh characteristic time. This is only used
145 /// when iteration_logic=fixed_fraction_of_rayleigh_time_step
146 const double fraction_of_rayleigh_characteristic_time;
147
148 /// Current value of the sub iteration number. This is initialized to zero.
149 unsigned int iteration_number;
150
151 /// Total number of iterations. This is the number of iterations that is
152 /// required until the completion of the sub simulation control
153 unsigned int total_number_of_iterations;
154
155 /// Time step value of the sub time stepping
156 double time_step;
157};
