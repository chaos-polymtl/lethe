31/**
32 * @brief Class that assembles the core of the heat transfer equation.
33 * This class assembles the weak form of:
34 * \f$ - k * \nabla^2 T + \rho * cp * \mathbf{u} * \nabla T - f - \tau :
35 * \nabla \mathbf{u} =0 \f$
36 *
37 * @tparam dim An integer that denotes the number of spatial dimensions
38 *
39 * @param simulation_control Shared pointer of the SimulationControl object
40 * controlling the current simulation
41 *
42 * @ingroup assemblers
43 */
44template <int dim>
45class HeatTransferAssemblerCore : public HeatTransferAssemblerBase<dim>
46{
47public:
48 HeatTransferAssemblerCore(
49 const std::shared_ptr<SimulationControl> &simulation_control)
50 : simulation_control(simulation_control)
51 {}
52
53 /**
54 * @brief assemble_matrix Assembles the matrix
55 * @param scratch_data (see base class)
56 * @param copy_data (see base class)
57 */
58 virtual void
59 assemble_matrix(const HeatTransferScratchData<dim> &scratch_data,
60 StabilizedMethodsCopyData &copy_data) override;
61
62
63 /**
64 * @brief assemble_rhs Assembles the rhs
65 * @param scratch_data (see base class)
66 * @param copy_data (see base class)
67 */
68 virtual void
69 assemble_rhs(const HeatTransferScratchData<dim> &scratch_data,
70 StabilizedMethodsCopyData &copy_data) override;
71
72 const std::shared_ptr<SimulationControl> simulation_control;
73};
