32/**
33 * @brief Class that assembles the core of the heat transfer equation.
34 * This class assembles the weak form of:
35 * \f$ - k * \nabla^2 T + \rho * cp * \mathbf{u} * \nabla T - f - \tau :
36 * \nabla \mathbf{u} =0 \f$
37 *
38 * @tparam dim An integer that denotes the number of spatial dimensions
39 *
40 * @param simulation_control Shared pointer of the SimulationControl object
41 * controlling the current simulation
42 *
43 * @ingroup assemblers
44 */
45template <int dim>
46class HeatTransferAssemblerCore : public HeatTransferAssemblerBase<dim>
47{
48public:
49 HeatTransferAssemblerCore(
50 const std::shared_ptr<SimulationControl> &simulation_control)
51 : simulation_control(simulation_control)
52 {}
53
54 /**
55 * @brief assemble_matrix Assembles the matrix
56 * @param scratch_data (see base class)
57 * @param copy_data (see base class)
58 */
59 virtual void
60 assemble_matrix(const HeatTransferScratchData<dim> &scratch_data,
61 StabilizedMethodsCopyData &copy_data) override;
62
63
64 /**
65 * @brief assemble_rhs Assembles the rhs

68 */

70 assemble_rhs(const HeatTransferScratchData<dim> &scratch_data,
