47template <int dim, int spacedim>
48template <bool assemble_matrix>
49void
50FluidDynamicsNitsche<dim, spacedim>::assemble_nitsche_restriction()
51{
52 TimerOutput::Scope t(this->computing_timer, "Nitsche assemble restriction");
53
54 Assert(
55 !this->simulation_parameters.physical_properties_manager.is_non_newtonian(),
56 RequiresConstantViscosity("assemble_nitsche_restriction"));
57
58 // Viscosity for stabilization constant
59 const double kinematic_viscosity =
60 this->simulation_parameters.physical_properties_manager
61 .get_kinematic_viscosity_scale();
62
63 // Time steps and inverse time steps which is used for stabilization constant
64 std::vector<double> time_steps_vector =
65 this->simulation_control->get_time_steps_vector();
66 const double dt = time_steps_vector[0];
67 const double sdt = 1. / dt;
68
69 for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
70 {
71 std::shared_ptr<Particles::ParticleHandler<spacedim>> &solid_ph =
72 solids[i_solid]->get_solid_particle_handler();
73
74
75 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
76
77 std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
78
79 FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
80 dealii::Vector<double> local_rhs(dofs_per_cell);
81
82 Tensor<1, spacedim> velocity;
83 Function<spacedim> *solid_velocity =
84 solids[i_solid]->get_solid_velocity();
85
86 solid_velocity->set_time(this->simulation_control->get_current_time());
87
88 // Penalization terms
89 const double beta =
90 this->simulation_parameters.nitsche->nitsche_solids[i_solid]->beta;
91
92 // Loop over all local particles
93 auto particle = solid_ph->begin();
94 while (particle != solid_ph->end())
95 {
96 local_matrix = 0;
97 local_rhs = 0;
98
99 const auto &cell = particle->get_surrounding_cell();
100 double h_cell = compute_cell_diameter<dim>(cell->measure(),
101 this->velocity_fem_degree);
102 const double penalty_parameter =
103 1. / std::pow(h_cell * h_cell, double(dim) / double(spacedim));
104 const auto &dh_cell =
105 typename DoFHandler<spacedim>::cell_iterator(*cell,
106 &this->dof_handler);
107 dh_cell->get_dof_indices(fluid_dof_indices);
108
109 const auto pic = solid_ph->particles_in_cell(cell);
110 Assert(pic.begin() == particle, ExcInternalError());
111 for (const auto &p : pic)
112 {
113 const ArrayView<const double> properties = p.get_properties();
114
115 velocity = 0;
116 const auto &ref_q = p.get_reference_location();
117 const auto &real_q = p.get_location();
118 const auto &JxW = properties[0];
119
120 for (unsigned int k = 0; k < dofs_per_cell; ++k)
121 {
122 const auto comp_k =
123 this->fe->system_to_component_index(k).first;
124 if (comp_k < spacedim)
125 {
126 // Get the velocity at non-quadrature point (particle in
127 // fluid)
128 auto &evaluation_point = this->evaluation_point;
129 velocity[comp_k] +=
130 evaluation_point[fluid_dof_indices[k]] *
131 this->fe->shape_value(k, ref_q);
132 }
133 }
134
135 const double u_mag = velocity.norm();
136
137
138 const double tau =
139 this->simulation_control->get_assembly_method() ==
140 Parameters::SimulationControl::TimeSteppingMethod::steady ?
141 1. / std::sqrt(std::pow(2. * u_mag / h_cell, 2) +
142 9 * std::pow(4 * kinematic_viscosity /
143 (h_cell * h_cell),
144 2)) :
145 1. /
146 std::sqrt(
147 std::pow(sdt, 2) + std::pow(2. * u_mag / h_cell, 2) +
148 9 * std::pow(4 * kinematic_viscosity / (h_cell * h_cell),
149 2));
150
151 for (unsigned int i = 0; i < dofs_per_cell; ++i)
152 {
153 const auto comp_i =
154 this->fe->system_to_component_index(i).first;
155 if (comp_i < spacedim)
156 {
157 if (assemble_matrix)
158 {
159 for (unsigned int j = 0; j < dofs_per_cell; ++j)
160 {
161 const auto comp_j =
162 this->fe->system_to_component_index(j).first;
163 if (comp_i == comp_j)
164 {
165 local_matrix(i, j) +=
166 penalty_parameter * beta *
167 this->fe->shape_value(i, ref_q) *
168 this->fe->shape_value(j, ref_q) * JxW;
169
170 // Partial jacobian of the strong residual
171 // term so that Nitsche's IB is taken into
172 // account in the stabilization
173 local_matrix(i, j) +=
174 penalty_parameter * beta *
175 this->fe->shape_value(j, ref_q) *
176 (this->fe->shape_grad(i, ref_q) *
177 velocity) *
178 tau * JxW;
179 }
180 }
181 }
182 // Regular residual
183 local_rhs(i) += -penalty_parameter * beta *
184 velocity[comp_i] *
185 this->fe->shape_value(i, ref_q) * JxW +
186 penalty_parameter * beta *
187 solid_velocity->value(real_q, comp_i) *
188 this->fe->shape_value(i, ref_q) * JxW;
189
190 // Residual due taking into account Nitsche IB in the
191 // strong residual used for GLS stabilization
192 const double strong_residual =
193 -penalty_parameter * beta *
194 (solid_velocity->value(real_q, comp_i) -
195 velocity[comp_i]);
196
197 const double test_function =
198 this->fe->shape_grad(i, ref_q) * velocity;
199
200 local_rhs(i) +=
201 -tau * (strong_residual)*test_function * JxW;
202 }
203 }
204 }
205 const AffineConstraints<double> &constraints_used =
206 this->zero_constraints;
207 auto &system_rhs = this->system_rhs;
208 constraints_used.distribute_local_to_global(local_matrix,
209 local_rhs,
210 fluid_dof_indices,
211 this->system_matrix,
212 system_rhs);
213 particle = pic.end();
214 }
215 this->system_matrix.compress(VectorOperation::add);
216 this->system_rhs.compress(VectorOperation::add);
217 }
218}

220template <>
221Tensor<1, 3>
222FluidDynamicsNitsche<2, 3>::calculate_forces_on_solid(
223 const unsigned int i_solid)
224{
225 std::shared_ptr<Particles::ParticleHandler<3>> &solid_ph =
226 solids[i_solid]->get_solid_particle_handler();
227
228 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
229
230 std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
231
232 Tensor<2, 3> velocity_gradient;
233 double pressure;
234 Tensor<1, 3> normal_vector;
235 Tensor<2, 3> fluid_stress;
236 Tensor<2, 3> fluid_pressure;
237 Tensor<1, 3> force; // to be changed for a vector of tensors when
238 // allowing multiple solids
239
240
241 Assert(
242 !this->simulation_parameters.physical_properties_manager.is_non_newtonian(),
243 RequiresConstantViscosity("assemble_nitsche_restriction"));
244
245 const double kinematic_viscosity =
246 this->simulation_parameters.physical_properties_manager
247 .get_kinematic_viscosity_scale();
248
249 // Loop over all local particles
250 auto particle = solid_ph->begin();
251 while (particle != solid_ph->end())
252 {
253 const auto &cell = particle->get_surrounding_cell();
254 const auto &dh_cell =
255 typename DoFHandler<3>::cell_iterator(*cell, &this->dof_handler);
256 dh_cell->get_dof_indices(fluid_dof_indices);
257
258 const auto pic = solid_ph->particles_in_cell(cell);
259 Assert(pic.begin() == particle, ExcInternalError());
260
261 // Generate FEField function to evaluate values and gradients
262 // at the particle location
263 auto &evaluation_point = this->evaluation_point;
264
265 Functions::FEFieldFunction<3, GlobalVectorType> fe_field(
266 this->dof_handler, evaluation_point);
267
268
269 fe_field.set_active_cell(dh_cell);
270
271 for (const auto &p : pic)
272 {
273 const ArrayView<const double> properties = p.get_properties();
274
275 velocity_gradient = 0;
276 pressure = 0;
277 const auto &q = p.get_location();
278 const auto &JxW = properties[0];
279 normal_vector[0] = -properties[1];
280 normal_vector[1] = -properties[2];
281 normal_vector[2] = -properties[3];
282
283 for (int k = 0; k < 3; ++k)
284 {
285 velocity_gradient[k] = fe_field.gradient(q, k);
286 }
287
288 pressure = fe_field.value(q, 3);
289
290 for (int d = 0; d < 2; ++d)
291 {
292 fluid_pressure[d][d] = pressure;
293 }
294
295 fluid_stress = kinematic_viscosity *
296 (velocity_gradient + transpose(velocity_gradient)) -
297 fluid_pressure;
298 force += fluid_stress * normal_vector * JxW;
299 }
300
301 particle = pic.end();
302 }
303 force = Utilities::MPI::sum(force, this->mpi_communicator);
304 return force;
305}

382template <int dim, int spacedim>
383Tensor<1, 3>
384FluidDynamicsNitsche<dim, spacedim>::calculate_torque_on_solid(
385 const unsigned int i_solid)
386{
387 std::shared_ptr<Particles::ParticleHandler<spacedim>> &solid_ph =
388 solids[i_solid]->get_solid_particle_handler();
389
390 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
391
392 std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
393
394 // Penalization terms
395 const double beta =
396 this->simulation_parameters.nitsche->nitsche_solids[i_solid]->beta;
397 Tensor<1, spacedim> velocity;
398 Function<spacedim> *solid_velocity = solids[i_solid]->get_solid_velocity();
399
400
401 Tensor<1, 3> torque;
402 torque = 0;
403
404 // Todo center of rotation should be parameter passed.
405 Point<spacedim> center_of_rotation =
406 this->simulation_parameters.nitsche->nitsche_solids[i_solid]->cor;
407
408 // Loop over all local particles
409 auto particle = solid_ph->begin();
410 while (particle != solid_ph->end())
411 {
412 const auto &cell = particle->get_surrounding_cell();
413 double h_cell =
414 compute_cell_diameter<dim>(cell->measure(), this->velocity_fem_degree);
415 const double penalty_parameter =
416 1. / std::pow(h_cell * h_cell, double(dim) / double(spacedim));
417 const auto &dh_cell =
418 typename DoFHandler<spacedim>::cell_iterator(*cell, &this->dof_handler);
419 dh_cell->get_dof_indices(fluid_dof_indices);
420
421 const auto pic = solid_ph->particles_in_cell(cell);
422 Assert(pic.begin() == particle, ExcInternalError());
423 for (const auto &p : pic)
424 {
425 const ArrayView<const double> properties = p.get_properties();
426
427 Tensor<1, spacedim> force;
428 force = 0;
429 velocity = 0;
430 const auto &ref_q = p.get_reference_location();
431 const auto &real_q = p.get_location();
432 const auto &JxW = properties[0];
433
434 for (unsigned int k = 0; k < dofs_per_cell; ++k)
435 {
436 const auto comp_k = this->fe->system_to_component_index(k).first;
437 if (comp_k < spacedim)
438 {
439 // Get the velocity at non-quadrature point (particle in
440 // fluid)
441 auto &evaluation_point = this->evaluation_point;
442 velocity[comp_k] += evaluation_point[fluid_dof_indices[k]] *
443 this->fe->shape_value(k, ref_q);
444 }
445 }
446 for (unsigned int i = 0; i < dofs_per_cell; ++i)
447 {
448 const auto comp_i = this->fe->system_to_component_index(i).first;
449 if (comp_i < spacedim)
450 {
451 force[comp_i] -=
452 penalty_parameter * beta * this->fe->shape_value(i, ref_q) *
453 JxW *
454 (solid_velocity->value(real_q, comp_i) - velocity[comp_i]);
455 }
456 }
457 // Calculate torque on particle location
458 auto distance = real_q - center_of_rotation;
459
460 if (dim == 2)
461 {
462 torque[0] = 0.;
463 torque[1] = 0.;
464 torque[2] += distance[0] * force[1] - distance[1] * force[0];
465 }
466 else if (dim == 3)
467 {
468 torque[0] += distance[1] * force[2] - distance[2] * force[1];
469 torque[1] += distance[2] * force[0] - distance[0] * force[2];
470 torque[2] += distance[0] * force[1] - distance[1] * force[0];
471 }
472 }
473 particle = pic.end();
474 }
475 torque = Utilities::MPI::sum(torque, this->mpi_communicator);
476 return torque;
477}

479template <int dim, int spacedim>
480void
481FluidDynamicsNitsche<dim, spacedim>::postprocess_solid_forces(
482 const unsigned int i_solid,
483 bool first_solid_forces)
484{
485 TimerOutput::Scope t(this->computing_timer, "Calculate force on solid");
486
487 std::vector<Tensor<1, spacedim>> force;
488 std::vector<unsigned int> solid_indices;
489
490 force.emplace_back(this->calculate_forces_on_solid(i_solid));
491 solid_indices.emplace_back(i_solid);
492
493 if (this->simulation_parameters.nitsche->verbosity ==
494 Parameters::Verbosity::verbose &&
495 this->this_mpi_process == 0)
496 {
497 std::string independent_column_names = "Solid ID";
498 std::vector<std::string> dependent_column_names;
499 dependent_column_names.emplace_back("f_x");
500 dependent_column_names.emplace_back("f_y");
501 if (spacedim == 3)
502 dependent_column_names.emplace_back("f_z");
503
504 TableHandler table = make_table_scalars_tensors(
505 solid_indices,
506 independent_column_names,
507 force,
508 dependent_column_names,
509 this->simulation_parameters.simulation_control.log_precision,
510 true);
511
512 // printed only for the first i_solid for which calculate_forces is true
513 // to be improved so that the column_names are not repeated
514 if (first_solid_forces)
515 {
516 std::cout << std::endl;
517 std::cout << "+------------------------------------------+"
518 << std::endl;
519 std::cout << "| Force on solid summary |"
520 << std::endl;
521 std::cout << "+------------------------------------------+"
522 << std::endl;
523 }
524 table.write_text(std::cout);
525 }
526
527 if (this->simulation_control->is_steady())
528 {
529 solid_forces_table[i_solid].add_value(
530 "cells", this->triangulation->n_global_active_cells());
531 }
532 else
533 {
534 solid_forces_table[i_solid].add_value(
535 "time", this->simulation_control->get_current_time());
536 solid_forces_table[i_solid].set_precision(
537 "time", this->simulation_parameters.forces_parameters.output_precision);
538 }
539 solid_forces_table[i_solid].add_value("f_x", force[0][0]);
540 solid_forces_table[i_solid].add_value("f_y", force[0][1]);
541 if (dim == 3)
542 solid_forces_table[i_solid].add_value("f_z", force[0][2]);
543 else
544 solid_forces_table[i_solid].add_value("f_z", 0.);
545
546 // Precision
547 solid_forces_table[i_solid].set_precision(
548 "f_x", this->simulation_parameters.forces_parameters.output_precision);
549 solid_forces_table[i_solid].set_precision(
550 "f_y", this->simulation_parameters.forces_parameters.output_precision);
551 solid_forces_table[i_solid].set_precision(
552 "f_z", this->simulation_parameters.forces_parameters.output_precision);
553
554 std::string filename_force =
555 this->simulation_parameters.simulation_control.output_folder +
556 this->simulation_parameters.nitsche->nitsche_solids[i_solid]
557 ->force_output_name +
558 "_" + Utilities::int_to_string(i_solid, 2) + ".dat";
559 std::ofstream output_force(filename_force.c_str());
560
561 solid_forces_table[i_solid].write_text(output_force);
562}

648template <int dim, int spacedim>
649void
650FluidDynamicsNitsche<dim, spacedim>::solve()
651{
652 // Fluid setup
653 read_mesh_and_manifolds(
654 *this->triangulation,
655 this->simulation_parameters.mesh,
656 this->simulation_parameters.manifolds_parameters,
657 this->simulation_parameters.restart_parameters.restart,
658 this->simulation_parameters.boundary_conditions);
659
660 this->setup_dofs();
661
662 // Solid setup
663 if (!this->simulation_parameters.restart_parameters.restart)
664 {
665 TimerOutput::Scope t(this->computing_timer,
666 "Nitsche setup solid mesh and particles");
667 for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
668 {
669 solids[i_solid]->initial_setup();
670
671 // Output initial configuration, if output_frequency!=0
672 if (this->simulation_control->is_output_iteration())
673 {
674 output_solid_particles(i_solid);
675 output_solid_triangulation(i_solid);
676 }
677 }
678 if constexpr (dim == spacedim)
679 {
680 // Parse the nitsche solids to the multiphysics interface
681 this->multiphysics->set_solid(&solids);
682 }
683 }
684
685 this->set_initial_condition(
686 this->simulation_parameters.initial_condition->type,
687 this->simulation_parameters.restart_parameters.restart);
688
689
690
691 // Time integration
692 while (this->simulation_control->integrate())
693 {
694 this->simulation_control->print_progression(this->pcout);
695
696 {
697 TimerOutput::Scope t(this->computing_timer, "Nitsche move particles");
698 for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
699 {
700 if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
701 ->enable_particles_motion)
702 {
703 // Particle and solid displacement is explicit, thus it must go
704 // from t to t+dt
705 const double time_step =
706 this->simulation_control->get_time_step();
707 const double initial_time =
708 this->simulation_control->get_current_time() - time_step;
709
710 solids[i_solid]->update_temperature_time(time_step);
711 solids[i_solid]->integrate_velocity(time_step, initial_time);
712 solids[i_solid]->move_solid_triangulation(time_step,
713 initial_time);
714 }
715 }
716 if constexpr (dim == spacedim)
717 {
718 // Parse the nitsche solids to the multiphysics interface
719 this->multiphysics->set_solid(&solids);
720 }
721 }
722 if (this->simulation_control->is_at_start())
723 this->iterate();
724 else
725 {
726 this->refine_mesh();
727 this->iterate();
728 }
729
730 this->postprocess(false);
731 bool first_solid_forces(true);
732 bool first_solid_torques(true);
733
734 for (unsigned int i_solid = 0;
735 i_solid < this->simulation_parameters.nitsche->number_solids;
736 ++i_solid)
737 {
738 if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
739 ->calculate_force_on_solid)
740 {
741 postprocess_solid_forces(i_solid, first_solid_forces);
742 first_solid_forces = false;
743 }
744 if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
745 ->calculate_torque_on_solid)
746 {
747 // for 22 and 33 simulations
748 //(to be implemented for 23 simulations)
749 if (dim == spacedim)
750 {
751 postprocess_solid_torques(i_solid, first_solid_torques);
752 first_solid_torques = false;
753 }
754 }
755
756 if (this->simulation_control->is_output_iteration())
757 {
758 output_solid_particles(i_solid);
759 output_solid_triangulation(i_solid);
760 }
761 }
762 this->finish_time_step();
763 }
764 if (this->simulation_parameters.test.enabled)
765 {
766 for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
767 {
768 solids[i_solid]->print_particle_positions();
769 }
770 }
771 this->finish_simulation();
772}

850template <int dim, int spacedim>
851void
852FluidDynamicsNitsche<dim, spacedim>::refine_mesh()
853{
854 bool refinement_step;
855 if (this->simulation_parameters.mesh_adaptation.refinement_at_frequency)
856 refinement_step = this->simulation_control->get_step_number() %
857 this->simulation_parameters.mesh_adaptation.frequency ==
858 0;
859 else
860 refinement_step = this->simulation_control->get_step_number() == 0;
861 if (refinement_step)
862 {
863 // If no adaptation is to be carried out, get out of the function
864 if (this->simulation_parameters.mesh_adaptation.type ==
865 Parameters::MeshAdaptation::Type::none)
866 return;
867
868 // Prepare the solid particle handlers for the mesh refinement
869 // All type of refinement except none will require that the particle
870 // handler be prepared for refinement
871 for (const auto &solid : solids)
872 solid->get_solid_particle_handler()
873 ->prepare_for_coarsening_and_refinement();
874
875
876 if (this->simulation_parameters.mesh_adaptation.type ==
877 Parameters::MeshAdaptation::Type::kelly)
878 this->refine_mesh_kelly();
879
880 else if (this->simulation_parameters.mesh_adaptation.type ==
881 Parameters::MeshAdaptation::Type::uniform)
882 this->refine_mesh_uniform();
883
884 // Unpact them after refinement has occured
885 for (const auto &solid : solids)
886 solid->get_solid_particle_handler()
887 ->unpack_after_coarsening_and_refinement();
888 }
889}

954FluidDynamicsNitsche<dim, spacedim>::read_checkpoint()
955{
956 this->FluidDynamicsMatrixBased<spacedim>::read_checkpoint();
957
958 TimerOutput::Scope t(this->computing_timer,
959 "Nitsche reload solid mesh and particles");
960
961 // Reload initial configurations
962 // This must be done after the background triangulation is read
963 // because the solid particle_handler is deserialized physically
