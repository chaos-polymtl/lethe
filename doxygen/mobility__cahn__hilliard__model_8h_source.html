143/**
144 * @brief Quartic mobility model.
145 *
146 * The mobility function is the following : \f$M(\phi) = D(1-\phi^2)^2 \f$
147 * where D is the mobility constant.
148 */
149class MobilityCahnHilliardModelQuartic : public MobilityCahnHilliardModel
150{
151public:
152 /**
153 * @brief Default constructor.
154 *
155 * @param[in] p_mobility_cahn_hilliard_constant the user defined mobility
156 * constant.
157 */
158 MobilityCahnHilliardModelQuartic(
159 const double p_mobility_cahn_hilliard_constant)
160 : mobility_cahn_hilliard_constant(p_mobility_cahn_hilliard_constant)
161 {
162 this->model_depends_on[field::phase_order_cahn_hilliard] = true;
163 }
164
165 /**
166 * @brief Destructor of derived class.
167 */
168 ~MobilityCahnHilliardModelQuartic() = default;
169
170 /**
171 * @brief Return the mobility constant.
172 * @return Value of the mobility constant.
173 */
174 double
175 get_mobility_constant() override
176 {
177 return mobility_cahn_hilliard_constant;
178 }
179
180 /**
181 * @brief Calculate the mobility.
182 * @param[in] fields_value Value of the various fields on which the mobility
183 * may depend.
184 * @return Value of the mobility calculated with the fields_value.
185 */
186 double
187 value(const std::map<field, double> &fields_value) override
188 {
189 Assert(
190 fields_value.find(field::phase_order_cahn_hilliard) != fields_value.end(),
191 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
192 "phase_order_cahn_hilliard"));
193 const double &phase_order_cahn_hilliard =
194 fields_value.at(field::phase_order_cahn_hilliard);
195
196 // The phase order values are clamped to avoid unphysical mobilities in the
197 // bulk phases.
198 return mobility_cahn_hilliard_constant *
199 std::pow((1 - std::min(1.0,
200 phase_order_cahn_hilliard *
201 phase_order_cahn_hilliard)),
202 2);
203 }
204
205 /**
206 * @brief Calculate the vector of mobility.
207 * @param[in] field_vectors Vectors of the fields on which the mobility
208 * may depend.
209 * @param[out] property_vector Vector of the mobility values.
210 */
211 void
212 vector_value(const std::map<field, std::vector<double>> &field_vectors,
213 std::vector<double> &property_vector) override
214 {
215 Assert(field_vectors.find(field::phase_order_cahn_hilliard) !=
216 field_vectors.end(),
217 PhysicialPropertyModelFieldUndefined(
218 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
219 const std::vector<double> &phase_order_cahn_hilliard =
220 field_vectors.at(field::phase_order_cahn_hilliard);
221 for (unsigned int i = 0; i < property_vector.size(); ++i)
222 {
223 property_vector[i] =
224 mobility_cahn_hilliard_constant *
225 std::pow((1 - std::min(1.0,
226 phase_order_cahn_hilliard[i] *
227 phase_order_cahn_hilliard[i])),
228 2);
229 }
230 }
231
232 /**
233 * @brief Calculate the jacobian (the partial derivative) of the
234 * mobility with respect to a field.
235 * @param[in] field_values Value of the various fields on which the mobility
236 * may depend.
237 * @param[in] id Indicator of the field with respect to which the Jacobian
238 * should be calculated.
239 * @return Value of the partial derivative of the mobility with respect to
240 * the field.
241 */
242
243 double
244 jacobian(const std::map<field, double> &fields_value, field /*id*/) override
245 {
246 Assert(
247 fields_value.find(field::phase_order_cahn_hilliard) != fields_value.end(),
248 PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
249 "phase_order_cahn_hilliard"));
250 const double &phase_order_cahn_hilliard =
251 fields_value.at(field::phase_order_cahn_hilliard);
252
253 return -4 *
254 (std::max(-1.0, std::min(phase_order_cahn_hilliard, 0.0)) +
255 std::min(1.0, std::max(phase_order_cahn_hilliard, 0.0))) *
256 mobility_cahn_hilliard_constant *
257 (1 -
258 std::min(1.0,
259 phase_order_cahn_hilliard * phase_order_cahn_hilliard));
260 }
261
262 /**
263 * @brief Calculate the derivative of the mobility with respect to a field.
264 * @param[in] field_vectors Vector for the values of the fields used to
265 * evaluate the mobility.
266 * @param[in] id Identifier of the field with respect to which a derivative
267 * should be calculated.
268 * @param[out] jacobian_vector Vector of the values of the derivatives of the
269 * mobility with respect to the field id.
270 */
271
272 void
273 vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
274 const field /*id*/,
275 std::vector<double> &jacobian_vector) override
276 {
277 Assert(field_vectors.find(field::phase_order_cahn_hilliard) !=
278 field_vectors.end(),
279 PhysicialPropertyModelFieldUndefined(
280 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
281 const std::vector<double> &phase_order_cahn_hilliard =
282 field_vectors.at(field::phase_order_cahn_hilliard);
283 for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
284 jacobian_vector[i] =
285 -4 *
286 (std::max(-1.0, std::min(phase_order_cahn_hilliard[i], 0.0)) +
287 std::min(1.0, std::max(phase_order_cahn_hilliard[i], 0.0))) *
288 mobility_cahn_hilliard_constant *
289 (1 -
290 std::min(1.0,
291 phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]));
292 }
293
294private:
295 const double mobility_cahn_hilliard_constant;
296};
