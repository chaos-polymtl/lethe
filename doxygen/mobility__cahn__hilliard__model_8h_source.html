142/**
143 * @brief Quartic mobility model.
144 *
145 * The mobility function is the following : \f$M(\phi) = D(1-\phi^2)^2 \f$
146 * where D is the mobility constant.
147 */
148class MobilityCahnHilliardModelQuartic : public MobilityCahnHilliardModel
149{
150public:
151 /**
152 * @brief Default constructor.
153 *
154 * @param[in] p_mobility_cahn_hilliard_constant the user defined mobility
155 * constant.
156 */
157 MobilityCahnHilliardModelQuartic(
158 const double p_mobility_cahn_hilliard_constant)
159 : mobility_cahn_hilliard_constant(p_mobility_cahn_hilliard_constant)
160 {
161 this->model_depends_on[field::phase_order_cahn_hilliard] = true;
162 }
163
164 /**
165 * @brief Destructor of derived class.
166 */
167 ~MobilityCahnHilliardModelQuartic() override = default;
168
169 /**
170 * @brief Return the mobility constant.
171 * @return Value of the mobility constant.
172 */
173 double
174 get_mobility_constant() override
175 {
176 return mobility_cahn_hilliard_constant;
177 }
178
179 /**
180 * @brief Calculate the mobility.
181 * @param[in] fields_value Value of the various fields on which the mobility
182 * may depend.
183 * @return Value of the mobility calculated with the fields_value.
184 */
185 double
186 value(const std::map<field, double> &fields_value) override
187 {
188 Assert(fields_value.contains(field::phase_order_cahn_hilliard),
189 PhysicialPropertyModelFieldUndefined(
190 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
191 const double &phase_order_cahn_hilliard =
192 fields_value.at(field::phase_order_cahn_hilliard);
193
194 // The phase order values are clamped to avoid unphysical mobilities in the
195 // bulk phases.
196 return mobility_cahn_hilliard_constant *
197 std::pow((1 - std::min(1.0,
198 phase_order_cahn_hilliard *
199 phase_order_cahn_hilliard)),
200 2);
201 }
202
203 /**
204 * @brief Calculate the vector of mobility.
205 * @param[in] field_vectors Vectors of the fields on which the mobility
206 * may depend.
207 * @param[out] property_vector Vector of the mobility values.
208 */
209 void
210 vector_value(const std::map<field, std::vector<double>> &field_vectors,
211 std::vector<double> &property_vector) override
212 {
213 Assert(field_vectors.contains(field::phase_order_cahn_hilliard),
214 PhysicialPropertyModelFieldUndefined(
215 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
216 const std::vector<double> &phase_order_cahn_hilliard =
217 field_vectors.at(field::phase_order_cahn_hilliard);
218 for (unsigned int i = 0; i < property_vector.size(); ++i)
219 {
220 property_vector[i] =
221 mobility_cahn_hilliard_constant *
222 std::pow((1 - std::min(1.0,
223 phase_order_cahn_hilliard[i] *
224 phase_order_cahn_hilliard[i])),
225 2);
226 }
227 }
228
229 /**
230 * @brief Calculate the jacobian (the partial derivative) of the
231 * mobility with respect to a field.
232 * @param[in] fields_value Value of the various fields on which the mobility
233 * may depend.
234 * @param[in] id Indicator of the field with respect to which the Jacobian
235 * should be calculated.
236 * @return Value of the partial derivative of the mobility with respect to
237 * the field.
238 */
239
240 double
241 jacobian(const std::map<field, double> &fields_value,
242 [[maybe_unused]] field id) override
243 {
244 Assert(fields_value.contains(field::phase_order_cahn_hilliard),
245 PhysicialPropertyModelFieldUndefined(
246 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
247 const double &phase_order_cahn_hilliard =
248 fields_value.at(field::phase_order_cahn_hilliard);
249
250 return -4 *
251 (std::max(-1.0, std::min(phase_order_cahn_hilliard, 0.0)) +
252 std::min(1.0, std::max(phase_order_cahn_hilliard, 0.0))) *
253 mobility_cahn_hilliard_constant *
254 (1 -
255 std::min(1.0,
256 phase_order_cahn_hilliard * phase_order_cahn_hilliard));
257 }
258
259 /**
260 * @brief Calculate the derivative of the mobility with respect to a field.
261 * @param[in] fields_vectors Vector for the values of the fields used to
262 * evaluate the mobility.
263 * @param[in] id Identifier of the field with respect to which a derivative
264 * should be calculated.
265 * @param[out] jacobian_vector Vector of the values of the derivatives of the
266 * mobility with respect to the field id.
267 */
268
269 void
270 vector_jacobian(const std::map<field, std::vector<double>> &fields_vectors,
271 [[maybe_unused]] const field id,
272 std::vector<double> &jacobian_vector) override
273 {
274 Assert(fields_vectors.contains(field::phase_order_cahn_hilliard),
275 PhysicialPropertyModelFieldUndefined(
276 "MobilityCahnHilliardModelQuartic", "phase_order_cahn_hilliard"));
277 const std::vector<double> &phase_order_cahn_hilliard =
278 fields_vectors.at(field::phase_order_cahn_hilliard);
279 for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
280 jacobian_vector[i] =
281 -4 *
282 (std::max(-1.0, std::min(phase_order_cahn_hilliard[i], 0.0)) +
283 std::min(1.0, std::max(phase_order_cahn_hilliard[i], 0.0))) *
284 mobility_cahn_hilliard_constant *
285 (1 -
286 std::min(1.0,
287 phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]));
288 }
289
290private:
291 const double mobility_cahn_hilliard_constant;
292};
