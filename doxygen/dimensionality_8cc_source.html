10 void
11 Dimensionality::define_all_scales()
12 {
13 // Define fundamental quantities using Wikipedia's notation
14 const double L = length;
15 const double M = mass;
16 const double theta = temperature;
17 const double T = time;
18
19 density_scaling = 1. * L * L * L / M;
20 specific_gas_constant_scaling = 1. / L / L * T * T * theta;
21 viscosity_scaling = 1. / L / L * T;
22 specific_heat_scaling = 1. / L / L * T * T * theta;
23 thermal_conductivity_scaling = 1. / M / L * T * T * T * theta;
24 enthalpy_scaling = 1. / M / L / L * T * T;
25 diffusivity_scaling = 1. / L / L * T;
26 thermal_expansion_scaling = 1. * theta;
27 surface_tension_scaling = 1. * T * T / M;
28 surface_tension_gradient_scaling = 1. * theta * T * T / M;
29 cahn_hilliard_mobility_scaling = 1. * M / L / L / L / T;
30 cahn_hilliard_epsilon_scaling = 1. / L;
31 }
