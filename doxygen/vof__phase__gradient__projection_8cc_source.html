6template <int dim>
7void
8VOFPhaseGradientProjection<dim>::assemble_system_matrix_and_rhs()
9{
10 // Reinitialize system matrix and right-hand side (rhs)
11 this->system_matrix = 0;
12 this->system_rhs = 0;
13
14 // Get VOF DoFHandler
15 const DoFHandler<dim> &dof_handler_vof =
16 this->subequations_interface.get_vof_dof_handler();
17
18 // Initialize FEValues for phase fraction gradient projection and VOF
19 FEValues<dim> fe_values_phase_gradient_projection(*this->mapping,
20 *this->fe,
21 *this->cell_quadrature,
22 update_values |
23 update_gradients |
24 update_JxW_values);
25 FEValues<dim> fe_values_vof(*this->mapping,
26 dof_handler_vof.get_fe(),
27 *this->cell_quadrature,
28 update_gradients);
29
30 // Initialize size of arrays
31 const unsigned int n_q_points =
32 fe_values_phase_gradient_projection.get_quadrature().size();
33 const unsigned int n_dofs_per_cell =
34 fe_values_phase_gradient_projection.get_fe().n_dofs_per_cell();
35
36 // Initialize local matrix and rhs
37 FullMatrix<double> local_matrix(n_dofs_per_cell, n_dofs_per_cell);
38 Vector<double> local_rhs(n_dofs_per_cell);
39
40 // Initialize local dof indices array
41 std::vector<types::global_dof_index> local_dof_indices(n_dofs_per_cell);
42
43 // Extractor for phase fraction gradient vector
44 FEValuesExtractors::Vector phase_fraction_gradients(0);
45
46 // Initialize filtered phase fraction gradient solution array
47 std::vector<Tensor<1, dim>> present_filtered_vof_phase_gradients(n_q_points);
48
49 // Initialize shape function arrays
50 std::vector<Tensor<1, dim>> phi(n_dofs_per_cell);
51 std::vector<Tensor<2, dim>> grad_phi(n_dofs_per_cell);
52
53 // Get the diffusion factor
54 const double diffusion_factor =
55 this->simulation_parameters.multiphysics.vof_parameters
56 .surface_tension_force.phase_fraction_gradient_diffusion_factor;
57
58 // Loop over phase gradient projection cells
59 for (const auto &cell : this->dof_handler->active_cell_iterators())
60 {
61 if (cell->is_locally_owned())
62 {
63 // Reinitialize local linear system
64 local_matrix = 0;
65 local_rhs = 0;
66
67 // Get VOF cell iterator
68 typename DoFHandler<dim>::active_cell_iterator vof_cell(
69 &(*this->triangulation),
70 cell->level(),
71 cell->index(),
72 &dof_handler_vof);
73
74 // Reinitialize FEValues with corresponding cells
75 fe_values_phase_gradient_projection.reinit(cell);
76 fe_values_vof.reinit(vof_cell);
77
78 // Get vector of Jacobi determinant times the quadrature weights
79 std::vector<double> JxW_vec =
80 fe_values_phase_gradient_projection.get_JxW_values();
81
82 // Compute cell size
83 auto &fe_phase_gradient_projection =
84 fe_values_phase_gradient_projection.get_fe();
85 const double h =
86 compute_cell_diameter<dim>(compute_cell_measure_with_JxW(JxW_vec),
87 fe_phase_gradient_projection.degree);
88
89 // Get VOF filtered phase fraction gradients
90 fe_values_vof.get_function_gradients(
91 this->subequations_interface.get_vof_filtered_solution(),
92 present_filtered_vof_phase_gradients);
93
94 // Loop over quadrature points
95 for (unsigned int q = 0; q < n_q_points; ++q)
96 {
97 // Shape functions
98 for (unsigned int k = 0; k < n_dofs_per_cell; ++k)
99 {
100 phi[k] = fe_values_phase_gradient_projection
101 [phase_fraction_gradients]
102 .value(k, q);
103 grad_phi[k] = fe_values_phase_gradient_projection
104 [phase_fraction_gradients]
105 .gradient(k, q);
106 }
107
108 // Local linear system assembly
109 for (unsigned int i = 0; i < n_dofs_per_cell; ++i)
110 {
111 // Assemble local matrix
112 for (unsigned int j = 0; j < n_dofs_per_cell; ++j)
113 {
114 local_matrix(i, j) +=
115 (phi[i] * phi[j] +
116 h * h * diffusion_factor *
117 scalar_product(grad_phi[i], grad_phi[j])) *
118 JxW_vec[q];
119 }
120 // Assemble local right-hand side (rhs)
121 local_rhs(i) += phi[i] *
122 present_filtered_vof_phase_gradients[q] *
123 JxW_vec[q];
124 }
125 }
126
127 // Distribute the local contributions to the global system
128 cell->get_dof_indices(local_dof_indices);
129 this->constraints.distribute_local_to_global(local_matrix,
130 local_rhs,
131 local_dof_indices,
132 this->system_matrix,
133 this->system_rhs);
134 }
135 }
136 this->system_matrix.compress(VectorOperation::add);
137 this->system_rhs.compress(VectorOperation::add);
138}
