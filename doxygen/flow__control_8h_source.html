21/**
22 * @brief Class allows to dynamically control the flow with a beta coefficient calculated at each step time.
23 */
24
25template <int dim>
26class FlowControl
27{
28public:
29 FlowControl(const Parameters::DynamicFlowControl &flow_control);
30 /**
31 * @brief calculate_beta. This function calculates a beta coefficient which
32 * applies a force to the flow in order to adjust the flow rate to a desired
33 * average velocity through the previous flow rate. Once the flow rate is
34 * reached, the algorithm calculates a new beta to keep a constant flow rate.
35 *
36 * @param average_velocity. The last step average_velocity
37 *
38 * @param dt. The current time step
39 *
40 * @param step_number. The current step
41 */
42 void
43 calculate_beta(const double &average_velocity,
44 const double &dt,
45 const unsigned int &step_number);
46
47 /**
48 * @brief proportional_flow_controller. This function calculates a beta only
49 * with the last step average velocity. It is used to initialize the beta
50 * if there's no initial beta value at 1st time step and the 2nd time step.
51 */
52 inline double
53 proportional_flow_controller(const double &average_velocity_n,
54 const double &dt)
55 {
56 return 0.5 * alpha * (average_velocity_0 - average_velocity_n) / dt;
57 }
58
59 /**
60 * @brief main_flow_controller. This function calculates a beta coefficient
61 * based on a formula described by Wang, 2009 in "A non-body conformal grid
62 * method for simulations
63 * of laminar and turbulent flows with a compressible
64 * large eddy simulation solver"
65 */
66 inline double
67 main_flow_controller(const double &average_velocity_n, const double &dt)
68 {
69 double beta_n1 = beta_n + alpha *
70 (average_velocity_0 - 2 * average_velocity_n +
71 average_velocity_1n) /
72 dt;
73
74 // If desired average velocity is reached, new beta only maintains the force
75 // to keep the flow at the desired value. If calculated beta is
76 // negative it is set to 0 to avoided +/- force.
77 if (average_velocity_0 * beta_n1 < 0 && no_force == false)
78 return 0.0;
79 else
80 return beta_n1;
81 }
82
83 /**
84 * @brief get_beta. This function gives the beta force of the step time
85 */
86 Tensor<1, dim>
87 get_beta()
88 {
89 return beta;
90 }
91
92 /**
93 * @brief get_beta_particle. This function gives the beta force of the
94 * CFD time step scaled for particles.
95 */
96 inline Tensor<1, 3>
97 get_beta_particles(const double &fluid_density,
98 const double &particle_density)
99 {
100 // The beta force for particles must have a tensor of dim = 3 since the body
101 // force g parameters is a tensor of dim 3 (those forces will add up)
102 Tensor<1, 3> beta_particle;
103 beta_particle[0] = beta[0];
104 beta_particle[1] = beta[1];
105
106 if constexpr (dim == 3)
107 beta_particle[2] = beta[2];
108
109 // From beta_p * rho_p = beta_f * rho_f [=] M/(T²L²) units of ΔP/L
110 return beta_particle * fluid_density / particle_density;
111 }
