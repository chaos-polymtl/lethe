12/**
13 * @brief EvaporationModel. Abstract class that allows to calculate the
14 * evaporation mass, heat and momentum fluxes. This model computes all the terms
15 * required for the assembly of the governing equations, i.e., Navier-Stokes
16 * momentum and energy, when evaporation is considered. The former terms are
17 * the momentum flux (i.e., pressure) acting on the evaporative front in the
18 * normal direction, the heat flux, acting at the same location as a cooling
19 * term, and the fluxes' jacobian, if required.
20 *
21 * The EvaporativeModel is the parent class, and its children are specific
22 * types of evaporative models, e.g., constant, temperature dependent. The
23 * selected model is instantiated in the assemblers' constructor
24 * (NavierStokesVOFAssemblerEvaporation and/or
25 * HeatTransferAssemblerVOFEvaporation), using the model_cast method according
26 * to the evaporation model parameters.
27 *
28 * The assemblers call the required method of the EvaporationModel for either
29 * the assembly of the rhs or matrix terms. Hence, no matter the selected
30 * evaporation model, the same assembler is called, avoiding repetitions of
31 * the assembler classes.
32 *
33 */
34class EvaporationModel
35{
36public:
37 EvaporationModel()
38 {}

383/**
384 * @brief EvaporationModelTemperature. Class that allows to calculate the
385 * evaporation mass, heat, and momentum fluxes when a temperature-dependent
386 * mass flux is considered.
387 *
388 * @param p_evaporation Evaporation model parameters.
389 */
390class EvaporationModelTemperature : public EvaporationModel
391{
392public:
393 EvaporationModelTemperature(const Parameters::Evaporation &p_evaporation)
394 : evaporation_coefficient(p_evaporation.evaporation_coefficient)
395 , recoil_pressure_coefficient(p_evaporation.recoil_pressure_coefficient)
396 , molar_mass(p_evaporation.molar_mass)
397 , boiling_temperature_inv(1.0 / p_evaporation.boiling_temperature)
398 , latent_heat_evaporation(p_evaporation.latent_heat_evaporation)
399 , ambient_pressure(p_evaporation.ambient_pressure)
400 , ambient_gas_density_inv(1.0 / p_evaporation.ambient_gas_density)
401 , liquid_density_inv(1.0 / p_evaporation.liquid_density)
402 , universal_gas_constant(p_evaporation.universal_gas_constant)
403 {
404 model_depends_on[field::temperature] = true;
405 }
406
407 /**
408 * @brief saturation_pressure Calculates the value of the saturation pressure.
409 * @param field_values Map storing the values of the various field on which the
410 * saturated vapor pressure evaluated at a point may depend.
411 * @return Value of the saturation pressure calculated with the field_values.
412 */
413 double
414 saturation_pressure(const std::map<field, double> &field_values)
415 {
416 AssertThrow(field_values.find(field::temperature) != field_values.end(),
417 PhysicialPropertyModelFieldUndefined(
418 "EvaporationModelTemperature", "temperature"));
419 const double temperature_inv =
420 1.0 / (field_values.at(field::temperature) + 1e-16);
421
422 const double R_inv = 1.0 / universal_gas_constant;
423
424 const double L_vap_x_M_x_R_inv =
425 latent_heat_evaporation * molar_mass * R_inv;
426
427 return ambient_pressure *
428 std::exp(-L_vap_x_M_x_R_inv *
429 (temperature_inv - boiling_temperature_inv));
430 }

432 /**
433 * @brief mass_flux Calculates the values of the saturation pressure
434 * for multiple points.
435 * @param field_vectors Map storing the vectors of the values for each field
436 * on which the saturated vapor pressure may depend. Each vector contains the
437 * field values at the points of interest.
438 * @param saturation_pressure_vector Vector of the saturation pressure values.
439 */
440 void
441 saturation_pressure(const std::map<field, std::vector<double>> &field_vectors,
442 std::vector<double> &saturation_pressure_vector)
443 {
444 AssertThrow(field_vectors.find(field::temperature) != field_vectors.end(),
445 PhysicialPropertyModelFieldUndefined(
446 "EvaporationModelTemperature", "temperature"));
447 const std::vector<double> &temperature =
448 field_vectors.at(field::temperature);
449
450 const double R_inv = 1.0 / universal_gas_constant;
451
452 const double L_vap_x_M_x_R_inv =
453 latent_heat_evaporation * molar_mass * R_inv;
454
455 const unsigned int n_pts = saturation_pressure_vector.size();
456
457 for (unsigned int i = 0; i < n_pts; ++i)
458 {
459 const double temperature_inv = 1.0 / (temperature[i] + 1e-16);
460
461 saturation_pressure_vector[i] =
462 ambient_pressure *
463 std::exp(-L_vap_x_M_x_R_inv *
464 (temperature_inv - boiling_temperature_inv));
465 }
466 }

468 /**
469 * @brief mass_flux Calculates the value of the evaporation mass flux.
470 * @param field_values Map storing the values of the various fields on which the
471 * evaporation mass flux may depend.
472 * @return Value of the evaporation mass flux calculated with the field_values.
473 */
474 double
475 mass_flux(const std::map<field, double> &field_values) override
476 {
477 const double saturation_pressure_value = saturation_pressure(field_values);
478 const double temperature_inv =
479 1.0 / (field_values.at(field::temperature) + 1e-16);
480
481 const double R_inv = 1.0 / universal_gas_constant;
482 const double M_x_2PI_inv_x_R_inv = molar_mass / (2.0 * M_PI) * R_inv;
483
484 const double mass_flux_value =
485 evaporation_coefficient * saturation_pressure_value *
486 std::sqrt(M_x_2PI_inv_x_R_inv * temperature_inv);
487
488 return mass_flux_value;
489 }

492 * @brief mass_flux Calculates the values of the evaporation mass flux
493 * for multiple points.
494 * @param field_vectors Map storing the vectors of the values for each field
495 * on which the evaporation mass flux may depend. Each vector contains the
496 * field values at the points of interest.
497 * @param mass_flux_vector Vector of mass flux values.

499 void
500 mass_flux(const std::map<field, std::vector<double>> &field_vectors,

520
521 mass_flux_vector[i] = evaporation_coefficient *
522 saturation_pressure_vector[i] *
523 std::sqrt(M_x_2PI_inv_x_R_inv * temperature_inv);
524 }
525 }

564 /**
565 * @brief heat_flux_jacobian Calculates the jacobian (the partial derivative)
566 * of the evaporation heat flux with respect to a field.
567 * @param field_values Map storing the values of the various fields on which the
568 * heat flux evaluated at a point may depend.
569 * @param id Identifier of the field with respect to which the jacobian
570 * should be calculated.
571 * @return Value of the partial derivative of the heat flux with respect to
572 * the field.
573 */
574 double
575 heat_flux_jacobian(const std::map<field, double> &field_values,
576 const field id) override
577 {
578 AssertThrow(field_values.find(field::temperature) != field_values.end(),
579 PhysicialPropertyModelFieldUndefined(
580 "EvaporationModelTemperature", "temperature"));
581 const double temperature_inv =
582 1.0 / (field_values.at(field::temperature) + 1e-16);
583
584 const double R_inv = 1.0 / universal_gas_constant;
585 const double L_vap_x_M_x_R_inv =
586 latent_heat_evaporation * molar_mass * R_inv;
587 const double M_x_2PI_inv_x_R_inv = molar_mass / (2.0 * M_PI) * R_inv;
588
589 if (id == field::temperature)
590 return latent_heat_evaporation * ambient_pressure *
591 evaporation_coefficient * temperature_inv *
592 std::sqrt(M_x_2PI_inv_x_R_inv * temperature_inv) *
593 std::exp(-L_vap_x_M_x_R_inv *
594 (temperature_inv - boiling_temperature_inv)) *
595 (L_vap_x_M_x_R_inv * temperature_inv - 0.5);
596 else
597 return 0.0;
598 }

666 * flux for multiple points.
667 * @param field_vectors Map storing the vectors of the values for each field
668 * on which the momentum flux may depend. Each vector contains the field
669 * values at the points of interest.
670 * @param momentum_flux_vector Vectors of the momentum flux values.
671 */
672 void
673 momentum_flux(const std::map<field, std::vector<double>> &field_vectors,
674 std::vector<double> &momentum_flux_vector) override
675 {
676 const unsigned int n_pts = momentum_flux_vector.size();
677
678 std::vector<double> saturation_pressure_vector(n_pts);
679 saturation_pressure(field_vectors, saturation_pressure_vector);
680
681 for (unsigned int i = 0; i < n_pts; ++i)
682 momentum_flux_vector[i] =
683 std::max(recoil_pressure_coefficient * saturation_pressure_vector[i] -
684 ambient_pressure,
685 0.0);
686 }
