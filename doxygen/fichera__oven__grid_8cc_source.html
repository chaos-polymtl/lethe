6template <int dim, int spacedim>
7FicheraOvenGrid<dim, spacedim>::FicheraOvenGrid(
8 const std::string &grid_arguments)
9{
10 if constexpr (dim != 3 || spacedim != 3)
11 {
12 AssertThrow(
13 false,
14 ExcMessage(
15 "The Fichera oven mesh is only supported in 3D space with 3D elements."));
16 return;
17 }
18
19 this->grid_arguments = grid_arguments;
20 const std::vector<std::string> arguments =
21 Utilities::split_string_list(grid_arguments, ':');
22
23 if (arguments.size() < 2)
24 {
25 AssertThrow(false,
26 ExcMessage(
27 "The Fichera oven grid requires at least 2 arguments: "
28 "(bottom_left : top_right [: colorize]). "
29 "Points are given as x,y,z and colorize as true/false."));
30 }
31
32 // Parse bottom_left point
33 std::stringstream bottom_left_stream(arguments[0]);
34 std::vector<double> bottom_left_coords;
35 std::string coord_str;
36 while (getline(bottom_left_stream, coord_str, ','))
37 bottom_left_coords.push_back(std::stod(coord_str));
38
39 AssertThrow(bottom_left_coords.size() == static_cast<unsigned int>(dim),
40 ExcMessage("The bottom_left point must have exactly " +
41 std::to_string(dim) + " coordinates (x,y,z format)."));
42
43 this->bottom_left = Point<dim>(bottom_left_coords[0],
44 bottom_left_coords[1],
45 bottom_left_coords[2]);
46
47 // Parse top_right point
48 std::stringstream top_right_stream(arguments[1]);
49 std::vector<double> top_right_coords;
50 while (getline(top_right_stream, coord_str, ','))
51 top_right_coords.push_back(std::stod(coord_str));
52
53 AssertThrow(top_right_coords.size() == static_cast<unsigned int>(dim),
54 ExcMessage("The top_right point must have exactly " +
55 std::to_string(dim) + " coordinates (x,y,z format)."));
56 this->top_right =
57 Point<dim>(top_right_coords[0], top_right_coords[1], top_right_coords[2]);
58
59 // Parse optional colorize flag (defaults to false)
60 this->colorize =
61 (arguments.size() > 2 && arguments[2] == "true") ? true : false;
62}

65template <>
66void
67FicheraOvenGrid<3, 3>::make_grid(Triangulation<3, 3> &triangulation)
68{
69 // Create the initial 2x2x3 subdivided rectangle to be carved
70 Triangulation<3> bulk_tria;
71 std::vector<unsigned int> repetitions = {2, 2, 3};
72 GridGenerator::subdivided_hyper_rectangle(bulk_tria,
73 repetitions,
74 bottom_left,
75 top_right);
76
77 // Compute the dimensions of the box for later use in cell removal
78 const Tensor<1, 3> dimensions = top_right - bottom_left;
79
80 // Remove cells to create the staircase geometry
81 std::set<Triangulation<3>::active_cell_iterator> cells_to_remove;
82 for (const auto &cell : bulk_tria.active_cell_iterators())
83 {
84 const Point<3> cell_center = cell->center();
85
86 // Remove cells in the (+x, +y) quadrant above z > 1/3 height
87 if ((cell_center[0] > dimensions[0] / 2) &&
88 (cell_center[1] > dimensions[1] / 2) &&
89 (cell_center[2] > dimensions[2] / 3))
90 {
91 cells_to_remove.insert(cell);
92 }
93 // Remove cells in the (+x, -y) above z > 2/3 height
94 if ((cell_center[0] > dimensions[0] / 2) &&
95 (cell_center[1] < dimensions[1] / 2) &&
96 (cell_center[2] > 2. * dimensions[2] / 3))
97 {
98 cells_to_remove.insert(cell);
99 }
100 // Remove cells in the (-x, +y) above z > 2/3 height
101 if ((cell_center[0] < dimensions[0] / 2) &&
102 (cell_center[1] > dimensions[1] / 2) &&
103 (cell_center[2] > 2. * dimensions[2] / 3))
104 {
105 cells_to_remove.insert(cell);
106 }
107 }
108
109 // Create the oven
110 GridGenerator::create_triangulation_with_removed_cells(bulk_tria,
111 cells_to_remove,
112 triangulation);
113
114 // Colorize the inlet. For this geometry, the top face of the remaining
115 // staircase (z = top_right[2]) is chosen as the inlet and assigned
116 // boundary_id = 1 for the waveguide problem the Fichera oven is used for; all
117 // other faces keep the default boundary_id = 0. The top face is identified by
118 // checking if the z-coordinate of the face center is close to top_right[2].
119 if (colorize)
120 {
121 for (const auto &face : triangulation.active_face_iterators())
122 if (face->at_boundary())
123 {
124 const Point<3> face_center = face->center();
125 if (std::abs(face_center[2] - dimensions[2]) < 1e-12)
126 face->set_boundary_id(1);
127 }
128 }
129}
