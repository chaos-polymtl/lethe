352 /**
353 * @brief Verify consistency of the input parameters for boundary
354 * conditions to ensure that for every boundary condition within the
355 * triangulation, a boundary condition has been specified in the input file.
356 */
357 void
358 verify_consistency_of_boundary_conditions()
359 {
360 // Sanity check all of the boundary conditions of the triangulation to
361 // ensure that they have a type.
362 std::vector<types::boundary_id> boundary_ids_in_triangulation =
363 this->triangulation->get_boundary_ids();
364 for (auto const &boundary_id_in_tria : boundary_ids_in_triangulation)
365 {
366 AssertThrow(simulation_parameters.boundary_conditions_vof.type.find(
367 boundary_id_in_tria) !=
368 simulation_parameters.boundary_conditions_vof.type.end(),
369 VOFBoundaryConditionMissing(boundary_id_in_tria));
370 }

480 /**
481 * @brief Assemble the system for interface sharpening
482 * * This function assembles the weak form of:
483 * \f$ \Phi = c ^ {(1 - \alpha)} * (\phi ^ \alpha)\f$ if \f$ 0 <=
484 * \phi <= c \ \f$
485 * \f$ \Phi = 1 - (1 - c) ^ {(1 - \alpha)} * (1 - \phi) ^ \alpha \f$ if \f$c <
486 * \phi <= 1 \f$
487 * Reference for sharpening method
488 * https://www.sciencedirect.com/science/article/pii/S0045782500002000
489 *
490 * @param solution VOF solution (phase fraction)
491 *
492 * @param sharpening_threshold Interface sharpening threshold that represents
493 * the mass conservation level
494 *
495 */

497 assemble_L2_projection_interface_sharpening(
498 GlobalVectorType &solution,
499 const double sharpening_threshold);
