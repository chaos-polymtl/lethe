358 /**
359 * @brief Return the metric for residual rescaling. By default, should return 1.
360 * If the rescale_residual_by_volume is set to true, the method
361 * returns the square root of the global volume of the triangulation.
362 *
363 * @return Rescale metric.
364 */
365 double
366 get_residual_rescale_metric() const override
367 {
368 return simulation_parameters.linear_solver.at(PhysicsID::VOF)
369 .rescale_residual_by_volume ?
370 std::sqrt(
371 GridTools::volume(*this->triangulation, *this->mapping)) :
372 1.;
373 }

376 /**
377 * @brief Verify consistency of the input parameters for boundary
378 * conditions to ensure that for every boundary condition within the
379 * triangulation, a boundary condition has been specified in the input file.
380 */
381 void
382 verify_consistency_of_boundary_conditions()
383 {
384 // Sanity check all of the boundary conditions of the triangulation to
385 // ensure that they have a type.
386 std::vector<types::boundary_id> boundary_ids_in_triangulation =
387 this->triangulation->get_boundary_ids();
388 for (auto const &boundary_id_in_tria : boundary_ids_in_triangulation)
389 {
390 AssertThrow(simulation_parameters.boundary_conditions_vof.type.find(
391 boundary_id_in_tria) !=
392 simulation_parameters.boundary_conditions_vof.type.end(),
393 VOFBoundaryConditionMissing(boundary_id_in_tria));
394 }
395 }

504 /**
505 * @brief Assemble the system for interface sharpening
506 * * This function assembles the weak form of:
507 * \f$ \Phi = c ^ {(1 - \alpha)} * (\phi ^ \alpha)\f$ if \f$ 0 <=
508 * \phi <= c \ \f$
509 * \f$ \Phi = 1 - (1 - c) ^ {(1 - \alpha)} * (1 - \phi) ^ \alpha \f$ if \f$c <
510 * \phi <= 1 \f$
511 * Reference for sharpening method
512 * https://www.sciencedirect.com/science/article/pii/S0045782500002000
513 *
514 * @param solution VOF solution (phase fraction)
515 *
516 * @param sharpening_threshold Interface sharpening threshold that represents
517 * the mass conservation level
518 *
519 */
520 void
521 assemble_L2_projection_interface_sharpening(
522 GlobalVectorType &solution,
523 const double sharpening_threshold);

671 * This prevents code duplication throughout the VOF class. The function looks
672 * at the multiphysics interface to decide if the velocity is a block velocity

676 * @param[in] velocity_cell the iterator of the cell where the velocity is to
677 * be reinitialized.
678 *
679 * @param[in] face_no the face index where the velocity is to be

682 * @param[in,out] scratch_data the scratch data to be used for the
683 * reinitialization.

687 const typename DoFHandler<dim>::active_cell_iterator &velocity_cell,
688 const unsigned int &face_no,
689 VOFScratchData<dim> &scratch_data)
690 {
691 if (multiphysics->fluid_dynamics_is_block())
692 {
693 scratch_data.reinit_face_velocity(velocity_cell,
694 face_no,
695 multiphysics->get_block_solution(
696 PhysicsID::fluid_dynamics),
697 this->simulation_parameters.ale);
698 }
699 else
700 {
701 scratch_data.reinit_face_velocity(velocity_cell,
702 face_no,
703 multiphysics->get_solution(
704 PhysicsID::fluid_dynamics),
705 this->simulation_parameters.ale);
706 }
