19/**
20 * @brief Class that stores the information required by the assembly procedure
21 * for the Cahn-Hilliard equations. Consequently, this class
22 * calculates the phase field parameter Phi (values, gradients, laplacians),
23 * the chemical potential eta (values, gradients, laplacians) and the shape
24 * function (values, gradients, laplacians) at all the Gauss points for all
25 * degrees of freedom and stores it into arrays. Additionally, the user can
26 * request that this class gathers additional fields for physics which are
27 * coupled to the Cahn-Hilliard equations, such as the velocity which is
28 * required. This class serves as a separation between the evaluation at the
29 * Gauss point of the variables of interest and their use in the assembly, which
30 * is carried out by the assembler functions. For more information on this
31 * design, the reader can consult deal.II step-9
32 * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
33 * example, the scratch is a struct instead of a templated class because of the
34 * simplicity of step-9.
35 *
36 * @tparam dim An integer that denotes the dimension of the space in which
37 * the flow is solved
38 *
39 * @ingroup solvers
40 **/
41
42template <int dim>
43class CahnHilliardScratchData : public PhysicsScratchDataBase
44{
45public:
46 /**
47 * @brief Constructor. The constructor creates the fe_values that will be used
48 * to fill the member variables of the scratch. It also allocated the
49 * necessary memory for all member variables. However, it does not do any
50 * evaluation, since this needs to be done at the cell level.
51 *
52 * @param properties_manager The physical properties manager (see physical_properties_manager.h)
53 * @param fe_cahn_hilliard The FESystem used to solve the Cahn-Hilliard equations
54 * @param quadrature The quadrature to use for the assembly
55 * @param mapping The mapping of the domain in which the Cahn-Hilliard equations are solved

57 *
58 */

120 /** @brief Reinitializes the content of the scratch.
121 *
122 * Using the FeValues and the content of the solutions and previous solutions,
123 * fills all of the class member of the scratch.
124 *
125 * @param[in] cell The cell over which the assembly is being carried.
126 * This cell must be compatible with the fe which is used to fill the
127 * FeValues.
128 *
129 * @param[in] current_solution The present value of the solution for [Phi,eta]
130 *
131 * @param[in] previous_solutions The solutions at the previous time steps.
132 *
133 * @param[in] source_function The function describing the source term in
134 * Cahn-Hilliard equations.
135 *
136 */
137
138 template <typename VectorType>
139 void
140 reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
141 const VectorType &current_solution,
142 const std::vector<VectorType> &previous_solutions,
143 Function<dim> *source_function)
144 {
145 this->phase_order.component = 0;
146 this->chemical_potential.component = 1;
147
148 this->fe_values_cahn_hilliard.reinit(cell);
149
150 quadrature_points = this->fe_values_cahn_hilliard.get_quadrature_points();
151 auto &fe_cahn_hilliard = this->fe_values_cahn_hilliard.get_fe();
152
153 source_function->value_list(quadrature_points, source_phase_order, 0);
154 source_function->value_list(quadrature_points,
155 source_chemical_potential,
156 1);
157
158 // Compute cell diameter
159 double cell_measure = compute_cell_measure_with_JxW(
160 this->fe_values_cahn_hilliard.get_JxW_values());
161 this->cell_size =
162 compute_cell_diameter<dim>(cell_measure, fe_cahn_hilliard.degree);
163
164 // Gather Phi and eta (values, gradient and laplacian)
165 this->fe_values_cahn_hilliard[phase_order].get_function_values(
166 current_solution, this->phase_order_values);
167 this->fe_values_cahn_hilliard[phase_order].get_function_gradients(
168 current_solution, this->phase_order_gradients);
169 this->fe_values_cahn_hilliard[phase_order].get_function_laplacians(
170 current_solution, this->phase_order_laplacians);
171
172 this->fe_values_cahn_hilliard[chemical_potential].get_function_values(
173 current_solution, this->chemical_potential_values);
174 this->fe_values_cahn_hilliard[chemical_potential].get_function_gradients(
175 current_solution, this->chemical_potential_gradients);
176 this->fe_values_cahn_hilliard[chemical_potential].get_function_laplacians(
177 current_solution, this->chemical_potential_laplacians);
178
179 // Gather previous phase order values
180 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
181 {
182 this->fe_values_cahn_hilliard[phase_order].get_function_values(
183 previous_solutions[p], previous_phase_order_values[p]);
184 }
185
186 // Gather previous chemical potential values
187 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
188 {
189 this->fe_values_cahn_hilliard[chemical_potential].get_function_values(
190 previous_solutions[p], previous_chemical_potential_values[p]);
191 }
192
193 for (unsigned int q = 0; q < n_q_points; ++q)
194 {
195 this->JxW[q] = this->fe_values_cahn_hilliard.JxW(q);
196
197 for (unsigned int k = 0; k < n_dofs; ++k)
198 {
199 // Shape functions for the phase order
200 this->phi_phase[q][k] =
201 this->fe_values_cahn_hilliard[phase_order].value(k, q);
202 this->grad_phi_phase[q][k] =
203 this->fe_values_cahn_hilliard[phase_order].gradient(k, q);
204 this->hess_phi_phase[q][k] =
205 this->fe_values_cahn_hilliard[phase_order].hessian(k, q);
206 this->laplacian_phi_phase[q][k] = trace(this->hess_phi_phase[q][k]);
207
208 // Shape functions for the chemical potential
209 this->phi_potential[q][k] =
210 this->fe_values_cahn_hilliard[chemical_potential].value(k, q);
211 this->grad_phi_potential[q][k] =
212 this->fe_values_cahn_hilliard[chemical_potential].gradient(k, q);
213 this->hess_phi_potential[q][k] =
214 this->fe_values_cahn_hilliard[chemical_potential].hessian(k, q);
215 this->laplacian_phi_potential[q][k] =
216 trace(this->hess_phi_potential[q][k]);
217 }
218 }
219
220 this->is_boundary_cell =
221 cell->at_boundary(); // The attribute needs to be updated because the
222 // assembler for the angle of contact boundary
223 // condition needs to know if the cell is at the
224 // boundary
225 if (this->is_boundary_cell)
226 {
227 n_faces = cell->n_faces();
228 is_boundary_face = std::vector<bool>(n_faces, false);
229 n_faces_q_points = fe_face_values_cahn_hilliard.get_quadrature().size();
230 boundary_face_id = std::vector<unsigned int>(n_faces);
231
232 face_JxW.reinit(n_faces, n_faces_q_points);
233 this->grad_phi_face_phase.reinit(n_faces, n_faces_q_points, n_dofs);
234
235 this->face_phase_grad_values = std::vector<std::vector<Tensor<1, dim>>>(
236 n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
237
238 this->face_normal = std::vector<std::vector<Tensor<1, dim>>>(
239 n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
240
241 for (const auto face : cell->face_indices())
242 {
243 this->is_boundary_face[face] = cell->face(face)->at_boundary();
244 if (this->is_boundary_face[face])
245 {
246 fe_face_values_cahn_hilliard.reinit(cell, face);
247 boundary_face_id[face] = cell->face(face)->boundary_id();
248
249 this->fe_face_values_cahn_hilliard[phase_order]
250 .get_function_gradients(current_solution,
251 this->face_phase_grad_values[face]);
252
253 for (unsigned int q = 0; q < n_faces_q_points; ++q)
254 {
255 face_JxW[face][q] = fe_face_values_cahn_hilliard.JxW(q);
256 this->face_normal[face][q] =
257 this->fe_face_values_cahn_hilliard.normal_vector(q);
258 for (const unsigned int k :
259 fe_face_values_cahn_hilliard.dof_indices())
260 {
261 this->grad_phi_face_phase[face][q][k] =
262 this->fe_face_values_cahn_hilliard[phase_order]
263 .gradient(k, q);
264 }
265 }
266 }
267 }
268 }
269 }

272 template <typename VectorType>
273 void
274 reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
275 const VectorType &current_solution,
276 const Parameters::ALE<dim> &ale)
277 {
278 FEValuesExtractors::Vector velocities(0);
279 this->fe_values_fd.reinit(cell);
280
281 this->fe_values_fd[velocities].get_function_values(current_solution,
282 velocity_values);
283
284 if (!ale.enabled())
285 return;
286
287
288 // ALE enabled, so extract the ALE velocity and subtract it from the
289 // velocity obtained from the fluid dynamics
290 Tensor<1, dim> velocity_ale;
291 std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
292 ale.velocity;
293 Vector<double> velocity_ale_vector(dim);
294
295 for (unsigned int q = 0; q < n_q_points; ++q)
296 {
297 velocity_ale_function->vector_value(quadrature_points[q],
298 velocity_ale_vector);
299 for (int d = 0; d < dim; ++d)
300 velocity_ale[d] = velocity_ale_vector[d];
301
302 velocity_values[q] -= velocity_ale;
303 }
304 }
