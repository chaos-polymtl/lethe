9template <int dim>
10void
11NavierStokesScratchData<dim>::allocate()
12{
13 // Initialize size of arrays
14 this->n_q_points = fe_values.get_quadrature().size();
15 this->n_dofs = fe_values.get_fe().n_dofs_per_cell();
16
17 // Initialize arrays related to quadrature
18 this->JxW = std::vector<double>(n_q_points);
19
20 // Initialize component array
21 this->components = std::vector<unsigned int>(n_dofs);
22
23 // Forcing term array
24 this->rhs_force =
25 std::vector<Vector<double>>(n_q_points, Vector<double>(dim + 1));
26 this->force = std::vector<Tensor<1, dim>>(n_q_points);
27 this->mass_source = std::vector<double>(n_q_points);
28
29 // Initialize arrays related to velocity and pressure
30 this->velocities.first_vector_component = 0;
31 this->pressure.component = dim;
32
33 // Velocity
34 this->velocity_values = std::vector<Tensor<1, dim>>(n_q_points);
35 this->velocity_divergences = std::vector<double>(n_q_points);
36 this->velocity_gradients = std::vector<Tensor<2, dim>>(n_q_points);
37 this->velocity_laplacians = std::vector<Tensor<1, dim>>(n_q_points);
38 this->velocity_hessians = std::vector<Tensor<3, dim>>(n_q_points);
39 this->shear_rate = std::vector<double>(n_q_points);
40
41 // For SDIRK method: sum(a_ij * k_j)
42 if (this->simulation_control->is_sdirk())
43 {
44 this->sdirk_stage_sum = std::vector<Tensor<1, dim>>(n_q_points);
45 }
46
47 // Velocity for BDF schemes
48 this->previous_velocity_values = std::vector<std::vector<Tensor<1, dim>>>(
49 maximum_number_of_previous_solutions(),
50 std::vector<Tensor<1, dim>>(n_q_points));
51
52 // Pressure
53 this->pressure_values = std::vector<double>(n_q_points);
54 this->pressure_gradients = std::vector<Tensor<1, dim>>(n_q_points);
55 this->pressure_scaling_factor = 1;
56
57 // Pressure for BDF schemes (compressible Navier-Stokes)
58 this->previous_pressure_values =
59 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
60 std::vector<double>(n_q_points));
61 // Initialize arrays related to shape functions
62 // Velocity shape functions
63 this->phi_u = std::vector<std::vector<Tensor<1, dim>>>(
64 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
65 this->grad_phi_u = std::vector<std::vector<Tensor<2, dim>>>(
66 n_q_points, std::vector<Tensor<2, dim>>(n_dofs));
67 this->div_phi_u =
68 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
69 this->hess_phi_u = std::vector<std::vector<Tensor<3, dim>>>(
70 n_q_points, std::vector<Tensor<3, dim>>(n_dofs));
71 this->laplacian_phi_u = std::vector<std::vector<Tensor<1, dim>>>(
72 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
73
74 // Pressure shape functions
75 this->phi_p =
76 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
77 this->grad_phi_p = std::vector<std::vector<Tensor<1, dim>>>(
78 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
79
80 // Physical properties
81 fields.insert(
82 std::pair<field, std::vector<double>>(field::pressure, n_q_points));
83 fields.insert(
84 std::pair<field, std::vector<double>>(field::shear_rate, n_q_points));
85
86 density = std::vector<double>(n_q_points);
87 dynamic_viscosity = std::vector<double>(n_q_points);
88 kinematic_viscosity = std::vector<double>(n_q_points);
89 dynamic_viscosity_for_stabilization = std::vector<double>(n_q_points);
90 kinematic_viscosity_for_stabilization = std::vector<double>(n_q_points);
91 thermal_expansion = std::vector<double>(n_q_points);
92 grad_kinematic_viscosity_shear_rate = std::vector<double>(n_q_points);
93
94 previous_density =
95 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
96 std::vector<double>(this->n_q_points));
97}

369template <int dim>
370void
371NavierStokesScratchData<dim>::reinit_mortar(
372 const typename DoFHandler<dim>::active_cell_iterator &cell,
373 const Parameters::Mortar<dim> &mortar_parameters,
374 const double &radius)
375{
376 auto cell_center = cell->center();
377
378 // Get updated rotor angular velocity
379 mortar_parameters.rotor_angular_velocity->set_time(
380 this->simulation_control->get_current_time());
381 const double rotor_angular_velocity =
382 mortar_parameters.rotor_angular_velocity->value(Point<dim>());
383
384 // Compute radius between center of rotation and current cell center
385 double radius_current =
386 cell_center.distance(mortar_parameters.center_of_rotation);
387
388 // Use prescribed rotor angular velocity only if cell is part of the rotor
389 double cell_rotor_angular_velocity;
390 if (radius_current > radius)
391 cell_rotor_angular_velocity = 0.0;
392 else
393 cell_rotor_angular_velocity = rotor_angular_velocity;
394
395 // Compute rotor linear velocity at quadrature points
396 rotor_linear_velocity_values = std::vector<Tensor<1, dim>>(this->n_q_points);
397 for (unsigned int q = 0; q < n_q_points; ++q)
398 {
399 const auto x = fe_values.quadrature_point(q)[0];
400 const auto y = fe_values.quadrature_point(q)[1];
401 rotor_linear_velocity_values[q][0] = -cell_rotor_angular_velocity * y;
402 rotor_linear_velocity_values[q][1] = cell_rotor_angular_velocity * x;
403 }
404}

431template <int dim>
432void
433NavierStokesScratchData<dim>::calculate_physical_properties()
434{
435 if (properties_manager.field_is_required(field::temperature) &&
436 gather_temperature)
437 {
438 set_field_vector(field::temperature,
439 this->temperature_values,
440 this->fields);
441 }
442
443 if (properties_manager.field_is_required(field::pressure))
444 {
445 set_field_vector(field::pressure, this->pressure_values, this->fields);
446 }
447
448 if (properties_manager.field_is_required(field::shear_rate))
449 {
450 for (unsigned int q = 0; q < n_q_points; ++q)
451 {
452 // Calculate shear rate (at each q)
453 const Tensor<2, dim> shear_rate_tensor =
454 velocity_gradients[q] + transpose(velocity_gradients[q]);
455
456 // Calculate the shear rate magnitude
457 shear_rate[q] = calculate_shear_rate_magnitude(shear_rate_tensor);
458 }
459
460 set_field_vector(field::shear_rate, shear_rate, this->fields);
461 }
462
463 switch (properties_manager.get_number_of_fluids())
464 {
465 case 1:
466 {
467 // In the case of an incompressible flow, viscosity is the only
468 // required property
469 const auto rheology_model = properties_manager.get_rheology();
470 rheology_model->vector_value(fields, kinematic_viscosity);
471
472 rheology_model->get_kinematic_viscosity_for_stabilization_vector(
473 fields, kinematic_viscosity_for_stabilization);
474
475 kinematic_viscosity_scale =
476 rheology_model->get_kinematic_viscosity_scale();
477
478 if (!properties_manager.density_is_constant())
479 {
480 // For a weakly compressible flow, density variations will play a
481 // role
482 const auto density_model = properties_manager.get_density();
483 density_model->vector_value(fields, density);
484 density_psi = density_model->get_psi();
485 density_ref = density_model->get_density_ref();
486 // Dynamic viscosity is also necessary for compressible flows
487 rheology_model->get_dynamic_viscosity_vector(density_ref,
488 fields,
489 dynamic_viscosity);
490 rheology_model->get_dynamic_viscosity_for_stabilization_vector(
491 density_ref, fields, dynamic_viscosity_for_stabilization);
492 }
493 else
494 {
495 density_scale = properties_manager.get_density_scale();
496 }
497
498 if (properties_manager.is_non_newtonian())
499 {
500 // Calculate derivative of viscosity with respect to shear rate
501 rheology_model->vector_jacobian(
502 fields, field::shear_rate, grad_kinematic_viscosity_shear_rate);
503 }
504
505 if (gather_temperature)
506 {
507 const auto thermal_expansion_model =
508 properties_manager.get_thermal_expansion();
509 thermal_expansion_model->vector_value(fields, thermal_expansion);
510 }
511 break;
512 }
513 case 2:
514 {
515 // We need both density and viscosity
516 const auto density_model_0 = properties_manager.get_density(0);
517 const auto rheology_model_0 = properties_manager.get_rheology(0);
518 const auto density_model_1 = properties_manager.get_density(1);
519 const auto rheology_model_1 = properties_manager.get_rheology(1);
520
521 density_ref_0 = density_model_0->get_density_ref();
522 density_ref_1 = density_model_1->get_density_ref();
523
524 kinematic_viscosity_scale_0 =
525 rheology_model_0->get_kinematic_viscosity_scale();
526 kinematic_viscosity_scale_1 =
527 rheology_model_1->get_kinematic_viscosity_scale();
528
529 kinematic_viscosity_scale =
530 std::max(kinematic_viscosity_scale_0, kinematic_viscosity_scale_1);
531
532 // Gather properties from material interactions if necessary
533 if (properties_manager.get_number_of_material_interactions() > 0)
534 {
535 const auto material_interaction_id =
536 properties_manager.get_material_interaction_id(
537 material_interactions_type::fluid_fluid, 0, 1);
538 // Gather surface tension
539 const auto surface_tension_model =
540 properties_manager.get_surface_tension(material_interaction_id);
541 surface_tension_model->vector_value(fields, surface_tension);
542 // Gather surface tension gradient only if necessary
543 if (!properties_manager.surface_tension_is_constant())
544 surface_tension_model->vector_jacobian(
545 fields, field::temperature, surface_tension_gradient);
546 }
547
548 density_model_0->vector_value(fields, density_0);
549 rheology_model_0->get_dynamic_viscosity_vector(density_ref_0,
550 fields,
551 dynamic_viscosity_0);
552 rheology_model_0->get_dynamic_viscosity_for_stabilization_vector(
553 density_ref_0, fields, dynamic_viscosity_for_stabilization_0);
554
555 density_model_1->vector_value(fields, density_1);
556 rheology_model_1->get_dynamic_viscosity_vector(density_ref_1,
557 fields,
558 dynamic_viscosity_1);
559 rheology_model_1->get_dynamic_viscosity_for_stabilization_vector(
560 density_ref_1, fields, dynamic_viscosity_for_stabilization_1);
561 if (gather_temperature)
562 {
563 const auto thermal_expansion_model_0 =
564 properties_manager.get_thermal_expansion(0);
565 const auto thermal_expansion_model_1 =
566 properties_manager.get_thermal_expansion(1);
567 thermal_expansion_model_0->vector_value(fields,
568 thermal_expansion_0);
569 thermal_expansion_model_1->vector_value(fields,
570 thermal_expansion_1);
571 }
572
573 if (gather_vof && !gather_cahn_hilliard)
574 {
575 for (unsigned int q = 0; q < this->n_q_points; ++q)
576 {
577 double filtered_phase_value = this->filtered_phase_values[q];
578
579 density[q] = calculate_point_property(filtered_phase_value,
580 this->density_0[q],
581 this->density_1[q]);
582
583 dynamic_viscosity[q] =
584 calculate_point_property(filtered_phase_value,
585 this->dynamic_viscosity_0[q],
586 this->dynamic_viscosity_1[q]);
587
588 kinematic_viscosity[q] = dynamic_viscosity[q] / density[q];
589
590 dynamic_viscosity_for_stabilization[q] =
591 calculate_point_property(
592 filtered_phase_value,
593 this->dynamic_viscosity_for_stabilization_0[q],
594 this->dynamic_viscosity_for_stabilization_1[q]);
595
596 thermal_expansion[q] =
597 calculate_point_property(filtered_phase_value,
598 this->thermal_expansion_0[q],
599 this->thermal_expansion_1[q]);
600 }
601
602 // Gather density_psi for isothermal compressible NS equations
603 if (!properties_manager.density_is_constant())
604 {
605 density_psi_0 = density_model_0->get_psi();
606 density_psi_1 = density_model_1->get_psi();
607
608 for (unsigned int q = 0; q < this->n_q_points; ++q)
609 {
610 double filtered_phase_value =
611 this->filtered_phase_values[q];
612 // To avoid calculating twice in the assemblers
613 compressibility_multiplier[q] =
614 filtered_phase_value *
615 (density_0[q] / density_1[q] * density_psi_1 -
616 density_psi_0) +
617 density_psi_0;
618 }
619 }
620
621 for (unsigned p = 0; p < previous_phase_values.size(); ++p)
622 {
623 // Blend the physical properties using the VOF field
624 for (unsigned int q = 0; q < this->n_q_points; ++q)
625 {
626 // Calculate previous density (right now assumes constant
627 // density model per phase)
628 previous_density[p][q] = calculate_point_property(
629 filter->filter_phase(this->previous_phase_values[p][q]),
630 this->density_0[q],
631 this->density_1[q]);
632 }
633 }
634 break;
635 }
636 else if (gather_cahn_hilliard && !gather_vof)
637 {
638 // Blend the physical properties using the CahnHilliard field
639 for (unsigned int q = 0; q < this->n_q_points; ++q)
640 {
641 double phase_order_cahn_hilliard_value =
642 this->filtered_phase_order_cahn_hilliard_values[q];
643
644 density[q] = calculate_point_property_cahn_hilliard(
645 phase_order_cahn_hilliard_value,
646 this->density_0[q],
647 this->density_1[q]);
648
649 dynamic_viscosity[q] = calculate_point_property_cahn_hilliard(
650 phase_order_cahn_hilliard_value,
651 this->dynamic_viscosity_0[q],
652 this->dynamic_viscosity_1[q]);
653
654 dynamic_viscosity_for_stabilization[q] =
655 calculate_point_property_cahn_hilliard(
656 phase_order_cahn_hilliard_value,
657 this->dynamic_viscosity_for_stabilization_0[q],
658 this->dynamic_viscosity_for_stabilization_1[q]);
659 }
660 break;
661 }
662 break;
663 }
664 default:
665 throw std::runtime_error("Unsupported number of fluids (>2)");
666 }
667}

688template <int dim>
689double
690NavierStokesScratchData<dim>::extract_particle_properties()
691{
692 average_particle_velocity = 0;
693 // Loop over particles in cell
694 double total_particle_volume = 0;
695 unsigned int i_particle = 0;
696
697 for (auto &particle : pic)
698 {
699 auto particle_properties = particle.get_properties();
700 // Stores the values of particle velocity in a tensor
701 particle_velocity[i_particle][0] =
702 particle_properties[DEM::CFDDEMProperties::PropertiesIndex::v_x];
703 particle_velocity[i_particle][1] =
704 particle_properties[DEM::CFDDEMProperties::PropertiesIndex::v_y];
705 if constexpr (dim == 3)
706 particle_velocity[i_particle][2] =
707 particle_properties[DEM::CFDDEMProperties::PropertiesIndex::v_z];
708
709 if (!interpolated_void_fraction)
710 total_particle_volume +=
711 M_PI *
712 pow(particle_properties[DEM::CFDDEMProperties::PropertiesIndex::dp],
713 dim) /
714 (2 * dim);
715
716 average_particle_velocity += particle_velocity[i_particle];
717 i_particle++;
718 }
719 number_of_particles = i_particle;
720 if (number_of_particles != 0)
721 { // Calculate the average particle velocity within the cell
722 average_particle_velocity =
723 average_particle_velocity / number_of_particles;
724 }
725 return total_particle_volume;
726}

728template <int dim>
729void
730NavierStokesScratchData<dim>::calculate_cell_void_fraction(
731 const double &total_particle_volume)
732{
733 cell_volume = compute_cell_measure_with_JxW(this->fe_values.get_JxW_values());
734
735 if (!this->interpolated_void_fraction)
736 {
737 double cell_void_fraction_bulk = 0;
738 cell_void_fraction_bulk =
739 (cell_volume - total_particle_volume) / cell_volume;
740
741 for (unsigned int j = 0; j < number_of_particles; ++j)
742 cell_void_fraction[j] = cell_void_fraction_bulk;
743 }
744 else
745 for (unsigned int j = 0; j < number_of_particles; ++j)
746 cell_void_fraction[j] = 0;

749template <int dim>
750Quadrature<dim>
751NavierStokesScratchData<dim>::gather_particles_reference_location()
752{
753 // Create local vector that will be used to spawn an in-situ quadrature to
754 // interpolate at the locations of the particles
755 std::vector<Point<dim>> particle_reference_location(number_of_particles);
756 std::vector<double> particle_weights(number_of_particles, 1);
757 unsigned int i_particle = 0;
758
759 // Loop over particles in cell and cache their reference location
760 for (auto &particle : pic)
761 {
762 // Store particle positions and weights
763 // Reference location of the particle
764 particle_reference_location[i_particle] =
765 particle.get_reference_location();
766 i_particle++;
767 }
768
769 // Return a quadrature for the Navier-Stokes equations that is based on the
770 // particle reference location
771 return Quadrature<dim>(particle_reference_location, particle_weights);
