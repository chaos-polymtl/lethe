41 using namespace dealii;
42
43 template <int dim>
44 struct cell_comparison
45 {
46 inline bool
47 operator()(
48 const typename Triangulation<dim>::active_cell_iterator &cell_1,
49 const typename Triangulation<dim>::active_cell_iterator &cell_2) const
50 {
51 return cell_1->global_active_cell_index() <
52 cell_2->global_active_cell_index();
53 }
54 };
55
56 /**
57 * @brief Operator overloading to enable using particle iterators as map keys.
58 */
59
60 template <int dim>
61 struct particle_comparison
62 {
63 inline bool
64 operator()(const Particles::ParticleIterator<dim> &particle_1,
65 const Particles::ParticleIterator<dim> &particle_2)
66 {
67 return particle_1->id() < particle_2->id();
68 }
69 };
