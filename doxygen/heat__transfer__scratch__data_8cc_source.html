10template <int dim>
11void
12HeatTransferScratchData<dim>::allocate()
13{
14 // Initialize size of arrays
15 this->n_q_points = fe_values_T.get_quadrature().size();
16 this->n_dofs = fe_values_T.get_fe().n_dofs_per_cell();
17
18 // Initialize arrays related to quadrature
19 this->JxW = std::vector<double>(n_q_points);
20
21 // Forcing term array
22 this->source = std::vector<double>(n_q_points);
23
24 // Initialize arrays related to velocity and pressure
25 // Velocity
26 this->velocity_values = std::vector<Tensor<1, dim>>(n_q_points);
27 this->velocity_gradient_values = std::vector<Tensor<2, dim>>(n_q_points);
28 this->shear_rate_values = std::vector<double>(n_q_points);
29 // Pressure
30 this->pressure_values = std::vector<double>(n_q_points);
31
32 // Initialize arrays related to temperature
33 this->present_temperature_values = std::vector<double>(n_q_points);
34 this->temperature_gradients = std::vector<Tensor<1, dim>>(n_q_points);
35 this->present_temperature_laplacians = std::vector<double>(n_q_points);
36
37 // Initial arrays for physical properties
38 this->specific_heat = std::vector<double>(n_q_points);
39 this->grad_specific_heat_temperature = std::vector<double>(n_q_points);
40 this->thermal_conductivity = std::vector<double>(n_q_points);
41 this->density = std::vector<double>(n_q_points);
42 this->dynamic_viscosity = std::vector<double>(n_q_points);
43
44 // Velocity for BDF schemes
45 this->previous_temperature_values =
46 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
47 std::vector<double>(n_q_points));
48 this->previous_temperature_gradients =
49 std::vector<std::vector<Tensor<1, dim>>>(
50 maximum_number_of_previous_solutions(),
51 std::vector<Tensor<1, dim>>(n_q_points));
52
53 // Initialize arrays related to shape functions
54 // Velocity shape functions
55 this->phi_T =
56 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
57 this->grad_phi_T = std::vector<std::vector<Tensor<1, dim>>>(
58 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
59 this->hess_phi_T = std::vector<std::vector<Tensor<2, dim>>>(
60 n_q_points, std::vector<Tensor<2, dim>>(n_dofs));
61 this->laplacian_phi_T =
62 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
63
64 // Physical properties
65 fields.insert(
66 std::pair<field, std::vector<double>>(field::temperature, n_q_points));
67 fields.insert(
68 std::pair<field, std::vector<double>>(field::temperature_p1, n_q_points));
69 fields.insert(
70 std::pair<field, std::vector<double>>(field::temperature_p2, n_q_points));
71 fields.insert(
72 std::pair<field, std::vector<double>>(field::shear_rate, n_q_points));
73 fields.insert(
74 std::pair<field, std::vector<double>>(field::pressure, n_q_points));
75}

77template <int dim>
78void
79HeatTransferScratchData<dim>::enable_vof(
80 const FiniteElement<dim> &fe,
81 const Quadrature<dim> &quadrature,
82 const Mapping<dim> &mapping,
83 const Parameters::VOF_PhaseFilter &phase_filter_parameters)
84{
85 gather_vof = true;
86 fe_values_vof = std::make_shared<FEValues<dim>>(
87 mapping, fe, quadrature, update_values | update_gradients);
88
89 // Allocate VOF values
90 filtered_phase_values = std::vector<double>(this->n_q_points);
91 filtered_phase_gradient_values =
92 std::vector<Tensor<1, dim>>(this->n_q_points);
93
94 // Allocate physical properties
95 specific_heat_0 = std::vector<double>(n_q_points);
96 density_0 = std::vector<double>(n_q_points);
97 thermal_conductivity_0 = std::vector<double>(n_q_points);
98 dynamic_viscosity_0 = std::vector<double>(n_q_points);
99 grad_specific_heat_temperature_0 = std::vector<double>(n_q_points);
100
101 specific_heat_1 = std::vector<double>(n_q_points);
102 density_1 = std::vector<double>(n_q_points);
103 thermal_conductivity_1 = std::vector<double>(n_q_points);
104 dynamic_viscosity_1 = std::vector<double>(n_q_points);
105 grad_specific_heat_temperature_1 = std::vector<double>(n_q_points);
106
107 // Create filter
108 filter = VolumeOfFluidFilterBase::model_cast(phase_filter_parameters);
109}

145template <int dim>
146void
147HeatTransferScratchData<dim>::calculate_physical_properties()
148{
149 set_field_vector(field::temperature,
150 this->present_temperature_values,
151 this->fields);
152
153 if (properties_manager.field_is_required(field::temperature_p1))
154 set_field_vector(field::temperature_p1,
155 this->previous_temperature_values[0],
156 this->fields);
157
158 if (properties_manager.field_is_required(field::temperature_p2))
159 set_field_vector(field::temperature_p2,
160 this->previous_temperature_values[1],
161 this->fields);
162
163 if (properties_manager.field_is_required(field::shear_rate))
164 {
165 for (unsigned int q = 0; q < n_q_points; ++q)
166 {
167 // Calculate shear rate (at each q)
168 const Tensor<2, dim> shear_rate_tensor =
169 velocity_gradient_values[q] +
170 transpose(velocity_gradient_values[q]);
171
172 // Calculate the shear rate magnitude
173 shear_rate_values[q] =
174 calculate_shear_rate_magnitude(shear_rate_tensor);
175 }
176
177 set_field_vector(field::shear_rate, shear_rate_values, this->fields);
178 }
179
180 if (properties_manager.field_is_required(field::pressure))
181 {
182 set_field_vector(field::pressure, this->pressure_values, this->fields);
183 }
184
185 if (material_id < 1 || properties_manager.get_number_of_solids() < 1)
186 {
187 switch (properties_manager.get_number_of_fluids())
188 {
189 case 1:
190 {
191 const auto density_model = properties_manager.get_density();
192 const auto specific_heat_model =
193 properties_manager.get_specific_heat();
194 const auto thermal_conductivity_model =
195 properties_manager.get_thermal_conductivity();
196 const auto rheology_model = properties_manager.get_rheology();
197
198 density_model->vector_value(fields, density);
199 specific_heat_model->vector_value(fields, specific_heat);
200 specific_heat_model->vector_jacobian(
201 fields, field::temperature, grad_specific_heat_temperature);
202 thermal_conductivity_model->vector_value(fields,
203 thermal_conductivity);
204 rheology_model->get_dynamic_viscosity_vector(
205 density_model->get_density_ref(), fields, dynamic_viscosity);
206
207 break;
208 }
209 case 2:
210 {
211 const auto density_models =
212 properties_manager.get_density_vector();
213 const auto specific_heat_models =
214 properties_manager.get_specific_heat_vector();
215 const auto thermal_conductivity_models =
216 properties_manager.get_thermal_conductivity_vector();
217 const auto rheology_models =
218 properties_manager.get_rheology_vector();
219
220 density_models[0]->vector_value(fields, density_0);
221 specific_heat_models[0]->vector_value(fields, specific_heat_0);
222 thermal_conductivity_models[0]->vector_value(
223 fields, thermal_conductivity_0);
224 rheology_models[0]->get_dynamic_viscosity_vector(
225 density_models[0]->get_density_ref(),
226 fields,
227 dynamic_viscosity_0);
228 specific_heat_models[0]->vector_jacobian(
229 fields, field::temperature, grad_specific_heat_temperature_0);
230
231 density_models[1]->vector_value(fields, density_1);
232 specific_heat_models[1]->vector_value(fields, specific_heat_1);
233 thermal_conductivity_models[1]->vector_value(
234 fields, thermal_conductivity_1);
235 rheology_models[1]->get_dynamic_viscosity_vector(
236 density_models[1]->get_density_ref(),
237 fields,
238 dynamic_viscosity_1);
239 specific_heat_models[1]->vector_jacobian(
240 fields, field::temperature, grad_specific_heat_temperature_1);
241
242 // Blend the physical properties using the VOF field
243 for (unsigned int q = 0; q < this->n_q_points; ++q)
244 {
245 auto filtered_phase_value = this->filtered_phase_values[q];
246
247 density[q] = calculate_point_property(filtered_phase_value,
248 this->density_0[q],
249 this->density_1[q]);
250
251 specific_heat[q] =
252 calculate_point_property(filtered_phase_value,
253 this->specific_heat_0[q],
254 this->specific_heat_1[q]);
255
256 thermal_conductivity[q] =
257 calculate_point_property(filtered_phase_value,
258 this->thermal_conductivity_0[q],
259 this->thermal_conductivity_1[q]);
260
261 dynamic_viscosity[q] =
262 calculate_point_property(filtered_phase_value,
263 this->dynamic_viscosity_0[q],
264 this->dynamic_viscosity_1[q]);
265
266 grad_specific_heat_temperature[q] = calculate_point_property(
267 filtered_phase_value,
268 this->grad_specific_heat_temperature_0[q],
269 this->grad_specific_heat_temperature_1[q]);
270 }
271 }
272 break;
273 default:
274 throw std::runtime_error("Unsupported number of fluids (>2)");
275 }
276 }
277 else
278 {
279 const auto density_model = properties_manager.get_density(0, material_id);
280 const auto specific_heat_model =
281 properties_manager.get_specific_heat(0, material_id);
282 const auto thermal_conductivity_model =
283 properties_manager.get_thermal_conductivity(0, material_id);
284 const auto rheology_model =
285 properties_manager.get_rheology(0, material_id);
286
287 density_model->vector_value(fields, density);
288 specific_heat_model->vector_value(fields, specific_heat);
289 specific_heat_model->vector_jacobian(fields,
290 field::temperature,
291 grad_specific_heat_temperature);
292 thermal_conductivity_model->vector_value(fields, thermal_conductivity);
293 rheology_model->get_dynamic_viscosity_vector(
294 density_model->get_density_ref(), fields, dynamic_viscosity);
295 }
296}
