17template <int dim>
18void
19AverageScalar<dim>::calculate_average_scalar(
20 const GlobalVectorType &local_evaluation_point,
21 const Parameters::PostProcessing &post_processing,
22 const double current_time,
23 const double time_step)
24{
25 const double epsilon = 1e-6;
26 const double initial_time =
27 post_processing.initial_time_for_average_temp_and_hf;
28 dt = time_step;
29
30 // When averaging scalar begins
31 if (current_time >= (initial_time - epsilon))
32 {
33 if (!has_started_averaging)
34 {
35 real_initial_time = current_time;
36
37 // Store the first dt value in case dt varies.
38 dt_0 = dt;
39
40 has_started_averaging = true;
41 }
42
43 // Calculate (scalar*dt) at each time step and accumulates the values
44 scalar_dt.equ(dt, local_evaluation_point);
45
46 sum_scalar_dt += scalar_dt;
47
48 // Get the inverse of the time since the beginning of the time averaging
49 total_time_for_average = (current_time - real_initial_time) + dt_0;
50 inv_range_time = 1. / total_time_for_average;
51
52 // Calculate the average scalars.
53 average_scalar.equ(inv_range_time, sum_scalar_dt);
54 }
55}

77template <int dim>
78void
79AverageScalar<dim>::update_average_scalar()
80{
81 if (total_time_for_average > 1e-16)
82 {
83 inv_range_time = 1.0 / total_time_for_average;
84 // Calculate the average scalar.
85 average_scalar.equ(inv_range_time, sum_scalar_dt);

109template <int dim>
110std::vector<const GlobalVectorType *>
111AverageScalar<dim>::save(const std::string &prefix)
112{
113 sum_scalar_dt_with_ghost_cells = sum_scalar_dt;
114
115 std::vector<const GlobalVectorType *> avg_scalar_set_transfer;
116 avg_scalar_set_transfer.push_back(&sum_scalar_dt_with_ghost_cells);
117
118 std::string filename = prefix + ".averagescalar";
119 std::ofstream output(filename.c_str());
120 output << "Average scalar" << std::endl;
121 output << "dt_0 " << dt_0 << std::endl;
122 output << "has_started_averaging_boolean " << has_started_averaging
123 << std::endl;
124 output << "Real_initial_time " << real_initial_time << std::endl;
125
126 return avg_scalar_set_transfer;
127}

129template <int dim>
130std::vector<GlobalVectorType *>
131AverageScalar<dim>::read(const std::string &prefix)
132{
133 std::vector<GlobalVectorType *> sum_vectors;
134 sum_vectors.push_back(&sum_scalar_dt_with_ghost_cells);
135
136 std::string filename = prefix + ".averagescalar";
137 std::ifstream input(filename.c_str());
138 AssertThrow(input, ExcFileNotOpen(filename));
139
140 std::string buffer;
141 std::getline(input, buffer);
142 input >> buffer >> dt_0;
143 input >> buffer >> has_started_averaging;
144 input >> buffer >> real_initial_time;
145
146 return sum_vectors;
