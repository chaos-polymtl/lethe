17 template <int dim>
18 void
19 AverageScalar<dim>::calculate_average_scalar(
20  const GlobalVectorType &local_evaluation_point,
21  const Parameters::PostProcessing &post_processing,
22  const double current_time,
23  const double time_step)
24 {
25  const double epsilon = 1e-6;
26  const double initial_time =
27  post_processing.initial_time_for_average_temp_and_hf;
28  dt = time_step;
29 
30  // When averaging scalar begins
31  if (current_time >= (initial_time - epsilon) && !average_calculation)
32  {
33  average_calculation = true;
34  real_initial_time = current_time;
35 
36  // Store the first dt value in case dt varies.
37  dt_0 = dt;
38  }
39 
40  // Calculate (scalar*dt) at each time step and accumulates the values
41  scalar_dt.equ(dt, local_evaluation_point);
42 
43  sum_scalar_dt += scalar_dt;
44 
45  // Get the inverse of the time since the beginning of the time averaging
46  total_time_for_average = (current_time - real_initial_time) + dt_0;
47  inv_range_time = 1. / total_time_for_average;
48 
49  // Calculate the average scalars.
50  average_scalar.equ(inv_range_time, sum_scalar_dt);
51 }
