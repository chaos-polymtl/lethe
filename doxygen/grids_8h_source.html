103/**
104 * @brief Refine a mesh around specific boundary ids
105 *
106 * @param[in] n_refinement The number of times the boundary should be refined
107 *
108 * @param[in] boundary_ids The boundary ids for which the cells should be
109 * refined
110 *
111 * @param[in, out] triangulation The triangulation on which refinement must be
112 * carried out
113 *
114 */
115template <int dim, int spacedim = dim>
116void
117refine_triangulation_at_boundaries(
118 const std::vector<int> boundary_ids,
119 const unsigned int n_refinement,
120 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation)
121{
122 // For the amount of refinements required
123 for (unsigned int r = 0; r < n_refinement; ++r)
124 {
125 // Loop over the cells and flag all cells which are within the list of
126 // boundary ids
127 for (const auto &cell : triangulation.active_cell_iterators())
128 {
129 if (cell->is_locally_owned())
130 {
131 for (const auto f : cell->face_indices())
132
133 {
134 if (cell->face(f)->at_boundary())
135 {
136 if (std::find(boundary_ids.begin(),
137 boundary_ids.end(),
138 cell->face(f)->boundary_id()) !=
139 boundary_ids.end())
140 {
141 cell->set_refine_flag();
142 }
143 }
144 }
145 }
146 }
147 triangulation.execute_coarsening_and_refinement();
148 }
149}
