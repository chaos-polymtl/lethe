23 /**
24 * @brief Implements analytical solutions for the Navier-Stokes equations
25 * and the other physics supported by lethe
26 *
27 * @tparam dim An integer that denotes the dimension of the space in which
28 * the flow is solved.
29
30 * The analytical solution class provides an all common
31 * element required for the calculation of analytical solution for all the
32 * physics supported by Lethe.
33 **/
34 template <int dim>
35 class AnalyticalSolution
36 {
37 private:
38 /**
39 * Establishes if the calculation of the analytical solution is enabled or
40 * not
41 */
42 bool enable;
43
44 /**
45 * Filename used to store the L2 norm of the error
46 */
47 std::string filename;
48
49
50 /**
51 * @brief Construct a new AnalyticalSolution object.
52 *
53 * The constructor automatically sets the number of components
54 * of each equation manually since all equations currently supported have
55 * a set number of equations.
56 */
57 public:
58 AnalyticalSolution()
59 : enable(false)
60 , uvwp(dim + 1)
61 , temperature(1)
62 , tracer(1)
63 , cahn_hilliard(2)
64 {}
65
66 /**
67 * @brief Declares the parameters required by the analytical solution
68 * within a parameter file
69 *
70 * @param prm ParameterHandler used to declare the parameters.
71 *
72 */
73 virtual void
74 declare_parameters(ParameterHandler &prm);
75
76 /**
77 * @brief Parses the parameters required by the analytical solution
78 * within a parameter file
79 *
80 * @param prm ParameterHandler used to parse the parameters.
81 *
82 */
83 virtual void
84 parse_parameters(ParameterHandler &prm);
85
86 /**
87 * @brief Checks if the calculation of the error using the analytical
88 * solution is enabled. This option is there to prevent unnecessary
89 * calculations when no analytical solutions are required.
90 */
91 bool
92 calculate_error()
93 {
94 return enable;
95 }
96
97
98 /**
99 * @brief Enables the usage of the analytical solution
100 */
101 void
102 set_enable(bool is_enable)
103 {
104 enable = is_enable;
105 }
106
107 /**
108 * @brief Get the file name associated with the output of the
109 * L2 norm of the error calculated using the analytical solution.
110 *
111 * @return filename A string that contains the filename used to output
112 * the L2 norm of the error.
113 */
114 std::string
115 get_filename()
116 {
117 return filename;
118 }
119
120
121 /**
122 * Controls if the L2 norm of the error is printed to the terminal
123 * during the simulation.
124 */
125 Parameters::Verbosity verbosity;
126
127 /**
128 * ParsedFunction that contains the analytical solution for the velocity
129 * components and the pressure.
130 */
131 Functions::ParsedFunction<dim> uvwp;
132
133 /**
134 * ParsedFunction that contains the analytical solution for the temperature.
135 */
136 Functions::ParsedFunction<dim> temperature;
137
138 /**
139 * ParsedFunction that contains the analytical solution for the tracer.
140 */
141 Functions::ParsedFunction<dim> tracer;
142
143 /**
144 * ParsedFunction that contains the analytical solution for the phase.
145 */
146 Functions::ParsedFunction<dim> phase;
147
148 /**
149 * ParsedFunction that contains the analytical solution for the phase order
150 * and the chemical potential.
151 */
152 Functions::ParsedFunction<dim> cahn_hilliard;
153 };
