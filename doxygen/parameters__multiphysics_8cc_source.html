194void
195Parameters::VOF_RegularizationMethod::parse_parameters(ParameterHandler &prm)
196{
197 prm.enter_subsection("interface regularization method");
198 {
199 const std::string t = prm.get("type");
200 if (t == "none")
201 this->regularization_method_type =
202 Parameters::RegularizationMethodType::none;
203 else if (t == "projection-based interface sharpening")
204 {
205 regularization_method_type =
206 Parameters::RegularizationMethodType::sharpening;
207 sharpening.enable = true;
208 }
209 else if (t == "algebraic interface reinitialization")
210 {
211 this->regularization_method_type =
212 Parameters::RegularizationMethodType::algebraic;
213 algebraic_interface_reinitialization.enable = true;
214 }
215 else if (t == "geometric interface reinitialization")
216 {
217 this->regularization_method_type =
218 Parameters::RegularizationMethodType::geometric;
219 geometric_interface_reinitialization.enable = true;
220 }
221 else
222 throw(
223 std::runtime_error("Invalid interface regularization method type!"));
224
225 this->frequency = prm.get_integer("frequency");
226 Assert(this->frequency > 0,
227 RegularizationMethodFrequencyError(this->frequency));
228
229 const std::string op2 = prm.get("verbosity");
230 if (op2 == "quiet")
231 this->verbosity = Parameters::Verbosity::quiet;
232 else if (op2 == "verbose")
233 this->verbosity = Parameters::Verbosity::verbose;
234 else if (op2 == "extra verbose")
235 this->verbosity = Parameters::Verbosity::extra_verbose;
236 else
237 throw(std::invalid_argument("Invalid verbosity level\n "
238 "Options are: \n"
239 " <quiet|verbose|extra verbose>"));
240
241 this->sharpening.parse_parameters(prm);
242 this->algebraic_interface_reinitialization.parse_parameters(prm);
243 this->geometric_interface_reinitialization.parse_parameters(prm);
244 }
245 prm.leave_subsection();
246}

315Parameters::VOF_InterfaceSharpening::parse_parameters(ParameterHandler &prm)
316{
317 prm.enter_subsection("projection-based interface sharpening");
318 {
319 interface_sharpness = prm.get_double("interface sharpness");
320
321 // Sharpening type
322 const std::string t = prm.get("type");
323 if (t == "constant")
324 type = Parameters::SharpeningType::constant;
325 if (t == "adaptive")
326 type = Parameters::SharpeningType::adaptive;
327
328 // Parameters for constant sharpening
329 threshold = prm.get_double("threshold");
330
331 // Parameters for adaptive sharpening
332 threshold_max_deviation = prm.get_double("threshold max deviation");
333 max_iterations = prm.get_integer("max iterations");
334 monitoring = prm.get_bool("monitoring");
335 tolerance = prm.get_double("tolerance");
336
337 // Monitored fluid
338 const std::string op_mf = prm.get("monitored fluid");
339 if (op_mf == "fluid 1")
340 monitored_fluid = Parameters::FluidIndicator::fluid1;
341 else if (op_mf == "fluid 0")
342 monitored_fluid = Parameters::FluidIndicator::fluid0;
343 else
344 throw(std::runtime_error("Invalid monitored fluid. "
345 "Options are 'fluid 0' or 'fluid 1'."));
346
347 // Error definitions
348 Assert(threshold > 0.0 && threshold < 1.0,
349 SharpeningThresholdError(threshold));
350
351 Assert(threshold_max_deviation > 0.0 && threshold_max_deviation < 0.5,
352 SharpeningThresholdErrorMaxDeviation(threshold_max_deviation));
353 }
354 prm.leave_subsection();
