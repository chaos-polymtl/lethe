215void
216Parameters::CLS_ReinitializationMethod::parse_parameters(ParameterHandler &prm)
217{
218 prm.enter_subsection("interface reinitialization method");
219 {
220 const std::string t = prm.get("type");
221 if (t == "none")
222 this->reinitialization_method_type =
223 Parameters::ReinitializationMethodType::none;
224 else if (t == "projection-based interface sharpening")
225 {
226 reinitialization_method_type =
227 Parameters::ReinitializationMethodType::sharpening;
228 sharpening.enable = true;
229 }
230 else if (t == "pde-based interface reinitialization")
231 {
232 this->reinitialization_method_type =
233 Parameters::ReinitializationMethodType::pde_based;
234 pde_based_interface_reinitialization.enable = true;
235 }
236 else if (t == "geometric interface reinitialization")
237 {
238 this->reinitialization_method_type =
239 Parameters::ReinitializationMethodType::geometric;
240 geometric_interface_reinitialization.enable = true;
241 }
242 else
243 throw(
244 std::runtime_error("Invalid interface reinitialization method type!"));
245
246 this->frequency = prm.get_integer("frequency");
247 Assert(this->frequency > 0,
248 ReinitializationMethodFrequencyError(this->frequency));
249
250 const std::string op2 = prm.get("verbosity");
251 if (op2 == "quiet")
252 this->verbosity = Parameters::Verbosity::quiet;
253 else if (op2 == "verbose")
254 this->verbosity = Parameters::Verbosity::verbose;
255 else if (op2 == "extra verbose")
256 this->verbosity = Parameters::Verbosity::extra_verbose;
257 else
258 throw(std::invalid_argument("Invalid verbosity level\n "
259 "Options are: \n"
260 " <quiet|verbose|extra verbose>"));
261
262 this->sharpening.parse_parameters(prm);
263 this->pde_based_interface_reinitialization.parse_parameters(prm);
264 this->geometric_interface_reinitialization.parse_parameters(prm);
265 }
266 prm.leave_subsection();
267}

335void
336Parameters::CLS_InterfaceSharpening::parse_parameters(ParameterHandler &prm)
337{
338 prm.enter_subsection("projection-based interface sharpening");
339 {
340 interface_sharpness = prm.get_double("interface sharpness");
341
342 // Sharpening type
343 const std::string t = prm.get("type");
344 if (t == "constant")
345 type = Parameters::SharpeningType::constant;
346 if (t == "adaptive")
347 type = Parameters::SharpeningType::adaptive;
348
349 // Parameters for constant sharpening
350 threshold = prm.get_double("threshold");
351
352 // Parameters for adaptive sharpening
353 threshold_max_deviation = prm.get_double("threshold max deviation");
354 max_iterations = prm.get_integer("max iterations");
355 monitoring = prm.get_bool("monitoring");
356 tolerance = prm.get_double("tolerance");
357
358 // Monitored fluid
359 const std::string op_mf = prm.get("monitored fluid");
360 if (op_mf == "fluid 1")
361 monitored_fluid = Parameters::FluidIndicator::fluid1;
362 else if (op_mf == "fluid 0")
363 monitored_fluid = Parameters::FluidIndicator::fluid0;
364 else
365 throw(std::runtime_error("Invalid monitored fluid. "
366 "Options are 'fluid 0' or 'fluid 1'."));
367
368 // Error definitions
369 Assert(threshold > 0.0 && threshold < 1.0,
370 SharpeningThresholdError(threshold));
371
372 Assert(threshold_max_deviation > 0.0 && threshold_max_deviation < 0.5,
373 SharpeningThresholdErrorMaxDeviation(threshold_max_deviation));
374 }
375 prm.leave_subsection();
376}

737Parameters::CahnHilliard::parse_parameters(ParameterHandler &prm,
738 const Dimensionality &dimensions)
739{
740 prm.enter_subsection("cahn hilliard");
741 {
742 cahn_hilliard_phase_filter.parse_parameters(prm);
743
744 CahnHilliard::potential_smoothing_coefficient =
745 prm.get_double("potential smoothing coefficient");
746
747 prm.enter_subsection("epsilon");
748 {
749 const std::string op_epsilon = prm.get("method");
750 if (op_epsilon == "automatic")
751 {
752 CahnHilliard::epsilon_set_method =
753 Parameters::EpsilonSetMethod::automatic;
754 }

930template <int dim>
931void
932Parameters::TimeHarmonicMaxwell<dim>::parse_parameters(
933 ParameterHandler &prm,
934 const Dimensionality &dimensions)
935{
936 prm.enter_subsection("time harmonic maxwell");
937 {
938 prm.enter_subsection("time coupling strategy");
939 {
940 const std::string op_coupling_type = prm.get("type");
941 if (op_coupling_type == "none")
942 time_coupling_strategy =
943 Parameters::TimeHarmonicMaxwellCouplingStrategy::none;
944 else if (op_coupling_type == "iteration")
945 time_coupling_strategy =
946 Parameters::TimeHarmonicMaxwellCouplingStrategy::iteration;
947 else if (op_coupling_type == "time")
948 time_coupling_strategy =
949 Parameters::TimeHarmonicMaxwellCouplingStrategy::time;
950 else if (op_coupling_type == "threshold")
951 time_coupling_strategy =
952 Parameters::TimeHarmonicMaxwellCouplingStrategy::threshold;
953 else
954 AssertThrow(false,
955 ExcMessage("Invalid time coupling strategy type. Options "
956 "are <none|iteration|time|threshold>."));
957
958 TimeHarmonicMaxwell::coupling_iteration =
959 prm.get_integer("coupling iteration");
960 TimeHarmonicMaxwell::coupling_time = prm.get_double("coupling time");
961 TimeHarmonicMaxwell::coupling_threshold =
962 prm.get_double("coupling threshold");
963 }
964 prm.leave_subsection();
965
966 TimeHarmonicMaxwell::electromagnetic_frequency =
967 prm.get_double("electromagnetic frequency") *
968 dimensions.electromagnetic_frequency_scaling;
969
970 const std::string op_scaling_type = prm.get("electromagnetic scaling type");
971 if (op_scaling_type == "none")
972 TimeHarmonicMaxwell::electromagnetic_scaling_type =
973 Parameters::ElectromagneticScalingType::none;
974 else if (op_scaling_type == "electric field")
975 TimeHarmonicMaxwell::electromagnetic_scaling_type =
976 Parameters::ElectromagneticScalingType::electric_field;
977 else if (op_scaling_type == "magnetic field")
978 TimeHarmonicMaxwell::electromagnetic_scaling_type =
979 Parameters::ElectromagneticScalingType::magnetic_field;
980 else if (op_scaling_type == "power")
981 TimeHarmonicMaxwell::electromagnetic_scaling_type =
982 Parameters::ElectromagneticScalingType::power;
983 else
984 AssertThrow(false,
985 ExcMessage(
986 "Invalid electromagnetic scaling type. "
987 "Options are <none|electric field|magnetic field|power>."));
988
989 // By default, the electric field dimensionality is in V/m, but if the user
990 // changed the dimensionality of the problem, we need to change the
991 // dimensionality of the electric field accordingly to ensure that the
992 // correct physical solution is obtained in dimensional units. This needs to
993 // be apply to the power later on as well if the user chooses power-based
994 // scaling which we cannot make dimensionless yet to recover the amplitude
995 // scaling.
996 TimeHarmonicMaxwell::electric_field_dimensionality =
997 dimensions.electric_amplitude_scaling;
998 // The same applies for the magnetic field, which is in A/m by default.
999 TimeHarmonicMaxwell::magnetic_field_dimensionality =
1000 dimensions.magnetic_amplitude_scaling;
1001
1002 // If the choose other scaling, we can already apply the dimensionality to
1003 // the amplitude provided.
1004 TimeHarmonicMaxwell::electric_field_amplitude =
1005 prm.get_double("electric field amplitude") *
1006 TimeHarmonicMaxwell::electric_field_dimensionality;
1007
1008 TimeHarmonicMaxwell::magnetic_field_amplitude =
1009 prm.get_double("magnetic field amplitude") *
1010 TimeHarmonicMaxwell::magnetic_field_dimensionality;
1011
1012 TimeHarmonicMaxwell::number_of_waveguide_inlets =
1013 prm.get_integer("number of waveguide inlets");
1014
1015 // Ensure that the number of waveguide inlets is smaller than the maximum
1016 // declared in declare_parameters.
1017 AssertThrow(
1018 TimeHarmonicMaxwell::number_of_waveguide_inlets <= 10,
1019 ExcMessage(
1020 "The number of waveguide inlets specified exceeds "
1021 "the maximum allowed. Please increase the value "
1022 "of max_waveguide_inlets in the code declare_parameters section of the TimeHarmonicMaxwell class."));
1023
1024 // Resize vectors to hold the correct number of inlets
1025 TimeHarmonicMaxwell::waveguide_mode.resize(number_of_waveguide_inlets);
1026 TimeHarmonicMaxwell::mode_order_m.resize(number_of_waveguide_inlets);
1027 TimeHarmonicMaxwell::mode_order_n.resize(number_of_waveguide_inlets);
1028 TimeHarmonicMaxwell::waveguide_boundary_ids.resize(
1029 number_of_waveguide_inlets);
1030 TimeHarmonicMaxwell::waveguide_power.resize(number_of_waveguide_inlets);
1031
1032 for (unsigned int inlet = 0; inlet < number_of_waveguide_inlets; ++inlet)
1033 {
1034 prm.enter_subsection("waveguide inlet " + std::to_string(inlet));
1035 {
1036 TimeHarmonicMaxwell::waveguide_boundary_ids[inlet] =
1037 prm.get_integer("port boundary id");
1038
1039 TimeHarmonicMaxwell::waveguide_power[inlet] =
1040 prm.get_double("waveguide power");
1041
1042 // Check that the waveguide power is not zero which would imply no
1043 // excitation at the inlet.
1044 AssertThrow(
1045 TimeHarmonicMaxwell::waveguide_power[inlet] > 0,
1046 ExcMessage(
1047 "The waveguide power for the inlet " + std::to_string(inlet) +
1048 " is zero. Please check the waveguide power parameter in the input prm file for this inlet. If you really want to have no excitation but an open boundary condition at this inlet, you should change the boundary condition type to the impedance boundary condition with zero excitation and the desired admittance."));
1049
1050 prm.enter_subsection("waveguide mode");
1051 {
1052 const std::string op_mode_type = prm.get("mode type");
1053 if (op_mode_type == "TE")
1054 {
1055 TimeHarmonicMaxwell::waveguide_mode[inlet] =
1056 Parameters::WaveguideMode::TE;
1057 }
1058 else if (op_mode_type == "TM")
1059 {
1060 TimeHarmonicMaxwell::waveguide_mode[inlet] =
1061 Parameters::WaveguideMode::TM;
1062 }
1063 else
1064 throw(std::runtime_error("Invalid waveguide mode type. "
1065 "Options are 'TE' or 'TM'."));
1066
1067 TimeHarmonicMaxwell::mode_order_m[inlet] =
1068 prm.get_integer("mode order m");
1069
1070 TimeHarmonicMaxwell::mode_order_n[inlet] =
1071 prm.get_integer("mode order n");
1072
1073 // Check that the choice of mode orders is valid
1074 if (TimeHarmonicMaxwell::mode_order_m[inlet] == 0 &&
1075 TimeHarmonicMaxwell::mode_order_n[inlet] == 0)
1076 {
1077 throw(std::runtime_error(
1078 "Invalid waveguide mode orders m and n. "
1079 "At least one of the two mode orders must be non-zero."));
1080 }
1081 }
1082 prm.leave_subsection();
1083
1084 const unsigned int num_corners = Utilities::fixed_power<2>(dim - 1);
1085 std::array<Tensor<1, dim>, num_corners> tmp_corners;
1086
1087 for (unsigned int corner = 0; corner < num_corners; ++corner)
1088 {
1089 const std::string corner_name =
1090 "corner " + std::to_string(corner);
1091 const Tensor<1, dim> corner_value =
1092 value_string_to_tensor<dim>(prm.get(corner_name));
1093
1094 tmp_corners[corner] = corner_value;
1095 }
1096
1097 // In 3D, verify that the provided corners define a coplanar
1098 // quadrilateral. This is not useful in 2D as any 2 points are always
1099 // collinear.
1100 if constexpr (dim == 3)
1101 {
1102 const Tensor<1, dim> vec1 =
1103 tmp_corners[1] -
1104 tmp_corners[0]; // Vector from corner 1 to corner 2
1105 const Tensor<1, dim> vec2 =
1106 tmp_corners[2] -
1107 tmp_corners[0]; // Vector from corner 1 to corner 3
1108 const Tensor<1, dim> vec3 =
1109 tmp_corners[3] -
1110 tmp_corners[0]; // Vector from corner 1 to corner 4
1111
1112 // The triple scalar product (determinant) scales as the volume of
1113 // the parallelepiped defined by the three vectors. So we make the
1114 // it dimensionless by a characteristic volume (product of the
1115 // norms of the three vectors). This way we can set a tolerance
1116 // that is independent of the actual size of the waveguide.
1117 const double determinant =
1118 scalar_product(vec3, cross_product_3d(vec1, vec2)) /
1119 (vec1.norm() * vec2.norm() * vec3.norm());
1120
1121 const double tolerance = 1e-10;
1122
1123 AssertThrow(std::abs(determinant) < tolerance,
1124 ExcMessage(
1125 "The provided corners for waveguide inlet " +
1126 std::to_string(inlet) +
1127 " do not define a coplanar quadrilateral. "
1128 "Please check the corner coordinates."));
1129 }
1130
1131 TimeHarmonicMaxwell::waveguide_corners.push_back(tmp_corners);
1132 }
1133 prm.leave_subsection();
1134 }
1135
1136 // Check that there is at least one waveguide inlet if the user use the
1137 // `power` electromagnetic scaling type.
1138 AssertThrow(
1139 !((TimeHarmonicMaxwell::electromagnetic_scaling_type ==
1140 Parameters::ElectromagneticScalingType::power) &&
1141 TimeHarmonicMaxwell::number_of_waveguide_inlets == 0),
1142 ExcMessage(
1143 "The power-based electromagnetic scaling type requires at least one waveguide inlet to be defined. Please check the number of waveguide inlets specified in the input prm file."));
1144 }
1145 prm.leave_subsection();
1146}
