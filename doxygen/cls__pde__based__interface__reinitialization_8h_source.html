38 /**
39 * @brief Constructor of the CLS PDE-based interface reinitialization.
40 *
41 * @param[in] p_simulation_parameters Simulation parameters.
42 *
43 * @param[in] p_pcout Parallel cout used to print the information.
44 *
45 * @param[in] p_triangulation Distributed mesh information.
46 *
47 * @param[in] p_simulation_control SimulationControl object.
48 *
49 * @param[in, out] p_subequations_interface Subequations interface object used
50 * to get information from other subequations and store information from the
51 * current one.
52 */
53 CLSPDEBasedInterfaceReinitialization(
54 const SimulationParameters<dim> &p_simulation_parameters,
55 const ConditionalOStream &p_pcout,
56 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
57 p_triangulation,
58 const std::shared_ptr<SimulationControl> &p_simulation_control,
59 CLSSubequationsInterface<dim> &p_subequations_interface)
60 : PhysicsNonlinearSubequationsSolver<dim, GlobalVectorType>(
61 p_simulation_parameters.cls_subequations_non_linear_solvers.at(
62 CLSSubequationsID::pde_based_interface_reinitialization),
63 p_pcout)
64 , subequation_id(CLSSubequationsID::pde_based_interface_reinitialization)
65 , subequations_interface(p_subequations_interface)
66 , simulation_parameters(p_simulation_parameters)
67 , simulation_control(p_simulation_control)
68 , triangulation(p_triangulation)
69 , dof_handler(std::make_shared<DoFHandler<dim>>(*this->triangulation))
70 , subequation_verbosity(p_simulation_parameters.multiphysics.cls_parameters
71 .reinitialization_method.verbosity)
72 {
73 this->pcout << std::setprecision(simulation_control->get_log_precision())
74 << std::scientific;
75
76 if (this->simulation_parameters.mesh.simplex)
77 {
78 // For simplex meshes
79 this->fe = std::make_shared<FE_SimplexP<dim>>(
80 this->simulation_parameters.fem_parameters.CLS_order);
81 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
82 this->cell_quadrature =
83 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
84 }
85 else
86 {
87 // Usual case, for quad/hex meshes
88 this->fe = std::make_shared<FE_Q<dim>>(
89 this->simulation_parameters.fem_parameters.CLS_order);
90 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
91 this->cell_quadrature =
92 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
93 }
94
95 // Ensure that the shared pointer is properly allocated
96 this->present_solution = std::make_shared<GlobalVectorType>();
97 }

236 /**
237 * @brief Computes mesh-dependant diffusivity coefficient value.
238 *
239 * @param[in] min_cell_size Smallest cell's measure.
240 *
241 * @return Constant diffusivity coefficient value for a given mesh.
242 */
243 inline double
244 compute_diffusivity(const double min_cell_size) const
245 {
246 const double multiplier =
247 this->simulation_parameters.multiphysics.cls_parameters
248 .reinitialization_method.pde_based_interface_reinitialization
249 .diffusivity_multiplier;
250 const double power =
251 this->simulation_parameters.multiphysics.cls_parameters
252 .reinitialization_method.pde_based_interface_reinitialization
253 .diffusivity_power;
254 return multiplier * std::pow(min_cell_size, power);
255 }
256
257 /**
258 * @brief Computes PDE-based reinitialization time step value with
259 * user-imposed CFL and minimum cell size.
260 *
261 * @return PDE-based reinitialization time step.
262 */
263 inline double
264 compute_time_step()
265 {
266 // Get artificial time-step factor
267 const double dtau_factor =
268 this->simulation_parameters.multiphysics.cls_parameters
269 .reinitialization_method.pde_based_interface_reinitialization
270 .dtau_factor;
271
272 // Get the minimum cell size
273 const double h_min =
274 identify_minimum_cell_size(*this->mapping,
275 *this->dof_handler,
276 *this->cell_quadrature,
277 this->triangulation->get_mpi_communicator());
278
279 // Compute diffusivity
280 const double diffusivity_inv = 1.0 / compute_diffusivity(h_min);
281
282 // Compute artificial time step
283 const double dtau = h_min * h_min * diffusivity_inv * dtau_factor;
284
285 // Use the smallest of the computed artificial time step and the simulation
286 // time step for the reinitialization scheme
287 return std::min(dtau, this->simulation_control->get_time_step());
288 }

326 /**
327 * @brief Indicate if the PDE-based reinitialization should continue by
328 * checking if at least one of the two stop criteria (steady-state criterion
329 * or maximum number of reinitialization steps) is met.
330 *
331 * @param[in] step_number PDE-based interface reinitialization step number.
332 *
333 * @return Boolean indicating if the PDE-based reinitialization should
334 * continue
335 */
336 inline bool
337 continue_iterating(const unsigned int step_number)
338 {
339 if (step_number == 1) // Initial condition
340 {
341 return (step_number <
342 this->simulation_parameters.multiphysics.cls_parameters
343 .reinitialization_method
344 .pde_based_interface_reinitialization.max_steps_number +
345 1);
346 }
347 else
348 {
349 // Get the stop criterion of the artificial time-stepping scheme
350 double steady_state_criterion =
351 this->simulation_parameters.multiphysics.cls_parameters
352 .reinitialization_method.pde_based_interface_reinitialization
353 .steady_state_criterion;
354
355 // Evaluate the solution difference between the 2 last solutions
356 auto solution_diff = local_evaluation_point;
357 solution_diff -= previous_local_evaluation_point;
358
359 // Evaluate the current steady-state criterion value
360 double stop_criterion =
361 solution_diff.l2_norm() /
362 (previous_local_evaluation_point.l2_norm() + 1e-14);
363
364 if (this->subequation_verbosity == Parameters::Verbosity::extra_verbose)
365 {
366 this->pcout
367 << "PDE-based reinitialization solution norm difference = "
368 << solution_diff.l2_norm() << std::endl;
369 this->pcout
370 << "PDE-based reinitialization steady-state criterion value = "
371 << stop_criterion << std::endl;
372 this->pcout
373 << "PDE-based reinitialization fixed steady-state criterion = "
374 << steady_state_criterion << std::endl;
375 }
376
377 return ((stop_criterion > steady_state_criterion) &&
378 (step_number <
379 this->simulation_parameters.multiphysics.cls_parameters
380 .reinitialization_method
381 .pde_based_interface_reinitialization.max_steps_number +
382 1));
383 }
384 }
