7/**
8 * @brief IDs associated to the different subequations solved in Lethe.
9 */
10enum class VOFSubequationsID : unsigned int
11{
12 /// VOF phase fraction gradient L2 projection
13 phase_gradient_projection = 0,
14 /// VOF curvature L2 projection
15 curvature_projection = 1,
16 /// VOF algebraic interface reinitialization
17 algebraic_interface_reinitialization = 2
18};
19
20/**
21 * @brief Get the VOFSubequationsID associated to a specific VOF subequation
22 * from a string.
23 *
24 * @param[in] vof_subequation_name String corresponding to a particular VOF
25 * subequation.
26 *
27 * @return VOFSubequationsID corresponding to @p vof_subequation_name.
28 */
29inline VOFSubequationsID
30get_vof_subequation_id(const std::string &vof_subequation_name)
31{
32 if (vof_subequation_name == "VOF phase gradient L2 projection")
33 return VOFSubequationsID::phase_gradient_projection;
34 else if (vof_subequation_name == "VOF curvature L2 projection")
35 return VOFSubequationsID::curvature_projection;
36 else if (vof_subequation_name == "VOF algebraic interface reinitialization")
37 return VOFSubequationsID::algebraic_interface_reinitialization;
38 else
39 throw(std::invalid_argument("Invalid VOF subequation name. Options are: \n"
40 " <VOF phase gradient L2 projection>\n"
41 " <VOF curvature L2 projection>\n"
42 " <VOF algebraic interface reinitialization>"));
43}
