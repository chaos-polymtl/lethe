48void
49SimulationControl::add_time_step(double p_timestep)
50{
51 time_step = p_timestep;
52 // Store previous time step in table
53 for (unsigned int i_time = time_step_vector.size() - 1; i_time > 0; --i_time)
54 time_step_vector[i_time] = time_step_vector[i_time - 1];
55
56 // Calculate time step, right now this is a dummy function
57 time_step_vector[0] = p_timestep;
58}

72void
73SimulationControl::update_assembly_method()
74{
75 if (iteration_number <= 1 &&
76 method == Parameters::SimulationControl::TimeSteppingMethod::bdf2 &&
77 bdf_start_method ==
78 Parameters::SimulationControl::BDFStartupMethods::multiple_step_bdf)
79 {
80 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf1;
81 set_current_time_step(initial_time_step * startup_timestep_scaling);
82 }
83 else if (iteration_number == 2 &&
84 method == Parameters::SimulationControl::TimeSteppingMethod::bdf2 &&
85 bdf_start_method == Parameters::SimulationControl::
86 BDFStartupMethods::multiple_step_bdf)
87 {
88 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf2;
89 set_suggested_time_step(initial_time_step *
90 (1 - startup_timestep_scaling));
91 }
92 else if (iteration_number == 3 &&
93 method == Parameters::SimulationControl::TimeSteppingMethod::bdf2 &&
94 bdf_start_method == Parameters::SimulationControl::
95 BDFStartupMethods::multiple_step_bdf)
96 {
97 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf2;
98 set_suggested_time_step(initial_time_step);
99 }
100 else if (iteration_number <= 1 &&
101 method == Parameters::SimulationControl::TimeSteppingMethod::bdf3 &&
102 bdf_start_method == Parameters::SimulationControl::
103 BDFStartupMethods::multiple_step_bdf)
104 {
105 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf1;
106 set_current_time_step(initial_time_step * startup_timestep_scaling);
107 }
108 else if (iteration_number == 2 &&
109 method == Parameters::SimulationControl::TimeSteppingMethod::bdf3 &&
110 bdf_start_method == Parameters::SimulationControl::
111 BDFStartupMethods::multiple_step_bdf)
112 {
113 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf2;
114 set_suggested_time_step(initial_time_step * startup_timestep_scaling);
115 }
116 else if (iteration_number == 3 &&
117 method == Parameters::SimulationControl::TimeSteppingMethod::bdf3 &&
118 bdf_start_method == Parameters::SimulationControl::
119 BDFStartupMethods::multiple_step_bdf)
120 {
121 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf3;
122 set_suggested_time_step(initial_time_step *
123 (1 - 2 * startup_timestep_scaling));
124 }
125 else if (iteration_number == 4 &&
126 method == Parameters::SimulationControl::TimeSteppingMethod::bdf3 &&
127 bdf_start_method == Parameters::SimulationControl::
128 BDFStartupMethods::multiple_step_bdf)
129 {
130 assembly_method = Parameters::SimulationControl::TimeSteppingMethod::bdf3;
131 set_suggested_time_step(initial_time_step);
132 }
133 else
134 {
135 assembly_method = method;

139unsigned int
140SimulationControl::get_number_of_stages(
141 const Parameters::SimulationControl::TimeSteppingMethod &method)
142{
143 using Method = Parameters::SimulationControl::TimeSteppingMethod;
144
145 switch (method)
146 {
147 case Method::steady:
148 case Method::steady_bdf:
149 case Method::bdf1:
150 case Method::bdf2:
151 case Method::bdf3:
152 return 1;
153
154 // When referencing to SDIRK methods the pattern used is
155 // sdirkOrderStage. For instance, sdirk22 indicates SDIRK method with
156 // order 2 and 2 stages
157 case Method::sdirk22:
158 return 2;
159
160 case Method::sdirk33:
161 case Method::sdirk43:
162 return 3;
163
164 default:
165 AssertThrow(
166 false,
167 ExcMessage(
168 "Unknown TimeSteppingMethod passed to "
169 "get_number_of_stages(). Please add the number of stages of the new method in core/simulation_control.cc."));
170 return 1;
171 }
172}

214std::vector<double>
215SimulationControl::get_checkpointed_simulation_control_info(
216 const std::string &prefix)
217{
218 std::string filename = prefix + ".simulationcontrol";
219 std::ifstream input(filename.c_str());
220 AssertThrow(input, ExcFileNotOpen(filename));
221
222 // Store the time steps and last checkpointed time without modifying the
223 // simulation control information
224 std::vector<double> time_steps(time_step_vector.size());
225 double last_checkpointed_time;
226
227 // Returned vector with the last checkpointed time and time step
228 std::vector<double> simulation_control_info(2);
229
230 std::string buffer;
231 std::getline(input, buffer);
232 for (unsigned int i = 0; i < time_step_vector.size(); ++i)
233 input >> buffer >> time_steps[i];
234 input >> buffer >> buffer;
235 input >> buffer >> last_checkpointed_time;
236 input >> buffer >> buffer;
237
238 // Fix the current time as final checkpointed time
239 simulation_control_info[0] = last_checkpointed_time;
240
241 // Fix time step to be the last time_step that was used
242 simulation_control_info[1] = time_steps[0];
243
244 return simulation_control_info;
245}

283bool
284SimulationControlTransient::integrate()
285{
286 if (!is_at_end())
287 {
288 previous_time = current_time;
289 first_assembly = true;
290 iteration_number++;
291 update_assembly_method();
292 add_time_step(calculate_time_step());
293 current_time += time_step;
294
295 if (is_bdf())
296 update_bdf_coefficients();
297 return true;
298 }
299
300 else
301 return false;
302}

311double
312SimulationControlTransient::calculate_time_step()
313{
314 double new_time_step = time_step;
315
316 if (adapt && iteration_number > 1)
317 {
318 new_time_step = time_step * adaptative_time_step_scaling;
319 if (CFL > 0 && max_CFL / CFL < adaptative_time_step_scaling)
320 new_time_step = time_step * max_CFL / CFL;
321
322 new_time_step = std::min(new_time_step, max_dt);
323 }
324
325 // Ensure that the time step for the last iteration is kept regardless of the
326 // end time set
327 if (time_step_independent_of_end_time)
328 return new_time_step;
329
330 // Modify last time step to ensure that the last iteration is exactly the end
331 // time specified in the parameter file
332 if (current_time + new_time_step > end_time)
333 new_time_step = end_time - current_time;
334
335 return new_time_step;
336}

338bool
339SimulationControlTransient::is_output_iteration()
340{
341 // If iteration control the only options are to not print or at certain
342 // frequency of iterations for the time interval given. In this mode, we also
343 // always output the last iteration in the simulation.
344 if (output_control == Parameters::SimulationControl::OutputControl::iteration)
345 {
346 if (output_iteration_frequency == 0)
347 return false;
348 else
349 {
350 // Check if the current step number matches the following condition:
351 // (The step number matches the output frequency OR is the last
352 // tiem-step) AND the current time is within the output time interval
353 return ((get_step_number() % output_iteration_frequency == 0 ||
354 is_at_end()) &&
355 get_current_time() >= output_time_interval[0] &&
356 get_current_time() <= output_time_interval[1]);
357 }
358 }
359
360 // If time control there are several options:
361
362 // Case 1. A specific output time frequency is given (with or without a
363 // specific time interval):
364 if (output_time_frequency != -1)
365 {
366 if ((current_time - time_last_output) - output_time_frequency >
367 -1e-12 * output_time_frequency &&
368 current_time >= output_time_interval[0] &&
369 current_time <= output_time_interval[1])
370 {
371 time_last_output = current_time;
372 return true;
373 }
374 else // if it does not match the time frequency we do not output
375 return false;
376 }
377
378 // For cases 2 and 3:
379 bool is_output_time = false;
380 double upper_bound, lower_bound, local_output_time;
381
382 // Case 2. If one or several specific output times are given
383 // We check the vector of output times one at a time. Once the list of
384 // specific times is over, we do not check anymore.
385 local_output_time = output_times_vector[output_times_counter];
386 if (local_output_time != -1)
387 {
388 if (!no_more_output_times)
389 {
390 upper_bound = local_output_time;
391 lower_bound = local_output_time;
392 }
393 else // if no more output times, we do not output anymore
394 return false;
395 }
396 else // Case 3. Only a specific time interval is specified
397 {
398 upper_bound = output_time_interval[1];
399 lower_bound = output_time_interval[0];
400 }
401
402 // We output in the current time.
403 is_output_time = (current_time >= lower_bound && current_time <= upper_bound);
404
405 // We always write one step before, in case the specific time or the
406 // interval lower bound does not correspond exactly to a time iteration
407 // performed (due to time step)
408 double next_time = current_time + calculate_time_step();
409 if (current_time < lower_bound && (next_time >= lower_bound))
410 {
411 is_output_time = true;
412
413 // If the end time is exactly the output time, there is not one step
414 // after to output and we need to update the output times counter for
415 // case 2
416 if (is_at_end() && local_output_time != -1 && is_output_time)
417 {
418 // Update the counter only if there are more elements in vector
419 if (output_times_vector.size() > output_times_counter + 1)
420 output_times_counter++;
421 else // otherwise set flag to false as no more times need to be
422 // checked
423 no_more_output_times = true;
424 }
425 }
426
427 // We always write one step after, in case the specific time or the
428 // interval upper bound does not correspond exactly to a time iteration
429 // performed (due to time step)
430 if (current_time >= upper_bound && (previous_time < upper_bound))
431 {
432 is_output_time = true;
433
434 // Update specific time for case 2 once we have written the upper bound
435 if (local_output_time != -1 && is_output_time)
436 {
437 // Update the counter only if there are more elements in vector
438 if (output_times_vector.size() > output_times_counter + 1)
439 output_times_counter++;
440 else // otherwise set flag to false as no more times need to be
441 // checked
442 no_more_output_times = true;
443 }
444 }
445
446 return is_output_time;
447}

463SimulationControlTransient::read(const std::string &prefix)
464{
465 SimulationControl::read(prefix);
466
467 if (output_control == Parameters::SimulationControl::OutputControl::time)
468 {
469 std::string filename = prefix + ".simulationcontrol";
470 std::ifstream input(filename.c_str());
471 unsigned int line_no = 0;
472 std::string buffer;
473 while (line_no != 8)
474 {

476 line_no++;

480
