14/**
15 * @brief Reports the ratio between the minimum cell size and the maximum
16 * particle diameter.
17 *
18 * This function loops over all locally owned cells in a distributed mesh
19 * to find the minimum vertex-to-vertex distance (cell size), then compares
20 * it to the maximum particle diameter. It prints out the results using
21 * a ConditionalOStream and issues a warning if the ratio is too low.
22 *
23 * @tparam dim The spatial dimension of the triangulation.
24 * @param triangulation A reference to the distributed triangulation (mesh) object.
25 * @param maximum_particle_diameter The largest particle diameter in the simulation.
26 * @param pcout A ConditionalOStream used for printing output. In parallel simulations,
27 * this stream ensures that only one processor (usually the master) prints
28 * messages to avoid cluttered output from all MPI ranks.
29 * @param mpi_communicator The MPI communicator used for the parallel simulation.
30 * This is required to compute the **global minimum cell size** across all
31 * MPI processes.
32 */
33template <int dim>
34void
35report_cell_size_to_particle_diameter_ratio(
36 const parallel::DistributedTriangulationBase<dim> &triangulation,
37 const double maximum_particle_diameter,
38 const ConditionalOStream &pcout,
39 const MPI_Comm &mpi_communicator)
40{
41 double min_vertex_distance = std::numeric_limits<double>::max();
42 for (const auto &cell : triangulation.active_cell_iterators())
43 if (cell->is_locally_owned())
44 min_vertex_distance =
45 std::min(min_vertex_distance, cell->minimum_vertex_distance());
46 min_vertex_distance =
47 Utilities::MPI::min(min_vertex_distance, mpi_communicator);
48 pcout << "Minimum vertex distance between cell vertices: "
49 << min_vertex_distance << std::endl;
50 double ratio = min_vertex_distance / maximum_particle_diameter;
51 pcout << "Minimum cell size to maximum particle diameter ratio: " << ratio
52 << std::endl;
53
54 if (ratio < 1.0)
55 {
56 AssertThrow(
57 false,
58 ExcMessage(
59 "Minimum cell size is smaller than the maximum particle diameter. "
60 "Consider coarsening the mesh to achieve a ratio larger than 1"));
61 }
62}
