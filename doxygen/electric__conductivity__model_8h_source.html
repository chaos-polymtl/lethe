41 /**

44 * @return value of the electric conductivity calculated with the fields_value.

52 /**

57 void
58 vector_value(
59 [[maybe_unused]] const std::map<field, std::vector<double>> &field_vectors,

62 property_vector.assign(property_vector.size(), electric_conductivity);
63 }
64
65 /**

67 * @param field_values Value of the various fields on which the property may depend.
68 * @param id Indicator of the field with respect to which the jacobian
69 * should be calculated
70 * @return value of the partial derivative of the electric conductivity with respect to the field.
71 */
72
73 double
74 jacobian(const std::map<field, double> & /*field_values*/,
75 field /*id*/) override

77 return 0;
78 };
79
80 /**
81 * @brief vector_jacobian Calculate the derivative of the electric conductivity with respect to a field
82 * @param field_vectors Vector for the values of the fields used to evaluate the property
83 * @param id Identifier of the field with respect to which a derivative should be calculated
84 * @param jacobian Vector of the value of the derivative of the electric conductivity with respect to the field id
85 */
86
87 void
88 vector_jacobian(
89 const std::map<field, std::vector<double>> & /*field_vectors*/,
90 [[maybe_unused]] const field id,
91 std::vector<double> &jacobian_vector) override
92 {
93 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0);
94 };
95
96private:
97 const double electric_conductivity;
98};
