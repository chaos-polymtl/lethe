22/**
23 * @brief A solver class for the Navier-Stokes equation using GLS stabilization
24 *
25 * @tparam dim An integer that denotes the dimension of the space in which
26 * the flow is solved
27 *
28 * @ingroup solvers
29 */
30
31template <int dim>
32class FluidDynamicsMatrixBased
33 : public NavierStokesBase<dim, GlobalVectorType, IndexSet>
34{
35public:
36 FluidDynamicsMatrixBased(SimulationParameters<dim> &nsparam);
37 ~FluidDynamicsMatrixBased();
38
39 /**
40 * @brief solve Solves the Navier-Stokes problem
41 *
42 * This functions solves the problem defined by the Navier-Stokes parameter
43 * by iterating through time or through the mesh refinements.
44 */
45 virtual void
46 solve();
47
48protected:
49 /**
50 * @brief setup_dofs_fd Setup the degree of freedom, system matrix and solution vectors
51 * Navier-Stokes problem
52 */
53 virtual void
54 setup_dofs_fd() override;
55
56 /**
57 * @brief Update the sum_over_previous_stages variable to use it in the solver.
58 *
59 * @param stage An unsigned integer which gives the index of the current stage
60 * @param method SDIRK method for now (BDF methods are single stage)
61 */
62 void
63 multi_stage_preresolution(
64 unsigned int stage,
65 Parameters::SimulationControl::TimeSteppingMethod method) override;
66
67 /**
68 * @brief Calculate \f$ k_i \f$ from \f$ u*_i : k_i = \frac{u*_i - u_n}{time_step*a_ii} - \sum{j=1}^{i-1} a_ij * k_j \f$
69 * Update \f$ \sum_{i=1}^{N_{stages}} b_i * k_i \f$
70 *
71 * @param stage An unsigned integer which gives the index of the current stage
72 * @param method SDIRK methods for now (BDF methods are single stage)
73 */
74 virtual void
75 multi_stage_postresolution(
76 unsigned int stage,
77 Parameters::SimulationControl::TimeSteppingMethod method,
78 double time_step) override;
79
80 /**
81 * @brief \f$ u_{n+1} = u_n + time_step * \sum_{i=1}^{N_{stages}} b_i * k_i \f$
82 *
83 * @param time_step The current time step to build the solution at the next time step
84 */
85 virtual void
86 update_multi_stage_solution(double time_step) override;
87
88 /**
89 * @brief Sets the initial condition for the solver
90 *
91 * If the simulation is restarted from a checkpoint, the initial solution
92 * setting is bypassed and the checkpoint is instead read.
93 *
94 * @param initial_condition_type The type of initial condition to be set
95 *
96 * @param restart A boolean that indicates if the simulation is being restarted.
97 * if set to true, the initial conditions are never set, but are instead
98 * overridden by the read_checkpoint functionality.
99 *
100 **/
101 virtual void
102 set_initial_condition_fd(
103 Parameters::FluidDynamicsInitialConditionType initial_condition_type,
104 bool restart = false) override;
105
106protected:
107 /**
108 * @brief Assembles the matrix associated with the solver
109 */
110 virtual void
111 assemble_system_matrix() override;
112
113 /**
114 * @brief Assemble the rhs associated with the solver
115 */
116 virtual void
117 assemble_system_rhs() override;
118
119 /**
120 * @brief Updates the average velocity field solution in the multiphyscics interface
121 */
122 virtual void
123 update_multiphysics_time_average_solution() override;
124
125 /**
126 * @brief Set-up the appropriate preconditioner
127 */
128 void
129 setup_preconditioner() override;
130
131 /**
132 * @brief Define the zero constraints used to solved the problem that change
133 * with other physics' solutions.
134 *
135 * It differs from FluidDynamicsMatrixBased::define_zero_constraints as it
136 * changes in time depending on the physics' solutions. Currently, it is only
137 * used to constraint solid with the temperature's evolution.
138 */
139 void
140 define_dynamic_zero_constraints();
141
142 /**
143 * @brief Fix a constant pressure on the minimum dof index. This can be enabled
144 * in the simulation control parameters.
145 */
146 void
147 define_pressure_constraints();
148
149 /**
150 * @brief Update mortar configuration.
151 *
152 * When the rotor domain is rotated, the mortar cells need to be reinitialized
153 * according to the new rotor-stator interface configuration.
154 */
155 virtual void
156 update_mortar_configuration() override;
157
158 /**
159 * @brief Assemble the local matrix for a given cell.
160 *
161 * This function is used by the WorkStream class to assemble
162 * the system matrix. It is a thread safe function.
163 *
164 * @param cell The cell for which the local matrix is assembled.
165 *
166 * @param scratch_data The scratch data which is used to store
167 * the calculated finite element information at the gauss point.
168 * See the documentation for NavierStokesScratchData for more
169 * information
170 *
171 * @param copy_data The copy data which is used to store
172 * the results of the assembly over a cell
173 */
174 virtual void
175 assemble_local_system_matrix(
176 const typename DoFHandler<dim>::active_cell_iterator &cell,
177 NavierStokesScratchData<dim> &scratch_data,
178 StabilizedMethodsTensorCopyData<dim> &copy_data);
179
180 /**
181 * @brief Assemble the local rhs for a given cell
182 *
183 * @param cell The cell for which the local matrix is assembled.
184 *
185 * @param scratch_data The scratch data which is used to store
186 * the calculated finite element information at the gauss point.
187 * See the documentation for NavierStokesScratchData for more
188 * information
189 *
190 * @param copy_data The copy data which is used to store
191 * the results of the assembly over a cell
192 */
193 virtual void
194 assemble_local_system_rhs(
195 const typename DoFHandler<dim>::active_cell_iterator &cell,
196 NavierStokesScratchData<dim> &scratch_data,
197 StabilizedMethodsTensorCopyData<dim> &copy_data);
198
199 /**
200 * @brief sets up the vector of assembler functions
201 */
202 virtual void
203 setup_assemblers();
204
205
206 /**
207 * @brief Copy local cell information to global matrix
208 */
209
210 virtual void
211 copy_local_matrix_to_global_matrix(
212 const StabilizedMethodsTensorCopyData<dim> &copy_data);
213
214 /**
215 * @brief Copy local cell rhs information to global rhs
216 */
217
218 virtual void
219 copy_local_rhs_to_global_rhs(
220 const StabilizedMethodsTensorCopyData<dim> &copy_data);
221
222 /**
223 * @brief Call for the assembly of the linear system of equation
224 *
225 * @param initial_step Indicates if this is the first solution of the linear system.
226 * If this is the case, the non_zero version of the constraints are used for
227 * the Dirichlet boundary conditions
228 */
229 void
230 solve_linear_system() override;
231
232private:
233 /**
234 * @brief Assembles an L2_projection matrix for the velocity and the pressure.
235 * This L2 projection matrix can be used to set the initial condition for
236 * the Navier-Stokes problem
237 */
238 void
239 assemble_L2_projection();
240
241
242 /**
243 * @brief Solver for the L2 Projection linear system at the initial time step to apply this initial condition. It always uses the GMRES solver with ILU(N) by default.
244 *
245 * @param[in] absolute_residual Absolute residual for the solver stopping
246 * criterion.
247 * @param[in] relative_residual Relative residual for the solver stopping
248 * criterion.
249 */
250
251 void
252 solve_L2_system(const double absolute_residual,
253 const double relative_residual);
254
255
256 /**
257 * @brief GMRES solver with ILU(N) preconditioning or AMG preconditioning
258 * @param[in] absolute_residual Absolute residual for the solver stopping
259 * criterion.
260 * @param[in] relative_residual Relative residual for the solver stopping
261 * criterion.
262 */
263 void
264 solve_system_GMRES(const double absolute_residual,
265 const double relative_residual);
266
267 /**
268 * @brief BiCGStab solver with ILU(N) preconditioning
269 * @param[in] absolute_residual Absolute residual for the solver stopping
270 * criterion.
271 * @param[in] relative_residual Relative residual for the solver stopping
272 * criterion.
273 */
274 void
275 solve_system_BiCGStab(const double absolute_residual,
276 const double relative_residual);
277 /**
278 * @brief Direct solver using TrilinosWrappers::SolverDirect
279 * The use of this solver should be avoided for 3D problems
280 */
281 void
282 solve_system_direct(const double absolute_residual,
283 const double relative_residual);
284
285 /**
286 * @brief Set-up AMG preconditioner
287 */
288 void
289 setup_AMG();
290
291 /**
292 * @brief Set-up ILU preconditioner
293 */
294 void
295 setup_ILU();
296
297
298 /**
299 * Members
300 */
301protected:
302 TrilinosWrappers::SparseMatrix system_matrix;
303
304private:
305 SparsityPattern sparsity_pattern;
306 std::shared_ptr<TrilinosWrappers::PreconditionILU> ilu_preconditioner;
307 std::shared_ptr<TrilinosWrappers::PreconditionAMG> amg_preconditioner;
308 int current_preconditioner_fill_level;
309 int initial_preconditioner_fill_level;
310};
