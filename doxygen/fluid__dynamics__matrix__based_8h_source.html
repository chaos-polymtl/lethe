22/**

64 * @param restart A boolean that indicates if the simulation is being restarted.

69 virtual void

74protected:

78 virtual void

81 /**
82 * @brief Assemble the rhs associated with the solver
83 */
84 virtual void
85 assemble_system_rhs() override;

87 /**

90 virtual void

100 * @brief Define the zero constraints used to solved the problem that change

122 * This function is used by the WorkStream class to assemble

124 *
125 * @param cell The cell for which the local matrix is assembled.
126 *
127 * @param scratch_data The scratch data which is used to store

132 * @param copy_data The copy data which is used to store

135 virtual void
136 assemble_local_system_matrix(

141 /**
142 * @brief Assemble the local rhs for a given cell

151 * @param copy_data The copy data which is used to store
152 * the results of the assembly over a cell

154 virtual void
155 assemble_local_system_rhs(

159

163 virtual void

183 /**
184 * @brief Call for the assembly of the linear system of equation
185 *
186 * @param initial_step Indicates if this is the first solution of the linear system.

190 * @param renewed_matrix Indicates if the matrix has been reassembled, and thus

196 void
197 solve_linear_system(const bool initial_step,
198 const bool renewed_matrix = true) override;

200private:
