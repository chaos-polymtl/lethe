4/*
5 * Implementation of the Cahn-Hilliard equations as an auxiliary physics.
6 * Equations solved:
7 * dPhi/dt + u * gradPhi = div(M(Phi)*grad eta)
8 * eta - f(Phi) + epsilon^2 * div(grad Phi) = 0
9 * with Phi the phase field parameter (or phase order), eta the chemical
10 * potential, and M the mobility function and epsilon the interface thickness.
11 * The phase field parameter must not be confused with the order (respectively
12 * phase_cahn_hilliard_order and potential_cahn_hilliard_order) of the finite
13 * elements related to the phase field parameter and the chemical potential.
14 */
15

57 CahnHilliard(MultiphysicsInterface<dim> *multiphysics_interface,
58 const SimulationParameters<dim> &p_simulation_parameters,
59 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
60 p_triangulation,
61 std::shared_ptr<SimulationControl> p_simulation_control)
62 : AuxiliaryPhysics<dim, GlobalVectorType>(
63 p_simulation_parameters.non_linear_solver.at(PhysicsID::cahn_hilliard))
64 , multiphysics(multiphysics_interface)
65 , computing_timer(p_triangulation->get_mpi_communicator(),

70 , triangulation(p_triangulation)
71 , simulation_control(p_simulation_control)
72 , dof_handler(*triangulation)
73 {
74 if (simulation_parameters.mesh.simplex)
75 {
76 // for simplex meshes
77 fe = std::make_shared<FESystem<dim>>(
78 FE_SimplexP<dim>(
79 simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
80 1,
81 FE_SimplexP<dim>(
82 simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
83 1);
84 mapping = std::make_shared<MappingFE<dim>>(*fe);
85 cell_quadrature = std::make_shared<QGaussSimplex<dim>>(
86 std::max(
87 simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
88 simulation_parameters.fem_parameters
89 .potential_cahn_hilliard_order) +
90 1);
91 face_quadrature = std::make_shared<QGaussSimplex<dim - 1>>(
92 std::max(
93 simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
94 simulation_parameters.fem_parameters
95 .potential_cahn_hilliard_order) +
96 1);
97 ;
98 }
99 else
100 {
101 // Usual case, for quad/hex meshes
102 fe = std::make_shared<FESystem<dim>>(
103 FE_Q<dim>(
104 simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
105 1,
106 FE_Q<dim>(
107 simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
108 1);
109 mapping = std::make_shared<MappingQ<dim>>(std::max(
110 simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
111 simulation_parameters.fem_parameters.potential_cahn_hilliard_order));
112 cell_quadrature = std::make_shared<QGauss<dim>>(
113 std::max(
114 simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
115 simulation_parameters.fem_parameters
116 .potential_cahn_hilliard_order) +
117 1);
118 face_quadrature = std::make_shared<QGauss<dim - 1>>(
119 std::max(
120 simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
121 simulation_parameters.fem_parameters
122 .potential_cahn_hilliard_order) +
123 1);
124 }
125
126 // Allocate solution transfer
127 solution_transfer =
128 std::make_shared<SolutionTransfer<dim, GlobalVectorType>>(dof_handler);
129
130 // Set size of previous solutions using BDF schemes information
131 previous_solutions.resize(maximum_number_of_previous_solutions());
132
133 // Prepare previous solutions transfer
134 previous_solutions_transfer.reserve(previous_solutions.size());
135 for (unsigned int i = 0; i < previous_solutions.size(); ++i)
136 {
137 previous_solutions_transfer.emplace_back(
138 SolutionTransfer<dim, GlobalVectorType>(this->dof_handler));
139 }
140
141 // Change the behavior of the timer for situations when you don't want
142 // outputs
143 if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
144 this->computing_timer.disable_output();
145 }

315 /**
316 * @brief Verify consistency of the input parameters for boundary
317 * conditions to ensure that for every boundary condition within the
318 * triangulation, a boundary condition has been specified in the input file.
319 */
320 void
321 verify_consistency_of_boundary_conditions()
322 {
323 // Sanity check all of the boundary conditions of the triangulation to
324 // ensure that they have a type.
325 std::vector<types::boundary_id> boundary_ids_in_triangulation =
326 this->triangulation->get_boundary_ids();
327 for (auto const &boundary_id_in_tria : boundary_ids_in_triangulation)
328 {
329 AssertThrow(
330 simulation_parameters.boundary_conditions_cahn_hilliard.type.find(
331 boundary_id_in_tria) !=
332 simulation_parameters.boundary_conditions_cahn_hilliard.type.end(),
333 CahnHilliardBoundaryConditionMissing(boundary_id_in_tria));
334 }
335 }
