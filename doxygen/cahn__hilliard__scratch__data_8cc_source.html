7template <int dim>
8void
9CahnHilliardScratchData<dim>::allocate()
10{
11 // Initialize size of arrays
12 this->n_q_points = fe_values_cahn_hilliard.get_quadrature().size();
13 this->n_dofs = fe_values_cahn_hilliard.get_fe().n_dofs_per_cell();
14
15 // Initialize arrays related to quadrature
16 this->JxW = std::vector<double>(n_q_points);
17
18 // Forcing term array
19 this->source_phase_order = std::vector<double>(n_q_points);
20 this->source_chemical_potential = std::vector<double>(n_q_points);
21
22 // Initialize arrays related to phase order and chemical potential
23 this->phase_order.component = 0;
24 this->chemical_potential.component = 1;
25
26 // Variables for the Cahn-Hilliard equations
27 this->phase_order_values = std::vector<double>(n_q_points);
28 this->phase_order_gradients = std::vector<Tensor<1, dim>>(n_q_points);
29 this->phase_order_laplacians = std::vector<double>(n_q_points);
30 this->chemical_potential_values = std::vector<double>(n_q_points);
31 this->chemical_potential_gradients = std::vector<Tensor<1, dim>>(n_q_points);
32 this->chemical_potential_laplacians = std::vector<double>(n_q_points);
33
34 // Phase order for BDF schemes
35 this->previous_phase_order_values =
36 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
37 std::vector<double>(n_q_points));
38
39 this->previous_chemical_potential_values =
40 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
41 std::vector<double>(n_q_points));
42
43 // Initialize arrays related to shape functions
44 // Phase-order shape functions
45 this->phi_phase =
46 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
47 this->grad_phi_phase = std::vector<std::vector<Tensor<1, dim>>>(
48 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
49 this->hess_phi_phase = std::vector<std::vector<Tensor<2, dim>>>(
50 n_q_points, std::vector<Tensor<2, dim>>(n_dofs));
51 this->laplacian_phi_phase =
52 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
53
54 // Chemical potential shape functions
55 this->phi_potential =
56 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
57 this->grad_phi_potential = std::vector<std::vector<Tensor<1, dim>>>(
58 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
59 this->hess_phi_potential = std::vector<std::vector<Tensor<2, dim>>>(
60 n_q_points, std::vector<Tensor<2, dim>>(n_dofs));
61 this->laplacian_phi_potential =
62 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
63
64 // Velocity values
65 this->velocity_values = std::vector<Tensor<1, dim>>(this->n_q_points);
66 this->previous_velocity_values = std::vector<std::vector<Tensor<1, dim>>>(
67 maximum_number_of_previous_solutions(),
68 std::vector<Tensor<1, dim>>(this->n_q_points));
69 this->velocity_gradient_values =
70 std::vector<Tensor<2, dim>>(this->n_q_points);
71
72 // Allocate physical properties
73 this->surface_tension = std::vector<double>(n_q_points);
74 this->mobility_cahn_hilliard = std::vector<double>(n_q_points);
75 this->mobility_cahn_hilliard_gradient = std::vector<double>(n_q_points);
76
77 // Physical properties
78 fields.insert(
79 std::pair<field, std::vector<double>>(field::phase_order_cahn_hilliard,
80 n_q_points));
81 fields.insert(std::pair<field, std::vector<double>>(
82 field::phase_order_cahn_hilliard_filtered, n_q_points));
83}

85template <int dim>
86void
87CahnHilliardScratchData<dim>::calculate_physical_properties()
88{
89 switch (properties_manager.get_number_of_fluids())
90 {
91 case 1:
92 {
93 throw std::runtime_error("Unsupported number of fluids (<2)");
94 }
95 case 2:
96 {
97 set_field_vector(field::phase_order_cahn_hilliard,
98 this->phase_order_values,
99 this->fields);
100
101 // Gather properties from material interactions
102 const auto material_interaction_id =
103 properties_manager.get_material_interaction_id(
104 material_interactions_type::fluid_fluid, 0, 1);
105
106 // Gather surface tension
107 const auto surface_tension_model =
108 properties_manager.get_surface_tension(material_interaction_id);
109 surface_tension_model->vector_value(fields, surface_tension);
110
111 // Gather mobility
112 const auto mobility_cahn_hilliard_model =
113 properties_manager.get_mobility_cahn_hilliard(
114 material_interaction_id);
115 mobility_cahn_hilliard_model->vector_value(fields,
116 mobility_cahn_hilliard);
117 mobility_cahn_hilliard_model->vector_jacobian(
118 fields, phase_order_cahn_hilliard, mobility_cahn_hilliard_gradient);
119
120 break;
121 }
122 default:
123 throw std::runtime_error("Unsupported number of fluids (>2)");
124 }
125}
