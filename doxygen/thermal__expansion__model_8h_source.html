101/**
102 * @brief Implements a phase-dependent thermal expansion coefficient
103 */
104class ThermalExpansionPhaseChange : public ThermalExpansionModel
105{
106public:
107 /**
108 * @brief Default constructor
109 */
110 ThermalExpansionPhaseChange(const Parameters::PhaseChange phase_change_params)
111 : p_phase_change_params(phase_change_params)
112 {
113 this->model_depends_on[field::temperature] = true;
114 }
115
116 /**
117 * @brief value Calculates the value of the thermal expansion coefficient
118 * @param fields_value Value of the various field on which the thermal expansion coefficient depends.
119 * @return value of the thermal expansion coefficient calculated with the fields_value.
120 */
121 double
122 value(const std::map<field, double> &fields_value) override
123 {
124 Assert(fields_value.find(field::temperature) != fields_value.end(),
125 PhysicialPropertyModelFieldUndefined("ThermalExpansionPhaseChange",
126 "temperature"));
127 if (fields_value.at(field::temperature) > p_phase_change_params.T_liquidus)
128 return p_phase_change_params.thermal_expansion_l;
129 else
130 return p_phase_change_params.thermal_expansion_s;
131 };
132
133 /**
134 * @brief vector_value Calculates the vector value of thermal expansion coefficients
135 * @param field_vectors Vector of properties on which the thermal expansion coefficients depend
136 * @param property_vector Values of the thermal expansion coefficients
137 */
138 void
139 vector_value(const std::map<field, std::vector<double>> &field_vectors,
140 std::vector<double> &property_vector) override
141 {
142 Assert(field_vectors.find(field::temperature) != field_vectors.end(),
143 PhysicialPropertyModelFieldUndefined("ThermalExpansionPhaseChange",
144 "temperature"));
145 const std::vector<double> &T = field_vectors.at(field::temperature);
146 for (unsigned int i = 0; i < property_vector.size(); ++i)
147 {
148 // Thermal expansion of solid phase
149 if (T[i] > p_phase_change_params.T_liquidus)
150 property_vector[i] = p_phase_change_params.thermal_expansion_l;
151 else
152 property_vector[i] = p_phase_change_params.thermal_expansion_s;
153 // Thermal expansion of liquid phase
154 }
155 };
156
157 /**
158 * @brief jacobian Calculates the jacobian (the partial derivative) of the thermal expansion with respect to a field
159 * @param field_values Value of the various fields on which the property may depend.
160 * @param id Indicator of the field with respect to which the jacobian
161 * should be calculated
162 * @return value of the partial derivative of the thermal expansion with respect to the field.
163 */
164
165 double
166 jacobian(const std::map<field, double> &field_values, field id) override
167 {
168 if (id == field::temperature)
169 {
170 Assert(field_values.find(field::temperature) != field_values.end(),
171 PhysicialPropertyModelFieldUndefined(
172 "ThermalExpansionPhaseChange", "temperature"));
173 return this->numerical_jacobian(field_values, field::temperature);
174 }
175 else
176 return 0;
177 };
178
179 /**
180 * @brief vector_jacobian Calculate the derivative of the thermal expansion with respect to a field
181 * @param field_vectors Vector for the values of the fields used to evaluate the property
182 * @param id Identifier of the field with respect to which a derivative should be calculated
183 * @param jacobian Vector of the value of the derivative of the thermal expansion with respect to the field id
184 */
185
186 void
187 vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
188 const field id,
189 std::vector<double> &jacobian_vector) override
190 {
191 Assert(field_vectors.find(field::temperature) != field_vectors.end(),
192 PhysicialPropertyModelFieldUndefined("ThermalExpansionPhaseChange",
193 "temperature"));
194 vector_numerical_jacobian(field_vectors, id, jacobian_vector);
195 };
196
197private:
198 Parameters::PhaseChange p_phase_change_params;
199};
