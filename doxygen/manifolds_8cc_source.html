44 void
45 Manifolds::parse_boundary(const ParameterHandler &prm,
46 const unsigned int i_bc)
47 {
48 const std::string op = prm.get("type");
49 if (op == "none")
50 types[i_bc] = ManifoldType::none;
51 else if (op == "spherical")
52 types[i_bc] = ManifoldType::spherical;
53 else if (op == "cylindrical")
54 types[i_bc] = ManifoldType::cylindrical;
55 else if (op == "iges")
56 types[i_bc] = ManifoldType::iges;
57
58 id[i_bc] = prm.get_integer("id");
59 manifold_point[i_bc] = prm.get("point coordinates");
60 manifold_direction[i_bc] = prm.get("direction vector");
61 cad_files[i_bc] = prm.get("cad file");
62 }

64 void
65 Manifolds::declare_parameters(ParameterHandler &prm,
66 unsigned int subsection_max_size)
67 {
68 manifold_point.resize(subsection_max_size);
69 manifold_direction.resize(subsection_max_size);
70 cad_files.resize(subsection_max_size);
71
72 prm.enter_subsection("manifolds");
73 {
74 prm.declare_entry("number",
75 "0",
76 Patterns::Integer(),
77 "Number of boundary conditions");
78 id.resize(subsection_max_size);
79 types.resize(subsection_max_size);
80 for (unsigned int i = 0; i < subsection_max_size; i++)
81 {
82 prm.enter_subsection("manifold " + Utilities::int_to_string(i));
83 declareDefaultEntry(prm, i);
84 prm.leave_subsection();
85 }
86 }
87 prm.leave_subsection();
88 }

92 {
93 prm.enter_subsection("manifolds");
94 {

102 prm.leave_subsection();

109template <int dim, int spacedim>
110void
111attach_manifolds_to_triangulation(
112 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
113 Parameters::Manifolds manifolds)
114{
115 for (unsigned int i = 0; i < manifolds.types.size(); ++i)
116 {
117 if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::spherical)
118 {
119 // Create a point using the parameter file input
120 Point<spacedim> circle_center(
121 value_string_to_tensor<spacedim>(manifolds.manifold_point[i]));
122
123 SphericalManifold<dim, spacedim> manifold_description(circle_center);
124 triangulation.set_manifold(manifolds.id[i], manifold_description);
125 }
126 else if (manifolds.types[i] ==
127 Parameters::Manifolds::ManifoldType::cylindrical)
128 {
129 if constexpr (spacedim == 3)
130 {
131 // Create a point using the parameter file input
132 Point<spacedim> point_on_axis(
133 value_string_to_tensor<spacedim>(manifolds.manifold_point[i]));
134
135 // Create a tensor representing the direction of the length of the
136 // cylinder
137 Tensor<1, spacedim> cylinder_axis(
138 value_string_to_tensor<spacedim>(
139 manifolds.manifold_direction[i]));
140
141 CylindricalManifold<dim, spacedim> manifold_description(
142 cylinder_axis, point_on_axis);
143 triangulation.set_manifold(manifolds.id[i], manifold_description);
144 }
145 else
146 throw std::runtime_error(
147 "Cylindrical manifolds are not supported in 2D");
148 }
149 else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::iges)
150 {
151 attach_cad_to_manifold(triangulation,
152 manifolds.cad_files[i],
153 manifolds.id[i]);
154 }
155 else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::none)
156 {
157 }
158 else
159 throw std::runtime_error("Unsupported manifolds type");
160 }
161}

179#ifdef DEAL_II_WITH_OPENCASCADE
180void
181attach_cad_to_manifold(parallel::DistributedTriangulationBase<3> &triangulation,
182 const std::string &cad_name,
183 const unsigned int manifold_id)
184{
185 TopoDS_Shape cad_surface = OpenCASCADE::read_IGES(cad_name, 1e-3);
186
187 // Enforce manifold over boundary ID
188 for (const auto &cell : triangulation.active_cell_iterators())
189 {
190 for (const auto &face : cell->face_iterators())
191 {
192 if (face->boundary_id() == manifold_id)
193 {
194 face->set_all_manifold_ids(manifold_id);
195 }
196 }
197 }
198
199 // Define tolerance for interpretation of CAD file
200 const double tolerance = OpenCASCADE::get_shape_tolerance(cad_surface) * 5;
201
202 // OpenCASCADE::NormalProjectionManifold<3,3> normal_projector(
203 // cad_surface, tolerance);
204 OpenCASCADE::NormalToMeshProjectionManifold<3, 3> normal_projector(
205 cad_surface, tolerance);
206
207 triangulation.set_manifold(manifold_id, normal_projector);
208}
209#else
210void
211attach_cad_to_manifold(parallel::DistributedTriangulationBase<3> &,
212 const std::string &,
213 const unsigned int)
214{
215 throw std::runtime_error(
216 "IGES manifolds require DEAL_II to be compiled with OPENCASCADE");
217}
218#endif // DEAL_II_WITH_OPENCASCADE
