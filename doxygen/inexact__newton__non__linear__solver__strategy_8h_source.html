11/**
12 * @brief Non-linear solver for non-linear systems of equations which uses an inexact
13 * Newton method with \f$\alpha\f$ relaxation to ensure that the residual is
14 * monotonically decreasing. It allows to reuse the matrix and avoid its
15 * reassembly for every Newton step.
16 *
17 */
18template <typename VectorType>
19class InexactNewtonNonLinearSolverStrategy
20 : public PhysicsSolverStrategy<VectorType>
21{
22public:
23 /**
24 * @brief Constructor.
25 *
26 * @param[in] physics_solver A pointer to the physics solver to which the
27 * non-linear solver is attached.
28 *
29 * @param[in] param Non-linear solver parameters as specified in the
30 * simulation parameter file.
31 *
32 */
33 InexactNewtonNonLinearSolverStrategy(
34 PhysicsSolver<VectorType> *physics_solver,
35 const Parameters::NonLinearSolver &param);
36
37
38 /**
39 * @brief Solve the non-linear system of equations.
40 *
41 */
42 void
43 solve() override;
44
45private:
46 bool matrix_requires_assembly;
47};

49template <typename VectorType>
50InexactNewtonNonLinearSolverStrategy<VectorType>::
51 InexactNewtonNonLinearSolverStrategy(
52 PhysicsSolver<VectorType> *physics_solver,

58template <typename VectorType>
59void
60InexactNewtonNonLinearSolverStrategy<VectorType>::solve()
61{
62 double global_res;
63 double current_res;
64 double last_res;
65 unsigned int outer_iteration = 0;
66 last_res = 1e6;
67 current_res = 1e6;
68 global_res = 1e6;
69
70 // current_res and global_res are different as one is defined based on the l2
71 // norm of the residual vector (current_res) and the other (global_res) is
72 // defined by the physical solver and may differ from the l2_norm of the
73 // residual vector. Only the global_res is compared to the tolerance in order
74 // to evaluate if the nonlinear system is solved. Only current_res is used for
75 // the alpha scheme as this scheme only monitors the convergence of the
76 // non-linear system of equation (the matrix problem).
77
78 PhysicsSolver<VectorType> *solver = this->physics_solver;
79
80 const double rescale_metric = solver->get_residual_rescale_metric();
81
82 auto &evaluation_point = solver->get_evaluation_point();
83 auto &present_solution = solver->get_present_solution();
84
85 if (!this->params.reuse_matrix)
86 matrix_requires_assembly = true;
87
88 while ((global_res > this->params.tolerance) &&
89 outer_iteration < this->params.max_iterations)
90 {
91 evaluation_point = present_solution;
92
93 if (matrix_requires_assembly)
94 {
95 solver->assemble_system_matrix();
96 solver->setup_preconditioner();
97 }
98
99 if (this->params.force_rhs_calculation || outer_iteration == 0)
100 solver->assemble_system_rhs();
101
102 if (outer_iteration == 0)
103 {
104 auto &system_rhs = solver->get_system_rhs();
105 current_res = system_rhs.l2_norm() / rescale_metric;
106 last_res = current_res;
107 }
108
109 if (this->params.verbosity != Parameters::Verbosity::quiet)
110 {
111 solver->pcout << "Newton iteration: " << outer_iteration
112 << " - Residual: " << current_res << std::endl;
113 }
114
115 solver->solve_linear_system();
116 double last_alpha_res = current_res;
117
118 unsigned int alpha_iter = 0;
119 for (double alpha = 1.0; alpha > 1e-1; alpha *= 0.5)
120 {
121 auto &local_evaluation_point = solver->get_local_evaluation_point();
122 auto &newton_update = solver->get_newton_update();
123 local_evaluation_point = present_solution;
124 local_evaluation_point.add(alpha, newton_update);
125 solver->apply_constraints();
126 evaluation_point = local_evaluation_point;
127 solver->assemble_system_rhs();
128
129 auto &system_rhs = solver->get_system_rhs();
130 current_res = system_rhs.l2_norm() / rescale_metric;
131
132 if (this->params.verbosity != Parameters::Verbosity::quiet)
133 {
134 solver->pcout << "\talpha = " << std::setw(6) << alpha
135 << std::setw(0) << " res = "
136 << std::setprecision(this->params.display_precision)
137 << std::setw(6) << current_res;
138
139 solver->output_newton_update_norms(
140 this->params.display_precision);
141 }
142
143 // If it's not the first iteration of alpha check if the residual is
144 // smaller then the last alpha iteration. If it's not smaller we fall
145 // back to the last alpha iteration, which is generally the first one.
146 if (alpha_iter != 0 and current_res > last_alpha_res)
147 {
148 alpha = 2 * alpha;
149 local_evaluation_point = present_solution;
150 local_evaluation_point.add(alpha, newton_update);
151 solver->apply_constraints();
152 evaluation_point = local_evaluation_point;
153
154 if (this->params.verbosity != Parameters::Verbosity::quiet)
155 {
156 solver->pcout
157 << "\t\talpha value was kept at alpha = " << alpha
158 << " since alpha = " << alpha / 2
159 << " increased the residual" << std::endl;
160 }
161 current_res = last_alpha_res;
162 break;
163 }
164 if (current_res < this->params.step_tolerance * last_res ||
165 last_res < this->params.tolerance)
166 {
167 // If the current residual has decreased by less than
168 // the last residual * matrix_tolerance or more than one alpha
169 // iteration was required, the jacobian approximation has become
170 // insufficiently performant and, consequently, it is preferable
171 // to renew the jacobian matrix
172 if (current_res > this->params.matrix_tolerance * last_res ||
173 alpha_iter > 0)
174 matrix_requires_assembly = true;
175 else
176 matrix_requires_assembly = false;
177
178 break;
179 }
180 last_alpha_res = current_res;
181 alpha_iter++;
182 }
183
184 global_res = solver->get_current_residual() / rescale_metric;
185 present_solution = evaluation_point;
186 last_res = current_res;
187 ++outer_iteration;
188 }
189
190 // If the non-linear solver has not converged abort simulation if
191 // abort_at_convergence_failure=true
192 if ((global_res > this->params.tolerance) &&
193 outer_iteration >= this->params.max_iterations &&
194 this->params.abort_at_convergence_failure)
195 {
196 throw(std::runtime_error(
197 "Stopping simulation because the non-linear solver has failed to converge"));
198 }
199}
