12template <int dim, typename PropertiesIndex>
13std::shared_ptr<ParticleWallContactForceBase<dim, PropertiesIndex>>
14set_particle_wall_contact_force_model(
15 const DEMSolverParameters<dim> &dem_parameters)
16{
17 ParticleWallContactForceModel particle_wall_contact_force_model =
18 dem_parameters.model_parameters.particle_wall_contact_force_method;
19
20 std::shared_ptr<ParticleWallContactForceBase<dim, PropertiesIndex>>
21 particle_wall_contact_force_object;
22
23 switch (particle_wall_contact_force_model)
24 {
25 case ParticleWallContactForceModel::linear:
26 {
27 set_rolling_resistance_model<dim,
28 PropertiesIndex,
29 ParticleWallContactForceModel::linear>(
30 dem_parameters, particle_wall_contact_force_object);
31 break;
32 }
33 case ParticleWallContactForceModel::nonlinear:
34 {
35 set_rolling_resistance_model<
36 dim,
37 PropertiesIndex,
38 ParticleWallContactForceModel::nonlinear>(
39 dem_parameters, particle_wall_contact_force_object);
40 break;
41 }
42 case ParticleWallContactForceModel::JKR:
43 {
44 set_rolling_resistance_model<dim,
45 PropertiesIndex,
46 ParticleWallContactForceModel::JKR>(
47 dem_parameters, particle_wall_contact_force_object);
48 break;
49 }
50 case ParticleWallContactForceModel::DMT:
51 {
52 set_rolling_resistance_model<dim,
53 PropertiesIndex,
54 ParticleWallContactForceModel::DMT>(
55 dem_parameters, particle_wall_contact_force_object);
56 break;
57 }
58 default:
59 {
60 throw std::runtime_error(
61 "The chosen particle-wall contact force model is invalid");
62 break;
63 }
64 }
65 return particle_wall_contact_force_object;
66}

69template <int dim,
70 typename PropertiesIndex,
71 Parameters::Lagrangian::ParticleWallContactForceModel
72 particle_wall_contact_force_model>
73void
74set_rolling_resistance_model(
75 const DEMSolverParameters<dim> &dem_parameters,
76 std::shared_ptr<ParticleWallContactForceBase<dim, PropertiesIndex>>
77 &particle_wall_contact_force_object)
78{
79 RollingResistanceMethod rolling_resistance_method =
80 dem_parameters.model_parameters.rolling_resistance_method;
81
82 switch (rolling_resistance_method)
83 {
84 case RollingResistanceMethod::none:
85 {
86 particle_wall_contact_force_object = std::make_shared<
87 ParticleWallContactForce<dim,
88 PropertiesIndex,
89 particle_wall_contact_force_model,
90 RollingResistanceMethod::none>>(
91 dem_parameters);
92 break;
93 }
94 case RollingResistanceMethod::constant:
95 {
96 particle_wall_contact_force_object = std::make_shared<
97 ParticleWallContactForce<dim,
98 PropertiesIndex,
99 particle_wall_contact_force_model,
100 RollingResistanceMethod::constant>>(
101 dem_parameters);
102 break;
103 }
104 case RollingResistanceMethod::viscous:
105 {
106 particle_wall_contact_force_object = std::make_shared<
107 ParticleWallContactForce<dim,
108 PropertiesIndex,
109 particle_wall_contact_force_model,
110 RollingResistanceMethod::viscous>>(
111 dem_parameters);
112 break;
113 }
114 case RollingResistanceMethod::epsd:
115 {
116 particle_wall_contact_force_object = std::make_shared<
117 ParticleWallContactForce<dim,
118 PropertiesIndex,
119 particle_wall_contact_force_model,
120 RollingResistanceMethod::epsd>>(
121 dem_parameters);
122 break;
123 }
124 default:
125 throw std::runtime_error("Invalid rolling resistance method");
126 }
127}
