9Vector<double>
10bdf_coefficients(const unsigned int p, const std::vector<double> &time_steps)
11{
12 // There should be at least p time steps
13 assert(time_steps.size() >= p);
14
15 // Create a time table for the bdf formula
16 Vector<double> times(p + 1);
17 for (unsigned int i = 0; i < p + 1; ++i)
18 {
19 times[i] = 0.;
20 for (unsigned int j = 0; j < i; ++j)
21 times[i] -= time_steps[j];
22 }
23
24 // The alphas are the bdf coefficients
25 Vector<double> alpha(p + 1);
26 alpha = 0;
27
28 for (unsigned int j = 1; j < p + 1; ++j)
29 {
30 double factor = 1.;
31 for (unsigned int i = 1; i < j; ++i)
32 factor *= times[0] - times[i];
33
34 Vector<double> term(p + 1);
35 term.equ(factor, delta(p, 0, j, times));
36 alpha += term;
37 }
38 return alpha;
39}

41Vector<double>
42calculate_bdf_coefficients(
43 const Parameters::SimulationControl::TimeSteppingMethod method,
44 const std::vector<double> &time_steps)
45{
46 switch (method)
47 {
48 case Parameters::SimulationControl::TimeSteppingMethod::bdf1:

50 case Parameters::SimulationControl::TimeSteppingMethod::steady_bdf:
51 return bdf_coefficients(1, time_steps);
52 case Parameters::SimulationControl::TimeSteppingMethod::bdf2:
53 return bdf_coefficients(2, time_steps);
54 case Parameters::SimulationControl::TimeSteppingMethod::bdf3:
55 return bdf_coefficients(3, time_steps);
56 default:
57 throw(std::runtime_error(
58 "BDF coefficients were requested without a BDF method"));
59 }
60}

62Vector<double>
63delta(const unsigned int order,
64 const unsigned int n,
65 const unsigned int j,
66 const Vector<double> &times)
67{
68 if (j == 0)
69 {
70 Vector<double> arr(order + 1);
71 arr = 0.;
72 arr[n] = 1;
73 return arr;
74 }
75
76 // else
77 Vector<double> delta_1 = delta(order, n, j - 1, times);
78 Vector<double> delta_2 = delta(order, n + 1, j - 1, times);
79 Vector<double> delta_sol(order + 1);
80 for (unsigned int i_del = 0; i_del < order + 1; ++i_del)
81 delta_sol[i_del] =
82 (delta_1[i_del] - delta_2[i_del]) / (times[n] - times[n + j]);
83 return delta_sol;
84}
