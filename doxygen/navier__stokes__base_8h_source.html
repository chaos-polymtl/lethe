217 /**
218 * @brief Set the initial condition
219 *
220 * @param initial_condition_type Type of method use to impose initial condition.
221 *
222 * @param restart Indicator if the simulation is being restarted or not.
223 *
224 **/
225
226 virtual void
227 set_initial_condition(
228 Parameters::FluidDynamicsInitialConditionType initial_condition_type,
229 bool restart = false)
230 {
231 unsigned int ref_iter = 0;
232 do
233 {
234 if (ref_iter > 0)
235 this->refine_mesh();
236
237 set_initial_condition_fd(initial_condition_type, restart);
238 if (!restart)
239 {
240 multiphysics->set_initial_conditions();
241 }
242 ref_iter++;
243 }
244 while (
245 ref_iter <
246 (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
247 restart == false);
248
249 if (!restart)
250 {
251 this->postprocess_fd(true);
252 multiphysics->postprocess(true);
253 if (this->simulation_control->is_output_iteration())
254 this->write_output_results(this->present_solution);
255 }

508 /**
509 * @brief Turn regions of the mesh where the @p material_id>0 into a solid
510 * block by injecting velocity and pressure DOFs into the zero constraints.
511 *
512 * It is achieved by imposing \f$\mathbf{u}=0\f$ within the cells which have a
513 * @p material_id>0. In addition, solid cells which are not connected to the
514 * fluid by any means also get a pressure Dirichlet boundary condition which
515 * fixes the pressure to 0. It ensures that the linear system is well-posed.
516 * Right now, this routine only supports the usage of 1 solid domain, but
517 * eventually it could be extended to more than one. By default, the fluid
518 * domain is assumed to have a @p material_id=0 and the rest of the domains
519 * have a @p material_id>0.
520 *
521 * @param[in] non_zero_constraints If this parameter is true, it indicates
522 * that non-zero constraints are being constrained for the solid domain. If
523 * this is set to false, homogeneous constraints are constrained in the solid
524 * domain.
525 */
526 void
527 establish_solid_domain(const bool non_zero_constraints);

529 /**
530 * @brief Checks and identifies if the cell is located in the constraining
531 * domain defined by a plane with its outward-pointing normal vector. The
532 * check is done through a scalar product between a vector formed by a vertex
533 * of the cell to @plane_point and @plane_normal_vector. If one of the
534 * vertices of the cell results in a negative scalar product result, the cell
535 * is excluded from the constrained domain.
536 *
537 * @param[in] cell Pointer to an active cell of the fluid dynamics DoFHandler.
538 *
539 * @param[in] plane_point Coordinates of a point on the restriction plane for
540 * the stasis constraint application domain.
541 *
542 * @param[in] plane_normal_vector Outward-pointing normal vector to define the
543 * restriction plane for the stasis constraint application domain.
544 *
545 * @return Boolean indicating if the cell is in the domain of interest
546 * (@p true) or not (@p false).
547 */
548 inline bool
549 cell_in_constraining_domain(
550 const typename DoFHandler<dim>::active_cell_iterator &cell,
551 const Point<dim> &plane_point,
552 const Tensor<1, dim> &plane_normal_vector)
553 {
554 for (unsigned int v = 0; v < cell->n_vertices(); ++v)
555 {
556 Point<dim> cell_vertex = cell->vertex(v);
557 Tensor<1, dim> cell_vertex_to_plane_point_vector =
558 plane_point - cell_vertex;
559 double scalar_product_result =
560 scalar_product(cell_vertex_to_plane_point_vector,
561 plane_normal_vector);
562 if (scalar_product_result <= 0)
563 return false;
564 }
565 return true;
566 }

645 /**
646 * @brief Constrain velocity DOFs of a solid cell.
647 *
648 * @param[in] non_zero_constraints If this parameter is true, it indicates
649 * that non-zero constraints are applied in the solid domain. If
650 * this is set to false, homogeneous constraints are applied in the solid
651 * domain.
652 *
653 * @param[in] local_dof_indices Vector of a cell's local DOF indices.
654 *
655 * @param[out] zero_constraints Homogeneous constraints holding object.
656 */
657 inline void
658 constrain_solid_cell_velocity_dofs(
659 const bool &non_zero_constraints,
660 const std::vector<types::global_dof_index> &local_dof_indices,
661 AffineConstraints<double> &zero_constraints)
662 {
663 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
664 {
665 const unsigned int component =
666 this->fe->system_to_component_index(i).first;
667 if (component < dim) // velocity DOFs
668 {
669 // We apply a constraint to all DOFs in the solid region, whether
670 // they are locally owned or not.
671 if (non_zero_constraints)
672 {
673 this->nonzero_constraints.add_line(local_dof_indices[i]);
674 this->nonzero_constraints.set_inhomogeneity(
675 local_dof_indices[i], 0);
676 }
677 else
678 zero_constraints.add_line(local_dof_indices[i]);
679 }
680 }

772 const bool &non_zero_constraints,
773 const std::vector<types::global_dof_index> &local_dof_indices,
774 AffineConstraints<double> &zero_constraints)
775 {
776 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
777 {
778 const unsigned int component =
779 this->fe->system_to_component_index(i).first;
780
781 // Only pressure DOFs have an additional Dirichlet condition
782 if (component == dim) // pressure DOFs
783 {
784 // We only apply the constraint on the locally owned pressure DOFs
785 // since we have no way of verifying if the locally relevant DOFs
786 // are connected to a fluid cell.
787 bool dof_is_locally_owned = false;
788
789 // For the GLS-family of solvers, we only have a single index set
790 if constexpr (std::is_same_v<DofsType, IndexSet>)
791 {
792 dof_is_locally_owned =
793 this->locally_owned_dofs.is_element(local_dof_indices[i]);
794 }
795
796 // For the GD-family of solvers, we have two index sets. One for
797 // velocities and one for pressure.
798 if constexpr (std::is_same_v<DofsType, std::vector<IndexSet>>)
799 {
800 dof_is_locally_owned =
801 this->locally_owned_dofs[1].is_element(local_dof_indices[i]);
802 }
803
804 if (dof_is_locally_owned)
805 {
806 if (non_zero_constraints)
807 {
808 this->nonzero_constraints.add_line(local_dof_indices[i]);
809 this->nonzero_constraints.set_inhomogeneity(
810 local_dof_indices[i], 0);
811 }
812 else
813 {
814 zero_constraints.add_line(local_dof_indices[i]);
815 }
816 }
817 }
818 }
819 }
