233 /**
234 * @brief Set the initial condition
235 *
236 * @param initial_condition_type Type of method use to impose initial condition.
237 *
238 * @param restart Indicator if the simulation is being restarted or not.
239 *
240 **/
241
242 virtual void
243 set_initial_condition(
244 const Parameters::FluidDynamicsInitialConditionType initial_condition_type,
245 const bool restart = false)
246 {
247 unsigned int ref_iter = 0;
248 do
249 {
250 if (ref_iter > 0)
251 this->refine_mesh();
252
253 set_initial_condition_fd(initial_condition_type, restart);
254 if (!restart)
255 {
256 multiphysics->set_initial_conditions();
257 }
258 ref_iter++;
259 }
260 while (
261 ref_iter <
262 (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
263 restart == false);
264
265 if (!restart)
266 {
267 this->postprocess_fd(true);
268 multiphysics->postprocess(true);
269 if (this->simulation_control->is_output_iteration())
270 this->write_output_results(*this->present_solution);
271 }
272 }

534 /**
535 * @brief Turn regions of the mesh where the @p material_id>0 into a solid
536 * block by injecting velocity and pressure DOFs into the zero constraints.
537 *
538 * It is achieved by imposing \f$\mathbf{u}=0\f$ within the cells which have a
539 * @p material_id>0. In addition, solid cells which are not connected to the
540 * fluid by any means also get a pressure Dirichlet boundary condition which
541 * fixes the pressure to 0. It ensures that the linear system is well-posed.
542 * Right now, this routine only supports the usage of 1 solid domain, but
543 * eventually it could be extended to more than one. By default, the fluid
544 * domain is assumed to have a @p material_id=0 and the rest of the domains
545 * have a @p material_id>0.
546 *
547 * @param[in] non_zero_constraints If this parameter is true, it indicates
548 * that non-zero constraints are being constrained for the solid domain. If
549 * this is set to false, homogeneous constraints are constrained in the solid
550 * domain.
551 */
552 void
553 establish_solid_domain(const bool non_zero_constraints);

555 /**
556 * @brief Checks and identifies if the cell is located in the constraining
557 * domain defined by a plane with its outward-pointing normal vector. The
558 * check is done through a scalar product between a vector formed by a vertex
559 * of the cell to @plane_point and @plane_normal_vector. If one of the
560 * vertices of the cell results in a negative scalar product result, the cell
561 * is excluded from the constrained domain.
562 *
563 * @param[in] cell Pointer to an active cell of the fluid dynamics DoFHandler.
564 *
565 * @param[in] plane_point Coordinates of a point on the restriction plane for
566 * the stasis constraint application domain.
567 *
568 * @param[in] plane_normal_vector Outward-pointing normal vector to define the
569 * restriction plane for the stasis constraint application domain.
570 *
571 * @return Boolean indicating if the cell is in the domain of interest
572 * (@p true) or not (@p false).
573 */
574 inline bool
575 cell_in_constraining_domain(
576 const typename DoFHandler<dim>::active_cell_iterator &cell,
577 const Point<dim> &plane_point,
578 const Tensor<1, dim> &plane_normal_vector)
579 {
580 for (unsigned int v = 0; v < cell->n_vertices(); ++v)
581 {
582 Point<dim> cell_vertex = cell->vertex(v);
583 Tensor<1, dim> cell_vertex_to_plane_point_vector =
584 plane_point - cell_vertex;
585 double scalar_product_result =
586 scalar_product(cell_vertex_to_plane_point_vector,
587 plane_normal_vector);
588 if (scalar_product_result <= 0)
589 return false;
590 }
591 return true;
592 }

671 /**
672 * @brief Constrain velocity DOFs of a solid cell.
673 *
674 * @param[in] non_zero_constraints If this parameter is true, it indicates
675 * that non-zero constraints are applied in the solid domain. If
676 * this is set to false, homogeneous constraints are applied in the solid
677 * domain.
678 *
679 * @param[in] local_dof_indices Vector of a cell's local DOF indices.
680 *
681 * @param[out] zero_constraints Homogeneous constraints holding object.
682 */
683 inline void
684 constrain_solid_cell_velocity_dofs(
685 const bool &non_zero_constraints,
686 const std::vector<types::global_dof_index> &local_dof_indices,
687 AffineConstraints<double> &zero_constraints)
688 {
689 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
690 {
691 const unsigned int component =
692 this->fe->system_to_component_index(i).first;
693 if (component < dim) // velocity DOFs
694 {
695 // We apply a constraint to all DOFs in the solid region, whether
696 // they are locally owned or not.
697 if (non_zero_constraints)
698 {
699 this->nonzero_constraints.add_line(local_dof_indices[i]);
700 this->nonzero_constraints.set_inhomogeneity(
701 local_dof_indices[i], 0);
702 }
703 else
704 zero_constraints.add_line(local_dof_indices[i]);
705 }
706 }
707 }

783 /**
784 * @brief Constrain pressure DOFs if cells are not connected to fluid and DOFs
785 * are locally owned.
786 *
787 * @param[in] non_zero_constraints If this parameter is true, it indicates
788 * that non-zero constraints are applied in the solid domain. If
789 * this is set to false, homogeneous constraints are applied in the solid
790 * domain.
791 *
792 * @param[in] local_dof_indices Vector of a cell's local DOF indices.
793 *
794 * @param[out] zero_constraints Homogeneous constraints holding object.
795 */
796 inline void
797 constrain_pressure(
798 const bool &non_zero_constraints,
799 const std::vector<types::global_dof_index> &local_dof_indices,
800 AffineConstraints<double> &zero_constraints)
801 {
802 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
803 {
804 const unsigned int component =
805 this->fe->system_to_component_index(i).first;
806
807 // Only pressure DOFs have an additional Dirichlet condition
808 if (component == dim) // pressure DOFs
809 {
810 // We only apply the constraint on the locally owned pressure DOFs
811 // since we have no way of verifying if the locally relevant DOFs
812 // are connected to a fluid cell.
813 bool dof_is_locally_owned = false;
814
815 // For the GLS-family of solvers, we only have a single index set
816 if constexpr (std::is_same_v<DofsType, IndexSet>)
817 {
818 dof_is_locally_owned =
819 this->locally_owned_dofs.is_element(local_dof_indices[i]);
820 }
821
822 // For the GD-family of solvers, we have two index sets. One for
823 // velocities and one for pressure.
824 if constexpr (std::is_same_v<DofsType, std::vector<IndexSet>>)
825 {
826 dof_is_locally_owned =
827 this->locally_owned_dofs[1].is_element(local_dof_indices[i]);
828 }
829
830 if (dof_is_locally_owned)
831 {
832 if (non_zero_constraints)
833 {
834 this->nonzero_constraints.add_line(local_dof_indices[i]);
835 this->nonzero_constraints.set_inhomogeneity(
836 local_dof_indices[i], 0);
837 }
838 else
839 {
840 zero_constraints.add_line(local_dof_indices[i]);
841 }
842 }
843 }
844 }
845 }
