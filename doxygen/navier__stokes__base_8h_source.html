221 /**
222 * @brief set_initial_conditions
223 *
224 * @param initial_condition_type Type of method use to impose initial condition.
225 *
226 * @param restart Indicator if the simulation is being restarted or not.
227 *
228 **/
229
230 virtual void
231 set_initial_condition(Parameters::InitialConditionType initial_condition_type,
232 bool restart = false)
233 {
234 unsigned int ref_iter = 0;
235 do
236 {
237 if (ref_iter > 0)
238 this->refine_mesh();
239
240 set_initial_condition_fd(initial_condition_type, restart);
241 if (!restart)
242 {
243 multiphysics->set_initial_conditions();
244 }
245 ref_iter++;
246 }
247 while (
248 ref_iter <
249 (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
250 restart == false);
251
252 if (!restart)
253 {
254 this->postprocess_fd(true);
255 multiphysics->postprocess(true);
256 if (this->simulation_control->is_output_iteration())
257 this->write_output_results(this->present_solution);
258 }
259 }

452 /**
453 * @brief Turn regions of the mesh where the @p material_id>0 into a solid
454 * block by injecting velocity and pressure DOFs into the zero constraints.
455 *
456 * It is achieved by imposing \f$\mathbf{u}=0\f$ within the cells which have a
457 * @p material_id>0. In addition, solid cells which are not connected to the
458 * fluid by any means also get a pressure Dirichlet boundary condition which
459 * fixes the pressure to 0. It ensures that the linear system is well-posed.
460 * Right now, this routine only supports the usage of 1 solid domain, but
461 * eventually it could be extended to more than one. By default, the fluid
462 * domain is assumed to have a @p material_id=0 and the rest of the domains
463 * have a @p material_id>0.
464 *
465 * @param[in] non_zero_constraints If this parameter is true, it indicates
466 * that non-zero constraints are being constrained for the solid domain. If
467 * this is set to false, homogeneous constraints are constrained in the solid
468 * domain.
469 */
470 void
471 establish_solid_domain(const bool non_zero_constraints);

473 /**
474 * @brief Checks and identifies if the cell is located in the constraining
475 * domain defined by a plane with its outward-pointing normal vector. The
476 * check is done through a scalar product between a vector formed by a vertex
477 * of the cell to @plane_point and @plane_normal_vector. If one of the
478 * vertices of the cell results in a negative scalar product result, the cell
479 * is excluded from the constrained domain.
480 *
481 * @param[in] cell Pointer to an active cell of the fluid dynamics DoFHandler.
482 *
483 * @param[in] plane_point Coordinates of a point on the restriction plane for
484 * the stasis constraint application domain.
485 *
486 * @param[in] plane_normal_vector Outward-pointing normal vector to define the
487 * restriction plane for the stasis constraint application domain.
488 *
489 * @return Boolean indicating if the cell is in the domain of interest
490 * (@p true) or not (@p false).
491 */
492 inline bool
493 cell_in_constraining_domain(
494 const typename DoFHandler<dim>::active_cell_iterator &cell,
495 const Point<dim> &plane_point,
496 const Tensor<1, dim> &plane_normal_vector)
497 {
498 for (unsigned int v = 0; v < cell->n_vertices(); ++v)
499 {
500 Point<dim> cell_vertex = cell->vertex(v);
501 Tensor<1, dim> cell_vertex_to_plane_point_vector =
502 plane_point - cell_vertex;
503 double scalar_product_result =
504 scalar_product(cell_vertex_to_plane_point_vector,
505 plane_normal_vector);
506 if (scalar_product_result <= 0)
507 return false;
508 }
509 return true;
510 }

589 /**
590 * @brief Constrain velocity DOFs of a solid cell.
591 *
592 * @param[in] non_zero_constraints If this parameter is true, it indicates
593 * that non-zero constraints are applied in the solid domain. If
594 * this is set to false, homogeneous constraints are applied in the solid
595 * domain.
596 *
597 * @param[in] local_dof_indices Vector of a cell's local DOF indices.
598 *
599 * @param[out] zero_constraints Homogeneous constraints holding object.
600 */
601 inline void
602 constrain_solid_cell_velocity_dofs(
603 const bool &non_zero_constraints,
604 const std::vector<types::global_dof_index> &local_dof_indices,
605 AffineConstraints<double> &zero_constraints)
606 {
607 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
608 {
609 const unsigned int component =
610 this->fe->system_to_component_index(i).first;
611 if (component < dim) // velocity DOFs
612 {
613 // We apply a constraint to all DOFs in the solid region, whether
614 // they are locally owned or not.
615 if (non_zero_constraints)
616 {
617 this->nonzero_constraints.add_line(local_dof_indices[i]);
618 this->nonzero_constraints.set_inhomogeneity(
619 local_dof_indices[i], 0);
620 }
621 else
622 zero_constraints.add_line(local_dof_indices[i]);
623 }
624 }
625 }

701 /**
702 * @brief Constrain pressure DOFs if cells are not connected to fluid and DOFs
703 * are locally owned.
704 *
705 * @param[in] non_zero_constraints If this parameter is true, it indicates
706 * that non-zero constraints are applied in the solid domain. If
707 * this is set to false, homogeneous constraints are applied in the solid
708 * domain.
709 *
710 * @param[in] local_dof_indices Vector of a cell's local DOF indices.
711 *
712 * @param[out] zero_constraints Homogeneous constraints holding object.
713 */
714 inline void
715 constrain_pressure(
716 const bool &non_zero_constraints,
717 const std::vector<types::global_dof_index> &local_dof_indices,
718 AffineConstraints<double> &zero_constraints)
719 {
720 for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
721 {
722 const unsigned int component =
723 this->fe->system_to_component_index(i).first;
724
725 // Only pressure DOFs have an additional Dirichlet condition
726 if (component == dim) // pressure DOFs
727 {
728 // We only apply the constraint on the locally owned pressure DOFs
729 // since we have no way of verifying if the locally relevant DOFs
730 // are connected to a fluid cell.
731 bool dof_is_locally_owned = false;
732
733 // For the GLS-family of solvers, we only have a single index set
734 if constexpr (std::is_same_v<DofsType, IndexSet>)
735 {
736 dof_is_locally_owned =
737 this->locally_owned_dofs.is_element(local_dof_indices[i]);
738 }
739
740 // For the GD-family of solvers, we have two index sets. One for
741 // velocities and one for pressure.
742 if constexpr (std::is_same_v<DofsType, std::vector<IndexSet>>)
743 {
744 dof_is_locally_owned =
745 this->locally_owned_dofs[1].is_element(local_dof_indices[i]);
746 }
747
748 if (dof_is_locally_owned)
749 {
750 if (non_zero_constraints)
751 {
752 this->nonzero_constraints.add_line(local_dof_indices[i]);
753 this->nonzero_constraints.set_inhomogeneity(
754 local_dof_indices[i], 0);
755 }
756 else
757 {
758 zero_constraints.add_line(local_dof_indices[i]);
759 }
760 }
761 }
762 }
