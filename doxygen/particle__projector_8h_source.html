36/**
37 * @brief Calculate the area of intersection between a circular (2D) particle and a circle.
38 *
39 * @param[in] r_particle Radius of the particle
40 *
41 * @param[in] r_circle Radius of the circle
42 *
43 * @param[in] neighbor_distance Distance between the particle and the circle
44 */
45inline double
46particle_circle_intersection_2d(double r_particle,
47 double r_circle,
48 double neighbor_distance)
49{
50 return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
51 cos((pow(neighbor_distance, 2) +
52 pow(r_particle, 2) - pow(r_circle, 2)) /
53 (2 * neighbor_distance * r_particle))) +
54 Utilities::fixed_power<2, double>(r_circle) *
55 Utilities::fixed_power<-1, double>(
56 cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
57 pow(r_circle, 2)) /
58 (2 * neighbor_distance * r_circle))) -
59 0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
60 (neighbor_distance + r_particle - r_circle) *
61 (neighbor_distance - r_particle + r_circle) *
62 (neighbor_distance + r_particle + r_circle));
63}

65/**
66 * @brief Calculate the volume of intersection between a spherical (3D) particle and a sphere.
67 *
68 * @param[in] r_particle Radius of the particle
69 * @param[in] r_sphere Radius of the sphere
70 * @param[in] neighbor_distance Distance between the particle and the sphere
71 */
72
73inline double
74particle_sphere_intersection_3d(double r_particle,
75 double r_sphere,
76 double neighbor_distance)
77{
78 return M_PI *
79 Utilities::fixed_power<2, double>(r_sphere + r_particle -
80 neighbor_distance) *
81 (Utilities::fixed_power<2, double>(neighbor_distance) +
82 (2 * neighbor_distance * r_particle) -
83 (3 * Utilities::fixed_power<2, double>(r_particle)) +
84 (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
85 (3 * Utilities::fixed_power<2, double>(r_sphere))) /
86 (12 * neighbor_distance);
87}

89/**
90 * @brief ParticleFieldQCM calculator.
91 * This class stores the required information for the calculation of a
92 * field or a particle-fluid interaction onto a mesh. This class does not solve
93 * any equation, but compartmentalize the necessary information for the
94 * projection of the particle field or interaction onto a mesh. Multiple
95 * instances of this function can be called to establish new projections.In the
96 * case of a particle-fluid interaction, there is no particle field to project
97 * and, consequently, the property_start_index parameter is given an invalid
98 * default value (-1).
99 *
100 * @tparam dim An integer that denotes the number of spatial dimensions.
101 * @tparam n_components The number of components in the field. This number should be either 1 (a scalar) or dim (a Tensor<1,dim>)
102 * @tparam property_start_index An integer that indicates at which particle property the field starts.
103 * If the propery_start_index is negative, then there is no property start and
104 * this project is not used to project a particle field.
105 */
106template <int dim, int n_components, int property_start_index = -1>
107class ParticleFieldQCM
108{
109public:
110 /**
111 * @brief Constructor
112 *
113 * @param triangulation The triangulation on which the particles reside
114 * @param fe_degree The finite element degree used to interpolate the field
115 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
116 * @param neumann_boundaries A flag to indicate what to do with cells that do not contain particles.
117 * If the neumann_boundaries is set to true, then cells without particles do
118 * not have a mass matrix assembled. This corresponds to setting a neumann
119 * boundary at the cells which do not contain particles.
120 * @param conservative_projection A flag to indicate if the projection
121 * of the field must be conservative. When a projection is conservative,
122 * the integral of the field over the triangulation will be strictly
123 * equal to the field summed over all particles. This is useful when
124 * projecting a quantity such as the force acting on the particles. When
125 * conservative is put to false, the projection is meant to be a smooth
126 * representation of the particle information and does not conserve it.
127 */
128 ParticleFieldQCM(parallel::DistributedTriangulationBase<dim> *triangulation,
129 const unsigned int fe_degree,
130 const bool simplex,
131 const bool neumann_boundaries,
132 const bool conservative_projection)
133 : dof_handler(*triangulation)
134 , neumann_boundaries(neumann_boundaries)
135 , conservative_projection(conservative_projection)
136 {
137 if (simplex)
138 {
139 const FE_SimplexP<dim> fe_temp(fe_degree);
140 fe = std::make_shared<FESystem<dim>>(fe_temp, n_components);
141 }
142 else
143 {
144 const FE_Q<dim> temp_fe(fe_degree);
145 fe = std::make_shared<FESystem<dim>>(temp_fe, n_components);
146 }
147 }

190/**
191 * @brief Particle information projection.
192 * This class manages the calculation of the projection of the particle
193 * fields onto a triangulation. Its main use is the calculation of the
194 * void fraction which is used as an auxiliary field for the solvers that solve
195 * the Volume-Averaged Navier-Stokes equations. The present architecture
196 * of the class support all of the different void fraction calculation methods
197 * within a single class instead of building a class hierarchy. This is
198 * because there are only a few void fraction calculation strategies.
199 *
200 * Using QCM, this class is also capable of projecting arbitrary fields
201 * belonging to the particle back onto the mesh by instantiating
202 * @ParticleFieldQCM class.
203 *
204 * @tparam dim An integer that denotes the number of spatial dimensions.
205 */
206template <int dim>
207class ParticleProjector : public PhysicsLinearSubequationsSolver
208{
209public:
210 /**
211 * @brief Constructor for the void fraction calculator.
212 *
213 * @param triangulation The triangulation on which the simulation is being done.
214 * @param input_parameters The parameters for the void fraction calculation.
215 * @param linear_solver_parameters The parameters for the linear solver. This is used to solve the L2 projection of the void fraction
216 * @param particle_handler The particle handler used to manage the particles in the simulation. This object must be provided even if the particles are not used to establish the void fraction.
217 * @param fe_degree The finite element degree used to interpolate the void fraction.
218 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
219 * @param pcout The ConditionalOStream used to print the information.
220 */
221 ParticleProjector(
222 parallel::DistributedTriangulationBase<dim> *triangulation,
223 std::shared_ptr<Parameters::VoidFractionParameters<dim>> input_parameters,
224 const Parameters::LinearSolver &linear_solver_parameters,
225 Particles::ParticleHandler<dim> *particle_handler,
226 const unsigned int fe_degree,
227 const bool simplex,
228 const ConditionalOStream &pcout)
229 : PhysicsLinearSubequationsSolver(pcout)
230 , dof_handler(*triangulation)
231 , triangulation(triangulation)
232 , void_fraction_parameters(input_parameters)
233 , linear_solver_parameters(linear_solver_parameters)
234 , particle_handler(particle_handler)
235 , particle_have_been_projected(false)
236 , particle_velocity(triangulation, fe_degree, simplex, true, false)
237 , fluid_force_on_particles_two_way_coupling(triangulation,
238 fe_degree,
239 simplex,
240 false,
241 true)
242 , fluid_drag_on_particles(triangulation, fe_degree, simplex, false, true)
243 , momentum_transfer_coefficient(triangulation,
244 fe_degree,
245 simplex,
246 false,
247 true)
248 {
249 if (simplex)
250 {
251 fe = std::make_shared<FE_SimplexP<dim>>(fe_degree);
252 mapping = std::make_shared<MappingFE<dim>>(*fe);
253 quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
254 }
255 else
256 {
257 // Usual case, for quad/hex meshes
258 fe = std::make_shared<FE_Q<dim>>(fe_degree);
259 mapping = std::make_shared<MappingQ<dim>>(fe->degree);
260 if (this->void_fraction_parameters->quadrature_rule ==
261 Parameters::VoidFractionQuadratureRule::gauss)
262 {
263 if (this->void_fraction_parameters->n_quadrature_points == 0)
264 quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
265 else
266 quadrature = std::make_shared<QGauss<dim>>(
267 this->void_fraction_parameters->n_quadrature_points);
268 }
269 else if (this->void_fraction_parameters->quadrature_rule ==
270 Parameters::VoidFractionQuadratureRule::gauss_lobatto)
271 {
272 if (this->void_fraction_parameters->n_quadrature_points == 0)
273 quadrature = std::make_shared<QGaussLobatto<dim>>(fe->degree + 2);
274 else if (this->void_fraction_parameters->n_quadrature_points >= 3)
275 quadrature = std::make_shared<QGaussLobatto<dim>>(
276 this->void_fraction_parameters->n_quadrature_points);
277 else
278 throw(std::runtime_error(
279 "For void fraction using Gauss-Lobatto ('gauss-lobatto') quadrature rule, the minimum number of quadrature points is 3"));
280 }
281 else
282 {
283 AssertThrow(
284 false,
285 ExcMessage(
286 "An invalid quadrature rule was provided to the ParticleProject class."));
287 }
288 }
289
290 AssertThrow(
291 !void_fraction_parameters->project_particle_velocity ||
292 void_fraction_parameters->l2_smoothing_length > 0,
293 ExcMessage(
294 "The projection of the particle velocity field requires that the l2 smoothing length be > 0. Otherwise, the Poisson equation used in the Neumann boundary is invalid."));
295
296 AssertThrow(
297 void_fraction_parameters->project_particle_velocity == false ||
298 (void_fraction_parameters->project_particle_velocity &&
299 void_fraction_parameters->mode == Parameters::VoidFractionMode::qcm),
300 ExcMessage(
301 "The projection of the particle velocity currently requires that the QCM method be used for the calculation of the void fraction."));
302 }

468 /**
469 * @brief Calculate the characteristic radius of a sphere with a given volume (area in 2D):
470 * R = (2*dim*V/pi)^(1/dim) / 2
471 *
472 * @param volume value for which the equivalent radius of a sphere is calculated.
473 *
474 * @return Radius of a sphere with equivalent volume
475 */
476 static double
477 radius_sphere_equivalent_volume(const double volume)
478 {
479 return 0.5 * pow(2.0 * static_cast<double>(dim) * volume / M_PI,
480 1.0 / static_cast<double>(dim));
481 }

486 * @param cell_measure The measure of the cell in wich QCM is calculated.

488 * @return The QCM radius used in the calculations.

507 /**
508 * @brief Calculate the intersection measure (volume in 3D and area in 2D) between a particle and an hypersphere.
509 * This is used when calculating the intersection volume between the averaging
510 * sphere and a particle in QCM. The calculation inherently assumes that the
511 * radius of the averaging sphere is larger than that of the particle.
512 *
513 * @param r_particle The radius of the particle
514 * @param r_sphere The radius of the averaging sphere
515 * @param distance_between_spheres The distance between the centers of the spheres
516 *
517 * @return The intersection volume (in 3D) or area (in 2d)
518 */
519
520 static double
521 calculate_intersection_measure(double r_particle,
522 double r_sphere,
523 double distance_between_spheres)
524 {
525 // Particle completely in the reference sphere
526 if (distance_between_spheres <= (r_sphere - r_particle))
527 {
528 if constexpr (dim == 2)
529 return M_PI * Utilities::fixed_power<2>(r_particle);
530 if constexpr (dim == 3)
531 return 4. / 3. * M_PI * Utilities::fixed_power<3>(r_particle);
532 }
533
534 // Particle partially in the reference sphere
535 if ((distance_between_spheres > (r_sphere - r_particle)) &&
536 (distance_between_spheres < (r_sphere + r_particle)))
537 {
538 if constexpr (dim == 2)
539 return particle_circle_intersection_2d(r_particle,
540 r_sphere,
541 distance_between_spheres);
542
543 else if constexpr (dim == 3)
544 return particle_sphere_intersection_3d(r_particle,
545 r_sphere,
546 distance_between_spheres);
547 }
548
549 // Particle completely outside the reference
550 // sphere. The intersection volume is zero.
551 return 0;
552 }

592 /**
593 * @brief Calculate and return the periodic offset distance vector of the domain which is needed
594 * for the periodic boundary conditions using the QCM or SPM for void fraction
595 * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
596 * periodic boundaries. This periodic boundary is then used as the reference
597 * for all particle locations shifted by this offset vector.
598 *
599 * @param[in] boundary_id The ID of one of the periodic boundaries
600 *
601 * @return The periodic offset vector.
602 */
603 inline Tensor<1, dim>
604 get_periodic_offset_distance(unsigned int boundary_id) const
605 {
606 Tensor<1, dim> offset;
607
608 // Iterating over the active cells in the triangulation
609 for (const auto &cell : (*this->triangulation).active_cell_iterators())
610 {
611 if (cell->is_locally_owned() || cell->is_ghost())
612 {
613 if (cell->at_boundary())
614 {
615 // Iterating over cell faces
616 for (unsigned int face_id = 0; face_id < cell->n_faces();
617 ++face_id)
618 {
619 unsigned int face_boundary_id =
620 cell->face(face_id)->boundary_id();
621
622 // Check if face is on the boundary, if so, get
623 // the periodic offset distance for one pair of periodic
624 // faces only since periodic boundaries are aligned with the
625 // direction and only axis are currently allowed
626 if (face_boundary_id == boundary_id)
627 {
628 Point<dim> face_center = cell->face(face_id)->center();
629 auto periodic_cell = cell->periodic_neighbor(face_id);
630 unsigned int periodic_face_id =
631 cell->periodic_neighbor_face_no(face_id);
632 Point<dim> periodic_face_center =
633 periodic_cell->face(periodic_face_id)->center();
634
635 offset = periodic_face_center - face_center;
636
637 return offset;
638 }
639 }
640 }
641 }
642 }
643
644 // A zero tensor is returned in case no cells are found on the periodic
645 // boundaries on this processor. This processor won't handle particle in
646 // cells at periodic boundaries, so it won't affect any computation.
647 return offset;
648 }
