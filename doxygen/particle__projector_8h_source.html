35/**
36 * @brief Calculate the area of intersection between a circular (2D) particle and a circle.
37 *
38 * @param[in] r_particle Radius of the particle
39 *
40 * @param[in] r_circle Radius of the circle
41 *
42 * @param[in] neighbor_distance Distance between the particle and the circle
43 */
44inline double
45particle_circle_intersection_2d(double r_particle,
46 double r_circle,
47 double neighbor_distance)
48{
49 return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
50 cos((pow(neighbor_distance, 2) +
51 pow(r_particle, 2) - pow(r_circle, 2)) /
52 (2 * neighbor_distance * r_particle))) +
53 Utilities::fixed_power<2, double>(r_circle) *
54 Utilities::fixed_power<-1, double>(
55 cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
56 pow(r_circle, 2)) /
57 (2 * neighbor_distance * r_circle))) -
58 0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
59 (neighbor_distance + r_particle - r_circle) *
60 (neighbor_distance - r_particle + r_circle) *
61 (neighbor_distance + r_particle + r_circle));
62}

64/**
65 * @brief Calculate the volume of intersection between a spherical (3D) particle and a sphere.
66 *
67 * @param[in] r_particle Radius of the particle
68 *
69 * @param[in] r_sphere Radius of the sphere
70 *
71 * @param[in] neighbor_distance Distance between the particle and the sphere
72 */
73
74inline double
75particle_sphere_intersection_3d(double r_particle,
76 double r_sphere,
77 double neighbor_distance)
78{
79 return M_PI *
80 Utilities::fixed_power<2, double>(r_sphere + r_particle -
81 neighbor_distance) *
82 (Utilities::fixed_power<2, double>(neighbor_distance) +
83 (2 * neighbor_distance * r_particle) -
84 (3 * Utilities::fixed_power<2, double>(r_particle)) +
85 (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
86 (3 * Utilities::fixed_power<2, double>(r_sphere))) /
87 (12 * neighbor_distance);
88}

90/**
91 * @brief ParticleFieldQCM calculator.
92 * This class stores the required information for the calculation of a
93 * field or a particle-fluid interaction onto a mesh. This class does not solve
94 * any equation, but compartmentalize the necessary information for the
95 * projection of the particle field or interaction onto a mesh. Multiple
96 * instances of this function can be called to establish new projections.In the
97 * case of a particle-fluid interaction, there is no particle field to project
98 * and, consequently, the property_start_index parameter is given an invalid
99 * default value (-1).
100 *
101 * @tparam dim An integer that denotes the number of spatial dimensions.
102 *
103 * @tparam n_components The number of components in the field. This number should be either 1 (a scalar) or dim (a Tensor<1,dim>)
104 *
105 * @tparam property_start_index An integer that indicates at which particle property the field starts.
106 * If the propery_start_index is negative, then there is no property start and
107 * this project is not used to project a particle field.
108 */
109template <int dim, int n_components, int property_start_index = -1>
110class ParticleFieldQCM
111{
112public:
113 /**
114 * @brief Constructor
115 *
116 * @param triangulation The triangulation on which the particles reside
117 * @param fe_degree The finite element degree used to interpolate the field
118 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
119 * @param neumann_boundaries A flag to indicate what to do with cells that do not contain particles.
120 * If the neumann_boundaries is set to true, then cells without particles do
121 * not have a mass matrix assembled. This corresponds to setting a neumann
122 * boundary at the cells which do not contain particles.
123 * @param distribute_contribution A flag to indicate if the contribution is meant to be distributed using the volume of the particle divided by the total volume contribution of the particle.
124 */
125 ParticleFieldQCM(parallel::DistributedTriangulationBase<dim> *triangulation,
126 const unsigned int fe_degree,
127 const bool simplex,
128 const bool neumann_boundaries,
129 const bool distribute_contribution)

131 , neumann_boundaries(neumann_boundaries)
132 , distribute_contribution(distribute_contribution)
133 {
134 if (simplex)
135 {
136 if constexpr (n_components == 1)
137 fe = FE_SimplexP<dim>(fe_degree);
138 else
139 {
140 const FE_SimplexP<dim> fe_temp(fe_degree);
141 fe = std::make_shared<FESystem<dim>>(fe_temp, n_components);
142 }
143 }
144 else
145 {
146 if constexpr (n_components == 1)
147 fe = FE_Q<dim>(fe_degree);
148 else
149 {
150 const FE_Q<dim> temp_fe(fe_degree);
151 fe = std::make_shared<FESystem<dim>>(temp_fe, dim);
152 }
153 }
154 }

204/**
205 * @brief Particle information projection.
206 * This class manages the calculation of the projection of the particle
207 * fields onto a triangulation. Its main use is the calculation of the
208 * void fraction which is used as an auxiliary field for the solvers that solve
209 * the Volume-Averaged Navier-Stokes equations. The present architecture
210 * of the class support all of the different void fraction calculation methods
211 * within a single class instead of building a class hierarchy. This is
212 * because there are only a few void fraction calculation strategies.
213 *
214 * Using QCM, this class is also capable of projecting arbitrary fields
215 * belonging to the particle back onto the mesh by instantiating
216 * @ParticleFieldQCM class.
217 *
218 * @tparam dim An integer that denotes the number of spatial dimensions.
219 */
220template <int dim>
221class ParticleProjector : public PhysicsLinearSubequationsSolver
222{
223public:
224 /**
225 * @brief Constructor for the void fraction calculator.
226 *
227 * @param triangulation The triangulation on which the simulation is being done.
228 * @param input_parameters The parameters for the void fraction calculation.
229 * @param linear_solver_parameters The parameters for the linear solver. This is used to solve the L2 projection of the void fraction
230 * @param particle_handler The particle handler used to manage the particles in the simulation. This object must be provided even if the particles are not used to establish the void fraction.
231 * @param fe_degree The finite element degree used to interpolate the void fraction.
232 * @param simplex A flag to indicate if the simulations are being done with simplex elements.
233 * @param pcout The ConditionalOStream used to print the information.
234 */
235 ParticleProjector(
236 parallel::DistributedTriangulationBase<dim> *triangulation,
237 std::shared_ptr<Parameters::VoidFractionParameters<dim>> input_parameters,
238 const Parameters::LinearSolver &linear_solver_parameters,
239 Particles::ParticleHandler<dim> *particle_handler,
240 const unsigned int fe_degree,
241 const bool simplex,
242 const ConditionalOStream &pcout)
243 : PhysicsLinearSubequationsSolver(pcout)
244 , dof_handler(*triangulation)
245 , triangulation(triangulation)
246 , void_fraction_parameters(input_parameters)
247 , linear_solver_parameters(linear_solver_parameters)
248 , particle_handler(particle_handler)
249 , particle_have_been_projected(false)
250 , particle_velocity(triangulation, fe_degree, simplex, true, false)
251 , particle_fluid_force(triangulation, fe_degree, simplex, false, true)
252 {
253 if (simplex)
254 {
255 fe = std::make_shared<FE_SimplexP<dim>>(fe_degree);
256 mapping = std::make_shared<MappingFE<dim>>(*fe);
257 quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
258 }
259 else
260 {
261 // Usual case, for quad/hex meshes
262 fe = std::make_shared<FE_Q<dim>>(fe_degree);
263 mapping = std::make_shared<MappingQ<dim>>(fe->degree);
264 if (this->void_fraction_parameters->quadrature_rule ==
265 Parameters::VoidFractionQuadratureRule::gauss)
266 {
267 if (this->void_fraction_parameters->n_quadrature_points == 0)
268 quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
269 else
270 quadrature = std::make_shared<QGauss<dim>>(
271 this->void_fraction_parameters->n_quadrature_points);
272 }
273 if (this->void_fraction_parameters->quadrature_rule ==
274 Parameters::VoidFractionQuadratureRule::gauss_lobatto)
275 {
276 if (this->void_fraction_parameters->n_quadrature_points == 0)
277 quadrature = std::make_shared<QGaussLobatto<dim>>(fe->degree + 2);
278 else if (this->void_fraction_parameters->n_quadrature_points >= 3)
279 quadrature = std::make_shared<QGaussLobatto<dim>>(
280 this->void_fraction_parameters->n_quadrature_points);
281 else
282 throw(std::runtime_error(
283 "For void fraction using Gauss-Lobatto ('gauss-lobatto') quadrature rule, the minimum number of quadrature points is 3"));
284 }
285 }
286
287 AssertThrow(
288 !void_fraction_parameters->project_particle_velocity ||
289 void_fraction_parameters->l2_smoothing_length > 0,
290 ExcMessage(
291 "The projection of the particle velocity field requires that the l2 smoothing length be > 0. Otherwise, the Poisson equation used in the Neumann boundary is invalid."));
292
293 AssertThrow(
294 void_fraction_parameters->project_particle_velocity == false ||
295 (void_fraction_parameters->project_particle_velocity &&
296 void_fraction_parameters->mode == Parameters::VoidFractionMode::qcm),
297 ExcMessage(
298 "The projection of the particle velocity currently requires that the QCM method be used for the calculation of the void fraction."));
299 }

407 /**
408 * @brief Calculate the characteristic radius of a sphere with a given volume (area in 2D):
409 * R = (2*dim*V/pi)^(1/dim) / 2
410 *
411 * @param volume value for which the equivalent radius of a sphere is calculated.
412 *
413 * @return Radius of a sphere with equivalent volume
414 */
415 static double
416 radius_sphere_equivalent_volume(const double volume)
417 {
418 return 0.5 * pow(2.0 * static_cast<double>(dim) * volume / M_PI,
419 1.0 / static_cast<double>(dim));
420 }

446 /**
447 * @brief Calculate the intersection measure (volume in 3D and area in 2D) between a particle and an hypersphere.
448 * This is used when calculating the intersection volume between the averaging
449 * sphere and a particle in QCM. The calculation inherently assumes that the
450 * radius of the averaging sphere is larger than that of the particle.
451 *
452 * @param r_particle The radius of the particle
453 * @param r_sphere The radius of the averaging sphere
454 * @param distance_between_spheres The distance between the centers of the spheres
455 *
456 * @return The intersection volume (in 3D) or area (in 2d)
457 */
458
459 static double
460 calculate_intersection_measure(double r_particle,
461 double r_sphere,
462 double distance_between_spheres)
463 {
464 // Particle completely in the reference sphere
465 if (distance_between_spheres <= (r_sphere - r_particle))
466 {
467 if constexpr (dim == 2)
468 return M_PI * Utilities::fixed_power<2>(r_particle);
469 if constexpr (dim == 3)
470 return 4. / 3. * M_PI * Utilities::fixed_power<3>(r_particle);
471 }
472
473 // Particle partially in the reference sphere
474 if ((distance_between_spheres > (r_sphere - r_particle)) &&
475 (distance_between_spheres < (r_sphere + r_particle)))
476 {
477 if constexpr (dim == 2)
478 return particle_circle_intersection_2d(r_particle,
479 r_sphere,
480 distance_between_spheres);
481
482 else if constexpr (dim == 3)
483 return particle_sphere_intersection_3d(r_particle,
484 r_sphere,
485 distance_between_spheres);
486 }
487
488 // Particle completely outside the reference
489 // sphere. The intersection volume is zero.
490 return 0;
491 }

547 /**
548 * @brief Calculate and return the periodic offset distance vector of the domain which is needed
549 * for the periodic boundary conditions using the QCM or SPM for void fraction
550 * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
551 * periodic boundaries. This periodic boundary is then used as the reference
552 * for all particle locations shifted by this offset vector.
553 *
554 * @param[in] boundary_id The ID of one of the periodic boundaries
555 *
556 * @return The periodic offset vector.
557 */
558 inline Tensor<1, dim>
559 get_periodic_offset_distance(unsigned int boundary_id) const
560 {
561 Tensor<1, dim> offset;
562
563 // Iterating over the active cells in the triangulation
564 for (const auto &cell : (*this->triangulation).active_cell_iterators())
565 {
566 if (cell->is_locally_owned() || cell->is_ghost())
567 {
568 if (cell->at_boundary())
569 {
570 // Iterating over cell faces
571 for (unsigned int face_id = 0; face_id < cell->n_faces();
572 ++face_id)
573 {
574 unsigned int face_boundary_id =
575 cell->face(face_id)->boundary_id();
576
577 // Check if face is on the boundary, if so, get
578 // the periodic offset distance for one pair of periodic
579 // faces only since periodic boundaries are aligned with the
580 // direction and only axis are currently allowed
581 if (face_boundary_id == boundary_id)
582 {
583 Point<dim> face_center = cell->face(face_id)->center();
584 auto periodic_cell = cell->periodic_neighbor(face_id);
585 unsigned int periodic_face_id =
586 cell->periodic_neighbor_face_no(face_id);
587 Point<dim> periodic_face_center =
588 periodic_cell->face(periodic_face_id)->center();
589
590 offset = periodic_face_center - face_center;
591
592 return offset;
593 }
594 }
595 }
596 }
597 }
598
599 // A zero tensor is returned in case no cells are found on the periodic
600 // boundaries on this processor. This processor won't handle particle in
601 // cells at periodic boundaries, so it won't affect any computation.
602 return offset;
603 }
