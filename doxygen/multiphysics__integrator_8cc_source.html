6template <int dim, typename PropertiesIndex>
7void
8integrate_temperature(Particles::ParticleHandler<dim> &particle_handler,
9 const double dt,
10 std::vector<double> &heat_transfer_rate,
11 const std::vector<double> &heat_source)
12{
13 for (auto particle = particle_handler.begin();
14 particle != particle_handler.end();
15 ++particle)
16 {
17 const types::particle_index particle_id = particle->get_local_index();
18 auto particle_properties = particle->get_properties();
19 double &particle_heat_transfer_rate = heat_transfer_rate[particle_id];
20 double particle_heat_source = heat_source[particle_id];
21 const double mass_inverse =
22 1 / particle_properties[PropertiesIndex::mass];
23 const double specific_heat_inverse =
24 1 / particle_properties[PropertiesIndex::specific_heat];
25
26 // Integration
27 particle_properties[PropertiesIndex::T] +=
28 dt * (particle_heat_transfer_rate + particle_heat_source) *
29 mass_inverse * specific_heat_inverse;
30
31 // Reinitialize heat_transfer_rate
32 particle_heat_transfer_rate = 0;
33 }
34}
