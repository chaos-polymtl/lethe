21/**
22 * @brief Store the information required by the assembly procedure
23 * for a VOF free surface equation. Consequently, this class
24 * calculates the phase values (values, gradients, laplacians) and the shape
25 * method (values, gradients, laplacians) at all the gauss points for all
26 * degrees of freedom and stores it into arrays.
27 * This class serves as a separation between the evaluation at the gauss point
28 * of the variables of interest and their use in the assembly, which is carried
29 * out by the assembler methods.
30 *
31 * @tparam dim An integer that denotes the dimension of the space in which
32 * the flow is solved
33 *
34 * @ingroup solvers
35 **/
36

42 * @brief Constructor. The constructor creates the fe_values that will be used

44 * necessary memory for all member variables. However, it does not do any

46 *
47 * @param simulation_control The SimulationControl object that holds
48 * information related to the control of the steady-state or transient
49 * simulation. This is used to extrapolate velocity solutions in time
50 * for transient simulation.
51 *
52 * @param properties_manager Manager to calculate the physical properties.
53 *
54 * @param fe_vof The FESystem used to solve the VOF equations.
55 *
56 * @param quadrature The quadrature to use for the assembly within the cells.
57 *
58 * @param face_quadrature The quadrature to use for the assembly on faces.
59 *
60 * @param mapping The mapping of the domain in which the Navier-Stokes

135 /** @brief Reinitialize the content of the scratch
136 *
137 * Using the FeValues and the content of the solutions and previous solutions,
138 * fills all of the class member of the scratch
139 *
140 * @tparam VectorType The Vector type used for the solvers
141 *
142 * @param[in] cell The cell over which the assembly is being carried.
143 * This cell must be compatible with the FE which is used to fill the
144 * FeValues.
145 *
146 * @param[in] current_solution The present value of the solution for the VOF.
147 *
148 * @param[in] previous_solutions The solutions at the previous time steps.
149 *
150 */
151
152 template <typename VectorType>
153 void
154 reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
155 const VectorType &current_solution,
156 const std::vector<VectorType> &previous_solutions)
157 {
158 fe_values_vof.reinit(cell);
159 this->quadrature_points = fe_values_vof.get_quadrature_points();
160 auto &fe_vof = fe_values_vof.get_fe();
161
162 // Compute cell diameter
163 double cell_measure =
164 compute_cell_measure_with_JxW(this->fe_values_vof.get_JxW_values());
165 this->cell_size = compute_cell_diameter<dim>(cell_measure, fe_vof.degree);
166
167 fe_values_vof.get_function_values(current_solution,
168 this->present_phase_values);
169 fe_values_vof.get_function_gradients(current_solution,
170 this->phase_gradients);
171 fe_values_vof.get_function_laplacians(current_solution,
172 this->phase_laplacians);
173 fe_values_vof.get_function_gradients(previous_solutions[0],
174 this->previous_phase_gradients);
175
176
177 // Gather previous vof values
178 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
179 {
180 fe_values_vof.get_function_values(previous_solutions[p],
181 this->previous_phase_values[p]);
182 }
183
184 for (unsigned int q = 0; q < this->n_q_points; ++q)
185 {
186 this->JxW[q] = fe_values_vof.JxW(q);
187
188 for (unsigned int k = 0; k < this->n_dofs; ++k)
189 {
190 // Shape function
191 this->phi[q][k] = fe_values_vof.shape_value(k, q);
192 this->grad_phi[q][k] = fe_values_vof.shape_grad(k, q);
193 this->hess_phi[q][k] = fe_values_vof.shape_hessian(k, q);
194 this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
195 }
196 }
197 }

223 const unsigned int &sub_face_no,

236

247 current_solution, values_there);
248
249 fe_interface_values_vof.get_jump_in_function_values(current_solution,
250 phase_value_jump);

253 /** @brief Reinitialize the velocity, calculated by the Fluid Dynamics
254 *
255 * @tparam VectorType The Vector type used for the solvers
256 *
257 * @param cell The cell for which the velocity is reinitialized
258 * This cell must be compatible with the Fluid Dynamics FE
259 *
260 * @param current_solution The present value of the solution for \f$[u,p]\f$
261 *
262 * @param previous_solutions Vector of \f$n\f$ @p VectorType containers of
263 * previous fluid dynamic solutions (\f$[u,p]\f$). \f$n\f$ depends on the BDF
264 * scheme selected for time-stepping.
265 *
266 * @param ale ALE parameters
267 *
268 */
269
270 template <typename VectorType>
271 void
272 reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
273 const VectorType &current_solution,
274 const std::vector<VectorType> &previous_solutions,
275 const Parameters::ALE<dim> &ale)
276 {
277 fe_values_fd.reinit(cell);
278
279 fe_values_fd[velocities_fd].get_function_values(current_solution,
280 velocity_values);
281 fe_values_fd[velocities_fd].get_function_gradients(
282 current_solution, velocity_gradient_values);
283
284 for (unsigned int q = 0; q < this->n_q_points; ++q)
285 {
286 this->velocity_divergences[q] =
287 trace(this->velocity_gradient_values[q]);
288 }
289
290 // Gather previous velocity values
291 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
292 {
293 fe_values_fd[velocities_fd].get_function_values(
294 previous_solutions[p], this->previous_velocity_values[p]);
295 }
296
297 if (ale.enabled())
298 {
299 // ALE enabled, so extract the ALE velocity and subtract it from the
300 // velocity obtained from the fluid dynamics
301 Tensor<1, dim> velocity_ale;
302 std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
303 ale.velocity;
304 Vector<double> velocity_ale_vector(dim);
305
306 for (unsigned int q = 0; q < n_q_points; ++q)
307 {
308 velocity_ale_function->vector_value(quadrature_points[q],
309 velocity_ale_vector);
310 for (int d = 0; d < dim; ++d)
311 velocity_ale[d] = velocity_ale_vector[d];
312
313 velocity_values[q] -= velocity_ale;
314
315 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
316 {
317 this->previous_velocity_values[p][q] -= velocity_ale;
318 }
319 }
320 }
321
322 // Extrapolate velocity to t+dt using the BDF scheme if the simulation is
323 // transient
324 const auto method = this->simulation_control->get_assembly_method();
325 if (is_bdf(method))
326 {
327 // Extrapolate velocity
328 std::vector<double> time_vector =
329 this->simulation_control->get_simulation_times();
330 bdf_extrapolate(time_vector,
331 this->previous_velocity_values,
332 number_of_previous_solutions(method),
333 this->velocity_values);
334 }
335 }

338 /** @brief Reinitialize the content of the scratch regarding the velocity for internal/boundary faces.
339 * The velocity is inherently assumed to have been solved using a CG scheme.
340 *
341 * @param[in] cell The cell over which the assembly is being carried.
342 *
343 * @param[in] face_no The face index associated with the cell
344 *
345 * @param[in] velocity_solution The present value of the velocity solution.
346 */
347 template <typename VectorType>
348 void
349 reinit_face_velocity(
350 const typename DoFHandler<dim>::active_cell_iterator &velocity_cell,
351 const unsigned int &face_no,
352 const VectorType &velocity_solution,
353 const Parameters::ALE<dim> &ale)
354 {
355 fe_face_values_fd.reinit(velocity_cell, face_no);
356
357 // BB note : Array could be pre-allocated
358 face_velocity_values.resize(face_quadrature_points.size());
359
360 fe_face_values_fd[velocities_fd].get_function_values(velocity_solution,
361 face_velocity_values);
362
363 if (!ale.enabled())
364 return;
365
366 // ALE enabled, so extract the ALE velocity and subtract it from the
367 // velocity obtained from the fluid dynamics
368 Tensor<1, dim> velocity_ale;
369 std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
370 ale.velocity;
371 Vector<double> velocity_ale_vector(dim);
372
373 for (unsigned int q = 0; q < face_quadrature_points.size(); ++q)
374 {
375 velocity_ale_function->vector_value(face_quadrature_points[q],
376 velocity_ale_vector);
377 for (int d = 0; d < dim; ++d)
378 velocity_ale[d] = velocity_ale_vector[d];
379
380 face_velocity_values[q] -= velocity_ale;
381 }
382 }

391 // FEValues for the VOF problem
392 FEValues<dim> fe_values_vof;
393 FEInterfaceValues<dim> fe_interface_values_vof;
394 unsigned int n_dofs;
395 unsigned int n_interface_dofs;
396 unsigned int n_q_points;
397 double cell_size;
398
399 // Quadrature
400 Table<1, double> JxW;
401 std::vector<Point<dim>> quadrature_points;
402 std::vector<Point<dim>> face_quadrature_points;
403
404 // VOF values

407 std::vector<Tensor<1, dim>> previous_phase_gradients;

416
417 // Shape functions

419 Table<2, Tensor<1, dim>> grad_phi;
420 Table<2, Tensor<2, dim>> hess_phi;
421 Table<2, double> laplacian_phi;

424 /**

426 */

428 FEFaceValues<dim> fe_face_values_fd;

430 FEValuesExtractors::Vector velocities_fd;

433 std::vector<std::vector<Tensor<1, dim>>> previous_velocity_values;
434 std::vector<Tensor<2, dim>> velocity_gradient_values;
435 std::vector<double> velocity_divergences;
436
437 // Face velocity value for DG
438 std::vector<Tensor<1, dim>> face_velocity_values;
