54/**
55 * @brief Class that assembles the core of the VOF solver.
56 * According to the following weak form:
57 * \f$\mathbf{u} \cdot T \cdot \nabla T =0 \f$ with an SUPG
58 * stabilization
59 *
60 * @tparam dim An integer that denotes the number of spatial dimensions
61 *
62 * @ingroup assemblers
63 */
64template <int dim>
65class VOFAssemblerCore : public VOFAssemblerBase<dim>
66{
67public:
68 VOFAssemblerCore(const std::shared_ptr<SimulationControl> &simulation_control,
69 const Parameters::FEM fem_parameters,
70 const Parameters::VOF vof_parameters)
71 : simulation_control(simulation_control)
72 , fem_parameters(fem_parameters)
73 , vof_parameters(vof_parameters)
74 , compressible(vof_parameters.compressible)
75 {}
76

79 * @param scratch_data (see base class)
80 * @param copy_data (see base class)
81 */
82 virtual void
83 assemble_matrix(const VOFScratchData<dim> &scratch_data,
84 StabilizedMethodsCopyData &copy_data) override;
85
86
87 /**
88 * @brief assemble_rhs Assembles the rhs
89 * @param scratch_data (see base class)
90 * @param copy_data (see base class)
91 */
92 virtual void
93 assemble_rhs(const VOFScratchData<dim> &scratch_data,
94 StabilizedMethodsCopyData &copy_data) override;
95
96 const std::shared_ptr<SimulationControl> simulation_control;
97 const Parameters::FEM fem_parameters;
98 const Parameters::VOF vof_parameters;
99
100 // Controls if the compressibility term is assembled in the VOF equation
101 const bool compressible;
102};

169 * transient simulations.
