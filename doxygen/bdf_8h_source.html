15/**
16 * @brief Calculate the coefficients required for backward differentiation
17 * formula (BDF) integration of @p order \f$n\f$.
18 *
19 * The coefficients are determined through a recursion algorithm.
20 * The algorithm is taken from: Hay, Alexander, et al. "hp-Adaptive time
21 * integration based on the BDF for viscous flows." Journal of Computational
22 * Physics 291 (2015): 151-176.
23 *
24 * @param[in] method Time stepping method.
25 *
26 * @param[in] time_steps Vector containing all the time steps. The time steps
27 * should be in decreasing order.
28 *
29 * For example, if the method is a BDF2 (\f$n=2\f$), it uses three values for
30 * the time: \f$t\f$, \f$t-\Delta t_1\f$ and \f$t-\Delta t_2\f$. Thus, the @p
31 * time_steps vector should contain \f$\Delta t_1\f$ and \f$\Delta t_2\f$.
32 *
33 * @return Vector containing BDF integration coefficient values.
34 *
35 * @note At the moment, the highest implemented scheme is @p bdf3 (@p order=3).
36 */
37Vector<double>
38calculate_bdf_coefficients(
39 Parameters::SimulationControl::TimeSteppingMethod method,
40 const std::vector<double> &time_steps);

43/**
44 * @brief Recursive function for calculating the \f$j^{th}\f$ order divided

48 * The algorithm is taken from: Hay, Alexander, et al. "hp-Adaptive time
49 * integration based on the BDF for viscous flows." Journal of Computational
50 * Physics 291 (2015): 151-176.
51 *
52 * @param[in] order Order of the BDF method.
53 *
54 * @param[in] n Integer required for the recursive function calculation.
55 *
56 * @param[in] j Order of the divided difference of the value of interest.
57 *
58 * @param[in] times Vector of times values corresponding to the various
59 * iteration in decreasing order.
60 *

64 * @return Vector of \f$n\f$ divided differences (\f$\delta^j\f$) used in the

89/**
90 * @brief Get the number of previous time steps for a specific BDF scheme.
91 *
92 * @param[in] method Time stepping method of the simulation.
93 *
94 * @return Number of previous time steps used in a given BDF scheme.
95 */
96inline unsigned int
97number_of_previous_solutions(
98 const Parameters::SimulationControl::TimeSteppingMethod method)
99{
100 if (method == Parameters::SimulationControl::TimeSteppingMethod::bdf1 ||
101 method == Parameters::SimulationControl::TimeSteppingMethod::steady_bdf)
102 return 1;
103 if (method == Parameters::SimulationControl::TimeSteppingMethod::bdf2)
104 return 2;
105 if (method == Parameters::SimulationControl::TimeSteppingMethod::bdf3)
106 return 3;
107
108 return 0;
109}

111/**
112 * @brief Extrapolate vector of solution to time_vector[0] using previous
113 * solution times and previous solutions.
114 *
115 * @tparam DataType Type of the variable being extrapolated (e.g. double,
116 * Tensor<1,dim>)
117 *
118 * @param[in] time_vector Vector of times. The solution will be extrapolated to
119 * time 0. It should be of size @p number_of_previous_solutions+1.
120 *
121 * @param[in] solution_vector Vector of solution. The solution will be
122 * extrapolated to index 0. It should be at least of size @p
123 * number_of_previous_solutions.
124 *
125 * @param[in] number_of_previous_solutions Number of previous solutions used
126 * to extrapolate.
127 *
128 * @param[out] extrapolated_solution Vector of extrapolated solution.
129 */
130
131template <typename DataType>
132void
133bdf_extrapolate(const std::vector<double> &time_vector,
134 const std::vector<std::vector<DataType>> &solution_vector,
135 const unsigned int number_of_previous_solutions,
136 std::vector<DataType> &extrapolated_solution)
137{
138 // If only one previous solution is used, we don't make a Lagrange
139 // polynomial but use the previous value
140 if (number_of_previous_solutions == 1)
141 {
142 for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
143 extrapolated_solution[q] = solution_vector[0][q];
144 return;
145 }
146
147 // Otherwise, we extrapolate with a Lagrange polynomial
148 for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
149 {
150 // Set extrapolated solution to zero
151 extrapolated_solution[q] = 0;
152
153 // For all the previous solutions which we will use to extrapolate
154 for (unsigned int p = 0; p < number_of_previous_solutions; ++p)
155 {
156 // Factor is the weight of the previous solution fixed by the Lagrange
157 // polynomial
158 double factor = 1;
159 for (unsigned int k = 0; k < number_of_previous_solutions; ++k)
160 {
161 if (p != k)
162 {
163 // The time vector also contains the time of the solution to
164 // be extrapolated to, hence the +1
165 factor *= (time_vector[0] - time_vector[k + 1]) /
166 (time_vector[p + 1] - time_vector[k + 1]);
167 }
168 }
169 // Calculate the contribution of this time step to the extrapolated
170 // solution
171 extrapolated_solution[q] += solution_vector[p][q] * factor;
172 }
173 }
174}
