40/**
41 * @brief Default filter applied to phase fraction. The VolumeOfFluidNoFilter
42 * filter does not modify the phase fraction value and returns it as it is.
43 */
44class VolumeOfFluidNoFilter : public VolumeOfFluidFilterBase
45{
46public:
47 /**
48 * @brief Constructor of the transparent filter.

74/**
75 * @brief Filter phase fraction with a hyperbolic tangent function
76 * for VOF simulations.
77 *
78 * The filtered phase is defined as:
79 *
80 * \f$\phi' = 0.5 \tanh(\beta(\phi-0.5)) + 0.5\f$
81 *
82 * where \f$\phi'\f$ is the filtered phase fraction value and \f$\beta\f$
83 * is a model parameter that enables sharper definition when increased.
84 *
85 * The filter allows a sharper definition of the interface and clamps the value
86 * of the phase fraction between 0 and 1.
87 */
88class VolumeOfFluidTanhFilter : public VolumeOfFluidFilterBase
89{
90public:
91 /**
92 * @brief Constructor of the hyperbolic tangent phase fraction filter.
93 *
94 * @param[in] beta Value of the \f$\beta\f$ parameter modulating the
95 * interface sharpness.
96 */
97 VolumeOfFluidTanhFilter(const double beta)
98 : beta(beta)
99 {}
100
101 /**
102 * @brief Computes the value of the filtered phase fraction.
103 *
104 * @param[in] unfiltered_phase Value of the phase fraction before applying the
105 * filter.
106 *
107 * @return Value of the computed phase fraction after applying the filter.
108 */
109 virtual double
110 filter_phase(const double &unfiltered_phase) override
111 {
112 return 0.5 * tanh(beta * (unfiltered_phase - 0.5)) + 0.5;
113 }
114
115private:
116 /**
117 * User-defined parameter that influences the sharpness of the interface
118 * between fluids. A greater value leads to a sharper interface. The
119 * recommended value is \f$\beta=20\f$.
120 */
121 const double beta;
122};
