17template <int dim, int spacedim>
18void
19attach_grid_to_triangulation(Triangulation<dim, spacedim> &triangulation,
20 const Parameters::Mesh &mesh_parameters)
21
22{
23 // GMSH input
24 if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
25 {
26 if (mesh_parameters.simplex)
27 {
28 auto comm = triangulation.get_mpi_communicator();
29 std::string file_name = mesh_parameters.file_name;
30
31 auto construction_data = TriangulationDescription::Utilities::
32 create_description_from_triangulation_in_groups<dim, spacedim>(
33 [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
34 GridIn<dim, spacedim> grid_in;
35 grid_in.attach_triangulation(basetria);
36 std::ifstream input_file(file_name);
37
38 grid_in.read_msh(input_file);
39 },
40 [](dealii::Triangulation<dim, spacedim> &basetria,
41 const MPI_Comm comm,
42 const unsigned int /*group_size*/) {
43#if defined(DEAL_II_WITH_METIS)
44 GridTools::partition_triangulation(
45 Utilities::MPI::n_mpi_processes(comm),
46 basetria,
47 SparsityTools::Partitioner::metis);
48#elif defined(DEAL_II_WITH_ZOLTAN)
49 GridTools::partition_triangulation(
50 Utilities::MPI::n_mpi_processes(comm),
51 basetria,
52 SparsityTools::Partitioner::zoltan);
53#else
54 AssertThrow(
55 false,
56 ExcMessage(
57 "Parallel simulation with simplex meshes require that deal.II be compiled with either Metis or Zoltan"));
58#endif
59 },
60 comm,
61 Utilities::MPI::n_mpi_processes(comm) /* group size */,
62 dealii::Triangulation<dim, spacedim>::none);
63
64 triangulation.create_triangulation(construction_data);
65 GridTools::scale(mesh_parameters.scale, triangulation);
66 }
67 else
68 {
69 GridIn<dim, spacedim> grid_in;
70 grid_in.attach_triangulation(triangulation);
71 std::ifstream input_file(mesh_parameters.file_name);
72 grid_in.read_msh(input_file);
73 GridTools::scale(mesh_parameters.scale, triangulation);
74 }
75 }
76 // Dealii grids
77 else if (mesh_parameters.type == Parameters::Mesh::Type::dealii)
78 {
79 if (mesh_parameters.simplex)
80 {
81 Triangulation<dim, spacedim> temporary_quad_triangulation;
82 GridGenerator::generate_from_name_and_arguments(
83 temporary_quad_triangulation,
84 mesh_parameters.grid_type,
85 mesh_parameters.grid_arguments);
86
87 GridTools::scale(mesh_parameters.scale, temporary_quad_triangulation);
88
89 // initial refinement
90 const unsigned int initial_refinement =
91 mesh_parameters.initial_refinement;
92 temporary_quad_triangulation.refine_global(initial_refinement);
93
94 // flatten the triangulation
95 Triangulation<dim, spacedim> flat_temp_quad_triangulation;
96 GridGenerator::flatten_triangulation(temporary_quad_triangulation,
97 flat_temp_quad_triangulation);
98
99 Triangulation<dim, spacedim> temporary_tri_triangulation(
100 Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
101 GridGenerator::convert_hypercube_to_simplex_mesh(
102 flat_temp_quad_triangulation, temporary_tri_triangulation);
103
104 GridTools::partition_triangulation_zorder(
105 Utilities::MPI::n_mpi_processes(
106 triangulation.get_mpi_communicator()),
107 temporary_tri_triangulation);
108 GridTools::partition_multigrid_levels(temporary_tri_triangulation);
109
110 // extract relevant information from distributed triangulation
111 auto construction_data = TriangulationDescription::Utilities::
112 create_description_from_triangulation(
113 temporary_tri_triangulation,
114 triangulation.get_mpi_communicator(),
115 TriangulationDescription::Settings::
116 construct_multigrid_hierarchy);
117 triangulation.create_triangulation(construction_data);
118 }
119 else
120 {
121 GridGenerator::generate_from_name_and_arguments(
122 triangulation,
123 mesh_parameters.grid_type,
124 mesh_parameters.grid_arguments);
125
126 GridTools::scale(mesh_parameters.scale, triangulation);
127
128 if constexpr (dim == 3)
129 {
130 // Initial mesh translation
131 GridTools::shift(mesh_parameters.translation, triangulation);
132
133 // Initial mesh rotation
134 GridTools::rotate(mesh_parameters.rotation_axis,
135 mesh_parameters.rotation_angle,
136 triangulation);
137 }
138 }
139 }
140 // Customizable cylinder mesh
141 else if (mesh_parameters.type == Parameters::Mesh::Type::cylinder)
142 {
143 if (mesh_parameters.simplex)
144 {
145 throw std::runtime_error(
146 "Unsupported mesh type - custom cylinder mesh with simplex is not supported. Use a dealii cylinder to use simplex mesh.");
147 }
148 else if (dim != 3)
149 {
150 throw std::runtime_error(
151 "Unsupported mesh type - custom cylinder mesh is only supported in 3d.");
152 }
153
154 if constexpr (dim == 3)
155 {
156 // Separate arguments of the string
157 std::vector<std::string> arguments;
158 std::stringstream s_stream(mesh_parameters.grid_arguments);
159 while (s_stream.good())
160 {
161 std::string substr;
162 getline(s_stream, substr, ':');
163 arguments.push_back(substr);
164 }
165
166 // Arguments declaration
167 unsigned int subdivisions;
168 double radius, half_height;
169 if (arguments.size() != 3)
170 {
171 throw std::runtime_error(
172 "Mandatory cylinder parameters are (x subdivisions: radius : half height)");
173 }
174 else
175 {
176 std::vector<double> arguments_double =
177 dealii::Utilities::string_to_double(arguments);
178 subdivisions = static_cast<int>(arguments_double[0]);
179 radius = arguments_double[1];
180 half_height = arguments_double[2];
181 }
182
183 if (mesh_parameters.grid_type == "classic")
184 {
185 // Create a subdivided cylinder from deal.ii
186 GridGenerator::subdivided_cylinder(triangulation,
187 subdivisions,
188 radius,
189 half_height);
190
191 GridTools::scale(mesh_parameters.scale, triangulation);
192 }
193 else
194 {
195 // Create a temporary 2d mesh
196 Triangulation<2, spacedim - 1> temporary_triangulation;
197
198 // Create a spherical manifold for 2d mesh
199 Point<2> center(0.0, 0.0);
200 const SphericalManifold<2, spacedim - 1> m0(center);
201
202 if (mesh_parameters.grid_type == "regularized" ||
203 mesh_parameters.grid_type == "squared")
204 {
205 // Create a square mesh
206 double real_radius = radius * std::sin(M_PI_4);
207 GridGenerator::hyper_cube(temporary_triangulation,
208 -real_radius,
209 real_radius,
210 true);
211
212 // Assign boundary 0 to perimeter as for cylinder
213 for (const auto &cell :
214 temporary_triangulation.active_cell_iterators())
215 {
216 if (cell->is_locally_owned())
217 {
218 // Looping through all the faces of the cell
219 for (const auto &face : cell->face_iterators())
220 {
221 // Check to see if the face is located at boundary
222 if (face->at_boundary())
223 {
224 face->set_boundary_id(0);
225 }
226 }
227 }
228 }
229 }
230 else if (mesh_parameters.grid_type == "balanced")
231 {
232 GridGenerator::hyper_ball_balanced(temporary_triangulation,
233 center,
234 radius);
235 }
236 else
237 {
238 throw std::runtime_error(
239 "Unknown grid type. Choices are <classic|balanced|squared|regularized>.");
240 }
241
242 temporary_triangulation.reset_all_manifolds();
243 temporary_triangulation.set_all_manifold_ids_on_boundary(0);
244 temporary_triangulation.set_manifold(0, m0);
245
246 if (mesh_parameters.grid_type == "regularized")
247 {
248 // Pre-refinement to reduce mesh size at corners before
249 // regularization
250 temporary_triangulation.refine_global(2);
251 GridTools::regularize_corner_cells(temporary_triangulation);
252
253 // Flatten the triangulation
254 Triangulation<2, spacedim - 1> flat_temporary_triangulation;
255 flat_temporary_triangulation.copy_triangulation(
256 temporary_triangulation);
257 temporary_triangulation.clear();
258 GridGenerator::flatten_triangulation(
259 flat_temporary_triangulation, temporary_triangulation);
260 }
261
262 // Extrude the 2d temporary mesh to 3d cylinder
263 GridGenerator::extrude_triangulation(temporary_triangulation,
264 subdivisions + 1,
265 2.0 * half_height,
266 triangulation,
267 true);
268
269 // Rotate mesh in x-axis and set the (0,0,0) at the barycenter
270 // to be comparable to dealii cylinder meshes
271 Tensor<1, spacedim> axis_vector({0.0, 1.0, 0.0});
272 GridTools::rotate(axis_vector, M_PI_2, triangulation);
273 Tensor<1, spacedim> shift_vector({-half_height, 0.0, 0.0});
274 GridTools::shift(shift_vector, triangulation);
275
276 // Force the manifold id to be zero in the case of the balanced
277 // cylinder
278 if (mesh_parameters.grid_type == "balanced")
279 triangulation.reset_manifold(1);
280
281 // Add a cylindrical manifold on the final unrefined mesh
282 const CylindricalManifold<3, spacedim> m1(0);
283 triangulation.set_manifold(0, m1);
284
285 GridTools::scale(mesh_parameters.scale, triangulation);
286 }
287 }
288 }
289
290 // Periodic Hills grid
291 else if (mesh_parameters.type == Parameters::Mesh::Type::periodic_hills &&
292 !mesh_parameters.simplex)
293 {
294 if (mesh_parameters.simplex)
295 {
296 throw std::runtime_error(
297 "Unsupported mesh type - periodic hills mesh with simplex is not supported");
298 }
299 else
300 {
301 PeriodicHillsGrid<dim, spacedim> grid(mesh_parameters.grid_arguments);
302 grid.make_grid(triangulation);
303
304 GridTools::scale(mesh_parameters.scale, triangulation);
305 }
306 }
307 else
308 throw std::runtime_error(
309 "Unsupported mesh type - mesh will not be created");
310}

313template <int dim, int spacedim>
314void
315setup_periodic_boundary_conditions(
316 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
317 const BoundaryConditions::BoundaryConditions &boundary_conditions)
318
319{
320 // Setup periodic boundary conditions
321 for (auto const &[id, type] : boundary_conditions.type)
322 {
323 if (type == BoundaryConditions::BoundaryType::periodic)
324 {
325 auto triangulation_ptr = &triangulation;
326
327 std::vector<GridTools::PeriodicFacePair<
328 typename Triangulation<dim, spacedim>::cell_iterator>>
329 periodicity_vector;
330 GridTools::collect_periodic_faces(
331 *dynamic_cast<Triangulation<dim, spacedim> *>(triangulation_ptr),
332 id,
333 boundary_conditions.periodic_neighbor_id.at(id),
334 boundary_conditions.periodic_direction.at(id),
335 periodicity_vector);
336 triangulation.add_periodicity(periodicity_vector);
337 }
338 }
339}

341template <int dim, int spacedim>
342void
343read_mesh_and_manifolds(
344 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
345 const Parameters::Mesh &mesh_parameters,
346 const Parameters::Manifolds &manifolds_parameters,
347 const bool restart,
348 const BoundaryConditions::BoundaryConditions &boundary_conditions)
349{
350 attach_grid_to_triangulation(triangulation, mesh_parameters);
351
352 setup_periodic_boundary_conditions(triangulation, boundary_conditions);
353
354 // If the mesh is a GMSH mesh, we need to manually set the manifold id
355 // of the face to be that of the boundary
356 // ID, otherwise, we will be unable to use external manifold.
357 // This is done manually by looping through all faces and giving them a
358 // manifold id if there is a manifold associated with this number.
359 if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
360 {
361 // Gather all the manifold ids within a set
362 std::set<int> manifold_ids;
363 for (unsigned int i = 0; i < manifolds_parameters.size; ++i)
364 manifold_ids.insert(manifolds_parameters.id[i]);
365
366 // Reset all the manifolds manually and force them to zero
367 triangulation.reset_all_manifolds();
368
369 // If the parameter file forces the occurrence of manifold,
370 // loop over the faces of the triangulation. If the face of the
371 // triangulation has a boundary id which corresponds to a manifold id
372 // identified within the parameter file, then fix the manifold id of this
373 // face manually to be that of the boundary id. In the past, this was done
374 // by default for every face, but since 2023-12 this throws (rightfully)
375 // an error in deal.II
376 if (manifolds_parameters.size > 0)
377 {
378 for (const auto &face : triangulation.active_face_iterators())
379 {
380 if (face->at_boundary() &&
381 manifold_ids.contains(face->boundary_id()))
382 face->set_all_manifold_ids(face->boundary_id());
383 }
384 }
385 }
386
387 // Finally attach the manifolds to the triangulation
388 // Right now this should only occur for GMSH mesh, but the function is generic
389 // enough.
390 attach_manifolds_to_triangulation(triangulation, manifolds_parameters);
391
392 if (mesh_parameters.simplex)
393 {
394 // Refinement isn't possible yet
395 }
396 else
397 {
398 if (mesh_parameters.refine_until_target_size)
399 {
400 double minimal_cell_size =
401 GridTools::minimal_cell_diameter(triangulation);
402 double target_size = mesh_parameters.target_size;
403 unsigned int number_refinement = static_cast<unsigned int>(floor(
404 std::log(minimal_cell_size / target_size) / std::numbers::ln2));
405 triangulation.refine_global(number_refinement);
406 }
407 else if (!restart)
408 {
409 const int initial_refinement = mesh_parameters.initial_refinement;
410 triangulation.refine_global(initial_refinement);
411 refine_triangulation_at_boundaries(
412 mesh_parameters.boundaries_to_refine,
413 mesh_parameters.initial_refinement_at_boundaries,
414 triangulation);
415 }
416 }
417}

419template <int dim, int spacedim>
420void
421read_mesh_and_manifolds_for_stator_and_rotor(
422 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
423 const Parameters::Mesh &mesh_parameters,
424 const Parameters::Manifolds &manifolds_parameters,
425 const bool restart,
426 const BoundaryConditions::BoundaryConditions &boundary_conditions,
427 const Parameters::Mortar<dim> &mortar_parameters)
428{
429 // First check if stator and rotor meshes are of the same type
430 AssertThrow(
431 mesh_parameters.type == mortar_parameters.rotor_mesh->type,
432 ExcMessage(
433 "The mesh types for the rotor and stator geometries must be the same."));
434
435 // Faces at rotor-stator interface
436 unsigned int n_faces_rotor_interface = 0;
437 unsigned int n_faces_stator_interface = 0;
438
439 // Since the rotor and stator meshes are read separately, a dummy
440 // triangulation is created for each part of the domain and then merged
441
442 // Stator triangulation
443 Triangulation<dim> stator_temp_tria;
444 attach_grid_to_triangulation(stator_temp_tria, mesh_parameters);
445
446 // Rotor triangulation
447 Triangulation<dim> rotor_temp_tria;
448 attach_grid_to_triangulation(rotor_temp_tria, *mortar_parameters.rotor_mesh);
449
450 if (mesh_parameters.type == Parameters::Mesh::Type::dealii)
451 {
452 // Get stator manifold ids without flat id
453 unsigned int stator_ids_no_flat = 0;
454 for (const auto &id : stator_temp_tria.get_manifold_ids())
455 {
456 if (id != numbers::flat_manifold_id)
457 stator_ids_no_flat++;
458 }
459
460 // Get rotor manifold ids without flat id
461 unsigned int rotor_ids_no_flat = 0;
462 for (const auto &id : rotor_temp_tria.get_manifold_ids())
463 {
464 if (id != numbers::flat_manifold_id)
465 rotor_ids_no_flat++;
466 }
467
468 // Shift rotor boundary IDs #
469 unsigned int n_boundary_ids_stator =
470 stator_temp_tria.get_boundary_ids().size();
471 // Check number of faces at rotor interface with stator
472 for (const auto &face : rotor_temp_tria.active_face_iterators())
473 {
474 if (face->at_boundary())
475 {
476 face->set_boundary_id(face->boundary_id() +
477 n_boundary_ids_stator);
478 if (face->boundary_id() == mortar_parameters.rotor_boundary_id)
479 n_faces_rotor_interface++;
480 }
481 }
482
483 // Keep track of modified IDs in faces and lines
484 std::vector<bool> changed_manifold_ids_faces(
485 rotor_temp_tria.n_active_faces(), false);
486 std::vector<bool> changed_manifold_ids_lines(
487 rotor_temp_tria.n_active_lines(), false);
488
489 // Shift rotor manifold IDs #
490 for (const auto &cell : rotor_temp_tria.active_cell_iterators())
491 {
492 if (cell->manifold_id() != numbers::flat_manifold_id)
493 cell->set_manifold_id(cell->manifold_id() + stator_ids_no_flat);
494 for (const auto &face : cell->face_iterators())
495 {
496 if (face->manifold_id() != numbers::flat_manifold_id &&
497 !changed_manifold_ids_faces[face->index()])
498 {
499 face->set_manifold_id(face->manifold_id() +
500 stator_ids_no_flat);
501 changed_manifold_ids_faces[face->index()] = true;
502 }
503 if constexpr (dim == 3)
504 {
505 for (const auto line_index : cell->line_indices())
506 {
507 if (cell->line(line_index)->manifold_id() !=
508 numbers::flat_manifold_id &&
509 !changed_manifold_ids_lines[line_index])
510 cell->line(line_index)
511 ->set_manifold_id(
512 cell->line(line_index)->manifold_id() +
513 stator_ids_no_flat);
514 changed_manifold_ids_lines[line_index] = true;
515 }
516 }
517 }
518 }
519
520 // Check number of faces at stator interface with rotor
521 for (const auto &face : stator_temp_tria.active_face_iterators())
522 {
523 if (face->at_boundary())
524 {
525 if (face->boundary_id() == mortar_parameters.stator_boundary_id)
526 n_faces_stator_interface++;
527 }
528 }
529
530 // Store rotor manifolds in shifted IDs #
531 for (unsigned int m = 0; m < rotor_temp_tria.get_manifold_ids().size();
532 m++)
533 {
534 unsigned int temp = rotor_temp_tria.get_manifold_ids()[m];
535 if (temp != numbers::flat_manifold_id)
536 rotor_temp_tria.set_manifold(m + stator_ids_no_flat,
537 rotor_temp_tria.get_manifold(m));
538 }
539
540 // Merge triangulations
541 GridGenerator::merge_triangulations(
542 stator_temp_tria, rotor_temp_tria, triangulation, 0.0, true, true);
543
544 // Attach manifolds to merged triangulation
545 unsigned int n = 0;
546 for (unsigned int i = 0; i < stator_ids_no_flat; i++)
547 {
548 triangulation.set_manifold(n, stator_temp_tria.get_manifold(i));
549 n++;
550 }
551 for (unsigned int j = stator_ids_no_flat;
552 j < stator_ids_no_flat + rotor_ids_no_flat;
553 j++)
554 {
555 triangulation.set_manifold(n, rotor_temp_tria.get_manifold(j));
556 n++;
557 }
558 }
559 else if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
560 {
561 // Merge triangulations
562 GridGenerator::merge_triangulations(
563 stator_temp_tria, rotor_temp_tria, triangulation, 0.0, true, true);
564
565 // Check number of faces at the rotor-stator interface
566 for (const auto &face : triangulation.active_face_iterators())
567 {
568 if (face->at_boundary())
569 {
570 if (face->boundary_id() == mortar_parameters.rotor_boundary_id)
571 n_faces_rotor_interface++;
572 if (face->boundary_id() == mortar_parameters.stator_boundary_id)
573 n_faces_stator_interface++;
574 }
575 }
576
577 // Gather all the manifold ids within a set
578 std::set<int> manifold_ids;
579 for (unsigned int i = 0; i < manifolds_parameters.size; ++i)
580 manifold_ids.insert(manifolds_parameters.id[i]);
581
582 // Reset all the manifolds manually and force them to zero
583 triangulation.reset_all_manifolds();
584
585 // Set manifolds
586 if (manifolds_parameters.size > 0)
587 {
588 for (const auto &face : triangulation.active_face_iterators())
589 {
590 if (face->at_boundary() &&
591 manifold_ids.contains(face->boundary_id()))
592 face->set_all_manifold_ids(face->boundary_id());
593 }
594 }
595
596 // Attach manifolds to rotor-stator triangulation
597 attach_manifolds_to_triangulation(triangulation, manifolds_parameters);
598 }
599 else
600 throw std::runtime_error(
601 "Unsupported mesh type - mesh will not be created");
602
603 AssertThrow(
604 n_faces_rotor_interface == n_faces_stator_interface,
605 ExcMessage(
606 "The number of faces at the rotor interface ID #" +
607 std::to_string(mortar_parameters.rotor_boundary_id) +
608 " is different from the number of faces at the stator interface ID #" +
609 std::to_string(mortar_parameters.stator_boundary_id) + "."));
610
611 // Setup boundary conditions
612 setup_periodic_boundary_conditions(triangulation, boundary_conditions);
613
614 // Initial mesh refinement
615 if (mesh_parameters.simplex)
616 {
617 // Refinement isn't possible yet
618 }
619 else
620 {
621 if (mesh_parameters.refine_until_target_size)
622 {
623 const double minimal_cell_size =
624 GridTools::minimal_cell_diameter(triangulation);
625 const double target_size = mesh_parameters.target_size;
626 const unsigned int number_refinement =
627 static_cast<unsigned int>(floor(
628 std::log(minimal_cell_size / target_size) / std::numbers::ln2));
629 triangulation.refine_global(number_refinement);
630 }
631 else if (!restart)
632 {
633 const int initial_refinement = mesh_parameters.initial_refinement;
634 triangulation.refine_global(initial_refinement);
635 refine_triangulation_at_boundaries(
636 mesh_parameters.boundaries_to_refine,
637 mesh_parameters.initial_refinement_at_boundaries,
638 triangulation);
639 }
640 }
641}
