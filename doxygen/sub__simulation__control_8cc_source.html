7SubSimulationControlDEM::SubSimulationControlDEM(
8 const DEMSubIterationLogic iteration_logic,
9 const double time_interval,
10 const unsigned int coupling_frequency,
11 const double rayleigh_characteristic_time,
12 const double fraction_of_rayleigh_characteristic_time)
13 : iteration_logic(iteration_logic)
14 , time_interval(time_interval)
15 , coupling_frequency(coupling_frequency)
16 , rayleigh_characteristic_time(rayleigh_characteristic_time)
17 , fraction_of_rayleigh_characteristic_time(
18 fraction_of_rayleigh_characteristic_time)
19 , iteration_number(0)
20{
21 Assert(time_interval > 0,
22 ExcMessage("The time interval must be strictly positive"));
23
24
25 if (iteration_logic == DEMSubIterationLogic::fixed_number_of_iterations)
26 {
27 Assert(
28 coupling_frequency > 0,
29 ExcMessage(
30 "The coupling frequency must be positive. A value of zero or a negative value was encountered. The simulation will stop."));
31
32 // A fixed number of iterations is requested, the time step is fixed
33 // using the coupling frequency and the time interval.
34 time_step = time_interval / coupling_frequency;
35 total_number_of_iterations = coupling_frequency;
36 }
37 else //(iteration_logic ==
38 // DEMSubIterationLogic::fixed_fraction_of_rayleigh_time_step)
39 {
40 Assert(rayleigh_characteristic_time > 0,
41 ExcMessage(
42 "The Rayleigh characteristic time must be strictly positive"));
43 Assert(
44 fraction_of_rayleigh_characteristic_time > 0,
45 ExcMessage(
46 "The fraction of the Rayleigh characteristic time must be strictly positive"));
47
48 // A fixed fraction of the Rayleigh time step is used. The number of
49 // iterations will be manually calculated
50 time_step =
51 fraction_of_rayleigh_characteristic_time * rayleigh_characteristic_time;
52
53 // We calculate the total number of iterations with the proposed time step
54 // We then take the ceiling of it to have a integer number of time steps
55 total_number_of_iterations =
56 static_cast<unsigned int>(std::ceil(time_interval / time_step));
57
58 // Since the total number of iterations needs to be an integer, the actual
59 // value of the time step may have changed as a result of the previous
60 // calculations. We then recalculate the time step.
61 time_step = time_interval / double(total_number_of_iterations);
62
63 Assert(
64 total_number_of_iterations > 0,
65 ExcMessage(
66 "The total number of iterations for the DEM sub simulation control is below 0."
67 " Something went wrong with either the fraction of the Rayleigh characteristic time or the Rayleigh characteristic time provided."));
68 }
69}
