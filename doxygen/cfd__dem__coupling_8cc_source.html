33template <int dim>
34void
35CFDDEMSolver<dim>::dem_setup_parameters()
36{
37 coupling_frequency =
38 this->cfd_dem_simulation_parameters.cfd_dem.coupling_frequency;
39
40 dem_action_manager = DEMActionManager::get_action_manager();
41
42 // Initialize DEM Parameters
43 dem_parameters.lagrangian_physical_properties =
44 this->cfd_dem_simulation_parameters.dem_parameters
45 .lagrangian_physical_properties;
46 g = dem_parameters.lagrangian_physical_properties.g;
47 dem_parameters.boundary_conditions =
48 this->cfd_dem_simulation_parameters.dem_parameters.boundary_conditions;
49 dem_parameters.insertion_info =
50 this->cfd_dem_simulation_parameters.dem_parameters.insertion_info;
51 dem_parameters.floating_walls =
52 this->cfd_dem_simulation_parameters.dem_parameters.floating_walls;
53 dem_parameters.model_parameters =
54 this->cfd_dem_simulation_parameters.dem_parameters.model_parameters;
55 dem_parameters.simulation_control =
56 this->cfd_dem_simulation_parameters.dem_parameters.simulation_control;
57 dem_parameters.post_processing =
58 this->cfd_dem_simulation_parameters.dem_parameters.post_processing;
59 dem_parameters.mesh = this->cfd_dem_simulation_parameters.dem_parameters.mesh;
60 dem_parameters.restart =
61 this->cfd_dem_simulation_parameters.dem_parameters.restart;
62 size_distribution_object_container.resize(
63 dem_parameters.lagrangian_physical_properties.particle_type_number);
64
65 const auto parallel_triangulation =
66 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
67 &*this->triangulation);
68
69 // Setup load balancing parameters
70 load_balancing.set_parameters(dem_parameters.model_parameters);
71 load_balancing.copy_references(this->simulation_control,
72 *parallel_triangulation,
73 this->particle_handler,
74 sparse_contacts_object);
75
76 // Attach the correct functions to the signals inside the triangulation
77 load_balancing.connect_weight_signals();
78
79 if (dem_parameters.model_parameters.sparse_particle_contacts)
80 {
81 dem_action_manager->set_sparse_contacts_enabled();
82 sparse_contacts_object.set_parameters(
83 dem_parameters.model_parameters.granular_temperature_threshold,
84 dem_parameters.model_parameters.solid_fraction_threshold,
85 dem_parameters.model_parameters.advect_particles);
86 }
87
88 setup_distribution_type();
89
90 dem_time_step =
91 this->simulation_control->get_time_step() / coupling_frequency;
92
93 // Calculate the Rayleigh critical time step ratio
94 double rayleigh_time_step = DBL_MAX;
95
96 for (unsigned int i = 0;
97 i < dem_parameters.lagrangian_physical_properties.particle_type_number;
98 ++i)
99 {
100 double youngs_modulus = dem_parameters.lagrangian_physical_properties
101 .youngs_modulus_particle[i];
102 double poisson_ratio =
103 dem_parameters.lagrangian_physical_properties.poisson_ratio_particle[i];
104 double density =
105 dem_parameters.lagrangian_physical_properties.density_particle[i];
106
107 double shear_modulus = youngs_modulus / (2.0 * (1.0 + poisson_ratio));
108
109 double min_diameter =
110 size_distribution_object_container.at(i)->find_min_diameter();
111
112 rayleigh_time_step =
113 std::min(M_PI_2 * min_diameter * sqrt(density / shear_modulus) /
114 (0.1631 * poisson_ratio + 0.8766),
115 rayleigh_time_step);
116 }
117
118 const double time_step_rayleigh_ratio = dem_time_step / rayleigh_time_step;
119 this->pcout << "DEM time-step is " << time_step_rayleigh_ratio * 100
120 << "% of Rayleigh time step" << std::endl;
121
122 // Check if there are periodic boundaries
123 for (unsigned int i_bc = 0;
124 i_bc < dem_parameters.boundary_conditions.bc_types.size();
125 ++i_bc)
126 {
127 if (dem_parameters.boundary_conditions.bc_types[i_bc] ==
128 Parameters::Lagrangian::BCDEM::BoundaryType::periodic)
129 {
130 dem_action_manager->set_periodic_boundaries_enabled();
131
132 periodic_boundaries_object.set_periodic_boundaries_information(
133 dem_parameters.boundary_conditions.periodic_boundary_0,
134 dem_parameters.boundary_conditions.periodic_direction);
135
136 break;
137 }
138 }
139
140 insertion_object = set_insertion_type();
141 // Initialize the total contact list counter
142 integrator_object = set_integrator_type();
143 particle_particle_contact_force_object =
144 set_particle_particle_contact_force_model<
145 dim,
146 DEM::CFDDEMProperties::PropertiesIndex>(
147 this->cfd_dem_simulation_parameters.dem_parameters);
148
149 // Initialize the contact search counter
150 contact_search_total_number = 0;
151}

153template <int dim>
154void
155CFDDEMSolver<dim>::setup_distribution_type()
156{
157 // Use namespace and alias to make the code more readable
158 using namespace Parameters::Lagrangian;
159 LagrangianPhysicalProperties &lpp =
160 dem_parameters.lagrangian_physical_properties;
161
162 maximum_particle_diameter = 0;
163 for (unsigned int particle_type = 0; particle_type < lpp.particle_type_number;
164 particle_type++)
165 {
166 switch (lpp.distribution_type.at(particle_type))
167 {
168 case SizeDistributionType::uniform:
169 size_distribution_object_container[particle_type] =
170 std::make_shared<UniformDistribution>(
171 lpp.particle_average_diameter.at(particle_type));
172 break;
173 case SizeDistributionType::normal:
174 size_distribution_object_container[particle_type] =
175 std::make_shared<NormalDistribution>(
176 lpp.particle_average_diameter.at(particle_type),
177 lpp.particle_size_std.at(particle_type),
178 lpp.seed_for_distributions[particle_type] + this_mpi_process);
179 break;
180 case SizeDistributionType::custom:
181 size_distribution_object_container[particle_type] =
182 std::make_shared<CustomDistribution>(
183 lpp.particle_custom_diameter.at(particle_type),
184 lpp.particle_custom_probability.at(particle_type),
185 lpp.seed_for_distributions[particle_type] + this_mpi_process);
186 break;
187 }
188
189 maximum_particle_diameter = std::max(
190 maximum_particle_diameter,
191 size_distribution_object_container[particle_type]->find_max_diameter());
192 }
193
194 neighborhood_threshold_squared =
195 std::pow(dem_parameters.model_parameters.neighborhood_threshold *
196 maximum_particle_diameter,
197 2);
198}

200template <int dim>
201std::shared_ptr<Insertion<dim, DEM::CFDDEMProperties::PropertiesIndex>>
202CFDDEMSolver<dim>::set_insertion_type()
203{
204 using namespace Parameters::Lagrangian;
205 typename InsertionInfo<dim>::InsertionMethod insertion_method =
206 dem_parameters.insertion_info.insertion_method;
207
208 const auto parallel_triangulation =
209 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
210 &*this->triangulation);
211
212 switch (insertion_method)
213 {
214 case InsertionInfo<dim>::InsertionMethod::file:
215 {
216 return std::make_shared<
217 InsertionFile<dim, DEM::CFDDEMProperties::PropertiesIndex>>(
218 size_distribution_object_container,
219 *parallel_triangulation,
220 dem_parameters);
221 }
222 case InsertionInfo<dim>::InsertionMethod::list:
223 {
224 return std::make_shared<
225 InsertionList<dim, DEM::CFDDEMProperties::PropertiesIndex>>(
226 size_distribution_object_container,
227 *parallel_triangulation,
228 dem_parameters);
229 }
230 case InsertionInfo<dim>::InsertionMethod::plane:
231 {
232 return std::make_shared<
233 InsertionPlane<dim, DEM::CFDDEMProperties::PropertiesIndex>>(
234 size_distribution_object_container,
235 *parallel_triangulation,
236 dem_parameters);
237 }
238 case InsertionInfo<dim>::InsertionMethod::volume:
239 {
240 return std::make_shared<
241 InsertionVolume<dim, DEM::CFDDEMProperties::PropertiesIndex>>(
242 size_distribution_object_container,
243 *parallel_triangulation,
244 dem_parameters,
245 maximum_particle_diameter);
246 }
247 default:
248 throw(std::runtime_error("Invalid insertion method."));
249 }

275template <int dim>
276void
277CFDDEMSolver<dim>::initialize_dem_parameters()
278{
279 this->pcout << "Initializing DEM parameters" << std::endl;
280
281 const auto parallel_triangulation =
282 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
283 &*this->triangulation);
284
285 // Set up the local and ghost cells (if ASC enabled)
286 sparse_contacts_object.update_local_and_ghost_cell_set(
287 this->particle_projector.dof_handler);
288
289 particle_wall_contact_force_object = set_particle_wall_contact_force_model<
290 dim,
291 DEM::CFDDEMProperties::PropertiesIndex>(
292 this->cfd_dem_simulation_parameters.dem_parameters);
293
294 // Finding the smallest contact search frequency criterion between (smallest
295 // cell size - largest particle radius) and (security factor * (blob diameter
296 // - 1) * largest particle radius). This value is used in
297 // find_contact_detection_frequency function
298 smallest_contact_search_criterion =
299 std::min((GridTools::minimal_cell_diameter(*this->triangulation) -
300 maximum_particle_diameter * 0.5),
301 (dem_parameters.model_parameters.dynamic_contact_search_factor *
302 (dem_parameters.model_parameters.neighborhood_threshold - 1) *
303 maximum_particle_diameter * 0.5));
304
305 // Remap periodic cells (if PBC enabled)
306 periodic_boundaries_object.map_periodic_cells(
307 *parallel_triangulation, periodic_boundaries_cells_information);
308
309 // Set the periodic offset to contact managers and particles contact forces
310 // for periodic contact detection (if PBC enabled)
311 contact_manager.set_periodic_offset(
312 periodic_boundaries_object.get_periodic_offset_distance());
313 particle_particle_contact_force_object->set_periodic_offset(
314 periodic_boundaries_object.get_periodic_offset_distance());
315
316 // Find cell neighbors
317 contact_manager.execute_cell_neighbors_search(
318 *parallel_triangulation, periodic_boundaries_cells_information);
319
320 // Find boundary cells with faces
321 boundary_cell_object.build(
322 *parallel_triangulation,
323 dem_parameters.floating_walls,
324 dem_parameters.boundary_conditions.outlet_boundaries,
325 this->cfd_dem_simulation_parameters.cfd_parameters.mesh
326 .check_for_diamond_cells,
327 this->cfd_dem_simulation_parameters.cfd_parameters.mesh
328 .expand_particle_wall_contact_search,
329 this->pcout);
330
331 sort_particles_into_subdomains_and_cells();
332
333 // Remap periodic nodes after setup of dofs (If ASC and PBC)
334 if (dem_action_manager->check_periodic_boundaries_enabled() &&
335 dem_action_manager->check_sparse_contacts_enabled())
336 {
337 sparse_contacts_object.map_periodic_nodes(
338 this->particle_projector.void_fraction_constraints);
339 }
340
341 this->pcout << "Finished initializing DEM parameters" << std::endl
342 << "DEM time-step is " << dem_time_step << " s" << std::endl;
343}
344template <int dim>
345void
346CFDDEMSolver<dim>::read_dem()
347{
348 this->pcout << "Reading DEM checkpoint" << std::endl;
349
350 std::string prefix =
351 this->cfd_dem_simulation_parameters.void_fraction->dem_file_name;
352
353 // Load checkpoint controller
354 std::string checkpoint_controller_object_filename =
355 prefix + ".checkpoint_controller";
356 std::ifstream iss_checkpoint_controller_obj(
357 checkpoint_controller_object_filename);
358 boost::archive::text_iarchive ia_checkpoint_controller_obj(
359 iss_checkpoint_controller_obj, boost::archive::no_header);
360
361 unsigned int checkpoint_id;
362 ia_checkpoint_controller_obj >> checkpoint_id;
363
364 // New prefix for the remaining files
365 prefix = prefix + "_" + Utilities::int_to_string(checkpoint_id);
366
367 // Gather particle serialization information
368 std::string particle_filename = prefix + ".particles";
369 std::ifstream input(particle_filename.c_str());
370 AssertThrow(input, ExcFileNotOpen(particle_filename));
371
372 std::string buffer;
373 std::getline(input, buffer);
374 std::istringstream iss(buffer);
375 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
376
377 // Create a temporary particle_handler with DEM properties
378 Particles::ParticleHandler<dim> temporary_particle_handler(
379 *this->triangulation,
380 this->particle_mapping,
381 DEM::DEMProperties::n_properties);
382
383 ia >> temporary_particle_handler;
384
385 const std::string filename = prefix + ".triangulation";
386 std::ifstream in(filename.c_str());
387 if (!in)
388 AssertThrow(false,
389 ExcMessage(
390 std::string(
391 "You are trying to restart a previous computation, "
392 "but the restart file <") +
393 filename + "> does not appear to exist!"));
394
395 if (auto parallel_triangulation =
396 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
397 &*this->triangulation))
398 {
399 try
400 {
401 parallel_triangulation->load(filename.c_str());
402
403 // Deserialize particles have the triangulation has been read
404 temporary_particle_handler.deserialize();
405 }
406 catch (...)
407 {
408 AssertThrow(false,
409 ExcMessage("Cannot open snapshot mesh file or read the"
410 "triangulation stored there."));
411 }
412
413 // Fill the existing particle handler using the temporary one
414 // This is done during the dynamic cast for the convert_particle_handler
415 // function which requires a pararallel::distributed::triangulation
416 convert_particle_handler<dim,
417 DEM::DEMProperties::PropertiesIndex,
418 DEM::CFDDEMProperties::PropertiesIndex>(
419 *parallel_triangulation,
420 temporary_particle_handler,
421 this->particle_handler);
422 }
423 else
424 {
425 throw std::runtime_error(
426 "VANS equations currently do not support triangulations other than parallel::distributed");
427 }
428
429 this->pcout << "Finished reading DEM checkpoint" << std::endl
430 << this->particle_handler.n_global_particles()
431 << " particles are in the simulation" << std::endl;
432
433 write_dem_output_results();
434}

438CFDDEMSolver<dim>::write_checkpoint()
439{
440 TimerOutput::Scope timer(this->computing_timer, "Write checkpoint");
441
442 std::string prefix =
443 this->simulation_parameters.simulation_control.output_folder +
444 this->simulation_parameters.restart_parameters.filename;
445 std::string prefix_particles = prefix + "_particles";
446 std::string prefix_grid = prefix + "_lagrangian_postprocessing";
447 if (Utilities::MPI::this_mpi_process(this->mpi_communicator) == 0)
448 {
449 this->simulation_control->save(prefix);
450 this->pvdhandler.save(prefix);
451 particles_pvdhandler.save(prefix_particles);
452
453 if (this->dem_parameters.post_processing

455 grid_pvdhandler.save(prefix_grid);
456
457 if (this->simulation_parameters.flow_control.enable_flow_control)
458 this->flow_control.save(prefix);
459 }
460
461 std::ostringstream oss;
462 boost::archive::text_oarchive oa(oss, boost::archive::no_header);
463 oa << this->particle_handler;
464
465 // Write additional particle information for deserialization
466 std::string particle_filename = prefix + ".particles";
467 std::ofstream output(particle_filename.c_str());
468 output << oss.str() << std::endl;
469
470 std::vector<const GlobalVectorType *> sol_set_transfer;
471 sol_set_transfer.push_back(&this->present_solution);
472 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
473 {
474 sol_set_transfer.push_back(&this->previous_solutions[i]);
475 }
476
477 if (this->simulation_parameters.post_processing.calculate_average_velocities)
478 {
479 std::vector<const GlobalVectorType *> av_set_transfer =
480 this->average_velocities->save(prefix);
481
482 // Insert average velocities vectors into the set transfer vector
483 sol_set_transfer.insert(sol_set_transfer.end(),
484 av_set_transfer.begin(),
485 av_set_transfer.end());
486 }
487
488 // Prepare for Serialization
489 SolutionTransfer<dim, GlobalVectorType> system_trans_vectors(
490 this->dof_handler);
491 system_trans_vectors.prepare_for_serialization(sol_set_transfer);
492
493 // Prepare particle handler for serialization
494 this->particle_handler.prepare_for_serialization();
495
496 // Void Fraction
497 std::vector<const GlobalVectorType *> vf_set_transfer;
498 vf_set_transfer.push_back(
499 &this->particle_projector.void_fraction_locally_relevant);
500 for (unsigned int i = 0;
501 i < this->particle_projector.previous_void_fraction.size();
502 ++i)
503 {
504 vf_set_transfer.push_back(
505 &this->particle_projector.previous_void_fraction[i]);
506 }
507
508 this->multiphysics->write_checkpoint();
509
510 // Prepare for Serialization
511 SolutionTransfer<dim, GlobalVectorType> vf_system_trans_vectors(
512 this->particle_projector.dof_handler);
513 vf_system_trans_vectors.prepare_for_serialization(vf_set_transfer);
514
515 if (auto parallel_triangulation =
516 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
517 &*this->triangulation))
518 {
519 std::string triangulationName = prefix + ".triangulation";
520 parallel_triangulation->save(prefix + ".triangulation");
521 }

524template <int dim>
525void
526CFDDEMSolver<dim>::read_checkpoint()
527{
528 TimerOutput::Scope timer(this->computing_timer, "Read checkpoint");
529 std::string prefix =
530 this->simulation_parameters.simulation_control.output_folder +
531 this->simulation_parameters.restart_parameters.filename;
532 std::string prefix_particles = prefix + "_particles";
533 std::string prefix_grid = prefix + "_lagrangian_postprocessing";
534
535 this->simulation_control->read(prefix);
536 this->pvdhandler.read(prefix);
537 particles_pvdhandler.read(prefix_particles);
538
539 if (this->dem_parameters.post_processing.lagrangian_post_processing_enabled)
540 {
541 grid_pvdhandler.read(prefix_grid);
542 }
543
544 // Gather particle serialization information
545 std::string particle_filename = prefix + ".particles";
546 std::ifstream input(particle_filename.c_str());
547 AssertThrow(input, ExcFileNotOpen(particle_filename));
548
549 std::string buffer;
550 std::getline(input, buffer);
551 std::istringstream iss(buffer);
552 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
553
554 ia >> this->particle_handler;
555
556 const std::string filename = prefix + ".triangulation";
557 std::ifstream in(filename.c_str());
558 if (!in)
559 AssertThrow(false,
560 ExcMessage(
561 std::string(
562 "You are trying to restart a previous computation, "
563 "but the restart file <") +
564 filename + "> does not appear to exist!"));
565
566 try
567 {
568 if (auto parallel_triangulation =
569 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
570 this->triangulation.get()))
571 parallel_triangulation->load(filename.c_str());
572 }
573 catch (...)
574 {
575 AssertThrow(false,
576 ExcMessage("Cannot open snapshot mesh file or read the "
577 "triangulation stored there."));
578 }
579
580 this->setup_dofs();
581
582 // TODO BB
583 // Remap periodic nodes after setup of dofs
584 if (dem_action_manager->check_periodic_boundaries_enabled() &&
585 dem_action_manager->check_sparse_contacts_enabled())
586 {
587 sparse_contacts_object.map_periodic_nodes(
588 this->particle_projector.void_fraction_constraints);
589 }
590
591 // Velocity Vectors
592 std::vector<GlobalVectorType *> x_system(1 + this->previous_solutions.size());
593
594 GlobalVectorType distributed_system(this->locally_owned_dofs,
595 this->mpi_communicator);
596
597 x_system[0] = &(distributed_system);
598
599 std::vector<GlobalVectorType> distributed_previous_solutions;
600
601 distributed_previous_solutions.reserve(this->previous_solutions.size());
602
603 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
604 {
605 distributed_previous_solutions.emplace_back(
606 GlobalVectorType(this->locally_owned_dofs, this->mpi_communicator));
607 x_system[i + 1] = &distributed_previous_solutions[i];
608 }
609
610 SolutionTransfer<dim, GlobalVectorType> system_trans_vectors(
611 this->dof_handler);
612
613 if (this->simulation_parameters.post_processing.calculate_average_velocities)
614 {
615 std::vector<GlobalVectorType *> sum_vectors =
616 this->average_velocities->read(prefix);
617
618 x_system.insert(x_system.end(), sum_vectors.begin(), sum_vectors.end());
619 }
620
621 system_trans_vectors.deserialize(x_system);
622
623 this->present_solution = distributed_system;
624 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
625 {
626 this->previous_solutions[i] = distributed_previous_solutions[i];
627 }
628
629 // Void Fraction Vectors
630 std::vector<GlobalVectorType *> vf_system(
631 1 + this->particle_projector.previous_void_fraction.size());
632
633 GlobalVectorType vf_distributed_system(
634 this->particle_projector.locally_owned_dofs, this->mpi_communicator);
635
636 vf_system[0] = &(vf_distributed_system);
637
638 std::vector<GlobalVectorType> vf_distributed_previous_solutions;
639
640 vf_distributed_previous_solutions.reserve(
641 this->particle_projector.previous_void_fraction.size());
642
643 for (unsigned int i = 0;
644 i < this->particle_projector.previous_void_fraction.size();
645 ++i)
646 {
647 vf_distributed_previous_solutions.emplace_back(
648 GlobalVectorType(this->particle_projector.locally_owned_dofs,
649 this->mpi_communicator));
650 vf_system[i + 1] = &vf_distributed_previous_solutions[i];
651 }
652
653 SolutionTransfer<dim, GlobalVectorType> vf_system_trans_vectors(
654 this->particle_projector.dof_handler);
655
656 vf_system_trans_vectors.deserialize(vf_system);
657
658 this->particle_projector.void_fraction_locally_relevant =
659 vf_distributed_system;
660 for (unsigned int i = 0;
661 i < this->particle_projector.previous_void_fraction.size();
662 ++i)
663 {
664 this->particle_projector.previous_void_fraction[i] =
665 vf_distributed_previous_solutions[i];
666 }
667
668 if (this->simulation_parameters.flow_control.enable_flow_control)
669 {
670 this->flow_control.read(prefix);
671 }
672
673 this->multiphysics->read_checkpoint();
674
675 // Deserialize particles have the triangulation has been read
676 this->particle_handler.deserialize();
677}

679template <int dim>
680void
681CFDDEMSolver<dim>::check_contact_detection_method(unsigned int counter)
682{
683 // Use namespace and alias to make the code more readable
684 using namespace Parameters::Lagrangian;
685 Parameters::Lagrangian::ModelParameters<dim> &model_parameters =
686 dem_parameters.model_parameters;
687
688 switch (model_parameters.contact_detection_method)
689 {
690 case ModelParameters<dim>::ContactDetectionMethod::constant:
691 {
692 if ((counter % model_parameters.contact_detection_frequency) == 0)
693 dem_action_manager->contact_detection_step();
694 break;
695 }
696 case ModelParameters<dim>::ContactDetectionMethod::dynamic:
697 {
698 double dt =
699 this->simulation_control->get_time_step() /
700 this->cfd_dem_simulation_parameters.cfd_dem.coupling_frequency;
701
702 find_particle_contact_detection_step<
703 dim,
704 DEM::CFDDEMProperties::PropertiesIndex>(
705 this->particle_handler,
706 dt,
707 smallest_contact_search_criterion,
708 this->mpi_communicator,
709 displacement);
710 break;
711 }
712 default:

714 }

717template <int dim>
718void
719CFDDEMSolver<dim>::load_balance()
720{
721 load_balancing.check_load_balance_iteration();
722
723 // If not a load balance iteration, exit the function
724 if (!dem_action_manager->check_load_balance())
725 return;
726
727 std::vector<const GlobalVectorType *> sol_set_transfer;
728 sol_set_transfer.push_back(&this->present_solution);
729 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
730 {
731 sol_set_transfer.push_back(&this->previous_solutions[i]);
732 }
733
734 // Prepare for Serialization
735 SolutionTransfer<dim, GlobalVectorType> system_trans_vectors(
736 this->dof_handler);
737 system_trans_vectors.prepare_for_coarsening_and_refinement(sol_set_transfer);
738
739 // Void Fraction
740 std::vector<const GlobalVectorType *> vf_set_transfer;
741 vf_set_transfer.push_back(
742 &this->particle_projector.void_fraction_locally_relevant);
743 for (unsigned int i = 0;
744 i < this->particle_projector.previous_void_fraction.size();
745 ++i)
746 {
747 vf_set_transfer.push_back(
748 &this->particle_projector.previous_void_fraction[i]);
749 }
750
751 // Prepare for Serialization
752 SolutionTransfer<dim, GlobalVectorType> vf_system_trans_vectors(
753 this->particle_projector.dof_handler);
754 vf_system_trans_vectors.prepare_for_coarsening_and_refinement(
755 vf_set_transfer);
756
757 // Prepare particle handle for serialization
758 this->particle_handler.prepare_for_coarsening_and_refinement();
759
760 this->pcout << "-->Repartitionning triangulation" << std::endl;
761
762 const auto parallel_triangulation =
763 dynamic_cast<parallel::distributed::Triangulation<dim> *>(
764 &*this->triangulation);
765
766 parallel_triangulation->repartition();
767
768 // If PBC are enabled remap periodic cells
769 periodic_boundaries_object.map_periodic_cells(
770 *parallel_triangulation, periodic_boundaries_cells_information);
771
772 // Update cell neighbors
773 contact_manager.update_cell_neighbors(*parallel_triangulation,
774 periodic_boundaries_cells_information);
775
776 boundary_cell_object.build(
777 *parallel_triangulation,
778 dem_parameters.floating_walls,
779 dem_parameters.boundary_conditions.outlet_boundaries,
780 this->cfd_dem_simulation_parameters.cfd_parameters.mesh
781 .check_for_diamond_cells,
782 this->cfd_dem_simulation_parameters.cfd_parameters.mesh
783 .expand_particle_wall_contact_search,
784 this->pcout);
785
786 const auto average_minimum_maximum_cells =
787 Utilities::MPI::min_max_avg(parallel_triangulation->n_active_cells(),
788 this->mpi_communicator);
789
790 const auto average_minimum_maximum_particles = Utilities::MPI::min_max_avg(
791 this->particle_handler.n_locally_owned_particles(), this->mpi_communicator);
792
793 this->pcout << "Load balance finished" << std::endl;
794 this->pcout
795 << "Average, minimum and maximum number of particles on the processors are "
796 << average_minimum_maximum_particles.avg << " , "
797 << average_minimum_maximum_particles.min << " and "
798 << average_minimum_maximum_particles.max << std::endl;
799 this->pcout
800 << "Minimum and maximum number of cells owned by the processors are "
801 << average_minimum_maximum_cells.min << " and "
802 << average_minimum_maximum_cells.max << std::endl;
803
804 this->pcout << "Setup DOFs" << std::endl;
805 this->setup_dofs();
806
807 // TODO BB
808 // Remap periodic nodes after setup of dofs
809 if (dem_action_manager->check_periodic_boundaries_enabled() &&
810 dem_action_manager->check_sparse_contacts_enabled())
811 {
812 sparse_contacts_object.map_periodic_nodes(
813 this->particle_projector.void_fraction_constraints);
814 }
815
816 // Update the local and ghost cells (if ASC enabled)
817 sparse_contacts_object.update_local_and_ghost_cell_set(
818 this->particle_projector.dof_handler);
819
820 // Velocity Vectors
821 std::vector<GlobalVectorType *> x_system(1 + this->previous_solutions.size());
822
823 GlobalVectorType distributed_system(this->locally_owned_dofs,
824 this->mpi_communicator);
825
826 x_system[0] = &(distributed_system);
827
828 std::vector<GlobalVectorType> distributed_previous_solutions;
829
830 distributed_previous_solutions.reserve(this->previous_solutions.size());
831
832 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
833 {
834 distributed_previous_solutions.emplace_back(
835 GlobalVectorType(this->locally_owned_dofs, this->mpi_communicator));
836 x_system[i + 1] = &distributed_previous_solutions[i];
837 }
838
839 system_trans_vectors.interpolate(x_system);
840
841 this->present_solution = distributed_system;
842 for (unsigned int i = 0; i < this->previous_solutions.size(); ++i)
843 {
844 this->previous_solutions[i] = distributed_previous_solutions[i];
845 }
846
847 x_system.clear();
848
849 // Void Fraction Vectors
850 std::vector<GlobalVectorType *> vf_system(
851 1 + this->particle_projector.previous_void_fraction.size());
852
853 GlobalVectorType vf_distributed_system(
854 this->particle_projector.locally_owned_dofs, this->mpi_communicator);
855
856 vf_system[0] = &(vf_distributed_system);
857
858 std::vector<GlobalVectorType> vf_distributed_previous_solutions;
859
860 vf_distributed_previous_solutions.reserve(
861 this->particle_projector.previous_void_fraction.size());
862
863 for (unsigned int i = 0;
864 i < this->particle_projector.previous_void_fraction.size();
865 ++i)
866 {
867 vf_distributed_previous_solutions.emplace_back(
868 GlobalVectorType(this->particle_projector.locally_owned_dofs,
869 this->mpi_communicator));
870 vf_system[i + 1] = &vf_distributed_previous_solutions[i];
871 }
872
873 vf_system_trans_vectors.interpolate(vf_system);
874
875 this->particle_projector.void_fraction_locally_relevant =
876 vf_distributed_system;
877 for (unsigned int i = 0;
878 i < this->particle_projector.previous_void_fraction.size();
879 ++i)
880 {
881 this->particle_projector.previous_void_fraction[i] =
882 vf_distributed_previous_solutions[i];
883 }
884
885 vf_system.clear();
886
887 // Unpack particle handler after load balancing step
888 this->particle_handler.unpack_after_coarsening_and_refinement();
889
890 // Regenerate vertex to cell map
891 this->vertices_cell_mapping();
892}

1333template <int dim>
1334inline void
1335CFDDEMSolver<dim>::sort_particles_into_subdomains_and_cells()
1336{
1337 this->particle_handler.sort_particles_into_subdomains_and_cells();
1338
1339 // Exchange ghost particles
1340 this->particle_handler.exchange_ghost_particles(true);
1341
1342 // Resize the displacement, force and torque containers only if the particles
1343 // have changed subdomains
1344 if (dem_action_manager->check_resize_containers())
1345 {
1346 unsigned int number_of_particles =
1347 this->particle_handler.get_max_local_particle_index();
1348 // Resize displacement container
1349 displacement.resize(number_of_particles);
1350 // Resize outcome containers
1351 contact_outcome.resize_interaction_containers(number_of_particles);
1352 MOI.resize(number_of_particles);
1353
1354 // Updating moment of inertia container
1355 for (auto &particle : this->particle_handler)
1356 {

1375{

1377 // broad and fine searches, sort_particles_into_subdomains_and_cells, and
1378 // exchange_ghost
1379 dem_contact_build(counter);

1381 // Particle-particle contact force

1383 contact_manager.get_local_adjacent_particles(),

1388 dem_time_step,
1389 contact_outcome);

1400 add_fluid_particle_interaction_torque();

1402 // The cell average velocities and accelerations are updated
1403 // from the fully computed forces at step 0, but we do not use the
1404 // mobility status to disabled contacts at this step. The reason

1406 // DEM time step (from the last CFD time step) are the force of
1407 // the fluid may have significantly changed the particulate

1425 dynamic_cast<parallel::distributed::Triangulation<dim> *>(

1434 sparse_contacts_object);
1435 }

1448 collision_event_log);
