10template <int dim>
11void
12CLSSubequationsInterface<dim>::initialize_subequations(
13 const SimulationParameters<dim> &p_simulation_parameters,
14 std::shared_ptr<parallel::DistributedTriangulationBase<dim>> p_triangulation,
15 const std::shared_ptr<SimulationControl> &p_simulation_control)
16{
17 // Reset all subequations
18 this->active_subequations.clear();
19 this->subequations.clear();
20
21 // Activate and add relevant subequations
22 if ((p_simulation_parameters.multiphysics.cls_parameters.surface_tension_force
23 .enable) ||
24 (p_simulation_parameters.multiphysics.cls_parameters
25 .reinitialization_method.pde_based_interface_reinitialization.enable))
26 {
27 // Phase gradient projection
28 this->active_subequations.push_back(
29 CLSSubequationsID::phase_gradient_projection);
30 this->subequations[CLSSubequationsID::phase_gradient_projection] =
31 std::make_shared<CLSPhaseGradientProjection<dim>>(
32 p_simulation_parameters, this->pcout, p_triangulation, *this);
33 // Phase curvature projection
34 this->active_subequations.push_back(
35 CLSSubequationsID::curvature_projection);
36 this->subequations[CLSSubequationsID::curvature_projection] =
37 std::make_shared<CLSCurvatureProjection<dim>>(p_simulation_parameters,
38 this->pcout,
39 p_triangulation,
40 *this);
41
42 if (p_simulation_parameters.multiphysics.cls_parameters
43 .reinitialization_method.pde_based_interface_reinitialization
44 .enable)
45 {
46 // PDE-based interface reinitialization
47 this->active_subequations.push_back(
48 CLSSubequationsID::pde_based_interface_reinitialization);
49 this->subequations
50 [CLSSubequationsID::pde_based_interface_reinitialization] =
51 std::make_shared<CLSPDEBasedInterfaceReinitialization<dim>>(
52 p_simulation_parameters,
53 this->pcout,
54 p_triangulation,
55 p_simulation_control,
56 *this);
57 }
58 }
59 reset_subequations_solutions_validity();
60}
