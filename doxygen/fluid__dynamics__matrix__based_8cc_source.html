56template <int dim>
57void
58FluidDynamicsMatrixBased<dim>::setup_dofs_fd()
59{
60 TimerOutput::Scope t(this->computing_timer, "Setup DOFs");
61
62 // Clear the preconditioner before the matrix they are associated with is
63 // cleared
64 amg_preconditioner.reset();
65 ilu_preconditioner.reset();
66 current_preconditioner_fill_level = initial_preconditioner_fill_level;
67
68
69 // Now reset system matrix
70 system_matrix.clear();
71
72 this->dof_handler.distribute_dofs(*this->fe);
73 DoFRenumbering::Cuthill_McKee(this->dof_handler);
74
75 this->locally_owned_dofs = this->dof_handler.locally_owned_dofs();
76 this->locally_relevant_dofs =
77 DoFTools::extract_locally_relevant_dofs(this->dof_handler);
78
79 // If enabled, rotate rotor mapping
80 this->rotate_rotor_mapping(false);
81
82 // Non Zero constraints
83 this->define_non_zero_constraints();
84
85 // Zero constraints
86 this->define_zero_constraints();
87
88 // If enabled, create mortar operators
89 this->reinit_mortar_operators();
90
91 // Operations on the following vectors (addition, multiplication, etc.) can
92 // only be done if these are reinitialized WITHOUT locally_relevant_dofs. This
93 // is why most of them are reinitialized both with and without
94 // locally_relevant_dofs.
95 this->present_solution.reinit(this->locally_owned_dofs,
96 this->locally_relevant_dofs,
97 this->mpi_communicator);
98 this->local_evaluation_point.reinit(this->locally_owned_dofs,
99 this->mpi_communicator);
100 this->evaluation_point.reinit(this->locally_owned_dofs,
101 this->locally_relevant_dofs,
102 this->mpi_communicator);
103
104 // Initialize vector of previous solutions
105 for (auto &solution : this->previous_solutions)
106 {
107 solution.reinit(this->locally_owned_dofs,
108 this->locally_relevant_dofs,
109 this->mpi_communicator);
110 }
111
112 if (this->simulation_control->is_sdirk())
113 {
114 // Reinitialize vectors used for the SDIRK methods
115 this->sdirk_vectors.sum_bi_ki.reinit(this->locally_owned_dofs,
116 this->locally_relevant_dofs,
117 this->mpi_communicator);
118 this->sdirk_vectors.local_sum_bi_ki.reinit(this->locally_owned_dofs,
119 this->mpi_communicator);
120 this->sdirk_vectors.sum_over_previous_stages.reinit(
121 this->locally_owned_dofs,
122 this->locally_relevant_dofs,
123 this->mpi_communicator);
124 this->sdirk_vectors.local_sum_over_previous_stages.reinit(
125 this->locally_owned_dofs, this->mpi_communicator);
126 this->sdirk_vectors.locally_owned_for_calculation.reinit(
127 this->locally_owned_dofs, this->mpi_communicator);
128
129 for (auto &solution : this->sdirk_vectors.previous_k_j_solutions)
130 {
131 solution.reinit(this->locally_owned_dofs,
132 this->locally_relevant_dofs,
133 this->mpi_communicator);
134 }
135 }
136
137 this->newton_update.reinit(this->locally_owned_dofs, this->mpi_communicator);
138 this->system_rhs.reinit(this->locally_owned_dofs, this->mpi_communicator);
139
140
141 auto &nonzero_constraints = this->get_nonzero_constraints();
142 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
143 DoFTools::make_sparsity_pattern(this->dof_handler,
144 dsp,
145 nonzero_constraints,
146 false);
147
148 // Add sparsity pattern entries
149 if (this->simulation_parameters.mortar_parameters.enable)
150 this->mortar_coupling_operator->add_sparsity_pattern_entries(dsp);
151
152 SparsityTools::distribute_sparsity_pattern(
153 dsp,
154 this->dof_handler.locally_owned_dofs(),
155 this->mpi_communicator,
156 this->locally_relevant_dofs);
157
158 system_matrix.reinit(this->locally_owned_dofs,
159 this->locally_owned_dofs,
160 dsp,
161 this->mpi_communicator);
162
163 if (this->simulation_parameters.post_processing
164 .calculate_average_velocities ||
165 this->simulation_parameters.initial_condition->type ==
166 Parameters::FluidDynamicsInitialConditionType::average_velocity_profile)
167 {
168 this->average_velocities->initialize_vectors(
169 this->locally_owned_dofs,
170 this->locally_relevant_dofs,
171 this->fe->n_dofs_per_vertex(),
172 this->mpi_communicator);
173 }
174
175 double global_volume =
176 GridTools::volume(*this->triangulation, *this->get_mapping());
177
178 this->pcout << " Number of active cells: "
179 << this->triangulation->n_global_active_cells() << std::endl
180 << " Number of degrees of freedom: "
181 << this->dof_handler.n_dofs() << std::endl;
182 this->pcout << " Volume of triangulation: " << global_volume
183 << std::endl;
184
185
186 // Provide the fluid dynamics dof_handler and present solution to the
187 // multiphysics interface
188 this->multiphysics->set_dof_handler(PhysicsID::fluid_dynamics,
189 &this->dof_handler);
190 this->multiphysics->set_solution(PhysicsID::fluid_dynamics,
191 &this->present_solution);
192 this->multiphysics->set_previous_solutions(PhysicsID::fluid_dynamics,
193 &this->previous_solutions);
194}

196template <int dim>
197void
198FluidDynamicsMatrixBased<dim>::update_mortar_configuration()
199{
200 if (!this->simulation_parameters.mortar_parameters.enable)
201 return;
202
203 TimerOutput::Scope t(this->computing_timer, "Update mortar configuration");
204
205 bool refinement_step;
206 if (this->simulation_parameters.mesh_adaptation.refinement_at_frequency)
207 refinement_step = this->simulation_control->get_step_number() %
208 this->simulation_parameters.mesh_adaptation.frequency ==
209 0;
210 else
211 refinement_step = this->simulation_control->get_step_number() == 0;
212
213 // We need to update the mortar operator/evaluator, as well as the sparsity
214 // pattern, at every iteration. Since this is already done within
215 // setup_dofs(), which is called in refine_mesh(), here we make sure that,
216 // when there is no mesh refinement, the mortar information is still updated
217 if (this->simulation_control->is_at_start() || !refinement_step ||
218 this->simulation_parameters.mesh_adaptation.type ==
219 Parameters::MeshAdaptation::Type::none)
220 {
221 // Clear the preconditioner before the matrix they are associated with is
222 // cleared
223 amg_preconditioner.reset();
224 ilu_preconditioner.reset();
225 current_preconditioner_fill_level = initial_preconditioner_fill_level;
226
227 // Now reset system matrix
228 system_matrix.clear();
229
230 // Rotate mapping
231 this->rotate_rotor_mapping(false);
232
233 // Non Zero constraints
234 this->define_non_zero_constraints();
235
236 // Zero constraints
237 this->define_zero_constraints();
238
239 // Create mortar manager, operator, and evaluator
240 this->reinit_mortar_operators();
241
242 // Create dynamic sparsity pattern
243 DynamicSparsityPattern dsp(this->locally_relevant_dofs);
244 DoFTools::make_sparsity_pattern(this->dof_handler,
245 dsp,
246 this->get_nonzero_constraints(),
247 false);
248
249 // Add mortar sparsity pattern entries
250 this->mortar_coupling_operator->add_sparsity_pattern_entries(dsp);
251
252 SparsityTools::distribute_sparsity_pattern(
253 dsp,
254 this->dof_handler.locally_owned_dofs(),
255 this->mpi_communicator,
256 this->locally_relevant_dofs);
257
258 system_matrix.reinit(this->locally_owned_dofs,
259 this->locally_owned_dofs,
260 dsp,
261 this->mpi_communicator);
262 }

280template <int dim>
281void
282FluidDynamicsMatrixBased<dim>::define_dynamic_zero_constraints()
283{
284 if (!this->simulation_parameters.constrain_solid_domain.enable)
285 return;
286
287 this->dynamic_zero_constraints.clear();
288 this->dynamic_zero_constraints.reinit(this->locally_owned_dofs,
289 this->locally_relevant_dofs);
290
291 DoFTools::make_hanging_node_constraints(this->dof_handler,
292 this->dynamic_zero_constraints);
293
294 const DoFHandler<dim> *dof_handler_ht =
295 this->multiphysics->get_dof_handler(PhysicsID::heat_transfer);
296
297 if (!this->simulation_parameters.multiphysics.VOF)
298 this->constrain_stasis_with_temperature(dof_handler_ht);
299 else
300 {
301 const DoFHandler<dim> *dof_handler_vof =
302 this->multiphysics->get_dof_handler(PhysicsID::VOF);
303 this->constrain_stasis_with_temperature_vof(dof_handler_vof,
304 dof_handler_ht);
305 }
306
307 this->dynamic_zero_constraints.merge(
308 this->zero_constraints,
309 AffineConstraints<double>::MergeConflictBehavior::right_object_wins);
310
311 this->dynamic_zero_constraints.close();
312
313 // Clear sets for next time step
314 for (StasisConstraintWithTemperature &stasis_constraint_struct :
315 this->stasis_constraint_structs)
316 {
317 stasis_constraint_struct.dofs_are_in_solid.clear();
318 stasis_constraint_struct.dofs_are_connected_to_fluid.clear();
319 }

322template <int dim>
323void
324FluidDynamicsMatrixBased<dim>::setup_assemblers()
325{
326 this->assemblers.clear();
327
328 if (this->check_existance_of_bc(
329 BoundaryConditions::BoundaryType::function_weak))
330 {
331 this->assemblers.emplace_back(
332 std::make_shared<WeakDirichletBoundaryCondition<dim>>(
333 this->simulation_control,
334 this->simulation_parameters.boundary_conditions));
335 }
336 if (this->check_existance_of_bc(
337 BoundaryConditions::BoundaryType::partial_slip))
338 {
339 this->assemblers.emplace_back(
340 std::make_shared<PartialSlipDirichletBoundaryCondition<dim>>(
341 this->simulation_control,
342 this->simulation_parameters.boundary_conditions));
343 }
344 if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::outlet))
345 {
346 this->assemblers.emplace_back(
347 std::make_shared<OutletBoundaryCondition<dim>>(
348 this->simulation_control,
349 this->simulation_parameters.boundary_conditions));
350 }
351 if (this->check_existance_of_bc(BoundaryConditions::BoundaryType::pressure))
352 {
353 this->assemblers.emplace_back(
354 std::make_shared<PressureBoundaryCondition<dim>>(
355 this->simulation_control,
356 this->simulation_parameters.boundary_conditions));
357 }
358
359 // Buoyancy force
360 if (this->simulation_parameters.multiphysics.buoyancy_force)
361 {
362 this->assemblers.emplace_back(std::make_shared<BuoyancyAssembly<dim>>(
363 this->simulation_control,
364 this->simulation_parameters.physical_properties_manager
365 .get_reference_temperature()));
366 }
367
368 // ALE
369 if (this->simulation_parameters.ale.enabled())
370 {
371 this->assemblers.emplace_back(
372 std::make_shared<NavierStokesAssemblerALE<dim>>(
373 this->simulation_control, this->simulation_parameters.ale));
374 }
375
376 if (this->simulation_parameters.multiphysics.cahn_hilliard)
377 {
378 // Time-stepping schemes
379 if (is_bdf(this->simulation_control->get_assembly_method()))
380 {
381 this->assemblers.emplace_back(
382 std::make_shared<GLSNavierStokesCahnHilliardAssemblerBDF<dim>>(
383 this->simulation_control));
384 }
385
386 AssertThrow(this->simulation_parameters.velocity_sources.darcy_type !=
387 Parameters::VelocitySource::DarcySourceType::phase_change,
388 PhaseChangeDarcyModelDoesNotSupportCHN());
389
390 this->assemblers.emplace_back(
391 std::make_shared<GLSNavierStokesCahnHilliardAssemblerCore<dim>>(
392 this->simulation_control, this->simulation_parameters));
393 }
394
395 if (this->simulation_parameters.multiphysics.VOF)
396 {
397 // Time-stepping schemes
398 if (is_bdf(this->simulation_control->get_assembly_method()) &&
399 this->simulation_parameters.physical_properties_manager
400 .density_is_constant())
401 {
402 this->assemblers.emplace_back(
403 std::make_shared<GLSNavierStokesVOFAssemblerBDF<dim>>(
404 this->simulation_control));
405 }
406 else if (is_bdf(this->simulation_control->get_assembly_method()))
407 {
408 this->assemblers.emplace_back(
409 std::make_shared<
410 GLSIsothermalCompressibleNavierStokesVOFAssemblerBDF<dim>>(
411 this->simulation_control));
412 }
413
414 // Surface tension force (STF)
415 if (this->simulation_parameters.multiphysics.vof_parameters
416 .surface_tension_force.enable)
417 {
418 if (this->simulation_parameters.multiphysics.vof_parameters
419 .surface_tension_force.enable_marangoni_effect)
420 this->assemblers.emplace_back(
421 std::make_shared<GLSNavierStokesVOFAssemblerMarangoni<dim>>(
422 this->simulation_control,
423 this->simulation_parameters.multiphysics.vof_parameters
424 .surface_tension_force));
425 else
426 {
427 this->assemblers.emplace_back(
428 std::make_shared<GLSNavierStokesVOFAssemblerSTF<dim>>(
429 this->simulation_control, this->simulation_parameters));
430 }
431 }
432
433 // Recoil pressure
434 if (this->simulation_parameters.evaporation.enable_recoil_pressure)
435 {
436 this->assemblers.emplace_back(
437 std::make_shared<NavierStokesVOFAssemblerEvaporation<dim>>(
438 this->simulation_control,
439 this->simulation_parameters.evaporation));
440 }
441
442 // Darcy force for phase change simulations
443 if (this->simulation_parameters.velocity_sources.darcy_type ==
444 Parameters::VelocitySource::DarcySourceType::phase_change)
445 {
446 AssertThrow(this->simulation_parameters.multiphysics.heat_transfer,
447 PhaseChangeDarcyModelRequiresTemperature());
448 this->assemblers.emplace_back(
449 std::make_shared<PhaseChangeDarcyVOFAssembler<dim>>(
450 this->simulation_parameters.physical_properties_manager
451 .get_phase_change_parameters_vector()));
452 }
453
454 if (this->simulation_parameters.physical_properties_manager
455 .is_non_newtonian())
456 {
457 // Core assembler with Non newtonian viscosity
458 this->assemblers.emplace_back(
459 std::make_shared<GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>>(
460 this->simulation_control, this->simulation_parameters));
461 }
462 else if (!this->simulation_parameters.physical_properties_manager
463 .density_is_constant())
464 {
465 this->assemblers.emplace_back(
466 std::make_shared<
467 GLSIsothermalCompressibleNavierStokesVOFAssemblerCore<dim>>(
468 this->simulation_control, this->simulation_parameters));
469 }
470 else
471 {
472 // Core assembler
473 this->assemblers.emplace_back(
474 std::make_shared<GLSNavierStokesVOFAssemblerCore<dim>>(
475 this->simulation_control, this->simulation_parameters));
476 }
477 }
478
479 if (!this->simulation_parameters.multiphysics.VOF &&
480 !this->simulation_parameters.multiphysics.cahn_hilliard)
481 {
482 // Time-stepping schemes
483 if (is_bdf(this->simulation_control->get_assembly_method()))
484 {
485 if (!this->simulation_parameters.physical_properties_manager
486 .density_is_constant())
487 {
488 this->assemblers.emplace_back(
489 std::make_shared<
490 GLSIsothermalCompressibleNavierStokesAssemblerBDF<dim>>(
491 this->simulation_control));
492 }
493 else
494 {
495 this->assemblers.emplace_back(
496 std::make_shared<GLSNavierStokesAssemblerBDF<dim>>(
497 this->simulation_control));
498 }
499 }
500
501 // sdirk methods
502 if (this->simulation_control->is_sdirk())
503 {
504 this->assemblers.emplace_back(
505 std::make_shared<GLSNavierStokesAssemblerSDIRK<dim>>(
506 this->simulation_control));
507 }
508
509 // Rotating frame sources term
510 if (this->simulation_parameters.velocity_sources.rotating_frame_type ==
511 Parameters::VelocitySource::RotatingFrameType::srf)
512 {
513 this->assemblers.emplace_back(
514 std::make_shared<GLSNavierStokesAssemblerSRF<dim>>(
515 this->simulation_parameters.velocity_sources));
516 }
517
518 // Darcy force for phase change simulations in single phase
519 if (this->simulation_parameters.velocity_sources.darcy_type ==
520 Parameters::VelocitySource::DarcySourceType::phase_change)
521 {
522 AssertThrow(this->simulation_parameters.multiphysics.heat_transfer,
523 PhaseChangeDarcyModelRequiresTemperature());
524 this->assemblers.emplace_back(
525 std::make_shared<PhaseChangeDarcyAssembly<dim>>(
526 this->simulation_parameters.physical_properties_manager
527 .get_physical_properties_parameters()
528 .fluids[0]
529 .phase_change_parameters));
530 }
531
532 if (this->simulation_parameters.physical_properties_manager
533 .is_non_newtonian())
534 {
535 // Core assembler with Non newtonian viscosity
536 this->assemblers.emplace_back(
537 std::make_shared<GLSNavierStokesAssemblerNonNewtonianCore<dim>>(
538 this->simulation_control));
539 }
540 else
541 {
542 // Core default assembler
543 if ((this->simulation_parameters.stabilization
544 .use_default_stabilization == true) ||
545 this->simulation_parameters.stabilization.stabilization ==
546 Parameters::Stabilization::NavierStokesStabilization::pspg_supg)
547 {
548 if (!this->simulation_parameters.physical_properties_manager
549 .density_is_constant())
550 {
551 this->assemblers.emplace_back(
552 std::make_shared<
553 GLSIsothermalCompressibleNavierStokesAssemblerCore<dim>>(
554 this->simulation_control));
555 }
556 else
557 {
558 this->assemblers.emplace_back(
559 std::make_shared<PSPGSUPGNavierStokesAssemblerCore<dim>>(
560 this->simulation_control));
561 }
562 }
563 else if (this->simulation_parameters.stabilization.stabilization ==
564 Parameters::Stabilization::NavierStokesStabilization::gls)
565 {
566 if (!this->simulation_parameters.physical_properties_manager
567 .density_is_constant())
568 {
569 this->assemblers.emplace_back(
570 std::make_shared<
571 GLSIsothermalCompressibleNavierStokesAssemblerCore<dim>>(
572 this->simulation_control));
573 }
574 else
575 {
576 this->assemblers.emplace_back(
577 std::make_shared<GLSNavierStokesAssemblerCore<dim>>(
578 this->simulation_control));
579 }
580 }
581 else
582 throw std::runtime_error(
583 "Using the GLS solver with a stabilization other than the pspg_supg or gls "
584 "stabilization will lead to an unstable solver that is unable to converge");
585 }
586 }
587
588 // Mortar ALE
589 // This assembler is added last because it needs access to the strong
590 // Jacobian term computed in the core default assembler
591 if (this->simulation_parameters.mortar_parameters.enable)
592 {
593 this->assemblers.emplace_back(
594 std::make_shared<NavierStokesAssemblerMortarALE<dim>>(
595 this->simulation_control,
596 this->simulation_parameters.mortar_parameters));
597 }

600template <int dim>
601void
602FluidDynamicsMatrixBased<dim>::assemble_system_matrix()
603{
604 TimerOutput::Scope t(this->computing_timer, "Assemble matrix");
605
606 this->system_matrix = 0;
607 setup_assemblers();
608
609 auto scratch_data = NavierStokesScratchData<dim>(
610 this->simulation_control,
611 this->simulation_parameters.physical_properties_manager,
612 *this->fe,
613 *this->cell_quadrature,
614 *this->get_mapping(),
615 *this->face_quadrature);
616
617 if (this->simulation_parameters.multiphysics.VOF)
618 {
619 const DoFHandler<dim> *dof_handler_vof =
620 this->multiphysics->get_dof_handler(PhysicsID::VOF);
621 scratch_data.enable_vof(
622 dof_handler_vof->get_fe(),
623 *this->cell_quadrature,
624 *this->get_mapping(),
625 this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
626
627 if (this->simulation_parameters.multiphysics.vof_parameters
628 .surface_tension_force.enable)
629 {
630 const DoFHandler<dim> &projected_phase_fraction_gradient_dof_handler =
631 this->multiphysics
632 ->get_projected_phase_fraction_gradient_dof_handler();
633 const DoFHandler<dim> &curvature_dof_handler =
634 this->multiphysics->get_curvature_dof_handler();
635 scratch_data.enable_projected_phase_fraction_gradient(
636 projected_phase_fraction_gradient_dof_handler.get_fe(),
637 *this->cell_quadrature,
638 *this->get_mapping());
639 scratch_data.enable_curvature(curvature_dof_handler.get_fe(),
640 *this->cell_quadrature,
641 *this->get_mapping());
642 }
643 }
644 else if (this->simulation_parameters.multiphysics.cahn_hilliard)
645 {
646 const DoFHandler<dim> *dof_handler_cahn_hilliard =
647 this->multiphysics->get_dof_handler(PhysicsID::cahn_hilliard);
648 scratch_data.enable_cahn_hilliard(
649 dof_handler_cahn_hilliard->get_fe(),
650 *this->cell_quadrature,
651 *this->get_mapping(),
652 this->simulation_parameters.multiphysics.cahn_hilliard_parameters);
653 }
654
655 if (this->simulation_parameters.multiphysics.heat_transfer)
656 {
657 const DoFHandler<dim> *dof_handler_ht =
658 this->multiphysics->get_dof_handler(PhysicsID::heat_transfer);
659 scratch_data.enable_heat_transfer(dof_handler_ht->get_fe(),
660 *this->cell_quadrature,
661 *this->get_mapping());
662 }
663

1063/**
1064 * Set the initial condition using a L2 or a viscous solver
1065 **/
1066template <int dim>
1067void
1068FluidDynamicsMatrixBased<dim>::set_initial_condition_fd(
1069 Parameters::FluidDynamicsInitialConditionType initial_condition_type,
1070 bool restart)
1071{
1072 if (restart)
1073 {
1074 this->pcout << "************************" << std::endl;
1075 this->pcout << "---> Simulation Restart " << std::endl;
1076 this->pcout << "************************" << std::endl;
1077 this->read_checkpoint();
1078 }
1079 else if (initial_condition_type ==
1080 Parameters::FluidDynamicsInitialConditionType::L2projection)
1081 {
1082 assemble_L2_projection();
1083 setup_preconditioner();
1084
1085 if (this->simulation_parameters.linear_solver
1086 .at(PhysicsID::fluid_dynamics)
1087 .solver == Parameters::LinearSolver::SolverType::gmres)
1088 solve_system_GMRES(true, 1e-15, 1e-15);
1089
1090 this->present_solution = this->newton_update;
1091 this->finish_time_step();
1092 }
1093 else if (initial_condition_type ==
1094 Parameters::FluidDynamicsInitialConditionType::nodal)
1095 {
1096 this->set_nodal_values();
1097 this->finish_time_step();
1098 }
1099 else if (initial_condition_type ==
1100 Parameters::FluidDynamicsInitialConditionType::viscous)
1101 {
1102 this->set_nodal_values();
1103 std::shared_ptr<RheologicalModel> original_viscosity_model =
1104 this->simulation_parameters.physical_properties_manager.get_rheology();
1105
1106 // Temporarily set the rheology to be newtonian with predefined viscosity
1107 std::shared_ptr<Newtonian> temporary_rheology =
1108 std::make_shared<Newtonian>(
1109 this->simulation_parameters.initial_condition->kinematic_viscosity);
1110
1111 this->simulation_parameters.physical_properties_manager.set_rheology(
1112 temporary_rheology);
1113
1114
1115 this->simulation_control->set_assembly_method(
1116 Parameters::SimulationControl::TimeSteppingMethod::steady);
1117 PhysicsSolver<GlobalVectorType>::solve_non_linear_system(false);
1118 this->finish_time_step();
1119
1120 this->simulation_parameters.physical_properties_manager.set_rheology(
1121 original_viscosity_model);
1122 }
1123 else if (initial_condition_type ==
1124 Parameters::FluidDynamicsInitialConditionType::ramp)
1125 {
1126 this->pcout << "*********************************" << std::endl;
1127 this->pcout << " Initial condition using ramp " << std::endl;
1128 this->pcout << "*********************************" << std::endl;
1129
1130 Timer timer(this->mpi_communicator);
1131
1132 this->set_nodal_values();
1133
1134 // Create a pointer to the current viscosity model
1135 std::shared_ptr<RheologicalModel> viscosity_model =
1136 this->simulation_parameters.physical_properties_manager.get_rheology();
1137
1138 // Gather n and viscosity final paramerters
1139 const double n_end = viscosity_model->get_n();
1140 const double viscosity_end = viscosity_model->get_kinematic_viscosity();
1141
1142 // n ramp parameters
1143 double n =
1144 this->simulation_parameters.initial_condition->ramp.ramp_n.n_init;
1145 const int n_iter_n =
1146 this->simulation_parameters.initial_condition->ramp.ramp_n.n_iter;
1147 const double alpha_n =
1148 this->simulation_parameters.initial_condition->ramp.ramp_n.alpha;
1149
1150 // Kinematic viscosity ramp parameters
1151 const int n_iter_viscosity =
1152 this->simulation_parameters.initial_condition->ramp.ramp_viscosity
1153 .n_iter;
1154 double kinematic_viscosity =
1155 n_iter_viscosity > 0 ?
1156 this->simulation_parameters.initial_condition->ramp.ramp_viscosity
1157 .kinematic_viscosity_init :
1158 viscosity_end;
1159 const double alpha_viscosity =
1160 this->simulation_parameters.initial_condition->ramp.ramp_viscosity
1161 .alpha;
1162
1163 viscosity_model->set_kinematic_viscosity(kinematic_viscosity);
1164
1165 // Ramp on n
1166 for (int i = 0; i < n_iter_n; ++i)
1167 {
1168 this->pcout << std::setprecision(4)
1169 << "********* Solution for n = " + std::to_string(n) +
1170 " and viscosity = " +
1171 std::to_string(kinematic_viscosity) + " *********"
1172 << std::endl;
1173
1174 viscosity_model->set_n(n);
1175
1176 this->simulation_control->set_assembly_method(
1177 Parameters::SimulationControl::TimeSteppingMethod::steady);
1178 PhysicsSolver<GlobalVectorType>::solve_non_linear_system(false);
1179 this->finish_time_step();
1180
1181 n += alpha_n * (n_end - n);
1182 }
1183
1184 // Reset n to simulation parameters
1185 viscosity_model->set_n(n_end);
1186
1187 // Ramp on kinematic viscosity
1188 for (int i = 0; i < n_iter_viscosity; ++i)
1189 {
1190 this->pcout << std::setprecision(4)
1191 << "********* Solution for n = " + std::to_string(n_end) +
1192 " and viscosity = " +
1193 std::to_string(kinematic_viscosity) + " *********"
1194 << std::endl;
1195
1196 viscosity_model->set_kinematic_viscosity(kinematic_viscosity);
1197
1198 this->simulation_control->set_assembly_method(
1199 Parameters::SimulationControl::TimeSteppingMethod::steady);
1200 PhysicsSolver<GlobalVectorType>::solve_non_linear_system(false);
1201 this->finish_time_step();
1202
1203 kinematic_viscosity +=
1204 alpha_viscosity * (viscosity_end - kinematic_viscosity);
1205 }
1206
1207 // Reset kinematic viscosity to simulation parameters
1208 viscosity_model->set_kinematic_viscosity(viscosity_end);
1209
1210 timer.stop();
1211
1212 if (this->simulation_parameters.timer.type !=
1213 Parameters::Timer::Type::none)
1214 {
1215 this->pcout << "*********************************" << std::endl;
1216 this->pcout << " Time spent in ramp: " << timer.wall_time() << "s"
1217 << std::endl;
1218 this->pcout << "*********************************" << std::endl;
1219 }
1220 }
1221 else if (initial_condition_type ==
1222 Parameters::FluidDynamicsInitialConditionType::
1223 average_velocity_profile)
1224 {
1225 announce_string(this->pcout,
1226 "Initial condition using average velocity profile");
1227
1228 std::string checkpoint_file_name =
1229 this->simulation_parameters.initial_condition->average_velocity_folder +
1230 this->simulation_parameters.initial_condition
1231 ->average_velocity_file_name;
1232
1233 // The average velocity profile needs to come from the results of a
1234 // simulation. The checkpoint/restart mechanism is used to obtain all the
1235 // simulations information necessary to restart the simulation. The same
1236 // mesh needs to be used, but new physics can be added to the simulation.
1237 this->set_solution_from_checkpoint(checkpoint_file_name);
1238
1239 auto simulation_control_info =
1240 this->simulation_control->get_checkpointed_simulation_control_info(
1241 checkpoint_file_name);
1242
1243 // Calculate the the average velocities
1244 this->average_velocities->calculate_average_velocities(
1245 this->local_evaluation_point,
1246 this->simulation_parameters.post_processing,
1247 simulation_control_info[0], // Checkpointed end time
1248 simulation_control_info[1]); // Checkpointed time step
1249
1250 this->local_evaluation_point =
1251 this->average_velocities->get_average_velocities();
1252 this->present_solution = this->local_evaluation_point;
1253 }
1254 else
1255 {
1256 throw std::runtime_error("GLSNS - Initial condition could not be set");
1257 }

1260template <int dim>
1261void
1262FluidDynamicsMatrixBased<dim>::assemble_L2_projection()
1263{
1264 system_matrix = 0;
1265 this->system_rhs = 0;
1266 FEValues<dim> fe_values(*this->get_mapping(),
1267 *this->fe,
1268 *this->cell_quadrature,
1269 update_values | update_quadrature_points |
1270 update_JxW_values);
1271 const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
1272 const unsigned int n_q_points = this->cell_quadrature->size();
1273 FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
1274 Vector<double> local_rhs(dofs_per_cell);
1275 std::vector<Vector<double>> initial_velocity(n_q_points,
1276 Vector<double>(dim + 1));
1277 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
1278 const FEValuesExtractors::Vector velocities(0);
1279 const FEValuesExtractors::Scalar pressure(dim);
1280
1281 Tensor<1, dim> rhs_initial_velocity_pressure;
1282 double rhs_initial_pressure;
1283
1284 std::vector<Tensor<1, dim>> phi_u(dofs_per_cell);
1285 std::vector<double> phi_p(dofs_per_cell);
1286
1287 for (const auto &cell : this->dof_handler.active_cell_iterators())
1288 {
1289 if (cell->is_locally_owned())
1290 {
1291 fe_values.reinit(cell);
1292 local_matrix = 0;
1293 local_rhs = 0;
1294 this->simulation_parameters.initial_condition->uvwp.vector_value_list(
1295 fe_values.get_quadrature_points(), initial_velocity);
1296 for (unsigned int q = 0; q < n_q_points; ++q)
1297 {
1298 for (unsigned int k = 0; k < dofs_per_cell; ++k)
1299 {
1300 phi_p[k] = fe_values[pressure].value(k, q);
1301 phi_u[k] = fe_values[velocities].value(k, q);
1302 }
1303
1304 // Establish the rhs tensor operator
1305 for (int i = 0; i < dim; ++i)
1306 {
1307 const unsigned int component_i =
1308 this->fe->system_to_component_index(i).first;
1309 rhs_initial_velocity_pressure[i] =
1310 initial_velocity[q](component_i);
1311 }
1312 rhs_initial_pressure = initial_velocity[q](dim);
1313
1314 for (unsigned int i = 0; i < dofs_per_cell; ++i)
1315 {
1316 // Matrix assembly
1317 for (unsigned int j = 0; j < dofs_per_cell; ++j)
1318 {
1319 local_matrix(i, j) +=
1320 (phi_u[j] * phi_u[i]) * fe_values.JxW(q);
1321 local_matrix(i, j) +=
1322 (phi_p[j] * phi_p[i]) * fe_values.JxW(q);
1323 }
1324 local_rhs(i) += (phi_u[i] * rhs_initial_velocity_pressure +
1325 phi_p[i] * rhs_initial_pressure) *
1326 fe_values.JxW(q);
1327 }
1328 }
1329
1330 cell->get_dof_indices(local_dof_indices);
1331 this->nonzero_constraints.distribute_local_to_global(
1332 local_matrix,
1333 local_rhs,
1334 local_dof_indices,
1335 system_matrix,
1336 this->system_rhs);
1337 }
1338 }
1339 system_matrix.compress(VectorOperation::add);
1340 this->system_rhs.compress(VectorOperation::add);
1341}
