11template <int dim, int spacedim>
12GridMotion<dim, spacedim>::GridMotion(
13 const Parameters::Lagrangian::GridMotion<spacedim> &grid_motion_parameters,
14 const double dem_time_step)
15{
16 auto *action_manager = DEMActionManager::get_action_manager();
17
18 switch (grid_motion_parameters.motion_type)
19 {
20 case Parameters::Lagrangian::GridMotion<spacedim>::MotionType::none:
21 {
22 grid_motion = &GridMotion<dim, spacedim>::no_motion;
23 break;
24 }
25 case Parameters::Lagrangian::GridMotion<spacedim>::MotionType::rotational:
26 {
27 // Communicate to the action manager that there is a grid motion
28 action_manager->set_grid_motion_enabled();
29
30 grid_motion = &GridMotion<dim, spacedim>::move_grid_rotational;
31 rotation_angle =
32 grid_motion_parameters.grid_rotational_speed * dem_time_step;
33 rotation_axis = grid_motion_parameters.grid_rotational_axis;
34 break;
35 }
36 case Parameters::Lagrangian::GridMotion<
37 spacedim>::MotionType::translational:
38 {
39 // Flag and communicate to the action manager that there is a
40 // grid motion
41 action_manager->set_grid_motion_enabled();
42
43 grid_motion = &GridMotion<dim, spacedim>::move_grid_translational;
44 shift_vector =
45 grid_motion_parameters.grid_translational_velocity * dem_time_step;
46 break;
47 }
48 default:
49 AssertThrow(false, ExcNotImplemented());
50 }
51}

95template <int dim, int spacedim>
96void
97GridMotion<dim, spacedim>::
98 update_boundary_points_and_normal_vectors_in_contact_list(
99 typename DEM::dem_data_structures<spacedim>::particle_wall_in_contact
100 &particle_wall_pairs_in_contact,
101 const typename DEM::dem_data_structures<
102 spacedim>::boundary_points_and_normal_vectors
103 &updated_boundary_points_and_normal_vectors)
104{
105 // If there is no grid motion, exit the function
106 if (!DEMActionManager::get_action_manager()->check_grid_motion_enabled())
107 return;
108
109 for (auto &[particle_id, pairs_in_contact_content] :
110 particle_wall_pairs_in_contact)
111 {
112 // Prevent compiler warning
113 (void)particle_id;
114 for (auto pairs_in_contact_iterator = pairs_in_contact_content.begin();
115 pairs_in_contact_iterator != pairs_in_contact_content.end();)
116 {
117 auto global_face_id = pairs_in_contact_iterator->first;
118 auto &contact_information = pairs_in_contact_iterator->second;
119
120 // Since we used the negative keys for diamond-shaped cells, we check
121 // and remove these elements from the particle_wall_pairs_in_contact
122 if (global_face_id >= 0)
123 {
124 contact_information.normal_vector =
125 updated_boundary_points_and_normal_vectors.at(global_face_id)
126 .first;
127 contact_information.point_on_boundary =
128 updated_boundary_points_and_normal_vectors.at(global_face_id)
129 .second;
130 ++pairs_in_contact_iterator;
131 }
132 else
133 {
134 pairs_in_contact_content.erase(pairs_in_contact_iterator++);
135 }
136 }
137 }
138}
