8template <int dim>
9void
10TracerScratchData<dim>::allocate()
11{
12 // Initialize size of arrays
13 this->n_q_points = fe_values_tracer.get_quadrature().size();
14 this->n_dofs = fe_values_tracer.get_fe().n_dofs_per_cell();
15
16 // Initialize arrays related to quadrature
17 this->JxW = std::vector<double>(n_q_points);
18
19 // Forcing term array
20 this->source = std::vector<double>(n_q_points);
21
22 // Initialize arrays related to velocity
23 this->velocities.first_vector_component = 0;
24 // Velocity
25 this->velocity_values = std::vector<Tensor<1, dim>>(n_q_points);
26 // Tracer
27 this->tracer_values = std::vector<double>(n_q_points);
28 this->tracer_gradients = std::vector<Tensor<1, dim>>(n_q_points);
29 this->previous_tracer_gradients =
30 std::vector<Tensor<1, dim>>(this->n_q_points);
31 this->tracer_laplacians = std::vector<double>(n_q_points);
32 this->tracer_diffusivity = std::vector<double>(n_q_points);
33 this->tracer_diffusivity_face = std::vector<double>(n_q_points);
34 this->tracer_diffusivity_0 = std::vector<double>(n_q_points);
35 this->tracer_diffusivity_1 = std::vector<double>(n_q_points);
36 this->tracer_reaction_prefactor = std::vector<double>(n_q_points);
37 this->grad_tracer_reaction_prefactor = std::vector<double>(n_q_points);
38
39 // Solid signed distance function
40 if (properties_manager.field_is_required(field::levelset))
41 this->sdf_values = std::vector<double>(n_q_points);
42
43 // Velocity for BDF schemes
44 this->previous_tracer_values =
45 std::vector<std::vector<double>>(maximum_number_of_previous_solutions(),
46 std::vector<double>(n_q_points));
47
48 // Initialize arrays related to shape functions
49 // Velocity shape functions
50 this->phi =
51 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
52 this->grad_phi = std::vector<std::vector<Tensor<1, dim>>>(
53 n_q_points, std::vector<Tensor<1, dim>>(n_dofs));
54 this->hess_phi = std::vector<std::vector<Tensor<2, dim>>>(
55 n_q_points, std::vector<Tensor<2, dim>>(n_dofs));
56 this->laplacian_phi =
57 std::vector<std::vector<double>>(n_q_points, std::vector<double>(n_dofs));
58
59 if (properties_manager.field_is_required(field::levelset))
60 fields.insert(
61 std::pair<field, std::vector<double>>(field::levelset, n_q_points));
62 if (properties_manager.field_is_required(field::tracer_concentration))
63 fields.insert(
64 std::pair<field, std::vector<double>>(field::tracer_concentration,
65 n_q_points));
66}

69template <int dim>
70void
71TracerScratchData<dim>::calculate_physical_properties()
72{
73 if (properties_manager.field_is_required(field::levelset))
74 set_field_vector(field::levelset, this->sdf_values, this->fields);
75
76 if (properties_manager.field_is_required(field::tracer_concentration))
77 set_field_vector(field::tracer_concentration,
78 this->tracer_values,
79 this->fields);
80
81 switch (properties_manager.get_number_of_fluids())
82 {
83 case 1:
84 {
85 const auto diffusivity_model =
86 properties_manager.get_tracer_diffusivity();
87 diffusivity_model->vector_value(fields, tracer_diffusivity);
88 const auto reaction_prefactor_model =
89 properties_manager.get_tracer_reaction_prefactor();
90 reaction_prefactor_model->vector_value(fields,
91 tracer_reaction_prefactor);
92 reaction_prefactor_model->vector_jacobian(
93 fields,
94 field::tracer_concentration,
95 grad_tracer_reaction_prefactor);
96
97 break;
98 }
99 case 2:
100 {
101 throw std::runtime_error(
102 "Unsupported number of fluids (2) with the tracer physics.");
103 // In this case, we need both density and viscosity
104 const auto diffusivity_models =
105 properties_manager.get_tracer_diffusivity_vector();
106
107 diffusivity_models[0]->vector_value(fields, tracer_diffusivity_0);
108 diffusivity_models[1]->vector_value(fields, tracer_diffusivity_1);
109
110 // TODO Incomplete at the present time because the tracer VOF
111 // complete is not finished Blend the physical properties
112 // using the VOF field
113 // for (unsigned int q = 0; q < this->n_q_points; ++q)
114 // {
115 // tracer_diffusivity[q] =
116 // calculate_point_property(this->phase_values[q],
117 // this->density_0[q],
118 // this->density_1[q]);
119 // }
120 break;
121 }
122 default:
123 throw std::runtime_error("Unsupported number of fluids (>2)");
124 }
125}
