27/**
28 * @brief Finds all the boundary cells and faces in the triangulation, for each
29 * cell the boundary face is specified and the normal vector as well as a point
30 * on the boundary face are obtained.
31 */
32
33template <int dim>
34class BoundaryCellsInformation
35{
36public:
37 BoundaryCellsInformation();
38
39 /**
40 * @brief The build function builds all the boundary cell information
41 * structure that finds which cell has a boundary face, boundary line or a
42 * boundary point. Additionally, it identifies which cells are located in the
43 * vicinity (controlled by the max cell diameter of floating walls.
44 *
45 * @param triangulation Triangulation to access the information of the cells
46 * @param floating_wall_properties Properties of floating walls specified in
47 * the parameter handler file
48 * @param outlet_boundaries A vector which contains the outlet boundary IDs
49 * @param check_diamond_cells If true, the diamond shaped cells are found
50 * and added to the particle-wall contact search cells
51 * @param expand_particle_wall_contact_search If true, expands the
52 * particle-wall contact search by adding the contact search between
53 * particles in the boundary cells with the boundary neighbors of these
54 * boundary cells
55 * @param pcout
56 */
57 void
58 build(
59 const parallel::distributed::Triangulation<dim> &triangulation,
60 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
61 const std::vector<unsigned int> &outlet_boundaries,
62 const bool &check_diamond_cells,
63 const bool &expand_particle_wall_contact_search,
64 const ConditionalOStream &pcout);
65
66 void
67 build(const parallel::distributed::Triangulation<dim> &triangulation,
68 const std::vector<unsigned int> &outlet_boundaries,
69 const bool &check_diamond_cells,
70 const ConditionalOStream &pcout);
71
72 std::map<int, boundary_cells_info_struct<dim>> &
73 get_boundary_cells_information()
74 {
75 return boundary_cells_information;
76 }

78 std::unordered_map<std::string, cell_point_info<dim>> &
79 get_boundary_cells_with_points()

81 return boundary_cells_with_points;

84 std::unordered_map<std::string, cell_line_info<dim>> &

90 std::unordered_map<
91 types::particle_index,
92 std::set<typename Triangulation<dim>::active_cell_iterator>> &
93 get_boundary_cells_with_floating_walls()
94 {
95 return boundary_cells_for_floating_walls;
96 }
97
98 /**
99 * Carries out updating the boundary information (point on boundary face and
100 * its normal vector) after moving the grid.
101 *
102 * @param updated_boundary_points_and_normal_vectors Normal vector and a point
103 * on the boundary faces which are updated after motion of the grid
104 */
105 void
106 update_boundary_info_after_grid_motion(
107 typename DEM::dem_data_structures<dim>::boundary_points_and_normal_vectors
108 &updated_boundary_points_and_normal_vectors);
109
110private:
111 /**
112 * Loops over all the cells to find boundary cells, find the boundary faces of
113 * boundary cells and for the boundary faces the normal vector and a point
114 * locating on the face are obtained
115 *
116 * @param triangulation Triangulation to access the information of the cells
117 * @param outlet_boundaries A vector which contains the outlet boundary IDs
118 */
119 void
120 find_boundary_cells_information(
121 const parallel::distributed::Triangulation<dim> &triangulation,
122 const std::vector<unsigned int> &outlet_boundaries);
123
124 /**
125 * Loops over all the cells to find boundary cells, find the global boundary
126 * cells with faces. Note that the boundary_cells_with_faces container only
127 * stores the local boundary cells, while global_boundary_cells_with_faces
128 * stores all the boundary cells with faces.
129 *
130 * @param triangulation Triangulation to access the information of the cells
131 * @param outlet_boundaries A vector which contains the outlet boundary IDs
132 */
133 void
134 find_global_boundary_cells_with_faces(
135 const parallel::distributed::Triangulation<dim> &triangulation,
136 const std::vector<unsigned int> &outlet_boundaries);
137
138 /**
139 * Loops over all the cells to find cells which should be searched for
140 * particle-line contact, boundary lines and a point locating on each line are
141 * obtained
142 *
143 * @param boundary_cells_with_faces A vector which contains all the boundary
144 * cells which has atleast one boundary face
145 * @param triangulation Triangulation to access the information of the cells
146 * @param boundary_cells_with_lines A vector of tuples which contains the
147 * cells with boundary lines and locations of the beginning and ending
148 * vertices of the boundary line
149 * @param boundary_cells_with_points A vector of pairs which contains the
150 * cells with boundary points, and the location of the point
151 */
152 void
153 find_particle_point_and_line_contact_cells(
154 const parallel::distributed::Triangulation<dim> &triangulation);
155
156 /**
157 * Adds the cells with boundary lines to the boundary cells information
158 * (boundary_cells_information), First , it loops through the
159 * boundary_cells_with_lines vector, then it loops through the neighbors of
160 * boundary cells with lines, if a neighbor is a member of
161 * boundary_cells_with_faces, then it adds the neighbor cell to the
162 * boundary_cells_information with the normal vector and point of the neighbor
163 * cell.
164 *
165 * @param triangulation Triangulation to access the information of the cells
166 * @param outlet_boundaries A vector which contains the outlet boundary IDs
167 * @param check_diamond_cells If true, the diamond shaped cells are found and added to the particle-wall contact search cells
168 * @param pcout
169 */
170 void
171 add_cells_with_boundary_lines_to_boundary_cells(
172 const parallel::distributed::Triangulation<dim> &triangulation,
173 const std::vector<unsigned int> &outlet_boundaries,
174 const bool &check_diamond_cells,
175 const ConditionalOStream &pcout);
176
177 /**
178 * Loops over all the cells to find cells which should be searched for
179 * particle-floating wall contact, normal vector of each cell is also obtained
180 *
181 * @param triangulation Triangulation to access the information of the cells
182 * @param floating_wall_properties Properties of floating walls specified in
183 * the parameter handler file
184 * @param boundary_cells_for_floating_walls An unordered_set which contains
185 * all the boundary cells of floating walls
186 * @param maximum_cell_diameter Maximum cell length in the triangulation
187 */
188 void
189 find_boundary_cells_for_floating_walls(
190 const parallel::distributed::Triangulation<dim> &triangulation,
191 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
192 const double maximum_cell_diameter);
193
194 /**
195 * Carries out adding new elements with the boundary cells (cells with
196 * boundary faces) and information (point and normal vector) of their neighbor
197 * boundary cells to the boundary_cells_information. By adding these
198 * elements, the broad and fine searches will search for potential contacts
199 * of the particles in each boundary cell with the boundary neighbors of these
200 * cells. This function is used to solve the problem of late contact detection
201 * in triangulations with curved boundaries.
202 *
203 * @param boundary_cells_information A container that contains the information
204 * of all the boundary cells with boundary faces
205 * @param outlet_boundaries A vector which contains the outlet boundary IDs
206 * @param global_boundary_cells_information A vector that contains the geometrical
207 * information of all (global) boundary cells
208 */
209 void
210 add_boundary_neighbors_of_boundary_cells(
211 const parallel::distributed::Triangulation<dim> &triangulation,
212 const std::vector<unsigned int> &outlet_boundaries,
213 std::map<int, boundary_cells_info_struct<dim>> &boundary_cells_information,
214 const std::map<int, boundary_cells_info_struct<dim>>
215 &global_boundary_cells_information);
216
217 // Structure that contains the necessary information for boundaries
218 std::map<int, boundary_cells_info_struct<dim>> boundary_cells_information;
219
220 // A vector that contains the geometrical information of all (global) boundary
221 // cells. This vector is used in
222 // add_cells_with_boundary_lines_to_boundary_cells function
223 std::map<int, boundary_cells_info_struct<dim>>
224 global_boundary_cells_information;
225
226 // Structure that contains the boundary cells which have a line
227 std::unordered_map<std::string, cell_line_info<dim>>
228 boundary_cells_with_lines;
229
230 // A vector of all the local cells with boundary lines. This vector is used in
231 // add_cells_with_boundary_lines_to_boundary_cells function.
232 std::vector<typename Triangulation<dim>::active_cell_iterator>
233 local_cells_with_boundary_lines;
234
235 // Structure that contains the boundary cells which have a point
236 std::unordered_map<std::string, cell_point_info<dim>>
237 boundary_cells_with_points;
238
239 // Structure that contains the boundary cells with floating walls
240 std::unordered_map<
241 types::global_dof_index,
242 std::set<typename Triangulation<dim>::active_cell_iterator>>
243 boundary_cells_for_floating_walls;
244
245 // Structure that contains the local boundary cells with faces
246 std::vector<typename Triangulation<dim>::active_cell_iterator>
247 boundary_cells_with_faces;
248
249 // Structure that contains the global boundary cells with faces
250 std::vector<typename Triangulation<dim>::active_cell_iterator>
251 global_boundary_cells_with_faces;
252
253 // A map that contains updated points on boundaries and normal vectors of the
254 // boundary faces. This container is used when the grid is moving. We use this
255 // vector to update the boundary points and normal vectors in particle-wall
256 // contact list.
257 std::map<unsigned int, std::pair<Tensor<1, dim>, Point<dim>>>
258 updated_boundary_points_and_normal_vectors;
259
260 // Boolean variables used for printing a warning only once
261 bool display_diamond_cells_warning = true;
262 bool display_pw_contact_expansion_warning = true;
263};
