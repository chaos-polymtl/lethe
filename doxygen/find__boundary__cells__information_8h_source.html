20/**
21 * @brief Finds all the boundary cells and faces in the triangulation, for each
22 * cell the boundary face is specified and the normal vector as well as a point
23 * on the boundary face are obtained.
24 */
25
26template <int dim>
27class BoundaryCellsInformation
28{
29public:
30 BoundaryCellsInformation();
31
32 /**
33 * @brief The build function builds all the boundary cell information
34 * structure that finds which cell has a boundary face, boundary line or a
35 * boundary point. Additionally, it identifies which cells are located in the
36 * vicinity (controlled by the max cell diameter of floating walls.
37 *
38 * @param triangulation Triangulation to access the information of the cells
39 * @param floating_wall_properties Properties of floating walls specified in
40 * the parameter handler file
41 * @param outlet_boundaries A vector which contains the outlet boundary IDs
42 * @param check_diamond_cells If true, the diamond shaped cells are found
43 * and added to the particle-wall contact search cells
44 * @param expand_particle_wall_contact_search If true, expands the
45 * particle-wall contact search by adding the contact search between
46 * particles in the boundary cells with the boundary neighbors of these
47 * boundary cells
48 * @param pcout
49 */
50 void
51 build(
52 const parallel::distributed::Triangulation<dim> &triangulation,
53 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
54 const std::vector<unsigned int> &outlet_boundaries,
55 const bool &check_diamond_cells,
56 const bool &expand_particle_wall_contact_search,
57 const ConditionalOStream &pcout);
58
59 void
60 build(const parallel::distributed::Triangulation<dim> &triangulation,
61 const std::vector<unsigned int> &outlet_boundaries,
62 const bool &check_diamond_cells,
63 const ConditionalOStream &pcout);
64
65 std::map<int, boundary_cells_info_struct<dim>> &
66 get_boundary_cells_information()
67 {
68 return boundary_cells_information;
69 }
70
71 std::unordered_map<std::string, cell_point_info<dim>> &
72 get_boundary_cells_with_points()
73 {
74 return boundary_cells_with_points;
75 }

77 std::unordered_map<std::string, cell_line_info<dim>> &

79 {
80 return boundary_cells_with_lines;
81 }
82
83 std::unordered_map<
84 types::particle_index,
85 std::set<typename Triangulation<dim>::active_cell_iterator>> &
86 get_boundary_cells_with_floating_walls()
87 {
88 return boundary_cells_for_floating_walls;
89 }
90
91 /**
92 * Carries out updating the boundary information (point on boundary face and
93 * its normal vector) after moving the grid.
94 *
95 * @param updated_boundary_points_and_normal_vectors Normal vector and a point
96 * on the boundary faces which are updated after motion of the grid
97 */
98 void
99 update_boundary_info_after_grid_motion(
100 typename DEM::dem_data_structures<dim>::boundary_points_and_normal_vectors
101 &updated_boundary_points_and_normal_vectors);
102
103private:
104 /**
105 * Loops over all the cells to find boundary cells, find the boundary faces of
106 * boundary cells and for the boundary faces the normal vector and a point
107 * locating on the face are obtained
108 *
109 * @param triangulation Triangulation to access the information of the cells
110 * @param outlet_boundaries A vector which contains the outlet boundary IDs
111 */
112 void
113 find_boundary_cells_information(
114 const parallel::distributed::Triangulation<dim> &triangulation,
115 const std::vector<unsigned int> &outlet_boundaries);
116
117 /**
118 * Loops over all the cells to find boundary cells, find the global boundary
119 * cells with faces. Note that the boundary_cells_with_faces container only
120 * stores the local boundary cells, while global_boundary_cells_with_faces
121 * stores all the boundary cells with faces.
122 *
123 * @param triangulation Triangulation to access the information of the cells
124 * @param outlet_boundaries A vector which contains the outlet boundary IDs
125 */
126 void
127 find_global_boundary_cells_with_faces(
128 const parallel::distributed::Triangulation<dim> &triangulation,
129 const std::vector<unsigned int> &outlet_boundaries);
130
131 /**
132 * Loops over all the cells to find cells which should be searched for
133 * particle-line contact, boundary lines and a point locating on each line are
134 * obtained
135 *
136 * @param boundary_cells_with_faces A vector which contains all the boundary
137 * cells which has atleast one boundary face
138 * @param triangulation Triangulation to access the information of the cells
139 * @param boundary_cells_with_lines A vector of tuples which contains the
140 * cells with boundary lines and locations of the beginning and ending
141 * vertices of the boundary line
142 * @param boundary_cells_with_points A vector of pairs which contains the
143 * cells with boundary points, and the location of the point
144 */
145 void
146 find_particle_point_and_line_contact_cells(
147 const parallel::distributed::Triangulation<dim> &triangulation);
148
149 /**
150 * Adds the cells with boundary lines to the boundary cells information
151 * (boundary_cells_information), First , it loops through the
152 * boundary_cells_with_lines vector, then it loops through the neighbors of
153 * boundary cells with lines, if a neighbor is a member of
154 * boundary_cells_with_faces, then it adds the neighbor cell to the
155 * boundary_cells_information with the normal vector and point of the neighbor
156 * cell.
157 *
158 * @param triangulation Triangulation to access the information of the cells
159 * @param outlet_boundaries A vector which contains the outlet boundary IDs
160 * @param check_diamond_cells If true, the diamond shaped cells are found and added to the particle-wall contact search cells
161 * @param pcout
162 */
163 void
164 add_cells_with_boundary_lines_to_boundary_cells(
165 const parallel::distributed::Triangulation<dim> &triangulation,
166 const std::vector<unsigned int> &outlet_boundaries,
167 const bool &check_diamond_cells,
168 const ConditionalOStream &pcout);
169
170 /**
171 * Loops over all the cells to find cells which should be searched for
172 * particle-floating wall contact, normal vector of each cell is also obtained
173 *
174 * @param triangulation Triangulation to access the information of the cells
175 * @param floating_wall_properties Properties of floating walls specified in
176 * the parameter handler file
177 * @param boundary_cells_for_floating_walls An unordered_set which contains
178 * all the boundary cells of floating walls
179 * @param maximum_cell_diameter Maximum cell length in the triangulation
180 */
181 void
182 find_boundary_cells_for_floating_walls(
183 const parallel::distributed::Triangulation<dim> &triangulation,
184 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
185 const double maximum_cell_diameter);
186
187 /**
188 * Carries out adding new elements with the boundary cells (cells with
189 * boundary faces) and information (point and normal vector) of their neighbor
190 * boundary cells to the boundary_cells_information. By adding these
191 * elements, the broad and fine searches will search for potential contacts
192 * of the particles in each boundary cell with the boundary neighbors of these
193 * cells. This function is used to solve the problem of late contact detection
194 * in triangulations with curved boundaries.
195 *
196 * @param boundary_cells_information A container that contains the information
197 * of all the boundary cells with boundary faces
198 * @param outlet_boundaries A vector which contains the outlet boundary IDs
199 * @param global_boundary_cells_information A vector that contains the geometrical
200 * information of all (global) boundary cells
201 */
202 void
203 add_boundary_neighbors_of_boundary_cells(
204 const parallel::distributed::Triangulation<dim> &triangulation,
205 const std::vector<unsigned int> &outlet_boundaries,
206 std::map<int, boundary_cells_info_struct<dim>> &boundary_cells_information,
207 const std::map<int, boundary_cells_info_struct<dim>>
208 &global_boundary_cells_information);
209
210 // Structure that contains the necessary information for boundaries
211 std::map<int, boundary_cells_info_struct<dim>> boundary_cells_information;
212
213 // A vector that contains the geometrical information of all (global) boundary
214 // cells. This vector is used in
215 // add_cells_with_boundary_lines_to_boundary_cells function
216 std::map<int, boundary_cells_info_struct<dim>>
217 global_boundary_cells_information;
218
219 // Structure that contains the boundary cells which have a line
220 std::unordered_map<std::string, cell_line_info<dim>>
221 boundary_cells_with_lines;
222
223 // A vector of all the local cells with boundary lines. This vector is used in
224 // add_cells_with_boundary_lines_to_boundary_cells function.
225 std::vector<typename Triangulation<dim>::active_cell_iterator>
226 local_cells_with_boundary_lines;
227
228 // Structure that contains the boundary cells which have a point
229 std::unordered_map<std::string, cell_point_info<dim>>
230 boundary_cells_with_points;
231
232 // Structure that contains the boundary cells with floating walls
233 std::unordered_map<
234 types::global_dof_index,
235 std::set<typename Triangulation<dim>::active_cell_iterator>>
236 boundary_cells_for_floating_walls;
237
238 // Structure that contains the local boundary cells with faces
239 std::vector<typename Triangulation<dim>::active_cell_iterator>
240 boundary_cells_with_faces;
241
242 // Structure that contains the global boundary cells with faces
243 std::vector<typename Triangulation<dim>::active_cell_iterator>
244 global_boundary_cells_with_faces;
245
246 // A map that contains updated points on boundaries and normal vectors of the
247 // boundary faces. This container is used when the grid is moving. We use this
248 // vector to update the boundary points and normal vectors in particle-wall
249 // contact list.
250 std::map<unsigned int, std::pair<Tensor<1, dim>, Point<dim>>>
251 updated_boundary_points_and_normal_vectors;
252
253 // Boolean variables used for printing a warning only once
254 bool display_diamond_cells_warning = true;
255 bool display_pw_contact_expansion_warning = true;
256};
