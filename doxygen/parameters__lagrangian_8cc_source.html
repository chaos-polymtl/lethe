108 void
109 LagrangianPhysicalProperties::parse_particle_properties(
110 const unsigned int &particle_type,
111 ParameterHandler &prm)
112 {
113 particle_average_diameter.at(particle_type) = prm.get_double("diameter");
114 particle_size_std.at(particle_type) =
115 prm.get_double("standard deviation");
116 particle_custom_diameter.at(particle_type) =
117 convert_string_to_vector<double>(prm, "custom diameters");
118 particle_custom_probability.at(particle_type) =
119 convert_string_to_vector<double>(prm, "custom volume fractions");
120 seed_for_distributions.push_back(
121 prm.get_integer("random seed distribution"));
122
123 double probability_sum =
124 std::reduce(particle_custom_probability.at(particle_type).begin(),
125 particle_custom_probability.at(particle_type).end());
126
127 // We make sure that the cumulative probability is equal to 1.
128 if (std::abs(probability_sum - 1.0) > 1.e-5)
129 {
130 throw(std::runtime_error(
131 "Invalid custom volume fraction. The sum of volume fractions should be equal to 1.0 "));
132 }
133 const std::string size_distribution_type_str =
134 prm.get("size distribution type");
135 if (size_distribution_type_str == "uniform")
136 {
137 distribution_type.at(particle_type) = SizeDistributionType::uniform;
138 }
139 else if (size_distribution_type_str == "normal")
140 {
141 distribution_type.at(particle_type) = SizeDistributionType::normal;
142 }
143 else if (size_distribution_type_str == "custom")
144 {
145 distribution_type.at(particle_type) = SizeDistributionType::custom;
146 }
147 else
148 {
149 throw(std::runtime_error(
150 "Invalid size distribution type. Choices are <uniform|normal|custom>."));
151 }
152 number.at(particle_type) = prm.get_integer("number of particles");
153 density_particle.at(particle_type) = prm.get_double("density particles");
154 youngs_modulus_particle.at(particle_type) =
155 prm.get_double("young modulus particles");
156 poisson_ratio_particle.at(particle_type) =
157 prm.get_double("poisson ratio particles");
158 restitution_coefficient_particle.at(particle_type) =
159 prm.get_double("restitution coefficient particles");
160 friction_coefficient_particle.at(particle_type) =
161 prm.get_double("friction coefficient particles");
162 rolling_viscous_damping_coefficient_particle.at(particle_type) =
163 prm.get_double("rolling viscous damping particles");
164 rolling_friction_coefficient_particle.at(particle_type) =
165 prm.get_double("rolling friction particles");
166 surface_energy_particle.at(particle_type) =
167 prm.get_double("surface energy particles");
168 hamaker_constant_particle.at(particle_type) =
169 prm.get_double("hamaker constant particles");
170 thermal_conductivity_particle.at(particle_type) =
171 prm.get_double("thermal conductivity particles");
172 specific_heat_particle.at(particle_type) =
173 prm.get_double("specific heat particles");
174 microhardness_particle.at(particle_type) =
175 prm.get_double("microhardness particles");
176 surface_slope_particle.at(particle_type) =
177 prm.get_double("surface slope particles");
178 surface_roughness_particle.at(particle_type) =
179 prm.get_double("surface roughness particles");
180 thermal_accommodation_particle.at(particle_type) =
181 prm.get_double("thermal accommodation particles");
182 }

540 void
541 InsertionInfo::parse_parameters(ParameterHandler &prm)
542 {
543 prm.enter_subsection("insertion info");
544 {
545 const std::string insertion = prm.get("insertion method");
546 if (insertion == "file")
547 insertion_method = InsertionMethod::file;
548 else if (insertion == "plane")
549 insertion_method = InsertionMethod::plane;
550 else if (insertion == "list")
551 insertion_method = InsertionMethod::list;
552 else if (insertion == "volume")
553 insertion_method = InsertionMethod::volume;
554 else
555 {
556 throw(std::runtime_error("Invalid insertion method "));
557 }
558 inserted_this_step =
559 prm.get_integer("inserted number of particles at each time step");
560 insertion_frequency = prm.get_integer("insertion frequency");
561
562 // Clear:
563 removing_particles_in_region = prm.get_bool("remove particles");
564
565 const std::vector<std::string> removal_box_point_coordinates_list(
566 Utilities::split_string_list(
567 prm.get("removal box points coordinates"), ":"));
568
569 std::vector<double> removal_point_coord_temp_1 =
570 Utilities::string_to_double(Utilities::split_string_list(
571 removal_box_point_coordinates_list.at(0)));
572 std::vector<double> removal_point_coord_temp_2 =
573 Utilities::string_to_double(Utilities::split_string_list(
574 removal_box_point_coordinates_list.at(1)));
575
576 if (removal_point_coord_temp_1.size() == 2 &&
577 removal_point_coord_temp_2.size() == 2)
578 {
579 removal_point_coord_temp_1.push_back(0.);
580 removal_point_coord_temp_2.push_back(0.);
581 }
582
583 for (unsigned int i = 0; i < 3; ++i)
584 {
585 clear_box_point_1[i] = removal_point_coord_temp_1.at(i);
586 clear_box_point_2[i] = removal_point_coord_temp_2.at(i);
587 }
588
589 // File:
590 // File for the insertion
591 list_of_input_files =
592 convert_string_to_vector<std::string>(prm, "list of input files");
593
594 // Plane:
595 // Insertion plane normal vector
596 insertion_plane_normal_vector =
597 value_string_to_tensor<3>(prm.get("insertion plane normal vector"));
598
599 // Insertion plane point
600 insertion_plane_point =
601 value_string_to_tensor<3>(prm.get("insertion plane point"));
602
603 // List:
604 // Read x, y and z lists
605 list_x = convert_string_to_vector<double>(prm, "list x");
606 list_y = convert_string_to_vector<double>(prm, "list y");
607 list_z = convert_string_to_vector<double>(prm, "list z");
608
609 // Find which vector is the longest
610 int max_size = std::max({list_x.size(), list_y.size(), list_z.size()});
611
612 // Resize the vectors in the case that one was longer
613 list_x.resize(max_size);
614 list_y.resize(max_size);
615 list_z.resize(max_size);
616
617 // Read vx, vv and vz
618 list_vx = convert_string_to_vector<double>(prm, "list velocity x");
619 list_vy = convert_string_to_vector<double>(prm, "list velocity y");
620 list_vz = convert_string_to_vector<double>(prm, "list velocity z");
621
622 // Fill the velocity vectors with zeros to match the size of list_x
623 if (list_vx != list_x)
624 list_vx.resize(max_size);
625 if (list_vy != list_x)
626 list_vy.resize(max_size);
627 if (list_vz != list_x)
628 list_vz.resize(max_size);
629
630 // Read wx, wy and wz
631 list_wx = convert_string_to_vector<double>(prm, "list omega x");
632 list_wy = convert_string_to_vector<double>(prm, "list omega y");
633 list_wz = convert_string_to_vector<double>(prm, "list omega z");
634
635 // Fill the angular velocity vectors with zeros to match the size of
636 // list_x
637 if (list_wx != list_x)
638 list_wx.resize(max_size);
639 if (list_wy != list_x)
640 list_wy.resize(max_size);
641 if (list_wz != list_x)
642 list_wz.resize(max_size);
643
644 // Read the diameters list
645 list_d = convert_string_to_vector<double>(prm, "list diameters");
646
647 // Volume:
648 std::vector<int> axis_order =
649 convert_string_to_vector<int>(prm, "insertion direction sequence");
650 if (axis_order.size() == 2)
651 axis_order.push_back(0.);
652
653 direction_sequence.reserve(3);
654 direction_sequence.push_back(axis_order[0]);
655 direction_sequence.push_back(axis_order[1]);
656 direction_sequence.push_back(axis_order[2]);
657
658 const std::vector<std::string> point_coordinates_list(
659 Utilities::split_string_list(
660 prm.get("insertion box points coordinates"), ":"));
661
662 std::vector<double> point_coord_temp_1 = Utilities::string_to_double(
663 Utilities::split_string_list(point_coordinates_list.at(0)));
664 std::vector<double> point_coord_temp_2 = Utilities::string_to_double(
665 Utilities::split_string_list(point_coordinates_list.at(1)));
666
667 if (point_coord_temp_1.size() == 2 && point_coord_temp_2.size() == 2)
668 {
669 point_coord_temp_1.push_back(0.);
670 point_coord_temp_2.push_back(0.);
671 }
