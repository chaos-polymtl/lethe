135 {
136 prm.enter_subsection("lagrangian physical properties");
137 initialize_containers(particle_average_diameter,
138 particle_size_std,
139 distribution_type,
140 particle_custom_diameter,
141 particle_custom_probability,
142 seed_for_distributions,
143 diameter_min_cutoff,
144 diameter_max_cutoff,

146 density_particle,

157 microhardness_particle,
158 surface_slope_particle,
159 surface_roughness_particle,
160 thermal_accommodation_particle,
161 real_youngs_modulus_particle);
162
163 // Deprecated parameter handling
164 // <g> used to be 3 parameters: <gx>, <gy> and <gz>
165 // If <gx> is in the input file, it will be used as the value for <g>
166 // as an alias. This way, the parameter <g> is not a tensor and allows the
167 // parsing of deprecated parameters.

169 prm.get_double("gy"),
170 prm.get_double("gz"));
171
172 particle_type_number = prm.get_integer("number of particle types");
173
174 for (unsigned int id = 0; id < particle_type_number; ++id)
175 {
176 prm.enter_subsection("particle type " +
177 Utilities::int_to_string(id, 1));
178 parse_particle_properties(id, prm);
179 prm.leave_subsection();
180 }
181

183 poisson_ratio_wall = prm.get_double("poisson ratio wall");
184 restitution_coefficient_wall =
185 prm.get_double("restitution coefficient wall");

187 rolling_friction_wall = prm.get_double("rolling friction wall");
188 rolling_viscous_damping_wall =

207 molecular_mean_free_path_gas =
208 prm.get_double("molecular mean free path gas");
209
210 prm.leave_subsection();
211 }

328 void
329 LagrangianPhysicalProperties::parse_particle_properties(
330 const unsigned int &particle_type,
331 const ParameterHandler &prm)
332 {
333 // unordered maps
334 particle_average_diameter.at(particle_type) =
335 prm.get_double("average diameter");
336 particle_size_std.at(particle_type) =
337 prm.get_double("standard deviation");
338 particle_custom_diameter.at(particle_type) =
339 convert_string_to_vector<double>(prm, "custom diameters");
340 particle_custom_probability.at(particle_type) =
341 convert_string_to_vector<double>(prm, "custom volume fractions");
342
343 // vectors
344 seed_for_distributions.push_back(
345 prm.get_integer("random seed distribution"));
346 diameter_min_cutoff.push_back(prm.get_double("minimum diameter cutoff"));
347 diameter_max_cutoff.push_back(prm.get_double("maximum diameter cutoff"));
348
349 double probability_sum =
350 std::reduce(particle_custom_probability.at(particle_type).begin(),
351 particle_custom_probability.at(particle_type).end());
352
353 // We make sure that the cumulative probability is equal to 1.
354 if (std::abs(probability_sum - 1.0) > 1.e-5)
355 {
356 throw(std::runtime_error(
357 "Invalid custom volume fraction. The sum of volume fractions should be equal to 1.0 "));
358 }
359 const std::string size_distribution_type_str =
360 prm.get("size distribution type");
361 if (size_distribution_type_str == "uniform")
362 {
363 distribution_type.at(particle_type) = SizeDistributionType::uniform;
364 }
365 else if (size_distribution_type_str == "normal")
366 {
367 distribution_type.at(particle_type) = SizeDistributionType::normal;
368 }
369 else if (size_distribution_type_str == "lognormal")
370 {
371 distribution_type.at(particle_type) = SizeDistributionType::lognormal;
372 }
373 else if (size_distribution_type_str == "custom")
374 {
375 distribution_type.at(particle_type) = SizeDistributionType::custom;
376 }
377 else
378 {
379 throw(std::runtime_error(
380 "Invalid size distribution type. Choices are <uniform|normal|custom>."));
381 }
382 number.at(particle_type) = prm.get_integer("number of particles");
383 density_particle.at(particle_type) = prm.get_double("density particles");
384 youngs_modulus_particle.at(particle_type) =
385 prm.get_double("young modulus particles");
386 poisson_ratio_particle.at(particle_type) =
387 prm.get_double("poisson ratio particles");
388 restitution_coefficient_particle.at(particle_type) =
389 prm.get_double("restitution coefficient particles");
390 friction_coefficient_particle.at(particle_type) =
391 prm.get_double("friction coefficient particles");
392 rolling_viscous_damping_coefficient_particle.at(particle_type) =
393 prm.get_double("rolling viscous damping particles");
394 rolling_friction_coefficient_particle.at(particle_type) =
395 prm.get_double("rolling friction particles");
396 surface_energy_particle.at(particle_type) =
397 prm.get_double("surface energy particles");
398 hamaker_constant_particle.at(particle_type) =
399 prm.get_double("hamaker constant particles");
400 thermal_conductivity_particle.at(particle_type) =
401 prm.get_double("thermal conductivity particles");
402 specific_heat_particle.at(particle_type) =
403 prm.get_double("specific heat particles");
404 microhardness_particle.at(particle_type) =
405 prm.get_double("microhardness particles");
406 surface_slope_particle.at(particle_type) =
407 prm.get_double("surface slope particles");
408 surface_roughness_particle.at(particle_type) =
409 prm.get_double("surface roughness particles");
410 thermal_accommodation_particle.at(particle_type) =
411 prm.get_double("thermal accommodation particles");
412 real_youngs_modulus_particle.at(particle_type) =
413 prm.get_double("real young modulus particles");
414 // Only use the real Young's modulus if it is higher than the Young's
415 // modulus
416 if (real_youngs_modulus_particle.at(particle_type) <
417 youngs_modulus_particle.at(particle_type))
418 {
419 real_youngs_modulus_particle.at(particle_type) =
420 youngs_modulus_particle.at(particle_type);
421 }
422 }

632 template <int dim>
633 void
634 InsertionInfo<dim>::parse_parameters(ParameterHandler &prm)
635 {
636 prm.enter_subsection("insertion info");
637 {
638 const std::string insertion = prm.get("insertion method");
639 if (insertion == "file")
640 insertion_method = InsertionMethod::file;
641 else if (insertion == "plane")
642 insertion_method = InsertionMethod::plane;
643 else if (insertion == "list")
644 insertion_method = InsertionMethod::list;
645 else if (insertion == "volume")
646 insertion_method = InsertionMethod::volume;
647 else
648 {
649 throw(std::runtime_error("Invalid insertion method "));
650 }
651 inserted_this_step =
652 prm.get_integer("inserted number of particles at each time step");
653 insertion_frequency = prm.get_integer("insertion frequency");
654
655 // Clear:
656 removing_particles_in_region = prm.get_bool("remove particles");
657
658 const std::vector<std::string> removal_box_point_coordinates_list(
659 Utilities::split_string_list(
660 prm.get("removal box points coordinates"), ":"));
661
662 std::vector<double> removal_point_coord_temp_1 =
663 Utilities::string_to_double(Utilities::split_string_list(
664 removal_box_point_coordinates_list.at(0)));
665 std::vector<double> removal_point_coord_temp_2 =
666 Utilities::string_to_double(Utilities::split_string_list(
667 removal_box_point_coordinates_list.at(1)));
668
669 if (removal_point_coord_temp_1.size() == 2 &&
670 removal_point_coord_temp_2.size() == 2)
