311 void
312 LagrangianPhysicalProperties::parse_particle_properties(
313 const unsigned int &particle_type,
314 const ParameterHandler &prm)
315 {
316 particle_average_diameter.at(particle_type) = prm.get_double("diameter");
317 particle_size_std.at(particle_type) =
318 prm.get_double("standard deviation");
319 particle_custom_diameter.at(particle_type) =
320 convert_string_to_vector<double>(prm, "custom diameters");
321 particle_custom_probability.at(particle_type) =
322 convert_string_to_vector<double>(prm, "custom volume fractions");
323 seed_for_distributions.push_back(
324 prm.get_integer("random seed distribution"));
325
326 double probability_sum =
327 std::reduce(particle_custom_probability.at(particle_type).begin(),
328 particle_custom_probability.at(particle_type).end());
329
330 // We make sure that the cumulative probability is equal to 1.
331 if (std::abs(probability_sum - 1.0) > 1.e-5)
332 {
333 throw(std::runtime_error(
334 "Invalid custom volume fraction. The sum of volume fractions should be equal to 1.0 "));
335 }
336 const std::string size_distribution_type_str =
337 prm.get("size distribution type");
338 if (size_distribution_type_str == "uniform")
339 {
340 distribution_type.at(particle_type) = SizeDistributionType::uniform;
341 }
342 else if (size_distribution_type_str == "normal")
343 {
344 distribution_type.at(particle_type) = SizeDistributionType::normal;
345 }
346 else if (size_distribution_type_str == "custom")
347 {
348 distribution_type.at(particle_type) = SizeDistributionType::custom;
349 }
350 else
351 {
352 throw(std::runtime_error(
353 "Invalid size distribution type. Choices are <uniform|normal|custom>."));
354 }
355 number.at(particle_type) = prm.get_integer("number of particles");
356 density_particle.at(particle_type) = prm.get_double("density particles");
357 youngs_modulus_particle.at(particle_type) =
358 prm.get_double("young modulus particles");
359 poisson_ratio_particle.at(particle_type) =
360 prm.get_double("poisson ratio particles");
361 restitution_coefficient_particle.at(particle_type) =
362 prm.get_double("restitution coefficient particles");
363 friction_coefficient_particle.at(particle_type) =
364 prm.get_double("friction coefficient particles");
365 rolling_viscous_damping_coefficient_particle.at(particle_type) =
366 prm.get_double("rolling viscous damping particles");
367 rolling_friction_coefficient_particle.at(particle_type) =
368 prm.get_double("rolling friction particles");
369 surface_energy_particle.at(particle_type) =
370 prm.get_double("surface energy particles");
371 hamaker_constant_particle.at(particle_type) =
372 prm.get_double("hamaker constant particles");
373 thermal_conductivity_particle.at(particle_type) =
374 prm.get_double("thermal conductivity particles");
375 specific_heat_particle.at(particle_type) =
376 prm.get_double("specific heat particles");
377 microhardness_particle.at(particle_type) =
378 prm.get_double("microhardness particles");
379 surface_slope_particle.at(particle_type) =
380 prm.get_double("surface slope particles");
381 surface_roughness_particle.at(particle_type) =
382 prm.get_double("surface roughness particles");
383 thermal_accommodation_particle.at(particle_type) =
384 prm.get_double("thermal accommodation particles");
385 real_youngs_modulus_particle.at(particle_type) =
386 prm.get_double("real young modulus particles");
387 // Only use the real Young's modulus if it is higher than the Young's
388 // modulus
389 if (real_youngs_modulus_particle.at(particle_type) <
390 youngs_modulus_particle.at(particle_type))
391 {
392 real_youngs_modulus_particle.at(particle_type) =
393 youngs_modulus_particle.at(particle_type);
394 }

601 template <int dim>
602 void
603 InsertionInfo<dim>::parse_parameters(ParameterHandler &prm)
604 {
605 prm.enter_subsection("insertion info");
606 {
607 const std::string insertion = prm.get("insertion method");
608 if (insertion == "file")
609 insertion_method = InsertionMethod::file;
610 else if (insertion == "plane")
611 insertion_method = InsertionMethod::plane;
612 else if (insertion == "list")
613 insertion_method = InsertionMethod::list;
614 else if (insertion == "volume")
615 insertion_method = InsertionMethod::volume;
616 else
617 {
618 throw(std::runtime_error("Invalid insertion method "));
619 }
620 inserted_this_step =
621 prm.get_integer("inserted number of particles at each time step");
622 insertion_frequency = prm.get_integer("insertion frequency");
623
624 // Clear:
625 removing_particles_in_region = prm.get_bool("remove particles");
626
627 const std::vector<std::string> removal_box_point_coordinates_list(
628 Utilities::split_string_list(
629 prm.get("removal box points coordinates"), ":"));
630
631 std::vector<double> removal_point_coord_temp_1 =
632 Utilities::string_to_double(Utilities::split_string_list(
633 removal_box_point_coordinates_list.at(0)));
634 std::vector<double> removal_point_coord_temp_2 =
635 Utilities::string_to_double(Utilities::split_string_list(
636 removal_box_point_coordinates_list.at(1)));
637
638 if (removal_point_coord_temp_1.size() == 2 &&
639 removal_point_coord_temp_2.size() == 2)
640 {
641 removal_point_coord_temp_1.push_back(0.);
642 removal_point_coord_temp_2.push_back(0.);
643 }
644
645 for (unsigned int i = 0; i < 3; ++i)
646 {
647 clear_box_point_1[i] = removal_point_coord_temp_1.at(i);
648 clear_box_point_2[i] = removal_point_coord_temp_2.at(i);
649 }
650
651 // File:
652 // File for the insertion
653 list_of_input_files =
654 convert_string_to_vector<std::string>(prm, "list of input files");
655
656 // Plane:
657 // Insertion plane normal vector
658 insertion_plane_normal_vector =
659 value_string_to_tensor<3>(prm.get("insertion plane normal vector"));
660
661 // Insertion plane point
662 insertion_plane_point =
663 value_string_to_tensor<3>(prm.get("insertion plane point"));
664
665 // List:
666 // Read x, y and z lists
667 list_x = convert_string_to_vector<double>(prm, "list x");
668 list_y = convert_string_to_vector<double>(prm, "list y");
669 list_z = convert_string_to_vector<double>(prm, "list z");
670
671 // Find which vector is the longest
672 int max_size = std::max({list_x.size(), list_y.size(), list_z.size()});
673

1527 BCDEM::parse_boundary_conditions(const ParameterHandler &prm)
1528 {
1529 const unsigned int boundary_id = prm.get_integer("boundary id");
1530 const std::string boundary_type = prm.get("type");
1531
1532 if (boundary_type == "outlet")
1533 {
1534 bc_types.push_back(BoundaryType::outlet);
1535 this->outlet_boundaries.push_back(boundary_id);
1536 }
1537 else if (boundary_type == "translational")
1538 {
1539 bc_types.push_back(BoundaryType::translational);
1540 Tensor<1, 3> translational_velocity;
1541 translational_velocity[0] = prm.get_double("speed x");
1542 translational_velocity[1] = prm.get_double("speed y");
1543 translational_velocity[2] = prm.get_double("speed z");
1544
1545 this->boundary_translational_velocity.at(boundary_id) =
1546 translational_velocity;
1547 }
1548 else if (boundary_type == "rotational")
1549 {
1550 bc_types.push_back(BoundaryType::rotational);
1551 double rotational_speed = prm.get_double("rotational speed");
1552
1553 // Read the rotational vector from a list of doubles
1554 Tensor<1, 3> rotational_vector =
1555 value_string_to_tensor<3>(prm.get("rotational vector"));
1556 if (rotational_vector.norm() == 0.)
1557 {
1558 throw(std::runtime_error(
1559 "Invalid rotational vector. Its norm cannot be equal to zero."));
1560 }
1561 // Read the point from a list of doubles
1562 Tensor<1, 3> point_on_rotation_axis_tensor =
1563 value_string_to_tensor<3>(prm.get("point on rotational vector"));
1564
1565 this->boundary_rotational_speed.at(boundary_id) = rotational_speed;
1566 this->boundary_rotational_vector.at(boundary_id) =
1567 rotational_vector / rotational_vector.norm();
1568 this->point_on_rotation_axis.at(boundary_id) =
1569 point_on_rotation_axis_tensor;
1570 }
1571 else if (boundary_type == "fixed_wall")
1572 {
1573 bc_types.push_back(BoundaryType::fixed_wall);
1574 }
1575 else if (boundary_type == "periodic")
1576 {
1577 bc_types.push_back(BoundaryType::periodic);
1578 this->periodic_boundary_0 = prm.get_integer("periodic id 0");
1579 this->periodic_boundary_1 = prm.get_integer("periodic id 1");
1580 this->periodic_direction = prm.get_integer("periodic direction");
1581 }
