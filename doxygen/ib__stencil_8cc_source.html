11template <int dim>
12unsigned int
13IBStencil<dim>::number_of_interpolation_support_points(const unsigned int order)
14{
15 // The number of points used in the stencil excluding the DOF is equal to the
16 // order.
17 unsigned int nb_points = order;
18
19 // In the case where the cell is used directly to find the solution at
20 // the IB only one point is needed.
21 if (order > 4)
22 nb_points = 1;
23 return nb_points;
24}

26template <int dim>
27void
28IBStencil<dim>::p_base(const unsigned int order)
29{
30 using numbers::PI;
31 // Define the sampling point position of the stencil on the reference 1D
32 // stencil 0 to 1, 1 being the position of the DOF.
33 reference_points.resize(order + 1);
34 for (unsigned int i = 0; i < order + 1; ++i)
35 {
36 reference_points[i] = 0.5 * (1 - cos(PI * i / order));
37 }
38}

40template <int dim>
41std::vector<double>
42IBStencil<dim>::coefficients(const unsigned int order,
43 const double length_ratio)
44{
45 p_base(order);
46 // Initialize the coefficient vector
47 std::vector<double> coef(order + 1);
48
49 FullMatrix<double> vandermonde(order + 1, order + 1);
50 FullMatrix<double> stencil(order + 1, order + 1);
51 FullMatrix<double> inv_vandermonde(order + 1, order + 1);
52
53 Vector<double> rhs(order + 1);
54 // Define the Vandermonde matrix
55 for (unsigned int i = 0; i < order + 1; ++i)
56 {
57 for (unsigned int j = 0; j < order + 1; ++j)
58 {
59 vandermonde[i][j] = std::pow(reference_points[i], j);
60 }
61 rhs[i] = std::pow(1 + length_ratio, i);
62 }
63 // Invert the Vandermonde matrix
64 inv_vandermonde.invert(vandermonde);
65
66 // Multiply each line of the inverted matrix by (1+length_ratio)^i
67 for (unsigned int i = 0; i < order + 1; ++i)
68 {
69 for (unsigned int j = 0; j < order + 1; ++j)
70 {
71 stencil[i][j] = inv_vandermonde[i][j] * rhs[i];
72 }
73 }
74 // Sum the columns to get the coefficients.
75 for (unsigned int i = 0; i < order + 1; ++i)
76 {
77 for (unsigned int j = 0; j < order + 1; ++j)
78 {
79 coef[order - i] += stencil[j][i];
80 }
81 }
82
83 // Fill the coefficient vector based on the order.
84 if (order > 4)
85 {
86 // In this case the cell is directly used to find the solution at the IB
87 // position. In this case only one point is needed (the position of the
88 // point on the IB) and its coefficient is 1.
89 coef.resize(1);
90 coef[0] = 1;
91 }
92
93 return coef;
94}

96template <int dim>
97std::tuple<Point<dim>, std::vector<Point<dim>>>
98IBStencil<dim>::support_points_for_interpolation(
99 const unsigned int order,
100 const double length_ratio,
101 IBParticle<dim> &p,
102 const Point<dim> &dof_point,
103 const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
104{
105 // Create the vector of points used for the stencil based on the order of the
106 // stencil. Also return the DOF position or the position of the point on the
107 // IB depending if the cell is used directly.
108 p_base(order);
109 Point<dim> point;
110 Point<dim> surface_point;
111 p.closest_surface_point(dof_point, surface_point, cell_guess);
112 std::vector<Point<dim>> interpolation_points;
113
114 if (order > 4)
115 {
116 // In this case the cell is directly used to find the solution at the IB
117 // position. In this case only one point is needed (the position of the
118 // point on the IB).
119 Tensor<1, dim, double> vect_ib = dof_point - surface_point;
120 point = surface_point;
121
122 Point<dim, double> interpolation_point_1(dof_point +
123 vect_ib * 1. / length_ratio);
124
125 interpolation_points.resize(1);
126 interpolation_points[0] = interpolation_point_1;
127 }
128 else
129 {
130 interpolation_points.resize(order);
131 Tensor<1, dim, double> vect_ib = dof_point - surface_point;
132 point = dof_point;
133 for (unsigned int i = 1; i < order + 1; ++i)
134 {
135 interpolation_points[i - 1] =
136 dof_point +
137 vect_ib * (1 - reference_points[order - i]) / (length_ratio);
138 }
139 }
140 return {point, interpolation_points};
141}

223template <int dim>
224double
225IBStencil<dim>::ib_velocity(
226 IBParticle<dim> &p,
227 const Point<dim> &dof_point,
228 unsigned int component,
229 const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
230{
231 double immersed_boundary_velocity = 0;
232
233 // Return the value of the IB condition for that specific stencil.
234 Tensor<1, 3, double> radial_vector;
235 Point<dim> closest_point;
236 p.closest_surface_point(dof_point, closest_point, cell_guess);
237 Point<dim> position_to_surface;
238 position_to_surface = closest_point - p.position;
239 radial_vector[0] = position_to_surface[0];
240 radial_vector[1] = position_to_surface[1];
241 // We have to do that conversion as there is no proper conversion from
242 // tensor of dim 2 to 3.
243 if constexpr (dim == 3)
244 radial_vector[2] = position_to_surface[2];
245
246 if constexpr (dim == 2)
247 {
248 Tensor<1, 3> relative_tangential_velocity =
249 cross_product_3d(p.omega, radial_vector);
250 if (component == 0)
251 {
252 // vx in 2D
253 immersed_boundary_velocity =
254 relative_tangential_velocity[0] + p.velocity[0];
255 }
256 if (component == 1)
257 {
258 // vy in 2D
259 immersed_boundary_velocity =
260 relative_tangential_velocity[1] + p.velocity[1];
261 }
262 }
263 if constexpr (dim == 3)
264 {
265 Tensor<1, 3> relative_tangential_velocity =
266 cross_product_3d(p.omega, radial_vector);
267 if (component == 0)
268 {
269 // vx in 3D
270 immersed_boundary_velocity =
271 relative_tangential_velocity[0] + p.velocity[0];
272 }
273 if (component == 1)
274 {
275 // vy in 3D
276 immersed_boundary_velocity =
277 relative_tangential_velocity[1] + p.velocity[1];
278 }
279 if (component == 2)
280 {
281 // vz in 3D
282 immersed_boundary_velocity =
283 relative_tangential_velocity[2] + p.velocity[2];
284 }
285 }
286
287 return immersed_boundary_velocity;
288}
