848 public:

852 // Type of laser model used in simulations. With "exponential_decay", the
853 // laser acts as a volumetric source, whereas, with
854 // "heat_flux_vof_interface", the laser behaves as a surface flux at the
855 // interface between fluids (VOF auxiliary physic must be enabled to use
856 // this model).
857 enum class LaserType : std::int8_t
858 {

860 gaussian_heat_flux_vof_interface,

871 // Absorptivity is defined as the fraction of the amount of incident
872 // radiation that is absorbed by the surface, and it is measured using
873 // diffuse reﬂectance spectroscopy (DRS). Generally, a constant value in
874 // the range of 0.3-0.8 (for welding processes with titanium) are used
875 // in the literature. However, recent studies show that it varies with
876 // powder particle size distribution, and the angle of incidence that
877 // changes due to the dynamic melt pool surface [Zhang, Z., Huang, Y.,
878 // Kasinathan, A.R., Shahabad, S.I., Ali, U., Mahmoodkhani, Y. and
879 // Toyserkani, E., 2019. 3-Dimensional heat transfer modeling for laser
880 // powder-bed fusion additive manufacturing with volumetric heat sources
881 // based on varied thermal conductivity and absorptivity. Optics & Laser
882 // Technology, 109, pp.297-312.]
883 double laser_absorptivity;
884
885 // Penetration depth of the laser
886 double penetration_depth;
887
888 // Laser beam radius on the melt pool surface
889 double beam_radius;
