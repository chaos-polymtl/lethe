858 public:
859 // A boolean parameter that enables the calculations of laser heat source

861
862 // Type of laser model used in simulations. With "exponential_decay", the
863 // laser acts as a volumetric source, whereas, with
864 // "heat_flux_vof_interface", the laser behaves as a surface flux at the

866 // this model).
867 enum class LaserType : std::int8_t
868 {

870 gaussian_heat_flux_vof_interface,
871 uniform_heat_flux_vof_interface

875 // In almost all the articles, it is assumed equal to 2.0
876 double concentration_factor;

878 // Laser power in W
879 double laser_power;
880
881 // Absorptivity is defined as the fraction of the amount of incident
882 // radiation that is absorbed by the surface, and it is measured using
883 // diffuse reectance spectroscopy (DRS). Generally, a constant value in
884 // the range of 0.3-0.8 (for welding processes with titanium) are used
885 // in the literature. However, recent studies show that it varies with
886 // powder particle size distribution, and the angle of incidence that
887 // changes due to the dynamic melt pool surface [Zhang, Z., Huang, Y.,
