9template <int dim, typename PropertiesIndex>
10class InsertionList : public Insertion<dim, PropertiesIndex>
11{
12public:
13 /**
14 * @brief Insert particles using a list specific positions. This allows the
15 * insertion of any number of particles at a well-controlled location which is
16 * especially useful from a testing perspective. The code ensures that the
17 * number of positions provided in the x,y (and possibly z) direction is
18 * coherent. If more particles than the number of positions in the list is
19 * requested, the class will continue inserting particles at the insertion
20 * frequency using the list of positions. There is no mechanism in place that
21 * prevents the overlap of these new particles with previous ones.
22 *
23 * @param size_distribution_object_container Contains all distribution for each
24 * particle type
25 * @param triangulation Triangulation to access the cells in which the
26 * particles are inserted
27 * @param dem_parameters DEM parameters declared in the .prm file
28 */
29 InsertionList(const std::vector<std::shared_ptr<Distribution>>
30 &size_distribution_object_container,
31 const parallel::distributed::Triangulation<dim> &triangulation,
32 const DEMSolverParameters<dim> &dem_parameters);

34 /**
35 * @brief The InsertionList class inserts particles using a list specific position.
