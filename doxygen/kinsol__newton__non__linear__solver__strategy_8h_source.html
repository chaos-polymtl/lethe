51template <typename VectorType>
52void
53KinsolNewtonNonLinearSolverStrategy<VectorType>::solve()
54{
55#ifdef DEAL_II_WITH_SUNDIALS
56
57 PhysicsSolver<VectorType> *solver = this->physics_solver;
58
59 auto &local_evaluation_point = solver->get_local_evaluation_point();
60
61 VectorType &evaluation_point = solver->get_evaluation_point();
62 VectorType &present_solution = solver->get_present_solution();
63
64 typename SUNDIALS::KINSOL<VectorType>::AdditionalData additional_data;
65 additional_data.function_tolerance = this->params.tolerance;
66 additional_data.maximum_non_linear_iterations = this->params.max_iterations;
67 additional_data.step_tolerance = this->params.tolerance;
68
69 // Kinsol solution strategy, the default one is line search
70 if (this->params.kinsol_strategy ==
71 Parameters::NonLinearSolver::KinsolStrategy::normal_newton)
72 {
73 additional_data.strategy =
74 SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::newton;
75 }
76 else if (this->params.kinsol_strategy ==
77 Parameters::NonLinearSolver::KinsolStrategy::line_search)
78 {
79 additional_data.strategy = SUNDIALS::KINSOL<
80 VectorType>::AdditionalData::SolutionStrategy::linesearch;
81 }
82 else if (this->params.kinsol_strategy ==
83 Parameters::NonLinearSolver::KinsolStrategy::picard)
84 {
85 additional_data.strategy =
86 SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::picard;
87 }
88
89 SUNDIALS::KINSOL<VectorType> nonlinear_solver(additional_data);
90
91 nonlinear_solver.reinit_vector = [&](VectorType &x) {
92 x.reinit(local_evaluation_point);
93 };
94
95 nonlinear_solver.residual = [&](const VectorType &evaluation_point_for_kinsol,
96 VectorType &residual) {
97 solver->pcout << "Computing residual vector..." << std::endl;
98 evaluation_point = evaluation_point_for_kinsol;
99 solver->apply_constraints();
100 solver->assemble_system_rhs();
101 auto &current_residual = solver->get_system_rhs();
102 residual = current_residual;
103 solver->pcout << " -Residual: " << residual.l2_norm() << std::endl;
104 return 0;
105 };
106
107 nonlinear_solver.setup_jacobian =
108 [&](const VectorType &present_solution_for_kinsol,
109 const VectorType & /*current_f*/) {
110 solver->pcout << "Computing jacobian matrix..." << std::endl;
111 evaluation_point = present_solution_for_kinsol;
112 solver->assemble_system_matrix();
113 solver->setup_preconditioner();
114 return 0;
115 };
116
117 nonlinear_solver.solve_with_jacobian = [&](const VectorType & /* residual */,
118 VectorType &dst,
119 const double /* tolerance */) {
120 solver->pcout << "Solving linear system..." << std::endl;
121 solver->solve_linear_system();
122 dst = solver->get_newton_update();
123 return 0;
124 };
125
126 local_evaluation_point = present_solution;
127
128 nonlinear_solver.solve(local_evaluation_point);
129 present_solution = local_evaluation_point;
130#else
131 throw std::runtime_error(
132 "Kinsol newton nonlinear solver requires DEAL_II to be compiled with SUNDIALS");
133#endif // DEAL_II_WITH_SUNDIALS
134}
