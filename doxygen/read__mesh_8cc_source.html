11template <int dim, int spacedim>
12void
13read_mesh(const Parameters::Mesh &mesh_parameters,
14 const bool restart,
15 const ConditionalOStream &pcout,
16 parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
17 const Parameters::Lagrangian::BCDEM &bc_params)
18{
19 pcout << "Reading triangulation" << std::endl;
20
21 attach_grid_to_triangulation(triangulation, mesh_parameters);
22
23 // Check if there are periodic boundaries
24 for (unsigned int i_bc = 0; i_bc < bc_params.bc_types.size(); ++i_bc)
25 {
26 if (bc_params.bc_types[i_bc] ==
27 Parameters::Lagrangian::BCDEM::BoundaryType::periodic)
28 match_periodic_boundaries(triangulation, bc_params);
29 }
30
31 if (!restart)
32 {
33 if (mesh_parameters.refine_until_target_size)
34 {
35 double minimal_cell_size =
36 GridTools::minimal_cell_diameter(triangulation);
37 double target_size = mesh_parameters.target_size;
38 unsigned int number_refinement = static_cast<unsigned int>(
39 floor(std::log(minimal_cell_size / target_size) / std::log(2)));
40 pcout << "Automatically refining grid until target size: "
41 << target_size << std::endl;
42 triangulation.refine_global(number_refinement);
43 pcout << "Mesh was automatically refined: " << number_refinement
44 << " times" << std::endl;
45 }
46 else
47 {
48 const int initial_refinement = mesh_parameters.initial_refinement;
49 triangulation.refine_global(initial_refinement);
50 }
51 }
52
53 pcout << std::endl << "Finished reading triangulation" << std::endl;
54}

56template <int dim, int spacedim>
57void
58match_periodic_boundaries(Triangulation<dim, spacedim> &triangulation,
59 const Parameters::Lagrangian::BCDEM &bc_param)
60{
61 std::vector<GridTools::PeriodicFacePair<
62 typename Triangulation<dim, spacedim>::cell_iterator>>
63 periodicity_vector;
64
65 GridTools::collect_periodic_faces(triangulation,
66 bc_param.periodic_boundary_0,
67 bc_param.periodic_boundary_1,
68 bc_param.periodic_direction,
69 periodicity_vector);
70 triangulation.add_periodicity(periodicity_vector);
71}
