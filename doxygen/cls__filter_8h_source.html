79/**
80 * @brief Filter phase indicator with a hyperbolic tangent function
81 * for CLS simulations.
82 *
83 * The filtered phase is defined as:
84 *
85 * \f$\phi' = 0.5 \tanh(\beta(\phi-0.5)) + 0.5\f$
86 *
87 * where \f$\phi'\f$ is the filtered phase indicator value and \f$\beta\f$
88 * is a model parameter that enables sharper definition when increased.
89 *
90 * The filter allows a sharper definition of the interface and clamps the value
91 * of the phase indicator between 0 and 1.
92 */
93class ConservativeLevelSetTanhFilter : public ConservativeLevelSetFilterBase
94{
95public:
96 /**
97 * @brief Constructor of the hyperbolic tangent phase indicator filter.
98 *
99 * @param[in] beta Value of the \f$\beta\f$ parameter modulating the
100 * interface sharpness.
101 */
102 ConservativeLevelSetTanhFilter(const double beta)
103 : beta(beta)
104 {}
105
106 /**
107 * @brief Computes the value of the filtered phase indicator.
108 *
109 * @param[in] unfiltered_phase Value of the phase indicator before applying
110 * the filter.
111 *
112 * @return Value of the computed phase indicator after applying the filter.
113 */
114 virtual double
115 filter_phase(const double &unfiltered_phase) override
116 {
117 return 0.5 * tanh(beta * (unfiltered_phase - 0.5)) + 0.5;
118 }
119
120private:
121 /**
122 * User-defined parameter that influences the sharpness of the interface
123 * between fluids. A greater value leads to a sharper interface. The
124 * recommended value is \f$\beta=20\f$.
125 */
126 const double beta;
127};
