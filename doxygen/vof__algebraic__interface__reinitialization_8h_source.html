41 * @param[in] p_simulation_parameters Simulation parameters.
42 *
43 * @param[in] p_pcout Parallel cout used to print the information.
44 *
45 * @param[in] p_triangulation Distributed mesh information.
46 *
47 * @param[in] p_simulation_control SimulationControl object.

49 * @param[in, out] p_subequations_interface Subequations interface object used

51 * current one.

53 VOFAlgebraicInterfaceReinitialization(
54 const SimulationParameters<dim> &p_simulation_parameters,
55 const ConditionalOStream &p_pcout,
56 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
57 p_triangulation,
58 const std::shared_ptr<SimulationControl> &p_simulation_control,
59 VOFSubequationsInterface<dim> &p_subequations_interface)
60 : PhysicsNonlinearSubequationsSolver<dim, GlobalVectorType>(
61 p_simulation_parameters.vof_subequations_non_linear_solvers.at(
62 VOFSubequationsID::algebraic_interface_reinitialization),
63 p_pcout)
64 , subequation_id(VOFSubequationsID::algebraic_interface_reinitialization)
65 , subequations_interface(p_subequations_interface)
66 , simulation_parameters(p_simulation_parameters)
67 , simulation_control(p_simulation_control)
68 , triangulation(p_triangulation)
69 , dof_handler(std::make_shared<DoFHandler<dim>>(*this->triangulation))
70 , subequation_verbosity(p_simulation_parameters.multiphysics.vof_parameters
71 .regularization_method.verbosity)
72 {
73 if (this->simulation_parameters.mesh.simplex)
74 {
75 // For simplex meshes
76 this->fe = std::make_shared<FE_SimplexP<dim>>(
77 this->simulation_parameters.fem_parameters.VOF_order);
78 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
79 this->cell_quadrature =
80 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
81 }
82 else
83 {
84 // Usual case, for quad/hex meshes
85 this->fe = std::make_shared<FE_Q<dim>>(
86 this->simulation_parameters.fem_parameters.VOF_order);
87 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
88 this->cell_quadrature =
89 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
90 }
91
92 // Ensure that the shared pointer is properly allocated
93 this->present_solution = std::make_shared<GlobalVectorType>();
94 }

239 /**
240 * @brief Computes mesh-dependant diffusivity coefficient value.
241 *
242 * @param[in] min_cell_size Smallest cell's measure.
243 *
244 * @return Constant diffusivity coefficient value for a given mesh.
245 */
246 inline double
247 compute_diffusivity(const double min_cell_size) const
248 {
249 const double multiplier =
250 this->simulation_parameters.multiphysics.vof_parameters
251 .regularization_method.algebraic_interface_reinitialization
252 .diffusivity_multiplier;
253 const double power =
254 this->simulation_parameters.multiphysics.vof_parameters
255 .regularization_method.algebraic_interface_reinitialization
256 .diffusivity_power;
257 return multiplier * std::pow(min_cell_size, power);
258 }

260 /**
261 * @brief Computes algebraic reinitialization time-step value with
262 * user-imposed CFL and minimum cell size.
263 *
264 * @return Algebraic reinitialization time-step.
265 */
266 inline double
267 compute_time_step()
268 {
269 // Get CFL value
270 const double cfl =
271 this->simulation_parameters.multiphysics.vof_parameters
272 .regularization_method.algebraic_interface_reinitialization
273 .reinitialization_cfl;
274
275 // Get the minimum cell size
276 const double h_min =
277 identify_minimum_cell_size(*this->mapping,
278 *this->dof_handler,
279 *this->cell_quadrature,
280 this->triangulation->get_mpi_communicator());
281
282 return h_min * cfl;
283 }

321 /**
322 * @brief Indicate if the algebraic reinitialization should continue by
323 * checking if at least one of the two stop criteria (steady-state criterion
324 * or maximum number of reinitialization steps) is met.
325 *
326 * @param[in] time_step_inv Inverse of the current reinitialization time-step.
327 *
328 * @param[in] step_number Algebraic interface reinitialization step number.
329 *
330 * @return Boolean indicating if the algebraic reinitialization should
331 * continue
332 */
333 inline bool
334 continue_iterating(const double time_step_inv, const unsigned int step_number)
335 {
336 if (step_number == 1)
337 {
338 return (step_number <
339 this->simulation_parameters.multiphysics.vof_parameters
340 .regularization_method.algebraic_interface_reinitialization
341 .max_steps_number +
342 1);
343 }
344 else
345 {
346 // Get the stop criterion of the pseudo-time-stepping scheme
347 double steady_state_criterion =
348 this->simulation_parameters.multiphysics.vof_parameters
349 .regularization_method.algebraic_interface_reinitialization
350 .steady_state_criterion;
351
352 // Evaluate the solution difference between the 2 last solutions
353 auto solution_diff = local_evaluation_point;
354 solution_diff -= previous_local_evaluation_point;
355
356 // Evaluate the current steady-state criterion value
357 double stop_criterion = time_step_inv * solution_diff.l2_norm();
358
359 if (this->subequation_verbosity == Parameters::Verbosity::extra_verbose)
360 {
361 this->pcout
362 << "Algebraic reinitialization solution norm difference = "
363 << solution_diff.l2_norm() << std::endl;
364 this->pcout
365 << "Algebraic reinitialization steady-state criterion value = "
366 << stop_criterion << std::endl;
367 this->pcout
368 << "Algebraic reinitialization fixed steady-state criterion = "
369 << steady_state_criterion << std::endl;
370 }
371
372 return ((stop_criterion > steady_state_criterion) &&
373 (step_number <
374 this->simulation_parameters.multiphysics.vof_parameters
375 .regularization_method.algebraic_interface_reinitialization
376 .max_steps_number +
377 1));
378 }
379 }
