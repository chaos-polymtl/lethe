43 * @param[in] p_pcout Parallel cout used to print the information.
44 *
45 * @param[in] p_triangulation Distributed mesh information.
46 *
47 * @param[in] p_simulation_control SimulationControl object.

53 VOFAlgebraicInterfaceReinitialization(
54 const SimulationParameters<dim> &p_simulation_parameters,
55 const ConditionalOStream &p_pcout,
56 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
57 p_triangulation,
58 const std::shared_ptr<SimulationControl> &p_simulation_control,
59 VOFSubequationsInterface<dim> &p_subequations_interface)
60 : PhysicsNonlinearSubequationsSolver<dim, GlobalVectorType>(
61 p_simulation_parameters.vof_subequations_non_linear_solvers.at(
62 VOFSubequationsID::algebraic_interface_reinitialization),
63 p_pcout)
64 , subequation_id(VOFSubequationsID::algebraic_interface_reinitialization)
65 , subequations_interface(p_subequations_interface)
66 , simulation_parameters(p_simulation_parameters)
67 , simulation_control(p_simulation_control)
68 , triangulation(p_triangulation)
69 , dof_handler(std::make_shared<DoFHandler<dim>>(*this->triangulation))
70 , subequation_verbosity(p_simulation_parameters.multiphysics.vof_parameters
71 .regularization_method.verbosity)
72 {
73 if (this->simulation_parameters.mesh.simplex)
74 {
75 // For simplex meshes
76 this->fe = std::make_shared<FE_SimplexP<dim>>(
77 this->simulation_parameters.fem_parameters.VOF_order);
78 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
79 this->cell_quadrature =
80 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
81 }
82 else
83 {
84 // Usual case, for quad/hex meshes
85 this->fe = std::make_shared<FE_Q<dim>>(
86 this->simulation_parameters.fem_parameters.VOF_order);
87 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
88 this->cell_quadrature =
89 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
90 }
91
92 // Ensure that the shared pointer is properly allocated
93 this->present_solution = std::make_shared<GlobalVectorType>();
94 }

222 /**
223 * @brief Computes mesh-dependant diffusivity coefficient value.
224 *
225 * @param[in] min_cell_size Smallest cell's measure.
226 *
227 * @return Constant diffusivity coefficient value for a given mesh.
228 */
229 inline double
230 compute_diffusivity(const double min_cell_size) const
231 {
232 const double multiplier =
233 this->simulation_parameters.multiphysics.vof_parameters
234 .regularization_method.algebraic_interface_reinitialization
235 .diffusivity_multiplier;
236 const double power =
237 this->simulation_parameters.multiphysics.vof_parameters
238 .regularization_method.algebraic_interface_reinitialization
239 .diffusivity_power;
240 return multiplier * std::pow(min_cell_size, power);
241 }

243 /**
244 * @brief Computes algebraic reinitialization time-step value with
245 * user-imposed CFL and minimum cell size.
246 *
247 * @return Algebraic reinitialization time-step.
248 */
249 inline double
250 compute_time_step()
251 {
252 // Get CFL value
253 const double cfl =
254 this->simulation_parameters.multiphysics.vof_parameters
255 .regularization_method.algebraic_interface_reinitialization
256 .reinitialization_cfl;
257
258 // Get the minimum cell size
259 const double h_min =
260 identify_minimum_cell_size(*this->mapping,
261 *this->dof_handler,
262 *this->cell_quadrature,
263 this->triangulation->get_mpi_communicator());
264
265 return h_min * cfl;
266 }

311 /**
312 * @brief Indicate if the algebraic reinitialization should continue by
313 * checking if at least one of the two stop criteria (steady-state criterion
314 * or maximum number of reinitialization steps) is met.
315 *
316 * @param[in] time_step_inv Inverse of the current reinitialization time-step.
317 *
318 * @param[in] step_number Algebraic interface reinitialization step number.
319 *
320 * @return Boolean indicating if the algebraic reinitialization should
321 * continue
322 */
323 inline bool
324 continue_iterating(const double time_step_inv, const unsigned int step_number)
325 {
326 if (step_number == 1)
327 {
328 return (step_number <
329 this->simulation_parameters.multiphysics.vof_parameters
330 .regularization_method.algebraic_interface_reinitialization
331 .max_steps_number +
332 1);
333 }
334 else
335 {
336 // Get the stop criterion of the pseudo-time-stepping scheme
337 double steady_state_criterion =
338 this->simulation_parameters.multiphysics.vof_parameters
339 .regularization_method.algebraic_interface_reinitialization
340 .steady_state_criterion;
341
342 // Evaluate the solution difference between the 2 last solutions
343 auto solution_diff = local_evaluation_point;
344 solution_diff -= previous_local_evaluation_point;
345
346 // Evaluate the current steady-state criterion value
347 double stop_criterion = time_step_inv * solution_diff.l2_norm();
348
349 if (this->subequation_verbosity == Parameters::Verbosity::extra_verbose)
350 {
351 this->pcout
352 << "Algebraic reinitialization solution norm difference = "
353 << solution_diff.l2_norm() << std::endl;
354 this->pcout
355 << "Algebraic reinitialization steady-state criterion value = "
356 << stop_criterion << std::endl;
357 this->pcout
358 << "Algebraic reinitialization fixed steady-state criterion = "
359 << steady_state_criterion << std::endl;
360 }
361
362 return ((stop_criterion > steady_state_criterion) &&
363 (step_number <
364 this->simulation_parameters.multiphysics.vof_parameters
365 .regularization_method.algebraic_interface_reinitialization
366 .max_steps_number +
367 1));
368 }
369 }
