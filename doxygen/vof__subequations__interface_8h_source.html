37/**

263 /**
264 * @brief Get the string associated with a specific subequation.
265 *
266 * @param[in] subequation_id Identifier associated with a specific
267 * subequation.
268 *
269 * @return String associated with the specified subequation.
270 *
271 * @note This is used for printing purposes.
272 */
273 std::string
274 get_subequation_string(const VOFSubequationsID &subequation_id)
275 {
276 AssertThrow((std::find(this->active_subequations.begin(),
277 this->active_subequations.end(),
278 subequation_id) != this->active_subequations.end()),
279 ExcInternalError());
280
281 std::string subequation_string;
282
283 if (subequation_id == VOFSubequationsID::phase_gradient_projection)
284 subequation_string = "VOF phase fraction gradient L2 projection";
285 else if (subequation_id == VOFSubequationsID::curvature_projection)
286 subequation_string = "VOF curvature L2 projection";
287 else if (subequation_id ==
288 VOFSubequationsID::algebraic_interface_reinitialization)
289 subequation_string = "VOF algebraic interface reinitialization";
290 else
291 throw(std::invalid_argument("Invalid VOFSubequationID. Options are: \n"
292 " <phase_gradient_projection>\n"
293 " <curvature_projection>\n"
294 " <algebraic_interface_reinitialization>"));
295
296 return subequation_string;
297 }

341 /**
342 * @brief Set a new VOF filtered solution vector and DoFHandler for the
343 * subequations to be solved and reset their validity.
344 *
345 * @param[in] vof_filtered_solution_vector Reference to the filtered VOF
346 * solution vector associated with the subequations to be solved.
347 *
348 * @param[in] dof_handler_vof Reference to the DoFHandler associated with the
349 * specified VOF filtered solution.
350 *
351 * @note At the moment, the VOF filtered solution is only used in
352 * VOFPhaseGradientProjection.
353 */
354 void
355 set_vof_filtered_solution_and_dof_handler(
356 const GlobalVectorType &vof_filtered_solution_vector,
357 const DoFHandler<dim> &dof_handler_vof)
358 {
359 // Reset validity map
360 reset_subequations_solutions_validity();
361
362 // Set solution values and DoFHandler
363 this->vof_filtered_solution_vector = vof_filtered_solution_vector;
364 this->dof_handler_vof = std::cref(dof_handler_vof);
365 }

367 /**
368 * @brief Set a new VOF solution vector and DoFHandler for the subequations to
369 * be solved.
370 *
371 * @param[in] vof_solution_vector VOF solution vector associated with the
372 * subequations to be solved.
373 *
374 * @note At the moment, the VOF unfiltered solution is only used in
375 * VOFAlgebraicInterfaceReinitialization.
376 */
377 void
378 set_vof_solution(const GlobalVectorType &vof_solution_vector)
379 {
380 // By checking if the validity map has been reset, we make sure that the VOF
381 // solution is consistent with the set filtered solution and the DoFHandler.
382 AssertThrow(validity_map_has_been_reset(), ValidityMapNotReset());
383 this->vof_solution_vector = vof_solution_vector;
384 }

386 /**

402 /**
403 * @brief Check if all booleans in subequations_solutions_validity have been
404 * reset to @p false.
405 *
406 * @return
407 * - @p true if all booleans in the map are set to @p false or;
408 * - @p false if at least one of the booleans is set to @p true
409 */
410 bool
411 validity_map_has_been_reset()
412 {
413 for (const auto &pair : subequations_solutions_validity)
414 {
415 if (pair.second == true)
416 return false;
417 }
418 return true;
419 }

465 /**
466 * Booleans indicating if the subequation has been solved with the set VOF
467 * filtered and unfiltered solution vectors set respectively with
468 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler
469 * and VOFSubequationsInterface<dim>::set_vof_solution.
470 *
471 * During initialization
472 * (VOFSubequationsInterface<dim>::initialize_subequations),
473 * all boolean are set to @p false.
474 *
475 * We start by setting a VOF filtered solution and DOFHandler
476 * (VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler),
477 * and then start solving equations making sure that dependencies are met
478 * (VOFPhaseGradientProjection<dim>::check_dependencies_validity,
479 * VOFCurvatureProjection<dim>::check_dependencies_validity, and
480 * VOFAlgebraicInterfaceReinitialization<dim>::check_dependencies_validity).
481 * For instance, to solve the curvature projection subequation, the phase
482 * gradient projection subequation has to be solved before as the projected
483 * phase gradient is used in the curvature projection subequation.
484 *
485 * A subequation in the validity map is to @p true only when it is solved and
486 * it remains valid until a new VOF filtered solution and/or DOFHandler are
487 * set through
488 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler.
489 *
490 * This validation mechanism allows, to check if:
491 * - A new VOF filtered/unfiltered solution has been set;
492 * - All dependencies of the subequation have been solved;
493 * - The current solution of the subequation is valid and avoid unnecessarily
494 * solving it once more.
495 *
496 * @note
497 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler
498 * also resets the validity map values to @p false.
499 *
500 * @note VOFAlgebraicInterfaceReinitialization also requires a VOF unfiltered
501 * solution. The solution is set by calling
502 * VOFSubequationsInterface<dim>::set_vof_solution, right after
503 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler.
504 */
505 std::map<VOFSubequationsID, bool> subequations_solutions_validity;
506};
