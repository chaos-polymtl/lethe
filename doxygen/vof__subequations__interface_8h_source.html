263 /**
264 * @brief Get the string associated with a specific subequation.
265 *
266 * @param[in] subequation_id Identifier associated with a specific
267 * subequation.
268 *
269 * @return String associated with the specified subequation.
270 *
271 * @note This is used for printing purposes.
272 */
273 std::string
274 get_subequation_string(const VOFSubequationsID &subequation_id)
275 {
276 AssertThrow((std::find(this->active_subequations.begin(),
277 this->active_subequations.end(),
278 subequation_id) != this->active_subequations.end()),
279 ExcInternalError());
280
281 std::string subequation_string;
282
283 if (subequation_id == VOFSubequationsID::phase_gradient_projection)
284 subequation_string = "VOF phase fraction gradient L2 projection";
285 else if (subequation_id == VOFSubequationsID::curvature_projection)
286 subequation_string = "VOF curvature L2 projection";
287 else if (subequation_id ==
288 VOFSubequationsID::algebraic_interface_reinitialization)
289 subequation_string = "VOF algebraic interface reinitialization";
290 else
291 throw(std::invalid_argument("Invalid VOFSubequationID. Options are: \n"
292 " <phase_gradient_projection>\n"
293 " <curvature_projection>\n"
294 " <algebraic_interface_reinitialization>"));
295
296 return subequation_string;
297 }

341 /**
342 * @brief Set a new VOF filtered solution vector and DoFHandler for the
343 * subequations to be solved and reset their validity.
344 *
345 * @param[in] vof_filtered_solution_vector Reference to the filtered VOF
346 * solution vector associated with the subequations to be solved.
347 *
348 * @param[in] dof_handler_vof Reference to the DoFHandler associated with the
349 * specified VOF filtered solution.
350 *
351 * @note At the moment, the VOF filtered solution is only used in
352 * VOFPhaseGradientProjection.
353 */
354 void
355 set_vof_filtered_solution_and_dof_handler(
356 const GlobalVectorType &vof_filtered_solution_vector,
357 const DoFHandler<dim> &dof_handler_vof)
358 {
359 // Reset validity map
360 reset_subequations_solutions_validity();
361
362 // Set solution values and DoFHandler
363 this->vof_filtered_solution_vector =
364 std::cref(vof_filtered_solution_vector);
365 this->dof_handler_vof = std::cref(dof_handler_vof);
366 }

368 /**
369 * @brief Set a new VOF solution vector and DoFHandler for the subequations to
370 * be solved.
371 *
372 * @param[in] vof_solution_vector VOF solution vector associated with the
373 * subequations to be solved.
374 *
375 * @note At the moment, the VOF unfiltered solution is only used in
376 * VOFAlgebraicInterfaceReinitialization.
377 */
378 void
379 set_vof_solution(const GlobalVectorType &vof_solution_vector)
380 {
381 // By checking if the validity map has been reset, we make sure that the VOF
382 // solution is consistent with the set filtered solution and the DoFHandler.
383 AssertThrow(validity_map_has_been_reset(), ValidityMapNotReset());
384 this->vof_solution_vector = std::cref(vof_solution_vector);
385 }

387 /**
388 * @brief Set the solution as valid after solving it.
389 *
390 * @param[in] subequation_id Identifier associated with a specific

395 {
396 AssertThrow((std::find(this->active_subequations.begin(),
397 this->active_subequations.end(),
398 subequation_id) != this->active_subequations.end()),
399 ExcInternalError());

403 /**
404 * @brief Check if all booleans in subequations_solutions_validity have been
405 * reset to @p false.
406 *
407 * @return
408 * - @p true if all booleans in the map are set to @p false or;
409 * - @p false if at least one of the booleans is set to @p true
410 */
411 bool
412 validity_map_has_been_reset()
413 {
414 for (const auto &pair : subequations_solutions_validity)
415 {
416 if (pair.second == true)
417 return false;
418 }
419 return true;
420 }

468 /**
469 * Booleans indicating if the subequation has been solved with the set VOF
470 * filtered and unfiltered solution vectors set respectively with
471 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler
472 * and VOFSubequationsInterface<dim>::set_vof_solution.
473 *
474 * During initialization
475 * (VOFSubequationsInterface<dim>::initialize_subequations),
476 * all boolean are set to @p false.
477 *
478 * We start by setting a VOF filtered solution and DOFHandler
479 * (VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler),
480 * and then start solving equations making sure that dependencies are met
481 * (VOFPhaseGradientProjection<dim>::check_dependencies_validity,
482 * VOFCurvatureProjection<dim>::check_dependencies_validity, and
483 * VOFAlgebraicInterfaceReinitialization<dim>::check_dependencies_validity).
484 * For instance, to solve the curvature projection subequation, the phase
485 * gradient projection subequation has to be solved before as the projected
486 * phase gradient is used in the curvature projection subequation.
487 *
488 * A subequation in the validity map is to @p true only when it is solved and
489 * it remains valid until a new VOF filtered solution and/or DOFHandler are
490 * set through
491 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler.
492 *
493 * This validation mechanism allows, to check if:
494 * - A new VOF filtered/unfiltered solution has been set;
495 * - All dependencies of the subequation have been solved;
496 * - The current solution of the subequation is valid and avoid unnecessarily
497 * solving it once more.
498 *
499 * @note
500 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler
501 * also resets the validity map values to @p false.
502 *
503 * @note VOFAlgebraicInterfaceReinitialization also requires a VOF unfiltered
504 * solution. The solution is set by calling
505 * VOFSubequationsInterface<dim>::set_vof_solution, right after
506 * VOFSubequationsInterface<dim>::set_vof_filtered_solution_and_dof_handler.
507 */
508 std::map<VOFSubequationsID, bool> subequations_solutions_validity;
509};
