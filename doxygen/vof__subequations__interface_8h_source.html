16/**
17 * @brief Interface for secondary equations (subequations) solved within
18 * the VOF auxiliary physics.
19 *
20 * @tparam dim Number of dimensions of the problem.
21 */
22template <int dim>
23class VOFSubequationsInterface
24{
25public:
26 /**
27 * @brief Constructor of an interface for subequations that require the use of
28 * a solver within the span of the VOF auxiliary physics' resolution.
29 *
30 * @param[in] p_simulation_parameters Simulation parameters.
31 *
32 * @param[in] p_pcout Parallel cout used to print the information.
33 *
34 * @param[in] p_triangulation Distributed mesh information.
35 *
36 * @param[in] p_simulation_control SimulationControl object.
37 *
38 * @param[in] p_multiphysics_interface Multiphysics interface object pointer
39 * used to get information from physics.
40 */
41 VOFSubequationsInterface(
42 const SimulationParameters<dim> &p_simulation_parameters,
43 const ConditionalOStream &p_pcout,
44 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
45 &p_triangulation,
46 const std::shared_ptr<SimulationControl> &p_simulation_control,
47 MultiphysicsInterface<dim> *p_multiphysics_interface);
48
49 /**
50 * @brief Default destructor.

52 ~VOFSubequationsInterface() = default;

54 /**
55 * @brief Setup the DofHandler and the degree of freedom associated with the
56 * subequations.
57 */
58 void
59 setup_dofs()
60 {
61 for (const auto &subequation : this->subequations)
62 {
63 subequation.second->setup_dofs();
64 }
65 };
66
67 /**
68 * @brief Setup the DofHandler and the degree of freedom associated with the
69 * physics.
70 *
71 * @param[in] subequation_id Identifier associated with the subequation wished
72 * to solve.
73 */
74 void
75 setup_specific_subequation_dofs(const VOFSubequationsID &subequation_id)
76 {
77 AssertThrow((std::find(this->active_subequations.begin(),
78 this->active_subequations.end(),
79 subequation_id) != this->active_subequations.end()),
80 ExcInternalError());
81
82 this->subequations.find(subequation_id)->second->setup_dofs();
83 };
84
85 /**
86 * @brief Call solving method of active subequations.
87 */
88 void
89 solve()
90 {
91 for (const auto &subequation : this->subequations)
92 {
93 subequation.second->solve();
94 }
95 }
96
97 /**
98 * @brief Call solving method for a specific subequation.

103 void
104 solve_specific_subequation(const VOFSubequationsID &subequation_id)

109 ExcInternalError());

162 /**
163 * @brief Get the string associated with a specific subequation.
164 *
165 * @param[in] subequation_id Identifier associated with a specific
166 * subequation.
167 *
168 * @return String associated with the specified subequation.
169 *
170 * @note This is used for printing purposes.
171 */
172 std::string
173 get_subequation_string(const VOFSubequationsID &subequation_id)
174 {
175 AssertThrow((std::find(this->active_subequations.begin(),
176 this->active_subequations.end(),
177 subequation_id) != this->active_subequations.end()),
178 ExcInternalError());
179
180 std::string subequation_string;
181
182 if (subequation_id == VOFSubequationsID::phase_gradient_projection)
183 subequation_string = "VOF phase fraction gradient L2 projection";
184 else if (subequation_id == VOFSubequationsID::curvature_projection)
185 subequation_string = "VOF curvature L2 projection";
186 else if (subequation_id ==
187 VOFSubequationsID::algebraic_interface_reinitialization)
188 subequation_string = "VOF algebraic interface reinitialization";
189 else
190 throw(std::invalid_argument("Invalid VOFSubequationID. Options are: \n"
191 " <phase_gradient_projection>\n"
192 " <curvature_projection>\n"
193 " <algebraic_interface_reinitialization>"));
194
195 return subequation_string;
196 }
197

199 * @brief Set the DoFHandler associated with a specific subequation in the

218 /**
219 * @brief Set the solution associated with a specific subequation in the

228 void

230 GlobalVectorType *solution_vector)
231 {
232 AssertThrow((std::find(this->active_subequations.begin(),
233 this->active_subequations.end(),
234 subequation_id) != this->active_subequations.end()),
235 ExcInternalError());
236 this->subequations_solutions[subequation_id] = solution_vector;
237 }
