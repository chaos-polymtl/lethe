20template <int dim, typename VectorType, typename DofsType>
21void
22AverageVelocities<dim, VectorType, DofsType>::calculate_average_velocities(
23 const VectorType &local_evaluation_point,
24 const Parameters::PostProcessing &post_processing,
25 const double &current_time,
26 const double &time_step)
27{
28 const double epsilon = 1e-6;
29 const double initial_time =
30 post_processing.initial_time_for_average_velocities;
31 dt = time_step;
32
33 // When averaging velocities begins
34 if (current_time >= (initial_time - epsilon))
35 {
36 if (!has_started_averaging)
37 {
38 has_started_averaging = true;
39 real_initial_time = current_time;
40
41 // Store the first dt value in case dt varies.
42 dt_0 = dt;
43 }
44 // Calculate (u*dt) at each time step and accumulate the values
45 velocity_dt.equ(dt, local_evaluation_point);
46 sum_velocity_dt += velocity_dt;
47
48 // Get the inverse of the time since the beginning of the time averaging
49 total_time_for_average = (current_time - real_initial_time) + dt_0;
50 inv_range_time = 1. / total_time_for_average;
51
52 // Calculate the average velocities.
53 average_velocities.equ(inv_range_time, sum_velocity_dt);
54
55 this->calculate_reynolds_stresses(local_evaluation_point);
56 }
57}

60template <int dim, typename VectorType, typename DofsType>
61void
62AverageVelocities<dim, VectorType, DofsType>::update_average_velocities()
63{
64 // Use the inverse of the time since the beginning of the time averaging to
65 // reevaluate the average velocity field and Reynolds stress.
66
67 if (total_time_for_average > 1e-16)
68 {
69 inv_range_time = 1.0 / total_time_for_average;
70 // Calculate the average velocities.
71 average_velocities.equ(inv_range_time, sum_velocity_dt);
72 reynolds_normal_stresses.equ(inv_range_time,
73 sum_reynolds_normal_stress_dt);
74 reynolds_shear_stresses.equ(inv_range_time, sum_reynolds_shear_stress_dt);
75 }
76}

78// Since Trilinos vectors and block vectors data doesn't have the same
79// structure, those vectors need to be processed in different ways.
80// Same about the dimension, where 2D solutions are stored in (u,v,p) groups
81// for vectors and [(u,v)][p] for block vectors and where 3D solutions
82// are stored in (u,v,w,p) groups for vectors and [(u,v,w)][p] for
83// block vectors.
84template <int dim, typename VectorType, typename DofsType>
85void
86AverageVelocities<dim, VectorType, DofsType>::calculate_reynolds_stresses(
87 const VectorType &local_evaluation_point)
88{
89 if constexpr (std::is_same_v<VectorType, GlobalVectorType> ||
90 std::is_same_v<VectorType, GlobalBlockVectorType>)
91 {
92 unsigned int begin_index, end_index;
93
94 // As explained in the comment above, about the structure of Trilinos
95 // vectors, a function is required to get the index of the turbulence
96 // kinetic energy. Trilinos vector : rss = (<u'u'>,<v'v'>,<w'w'>,k) =>
97 // k_index = i + dim Trilinos blocks : rss = [(<u'u'>,<v'v'>,<w'w'>)][k])
98 // => k_index = i / dim The k_index lambda is later changed for block
99 // vectors.
100 unsigned int (*k_index)(unsigned int) = [](unsigned int i) {
101 return i + dim;
102 };
103
104 const GlobalVectorType *local_solution, *local_average;
105 GlobalVectorType *rns_dt, *rss_dt, *k_dt;
106
107 if constexpr (std::is_same_v<VectorType, GlobalVectorType>)
108 {
109 if constexpr (std::is_same_v<VectorType,
110 dealii::TrilinosWrappers::MPI::Vector>)
111 {
112 begin_index = local_evaluation_point.local_range().first;
113 end_index = local_evaluation_point.local_range().second;
114 }
115 else if constexpr (std::is_same_v<VectorType,
116 dealii::LinearAlgebra::distributed::
117 Vector<double>>)
118 {
119 begin_index =
120 local_evaluation_point.get_partitioner()->local_range().first;
121 end_index =
122 local_evaluation_point.get_partitioner()->local_range().second;
123 }
124
125 else
126 {
127 AssertThrow(false, ExcNotImplemented());
128 }
129 local_solution = &local_evaluation_point;
130 local_average = &average_velocities;
131 rns_dt = &reynolds_normal_stress_dt;
132 rss_dt = &reynolds_shear_stress_dt;
133 k_dt = &reynolds_normal_stress_dt;
134 }
135 else if constexpr (std::is_same_v<VectorType, GlobalBlockVectorType>)
136 {
137 if constexpr (std::is_same_v<
138 VectorType,
139 dealii::TrilinosWrappers::MPI::BlockVector>)
140 {
141 begin_index = local_evaluation_point.block(0).local_range().first;
142 end_index = local_evaluation_point.block(0).local_range().second;
143 }
144 else if constexpr (std::is_same_v<VectorType,
145 dealii::LinearAlgebra::distributed::
146 BlockVector<double>>)
147 {
148 begin_index = local_evaluation_point.block(0)
149 .get_partitioner()
150 ->local_range()
151 .first;
152 end_index = local_evaluation_point.block(0)
153 .get_partitioner()
154 ->local_range()
155 .second;
156 }
157 else
158 {
159 }
160 local_solution = &local_evaluation_point.block(0);
161 local_average = &average_velocities.block(0);
162 rns_dt = &reynolds_normal_stress_dt.block(0);
163 rss_dt = &reynolds_shear_stress_dt.block(0);
164 k_dt = &reynolds_normal_stress_dt.block(1);
165 k_index = [](unsigned int i) { return i / dim; };
166 }
167
168 for (unsigned int i = begin_index; i < end_index; i += n_dofs_per_vertex)
169 {
170 // u'u'*dt
171 (*rns_dt)[i] = ((*local_solution)[i] - (*local_average)[i]) *
172 ((*local_solution)[i] - (*local_average)[i]) * dt;
173
174 // v'v'*dt
175 (*rns_dt)[i + 1] =
176 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
177 ((*local_solution)[i + 1] - (*local_average)[i + 1]) * dt;
178
179 // u'v'*dt
180 (*rss_dt)[i] = ((*local_solution)[i] - (*local_average)[i]) *
181 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
182 dt;
183
184 // k*dt = 1/2(u'u'+v'v')*dt (turbulence kinetic energy)
185 // Note : k_dt and rns_dt are both pointers of
186 // reynolds_normal_stress_dt for Trilinos vector (not block vectors)
187 (*k_dt)[k_index(i)] = ((*rns_dt)[i] + (*rns_dt)[i + 1]) / 2;
188
189 if (dim == 3)
190 {
191 // w'w'*dt
192 (*rns_dt)[i + 2] =
193 ((*local_solution)[i + 2] - (*local_average)[i + 2]) *
194 ((*local_solution)[i + 2] - (*local_average)[i + 2]) * dt;
195
196 // v'w'*dt
197 (*rss_dt)[i + 1] =
198 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
199 ((*local_solution)[i + 2] - (*local_average)[i + 2]) * dt;
200
201 // w'u'*dt
202 (*rss_dt)[i + 2] =
203 ((*local_solution)[i + 2] - (*local_average)[i + 2]) *
204 ((*local_solution)[i] - (*local_average)[i]) * dt;
205
206 // k*dt = 1/2(u'u'+v'v'+w'w')*dt
207 (*k_dt)[k_index(i)] = (*k_dt)[k_index(i)] + (*rns_dt)[i + 2] / 2;
208 }
209 }
210 }
211
212#ifndef LETHE_USE_LDV
213 if constexpr (std::is_same_v<VectorType,
214 LinearAlgebra::distributed::Vector<double>>)
215 {
216 unsigned int begin_index, end_index;
217 unsigned int (*k_index)(unsigned int) = [](unsigned int i) {
218 return i + dim;
219 };
220 const LinearAlgebra::distributed::Vector<double> *local_solution,
221 *local_average;
222 LinearAlgebra::distributed::Vector<double> *rns_dt, *rss_dt, *k_dt;
223
224 local_evaluation_point.update_ghost_values();
225 begin_index =
226 local_evaluation_point.get_partitioner()->local_range().first;
227 end_index =
228 local_evaluation_point.get_partitioner()->local_range().second;
229 local_solution = &local_evaluation_point;
230 local_average = &average_velocities;
231 rns_dt = &reynolds_normal_stress_dt;
232 rss_dt = &reynolds_shear_stress_dt;
233 k_dt = &reynolds_normal_stress_dt;
234
235 for (unsigned int i = begin_index; i < end_index; i += n_dofs_per_vertex)
236 {
237 // u'u'*dt
238 (*rns_dt)[i] = ((*local_solution)[i] - (*local_average)[i]) *
239 ((*local_solution)[i] - (*local_average)[i]) * dt;
240
241 // v'v'*dt
242 (*rns_dt)[i + 1] =
243 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
244 ((*local_solution)[i + 1] - (*local_average)[i + 1]) * dt;
245
246 // u'v'*dt
247 (*rss_dt)[i] = ((*local_solution)[i] - (*local_average)[i]) *
248 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
249 dt;
250
251 // // k*dt = 1/2(u'u'+v'v')*dt (turbulence kinetic energy)
252 // // Note : k_dt and rns_dt are both pointers of
253 // // reynolds_normal_stress_dt
254 (*k_dt)[k_index(i)] = ((*rns_dt)[i] + (*rns_dt)[i + 1]) / 2;
255
256 if (dim == 3)
257 {
258 // w'w'*dt
259 (*rns_dt)[i + 2] =
260 ((*local_solution)[i + 2] - (*local_average)[i + 2]) *
261 ((*local_solution)[i + 2] - (*local_average)[i + 2]) * dt;
262
263 // v'w'*dt
264 (*rss_dt)[i + 1] =
265 ((*local_solution)[i + 1] - (*local_average)[i + 1]) *
266 ((*local_solution)[i + 2] - (*local_average)[i + 2]) * dt;
267
268 // w'u'*dt
269 (*rss_dt)[i + 2] =
270 ((*local_solution)[i + 2] - (*local_average)[i + 2]) *
271 ((*local_solution)[i] - (*local_average)[i]) * dt;
272
273 // k*dt = 1/2(u'u'+v'v'+w'w')*dt
274 (*k_dt)[k_index(i)] = (*k_dt)[k_index(i)] + (*rns_dt)[i + 2] / 2;
275 }
276 }
277 }
278#endif
279
280 // Sum of all reynolds stresses during simulation.
281 sum_reynolds_normal_stress_dt += reynolds_normal_stress_dt;
282 sum_reynolds_shear_stress_dt += reynolds_shear_stress_dt;
283
284 // Calculate the reynolds stresses.
285 reynolds_normal_stresses.equ(inv_range_time, sum_reynolds_normal_stress_dt);
286 reynolds_shear_stresses.equ(inv_range_time, sum_reynolds_shear_stress_dt);
287}

290template <int dim, typename VectorType, typename DofsType>
291void
292AverageVelocities<dim, VectorType, DofsType>::initialize_vectors(
293 const DofsType &locally_owned_dofs,
294 const DofsType &locally_relevant_dofs,
295 const unsigned int &dofs_per_vertex,
296 const MPI_Comm &mpi_communicator)
297{
298 // Save the number of dofs per vertex. If solution is in block vectors,
299 // this is the number of dofs about velocity, dim.
300 n_dofs_per_vertex = dofs_per_vertex;
301
302 // Reinitialisation of the average velocity and reynolds stress vectors
303 // to get the right length.
304 velocity_dt.reinit(locally_owned_dofs, mpi_communicator);
305 sum_velocity_dt.reinit(locally_owned_dofs, mpi_communicator);
306 average_velocities.reinit(locally_owned_dofs, mpi_communicator);
307 get_av.reinit(locally_owned_dofs, locally_relevant_dofs, mpi_communicator);
308
309 // Reinitialize independent components of stress tensor vectors.
310 reynolds_normal_stress_dt.reinit(locally_owned_dofs, mpi_communicator);
311 sum_reynolds_normal_stress_dt.reinit(locally_owned_dofs, mpi_communicator);
312 reynolds_normal_stresses.reinit(locally_owned_dofs, mpi_communicator);
313 get_rns.reinit(locally_owned_dofs, locally_relevant_dofs, mpi_communicator);
314
315 reynolds_shear_stress_dt.reinit(locally_owned_dofs, mpi_communicator);
316 sum_reynolds_shear_stress_dt.reinit(locally_owned_dofs, mpi_communicator);
317 reynolds_shear_stresses.reinit(locally_owned_dofs, mpi_communicator);
318 get_rss.reinit(locally_owned_dofs, locally_relevant_dofs, mpi_communicator);
319
320 // Initializing vector with locally_relevant_dofs because writing checkpoint
321 // needs to read ghost cells.
322 sum_velocity_dt_with_ghost_cells.reinit(locally_owned_dofs,
323 locally_relevant_dofs,
324 mpi_communicator);
325 sum_rns_dt_with_ghost_cells.reinit(locally_owned_dofs,
326 locally_relevant_dofs,
327 mpi_communicator);
328 sum_rss_dt_with_ghost_cells.reinit(locally_owned_dofs,
329 locally_relevant_dofs,
330 mpi_communicator);
331}

348template <int dim, typename VectorType, typename DofsType>
349void
350AverageVelocities<dim, VectorType, DofsType>::post_mesh_adaptation()
351{
352 if constexpr (std::is_same_v<VectorType,
353 LinearAlgebra::distributed::Vector<double>>)
354 {
355 // When the vector are deal.II vectors, the solution transfer expects a
356 // vector with ghost entries
357 solution_transfer_sum_velocity_dt.interpolate(
358 sum_velocity_dt_with_ghost_cells);
359 solution_transfer_sum_reynolds_normal_stress_dt.interpolate(
360 sum_rns_dt_with_ghost_cells);
361 solution_transfer_sum_reynolds_shear_stress_dt.interpolate(
362 sum_rss_dt_with_ghost_cells);
363
364 sum_velocity_dt = sum_velocity_dt_with_ghost_cells;
365 sum_reynolds_normal_stress_dt = sum_rns_dt_with_ghost_cells;
366 sum_reynolds_shear_stress_dt = sum_rss_dt_with_ghost_cells;
367 }
368 else
369 {
370 // When the vectors are trilinos vectors, the solution transfer expects a
371 // locally_owned vector
372 solution_transfer_sum_velocity_dt.interpolate(sum_velocity_dt);
373 solution_transfer_sum_reynolds_normal_stress_dt.interpolate(
374 sum_reynolds_normal_stress_dt);
375 solution_transfer_sum_reynolds_shear_stress_dt.interpolate(
376 sum_reynolds_shear_stress_dt);
377
378
379
380 sum_velocity_dt_with_ghost_cells = sum_velocity_dt;
381 sum_rns_dt_with_ghost_cells = sum_reynolds_normal_stress_dt;
382 sum_rss_dt_with_ghost_cells = sum_reynolds_shear_stress_dt;
383 }
384
385
386
387 update_average_velocities();
388}

390template <int dim, typename VectorType, typename DofsType>
391std::vector<const VectorType *>
392AverageVelocities<dim, VectorType, DofsType>::save(const std::string &prefix)
393{
394 sum_velocity_dt_with_ghost_cells = sum_velocity_dt;
395 sum_rns_dt_with_ghost_cells = sum_reynolds_normal_stress_dt;
396 sum_rss_dt_with_ghost_cells = sum_reynolds_shear_stress_dt;
397
398
399 std::vector<const VectorType *> av_set_transfer;
400 av_set_transfer.push_back(&sum_velocity_dt_with_ghost_cells);
401 av_set_transfer.push_back(&sum_rns_dt_with_ghost_cells);
402 av_set_transfer.push_back(&sum_rss_dt_with_ghost_cells);
403
404
405 std::string filename = prefix + ".averagevelocities";
406 std::ofstream output(filename.c_str());
407 output << "Average velocities" << std::endl;
408 output << "dt_0 " << dt_0 << std::endl;
409 output << "has_started_averaging_boolean " << has_started_averaging
410 << std::endl;
411 output << "Real_initial_time " << real_initial_time << std::endl;
412
413 return av_set_transfer;
