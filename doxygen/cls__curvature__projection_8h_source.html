20 /**
21 * @brief Constructor for the L2 projection of the CLS interface curvature.

23 * @param[in] p_simulation_parameters Simulation parameters.
24 *
25 * @param[in] p_pcout Parallel cout used to print the information.
26 *
27 * @param[in] p_triangulation Distributed mesh information.
28 *
29 * @param[in,out] p_subequations_interface Subequations interface object used
30 * to get information from other subequations and store information from the
31 * current one.
32 */
33 CLSCurvatureProjection(
34 const SimulationParameters<dim> &p_simulation_parameters,
35 const ConditionalOStream &p_pcout,
36 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
37 p_triangulation,
38 CLSSubequationsInterface<dim> &p_subequations_interface)
39 : CLSLinearSubequationsSolver<dim>(
40 CLSSubequationsID::curvature_projection,
41 p_simulation_parameters,
42 ((p_simulation_parameters.multiphysics.cls_parameters
43 .surface_tension_force.verbosity != Parameters::Verbosity::quiet) ||
44 ((p_simulation_parameters.multiphysics.cls_parameters
45 .regularization_method.algebraic_interface_reinitialization
46 .enable) &&
47 (p_simulation_parameters.multiphysics.cls_parameters
48 .regularization_method.verbosity !=
49 Parameters::Verbosity::quiet))) ?
50 Parameters::Verbosity::verbose :
51 Parameters::Verbosity::quiet, // Set to verbose if surface tension
52 // verbosity is enabled or if algebraic
53 // interface reinitialization is enabled
54 // and set to verbose
55 p_pcout,
56 p_triangulation,
57 p_subequations_interface)
58 {
59 if (this->simulation_parameters.mesh.simplex)
60 {
61 // For simplex meshes
62 this->fe = std::make_shared<FE_SimplexP<dim>>(
63 this->simulation_parameters.fem_parameters.CLS_order);
64 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
65 this->cell_quadrature =
66 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
67 }
68 else
69 {
70 // Usual case, for quad/hex meshes
71 this->fe = std::make_shared<FE_Q<dim>>(
72 this->simulation_parameters.fem_parameters.CLS_order);
73 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
74 this->cell_quadrature =
75 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
76 }
77 }
78
79 /**

95 check_dependencies_validity() override;
