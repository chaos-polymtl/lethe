70 * @param rpt_parameters General parameters for photon count calculation with the Monte Carlo method and setting up the grid

179template <int dim>
180class RPTFEMReconstruction
181{
182public:
183 /**
184 * @brief Constructor for the RPTFEMReconstruction
185 *
186 * @param rpt_parameters General parameters for photon count calculation with the Monte Carlo method and setting up the grid
187 * @param rpt_fem_reconstruction_parameters Parameters for the particle position reconstruction
188 * @param rpt_detector_parameters Parameters specifying information about the detectors for photon count calculation with the Monte Carlo method
189 */
190 RPTFEMReconstruction(Parameters::RPTParameters &rpt_parameters,
191 Parameters::RPTFEMReconstructionParameters
192 &rpt_fem_reconstruction_parameters,
193 Parameters::DetectorParameters &rpt_detector_parameters)
194 : fe(1)
195 , dof_handler(triangulation)
196 , mapping(FE_SimplexP<dim>(1))
197 , parameters(rpt_parameters)
198 , fem_reconstruction_parameters(rpt_fem_reconstruction_parameters)
199 , detector_parameters(rpt_detector_parameters)
200 , computing_timer(std::cout, TimerOutput::summary, TimerOutput::wall_times)
201 {}
202
203 /**
204 * @brief Reconstruction of the particle positions with the counts from the
205 * detectors.
206 */
207 void
208 rpt_fem_reconstruct();
209
210
211private:
212 /**
213 * @brief Set up the triangulation that covers the domain of the reactor.
214 */
215 void
216 setup_triangulation();
217
218 /**
219 * @brief Calculates the cost with the selected cost function of the found
220 * position and returns it.
221 *
222 * @param cell Iterator on active cells of the grid.
223 *
224 * @param reference_location Location of the particle in reference
225 * coordinates
226 *
227 * @param last_constraint Constraint on the particle's location
228 *
229 * @param experimental_count Experimental counts from all detectors for a
230 * given particle position
231 *
232 * @return cost It is calculated with the cost function mentioned in the
233 * parameter file.
234 */
235 double
236 calculate_cost(
237 const TriaActiveIterator<DoFCellAccessor<dim, dim, false>> &cell,
238 Vector<double> &reference_location,
239 const double &last_constraint,
240 std::vector<double> &experimental_count);
241
242 /**
243 * @brief Evaluates the error vector of the found location in reference
244 * coordinates and returns the error vector's L2 norm
245 *
246 * @param reference_location Location of the point in reference coordinates
247 *
248 * @param last_constraint 4th constraint on the location of the particle in
249 * reference coordinates
250 *
251 * @return norm_error_coordinates
252 */
253 double
254 calculate_reference_location_error(Vector<double> &reference_location,
255 const double &last_constraint);
256
257 /**
258 * @brief Searches in the reference space the position of the particle.
259 * If the found position respects the set tolerance, and the calculated
260 * cost is minimal it calculates and stores the position of the particle
261 * in system coordinates.
262 *
263 * @param count_from_all_detectors Contains the photon counts of all
264 * detectors for all vertices of a cell.
265 * @param experimental_count Experimental counts from all detectors for a
266 * given particle position
267 * @param cell Cell where the particle's position is being searched
268 * @param tol_reference_location Tolerance on the validity of the found
269 * location in the reference space
270 * @param max_cost Cost that has to be minimized in order to find a
271 * particle's real position
272 * @param reference_location Position of the particle in reference
273 * coordinates
274 * @param position_found Bool,'true' if the particle's position was found,
275 * and 'false' if the particle's position couldn't be found
276 * @param real_location Position of the particle in system coordinates
277 */
278 void
279 search_position_in_reference_space(
280 std::vector<std::vector<double>> &count_from_all_detectors,
281 std::vector<double> &experimental_count,
282 const typename DoFHandler<dim>::active_cell_iterator &cell,
283 const double &tol_reference_location,
284 double &max_cost,
285 Vector<double> &reference_location,
286 bool &position_found,
287 Point<dim> &real_location);
288
289 /**
290 * @brief Finds and saves the position of the particle in system coordinates.
291 *
292 * @param experimental_counts experimental_count contains the experimental
293 * counts of every detector for a given position.
294 *
295 * @param tol_reference_location tolerance on the validity of the found
296 * location in the reference space
297 *
298 * @return 'true' if the particle's position was found and 'false' if the
299 * particle's position couldn't be found
300 */
301 bool
302 find_position_global_search(std::vector<double> &experimental_counts,
303 const double tol_reference_location);
304
305 /**
306 * @brief Finds and saves the position of the particle in system coordinates
307 * by doing a local search around the previously found position's cell's
308 * neighbors
309 *
310 * @param experimental_counts experimental_count contains the experimental
311 * counts of every detector for a given position.
312 *
313 * @param tol_reference_location tolerance on the validity of the found
314 * location in the reference space
315 *
316 * @return 'true' if the particle's position was found and 'false' if the
317 * particle's position couldn't be found
318 */
319 bool
320 find_position_local_search(std::vector<double> &experimental_counts,
321 const double tol_reference_location);
322
323 /**
324 * @brief Reads the file with the experimental counts and finds the
325 * particles' positions by calling the "find_cell" function.
326 */
327 void
328 trajectory();
329
330 /**
331 * @brief Saves the built dictionary so it may be used later for the 3D particle position
332 * reconstruction.
333 */
334 void
335 checkpoint();
336
337 /**
338 * @brief Loads the previously built dictionary for the 3D particle position
339 * reconstruction.
340 */
341 void
342 load_from_checkpoint();
343
344 /**
345 * @brief Exports found particle positions in a the specified file format.
346 * If not specified, it will be exported in a ".csv" file format.
347 */
348 void
349 export_found_positions();
350
351 Triangulation<dim> triangulation;
352 FE_SimplexP<dim> fe;
353 DoFHandler<dim> dof_handler;
354 RPTCalculatingParameters rpt_parameters;
355 MappingFE<dim> mapping;
356 Parameters::RPTParameters parameters;
357 Parameters::RPTFEMReconstructionParameters fem_reconstruction_parameters;
358 Parameters::DetectorParameters detector_parameters;
359
360 std::vector<Detector<dim>> detectors;
361 unsigned int n_detector;
362 std::vector<Vector<double>> nodal_counts;
363 std::vector<Point<dim>> found_positions;
364 typename DoFHandler<dim>::active_cell_iterator previous_position_cell;
365
366 TimerOutput computing_timer;
367};
