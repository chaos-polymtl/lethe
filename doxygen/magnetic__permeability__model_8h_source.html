37/**
38 * @brief Constant magnetic permeability.
39 */
40class ConstantMagneticPermeability : public MagneticPermeabilityModel
41{
42public:
43 /**
44 * @brief Default constructor
45 */
46 ConstantMagneticPermeability(const double p_magnetic_permeability)
47 : magnetic_permeability(p_magnetic_permeability)
48 {}
49
50 /**
51 * @brief value Calculates the value the magnetic permeability
52 * @param fields_value Value of the various field on which the magnetic permeability depends.
53 * @return value of the magnetic permeability calculated with the fields_value.
54 */
55 double
56 value([[maybe_unused]] const std::map<field, double> &fields_value) override
57 {
58 return magnetic_permeability;
59 };
60
61 /**
62 * @brief vector_value Calculates the vector value of magnetic conductivities
63 * @param field_vectors Vector of properties on which the magnetic conductivities depend
64 * @param property_vector Values of the magnetic conductivities
65 */
66 void
67 vector_value(
68 [[maybe_unused]] const std::map<field, std::vector<double>> &field_vectors,
69 std::vector<double> &property_vector) override
70 {
71 property_vector.assign(property_vector.size(), magnetic_permeability);
72 }
73
74 /**
75 * @brief jacobian Calculates the jacobian (the partial derivative) of the magnetic permeability with respect to a field
76 * @param field_values Value of the various fields on which the property may depend.
77 * @param id Indicator of the field with respect to which the jacobian
78 * should be calculated
79 * @return value of the partial derivative of the magnetic permeability with respect to the field.
80 */
81
82 double
83 jacobian(const std::map<field, double> & /*field_values*/,
84 field /*id*/) override
85 {
86 return 0;
87 };
88
89 /**
90 * @brief vector_jacobian Calculate the derivative of the magnetic permeability with respect to a field
91 * @param field_vectors Vector for the values of the fields used to evaluate the property
92 * @param id Identifier of the field with respect to which a derivative should be calculated
93 * @param jacobian Vector of the value of the derivative of the magnetic permeability with respect to the field id
94 */
95
96 void
97 vector_jacobian(
98 const std::map<field, std::vector<double>> & /*field_vectors*/,
99 [[maybe_unused]] const field id,
100 std::vector<double> &jacobian_vector) override
101 {
102 std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0);
103 };
104
105private:
106 const double magnetic_permeability;
107};
