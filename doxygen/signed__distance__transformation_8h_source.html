43/**
44 * @brief Transforms the signed distance to the phase fraction with a hyperbolic
45 * tangent function.
46 *
47 * The phase fraction is defined as:
48 *
49 * \f$\phi = 0.5 - 0.5 \tanh(d/\varepsilon)\f$
50 *
51 * where \f$d\f$ is the signed distance and \f$\varepsilon\f$
52 * is a model parameter giving a measure of the interface thickness.
53 *
54 */
55class SignedDistanceTransformationTanh : public SignedDistanceTransformationBase
56{
57public:
58 /**
59 * @brief Constructor of the hyperbolic tangent signed distance transformation.
60 *
61 * @param[in] tanh_thickness Value of the \f$\varepsilon\f$ parameter
62 * modulating the interface thickness.
63 */
64 SignedDistanceTransformationTanh(const double tanh_thickness)
65 : tanh_thickness(tanh_thickness)
66 {}
67
68 /**
69 * @brief Computes the value of the phase fraction from the signed distance.
70 *
71 * @param[in] signed_distance Value of the signed distance
72 *
73 * @return Value of the computed phase fraction after applying the
74 * transformation.
75 */
76 inline double
77 transform_signed_distance(const double signed_distance) override
78 {
79 return 0.5 - 0.5 * tanh(signed_distance / tanh_thickness);
80 }
81
82private:
83 /**
84 * User-defined parameter that defines the thickness of the interface. It
85 * influences its sharpness: a smaller value leads to a sharper interface.
86 */
87 const double tanh_thickness;
88};

90/**
91 * @brief Transforms the signed distance to the phase fraction with a piecewise
92 * polynomial function (degree 4).
93 *
94 * The phase fraction is defined as:
95 *
96 * if \f$d < 0 \f$
97 * \f$\phi = 0.5 - 0.5(4d' + 6d'^2 + 4*d'^3 + d'^4)\f$
98 *
99 * else
100 * \f$\phi = 0.5 - 0.5(4d' - 6d'^2 + 4*d'^3 - d'^4)\f$
101 *
102 * with \f$d' = d/d_\mathrm{max}\f$, where \f$d\f$ the signed distance and
103 * \f$d_\mathrm{max}\f$ the maximum redistanciation distance. This
104 * transformation clamps the phase fraction to 0 or 1 when \f$d = \pm
105 * d_\mathrm{max}\f$.
106 *
107 */
108class SignedDistanceTransformationPiecewisePolynomial
109 : public SignedDistanceTransformationBase
110{
111public:
112 /**
113 * @brief Constructor of the piecewise polynomial signed distance
114 * transformation.
115 *
116 * @param[in] max_reinitialization_distance Value of the maximum distance
117 */
118 SignedDistanceTransformationPiecewisePolynomial(
119 const double max_reinitialization_distance)
120 : max_reinitialization_distance_inv(1.0 / max_reinitialization_distance)
121 {}
122
123 /**
124 * @brief Computes the value of the phase fraction from the signed distance.
125 *
126 * @param[in] signed_distance Value of the signed distance
127 *
128 * @return Value of the computed phase fraction after applying the transtion.
129 */
130 inline double
131 transform_signed_distance(const double signed_distance) override
132 {
133 const double dimentionless_d =
134 signed_distance * max_reinitialization_distance_inv;
135
136 double piecewise_polynomial_value;
137
138 if (dimentionless_d < 0.0)
139 {
140 piecewise_polynomial_value =
141 4.0 * dimentionless_d +
142 6.0 * Utilities::fixed_power<2>(dimentionless_d) +
143 4.0 * Utilities::fixed_power<3>(dimentionless_d) +
144 Utilities::fixed_power<4>(dimentionless_d);
145
146 return 0.5 - 0.5 * piecewise_polynomial_value;
147 }
148 else
149 {
150 piecewise_polynomial_value =
151 4.0 * dimentionless_d -
152 6.0 * Utilities::fixed_power<2>(dimentionless_d) +
153 4.0 * Utilities::fixed_power<3>(dimentionless_d) -
154 Utilities::fixed_power<4>(dimentionless_d);
155
156 return 0.5 - 0.5 * piecewise_polynomial_value;
157 }
158 }
159
160private:
161 /**
162 * Inverse of the maximum redistanciation distance.
163 */
164 const double max_reinitialization_distance_inv;
165};
