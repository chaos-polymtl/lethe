15/// Function to construct the Butcher's table for a given SDIRK method
16// The coefficients are fixed data during the simulation.
17SDIRKTable
18sdirk_table(const Parameters::SimulationControl::TimeSteppingMethod method)
19{
20 SDIRKTable table;
21
22 // When referencing to SDIRK methods the pattern used is sdirkOrderStage. For
23 // instance, sdirk22 indicates SDIRK method with order 2 and 2 stages
24 if (method == Parameters::SimulationControl::TimeSteppingMethod::sdirk22)
25 {
26 // SDIRK22 - Alexander's method (L-stable)
27 // Reference: R. Alexander, Diagonally implicit Runge–Kutta methods for
28 // stiff ODEs, SIAM Journal on Numerical Analysis, 14(6), 1006–1021, 1977.
29 // DOI: https://doi.org/10.1137/0714068
30
31 // alpha is a constant coefficient chosen to ensure the method is stable
32 // and consistent
33 const double alpha = (2. - std::sqrt(2.)) / 2.;
34
35 table.A.reinit(2, 2);
36 table.A(0, 0) = alpha;
37 table.A(1, 0) = 1.0 - alpha;
38 table.A(1, 1) = alpha;
39
40 table.c.resize(2);
41 table.c[0] = alpha;
42 table.c[1] = 1.0;
43
44 table.b.resize(2);
45 table.b[0] = 1.0 - alpha;
46 table.b[1] = alpha;
47
48 return table;
49 }
50
51 else if (method == Parameters::SimulationControl::TimeSteppingMethod::sdirk33)
52 {
53 // SDIRK33 - 3-stage, 3rd-order L-stable method (Kennedy-Carpenter)
54 // Reference: J. F. G. Kennedy and A. D. Carpenter, "Diagonally Implicit
55 // Runge-KuttaMethods for Ordinary Differential Equation"
56 // NASA/TM–2016–219173 pp. 77, March 2016. No DOI available URL :
57 // https://ntrs.nasa.gov/api/citations/20160005923/downloads/20160005923.pdf
58
59 // gamma is a constant coefficient chosen to ensure the method is stable
60 // and consistent
61 constexpr double gamma = 0.4358665215;
62
63 table.A.reinit(3, 3);
64 table.A(0, 0) = gamma;
65 table.A(1, 0) = 0.2820667393;
66 table.A(1, 1) = gamma;
67 table.A(2, 0) = 1.208496649;
68 table.A(2, 1) = -0.6443631707;
69 table.A(2, 2) = gamma;
70
71 table.c.resize(3);
72 table.c[0] = gamma;
73 table.c[1] = 0.7179332608;
74 table.c[2] = 1.0;
75
76 table.b.resize(3);
77 table.b[0] = 1.208496649;
78 table.b[1] = -0.6443631707;
79 table.b[2] = gamma;
80
81 return table;
82 }
83
84 else if (method == Parameters::SimulationControl::TimeSteppingMethod::sdirk43)
85 {
86 // SDIRK43 - 3-stage, 4th-order method
87 // Reference: L. Ferracina, M.N. Spijker, "Strong stability of
88 // singly-diagonally-implicit Runge–Kutta methods" Applied Numerical
89 // Mathematics 58 (2008) 1675–1686 DOI: doi:10.1016/j.apnum.2007.10.004
90
91 // gamma is a constant coefficient chosen to ensure the method is stable
92 // and consistent const double gamma = 0.128886400515;
93 constexpr double gamma = 1.068579021301629;
94
95 table.A.reinit(3, 3);
96 table.A(0, 0) = gamma;
97 table.A(1, 0) = 1.0 / 2.0 - gamma;
98 table.A(1, 1) = gamma;
99 table.A(2, 0) = 2 * gamma;
100 table.A(2, 1) = 1 - 4 * gamma;
101 table.A(2, 2) = gamma;
102
103 table.c.resize(3);
104 table.c[0] = gamma;
105 table.c[1] = 1.0 / 2.0;
106 table.c[2] = 1 - gamma;
107
108 table.b.resize(3);
109 table.b[0] = 1.0 / (6 * std::pow((2 * gamma - 1), 2.0));
110 table.b[1] = 2 * (6 * std::pow(gamma, 2.0) - 6 * gamma + 1) /
111 (3 * std::pow((2 * gamma - 1), 2.0));
112 table.b[2] = 1 / (6 * std::pow((2 * gamma - 1), 2.0));
113
114 return table;
115 }
116
117 else
118 {
119 AssertThrow(false, ExcMessage("Unknown SDIRK method in sdirk_table()."));
120 }
121}

124// Function to extract the coefficients of a given stage from a SDIRK Butcher's
125// table
126
127SDIRKStageData::SDIRKStageData(const SDIRKTable &table,
128 const unsigned int stage_i)
129{
130 AssertThrow(table.c.size() == table.b.size(),
131 ExcMessage("b and c vectors must have the same size."));
132
133 AssertThrow(table.A.m() == table.b.size(),
134 ExcMessage(
135 "Matrix A dimensions must match size of b and c vectors."));
136
137 // The stages take the values 1, ..., nstages
138
139 a_ij.resize(stage_i);
140
141 for (unsigned int j = 0; j < stage_i; ++j)
142 // The Butcher's table starts at index 0 whereas the stages take the values
143 // 1, ..., nstages
144 a_ij[j] = table.A(stage_i - 1, j);
145
146 c_i = table.c[stage_i - 1];
147 b_i = table.b[stage_i - 1];
148}
