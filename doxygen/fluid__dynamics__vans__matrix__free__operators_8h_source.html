13/**
14 * @brief Implements the matrix-free operator for the VANS equations
15 *
16 * @tparam dim An integer that denotes the number of spatial dimensions.
17 * @tparam number Abstract type for number across the class (i.e., double).
18 */
19template <int dim, typename number>
20class VANSOperator : public NavierStokesOperatorBase<dim, number>
21{
22public:
23 using FECellIntegrator = FEEvaluation<dim, -1, 0, dim + 1, number>;
24 using FEFaceIntegrator = FEFaceEvaluation<dim, -1, 0, dim + 1, number>;
25 using VectorType = LinearAlgebra::distributed::Vector<number>;
26
27
28 /**
29 * @brief Constructor for the VANS operator
30 *
31 * @param[in] cfd_dem_parameters The CFD-DEM parameters used for the
32 * simulation. These parameters are notably used to distinguish between
33 * model A and model B as well as to establish grad-div stabilization.
34
35 */
36 VANSOperator(const Parameters::CFDDEM &cfd_dem_parameters)
37 : NavierStokesOperatorBase<dim, number>()
38 , cfd_dem_parameters(cfd_dem_parameters)
39 {
40 AssertThrow(
41 this->cfd_dem_parameters.vans_model == Parameters::VANSModel::modelA,
42 ExcMessage(
43 "Only model A of the VANS equation is supported by the VANS matrix-free solver"));
44 }
45
46 /**
47 * @brief Compute the void fraction and the void fraction gradient
48 * at the quadrature points.
49 *
50 * @param[in] void_fraction_solution The solution of the void fraction
51 *
52 * @param[in] void_fraction_dof_handler The dof handler associated with the
53 * void fraction
54 */
55 void
56 compute_void_fraction(
57 const LinearAlgebra::distributed::Vector<double> &void_fraction_solution,
58 const DoFHandler<dim> &void_fraction_dof_handler);
59
60protected:
61 /**
62 * @brief Store relevant values of the vector of the last newton step to use it
63 * in the Jacobian and pre-calculate the stabilization parameters tau and
64 * gamma.
65 *
66 * @param[in] newton_step Vector of the last newton step.
67 */
68 virtual void
69 evaluate_non_linear_term_and_calculate_tau(
70 const VectorType &newton_step) override;
71
72 /**
73 * @brief Perform cell integral on a cell batch without gathering and scattering
74 * the values, and according to the Jacobian of the discretized
75 * steady/transient VANS equations with stabilization.
76 *
77 * @param[in] integrator FEEvaluation object that allows to evaluate functions
78 * at quadrature points and perform cell integrations.
79 */
80 void
81 do_cell_integral_local(FECellIntegrator &integrator) const override;
82
83 /**
84 * @brief Perform cell integral on a cell batch with gathering and scattering
85 * the values, and according to the residual of the discretized
86 * steady/transient VANS equations with stabilization.
87 *
88 * @param[in] matrix_free Object that contains all data.
89 * @param[in,out] dst Global vector where the final result is added.
90 * @param[in] src Input vector with all values in all cells.
91 * @param[in] range Range of the cell batch.
92 */
93 void
94 local_evaluate_residual(
95 const MatrixFree<dim, number> &matrix_free,
96 VectorType &dst,
97 const VectorType &src,
98 const std::pair<unsigned int, unsigned int> &range) const override;
99
100 /// Table with correct alignment for vectorization to store the values of the
101 /// void fraction
102 Table<2, VectorizedArray<number>> void_fraction;
103
104 /// Table with correct alignment for vectorization to store the values of the
105 /// void fraction gradient
106 Table<2, Tensor<1, dim, VectorizedArray<number>>> void_fraction_gradient;
107
108 /// Table with correct alignment for vectorization to store the values of the
109 /// grad-div gamma parameter
110 Table<2, VectorizedArray<number>> grad_div_gamma;
111
112 /// Internal copy of the CFD-DEM parameters. This is used for grad-div
113 /// stabilization, but also to switch between form A and B of the VANS
114 /// equations.
115 const Parameters::CFDDEM cfd_dem_parameters;
116};
