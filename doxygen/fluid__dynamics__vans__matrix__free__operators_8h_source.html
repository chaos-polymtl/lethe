13/**
14 * @brief Implements the matrix-free operator for the VANS equations
15 *
16 * @tparam dim An integer that denotes the number of spatial dimensions.
17 * @tparam number Abstract type for number across the class (i.e., double).
18 */
19template <int dim, typename number>
20class VANSOperator : public NavierStokesOperatorBase<dim, number>
21{
22public:
23 using FECellIntegrator = FEEvaluation<dim, -1, 0, dim + 1, number>;
24 using FEFaceIntegrator = FEFaceEvaluation<dim, -1, 0, dim + 1, number>;
25 using VectorType = LinearAlgebra::distributed::Vector<number>;
26
27
28 /**
29 * @brief Constructor for the VANS operator
30 *
31 * @param[in] cfd_dem_parameters The CFD-DEM parameters used for the
32 * simulation. These parameters are notably used to distinguish between
33 * model A and model B as well as to establish grad-div stabilization.
34 */
35 VANSOperator(const Parameters::CFDDEM &cfd_dem_parameters)
36 : NavierStokesOperatorBase<dim, number>()
37 , cfd_dem_parameters(cfd_dem_parameters)
38 {
39 AssertThrow(
40 this->cfd_dem_parameters.vans_model == Parameters::VANSModel::modelA,
41 ExcMessage(
42 "Only model A of the VANS equation is supported by the VANS matrix-free solver"));
43 }
44
45 /**
46 * @brief Compute the void fraction at the fluid quadrature point.
47 * This essentially calculates the void fraction and the void fraction
48 * gradient.
49 *
50 * @param[in] void_fraction_dof_handler The dof handler associated with the
51 * void fraction
52 *
53 * @param[in] void_fraction_solution The solution of the void fraction
54 *
55 * @param[in] time_derivative_void_fraction_solution The time derivative of
56 * the void fraction.
57 *
58 */
59 void
60 compute_void_fraction(
61 const DoFHandler<dim> &void_fraction_dof_handler,
62 const LinearAlgebra::distributed::Vector<double> &void_fraction_solution,
63 const LinearAlgebra::distributed::Vector<double>
64 &time_derivative_void_fraction_solution);
65
66 /**
67 * @brief Compute the terms required to assemble the particle-fluid forces at
68 * the gauss points. The function gathers the particle-fluid force (without
69 * drag), the particle-fluid drag and the particle velocity. The function
70 * assumes that the all of the fields gathered from the projectors are the
71 * forces acting on the particles due to the fluid. Consequently, since this
72 * aims to gather the forces that apply on the fluid due to the particles, a
73 * (-) multiplicator is applied during the gathering stage. With these three
74 * elements, the full particle-fluid coupling and its jacobian can be
75 * established within the matrix-free operator.
76 *
77 * @param[in] fp_force_dof_handler The dof handler associated with the
78 * particle-fluid forces.
79 * @param[in] fp_force_solution The solution of the particle-fluid forces.
80 * @param[in] fp_drag_dof_handler The dof handler associated with the
81 * drag force.
82 * @param[in] fp_drag_solution The solution of the drag force.
83 * @param[in] particle_velocity_dof_handler The dof handler associated with
84 * the particle velocity.
85 * @param[in] fp_drag_solution The solution of the particle velocity.
86 *
87 */
88 void
89 compute_particle_fluid_interaction(
90 const DoFHandler<dim> &fp_force_dof_handler,
91 const LinearAlgebra::distributed::Vector<double> &fp_force_solution,
92 const DoFHandler<dim> &fp_drag_dof_handler,
93 const LinearAlgebra::distributed::Vector<double> &fp_drag_solution,
94 const DoFHandler<dim> &particle_velocity_dof_handler,
95 const LinearAlgebra::distributed::Vector<double>
96 &particle_velocity_solution,
97 const DoFHandler<dim> &momentum_transfer_coefficient_dof_handler,
98 const LinearAlgebra::distributed::Vector<double>
99 &momentum_transfer_coefficient_solution);
100
101protected:
102 /**
103 * @brief Store relevant values of the vector of the last newton step to use it
104 * in the Jacobian and pre-calculate the stabilization parameters tau and
105 * gamma.
106 *
107 * @param[in] newton_step Vector of the last newton step.
108 */
109 virtual void
110 evaluate_non_linear_term_and_calculate_tau(
111 const VectorType &newton_step) override;
112
113 /**
114 * @brief Perform cell integral on a cell batch without gathering and scattering
115 * the values, and according to the Jacobian of the discretized
116 * steady/transient VANS equations with stabilization.
117 *
118 * @param[in] integrator FEEvaluation object that allows to evaluate functions
119 * at quadrature points and perform cell integrations.
120 */
121 void
122 do_cell_integral_local(FECellIntegrator &integrator) const override;
123
124 /**
125 * @brief Perform cell integral on a cell batch with gathering and scattering
126 * the values, and according to the residual of the discretized
127 * steady/transient VANS equations with stabilization.
128 *
129 * @param[in] matrix_free Object that contains all data.
130 * @param[in,out] dst Global vector where the final result is added.
131 * @param[in] src Input vector with all values in all cells.
132 * @param[in] range Range of the cell batch.
133 */
134 void
135 local_evaluate_residual(
136 const MatrixFree<dim, number> &matrix_free,
137 VectorType &dst,
138 const VectorType &src,
139 const std::pair<unsigned int, unsigned int> &range) const override;
140
141 /// Table with correct alignment for vectorization to store the values of the
142 /// void fraction
143 Table<2, VectorizedArray<number>> void_fraction;
144
145 /// Table with correct alignment for vectorization to store the values of the
146 /// void fraction gradient
147 Table<2, Tensor<1, dim, VectorizedArray<number>>> void_fraction_gradient;
148
149 /// Table with correct alignment for vectorization to store the values of the
150 /// time deriative of the void fraction
151 Table<2, VectorizedArray<number>> time_derivative_void_fraction;
152
153 /// Table with correct alignment for vectorization to store the values of the
154 /// particle-fluid forces (without drag). These are the resulting forces
155 /// applied on the particles due to hydrodynamic interaction, normalized by
156 /// the volume of the cells.
157 Table<2, Tensor<1, dim, VectorizedArray<number>>> particle_fluid_force;
158
159 /// Table with correct alignment for vectorization to store the values of the
160 /// particle-fluid drag. The drag force in this table is volumetric, resulting
161 /// from the projection used.
162 Table<2, Tensor<1, dim, VectorizedArray<number>>> particle_fluid_drag;
163
164 /// Table with correct alignment for vectorization to store the values of the
165 /// particle velocity.
166 Table<2, Tensor<1, dim, VectorizedArray<number>>> particle_velocity;
167
168 /// Table with correct alignment for vectorization to store the values of the
169 /// momentum transfer coefficient.
170 Table<2, VectorizedArray<number>> momentum_transfer_coefficient;
171
172 /// Table with correct alignment for vectorization to store the values of the
173 /// grad-div gamma parameter
174 Table<2, VectorizedArray<number>> grad_div_gamma;
175
176 /// Internal copy of the CFD-DEM parameters. This is used for grad-div
177 /// stabilization, but also to switch between form A and B of the VANS
178 /// equations.
179 const Parameters::CFDDEM cfd_dem_parameters;
180};
