16/**
17 * @brief A solver class for the DEM used in conjunction with IB particles and
18 * fluid_dynamics_sharp. This class defines and uses some functions of the
19 * DEM class that have been modified and simplified to be compatible with
20 * IB_particles.
21 *
22 * @tparam dim An integer that denotes the dimension of the space in which
23 * the flow is solved
24 *
25 * @ingroup solvers
26 */
27

141 /**
142 * @brief Calculates non-linear (Hertzian) force between two objects. Take into input the physical properties and contact state of the two objects.
143 *
144 * @param normal_overlap Contact normal overlap
145 * @param contact_info A container that contains the required information for
146 * calculation of the contact force for a particle pair in contact
147 * @param contact_point The contact point between the two objects
148 * @param contact_normal Contact normal unit vector
149 * @param normal_force Contact normal force
150 * @param tangential_force Contact tangential force
151 * @param rolling_resistance_torque Contact rolling resistance torque
152 * @param particle_one_position the position of particle one
153 * @param particle_one_velocity the velocity of particle one
154 * @param particle_one_omega the angular velocity of particle one
155 * @param particle_one_properties the physical properties of particle one
156 * @param particle_two_position the position of particle two
157 * @param particle_two_velocity the velocity of particle two
158 * @param particle_two_omega the angular velocity of particle two
159 * @param particle_two_properties the physical properties of particle two
160 * @param dt the dem time step.
161 */
162 void
163 calculate_force_model(const double normal_overlap,
164 ContactInfo &contact_info,

174 Point<dim> &particle_two_position,

181 /**
182 * @brief Calculates particle-particle lubrication force. The force is based on the formula from
183 * Microhydrodynamics: Principles and Selected Applications by Kim, Sangtae;
184 * Karrila, Seppo J. ISBN 13: 9780750691734
185 *
186 * @param dt_dem The sub time stepping time step.
187 * @param h_max The gap between the two particle.
188 * @param h_min The minimal gap distance considered in the force calculation to avoid the singularity of the model.
189 * @param mu The fluid viscosity.
190 * @param lubrication_force a vector containing the lubrication force on the particles.
191 * @param lubrication_torque a vector containing the lubrication torques on the particles.
192 */
193 void
194 calculate_pp_lubrication_force(const double dt_dem,
195 const double h_max,
196 const double h_min,
197 const double mu,
198 std::vector<Tensor<1, 3>> &lubrication_force,
199 std::vector<Tensor<1, 3>> &lubrication_torque);

202 /**

207 * @param h_min The minimal gap distance considered in the force calculation to avoid the singularity of the model.
208 * @param mu The fluid viscosity.
209 * @param lubrication_force a vector containing the lubrication force on the particles.
