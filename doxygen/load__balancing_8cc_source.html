34template <int dim, typename PropertiesIndex>
35inline void
36LagrangianLoadBalancing<dim, PropertiesIndex>::check_load_balance_dynamic()
37{
38 if (simulation_control->get_step_number() % dynamic_check_frequency == 0)
39 {
40 unsigned int maximum_particle_number_on_proc =
41 Utilities::MPI::max(particle_handler->n_locally_owned_particles(),
42 mpi_communicator);
43 unsigned int minimum_particle_number_on_proc =
44 Utilities::MPI::min(particle_handler->n_locally_owned_particles(),
45 mpi_communicator);
46 unsigned int average_particle_number_on_proc =
47 particle_handler->n_locally_owned_particles() / n_mpi_processes;
48
49 // Execute load balancing if difference of load between processors is
50 // larger than threshold of the load per processor
51 if ((maximum_particle_number_on_proc - minimum_particle_number_on_proc) >
52 load_threshold * average_particle_number_on_proc)
53 DEMActionManager::get_action_manager()->load_balance_step();
54 }
55}

57template <int dim, typename PropertiesIndex>
58inline void
59LagrangianLoadBalancing<dim, PropertiesIndex>::
60 check_load_balance_with_sparse_contacts()
61{
62 if (simulation_control->get_step_number() % dynamic_check_frequency == 0)
63 {
64 // Process to accumulate the load of each process regards the number
65 // of cells and particles with their selected weight and with a factor
66 // related to the mobility status of the cells
67 double load_weight = 0.0;
68
69 for (const auto &cell : triangulation->active_cell_iterators())
70 {
71 if (cell->is_locally_owned())
72 {
73 // Apply the cell weight
74#if DEAL_II_VERSION_GTE(9, 7, 0)
75 Point<dim> cell_barycenter = cell->center();
76 load_weight +=
77 static_cast<int>(cell_weight_function->value(cell_barycenter));
78#else
79 load_weight += cell_weight;
80#endif
81
82
83 // Get the mobility status of the cell and the number of particles
84 const unsigned int cell_mobility_status =
85 adaptive_sparse_contacts->check_cell_mobility(cell);
86 const unsigned int n_particles_in_cell =
87 particle_handler->n_particles_in_cell(cell);
88
89 // Apply a factor on the particle weight regards the
90 // mobility status. alpha = 1 by default for mobile cell, but
91 // is modified if cell is active or inactive
92 double alpha = 1.0;
93 if (cell_mobility_status ==
94 AdaptiveSparseContacts<dim,
95 PropertiesIndex>::static_active ||
96 cell_mobility_status ==
97 AdaptiveSparseContacts<dim,
98 PropertiesIndex>::advected_active)
99 {
100 alpha = active_status_factor;
101 }
102 else if (cell_mobility_status ==
103 AdaptiveSparseContacts<dim,
104 PropertiesIndex>::inactive ||
105 cell_mobility_status ==
106 AdaptiveSparseContacts<dim, PropertiesIndex>::advected)
107 {
108 alpha = inactive_status_factor;
109 }
110
111 // Add the particle weight time the number of particles in the
112 // cell to the processor load
113 load_weight += alpha * n_particles_in_cell * particle_weight;
114 }
115 }
116
117 // Find the minimum load on a processor
118 double maximum_load_on_proc =
119 Utilities::MPI::max(load_weight, mpi_communicator);
120
121 // Find the minimum load on a processor
122 double minimum_load_on_proc =
123 Utilities::MPI::min(load_weight, mpi_communicator);
124
125 // Get the total load
126 double total_load = Utilities::MPI::sum(load_weight, mpi_communicator);
127
128 if ((maximum_load_on_proc - minimum_load_on_proc) >
129 load_threshold * (total_load / n_mpi_processes))
130 {
131 // Clear and connect a new cell weight function
132 connect_mobility_status_weight_signals();
133
134 DEMActionManager::get_action_manager()->load_balance_step();
135 }
136 }
137
138 // Clear and connect a new cell weight function with new mobility status
139 connect_mobility_status_weight_signals();
140}

142template <int dim, typename PropertiesIndex>
143unsigned int
144LagrangianLoadBalancing<dim, PropertiesIndex>::calculate_total_cell_weight(
145 const typename parallel::distributed::Triangulation<dim>::cell_iterator &cell,
146 const CellStatus status) const
147{
148 // Assign no weight to cells we do not own.
149 if (!cell->is_locally_owned())
150 return 0;
151
152 switch (status)
153 {
154 case CellStatus::cell_will_persist:
155 case CellStatus::cell_will_be_refined:
156 // If CELL_PERSIST, do as CELL_REFINE
157 {
158 const unsigned int n_particles_in_cell =
159 particle_handler->n_particles_in_cell(cell);
160 return n_particles_in_cell * particle_weight;
161 }
162 case CellStatus::cell_invalid:
163 break;
164
165 case CellStatus::children_will_be_coarsened:
166 {
167 unsigned int n_particles_in_cell = 0;
168
169 for (unsigned int child_index = 0;
170 child_index < GeometryInfo<dim>::max_children_per_cell;
171 ++child_index)
172 n_particles_in_cell +=
173 particle_handler->n_particles_in_cell(cell->child(child_index));
174
175 return n_particles_in_cell * particle_weight;
176 }
177 default:
178 Assert(false, ExcInternalError());
179 break;
180 }
181
182 return 0;
183}

185template <int dim, typename PropertiesIndex>
186unsigned int
187LagrangianLoadBalancing<dim, PropertiesIndex>::
188 calculate_total_cell_weight_with_mobility_status(
189 const typename parallel::distributed::Triangulation<dim>::cell_iterator
190 &cell,
191 const CellStatus status) const
192{
193 // Assign no weight to cells we do not own.
194 if (!cell->is_locally_owned())
195 return 0;
196
197 // Get mobility status of the cell
198 const unsigned int cell_mobility_status =
199 adaptive_sparse_contacts->check_cell_mobility(cell);
200
201 // Applied a factor on the particle weight regards the mobility status
202 // Factor of 1 when mobile cell
203 double alpha = 1.0;
204 if (cell_mobility_status ==
205 AdaptiveSparseContacts<dim, PropertiesIndex>::static_active ||
206 cell_mobility_status ==
207 AdaptiveSparseContacts<dim, PropertiesIndex>::advected_active)
208 {
209 alpha = active_status_factor;
210 }
211 else if (cell_mobility_status ==
212 AdaptiveSparseContacts<dim, PropertiesIndex>::inactive ||
213 cell_mobility_status ==
214 AdaptiveSparseContacts<dim, PropertiesIndex>::advected)
215 {
216 alpha = inactive_status_factor;
217 }
218
219 switch (status)
220 {
221 case dealii::CellStatus::cell_will_persist:
222 case dealii::CellStatus::cell_will_be_refined:
223 {
224 const unsigned int n_particles_in_cell =
225 particle_handler->n_particles_in_cell(cell);
226 return static_cast<int>(alpha * n_particles_in_cell *
227 particle_weight);
228 }
229 case dealii::CellStatus::cell_invalid:
230 break;
231 case dealii::CellStatus::children_will_be_coarsened:
232 {
233 unsigned int n_particles_in_cell = 0;
234
235 for (unsigned int child_index = 0;
236 child_index < GeometryInfo<dim>::max_children_per_cell;
237 ++child_index)
238 n_particles_in_cell +=
239 particle_handler->n_particles_in_cell(cell->child(child_index));
240
241 return static_cast<int>(alpha * n_particles_in_cell *
242 particle_weight);
243 }
244 default:
245 Assert(false, ExcInternalError());
246 break;
247 }
248
249 return 0;
250}
