16template <int dim, typename PropertiesIndex>
17void
18AdaptiveSparseContacts<dim, PropertiesIndex>::update_local_and_ghost_cell_set(
19 const DoFHandler<dim> &background_dh)
20{
21 if (!sparse_contacts_enabled)
22 return;
23
24 local_and_ghost_cells.clear();
25 for (const auto &cell : background_dh.active_cell_iterators())

32template <int dim, typename PropertiesIndex>
33void
34AdaptiveSparseContacts<dim, PropertiesIndex>::
35 calculate_granular_temperature_and_solid_fraction(
36 const Particles::ParticleHandler<dim> &particle_handler,
37 const std::set<typename DoFHandler<dim>::active_cell_iterator>
38 &local_and_ghost_cells_with_particles,
39 Vector<double> &granular_temperature_average,
40 Vector<double> &solid_fractions)
41{
42 // Iterating through the active cells in cell set with particles
43 for (const auto &cell : local_and_ghost_cells_with_particles)
44 {
45 // Particles in the cell
46 auto particles_in_cell = particle_handler.particles_in_cell(cell);
47 const unsigned int n_particles_in_cell =
48 particle_handler.n_particles_in_cell(cell);
49
50 // Initialize variables for solid fraction computation
51 double solid_fraction = 0.0;
52 double solid_volume = 0.0;
53 const double cell_volume = cell->measure();
54
55 // Initialize variables for granular temperature computation
56 double granular_temperature_cell = 0.0;
57 Tensor<1, dim> velocity_cell_average;
58 Tensor<1, dim> cell_velocity_fluctuation_squared_average;
59
60 // First loop over particles in cell to compute the sum of
61 // particle velocity and the solid volume of the current cell
62 for (auto particles_in_cell_iterator = particles_in_cell.begin();
63 particles_in_cell_iterator != particles_in_cell.end();
64 ++particles_in_cell_iterator)
65 {
66 // Get particle properties
67 auto particle_properties =
68 particles_in_cell_iterator->get_properties();
69 const double dp = particle_properties[PropertiesIndex::dp];
70
71 for (int d = 0; d < dim; ++d)
72 {
73 // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
74 int v_axis = PropertiesIndex::v_x + d;
75
76 // Add the velocity component value
77 velocity_cell_average[d] += particle_properties[v_axis];
78 }
79
80 solid_volume += M_PI * Utilities::fixed_power<dim>(dp) / (2.0 * dim);
81 }
82
83 // Calculate average velocity in the cell (sum/n_particles)
84 velocity_cell_average /= n_particles_in_cell;
85
86 // Calculate solid fraction of cell
87 solid_fraction = solid_volume / cell_volume;
88
89 // Second loop over particle to compute the sum of the cell velocity
90 // fluctuations
91 for (auto particles_in_cell_iterator = particles_in_cell.begin();
92 particles_in_cell_iterator != particles_in_cell.end();
93 ++particles_in_cell_iterator)
94 {
95 auto particle_properties =
96 particles_in_cell_iterator->get_properties();
97
98 for (int d = 0; d < dim; ++d)
99 {
100 // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
101 int v_axis = PropertiesIndex::v_x + d;
102
103 cell_velocity_fluctuation_squared_average[d] +=
104 Utilities::fixed_power<2>(particle_properties[v_axis] -
105 velocity_cell_average[d]);
106 }
107 }
108
109 // Calculate average granular temperature in the cell
110 for (int d = 0; d < dim; ++d)
111 {
112 cell_velocity_fluctuation_squared_average[d] /= n_particles_in_cell;
113 granular_temperature_cell +=
114 cell_velocity_fluctuation_squared_average[d] / dim;
115 }
116
117 // Store the average granular temperature and solid fraction with
118 // active cell index
119 granular_temperature_average[cell->active_cell_index()] =
120 granular_temperature_cell;
121 solid_fractions[cell->active_cell_index()] = solid_fraction;
122 }
123}

125template <int dim, typename PropertiesIndex>
126void
127AdaptiveSparseContacts<dim, PropertiesIndex>::identify_mobility_status(
128 const DoFHandler<dim> &background_dh,
129 const Particles::ParticleHandler<dim> &particle_handler,
130 const unsigned int n_active_cells,
131 MPI_Comm mpi_communicator)
132{
133 // If the sparse contacts are not enabled, exit the function
134 if (!sparse_contacts_enabled)
135 return;
136
137 // Reset cell status containers
138 cell_mobility_status.clear();
139
140 // Reset all status to mobile if mobility status needs to be reset
141 if (DEMActionManager::get_action_manager()->check_mobility_status_reset())
142 {
143 for (auto cell : local_and_ghost_cells)
144 {
145 assign_mobility_status(cell->active_cell_index(),
146 mobility_status::mobile);
147 }
148
149 return;
150 }
151
152 // Get a copy of the active & ghost cells set to iterate over and remove cell
153 // of the set when the mobility status is known. It avoids unnecessary
154 // iteration for next loops. We don't want to modify the original set since
155 // it is only updated when there's load balancing or reading of checkpoints.
156 auto local_and_ghost_cells_copy = local_and_ghost_cells;
157
158 // Create dummy dofs for background dof handler for the mobility_at_nodes
159 const FE_Q<dim> fe(1);
160 const unsigned int dofs_per_cell = fe.dofs_per_cell;
161
162 // Get locally owned and relevant dofs
163 const IndexSet &locally_owned_dofs = background_dh.locally_owned_dofs();
164 IndexSet locally_relevant_dofs =
165 DoFTools::extract_locally_relevant_dofs(background_dh);
166
167 // Reinit all value of mobility at nodes as inactive
168 mobility_at_nodes.reinit(locally_owned_dofs,
169 locally_relevant_dofs,
170 mpi_communicator);
171 mobility_at_nodes = 0;
172
173 // If the advection of particles setting is enabled (useful for CFD-DEM),
174 // mobility status are different: inactive and static_active status are
175 // advected and advected_active since they are not handled in the same way as
176 // in DEM only. They have a special handling in the velocity integration step
177 // because cell averaged velocity and acceleration are applied to particles,
178 // so they need this different status. The criteria for those status are the
179 // same as for DEM.
180 mobility_status inactive_status = (!advect_particles_enabled) ?
181 mobility_status::inactive :
182 mobility_status::advected;
183 mobility_status active_status = (!advect_particles_enabled) ?
184 mobility_status::static_active :
185 mobility_status::advected_active;
186
187 // Check if the cell is empty (n_particle = 0), if so, nodes and cells are
188 // flagged as empty mobility status
189 for (auto cell = local_and_ghost_cells_copy.begin();
190 cell != local_and_ghost_cells_copy.end();)
191 {
192 // Check if the cell has any particles
193 if (particle_handler.n_particles_in_cell(*cell) == 0)
194 {
195 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
196 (*cell)->get_dof_indices(local_dof_indices);
197
198 // Assign inactive status to cell and empty status to nodes
199 assign_mobility_status((*cell)->active_cell_index(),
200 local_dof_indices,
201 mobility_status::inactive,
202 mobility_status::empty);
203
204 // Remove cell from cell set and iterate to the following cell,
205 // erase() function returns the next iterator
206 cell = local_and_ghost_cells_copy.erase(cell);
207 }
208 else
209 {
210 // Since erase() is not called, iterator is incremented
211 ++cell;
212 }
213 }
214
215 // Update ghost values of mobility_at_nodes
216 mobility_at_nodes.update_ghost_values();
217
218 // Calculate the average granular temperature and solid fraction for each
219 // cell currently in the local_and_ghost_cells_copy set (empty cells are
220 // already removed)
221 Vector<double> granular_temperature_average(n_active_cells);
222 Vector<double> solid_fractions(n_active_cells);
223 calculate_granular_temperature_and_solid_fraction(
224 particle_handler,
225 local_and_ghost_cells_copy,
226 granular_temperature_average,
227 solid_fractions);
228
229 // Check if the cell is mobile by criteria:
230 // * granular temperature > threshold or
231 // * solid fraction of cell < threshold or
232 // * is next to an empty cell
233 // If so, nodes are flagged and cells are stored with mobile status
234 for (auto cell = local_and_ghost_cells_copy.begin();
235 cell != local_and_ghost_cells_copy.end();)
236 {
237 const unsigned int cell_id = (*cell)->active_cell_index();
238 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
239 (*cell)->get_dof_indices(local_dof_indices);
240
241 // Check if the cell has any empty neighbor cell by the value at nodes
242 bool has_empty_node = false;
243 for (auto node_id : local_dof_indices)
244 {
245 if (mobility_at_nodes(node_id) == mobility_status::empty)
246 {
247 has_empty_node = true;
248 break; // No need to check other nodes
249 }
250 }
251
252 // Check if the cell is mobile by criteria (granular temperature, solid
253 // fraction, next to empty cell)
254 if (granular_temperature_average[cell_id] >
255 granular_temperature_threshold ||
256 solid_fractions[cell_id] < solid_fraction_threshold || has_empty_node)
257 {
258 // Assign mobile status to cell and nodes
259 assign_mobility_status(cell_id,
260 local_dof_indices,
261 mobility_status::mobile,
262 mobility_status::mobile);
263
264 // Remove cell from cell set
265 cell = local_and_ghost_cells_copy.erase(cell);
266 }
267 else
268 {
269 ++cell;
270 }
271 }
272
273 mobility_at_nodes.update_ghost_values();
274
275 // Check if the cell is mobile by neighbor (at least a node is flagged as
276 // mobile from previous check), this is the additional mobile layer.
277 // If so, cells are stored in map as mobile status and nodes that are not
278 // mobile are flagged as active
279 for (auto cell = local_and_ghost_cells_copy.begin();
280 cell != local_and_ghost_cells_copy.end();)
281 {
282 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
283 (*cell)->get_dof_indices(local_dof_indices);
284
285 // Check if the cell has any mobile nodes by the value at nodes
286 bool has_mobile_node = false;
287 for (auto node_id : local_dof_indices)
288 {
289 if (mobility_at_nodes(node_id) == mobility_status::mobile)
290 {
291 has_mobile_node = true;
292 break;
293 }
294 }
295
296 if (has_mobile_node)
297 {
298 // Assign mobile status to cell and active to nodes
299 assign_mobility_status((*cell)->active_cell_index(),
300 local_dof_indices,
301 mobility_status::mobile,
302 active_status);
303
304 // Remove cell from cell set and iterate to the following cell
305 cell = local_and_ghost_cells_copy.erase(cell);
306 }
307 else
308 {
309 ++cell;
310 }
311 }
312
313 mobility_at_nodes.update_ghost_values();
314
315 // Check if the cell is active (at least a node is flagged as active from
316 // previous check), this is the layer of active cells
317 // If so, cells are stored with active status in the map
318 for (auto cell = local_and_ghost_cells_copy.begin();
319 cell != local_and_ghost_cells_copy.end();)
320 {
321 std::vector<types::global_dof_index> local_dofs_indices(dofs_per_cell);
322 (*cell)->get_dof_indices(local_dofs_indices);
323
324 // Check if cell has active nodes
325 bool has_active_nodes = false;
326 for (auto node_id : local_dofs_indices)
327 {
328 if (mobility_at_nodes(node_id) == (int)active_status)
329 {
330 has_active_nodes = true;
331 break;
332 }
333 }
334
335 if (has_active_nodes)
336 {
337 // Assign active status to cell (no need to assign active to nodes)
338 assign_mobility_status((*cell)->active_cell_index(), active_status);
339 cell = local_and_ghost_cells_copy.erase(cell);
340 }
341 else
342 {
343 ++cell;
344 }
345 }
346
347 // Store the inactive cells in the map, those are all the remaining cells
348 // in the local & ghost cells set
349 for (auto cell = local_and_ghost_cells_copy.begin();
350 cell != local_and_ghost_cells_copy.end();
351 ++cell)
352 {
353 // Assign inactive status to cell in map
354 assign_mobility_status((*cell)->active_cell_index(), inactive_status);
355 }
356}

358template <int dim, typename PropertiesIndex>
359void
360AdaptiveSparseContacts<dim, PropertiesIndex>::
361 update_average_velocities_acceleration(
362 Particles::ParticleHandler<dim> &particle_handler,
363 const Tensor<1, 3> &g,
364 const std::vector<Tensor<1, 3>> &force,
365 const double dt)
366{
367 // If the sparse contacts is disabled, exit the function.
368 // Also, if the mobility status do not need to be reset, which only happens
369 // at first dem iteration of the cfd iteration
370 if (!sparse_contacts_enabled ||
371 !DEMActionManager::get_action_manager()->check_mobility_status_reset())
372 return;
373
374 cell_velocities_accelerations.clear();
375
376 // Tensors for velocity and acceleration * dt computation
377 Tensor<1, 3> velocity_cell_average;
378 Tensor<1, 3> acc_dt_cell_average;
379
380 const Tensor<1, 3> dt_g = g * dt;
381
382 // Loop over all the cells even if they are not mobile to reset the force
383 // and torque values
384 for (auto cell : local_and_ghost_cells)
385 {
386 if (cell->is_locally_owned())
387 {
388 const unsigned int n_particles_in_cell =
389 particle_handler.n_particles_in_cell(cell);
390
391 velocity_cell_average.clear();
392 acc_dt_cell_average.clear();
393
394 if (n_particles_in_cell > 0)
395 {
396 auto particles_in_cell = particle_handler.particles_in_cell(cell);
397 for (auto &particle : particles_in_cell)
398 {
399 // Get particle properties
400 auto particle_properties = particle.get_properties();
401 types::particle_index particle_id =
402 particle.get_local_index();
403
404 double dt_mass_inverse =
405 dt / particle_properties[PropertiesIndex::mass];
406
407 // Calculate the acceleration of the particle times the time
408 // step
409 Tensor<1, 3> acc_dt_particle =
410 dt_g + force[particle_id] * dt_mass_inverse;
411
412 // Add up the acc. * dt for the average
413 acc_dt_cell_average += acc_dt_particle;
414
415 for (int d = 0; d < dim; ++d)
416 {
417 // Add up the current velocity for the average velocity
418 velocity_cell_average[d] +=
419 particle_properties[PropertiesIndex::v_x + d] +
420 acc_dt_particle[d];
421 }
422 }
423
424 // Compute the average velocity and acceleration * dt
425 velocity_cell_average /= n_particles_in_cell;
426 acc_dt_cell_average /= n_particles_in_cell;
427 }
428
429 // Insert the average pair in map
430 cell_velocities_accelerations.insert(std::make_pair(
431 cell, std::make_pair(velocity_cell_average, acc_dt_cell_average)));
432 }
433 }
434}
