11template <int dim>
12void
13TracerAssemblerCore<dim>::assemble_matrix(
14 const TracerScratchData<dim> &scratch_data,
15 StabilizedMethodsCopyData &copy_data)
16{
17 // Scheme and physical properties
18 const std::vector<double> &diffusivity_vector =
19 scratch_data.tracer_diffusivity;
20 const auto method = this->simulation_control->get_assembly_method();
21
22 // Loop and quadrature informations
23 const auto &JxW_vec = scratch_data.JxW;
24 const unsigned int n_q_points = scratch_data.n_q_points;
25 const unsigned int n_dofs = scratch_data.n_dofs;
26 const double h = scratch_data.cell_size;
27
28 // Time steps and inverse time steps which is used for stabilization constant
29 std::vector<double> time_steps_vector =
30 this->simulation_control->get_time_steps_vector();
31 const double dt = time_steps_vector[0];
32 const double sdt = 1. / dt;
33
34
35 // Copy data elements
36 auto &strong_jacobian_vec = copy_data.strong_jacobian;
37 auto &local_matrix = copy_data.local_matrix;
38
39 // assembling local matrix and right hand side
40 for (unsigned int q = 0; q < n_q_points; ++q)
41 {
42 // Gather into local variables the relevant fields
43 const double diffusivity = diffusivity_vector[q];
44 const Tensor<1, dim> tracer_gradient = scratch_data.tracer_gradients[q];
45 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
46
47 // Store JxW in local variable for faster access;
48 const double JxW = JxW_vec[q];
49
50 // We use the previous concentration for the shock capture if the
51 // simulation is transient
52 const Tensor<1, dim> tracer_gradient_for_dcdd =
53 is_steady(method) ? tracer_gradient :
54 scratch_data.previous_tracer_gradients[q];
55 const double tracer_gradient_for_dcdd_norm =
56 tracer_gradient_for_dcdd.norm();
57
58 // Shock capturing viscosity term
59 const double order = scratch_data.fe_values_tracer.get_fe().degree;
60
61 const double vdcdd = (0.5 * h) * (velocity.norm() * velocity.norm()) *
62 pow(tracer_gradient_for_dcdd.norm() * h, order);
63
64 const double tolerance = 1e-12;
65 Tensor<1, dim> s = velocity / (velocity.norm() + tolerance);
66 Tensor<1, dim> r =
67 tracer_gradient_for_dcdd / (tracer_gradient_for_dcdd_norm + tolerance);
68
69 const Tensor<2, dim> k_corr = (r * s) * outer_product(s, s);
70 const Tensor<2, dim> rr = outer_product(r, r);
71 const Tensor<2, dim> dcdd_factor = rr - k_corr;
72
73 // If the method is not steady, we use the previous gradient to calculate
74 // the DCDD schock capture term and consequently we do not need an
75 // estimation of the derivative of the vdcdd term.
76 const double d_vdcdd =
77 is_steady(method) ?
78 order * (0.5 * h * h) * (velocity.norm() * velocity.norm()) *
79 pow(tracer_gradient_for_dcdd_norm * h, order - 1) :
80 0;
81
82
83
84 // Calculation of the magnitude of the velocity for the
85 // stabilization parameter
86 const double u_mag = std::max(velocity.norm(), tolerance);
87
88 // Calculation of the GLS stabilization parameter. The
89 // stabilization parameter used is different if the simulation is
90 // steady or unsteady. In the unsteady case it includes the value
91 // of the time-step
92 const double tau =
93 is_steady(method) ?
94 1. / std::sqrt(std::pow(2. * u_mag / h, 2) +
95 9 * std::pow(4 * diffusivity / (h * h), 2)) :
96 1. / std::sqrt(std::pow(sdt, 2) + std::pow(2. * u_mag / h, 2) +
97 9 * std::pow(4 * diffusivity / (h * h), 2));
98
99 for (unsigned int j = 0; j < n_dofs; ++j)
100 {
101 const Tensor<1, dim> grad_phi_T_j = scratch_data.grad_phi[q][j];
102 const double laplacian_phi_T_j = scratch_data.laplacian_phi[q][j];
103 strong_jacobian_vec[q][j] +=
104 velocity * grad_phi_T_j - diffusivity * laplacian_phi_T_j;
105 }
106
107 for (unsigned int i = 0; i < n_dofs; ++i)
108 {
109 const auto phi_T_i = scratch_data.phi[q][i];
110 const auto grad_phi_T_i = scratch_data.grad_phi[q][i];
111
112 for (unsigned int j = 0; j < n_dofs; ++j)
113 {
114 const Tensor<1, dim> grad_phi_T_j = scratch_data.grad_phi[q][j];
115
116 // Weak form : - D * laplacian T + u * gradT - f=0
117 local_matrix(i, j) += (diffusivity * grad_phi_T_i * grad_phi_T_j +
118 phi_T_i * velocity * grad_phi_T_j) *
119 JxW;
120
121
122
123 local_matrix(i, j) += tau * strong_jacobian_vec[q][j] *
124 (grad_phi_T_i * velocity) * JxW;
125
126 local_matrix(i, j) +=
127 (vdcdd *
128 scalar_product(grad_phi_T_i, dcdd_factor * grad_phi_T_j) +
129 d_vdcdd * grad_phi_T_j.norm() *
130 scalar_product(grad_phi_T_i,
131 dcdd_factor * tracer_gradient)) *
132 JxW;
133 }
134 }
135 } // end loop on quadrature points
136}

139template <int dim>
140void
141TracerAssemblerCore<dim>::assemble_rhs(
142 const TracerScratchData<dim> &scratch_data,
143 StabilizedMethodsCopyData &copy_data)
144{
145 // Scheme and physical properties
146 const std::vector<double> &diffusivity_vector =
147 scratch_data.tracer_diffusivity;
148 const auto method = this->simulation_control->get_assembly_method();
149
150 // Loop and quadrature information
151 const auto &JxW_vec = scratch_data.JxW;
152 const unsigned int n_q_points = scratch_data.n_q_points;
153 const unsigned int n_dofs = scratch_data.n_dofs;
154 const double h = scratch_data.cell_size;
155
156 // Time steps and inverse time steps which is used for stabilization constant
157 std::vector<double> time_steps_vector =
158 this->simulation_control->get_time_steps_vector();
159 const double dt = time_steps_vector[0];
160 const double sdt = 1. / dt;
161
162 // Copy data elements
163 auto &strong_residual_vec = copy_data.strong_residual;
164 auto &local_rhs = copy_data.local_rhs;
165
166 // assembling local matrix and right hand side
167 for (unsigned int q = 0; q < n_q_points; ++q)
168 {
169 // Gather into local variables the relevant fields
170 const double diffusivity = diffusivity_vector[q];
171 const Tensor<1, dim> tracer_gradient = scratch_data.tracer_gradients[q];
172 const Tensor<1, dim> previous_tracer_gradient =
173 scratch_data.previous_tracer_gradients[q];
174 const double tracer_laplacian = scratch_data.tracer_laplacians[q];
175 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
176
177 // Store JxW in local variable for faster access;
178 const double JxW = JxW_vec[q];
179
180 // We use the previous concentration for the shock capture if the
181 // simulation is transient
182 const Tensor<1, dim> tracer_gradient_for_dcdd =
183 is_steady(method) ? tracer_gradient : previous_tracer_gradient;
184 const double tracer_gradient_for_dcdd_norm =
185 tracer_gradient_for_dcdd.norm();
186
187 // Shock capturing viscosity term
188 const double order = scratch_data.fe_values_tracer.get_fe().degree;
189
190
191 const double vdcdd = (0.5 * h) * (velocity.norm() * velocity.norm()) *
192 pow(tracer_gradient_for_dcdd_norm * h, order);
193
194 const double tolerance = 1e-12;
195 Tensor<1, dim> s = velocity / (velocity.norm() + tolerance);
196 Tensor<1, dim> r =
197 tracer_gradient_for_dcdd / (tracer_gradient_for_dcdd_norm + tolerance);
198
199 const Tensor<2, dim> k_corr = (r * s) * outer_product(s, s);
200 const Tensor<2, dim> rr = outer_product(r, r);
201 const Tensor<2, dim> dcdd_factor = rr - k_corr;
202
203 // Calculation of the magnitude of the velocity for the
204 // stabilization parameter
205 const double u_mag = std::max(velocity.norm(), tolerance);
206
207 // Calculation of the GLS stabilization parameter. The
208 // stabilization parameter used is different if the simulation is
209 // steady or unsteady. In the unsteady case it includes the value
210 // of the time-step
211 const double tau =
212 is_steady(method) ?
213 1. / std::sqrt(std::pow(2. * u_mag / h, 2) +
214 9 * std::pow(4 * diffusivity / (h * h), 2)) :
215 1. / std::sqrt(std::pow(sdt, 2) + std::pow(2. * u_mag / h, 2) +
216 9 * std::pow(4 * diffusivity / (h * h), 2));
217
218 // Calculate the strong residual for GLS stabilization
219 strong_residual_vec[q] +=
220 velocity * tracer_gradient - diffusivity * tracer_laplacian;
221
222 for (unsigned int i = 0; i < n_dofs; ++i)
223 {
224 const auto phi_T_i = scratch_data.phi[q][i];
225 const auto grad_phi_T_i = scratch_data.grad_phi[q][i];
226
227 // rhs for : - D * laplacian T + u * grad T - f=0
228 local_rhs(i) -= (diffusivity * grad_phi_T_i * tracer_gradient +
229 phi_T_i * velocity * tracer_gradient -
230 scratch_data.source[q] * phi_T_i) *
231 JxW;
232
233 local_rhs(i) -=
234 tau * (strong_residual_vec[q] * (grad_phi_T_i * velocity)) * JxW;
235
236 local_rhs(i) -=
237 vdcdd *
238 scalar_product(grad_phi_T_i, dcdd_factor * tracer_gradient) * JxW;
239 }
240 } // end loop on quadrature points
241}

248TracerAssemblerDGCore<dim>::assemble_matrix(
249 const TracerScratchData<dim> &scratch_data,
250 StabilizedMethodsCopyData &copy_data)
251{
252 // Scheme and physical properties
253 const std::vector<double> &diffusivity_vector =
254 scratch_data.tracer_diffusivity;
255
256 // Loop and quadrature informations
257 const auto &JxW_vec = scratch_data.JxW;
258 const unsigned int n_q_points = scratch_data.n_q_points;
259 const unsigned int n_dofs = scratch_data.n_dofs;
260
261 // Copy data elements
262 auto &local_matrix = copy_data.local_matrix;
263
264 // assembling local matrix and right hand side
265 for (unsigned int q = 0; q < n_q_points; ++q)
266 {
267 // Gather into local variables the relevant fields
268 const double diffusivity = diffusivity_vector[q];
269 const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
270
271 // Store JxW in local variable for faster access;
272 const double JxW = JxW_vec[q];
273
274 for (unsigned int i = 0; i < n_dofs; ++i)
275 {
276 const auto grad_phi_T_i = scratch_data.grad_phi[q][i];
277
278 for (unsigned int j = 0; j < n_dofs; ++j)
279 {
280 const Tensor<1, dim> grad_phi_T_j = scratch_data.grad_phi[q][j];
281 const auto phi_T_j = scratch_data.phi[q][j];
282
283 // Weak form : - D * laplacian T + u * gradT - f=0
284 // Note that the advection term has been weakened for it to appear
285 // explicitly in the weak form as a boundary term.
286 local_matrix(i, j) += (diffusivity * grad_phi_T_i * grad_phi_T_j -
287 grad_phi_T_i * velocity * phi_T_j) *
288 JxW;
289 }
290 }
291 } // end loop on quadrature points
292}

343template <int dim>
344void
345TracerAssemblerBDF<dim>::assemble_matrix(
346 const TracerScratchData<dim> &scratch_data,
347 StabilizedMethodsCopyData &copy_data)
348{
349 // Loop and quadrature informations
350 const auto &JxW = scratch_data.JxW;
351 const unsigned int n_q_points = scratch_data.n_q_points;
352 const unsigned int n_dofs = scratch_data.n_dofs;
353
354 // Copy data elements
355 auto &strong_residual = copy_data.strong_residual;
356 auto &strong_jacobian = copy_data.strong_jacobian;
357 auto &local_matrix = copy_data.local_matrix;
358
359 // Time stepping information
360 const auto method = this->simulation_control->get_assembly_method();
361 // Vector for the BDF coefficients
362 const Vector<double> &bdf_coefs =
363 this->simulation_control->get_bdf_coefficients();
364 std::vector<double> tracer(1 + number_of_previous_solutions(method));
365
366 // Loop over the quadrature points
367 for (unsigned int q = 0; q < n_q_points; ++q)
368 {
369 tracer[0] = scratch_data.tracer_values[q];
370 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
371 tracer[p + 1] = scratch_data.previous_tracer_values[p][q];
372
373 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
374 ++p)
375 {
376 strong_residual[q] += bdf_coefs[p] * tracer[p];
377 }
378
379 for (unsigned int j = 0; j < n_dofs; ++j)
380 {
381 strong_jacobian[q][j] += bdf_coefs[0] * scratch_data.phi[q][j];
382 }
383
384
385 for (unsigned int i = 0; i < n_dofs; ++i)
386 {
387 const double phi_u_i = scratch_data.phi[q][i];
388 for (unsigned int j = 0; j < n_dofs; ++j)
389 {
390 const double phi_u_j = scratch_data.phi[q][j];
391
392 local_matrix(i, j) += phi_u_j * phi_u_i * bdf_coefs[0] * JxW[q];
393 }
394 }
395 }
396}

399void
400TracerAssemblerBDF<dim>::assemble_rhs(
401 const TracerScratchData<dim> &scratch_data,
402 StabilizedMethodsCopyData &copy_data)
403{
404 // Loop and quadrature informations
405 const auto &JxW = scratch_data.JxW;
406 const unsigned int n_q_points = scratch_data.n_q_points;
407 const unsigned int n_dofs = scratch_data.n_dofs;
408
409 // Copy data elements
410 auto &strong_residual = copy_data.strong_residual;
411 auto &local_rhs = copy_data.local_rhs;
412
413 // Time stepping information
414 const auto method = this->simulation_control->get_assembly_method();
415 // Vector for the BDF coefficients
416 const Vector<double> &bdf_coefs =
417 this->simulation_control->get_bdf_coefficients();
418 std::vector<double> tracer(1 + number_of_previous_solutions(method));
419
420 // Loop over the quadrature points
421 for (unsigned int q = 0; q < n_q_points; ++q)
422 {
423 tracer[0] = scratch_data.tracer_values[q];
424 for (unsigned int p = 0; p < number_of_previous_solutions(method); ++p)
425 tracer[p + 1] = scratch_data.previous_tracer_values[p][q];
426
427 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
428 ++p)
429 {
430 strong_residual[q] += (bdf_coefs[p] * tracer[p]);
431 }
432
433 for (unsigned int i = 0; i < n_dofs; ++i)
434 {
435 const double phi_u_i = scratch_data.phi[q][i];
436 double local_rhs_i = 0;
437 for (unsigned int p = 0; p < number_of_previous_solutions(method) + 1;
438 ++p)
439 {
440 local_rhs_i -= bdf_coefs[p] * (tracer[p] * phi_u_i);
441 }
442 local_rhs(i) += local_rhs_i * JxW[q];
443 }
444 }
445}

452template <int dim>
453void
454TracerAssemblerSIPG<dim>::assemble_matrix(
455 const TracerScratchData<dim> &scratch_data,
456 StabilizedDGMethodsCopyData &copy_data)
457{
458 const double penalty_factor = scratch_data.penalty_factor;
459 const FEInterfaceValues<dim> &fe_iv = scratch_data.fe_interface_values_tracer;
460 const auto &q_points = fe_iv.get_quadrature_points();
461 auto &copy_data_face = copy_data.face_data.back();
462 const unsigned int n_dofs = fe_iv.n_current_interface_dofs();
463
464
465 const std::vector<double> &JxW = fe_iv.get_JxW_values();
466 const std::vector<Tensor<1, dim>> &normals = fe_iv.get_normal_vectors();
467
468 for (unsigned int q = 0; q < q_points.size(); ++q)
469 {
470 const double velocity_dot_n =
471 scratch_data.face_velocity_values[q] * normals[q];
472 for (unsigned int i = 0; i < n_dofs; ++i)
473 for (unsigned int j = 0; j < n_dofs; ++j)
474 {
475 copy_data_face.face_matrix(i, j) +=
476 fe_iv.jump_in_shape_values(i, q) // [\phi_i]
477 * fe_iv.shape_value((velocity_dot_n > 0.),
478 j,
479 q) // phi_j^{upwind}
480 * velocity_dot_n // (u . n)
481 * JxW[q]; // dx
482
483 // Assemble the diffusion term using Nitsche
484 // symmetric interior penalty method. See Larson Chap. 14. P.362
485 copy_data_face.face_matrix(i, j) +=
486 (scratch_data.tracer_diffusivity_face[q] *
487 (-fe_iv.average_of_shape_gradients(j, q) * normals[q] *
488 fe_iv.jump_in_shape_values(i, q) -
489 fe_iv.average_of_shape_gradients(i, q) * normals[q] *
490 fe_iv.jump_in_shape_values(j, q)) +
491 (scratch_data.tracer_diffusivity_face[q]) * penalty_factor *
492 fe_iv.jump_in_shape_values(j, q) *
493 fe_iv.jump_in_shape_values(i, q)) *
494 JxW[q];
495 }
496 }
497}

499template <int dim>
500void
501TracerAssemblerSIPG<dim>::assemble_rhs(
502 const TracerScratchData<dim> &scratch_data,
503 StabilizedDGMethodsCopyData &copy_data)
504{
505 const double penalty_factor = scratch_data.penalty_factor;
506 const FEInterfaceValues<dim> &fe_iv = scratch_data.fe_interface_values_tracer;
507 const auto &q_points = fe_iv.get_quadrature_points();
508 const unsigned int n_dofs = fe_iv.n_current_interface_dofs();
509
510 auto &copy_data_face = copy_data.face_data.back();
511
512 const std::vector<double> &JxW = fe_iv.get_JxW_values();
513 const std::vector<Tensor<1, dim>> &normals = fe_iv.get_normal_vectors();
514
515 for (unsigned int q = 0; q < q_points.size(); ++q)
516 {
517 const double velocity_dot_n =
518 scratch_data.face_velocity_values[q] * normals[q];
519 for (unsigned int i = 0; i < n_dofs; ++i)
520 {
521 // Assemble advection terms with upwinding
522 if (velocity_dot_n > 0)
523 {
524 copy_data_face.face_rhs(i) -=
525 fe_iv.jump_in_shape_values(i, q) // [\phi_i]
526 * scratch_data.values_here[q] // \phi_i^{upwind}
527 * velocity_dot_n // (u . n)
528 * JxW[q]; // dx
529 }
530 else
531 {
532 copy_data_face.face_rhs(i) -=
533 fe_iv.jump_in_shape_values(i, q) // [\phi_i]
534 * scratch_data.values_there[q] // \phi_i^{upwind}
535 * velocity_dot_n // (u . n)
536 * JxW[q]; // dx
537 }
538
539 // Assemble the diffusion term using Nitsche symmetric interior
540 // penalty method. See Larson Chap. 14. P.362
541 copy_data_face.face_rhs(i) -=
542 (scratch_data.tracer_diffusivity_face[q] *
543 (-scratch_data.tracer_average_gradient[q] * normals[q] *
544 fe_iv.jump_in_shape_values(i, q) -
545 scratch_data.tracer_value_jump[q] * normals[q] *
546 fe_iv.average_of_shape_gradients(i, q)) +
547 (scratch_data.tracer_diffusivity_face[q]) * penalty_factor *
548 scratch_data.tracer_value_jump[q] *
549 fe_iv.jump_in_shape_values(i, q)) *
550 JxW[q];
551 }
552 }
553}

559template <int dim>
560void
561TracerAssemblerBoundaryNitsche<dim>::assemble_matrix(
562 const TracerScratchData<dim> &scratch_data,
563 StabilizedDGMethodsCopyData &copy_data)
564{
565 const unsigned int boundary_index = scratch_data.boundary_index;
566 const FEFaceValuesBase<dim> &fe_face =
567 scratch_data.fe_interface_values_tracer.get_fe_face_values(0);
568 const auto &q_points = fe_face.get_quadrature_points();
569
570 const double beta = scratch_data.penalty_factor;
571
572 const unsigned int n_facet_dofs = fe_face.get_fe().n_dofs_per_cell();
573 const std::vector<double> &JxW = fe_face.get_JxW_values();
574 const std::vector<Tensor<1, dim>> &normals = fe_face.get_normal_vectors();
575
576 for (unsigned int point = 0; point < q_points.size(); ++point)
577 {
578 const double velocity_dot_n =
579 scratch_data.face_velocity_values[point] * normals[point];
580 for (unsigned int i = 0; i < n_facet_dofs; ++i)
581 {
582 for (unsigned int j = 0; j < n_facet_dofs; ++j)
583 {
584 if (boundary_conditions_tracer.type[boundary_index] ==
585 BoundaryConditions::BoundaryType::outlet)
586 {
587 if (velocity_dot_n > 0)
588 copy_data.local_matrix(i, j) +=
589 fe_face.shape_value(i, point) *
590 fe_face.shape_value(j, point) * velocity_dot_n *
591 JxW[point];
592 }
593 else if (boundary_conditions_tracer.type[boundary_index] ==
594 BoundaryConditions::BoundaryType::tracer_dirichlet)
595 {
596 if (velocity_dot_n > 0)
597 copy_data.local_matrix(i, j) +=
598 fe_face.shape_value(i, point) *
599 fe_face.shape_value(j, point) * velocity_dot_n *
600 JxW[point];
601
602 copy_data.local_matrix(i, j) +=
603 scratch_data.tracer_diffusivity_face[point] *
604 (-fe_face.shape_value(i, point) *
605 fe_face.shape_grad(j, point) * normals[point] -
606 fe_face.shape_value(j, point) *
607 fe_face.shape_grad(i, point) * normals[point] +
608 beta * fe_face.shape_value(i, point) *
609 fe_face.shape_value(j, point)) *
610 JxW[point];
611 }
612 }
613 }
614 }

617template <int dim>
618void
619TracerAssemblerBoundaryNitsche<dim>::assemble_rhs(
620 const TracerScratchData<dim> &scratch_data,
621 StabilizedDGMethodsCopyData &copy_data)
622{
623 const unsigned int boundary_index = scratch_data.boundary_index;
624 const FEFaceValuesBase<dim> &fe_face =
625 scratch_data.fe_interface_values_tracer.get_fe_face_values(0);
626 const auto &q_points = fe_face.get_quadrature_points();
627
628 const double beta = scratch_data.penalty_factor;
629
630 const unsigned int n_facet_dofs = fe_face.get_fe().n_dofs_per_cell();
631 const std::vector<double> &JxW = fe_face.get_JxW_values();
632 const std::vector<Tensor<1, dim>> &normals = fe_face.get_normal_vectors();
633
634 // If the boundary condition is an outlet, assumes that advection comes
635 // out since there is no inflow
636 if (boundary_conditions_tracer.type[boundary_index] ==
637 BoundaryConditions::BoundaryType::outlet)
638 {
639 for (unsigned int point = 0; point < q_points.size(); ++point)
640 {
641 const double velocity_dot_n =
642 scratch_data.face_velocity_values[point] * normals[point];
643
644 for (unsigned int i = 0; i < n_facet_dofs; ++i)
645 copy_data.local_rhs(i) -= fe_face.shape_value(i, point) // \phi_i
646 * scratch_data.values_here[point] *
647 velocity_dot_n // u . n
648 * JxW[point]; // dx
649 }
650 }
651 // Else the boundary condition is a dirichlet. Evaluate the function and
652 // process it accordingly
653 else if (boundary_conditions_tracer.type[boundary_index] ==
654 BoundaryConditions::BoundaryType::tracer_dirichlet)
655 {
656 std::vector<double> function_value(q_points.size());
657 boundary_conditions_tracer.tracer[boundary_index]->value_list(
658 q_points, function_value);
659
660 for (unsigned int point = 0; point < q_points.size(); ++point)
661 {
662 const double velocity_dot_n =
663 scratch_data.face_velocity_values[point] * normals[point];
664
665 for (unsigned int i = 0; i < n_facet_dofs; ++i)
666 {
667 if (velocity_dot_n < 0)
668 copy_data.local_rhs(i) -= fe_face.shape_value(i, point) *
669 function_value[point] *
670 velocity_dot_n * JxW[point];
671
672 if (velocity_dot_n > 0)
673 copy_data.local_rhs(i) -= fe_face.shape_value(i, point) *
674 scratch_data.values_here[point] *
675 velocity_dot_n * JxW[point];
676
677 copy_data.local_rhs(i) -=
678 scratch_data.tracer_diffusivity_face[point] *
679 (-fe_face.shape_value(i, point) *
680 scratch_data.gradients_here[point] * normals[point] -
681 (scratch_data.values_here[point] - function_value[point]) *
682 fe_face.shape_grad(i, point) * normals[point] +
683 beta * fe_face.shape_value(i, point) *
684 (scratch_data.values_here[point] - function_value[point])) *
685 JxW[point];
686 }
687 }
688 }
689 else if (boundary_conditions_tracer.type[boundary_index] ==
690 BoundaryConditions::BoundaryType::periodic ||
691 boundary_conditions_tracer.type[boundary_index] ==
692 BoundaryConditions::BoundaryType::periodic_neighbor)
693 {
694 AssertThrow(
695 false,
696 ExcMessage(
697 "Periodic boundary conditions cannot be used with the tracer physics when discontinous Galerkin elements are used."));
698 }
699 // If it is an unknown boundary condition, throw an exception. This case
700 // should never occur.
701 else
702 {
703 AssertThrow(false,
704 ExcMessage(
705 "No valid boundary conditions types were identified."));
706 }
707}

712template <int dim>
713void
714TracerAssemblerReaction<dim>::assemble_matrix(
715 const TracerScratchData<dim> &scratch_data,
716 StabilizedMethodsCopyData &copy_data)
717{
718 // Scheme and physical properties
719 const std::vector<double> &k = scratch_data.tracer_reaction_prefactor;
720
721 // Loop and quadrature information
722 const auto &JxW_vec = scratch_data.JxW;
723 const unsigned int n_q_points = scratch_data.n_q_points;
724 const unsigned int n_dofs = scratch_data.n_dofs;
725
726 auto &strong_jacobian_vec = copy_data.strong_jacobian;
727 auto &local_matrix = copy_data.local_matrix;
728
729 // Assembling local matrix
730 for (unsigned int q = 0; q < n_q_points; ++q)
731 {
732 const double reaction_coeff = k[q];
733 const double C = scratch_data.tracer_values[q];
734
735 // Update the strong Jacobian with the reaction term contribution:
736 // It comes from the derivative -d/dC (k * C) = -(k + C*dk/dC).
737 for (unsigned int j = 0; j < n_dofs; ++j)
738 {
739 const auto phi_T_j = scratch_data.phi[q][j];
740 strong_jacobian_vec[q][j] +=
741 -(reaction_coeff +
742 scratch_data.grad_tracer_reaction_prefactor[q] * C) *
743 phi_T_j;
744 }
745
746 // Store JxW in a local variable for faster access.
747 const double JxW = JxW_vec[q];
748
749 for (unsigned int i = 0; i < n_dofs; ++i)
750 {
751 const auto &phi_T_i = scratch_data.phi[q][i];
752
753 for (unsigned int j = 0; j < n_dofs; ++j)
754 {
755 const auto &phi_T_j = scratch_data.phi[q][j];
756
757 local_matrix(i, j) += reaction_coeff * phi_T_i * phi_T_j * JxW;
758 }
759 }
760 } // end loop on quadrature points
761}

763template <int dim>
764void
765TracerAssemblerReaction<dim>::assemble_rhs(
766 const TracerScratchData<dim> &scratch_data,
767 StabilizedMethodsCopyData &copy_data)
768{
769 // Scheme and physical properties
770 const std::vector<double> &k = scratch_data.tracer_reaction_prefactor;
771
772 // Loop and quadrature information
773 const auto &JxW_vec = scratch_data.JxW;
774 const unsigned int n_q_points = scratch_data.n_q_points;
775 const unsigned int n_dofs = scratch_data.n_dofs;
776
777 // Copy data elements
778 auto &strong_residual_vec = copy_data.strong_residual;
779 auto &local_rhs = copy_data.local_rhs;
780
781 // Assembling right-hand side
782 for (unsigned int q = 0; q < n_q_points; ++q)
783 {
784 // Store JxW in local variable for faster access
785 const double JxW = JxW_vec[q];
786
787 // Get the tracer concentration value at this quadrature point
788 const double C = scratch_data.tracer_values[q];
789
790 // Calculate the strong residual for GLS stabilization
791 strong_residual_vec[q] += -k[q] * C;
792
793 for (unsigned int i = 0; i < n_dofs; ++i)
794 {
795 const auto &phi_T_i = scratch_data.phi[q][i];
796
797 // Add reaction term to the RHS
798 local_rhs(i) -= k[q] * phi_T_i * C * JxW;
799 }
800 } // end loop on quadrature points
801}
