21/**
22 * @brief Solver class for coupled Computational Fluid Dynamics and Discrete Element Method (CFD-DEM) simulations using a matrix-free formulation.
23 *
24 * This class implements a two-way coupled CFD-DEM solver that combines the
25 * Volume-averaged Navier-Stokes (VANS) equations for fluid flow with the
26 * soft-sphere Discrete Element Method for particle dynamics. The solver
27 * enables simulation of fluid-particle interactions in systems containing
28 * dense particle suspensions.
29 *
30 * The CFD component uses Galerkin Least Squares (GLS) stabilization for
31 * the VANS equations, while the DEM component handles particle-particle
32 * and particle-wall collisions using the soft-sphere contact model. A
33 * matrix-free formulation is used to solve the VANS equation.
34 *
35 * @tparam dim Spatial dimension of the simulation (2 or 3).
36 * While 2D simulation are theoretically supported, they have not been tested.
37 *
38 * @ingroup solvers
39 */

122 /**
123 * @brief Set the initial condition. The version in this class does not output
124 * the solution, since the void fraction is not initialized yet.
125 *
126 * @param[in] initial_condition_type Type of method use to impose initial
127 *condition.
128 *
129 * @param[in] restart Indicator if the simulation is being restarted or not.
130 *
131 **/
132
133 void
134 set_initial_condition(
135 const Parameters::FluidDynamicsInitialConditionType initial_condition_type,
136 const bool restart = false) override
137 {
138 unsigned int ref_iter = 0;
139 do
140 {
141 if (ref_iter > 0)
142 this->refine_mesh();
143
144 this->set_initial_condition_fd(initial_condition_type, restart);
145 if (!restart)
146 {
147 this->multiphysics->set_initial_conditions();
148 }
149 ref_iter++;
150 }
151 while (
152 ref_iter <
153 (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
154 restart == false);
155
156 this->pcout
157 << "---------------------------------------------------------------"
158 << std::endl;
159 }
