15/**
16 * @brief Finds the neighbor list (without repetition) of all the active
17 * cells in the triangulation. It gets the vertices of the cells to get lists
18 * of the neighbor for each cells. There is some check to prevent repetition
19 * of a cell in a list (up to 8 vertices can have the same cell in common in
20 * 3D).
21 * 2 types of container are used for cell neighbors : local-local cells
22 * and local-ghost cells.
23 *
24 * @tparam dim An integer that denotes the dimension of the space in which
25 * the problem is solved.
26 * @tparam reciprocal An boolean that denotes if the cells_local_neighbor_list of
27 * cell i and j contains cell j and i, respectively.
28 *
29 * @param triangulation Triangulation to access the information of the cells
30 * @param cells_local_neighbor_list A vector (with size of the local cell
31 * number) of vectors (local adjacent cells of each local cell). First element
32 * of each set shows the main cell itself
33 * @param cells_ghost_neighbor_list A vector (with size of the local cell
34 * number) of vectors (ghost adjacent cells of each local cell). First element
35 * of each set shows the main cell itself
36 */
37template <int dim, bool reciprocal = false>
38void
39find_cell_neighbors(
40 const parallel::distributed::Triangulation<dim> &triangulation,
41 typename DEM::dem_data_structures<dim>::cells_neighbor_list
42 &cells_local_neighbor_list,
43 typename DEM::dem_data_structures<dim>::cells_neighbor_list
44 &cells_ghost_neighbor_list);

46/**
47 * @brief Finds the periodic neighbor list (without repetition) of all the
48 * active cells in the triangulation. It gets the coinciding vertices of the
49 * cells at the periodic boundary 0 to get lists of the periodic neighbor
50 * cells on the periodic boundary 1 for each cells. There is some check to
51 * prevent repetition of a cell in a list (up to 8 vertices can have the
52 * same cell in common in 3D).
53 * 3 types of container are used for periodic mapping of cell neighbors :
54 * local-local cells, local-ghost cells and ghost-local cells. The last
55 * container is necessary since the mapping are only from periodic boundary 0
56 * to periodic boundary 1 and the ghost-local particle pairs need distinction
57 * for proper handling of search of particle pairs and contact forces.
58 *
59 * @param triangulation Triangulation to access the information of the cells
60 * @param periodic_boundaries_cells_information A container of information
61 * related to the pairs of cell at periodic boundaries, used to get the cells
62 * on periodic boundary 0
63 * @param cells_local_periodic_neighbor_list A vector (with size of the local
64 * cell number) of vectors (local adjacent cells of each local cell). First
65 * element of each set shows the main cell itself
66 * @param cells_ghost_periodic_neighbor_list A vector (with size of the local
67 * cell number) of vectors (ghost adjacent cells of each local cell). First
68 * element of each set shows the main cell itself
69 * @param cells_ghost_local_periodic_neighbor_list A vector (with size of the
70 * ghost cell number) of vectors (local adjacent cells of each ghost cell).
71 * First element of each set shows the main ghost cell itself
72 */
73template <int dim>
74void
75find_cell_periodic_neighbors(
76 const parallel::distributed::Triangulation<dim> &triangulation,
77 const typename DEM::dem_data_structures<dim>::periodic_boundaries_cells_info
78 &periodic_boundaries_cells_information,
79 typename DEM::dem_data_structures<dim>::cells_neighbor_list
80 &cells_local_periodic_neighbor_list,
81 typename DEM::dem_data_structures<dim>::cells_neighbor_list
82 &cells_ghost_periodic_neighbor_list,
83 typename DEM::dem_data_structures<dim>::cells_neighbor_list
84 &cells_ghost_local_periodic_neighbor_list);

104/**
105 * @brief Generate a periodic neighbor cells list of the cell. With the
106 * coinciding_vertex_groups and the vertex_to_coinciding_vertex_group, we can
107 * get the coinciding vertices on a periodic boundary and with v_to_c we can
108 * get the list of the periodic neighbor cells to the main cell.
109 *
110 * Here is an example to understand how the search works :
111 * Cell 8 have vertices 0, 1, 2, 3 on the periodic boundary. First,
112 * it checks in the coinciding_vertex_groups if the vertex 0 is a key in the
113 * map, and if it is, it gets a label, let's say label is 10. In the
114 * vertex_to_coinciding_vertex_group, it can finds the the coinciding vertices
115 * with the label. Using the label 10, it gets the vertices 0 and 34, which
116 * means that vertices 0 and 34 are periodic. We do not want the cells
117 * attached to the vertex 0 since they are already found with the regular
118 * find_cell_neighbor function, so it skips vertex 0, but gets the cells 21,
119 * 22, 23, 24 attached to the vertex 34. The same checks are done for vertices
120 * 1, 2 and 3 and all periodic cells are return as a vector.
121 *
122 * @param cell The cell that needs the periodic neighbor list
123 * @param coinciding_vertex_groups A map of coinciding vertices labeled by an
124 * arbitrary element from them
125 * @param vertex_to_coinciding_vertex_group Map of a vertex to the label of a
126 * group of coinciding vertices
127 * @param v_to_c A vector of set with adjacent cells of all the vertices
128 * @param periodic_neighbor_list A vector which is the list of periodic cell
129 * neighbors
130 */
131template <int dim>
132void
133get_periodic_neighbor_list(
134 const typename Triangulation<dim>::active_cell_iterator &cell,
135 const std::map<unsigned int, std::vector<unsigned int>>
136 &coinciding_vertex_groups,
137 const std::map<unsigned int, unsigned int> &vertex_to_coinciding_vertex_group,
138 const std::vector<std::set<typename Triangulation<dim>::active_cell_iterator>>
139 &v_to_c,
140 typename DEM::dem_data_structures<dim>::cell_vector &periodic_neighbor_list);
