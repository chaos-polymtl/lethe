78template <int dim>
79MultiphysicsInterface<dim>::MultiphysicsInterface(
80 const SimulationParameters<dim> &nsparam,
81 std::shared_ptr<parallel::DistributedTriangulationBase<dim>> p_triangulation,
82 std::shared_ptr<SimulationControl> p_simulation_control,
83 ConditionalOStream &p_pcout)
84 : multiphysics_parameters(nsparam.multiphysics)
85 , pcout(p_pcout)
86{
87 inspect_multiphysics_models_dependencies(nsparam);
88
89 // Fluid dynamics is always considered active
90 // since its DofHandler is required at all time by
91 // the other physics. Consequently, disabling it only
92 // prevents solving it, but not allocating it.
93 {
94 active_physics.push_back(PhysicsID::fluid_dynamics);
95 }
96 if (multiphysics_parameters.heat_transfer)
97 {
98 verbosity[PhysicsID::heat_transfer] =
99 (nsparam.physics_solving_strategy.at(PhysicsID::heat_transfer)
100 .verbosity != Parameters::Verbosity::quiet ||
101 nsparam.linear_solver.at(PhysicsID::heat_transfer).verbosity !=
102 Parameters::Verbosity::quiet) ?
103 Parameters::Verbosity::verbose :
104 Parameters::Verbosity::quiet;
105 active_physics.push_back(PhysicsID::heat_transfer);
106 physics[PhysicsID::heat_transfer] = std::make_shared<HeatTransfer<dim>>(
107 this, nsparam, p_triangulation, p_simulation_control);
108 }
109 if (multiphysics_parameters.tracer)
110 {
111 verbosity[PhysicsID::tracer] =
112 (nsparam.physics_solving_strategy.at(PhysicsID::tracer).verbosity !=
113 Parameters::Verbosity::quiet ||
114 nsparam.linear_solver.at(PhysicsID::tracer).verbosity !=
115 Parameters::Verbosity::quiet) ?
116 Parameters::Verbosity::verbose :
117 Parameters::Verbosity::quiet;
118 active_physics.push_back(PhysicsID::tracer);
119 physics[PhysicsID::tracer] = std::make_shared<Tracer<dim>>(
120 this, nsparam, p_triangulation, p_simulation_control);
121 }
122 if (multiphysics_parameters.VOF)
123 {
124 verbosity[PhysicsID::VOF] =
125 (nsparam.physics_solving_strategy.at(PhysicsID::VOF).verbosity !=
126 Parameters::Verbosity::quiet ||
127 nsparam.linear_solver.at(PhysicsID::VOF).verbosity !=
128 Parameters::Verbosity::quiet) ?
129 Parameters::Verbosity::verbose :
130 Parameters::Verbosity::quiet;
131 active_physics.push_back(PhysicsID::VOF);
132 physics[PhysicsID::VOF] = std::make_shared<VolumeOfFluid<dim>>(
133 this, nsparam, p_triangulation, p_simulation_control);
134 }
135
136 if (multiphysics_parameters.cahn_hilliard)
137 {
138 verbosity[PhysicsID::cahn_hilliard] =
139 (nsparam.physics_solving_strategy.at(PhysicsID::cahn_hilliard)
140 .verbosity != Parameters::Verbosity::quiet ||
141 nsparam.linear_solver.at(PhysicsID::cahn_hilliard).verbosity !=
142 Parameters::Verbosity::quiet) ?
143 Parameters::Verbosity::verbose :
144 Parameters::Verbosity::quiet;
145 active_physics.push_back(PhysicsID::cahn_hilliard);
146 physics[PhysicsID::cahn_hilliard] = std::make_shared<CahnHilliard<dim>>(
147 this, nsparam, p_triangulation, p_simulation_control);
148 }
149
150 if (multiphysics_parameters.electromagnetics)
151 {
152 verbosity[PhysicsID::electromagnetics] =
153 (nsparam.linear_solver.at(PhysicsID::electromagnetics).verbosity !=
154 Parameters::Verbosity::quiet) ?
155 Parameters::Verbosity::verbose :
156 Parameters::Verbosity::quiet;
157
158 active_physics.push_back(PhysicsID::electromagnetics);
159 physics[PhysicsID::electromagnetics] =
160 std::make_shared<TimeHarmonicMaxwell<dim>>(this,
161 nsparam,
162 p_triangulation,
163 p_simulation_control);
164 }
165}

167template <int dim>
168const GlobalVectorType &
169MultiphysicsInterface<dim>::get_projected_phase_fraction_gradient_solution()
170{
171 // Throw error if VOF is not enabled
172 AssertThrow((std::ranges::find(active_physics, PhysicsID::VOF) !=
173 active_physics.end()),
174 ExcInternalError());
175 // Throw error if surface tension force is not enabled
176 AssertThrow(
177 (multiphysics_parameters.vof_parameters.surface_tension_force.enable),
178 ExcInternalError());
179
180 return dynamic_cast<VolumeOfFluid<dim> &>(*physics[PhysicsID::VOF])
181 .get_projected_phase_fraction_gradient_solution();
182}

252template <int dim>
253void
254MultiphysicsInterface<dim>::inspect_multiphysics_models_dependencies(
255 const SimulationParameters<dim> &nsparam)
256{
257 bool thermal_buoyancy_force_enabled =
258 nsparam.multiphysics.thermal_buoyancy_force;
259 bool heat_transfer_enabled = nsparam.multiphysics.heat_transfer;
260 bool marangoni_effect_enabled =
261 nsparam.multiphysics.vof_parameters.surface_tension_force
262 .enable_marangoni_effect;
263 bool surface_tension_force_enabled =
264 nsparam.multiphysics.vof_parameters.surface_tension_force.enable;
265 bool fluid_dynamics_enabled = nsparam.multiphysics.fluid_dynamics;
266 bool interface_sharpening_enabled =
267 nsparam.multiphysics.vof_parameters.regularization_method.sharpening.enable;
268 bool VOF_enabled = nsparam.multiphysics.VOF;
269 bool cahn_hilliard_enabled = nsparam.multiphysics.cahn_hilliard;
270
271 // To avoid getting unused parameter warning
272 _unused(thermal_buoyancy_force_enabled && heat_transfer_enabled &&
273 marangoni_effect_enabled && surface_tension_force_enabled &&
274 fluid_dynamics_enabled && interface_sharpening_enabled &&
275 VOF_enabled && cahn_hilliard_enabled);
276
277 // Dependence of thermal buoyancy force on fluid dynamics
278 AssertThrow(!(thermal_buoyancy_force_enabled == true &&
279 fluid_dynamics_enabled == false),
280 ThermalBuoyancyWithoutFluidDynamicsError(
281 thermal_buoyancy_force_enabled));
282
283 // Dependence of thermal buoyancy force on heat transfer
284 AssertThrow(
285 !(thermal_buoyancy_force_enabled == true && heat_transfer_enabled == false),
286 ThermalBuoyancyWithoutHeatTransferError(thermal_buoyancy_force_enabled));
287
288 // Dependence of Marangoni effect on fluid dynamics
289 AssertThrow(!(marangoni_effect_enabled == true &&
290 fluid_dynamics_enabled == false),
291 MarangoniWithoutFluidDynamicsError(marangoni_effect_enabled));
292
293 // Dependence of Marangoni effect on heat transfer
294 AssertThrow(!(marangoni_effect_enabled == true &&
295 heat_transfer_enabled == false),
296 MarangoniWithoutHeatTransferError(marangoni_effect_enabled));
297
298 // Dependence of Marangoni effect on surface tension force
299 AssertThrow(!(marangoni_effect_enabled == true &&
300 surface_tension_force_enabled == false),
301 MarangoniWithoutSurfaceTensionForceError(
302 marangoni_effect_enabled));
303
304 // Dependence of surface tension force on VOF
305 AssertThrow(!(surface_tension_force_enabled == true && VOF_enabled == false),
306 SurfaceTensionForceWithoutVOFError(
307 surface_tension_force_enabled));
308
309 // Dependence of Marangoni effect on VOF
310 AssertThrow(!(marangoni_effect_enabled == true && VOF_enabled == false),
311 MarangoniWithoutVOFError(marangoni_effect_enabled));
312
313 // Dependence of interface sharpening on VOF
314 AssertThrow(!(interface_sharpening_enabled == true && VOF_enabled == false),
315 InterfaceSharpeningWithoutVOFError(interface_sharpening_enabled));
316
317 // Cahn-Hilliard does not support heat transfer
318 AssertThrow(!(cahn_hilliard_enabled == true && heat_transfer_enabled == true),
319 CahnHilliardWithHeatTransferError());
320
321 // Cahn-Hilliard does not support thermal buoyancy force
322 AssertThrow(!(cahn_hilliard_enabled == true &&
323 thermal_buoyancy_force_enabled == true),
324 CahnHilliardWithThermalBuoyancyForceError());
325}
