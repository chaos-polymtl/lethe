69template <int dim>
70MultiphysicsInterface<dim>::MultiphysicsInterface(
71 const SimulationParameters<dim> &nsparam,
72 std::shared_ptr<parallel::DistributedTriangulationBase<dim>> p_triangulation,
73 std::shared_ptr<SimulationControl> p_simulation_control,
74 ConditionalOStream &p_pcout)
75 : multiphysics_parameters(nsparam.multiphysics)
76 // ,
77 // verbosity(nsparam.non_linear_solver.at(PhysicsID::fluid_dynamics).verbosity)
78 , pcout(p_pcout)
79{
80 inspect_multiphysics_models_dependencies(nsparam);
81
82 // Fluid dynamics is always considered active
83 // since its DofHandler is required at all time by
84 // the other physics. Consequently, disabling it only
85 // prevents solving it, but not allocating it.
86 {
87 // verbosity[PhysicsID::fluid_dynamics] =
88 // (nsparam.non_linear_solver.at(PhysicsID::fluid_dynamics).verbosity !=
89 // Parameters::Verbosity::quiet ||
90 // nsparam.linear_solver.at(PhysicsID::fluid_dynamics).verbosity !=
91 // Parameters::Verbosity::quiet) ? Parameters::Verbosity::verbose :
92 // Parameters::Verbosity::quiet;
93 active_physics.push_back(PhysicsID::fluid_dynamics);
94 }
95 if (multiphysics_parameters.heat_transfer)
96 {
97 verbosity[PhysicsID::heat_transfer] =
98 (nsparam.non_linear_solver.at(PhysicsID::heat_transfer).verbosity !=
99 Parameters::Verbosity::quiet ||
100 nsparam.linear_solver.at(PhysicsID::heat_transfer).verbosity !=
101 Parameters::Verbosity::quiet) ?
102 Parameters::Verbosity::verbose :
103 Parameters::Verbosity::quiet;
104 active_physics.push_back(PhysicsID::heat_transfer);
105 physics[PhysicsID::heat_transfer] = std::make_shared<HeatTransfer<dim>>(
106 this, nsparam, p_triangulation, p_simulation_control);
107 }
108 if (multiphysics_parameters.tracer)
109 {
110 verbosity[PhysicsID::tracer] =
111 (nsparam.non_linear_solver.at(PhysicsID::tracer).verbosity !=
112 Parameters::Verbosity::quiet ||
113 nsparam.linear_solver.at(PhysicsID::tracer).verbosity !=
114 Parameters::Verbosity::quiet) ?
115 Parameters::Verbosity::verbose :
116 Parameters::Verbosity::quiet;
117 active_physics.push_back(PhysicsID::tracer);
118 physics[PhysicsID::tracer] = std::make_shared<Tracer<dim>>(
119 this, nsparam, p_triangulation, p_simulation_control);
120 }
121 if (multiphysics_parameters.VOF)
122 {
123 verbosity[PhysicsID::VOF] =
124 (nsparam.non_linear_solver.at(PhysicsID::VOF).verbosity !=
125 Parameters::Verbosity::quiet ||
126 nsparam.linear_solver.at(PhysicsID::VOF).verbosity !=
127 Parameters::Verbosity::quiet) ?
128 Parameters::Verbosity::verbose :
129 Parameters::Verbosity::quiet;
130 active_physics.push_back(PhysicsID::VOF);
131 physics[PhysicsID::VOF] = std::make_shared<VolumeOfFluid<dim>>(
132 this, nsparam, p_triangulation, p_simulation_control);
133 }
134
135 if (multiphysics_parameters.cahn_hilliard)
136 {
137 verbosity[PhysicsID::cahn_hilliard] =
138 (nsparam.non_linear_solver.at(PhysicsID::cahn_hilliard).verbosity !=
139 Parameters::Verbosity::quiet ||
140 nsparam.linear_solver.at(PhysicsID::cahn_hilliard).verbosity !=
141 Parameters::Verbosity::quiet) ?
142 Parameters::Verbosity::verbose :
143 Parameters::Verbosity::quiet;
144 active_physics.push_back(PhysicsID::cahn_hilliard);
145 physics[PhysicsID::cahn_hilliard] = std::make_shared<CahnHilliard<dim>>(
146 this, nsparam, p_triangulation, p_simulation_control);
147 }
148}

150template <int dim>
151const GlobalVectorType &
152MultiphysicsInterface<dim>::get_projected_phase_fraction_gradient_solution()
153{
154 // Throw error if VOF is not enabled
155 AssertThrow((std::find(active_physics.begin(),
156 active_physics.end(),
157 PhysicsID::VOF) != active_physics.end()),
158 ExcInternalError());
159 // Throw error if surface tension force is not enabled
160 AssertThrow(
161 (multiphysics_parameters.vof_parameters.surface_tension_force.enable),
162 ExcInternalError());
163
164 return dynamic_cast<VolumeOfFluid<dim> &>(*physics[PhysicsID::VOF])
165 .get_projected_phase_fraction_gradient_solution();
166}

238template <int dim>
239void
240MultiphysicsInterface<dim>::inspect_multiphysics_models_dependencies(
241 const SimulationParameters<dim> &nsparam)
242{
243 bool buoyancy_force_enabled = nsparam.multiphysics.buoyancy_force;
244 bool heat_transfer_enabled = nsparam.multiphysics.heat_transfer;
245 bool marangoni_effect_enabled =
246 nsparam.multiphysics.vof_parameters.surface_tension_force
247 .enable_marangoni_effect;
248 bool surface_tension_force_enabled =
249 nsparam.multiphysics.vof_parameters.surface_tension_force.enable;
250 bool fluid_dynamics_enabled = nsparam.multiphysics.fluid_dynamics;
251 bool interface_sharpening_enabled =
252 nsparam.multiphysics.vof_parameters.regularization_method.sharpening.enable;
253 bool VOF_enabled = nsparam.multiphysics.VOF;
254
255 // To avoid getting unused parameter warning
256 _unused(buoyancy_force_enabled && heat_transfer_enabled &&
257 marangoni_effect_enabled && surface_tension_force_enabled &&
258 fluid_dynamics_enabled && interface_sharpening_enabled &&
259 VOF_enabled);
260
261 // Dependence of buoyant force on fluid dynamics
262 AssertThrow(!(buoyancy_force_enabled == true &&
263 fluid_dynamics_enabled == false),
264 BuoyancyWithoutFluidDynamicsError(buoyancy_force_enabled));
265
266 // Dependence of buoyant force on heat transfer
267 AssertThrow(!(buoyancy_force_enabled == true &&
268 heat_transfer_enabled == false),
269 BuoyancyWithoutHeatTransferError(buoyancy_force_enabled));
270
271 // Dependence of Marangoni effect on fluid dynamics
272 AssertThrow(!(marangoni_effect_enabled == true &&
273 fluid_dynamics_enabled == false),
274 MarangoniWithoutFluidDynamicsError(marangoni_effect_enabled));
275
276 // Dependence of Marangoni effect on heat transfer
277 AssertThrow(!(marangoni_effect_enabled == true &&
278 heat_transfer_enabled == false),
279 MarangoniWithoutHeatTransferError(marangoni_effect_enabled));
280
281 // Dependence of Marangoni effect on surface tension force
282 AssertThrow(!(marangoni_effect_enabled == true &&
283 surface_tension_force_enabled == false),
284 MarangoniWithoutSurfaceTensionForceError(
285 marangoni_effect_enabled));
286
287 // Dependence of surface tension force on VOF
288 AssertThrow(!(surface_tension_force_enabled == true && VOF_enabled == false),
289 SurfaceTensionForceWithoutVOFError(
290 surface_tension_force_enabled));
291
292 // Dependence of Marangoni effect on VOF
293 AssertThrow(!(marangoni_effect_enabled == true && VOF_enabled == false),
294 MarangoniWithoutVOFError(marangoni_effect_enabled));
295
296 // Dependence of interface sharpening on VOF
297 AssertThrow(!(interface_sharpening_enabled == true && VOF_enabled == false),
298 InterfaceSharpeningWithoutVOFError(interface_sharpening_enabled));
299}
