17/**
18 * @brief Generates a rectangular channel mesh with a meshed cylinder
19 * obstacle. By default, this geometry defines two material IDs. The
20 * obstacle is assigned the material_id = 1 and the remaining domain is
21 * assigned material_id = 0.
22 *
23 * The mesh consists of three concentric regions around the cylinder:
24 * -# An inner balanced disk of radius @p inner_radius (hyper_ball_balanced).
25 * -# A transition shell between @p inner_radius and @p outer_radius, whose
26 * diagonal vertices are adjusted to form a square.
27 * -# Rectangular padding regions that fill the gap between the transition
28 * square and the channel boundary defined by @p bottom_left / @p top_right.
29 *
30 * The number of subdivisions in each padding direction (bottom, top, left,
31 * right) can be controlled independently. In 3D, the 2D cross-section is
32 * extruded along the z-axis with a configurable height and number of slices.
33 * The slices refer to the number of layers in the z direction, so the total
34 * number of cells in the z direction will be n_slices - 1. It follows that the
35 * minimum number of slices is 2, which corresponds to a single layer of cells
36 * in the z direction. Note that it is always assumed that the channel bottom is
37 * at z=0 and the channel top is at z=height.
38 *
39 * The manifold objects attached for proper mesh refinement are:
40 * - PolarManifold (2D) or CylindricalManifold (3D) on the inner cylinder
41 * boundary faces (manifold_id = 1).
42 * - TransfiniteInterpolationManifold on the remaining domain (manifold_id = 0).
43 */
44template <int dim, int spacedim>
45class UniformChannelWithMeshedCylinderGrid
46{
47public:
48 /**
49 * @brief Constructor that parses geometry parameters from a colon-separated
50 * string.
51 *
52 * @param[in] grid_arguments A colon-separated string with the following
53 * fields (coordinates are comma-separated):
54 *
55 * | # | Field | Format | Required | Description |
56 * |----|--------------|------------|----------|----------------------------------------------------|
57 * | 0 | bottom_left | x,y | yes | Bottom-left corner of the channel before extrusion |
58 * | 1 | top_right | x,y | yes | Top-right corner of the channel before extrusion |
59 * | 2 | center | x,y | yes | Center of the cylinder |
60 * | 3 | inner_radius | double | yes | Radius of the cylinder |
61 * | 4 | outer_radius | double | yes | Radius of the transition region |
62 * | 5 | pad_bottom | int | no | N cells below the transition (default: 0) |
63 * | 6 | pad_top | int | no | N cells above the transition (default: 0) |
64 * | 7 | pad_left | int | no | N cells left of the transition (default: 0) |
65 * | 8 | pad_right | int | no | N cells right of the transition (default: 0) |
66 * | 9 | height | double | no | Extrusion height in z, 3D only (default: 1.0) |
67 * | 10 | n_slices | int | no | Number of z-layers, 3D only (default and min: 2) |
68 * | 11 | colorize | true/false | no | Assign distinct boundary IDs (default: false) |
69 *
70 * Example: @code "0,0 : 10,2 : 5,1 : 0.1 : 0.3 : 2 : 2 : 5 : 5 : 2. : 2 :
71 * true"
72 * @endcode
73 *
74 * When @p colorize is true, boundary IDs follow the
75 * subdivided_hyper_rectangle convention: 0 (-x), 1 (+x), 2 (-y), 3 (+y).
76 * In 3D, the extruded bottom (z=0) gets id 4 and top (z=height) gets id 5.
77 */
78 UniformChannelWithMeshedCylinderGrid(const std::string &grid_arguments);
79
80 /**
81 * @brief Generate the 2D or 3D channel mesh with a meshed cylinder and attach
82 * manifold objects for proper mesh refinement.
83 *
84 * For 2D: Delegates geometry construction to generate_2d_channel_mesh(), then
85 * attaches:
86 * - PolarManifold on the inner-cylinder boundary faces (manifold_id = 1).
87 * - TransfiniteInterpolationManifold on the remaining domain (manifold_id =
88 * 0).
89 * The boundary IDs are assigned following the subdivided_hyper_rectangle
90 * convention when colorize is enabled: 0 (-x), 1 (+x), 2 (-y), 3 (+y).
91 *
92 * For 3D: Delegates geometry construction to generate_2d_channel_mesh(), then
93 * extrudes the mesh along the z-axis and attaches:
94 * - CylindricalManifold on the cylinder surface (manifold_id = 1).
95 * - TransfiniteInterpolationManifold on the remaining domain (manifold_id =
96 * 0). The lateral boundary IDs are inherited from the 2D mesh when colorize
97 * is enabled. The extruded bottom (z = 0) gets boundary_id = 4 and the top (z
98 * = height) gets boundary_id = 5.
99 *
100 * @param[out] triangulation The triangulation to fill with the channel mesh.
101 */
102 void
103 make_grid(Triangulation<dim, spacedim> &triangulation);
104
105private:
106 /**
107 * @brief Generate the 2D channel mesh geometry with a meshed cylinder.
108 *
109 * This private helper creates the composite 2D triangulation by merging:
110 * - A balanced disk of radius @p inner_radius (the cylinder cross-section).
111 * - A hyper_shell transition region with squared diagonal vertices between
112 * @p inner_radius and @p outer_radius.
113 * - Up to 8 rectangular padding sub-meshes filling the gap between the
114 * transition square and the channel boundary.
115 *
116 * Manifold integer IDs are assigned (0 = TFI, 1 = polar) but no manifold
117 * objects are attached — the caller must do that after this function returns.
118 *
119 * If @p colorize is true, boundary IDs are set following the
120 * subdivided_hyper_rectangle convention: 0 (-x), 1 (+x), 2 (-y), 3 (+y).
121 *
122 * @param[out] triangulation The triangulation to fill.
123 * @param[in] bottom_left Bottom-left corner of the channel.
124 * @param[in] top_right Top-right corner of the channel.
125 * @param[in] center Center of the cylinder.
126 * @param[in] inner_radius Radius of the inner cylinder.
127 * @param[in] outer_radius Radius of the transition region.
128 * @param[in] pad_bottom Number of subdivisions in the bottom padding.
129 * @param[in] pad_top Number of subdivisions in the top padding.
130 * @param[in] pad_left Number of subdivisions in the left padding.
131 * @param[in] pad_right Number of subdivisions in the right padding.
132 * @param[in] colorize Whether to assign distinct boundary IDs.
133 */
134 void
135 generate_2d_channel_mesh(Triangulation<2> &triangulation,
136 const Point<2> &bottom_left,
137 const Point<2> &top_right,
138 const Point<2> &center,
139 const double inner_radius,
140 const double outer_radius,
141 const unsigned int pad_bottom,
142 const unsigned int pad_top,
143 const unsigned int pad_left,
144 const unsigned int pad_right,
145 const bool colorize);

147 /// The original input string with the geometry parameters.
148 std::string grid_arguments;
149 /// Point that define the bottom-left of the channel in the xy-plane.
150 Point<dim> bottom_left;
151 /// Point that define the top-right of the channel in the xy-plane.
152 Point<dim> top_right;
153 /// Point that define the center of the cylinder in the xy-plane.
154 Point<dim> center;
155 /// Radius of the cylinder.
156 double inner_radius;
157 /// Radius of the transition region between the cylinder and the channel.
158 double outer_radius;
159 /// Number of additional cells to be added to pad the channel from center -
160 /// inner_radius - outer_radius to the channel bottom boundary (-y direction).
161 unsigned int pad_bottom;
162 /// Number of additional cells to be added to pad the channel from center +
163 /// inner_radius + outer_radius to the channel top boundary (+y direction).
164 unsigned int pad_top;
165 /// Number of additional cells to be added to pad the channel from center -
166 /// inner_radius - outer_radius to the channel left boundary (-x direction).
167 unsigned int pad_left;
168 /// Number of additional cells to be added to pad the channel from center +
169 /// inner_radius + outer_radius to the channel right boundary (+x direction).
170 unsigned int pad_right;
171 /// Extrusion height in the z direction, 3D only.
172 double height;
173 /// Number of layers in the z direction, 3D only. Minimum is 2, which
174 /// corresponds to a single layer of cells in the z direction.
175 unsigned int n_slices;
176 /// Whether to assign distinct boundary IDs to the channel boundaries
177 /// following the subdivided_hyper_rectangle convention.
178 bool colorize;
179};
