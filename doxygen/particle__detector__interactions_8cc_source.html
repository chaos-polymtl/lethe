9template <int dim>
10void
11ParticleDetectorInteractions<dim>::calculate_position_parameters()
12{
13 // Vector from the detector face center to the particle position (Xdp)
14 Tensor<1, dim> face_detector_particle_distance =
15 particle_position - detector_face_position;
16
17 // Detector orientation vector parallel to z' in particle-detector coordinate
18 // (e'z)
19 Tensor<1, dim> face_middle_vector =
20 detector_face_position - detector_middle_position;
21 detector_orientation_z = face_middle_vector / face_middle_vector.norm();
22
23 // Projection of the distance vector on the detector axis which is parallel to
24 // the z' axis Particle position in particle-detector coordinate frame (h =
25 // z'p)
26 h = scalar_product(face_detector_particle_distance, detector_orientation_z);
27
28 // x' coordinate of the particle-detector frame (e'x) (see equations 29 to 31)
29 Tensor<1, dim> Z = h * detector_orientation_z;
30 Tensor<1, dim> X = Z - face_detector_particle_distance;
31
32 if (X.norm() < 1e-6) // When particle is aligned to the detector
33 {
34 detector_orientation_x[0] = 0;
35 detector_orientation_x[1] = 0;
36 detector_orientation_x[2] = 1;
37 }
38 else
39 detector_orientation_x = X / X.norm();
40
41 // y' coordinate of the particle-detector frame (e'y)
42 detector_orientation_y =
43 cross_product_3d(detector_orientation_z, detector_orientation_x);
44
45 // Rho parameter : distance between the center of the detector and a line
46 // parallel to the detector axis containing the particle position
47 rho = X.norm();
48}

50template <int dim>
51void
52ParticleDetectorInteractions<dim>::calculate_solid_angle(double n_alpha,
53 double n_theta)
54{
55 double alpha_max;
56 double theta_max = 0, theta_min = 0;
57 double r = detector_radius;
58 double l = detector_length;
59
60 // Case 1 or 2: S1 or S2 (Tracer views the detector from both top and/or
61 // lateral side of the detector)
62 if (rho > detector_radius)
63 {
64 // Calculate alpha angles and its weigthting factor for the Monte Carlo
65 // iteration
66 alpha_max = std::asin(r / rho);
67 alpha = alpha_max * (2 * n_alpha - 1);
68 weighting_factor_alpha = alpha_max / M_PI;
69
70 // Calculate relevant distances prior next and other calculations
71 OB_distance =
72 rho * std::cos(alpha) -
73 std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
74 OA_distance =
75 rho * std::cos(alpha) +
76 std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
77
78 if (h > 0) // Case 1 : S1 (h > 0)
79 {
80 theta_max = std::atan(OA_distance / h);
81 theta_cri = std::atan(OB_distance / h);
82 theta_min = std::atan(OB_distance / (h + l));
83 }
84 else if (std::abs(h) < 1e-6) // Case 1 : S1 (h = 0)
85 {
86 theta_max = M_PI_2; // pi/2
87 theta_cri = M_PI_2; // pi/2
88 theta_min = std::atan(OB_distance / l);
89 }
90 else // Cases where h < 0 happens are not implemented yet since they are
91 // not usual
92 {
93 AssertThrow(
94 h >= 0,
95 ExcMessage(
96 "For now, it is not possible to calculate count for particle positions on the side of the detector."));
97 }
98
99 // Calculate theta angle and its weighting factor for the Monte Carlo
100 // iteration
101 theta = std::acos(std::cos(theta_min) -
102 n_theta * (std::cos(theta_min) - std::cos(theta_max)));
103 weighting_factor_theta = (std::cos(theta_min) - std::cos(theta_max)) / 2;
104 }
105 // Case 3 : S3 (Tracer views the detector only from the top surface)
106 else // (rho < detector_radius)
107 {
108 theta_max = std::atan((r + rho) / h);
109 theta_cri = std::atan((r - rho) / h);
110 theta_min = 0;
111
112 // Calculate theta angle and its weigthting factor for the Monte Carlo
113 // iteration
114 theta = std::acos(std::cos(theta_min) -
115 n_theta * (std::cos(theta_min) - std::cos(theta_max)));
116 weighting_factor_theta = (std::cos(theta_min) - std::cos(theta_max)) / 2;
117
118 if (theta <= theta_cri)
119 {
120 alpha_max = M_PI;
121 weighting_factor_alpha = 1;
122 }
123 else // theta > theta_cri
124 {
125 alpha_max =
126 std::acos((std::pow(rho, 2) + std::pow(h * std::tan(theta), 2) -
127 std::pow(r, 2)) /
128 (2 * h * rho * std::tan(theta)));
129 weighting_factor_alpha = alpha_max / M_PI;
130 }
131 alpha = alpha_max * (2 * n_alpha - 1);
132
133 // Calculate relevant distances prior next and other calculations
134 OB_distance =
135 rho * std::cos(alpha) -
136 std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
137 OA_distance =
138 rho * std::cos(alpha) +
139 std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
140 }
141}

143template <int dim>
144double
145ParticleDetectorInteractions<dim>::calculate_detector_path_length()
146{
147 double detector_path_length;
148 double l = detector_length;
149 double theta_1 = std::atan(OA_distance / (h + l));
150
151 // Case 1 : S1 (Tracer views the detector from both top and lateral side of
152 // the detector)
153 if (rho > detector_radius)
154 {
155 if (theta <= theta_cri && theta <= theta_1)
156 detector_path_length =
157 (h + l) / std::cos(theta) - OB_distance / std::sin(theta);
158 else if (theta <= theta_cri && theta > theta_1)
159 detector_path_length = (OA_distance - OB_distance) / std::sin(theta);
160 else if (theta > theta_cri && theta <= theta_1)
161 detector_path_length = l / std::cos(theta);
162 else // (theta > theta_cri && theta > theta_1)
163 detector_path_length =
164 OA_distance / std::sin(theta) - h / std::cos(theta);
165 }
166 // Case 3 : S3 (Tracer views the detector only from the top surface)
167 else
168 {
169 double theta_2 = std::atan(OA_distance / h);
170
171 if (theta <= theta_1)
172 detector_path_length = l / std::cos(theta);
173 else if (theta > theta_1 && theta <= theta_2)
174 detector_path_length =
175 OA_distance / std::sin(theta) - h / std::cos(theta);
176 else // Do not enters in the detector
177 detector_path_length = 0;
178 }
179
180 return detector_path_length;
181}

183template <int dim>
184std::vector<double>
185ParticleDetectorInteractions<dim>::solve_t(
186 Tensor<2, dim> e_inverse,
187 Tensor<1, dim> detector_particle_origin,
188 Tensor<1, dim> particle_position_rotation)
189{
190 // Function value when evaluate with t (circle equation)
191 auto F = [=, *this](const double t) {
192 double R = parameters.reactor_radius;
193
194 Tensor<1, dim> line_equations(
195 {particle_position_rotation[0] + t * std::sin(theta) * std::cos(alpha),
196 particle_position_rotation[1] + t * std::sin(theta) * std::sin(alpha),
197 particle_position_rotation[2] + t * std::cos(M_PI - theta)});
198
199 return std::pow((e_inverse * (line_equations) +
200 detector_particle_origin)[0],
201 2) +
202 std::pow((e_inverse * (line_equations) +
203 detector_particle_origin)[1],
204 2) -
205 std::pow(R, 2);
206 };
207
208 // Numerical derivative of the function value when evaluate with t
209 auto dF = [=, *this](const double t) {
210 double dt = 0.0001;
211
212 return (F(t + dt) - F(t - dt)) / (2 * dt);
213 };
214
215 // Values for solving (positive/negative)
216 std::vector<double> t0 = {-1, 1};
217 double tolerance, dt0, max_iteration;
218 tolerance = 1e-6;
219 max_iteration = 1000;
220 dt0 = 1;
221
222 // Newton method
223 unsigned int i;
224 double t, dtn;
225 std::vector<double> t_solutions(t0.size(), 0);
226
227 for (unsigned int n = 0; n < t0.size(); n++)
228 {
229 i = 0;
230 dtn = dt0;
231 t = t0[n];
232
233 while (std::abs(dtn) > tolerance && i < max_iteration)
234 {
235 dtn = -F(t) / dF(t);
236 t += dtn;
237 i++;
238
239 if (i == max_iteration)
240 std::cout << "Solving t didn't work" << std::endl;
241 }
242
243 t_solutions[n] = t;
244 }
245
246 return t_solutions;
247}

249template <int dim>
250double
251ParticleDetectorInteractions<dim>::calculate_reactor_path_length()
252{
253 Tensor<1, dim> h_vector, detector_particle_origin,
254 particle_position_translation, particle_position_rotation;
255 Point<dim> intersection_position;
256
257 // Calculate translated particle position (eq 59 - 63)
258 h_vector = h * detector_orientation_z;
259 detector_particle_origin = particle_position - h_vector;
260 particle_position_translation = particle_position - detector_particle_origin;
261
262 // Calculate rotated particle position (eq 64)
263 Tensor<1, dim, Tensor<1, dim>> detector_orientation_vector(
264 {detector_orientation_x, detector_orientation_y, detector_orientation_z});
265 Tensor<2, dim> detector_orientation_matrix(detector_orientation_vector);
266 particle_position_rotation =
267 detector_orientation_matrix * particle_position_translation;
268
269 // Find t value of the parametric equation for
270 std::vector<double> t = solve_t(invert(detector_orientation_matrix),
271 detector_particle_origin,
272 particle_position_rotation);
273
274 // Evaluate and determinate the intersection point closer to the detector
275 std::vector<Point<dim>> intersection_point(t.size());
276 std::vector<Tensor<1, dim>> intersection_detector_distance_vector(t.size());
277 std::vector<double> intersection_detector_distance(t.size());
278
279 for (unsigned int i = 0; i < t.size(); i++)
280 {
281 Tensor<1, dim> line_equations(
282 {particle_position_rotation[0] +
283 t[i] * std::sin(theta) * std::cos(alpha),
284 particle_position_rotation[1] +
285 t[i] * std::sin(theta) * std::sin(alpha),
286 particle_position_rotation[2] + t[i] * std::cos(M_PI - theta)});
287
288 intersection_point[i] =
289 detector_particle_origin +
290 invert(detector_orientation_matrix) * line_equations;
291
292 intersection_detector_distance_vector[i] =
293 detector_face_position - intersection_point[i];
294 intersection_detector_distance[i] =
295 std::abs(intersection_detector_distance_vector[i].norm());
296 }
297
298 double reactor_path_length =
299 (intersection_detector_distance[0] < intersection_detector_distance[1]) ?
300 std::abs((particle_position - intersection_point[0]).norm()) :
301 std::abs((particle_position - intersection_point[1]).norm());
302
303 return reactor_path_length;
304}

329template <int dim>
330void
331ParticleDetectorInteractions<dim>::calculate_efficiency()
332{
333 // Initialize efficiency
334 efficiency = 0;
335
336 // Calculate position parameters for the particle position
337 calculate_position_parameters();
338
339 const double denum = 1. / RAND_MAX;
340
341 for (unsigned int i = 0; i < parameters.n_monte_carlo_iteration; i++)
342 {
343 // Generate random values for Monte Carlo
344 double n_alpha = (double)rand() * denum;
345 double n_theta = (double)rand() * denum;
346
347 // Initialize variables
348 double detector_path_length, reactor_path_length,
349 detector_interaction_probability, non_interaction_probability;
350
351 // Calculate all parameters and distances for efficiency
352 calculate_solid_angle(n_alpha, n_theta);
353 detector_path_length = calculate_detector_path_length();
354 detector_interaction_probability =
355 calculate_detector_interaction_probability(detector_path_length);
356 reactor_path_length = calculate_reactor_path_length();
357 non_interaction_probability =
358 calculate_non_interaction_probability(reactor_path_length);
359
360 efficiency += weighting_factor_alpha * weighting_factor_theta *
361 detector_interaction_probability *
362 non_interaction_probability;
363 }
364
365 efficiency /= parameters.n_monte_carlo_iteration;
366}

368template <int dim>
369double
370ParticleDetectorInteractions<dim>::calculate_count()
371{
372 double T, nu, R, phi, tau, count;
373
374 T = parameters.sampling_time;
375 nu = parameters.gamma_rays_emitted;
376 R = activity;
377 phi = parameters.peak_to_total_ratio;
378 tau = dead_time;
379
380 calculate_efficiency();
381
382 // Calculate count for a particle position and one detector
383 count =
384 (T * nu * R * phi * efficiency) / (1 + tau * nu * R * phi * efficiency);
385
386 return count;
387}
