8 ParticleDetectorInteractions<dim>::calculate_position_parameters()
9 {
10  // Vector from the detector face center to the particle position (Xdp)
11  Tensor<1, dim> face_detector_particle_distance =
12  particle_position - detector_face_position;
13 
14  // Detector orientation vector parallel to z' in particle-detector coordinate
15  // (e'z)
16  Tensor<1, dim> face_middle_vector =
17  detector_face_position - detector_middle_position;
18  detector_orientation_z = face_middle_vector / face_middle_vector.norm();
19 
20  // Projection of the distance vector on the detector axis which is parallel to
21  // the z' axis Particle position in particle-detector coordinate frame (h =
22  // z'p)
23  h = scalar_product(face_detector_particle_distance, detector_orientation_z);
24 
25  // x' coordinate of the particle-detector frame (e'x) (see equations 29 to 31)
26  Tensor<1, dim> Z = h * detector_orientation_z;
27  Tensor<1, dim> X = Z - face_detector_particle_distance;
28 
29  if (X.norm() < 1e-6) // When particle is aligned to the detector
30  {
31  detector_orientation_x[0] = 0;
32  detector_orientation_x[1] = 0;
33  detector_orientation_x[2] = 1;
34  }
35  else
36  detector_orientation_x = X / X.norm();
37 
38  // y' coordinate of the particle-detector frame (e'y)
39  detector_orientation_y =
40  cross_product_3d(detector_orientation_z, detector_orientation_x);
41 
42  // Rho parameter : distance between the center of the detector and a line
43  // parallel to the detector axis containing the particle position
44  rho = X.norm();
45 }

49 ParticleDetectorInteractions<dim>::calculate_solid_angle(double n_alpha,
50  double n_theta)
51 {
52  double alpha_max;
53  double theta_max = 0, theta_min = 0;
54  double r = detector_radius;
55  double l = detector_length;
56 
57  // Case 1 or 2: S1 or S2 (Tracer views the detector from both top and/or
58  // lateral side of the detector)
59  if (rho > detector_radius)
60  {
61  // Calculate alpha angles and its weigthting factor for the Monte Carlo
62  // iteration
63  alpha_max = std::asin(r / rho);
64  alpha = alpha_max * (2 * n_alpha - 1);
65  weighting_factor_alpha = alpha_max / M_PI;
66 
67  // Calculate relevant distances prior next and other calculations
68  OB_distance =
69  rho * std::cos(alpha) -
70  std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
71  OA_distance =
72  rho * std::cos(alpha) +
73  std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
74 
75  if (h > 0) // Case 1 : S1 (h > 0)
76  {
77  theta_max = std::atan(OA_distance / h);
78  theta_cri = std::atan(OB_distance / h);
79  theta_min = std::atan(OB_distance / (h + l));
80  }
81  else if (std::abs(h) < 1e-6) // Case 1 : S1 (h = 0)
82  {
83  theta_max = M_PI_2; // pi/2
84  theta_cri = M_PI_2; // pi/2
85  theta_min = std::atan(OB_distance / l);
86  }
87  else // Cases where h < 0 happens are not implemented yet since they are
88  // not usual
89  {
90  AssertThrow(
91  h >= 0,
92  ExcMessage(
93  "For now, it is not possible to calculate count for particle positions on the side of the detector."));
94  }
95 
96  // Calculate theta angle and its weighting factor for the Monte Carlo
97  // iteration
98  theta = std::acos(std::cos(theta_min) -
99  n_theta * (std::cos(theta_min) - std::cos(theta_max)));
100  weighting_factor_theta = (std::cos(theta_min) - std::cos(theta_max)) / 2;
101  }
102  // Case 3 : S3 (Tracer views the detector only from the top surface)
103  else // (rho < detector_radius)
104  {
105  theta_max = std::atan((r + rho) / h);
106  theta_cri = std::atan((r - rho) / h);
107  theta_min = 0;
108 
109  // Calculate theta angle and its weigthting factor for the Monte Carlo
110  // iteration
111  theta = std::acos(std::cos(theta_min) -
112  n_theta * (std::cos(theta_min) - std::cos(theta_max)));
113  weighting_factor_theta = (std::cos(theta_min) - std::cos(theta_max)) / 2;
114 
115  if (theta <= theta_cri)
116  {
117  alpha_max = M_PI;
118  weighting_factor_alpha = 1;
119  }
120  else // theta > theta_cri
121  {
122  alpha_max =
123  std::acos((std::pow(rho, 2) + std::pow(h * std::tan(theta), 2) -
124  std::pow(r, 2)) /
125  (2 * h * rho * std::tan(theta)));
126  weighting_factor_alpha = alpha_max / M_PI;
127  }
128  alpha = alpha_max * (2 * n_alpha - 1);
129 
130  // Calculate relevant distances prior next and other calculations
131  OB_distance =
132  rho * std::cos(alpha) -
133  std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
134  OA_distance =
135  rho * std::cos(alpha) +
136  std::sqrt(std::pow(r, 2) - std::pow(rho * std::sin(alpha), 2));
137  }

142 ParticleDetectorInteractions<dim>::calculate_detector_path_length()
143 {
144  double detector_path_length;
145  double l = detector_length;
146  double theta_1 = std::atan(OA_distance / (h + l));
147 
148  // Case 1 : S1 (Tracer views the detector from both top and lateral side of
149  // the detector)
150  if (rho > detector_radius)
151  {
152  if (theta <= theta_cri && theta <= theta_1)
153  detector_path_length =
154  (h + l) / std::cos(theta) - OB_distance / std::sin(theta);
155  else if (theta <= theta_cri && theta > theta_1)
156  detector_path_length = (OA_distance - OB_distance) / std::sin(theta);
157  else if (theta > theta_cri && theta <= theta_1)
158  detector_path_length = l / std::cos(theta);
159  else // (theta > theta_cri && theta > theta_1)
160  detector_path_length =
161  OA_distance / std::sin(theta) - h / std::cos(theta);
162  }
163  // Case 3 : S3 (Tracer views the detector only from the top surface)
164  else
165  {
166  double theta_2 = std::atan(OA_distance / h);
167 
168  if (theta <= theta_1)
169  detector_path_length = l / std::cos(theta);
170  else if (theta > theta_1 && theta <= theta_2)
171  detector_path_length =
172  OA_distance / std::sin(theta) - h / std::cos(theta);
173  else // Do not enters in the detector
174  detector_path_length = 0;
175  }
176 
177  return detector_path_length;

181 std::vector<double>
182 ParticleDetectorInteractions<dim>::solve_t(
183  Tensor<2, dim> e_inverse,
184  Tensor<1, dim> detector_particle_origin,
185  Tensor<1, dim> particle_position_rotation)
186 {
187  // Function value when evaluate with t (circle equation)
188  auto F = [=](double t) {
189  double R = parameters.reactor_radius;
190 
191  Tensor<1, dim> line_equations(
192  {particle_position_rotation[0] + t * std::sin(theta) * std::cos(alpha),
193  particle_position_rotation[1] + t * std::sin(theta) * std::sin(alpha),
194  particle_position_rotation[2] + t * std::cos(M_PI - theta)});
195 
196  return std::pow((e_inverse * (line_equations) +
197  detector_particle_origin)[0],
198  2) +
199  std::pow((e_inverse * (line_equations) +
200  detector_particle_origin)[1],
201  2) -
202  std::pow(R, 2);
203  };
204 
205  // Numerical derivative of the function value when evaluate with t
206  auto dF = [=](double t) {
207  double dt = 0.0001;
208 
209  return (F(t + dt) - F(t - dt)) / (2 * dt);
210  };
211 
212  // Values for solving (positive/negative)
213  std::vector<double> t0 = {-1, 1};
214  double tolerance, dt0, max_iteration;
215  tolerance = 1e-6;
216  max_iteration = 1000;
217  dt0 = 1;
218 
219  // Newton method
220  unsigned int i;
221  double t, dtn;
222  std::vector<double> t_solutions(t0.size(), 0);
223 
224  for (unsigned int n = 0; n < t0.size(); n++)
225  {
226  i = 0;
227  dtn = dt0;
228  t = t0[n];
229 
230  while (std::abs(dtn) > tolerance && i < max_iteration)
231  {
232  dtn = -F(t) / dF(t);
233  t += dtn;
234  i++;
235 
236  if (i == max_iteration)
237  std::cout << "Solving t didn't work" << std::endl;
238  }
239 
240  t_solutions[n] = t;
241  }
242 
243  return t_solutions;
244 }

248 ParticleDetectorInteractions<dim>::calculate_reactor_path_length()
249 {
250  Tensor<1, dim> h_vector, detector_particle_origin,
251  particle_position_translation, particle_position_rotation;
252  Point<dim> intersection_position;
253 
254  // Calculate translated particle position (eq 59 - 63)
255  h_vector = h * detector_orientation_z;
256  detector_particle_origin = particle_position - h_vector;
257  particle_position_translation = particle_position - detector_particle_origin;
258 
259  // Calculate rotated particle position (eq 64)
260  Tensor<1, dim, Tensor<1, dim>> detector_orientation_vector(
261  {detector_orientation_x, detector_orientation_y, detector_orientation_z});
262  Tensor<2, dim> detector_orientation_matrix(detector_orientation_vector);
263  particle_position_rotation =
264  detector_orientation_matrix * particle_position_translation;
265 
266  // Find t value of the parametric equation for
267  std::vector<double> t = solve_t(invert(detector_orientation_matrix),
268  detector_particle_origin,
269  particle_position_rotation);
270 
271  // Evaluate and determinate the intersection point closer to the detector
272  std::vector<Point<dim>> intersection_point(t.size());
273  std::vector<Tensor<1, dim>> intersection_detector_distance_vector(t.size());
274  std::vector<double> intersection_detector_distance(t.size());
275 
276  for (unsigned int i = 0; i < t.size(); i++)
277  {
278  Tensor<1, dim> line_equations(
279  {particle_position_rotation[0] +
280  t[i] * std::sin(theta) * std::cos(alpha),
281  particle_position_rotation[1] +
282  t[i] * std::sin(theta) * std::sin(alpha),
283  particle_position_rotation[2] + t[i] * std::cos(M_PI - theta)});
284 
285  intersection_point[i] =
286  detector_particle_origin +
287  invert(detector_orientation_matrix) * line_equations;
288 
289  intersection_detector_distance_vector[i] =
290  detector_face_position - intersection_point[i];
291  intersection_detector_distance[i] =
292  std::abs(intersection_detector_distance_vector[i].norm());
293  }
294 
295  double reactor_path_length =
296  (intersection_detector_distance[0] < intersection_detector_distance[1]) ?
297  std::abs((particle_position - intersection_point[0]).norm()) :
298  std::abs((particle_position - intersection_point[1]).norm());
299 
300  return reactor_path_length;

329 {
330  // Initialize efficiency
331  efficiency = 0;
332 
333  // Calculate position parameters for the particle position
334  calculate_position_parameters();
335 
336  const double denum = 1. / RAND_MAX;
337 
338  for (unsigned int i = 0; i < parameters.n_monte_carlo_iteration; i++)
339  {
340  // Generate random values for Monte Carlo
341  double n_alpha = (double)rand() * denum;
342  double n_theta = (double)rand() * denum;
343 
344  // Initialize variables
345  double detector_path_length, reactor_path_length,
346  detector_interaction_probability, non_interaction_probability;
347 
348  // Calculate all parameters and distances for efficiency
349  calculate_solid_angle(n_alpha, n_theta);
350  detector_path_length = calculate_detector_path_length();
351  detector_interaction_probability =
352  calculate_detector_interaction_probability(detector_path_length);
353  reactor_path_length = calculate_reactor_path_length();
354  non_interaction_probability =
355  calculate_non_interaction_probability(reactor_path_length);
356 
357  efficiency += weighting_factor_alpha * weighting_factor_theta *
358  detector_interaction_probability *
359  non_interaction_probability;
360  }
361 
362  efficiency /= parameters.n_monte_carlo_iteration;
