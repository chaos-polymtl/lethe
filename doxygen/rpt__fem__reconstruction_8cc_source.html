39template <int dim>
40void
41RPTL2Projection<dim>::setup_triangulation()
42{
43 if (fem_reconstruction_parameters.mesh_type ==
44 Parameters::RPTFEMReconstructionParameters::FEMMeshType::gmsh)
45 {
46 // Warn user
47 pcout << "-----------------------------------------------" << std::endl;
48 pcout << "At the moment, the application calculates" << std::endl;
49 pcout << "the gamma-ray attenuation in cylindrical" << std::endl;
50 pcout << "vessels only. Using another type of geometry" << std::endl;
51 pcout << "would yield in committing an error when" << std::endl;
52 pcout << "calculating photon counts. Furthermore, the" << std::endl;
53 pcout << "axis of symmetry of the cylinder should" << std::endl;
54 pcout << "be oriented in z-direction." << std::endl;
55 pcout << "-----------------------------------------------" << std::endl;
56
57 // Import and attach mesh to triangulation
58 GridIn<dim> grid_in;
59 grid_in.attach_triangulation(triangulation);
60 std::ifstream input_file(fem_reconstruction_parameters.mesh_file);
61 grid_in.read_msh(input_file);
62
63 // Attach a cylindrical manifold object to the triangulation
64 const CylindricalManifold<dim> manifold(2); // Cylinder along the z-axis
65 triangulation.set_all_manifold_ids(0);
66 triangulation.set_manifold(0, manifold);
67
68 // Check if all cells are tetrahedral elements
69 for (const auto &cell : triangulation.active_cell_iterators())
70 {
71 AssertThrow(
72 cell->n_vertices() == 4,
73 ExcMessage(
74 "The imported mesh has to have only tetrahedral elements"));
75 }
76 }
77 else
78 {
79 Triangulation<dim> temp_triangulation;
80 Triangulation<dim> flat_temp_triangulation;
81
82 GridGenerator::subdivided_cylinder(
83 temp_triangulation,
84 fem_reconstruction_parameters.z_subdivisions,
85 parameters.reactor_radius,
86 parameters.reactor_height * 0.5);
87 temp_triangulation.refine_global(
88 fem_reconstruction_parameters.mesh_refinement);
89
90 // Flatten the triangulation
91 GridGenerator::flatten_triangulation(temp_triangulation,
92 flat_temp_triangulation);
93 // Convert to simplex elements
94 GridGenerator::convert_hypercube_to_simplex_mesh(flat_temp_triangulation,
95 triangulation);
96 triangulation.set_all_manifold_ids(0);
97
98 // Grid transformation
99 Tensor<1, dim, double> axis({0, 1, 0});
100 GridTools::rotate(axis, M_PI_2, triangulation);
101 Tensor<1, dim> shift_vector({0, 0, parameters.reactor_height * 0.5});
102 GridTools::shift(shift_vector, triangulation);
103 }
104}

179template <int dim>
180void
181RPTL2Projection<dim>::setup_system()
182{
183 TimerOutput::Scope t(computing_timer, "setup_system");
184
185 // Partition triangulation between the processes
186 GridTools::partition_triangulation(
187 Utilities::MPI::n_mpi_processes(mpi_communicator), triangulation);
188
189 dof_handler.distribute_dofs(fe);
190 DoFRenumbering::subdomain_wise(dof_handler);
191
192 constraints.clear();
193 DoFTools::make_hanging_node_constraints(dof_handler, constraints);
194 constraints.close();
195
196 DynamicSparsityPattern dsp(dof_handler.n_dofs(), dof_handler.n_dofs());
197 DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints, false);
198
199 const std::vector<IndexSet> locally_owned_dofs_per_proc =
200 DoFTools::locally_owned_dofs_per_subdomain(dof_handler);
201
202 // Get the DoFs owned by each process
203 const IndexSet &locally_owned_dofs =
204 locally_owned_dofs_per_proc[Utilities::MPI::this_mpi_process(
205 mpi_communicator)];
206
207 nodal_counts.resize(n_detector);
208 for (Vector<double> &nodal_counts_for_one_detector : nodal_counts)
209 nodal_counts_for_one_detector.reinit(locally_owned_dofs.size());
210
211 system_matrix.reinit(locally_owned_dofs,
212 locally_owned_dofs,
213 dsp,
214 mpi_communicator);
215 solution.reinit(locally_owned_dofs, mpi_communicator);
216 system_rhs.reinit(locally_owned_dofs, mpi_communicator);
217}

219template <int dim>
220void
221RPTL2Projection<dim>::solve_linear_system(unsigned detector_no)
222{
223 TimerOutput::Scope t(computing_timer, "solve_linear_system");
224
225 this->pcout << "Norm of RHS is : " << system_rhs.l2_norm() << std::endl;
226
227 SolverControl solver_control(10000000,
228 std::max(1e-6, system_rhs.l2_norm() * 1e-6));
229 LA::SolverGMRES solver(solver_control);
230
231 LA::MPI::PreconditionILU preconditioner;
232 LA::MPI::PreconditionILU::AdditionalData data(0, 1e-10, 1, 0);
233 preconditioner.initialize(system_matrix, data);
234
235 solver.solve(system_matrix, solution, system_rhs, preconditioner);
236
237 Vector<double> localized_solution(
238 solution); // constructor Trilinos MPI Vector -> deal.II Vector
239
240 constraints.distribute(localized_solution);
241 nodal_counts[detector_no] = localized_solution;
242}

244template <int dim>
245void
246RPTL2Projection<dim>::assemble_system(unsigned no_detector)
247{
248 TimerOutput::Scope t(computing_timer, "assemble_system");
249
250 system_rhs = 0;
251 system_matrix = 0;
252
253 const QGaussSimplex<dim> quadrature_formula(fe.degree + 1);
254
255 FEValues<dim> fe_values(mapping,
256 fe,
257 quadrature_formula,
258 update_values | update_quadrature_points |
259 update_JxW_values);
260
261 const unsigned int dofs_per_cell = fe.n_dofs_per_cell();
262
263 FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
264 Vector<double> cell_rhs(dofs_per_cell);
265
266 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
267
268 for (const auto &cell : dof_handler.active_cell_iterators())
269 {
270 if (cell->subdomain_id() ==
271 Utilities::MPI::this_mpi_process(mpi_communicator))
272 {
273 cell_matrix = 0;
274 cell_rhs = 0;
275 fe_values.reinit(cell);
276
277 for (const unsigned int q_index :
278 fe_values.quadrature_point_indices())
279 {
280 Point<dim> q_point_position = fe_values.quadrature_point(q_index);
281 RadioParticle<dim> particle(q_point_position, 0);
282
283 ParticleDetectorInteractions<dim> p_q_interaction(
284 particle, detectors[no_detector], parameters);
285
286 double count = p_q_interaction.calculate_count();
287
288 if (count < 0 || count != count)
289 this->pcout
290 << "Invalid value encountered in count calculation, this should be investigated furthermore : "
291 << count <<
292
293 std::endl;
294 else
295 {
296 for (const unsigned int i : fe_values.dof_indices())
297 {
298 for (const unsigned int j : fe_values.dof_indices())
299 {
300 cell_matrix(i, j) +=
301 (fe_values.shape_value(i, q_index) * // phi_i(x_q)
302 fe_values.shape_value(j, q_index) * // phi_j(x_q)
303 fe_values.JxW(q_index)); // dx
304 }
305
306
307 cell_rhs(i) +=
308 (count * // f(x)
309 fe_values.shape_value(i, q_index) * // phi_i(x_q)
310 fe_values.JxW(q_index)); // dx
311 }
312 }
313 }
314 cell->get_dof_indices(local_dof_indices);
315 constraints.distribute_local_to_global(cell_matrix,
316 cell_rhs,
317 local_dof_indices,
318 system_matrix,
319 system_rhs);
320 }
321 }
322 system_matrix.compress(VectorOperation::add);
323 system_rhs.compress(VectorOperation::add);
324}

497template <int dim>
498Vector<double>
499assemble_matrix_and_rhs(
500 std::vector<std::vector<double>> &vertex_count,
501 std::vector<double> &experimental_count,
502 Parameters::RPTFEMReconstructionParameters::FEMCostFunction
503 &cost_function_type)
504{
505 Vector<double> reference_location;
506 unsigned int detector_size = vertex_count.size();
507 double sigma = 0;
508 LAPACKFullMatrix<double> sys_matrix;
509 Vector<double> sys_rhs;
510
511 sys_matrix.reinit(dim);
512 sys_rhs.reinit(dim);
513
514 if (cost_function_type ==
515 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute)
516 {
517 // Assembling sys_matrix (Jacobian)
518 for (unsigned int i = 0; i < dim; ++i)
519 {
520 for (unsigned int j = 0; j < dim; ++j)
521 {
522 for (unsigned int d = 0; d < detector_size; ++d)
523 {
524 sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
525 (-vertex_count[d][0] + vertex_count[d][i + 1]);
526 }
527 sys_matrix.set(i, j, sigma);
528 sigma = 0;
529 }
530 }
531
532 // Assembling sys_rhs
533 for (unsigned int i = 0; i < dim; ++i)
534 {
535 for (unsigned int d = 0; d < detector_size; ++d)
536 {
537 sigma += (vertex_count[d][0] - experimental_count[d]) *
538 (-vertex_count[d][0] + vertex_count[d][i + 1]);
539 }
540 sys_rhs[i] = -sigma;
541 sigma = 0;
542 }
543 }
544 else if (cost_function_type == Parameters::RPTFEMReconstructionParameters::
545 FEMCostFunction::relative)
546 {
547 std::vector<double> denom(detector_size);
548
549 for (unsigned int d = 0; d < experimental_count.size(); ++d)
550 denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
551
552 // Assembling sys_matrix
553 for (unsigned int i = 0; i < dim; ++i)
554 {
555 for (unsigned int j = 0; j < dim; ++j)
556 {
557 for (unsigned int d = 0; d < detector_size; ++d)
558 {
559 sigma += (-vertex_count[d][0] + vertex_count[d][j + 1]) *
560 (-vertex_count[d][0] + vertex_count[d][i + 1]) *
561 denom[d];
562 }
563 sys_matrix.set(i, j, sigma);
564 sigma = 0;
565 }
566 }
567
568 // Assembling sys_rhs
569 for (unsigned int i = 0; i < dim; ++i)
570 {
571 for (unsigned int d = 0; d < detector_size; ++d)
572 {
573 sigma += (vertex_count[d][0] - experimental_count[d]) *
574 (-vertex_count[d][0] + vertex_count[d][i + 1]) *
575 denom[d];
576 }
577 sys_rhs[i] = -sigma;
578 sigma = 0;
579 }
580 }
581
582 // Setup and solve linear system
583 sys_matrix.set_property(LAPACKSupport::general);
584 sys_matrix.compute_lu_factorization();
585 sys_matrix.solve(sys_rhs);
586
587 reference_location = sys_rhs;
588 return reference_location;
589}

592template <int dim>
593double
594RPTFEMReconstruction<dim>::calculate_reference_location_error(
595 Vector<double> &reference_location,
596 const double &last_constraint)
597{
598 std::vector<double> err_coordinates(dim + 1, 0);
599 double norm_error_coordinates = 0;
600
601 // Check if the location is a valid one
602 for (unsigned int i = 0; i < dim; ++i)
603 {
604 // e_1 = \xi - 1; e_2 = \eta - 1; e_3 = \zeta - 1;
605 if (reference_location[i] > 1)
606 err_coordinates[i] = (reference_location[i] - 1);
607
608 // e_1 = - \xi; e_2 = - \eta; e_3 = - \zeta;
609 if (reference_location[i] < 0)
610 err_coordinates[i] = std::abs(reference_location[i]);
611 }
612
613 // Check the last constraint
614 if (last_constraint < 0)
615 {
616 // e_4 = |(1 - \xi - \eta - \zeta)|
617 err_coordinates[3] = std::abs(last_constraint);
618 }
619 else if (last_constraint > 1)
620 {
621 // e_4 = (1 - \xi - \eta - \zeta) - 1
622 err_coordinates[3] = last_constraint - 1;
623 }
624
625 // Calculate the norm of the error coordinate vector
626 for (const double &error : err_coordinates)
627 {
628 norm_error_coordinates += error * error;
629 }
630
631 return norm_error_coordinates;
632}

635template <int dim>
636double
637RPTFEMReconstruction<dim>::calculate_cost(
638 const TriaActiveIterator<DoFCellAccessor<dim, dim, false>> &cell,
639 Vector<double> &reference_location,
640 const double &last_constraint,
641 std::vector<double> &experimental_count)
642{
643 double cost = 0;
644 double count = 0;
645
646 if (fem_reconstruction_parameters.fem_cost_function ==
647 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::absolute)
648 {
649 // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
650 // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2
651 for (unsigned int d = 0; d < n_detector; ++d)
652 {
653 count +=
654 (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
655
656 for (unsigned int i = 0; i < dim; ++i)
657 {
658 count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
659 reference_location[i]);
660 }
661 count -= experimental_count[d];
662 cost += count * count;
663 count = 0;
664 }
665 }
666 else if (fem_reconstruction_parameters.fem_cost_function ==
667 Parameters::RPTFEMReconstructionParameters::FEMCostFunction::
668 relative)
669 {
670 std::vector<double> denom(n_detector);
671
672 // cost = \sum_{d=1}^{n_detector} [C_{0,d} * (1 - \xi - \eta - \zeta) +
673 // C_{1,d} * \xi + C_{2,d} * \eta + C_{3,d} * \zeta - C_{exp,d}]^2 / (
674 // C_{exp,d}^2)
675 for (unsigned int d = 0; d < n_detector; ++d)
676 {
677 count +=
678 (nodal_counts[d][cell->vertex_dof_index(0, 0)] * last_constraint);
679
680 denom[d] = 1 / (experimental_count[d] * experimental_count[d]);
681
682 for (unsigned int i = 0; i < dim; ++i)
683 {
684 count += (nodal_counts[d][cell->vertex_dof_index(i + 1, 0)] *
685 reference_location[i]);
686 }
687
688 count -= experimental_count[d];
689 cost += count * count * denom[d];
690 count = 0;
691 }
692 }
693
694 return cost;

697template <int dim>
698void
699RPTFEMReconstruction<dim>::search_position_in_reference_space(
700 std::vector<std::vector<double>> &count_from_all_detectors,
701 std::vector<double> &experimental_count,
702 const typename DoFHandler<dim>::active_cell_iterator &cell,
703 const double &tol_reference_location,
704 double &max_cost,
705 Vector<double> &reference_location,
706 bool &position_found,
707 Point<dim> &real_location)
708{
709 double last_constraint_reference_location;
710 double norm_error_coordinates;
711 double calculated_cost;
712
713
714 // Solve linear system to find the location in reference coordinates
715 reference_location = assemble_matrix_and_rhs<dim>(
716 count_from_all_detectors,
717 experimental_count,
718 fem_reconstruction_parameters.fem_cost_function);
719
720 // 4th constraint on the location of the particle in reference coordinates
721 last_constraint_reference_location =
722 1 - reference_location[0] - reference_location[1] - reference_location[2];
723
724 // Evaluate the error of the reference position (Is it outside the
725 // reference tetrahedron ?)
726 norm_error_coordinates =
727 calculate_reference_location_error(reference_location,
728 last_constraint_reference_location);
729
730 // Extrapolation limit
731 if (norm_error_coordinates < tol_reference_location)
732 {
733 // Calculate cost with the selected cost function
734 calculated_cost = calculate_cost(cell,
735 reference_location,
736 last_constraint_reference_location,
737 experimental_count);
738 // Check if the new cost is lower than the previously stored
739 // real_location's
740 if (calculated_cost < max_cost)
741 {
742 max_cost = calculated_cost;
743 position_found = true;
744 previous_position_cell = cell;
745
746 // Evaluate the real location of the particle
747 for (unsigned int v = 0; v < cell->n_vertices(); ++v)
748 {
749 if (v == 0)
750 real_location = cell->vertex(v);
751 else
752 real_location += reference_location[v - 1] *
753 (cell->vertex(v) - cell->vertex(0));
754 }
755 }
756 }
757}

761RPTFEMReconstruction<dim>::find_position_global_search(
762 std::vector<double> &experimental_counts,
763 const double tol_reference_location)

765 double max_cost = DBL_MAX;
766 bool position_found = false;
767 Point<dim> real_location;
768 Vector<double> reference_location;
769 std::vector<std::vector<double>> count_from_all_detectors(

789 reference_location,
790 position_found,
791 real_location);
792 }

794 return position_found;

797template <int dim>
798bool
799RPTFEMReconstruction<dim>::find_position_local_search(
800 std::vector<double> &experimental_counts,
801 const double tol_reference_location)
802{
803 double max_cost = DBL_MAX;
804 bool position_found = false;
805 Point<dim> real_location;
806 Vector<double> reference_location;
807 std::vector<std::vector<double>> count_from_all_detectors(
808 n_detector, std::vector<double>(4));
809 const typename DoFHandler<dim>::active_cell_iterator cell =
810 previous_position_cell;
811
812 // Find cells adjacent to the cell were found the previous position
813 std::vector<typename DoFHandler<dim>::active_cell_iterator> adjacent_cells;
814 std::set<typename DoFHandler<dim>::active_cell_iterator> all_adjacent_cells;
815
816 // Add previous solution's cell
817 all_adjacent_cells.insert(cell);
818
819 // Search for adjacent cells and stock them in the set container
820 for (unsigned int v = 0; v < cell->n_vertices(); ++v)
821 {
822 auto v_index = cell->vertex_index(v);
823 adjacent_cells =
824 GridTools::find_cells_adjacent_to_vertex(dof_handler, v_index);
825
826 for (const auto &adjacent_cell : adjacent_cells)
827 all_adjacent_cells.insert(adjacent_cell);
828 }
829
830 // Add adjacent cells according to the level of proximity
831 if (fem_reconstruction_parameters.search_proximity_level > 1)
832 {
833 std::set<typename DoFHandler<dim>::active_cell_iterator>
834 previous_adjacent_cells;
835 std::set<typename DoFHandler<dim>::active_cell_iterator>
836 previous_main_cells;
837
838 previous_main_cells.insert(cell);
839
840 for (unsigned int proximity_level = 2;
841 proximity_level <
842 fem_reconstruction_parameters.search_proximity_level + 1;
843 ++proximity_level)
844 {
845 previous_adjacent_cells = all_adjacent_cells;
846
847 for (const auto &current_main_cell : previous_adjacent_cells)
848 {
849 if (*previous_main_cells.find(current_main_cell) !=
850 current_main_cell)
851 {
852 for (unsigned int v = 0; v < current_main_cell->n_vertices();
853 ++v)
854 {
855 auto v_index = current_main_cell->vertex_index(v);
856 adjacent_cells =
857 GridTools::find_cells_adjacent_to_vertex(dof_handler,
858 v_index);
859
860 for (const auto &adjacent_cell : adjacent_cells)
861 all_adjacent_cells.insert(adjacent_cell);
862 }
863 }
864 previous_main_cells = previous_adjacent_cells;
865 }
866 }
867 }
868
869 // Loop over adjacent cells, loop over detectors and get nodal count values
870 // for each vertex of the cell
871 for (const auto &adjacent_cell : all_adjacent_cells)
872 {
873 for (unsigned int d = 0; d < n_detector; ++d)
874 {
875 for (unsigned int v = 0; v < adjacent_cell->n_vertices(); ++v)
876 {
877 auto dof_index = adjacent_cell->vertex_dof_index(v, 0);
878 count_from_all_detectors[d][v] = nodal_counts[d][dof_index];
879 }
880 }
881
882 search_position_in_reference_space(count_from_all_detectors,
883 experimental_counts,
884 adjacent_cell,
885 tol_reference_location,
886 max_cost,
887 reference_location,
888 position_found,
889 real_location);
890 }
891
892 if (position_found)
893 {
894 found_positions.push_back(real_location);
895 return position_found;
896 }
897 else
898 {
899 position_found = find_position_global_search(experimental_counts,
900 tol_reference_location);
901 return position_found;
902 }
903}

905template <int dim>
906void
907RPTFEMReconstruction<dim>::trajectory()
908{
909 // Tolerance or the extrapolation limit in the reference space for a found
910 // position
911 double tol_reference_location =
912 fem_reconstruction_parameters.extrapolation_tolerance;
913
914 if (fem_reconstruction_parameters.mesh_type ==
915 Parameters::RPTFEMReconstructionParameters::FEMMeshType::dealii)
916 {
917 if (fem_reconstruction_parameters.extrapolation_tolerance == 0.005)
918 {
919 const unsigned int power = static_cast<unsigned int>(
920 pow(2, fem_reconstruction_parameters.mesh_refinement));
921 const unsigned int n_cell_z =
922 2 * fem_reconstruction_parameters.z_subdivisions * power;
923 tol_reference_location = parameters.reactor_height / n_cell_z * 1.15;
924 }
925 }
926
927 // Read and store all experimental counts
928 std::vector<std::vector<double>> all_experimental_counts;
929 {
930 TimerOutput::Scope t(computing_timer, "input_experimental_counts_file");
931 all_experimental_counts = read_detectors_counts<dim>(
932 fem_reconstruction_parameters.experimental_counts_file, n_detector);
933 }
934
935 {
936 TimerOutput::Scope t(computing_timer, "find_particle_positions");
937
938 if (fem_reconstruction_parameters.search_type ==
939 Parameters::RPTFEMReconstructionParameters::FEMSearchType::local)
940 {
941 // It is set to "false" to force the global search on the first search
942 bool adjacent_cell_search = false;
943
944 // Find the position of the particle with the experimental counts by
945 // prioritizing local search
946 for (std::vector<double> &experimental_counts : all_experimental_counts)
947 {
948 if (adjacent_cell_search)
949 adjacent_cell_search =
950 find_position_local_search(experimental_counts,
951 tol_reference_location);
952 else
953 adjacent_cell_search =
954 find_position_global_search(experimental_counts,
955 tol_reference_location);
956 }
957 }
958 else
959 {
960 // Find the position of the particle with the experimental counts by
961 // global search only
962 for (std::vector<double> &experimental_counts : all_experimental_counts)
963 {
964 find_position_global_search(experimental_counts,
965 tol_reference_location);
966 }
967 }
968 }
969}
