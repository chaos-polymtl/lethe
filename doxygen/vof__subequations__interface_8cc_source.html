10template <int dim>
11void
12VOFSubequationsInterface<dim>::initialize_subequations(
13 const SimulationParameters<dim> &p_simulation_parameters,
14 std::shared_ptr<parallel::DistributedTriangulationBase<dim>> p_triangulation,
15 const std::shared_ptr<SimulationControl> &p_simulation_control)
16{
17 // Reset all subequations
18 this->active_subequations.clear();
19 this->subequations.clear();
20
21 // Activate and add relevant subequations
22 if ((p_simulation_parameters.multiphysics.vof_parameters.surface_tension_force
23 .enable) ||
24 (p_simulation_parameters.multiphysics.vof_parameters.regularization_method
25 .algebraic_interface_reinitialization.enable))
26 {
27 // Phase gradient projection
28 this->active_subequations.push_back(
29 VOFSubequationsID::phase_gradient_projection);
30 this->subequations[VOFSubequationsID::phase_gradient_projection] =
31 std::make_shared<VOFPhaseGradientProjection<dim>>(
32 p_simulation_parameters, this->pcout, p_triangulation, *this);
33 // Phase curvature projection
34 this->active_subequations.push_back(
35 VOFSubequationsID::curvature_projection);
36 this->subequations[VOFSubequationsID::curvature_projection] =
37 std::make_shared<VOFCurvatureProjection<dim>>(p_simulation_parameters,
38 this->pcout,
39 p_triangulation,
40 *this);
41
42 if (p_simulation_parameters.multiphysics.vof_parameters
43 .regularization_method.algebraic_interface_reinitialization.enable)
44 {
45 // Algebraic interface reinitialization
46 this->active_subequations.push_back(
47 VOFSubequationsID::algebraic_interface_reinitialization);
48 this->subequations
49 [VOFSubequationsID::algebraic_interface_reinitialization] =
50 std::make_shared<VOFAlgebraicInterfaceReinitialization<dim>>(
51 p_simulation_parameters,
52 this->pcout,
53 p_triangulation,
54 p_simulation_control,
55 *this);
56 }
57 }
58 reset_subequations_solutions_validity();
59}
