85 /**
86 * @brief This class is the base class for all boundary conditions. It stores
87 * the general information that all boundary condition share.
88 * In Lethe, boundary conditions are identified with an id, a type and, in the
89 * special case of periodic boundary condition, a periodic matching id
90 * (periodic_id) and a periodic direction (0, 1 or 2).
91 */
92 class BoundaryConditions
93 {
94 public:
95 /// Map containing the boundary id and the boundary type
96 std::map<types::boundary_id, BoundaryType> type;
97
98 /// Map containing the penalization parameter for weak dirichlet BCs and
99 /// outlets for a corresponding id
100 std::map<types::boundary_id, double> beta;
101
102 /// Map containing the boundary layer thickness tangent component parameter
103 /// for partial slip for a corresponding id Map containing Dirichlet BCs for
104 /// a corresponding boundary id
105 std::map<types::boundary_id, double> boundary_layer_thickness;
106
107 /// Number of boundary conditions
108 unsigned int number_of_boundary_conditions;
109
110 /// indicator for transient BCs
111 bool time_dependent;
112
113 /// Map containing the boundary id and its corresponding periodic boundary
114 /// condition match
115 std::map<types::boundary_id, types::boundary_id> periodic_neighbor_id;
116
117 /// Map containing the boundary id and its periodic direction
118 std::map<types::boundary_id, unsigned int> periodic_direction;
119 };

301 /**
302 * @brief Parse the information for a boundary condition
303 *
304 * @param prm A parameter handler which is currently used to parse the simulation information
305 */
306 template <int dim>
307 void
308 NSBoundaryConditions<dim>::parse_boundary(ParameterHandler &prm)
309 {
310 // Parse the list of boundary ids
311 std::vector<types::boundary_id> boundary_ids =
312 convert_string_to_vector<types::boundary_id>(prm, "id");
313
314 // Check if the list contains at least one boundary id
315 AssertThrow(
316 boundary_ids.size() > 0,
317 ExcMessage(
318 "A boundary id has not been set for one of the fluid dynamics boundary conditions. Please ensure that the id is set for every boundary condition."));
319
320 // and unique
321 for (const auto boundary_id : boundary_ids)
322 {
323 AssertThrow(this->type.find(boundary_id) == this->type.end(),
324 NavierStokesBoundaryDuplicated(boundary_id));
325
326 // Allocate the navier_stokes_functions object for every boundary
327 // condition to ensure that they have a defined function and a center of
328 // rotation.
329 navier_stokes_functions[boundary_id] =
330 std::make_shared<NSBoundaryFunctions<dim>>();
331
332 prm.enter_subsection("u");
333 navier_stokes_functions[boundary_id]->u.parse_parameters(prm);
334 prm.leave_subsection();
335
336 prm.enter_subsection("v");
337 navier_stokes_functions[boundary_id]->v.parse_parameters(prm);
338 prm.leave_subsection();
339
340 prm.enter_subsection("w");
341 navier_stokes_functions[boundary_id]->w.parse_parameters(prm);
342 prm.leave_subsection();
343
344 prm.enter_subsection("p");
345 navier_stokes_functions[boundary_id]->p.parse_parameters(prm);
346 prm.leave_subsection();
347
348 prm.enter_subsection("center of rotation");
349 navier_stokes_functions[boundary_id]->center_of_rotation[0] =
350 prm.get_double("x");
351 navier_stokes_functions[boundary_id]->center_of_rotation[1] =
352 prm.get_double("y");
353
354 if (dim == 3)
355 navier_stokes_functions[boundary_id]->center_of_rotation[2] =
356 prm.get_double("z");
357 prm.leave_subsection();
358
359 // Establish the type of boundary condition
360 const std::string op = prm.get("type");
361 if (op == "none")
362 this->type[boundary_id] = BoundaryType::none;
363 if (op == "noslip")
364 this->type[boundary_id] = BoundaryType::noslip;
365 if (op == "slip")
366 this->type[boundary_id] = BoundaryType::slip;
367 if (op == "function" || op == "function weak" || op == "partial slip")
368 {
369 if (op == "function")
370 this->type[boundary_id] = BoundaryType::function;
371 else if (op == "partial slip")
372 this->type[boundary_id] = BoundaryType::partial_slip;
373 else
374 this->type[boundary_id] = BoundaryType::function_weak;
375 }
376 if (op == "pressure")
377 {
378 this->type[boundary_id] = BoundaryType::pressure;
379 }
380 if (op == "periodic")
381 {
382 types::boundary_id periodic_boundary_id =
383 prm.get_integer("periodic id");
384
385 this->type[boundary_id] = BoundaryType::periodic;
386
387 // We attribute a periodic neighbor boundary type to the neighbor
388 // boundary to ensure that all boundaries have a defined type
389 this->type[periodic_boundary_id] = BoundaryType::periodic_neighbor;
390
391 // We store the periodic id and direction
392 this->periodic_neighbor_id[boundary_id] = periodic_boundary_id;
393 this->periodic_direction[boundary_id] =
394 prm.get_integer("periodic direction");
395
396 // Allocate the navier_stokes_functions object for the periodic
397 // neighbor condition to ensure that they have a defined function
398 // and a center of rotation. Parse the information of the boundary
399 // id to duplicate it.
400 navier_stokes_functions[periodic_boundary_id] =
401 std::make_shared<NSBoundaryFunctions<dim>>();
402
403 prm.enter_subsection("u");
404 navier_stokes_functions[periodic_boundary_id]->u.parse_parameters(
405 prm);
406 prm.leave_subsection();
407
408 prm.enter_subsection("v");
409 navier_stokes_functions[periodic_boundary_id]->v.parse_parameters(
410 prm);
411 prm.leave_subsection();
412
413 prm.enter_subsection("w");
414 navier_stokes_functions[periodic_boundary_id]->w.parse_parameters(
415 prm);
416 prm.leave_subsection();
417
418 prm.enter_subsection("p");
419 navier_stokes_functions[periodic_boundary_id]->p.parse_parameters(
420 prm);
421 prm.leave_subsection();
422
423 prm.enter_subsection("center of rotation");
424 navier_stokes_functions[periodic_boundary_id]
425 ->center_of_rotation[0] = prm.get_double("x");
426 navier_stokes_functions[periodic_boundary_id]
427 ->center_of_rotation[1] = prm.get_double("y");
428
429 if (dim == 3)
430 navier_stokes_functions[periodic_boundary_id]
431 ->center_of_rotation[2] = prm.get_double("z");
432 prm.leave_subsection();
433 }
434 if (op == "outlet")
435 {
436 this->type[boundary_id] = BoundaryType::outlet;
437 }
438
439 this->beta[boundary_id] = prm.get_double("beta");
440 this->boundary_layer_thickness[boundary_id] =
441 prm.get_double("boundary layer thickness");
442 }
443 }

680 /**
681 * @brief Parse the information for a boundary condition
682 *
683 * @param prm A parameter handler which is currently used to parse the simulation information
684 */
685
686 template <int dim>
687 void
688 HTBoundaryConditions<dim>::parse_boundary(ParameterHandler &prm)
689 {
690 // Parse the list of boundary ids
691 std::vector<types::boundary_id> boundary_ids =
692 convert_string_to_vector<types::boundary_id>(prm, "id");
693
694 // Check if the list contains at least one boundary id
695 AssertThrow(
696 boundary_ids.size() > 0,
697 ExcMessage(
698 "A boundary id has not been set for one of the heat transfer boundary conditions. Please ensure that the id is set for every boundary condition."));
699
700 // Loop through all boundary ids to ensure that they are all non-negative
701 // and unique
702 for (const auto boundary_id : boundary_ids)
703 {
704 AssertThrow(this->type.find(boundary_id) == this->type.end(),
705 HeatTransferBoundaryDuplicated(boundary_id));
706
707
708 const std::string op = prm.get("type");
709 if (op == "noflux")
710 {
711 this->type[boundary_id] = BoundaryType::noflux;
712 }
713 else if (op == "temperature")
714 {
715 this->type[boundary_id] = BoundaryType::temperature;
716 }
717 else if (op == "convection-radiation-flux")
718 {
719 this->type[boundary_id] = BoundaryType::convection_radiation;
720 this->has_convection_radiation_bc = true;
721
722 // Emissivity validity (0 <= emissivity <= 1) will be checked at
723 // evaluation.
724 }
725 else if (op == "periodic")
726 {
727 types::boundary_id periodic_boundary_id =
728 prm.get_integer("periodic id");
729
730 this->type[boundary_id] = BoundaryType::periodic;
731
732 // We attribute a periodic neighbor boundary type to the neighbor
733 // boundary to ensure that all boundaries have a defined type
734 this->type[periodic_boundary_id] = BoundaryType::periodic_neighbor;
735
736 // We store the periodic id and direction
737 this->periodic_neighbor_id[boundary_id] = periodic_boundary_id;
738 this->periodic_direction[boundary_id] =
739 prm.get_integer("periodic direction");
740 }
741 else
742 {
743 AssertThrow(
744 false,
745 ExcMessage("Unknown boundary condition type for heat transfer."));
746 }
747
748 // All the functions are parsed since they might be used for
749 // post-processing
750 prm.enter_subsection("value");
751 this->dirichlet_value[boundary_id] =
752 std::make_shared<Functions::ParsedFunction<dim>>();
753 this->dirichlet_value[boundary_id]->parse_parameters(prm);
754 prm.leave_subsection();
755 prm.enter_subsection("h");
756 this->h[boundary_id] =
757 std::make_shared<Functions::ParsedFunction<dim>>();
758 this->h[boundary_id]->parse_parameters(prm);
759 prm.leave_subsection();
760 prm.enter_subsection("Tinf");
761 this->Tinf[boundary_id] =
762 std::make_shared<Functions::ParsedFunction<dim>>();
763 this->Tinf[boundary_id]->parse_parameters(prm);
