84 /**
85 * @brief This class is the base class for all boundary conditions. It stores
86 * the general information that all boundary condition share.
87 * In Lethe, boundary conditions are identified with an id, a type and, in the
88 * special case of periodic boundary condition, a periodic matching id
89 * (periodic_id) and a periodic direction (0, 1 or 2).
90 */
91 template <int dim>
92 class BoundaryConditions
93 {
94 public:
95 /// Map containing the boundary id and the boundary type
96 std::map<types::boundary_id, BoundaryType> type;
97
98 /// Map containing the penalization parameter for weak dirichlet BCs and
99 /// outlets for a corresponding id
100 std::map<types::boundary_id, double> beta;
101
102 /// Map containing the boundary layer thickness tangent component parameter
103 /// for partial slip for a corresponding id Map containing Dirichlet BCs for
104 /// a corresponding boundary id
105 std::map<types::boundary_id, double> boundary_layer_thickness;
106
107 /// Number of boundary conditions
108 unsigned int number_of_boundary_conditions;
109
110 /// indicator for transient BCs
111 bool time_dependent;
112
113 /// Map containing the boundary id and its corresponding periodic boundary
114 /// condition match
115 std::map<types::boundary_id, types::boundary_id> periodic_neighbor_id;
116
117 /// Map containing the boundary id and its periodic direction
118 std::map<types::boundary_id, unsigned int> periodic_direction;
119 };

291 "0",
292 Patterns::Double(),

297 /**
298 * @brief Parse the information for a boundary condition
299 *
300 * @param prm A parameter handler which is currently used to parse the simulation information
301 *
302 * @param i_bc The boundary condition number (and not necessarily it's id).
303 */
304 template <int dim>
305 void
306 NSBoundaryConditions<dim>::parse_boundary(ParameterHandler &prm)
307 {
308 AssertThrow(
309 prm.get_integer("id") >= 0,
310 ExcMessage(
311 "A boundary id has not been set for one of the fluid dynamics boundary condition. Please ensure that the id is set for every boundary condition."));
312
313 types::boundary_id boundary_id = prm.get_integer("id");
314
315 AssertThrow(this->type.find(boundary_id) == this->type.end(),
316 NavierStokesBoundaryDuplicated(boundary_id));
317
318
319 // Allocate the navier_stokes_functions object for every boundary condition
320 // to ensure that they have a defined function and a center of rotation.
321 navier_stokes_functions[boundary_id] =
322 std::make_shared<NSBoundaryFunctions<dim>>();
323
324 prm.enter_subsection("u");
325 navier_stokes_functions[boundary_id]->u.parse_parameters(prm);
326 prm.leave_subsection();
327
328 prm.enter_subsection("v");
329 navier_stokes_functions[boundary_id]->v.parse_parameters(prm);
330 prm.leave_subsection();
331
332 prm.enter_subsection("w");
333 navier_stokes_functions[boundary_id]->w.parse_parameters(prm);
334 prm.leave_subsection();
335
336 prm.enter_subsection("p");
337 navier_stokes_functions[boundary_id]->p.parse_parameters(prm);
338 prm.leave_subsection();
339
340 prm.enter_subsection("center of rotation");
341 navier_stokes_functions[boundary_id]->center_of_rotation[0] =
342 prm.get_double("x");
343 navier_stokes_functions[boundary_id]->center_of_rotation[1] =
344 prm.get_double("y");
345
346 if (dim == 3)
347 navier_stokes_functions[boundary_id]->center_of_rotation[2] =
348 prm.get_double("z");
349 prm.leave_subsection();
350
351 // Establish the type of boundary condition
352 const std::string op = prm.get("type");
353 if (op == "none")
354 this->type[boundary_id] = BoundaryType::none;
355 if (op == "noslip")
356 this->type[boundary_id] = BoundaryType::noslip;
357 if (op == "slip")
358 this->type[boundary_id] = BoundaryType::slip;
359 if (op == "function" || op == "function weak" || op == "partial slip")
360 {
361 if (op == "function")
362 this->type[boundary_id] = BoundaryType::function;
363 else if (op == "partial slip")
364 this->type[boundary_id] = BoundaryType::partial_slip;
365 else
366 this->type[boundary_id] = BoundaryType::function_weak;
367 }
368 if (op == "pressure")
369 {
370 this->type[boundary_id] = BoundaryType::pressure;
371 }
372 if (op == "periodic")
373 {
374 types::boundary_id periodic_boundary_id =
375 prm.get_integer("periodic_id");
376
377 this->type[boundary_id] = BoundaryType::periodic;
378
379 // We attribute a periodic neighbor boundary type to the neighbor
380 // boundary to ensure that all boundaries have a defined type
381 this->type[periodic_boundary_id] = BoundaryType::periodic_neighbor;
382
383 // We store the periodic id and direction
384 this->periodic_neighbor_id[boundary_id] = periodic_boundary_id;
385 this->periodic_direction[boundary_id] =
386 prm.get_integer("periodic_direction");
387
388 // Allocate the navier_stokes_functions object for the periodic neighbor
389 // condition to ensure that they have a defined function and a center of
390 // rotation. Parse the information of the boundary id to duplicate it.
391 navier_stokes_functions[periodic_boundary_id] =
392 std::make_shared<NSBoundaryFunctions<dim>>();
393
394 prm.enter_subsection("u");
395 navier_stokes_functions[periodic_boundary_id]->u.parse_parameters(prm);
396 prm.leave_subsection();
397
398 prm.enter_subsection("v");
399 navier_stokes_functions[periodic_boundary_id]->v.parse_parameters(prm);
400 prm.leave_subsection();
401
402 prm.enter_subsection("w");
403 navier_stokes_functions[periodic_boundary_id]->w.parse_parameters(prm);
404 prm.leave_subsection();
405
406 prm.enter_subsection("p");
407 navier_stokes_functions[periodic_boundary_id]->p.parse_parameters(prm);
408 prm.leave_subsection();
409
410 prm.enter_subsection("center of rotation");
411 navier_stokes_functions[periodic_boundary_id]->center_of_rotation[0] =
412 prm.get_double("x");
413 navier_stokes_functions[periodic_boundary_id]->center_of_rotation[1] =
414 prm.get_double("y");
415
416 if (dim == 3)
417 navier_stokes_functions[periodic_boundary_id]->center_of_rotation[2] =
418 prm.get_double("z");
419 prm.leave_subsection();
420 }
421 if (op == "outlet")
422 {
423 this->type[boundary_id] = BoundaryType::outlet;
424 }
425
426 this->beta[boundary_id] = prm.get_double("beta");
427 this->boundary_layer_thickness[boundary_id] =
428 prm.get_double("boundary layer thickness");
429 }

496 /**
497 * @brief This class manages the boundary conditions for Heat-Transfer solver
498 * It introduces the boundary functions and declares the boundary conditions
499 * coherently.
500 * The members "value", "h" and "Tinf" contain double used for bc calculation:

510 * coefficient, and "Stefan-Boltzmann constant" is the Stefan-Boltzmann

513 *
514 */

525 h;
526 std::map<types::boundary_id,
527 std::shared_ptr<Functions::ParsedFunction<dim>>>
528 Tinf;
529 std::map<types::boundary_id,
530 std::shared_ptr<Functions::ParsedFunction<dim>>>
531 emissivity;
532 std::map<types::boundary_id,
533 std::shared_ptr<Functions::ParsedFunction<dim>>>
534 heat_flux_bc;
535 double Stefan_Boltzmann_constant;
536
537 void
538 declare_default_entry(ParameterHandler &prm,
539 const types::boundary_id default_boundary_id);
540 void
541 declare_parameters(ParameterHandler &prm,
542 const unsigned int number_of_boundary_conditions);
543 void
544 parse_boundary(ParameterHandler &prm);

546 parse_parameters(ParameterHandler &prm);

548
549 // TODO REMOVE THIS FLAG IT IS NOT USEFUL

647 /**
648 * @brief Parse the information for a boundary condition
649 *
650 * @param prm A parameter handler which is currently used to parse the simulation information
651 *
652 * @param i_bc The boundary condition number (and not necessarily it's id).
653 */
654
655 template <int dim>
656 void
657 HTBoundaryConditions<dim>::parse_boundary(ParameterHandler &prm)
658 {
659 AssertThrow(
660 prm.get_integer("id") >= 0,
661 ExcMessage(
662 "An invalid boundary id has been given for a heat transfer boundary condition."));
663
664 types::boundary_id boundary_id = prm.get_integer("id");
665
666 AssertThrow(this->type.find(boundary_id) == this->type.end(),
667 HeatTransferBoundaryDuplicated(boundary_id));
668
669 const std::string op = prm.get("type");
670 if (op == "noflux")
671 {
672 this->type[boundary_id] = BoundaryType::noflux;
673 }
674 if (op == "temperature")
675 {
676 this->type[boundary_id] = BoundaryType::temperature;
677 }
678 else if (op == "convection-radiation-flux")
679 {
680 this->type[boundary_id] = BoundaryType::convection_radiation;
681 this->has_convection_radiation_bc = true;
682
683 // Emissivity validity (0 <= emissivity <= 1) will be checked at
684 // evaluation.
685 }
686
687 // All the functions are parsed since they might be used for post-processing
688 prm.enter_subsection("value");
689 this->dirichlet_value[boundary_id] =
690 std::make_shared<Functions::ParsedFunction<dim>>();
691 this->dirichlet_value[boundary_id]->parse_parameters(prm);
692 prm.leave_subsection();
693 prm.enter_subsection("h");
694 this->h[boundary_id] = std::make_shared<Functions::ParsedFunction<dim>>();
695 this->h[boundary_id]->parse_parameters(prm);
696 prm.leave_subsection();
697 prm.enter_subsection("Tinf");
698 this->Tinf[boundary_id] =
699 std::make_shared<Functions::ParsedFunction<dim>>();
700 this->Tinf[boundary_id]->parse_parameters(prm);
701 prm.leave_subsection();
702 prm.enter_subsection("emissivity");
703 this->emissivity[boundary_id] =
704 std::make_shared<Functions::ParsedFunction<dim>>();
705 this->emissivity[boundary_id]->parse_parameters(prm);
706 prm.leave_subsection();
707 prm.enter_subsection("heat_flux");
708 this->heat_flux_bc[boundary_id] =
709 std::make_shared<Functions::ParsedFunction<dim>>();
710 this->heat_flux_bc[boundary_id]->parse_parameters(prm);
711 prm.leave_subsection();
712 }
