32/**
33 * @brief CLS phase fraction gradient L2 projection solver.
34 *
35 * @tparam dim Number of dimensions of the problem.
36 */
37template <int dim>
38class CLSPhaseGradientProjection : public CLSLinearSubequationsSolver<dim>
39{
40public:
41 /**
42 * @brief Constructor for the L2 projection of the CLS phase fraction gradient
43 * (pfg).
44 *
45 * @param[in] p_simulation_parameters Simulation parameters.
46 *
47 * @param[in] p_pcout Parallel cout used to print the information.
48 *
49 * @param[in] p_triangulation Distributed mesh information.
50 *
51 * @param[in,out] p_subequations_interface Subequations interface object used
52 * to get information from other subequations and store information from the
53 * current one.
54 */
55 CLSPhaseGradientProjection(
56 const SimulationParameters<dim> &p_simulation_parameters,
57 const ConditionalOStream &p_pcout,
58 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
59 p_triangulation,
60 CLSSubequationsInterface<dim> &p_subequations_interface)
61 : CLSLinearSubequationsSolver<dim>(
62 CLSSubequationsID::phase_gradient_projection,
63 p_simulation_parameters,
64 ((p_simulation_parameters.multiphysics.cls_parameters
65 .surface_tension_force.verbosity != Parameters::Verbosity::quiet) ||
66 ((p_simulation_parameters.multiphysics.cls_parameters
67 .regularization_method.algebraic_interface_reinitialization
68 .enable) &&
69 (p_simulation_parameters.multiphysics.cls_parameters
70 .regularization_method.verbosity !=
71 Parameters::Verbosity::quiet))) ?
72 Parameters::Verbosity::verbose :
73 Parameters::Verbosity::quiet, // Set to verbose if surface tension
74 // verbosity is enabled or if algebraic
75 // interface reinitialization is enabled
76 // and set to verbose
77 p_pcout,
78 p_triangulation,
79 p_subequations_interface)
80 {
81 if (this->simulation_parameters.mesh.simplex)
82 {
83 // For simplex meshes
84 const FE_SimplexP<dim> subequation_fe(
85 this->simulation_parameters.fem_parameters.CLS_order);
86 this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
87 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
88
89 this->cell_quadrature =
90 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
91 }
92 else
93 {
94 // Usual case, for quad/hex meshes
95 const FE_Q<dim> subequation_fe(
96 this->simulation_parameters.fem_parameters.CLS_order);
97 this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
98 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
99
100 this->cell_quadrature =
101 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
102 }
103 }
104
105 /**
106 * @brief Default destructor.
107 */
108 ~CLSPhaseGradientProjection() = default;
109
110private:
111 /**
112 * @brief Assemble system matrix and right-hand side (rhs).
113 */
114 void
115 assemble_system_matrix_and_rhs() override;
116
117 /**
118 * @brief Check if a new CLS solution has been set.
119 */
120 void
121 check_dependencies_validity() override;
122};
