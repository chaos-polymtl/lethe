180 /**
181 * @brief Execute the contact calculation step for the particle-particle
182 * contact only for local-local and local-ghost contacts with no periodicity.
183 * This is a simplified version of the contact calculation of the
184 * particle-particle contact forces class, without the other contact types and
185 * the update of the particles forces, torques and tangential displacement.
186 *
187 * @tparam contact_type The type of contact to be calculated.
188 * @param[in] adjacent_particles_list Container of the adjacent particles of a
189 * particles.
190 * @param[out] vertices the vector of positions of touching particles.
191 * @param[out] normal_forces_vector the vector of normal forces between each
192 * touching particles.
193 */
194 template <ContactType contact_type>
195 inline void
196 execute_contact_calculation(
197 typename DEM::dem_data_structures<dim>::particle_contact_info
198 &adjacent_particles_list,
199 std::vector<Point<3>> &vertices,
200 std::vector<double> &normal_forces_vector)
201 {
202 // No contact calculation if no adjacent particles
203 if (adjacent_particles_list.empty())
204 return;
205
206 const double force_calculation_threshold_distance =
207 this->get_force_calculation_threshold_distance();
208
209 // Define local variables which will be used within the contact calculation
210 Tensor<1, 3> normal_unit_vector;
211 Tensor<1, 3> normal_force;
212 Tensor<1, 3> tangential_force;
213 Tensor<1, 3> particle_one_tangential_torque;
214 Tensor<1, 3> particle_two_tangential_torque;
215 Tensor<1, 3> rolling_resistance_torque;
216 double normal_relative_velocity_value;
217 Tensor<1, 3> tangential_relative_velocity;
218
219 // Gather information about particle 1 and set it up.
220 auto first_contact_info = adjacent_particles_list.begin();
221 auto particle_one = first_contact_info->second.particle_one;
222 auto particle_one_properties = particle_one->get_properties();
223
224 // Fix particle one location for 2d and 3d
225 Point<3> particle_one_location = this->get_location(particle_one);
226
227 for (auto &&contact_info :
228 adjacent_particles_list | boost::adaptors::map_values)
229 {
230 // Getting information (location and properties) of particle 2 in
231 // contact with particle 1
232 auto particle_two = contact_info.particle_two;
233 auto particle_two_properties = particle_two->get_properties();
234
235 if constexpr (contact_type == ghost_particle_particle)
236 {
237 // We create an arbitrary rule so that forces between local-ghost
238 // particle are not written twice by each processor.
239 if (particle_one->get_id() < particle_two->get_id())
240 continue;
241 }
242
243 // Get particle 2 location in dimension independent way
244 Point<3> particle_two_location = this->get_location(particle_two);
245
246 // Calculation of normal overlap
247 double normal_overlap =
248 0.5 * (particle_one_properties[PropertiesIndex::dp] +
249 particle_two_properties[PropertiesIndex::dp]) -
250 particle_one_location.distance(particle_two_location);
251
252 if (normal_overlap > force_calculation_threshold_distance)
253 {
254 update_contact_information(tangential_relative_velocity,
255 normal_relative_velocity_value,
256 normal_unit_vector,
257 particle_one_properties,
258 particle_two_properties,
259 particle_one_location,
260 particle_two_location);
261
262 this->calculate_contact(
263 contact_info,
264 tangential_relative_velocity,
265 normal_relative_velocity_value,
266 normal_unit_vector,
267 normal_overlap,
268 0., // dt is not required for the force chains
269 particle_one_properties,
270 particle_two_properties,
271 normal_force,
272 tangential_force,
273 particle_one_tangential_torque,
274 particle_two_tangential_torque,
275 rolling_resistance_torque);
276
277 vertices.push_back(particle_one_location);
278 vertices.push_back(particle_two_location);
279 normal_forces_vector.push_back(
280 scalar_product(normal_force, normal_unit_vector));
281 }
282 }
283 }

285 /**
286 * @brief Update the contact pair information for all contact force
287 * calculations.
288 *
289 * @param[out] tangential_relative_velocity Tangential relative velocity.
290 * @param[out] normal_relative_velocity_value Normal relative velocity.
291 * @param[out] normal_unit_vector Normal vector of the contact.
292 * @param[in] particle_one_properties Properties of particle one in contact.
293 * @param[in] particle_two_properties Properties of particle two in contact.
294 * @param[in] particle_one_location Location of particle one in contact.
295 * @param[in] particle_two_location Location of particle two in contact.
296 */
297 inline void
298 update_contact_information(
299 Tensor<1, 3> &tangential_relative_velocity,
300 double &normal_relative_velocity_value,
301 Tensor<1, 3> &normal_unit_vector,
302 const ArrayView<const double> &particle_one_properties,
303 const ArrayView<const double> &particle_two_properties,
304 const Point<3> &particle_one_location,
305 const Point<3> &particle_two_location)
306 {
307 // Calculation of the contact vector from particle one to particle two
308 Tensor<1, 3> contact_vector = particle_two_location - particle_one_location;
309
310 // Calculation of the normal unit contact vector
311 normal_unit_vector = contact_vector / contact_vector.norm();
312
313 // Defining velocities and angular velocities of particles one and
314 // two as vectors
315 Tensor<1, 3> particle_one_omega, particle_two_omega;
316
317 // Defining relative contact velocity
318 Tensor<1, 3> contact_relative_velocity;
319
320 // Assigning velocities and angular velocities of particles
321 contact_relative_velocity[0] =
322 particle_one_properties[PropertiesIndex::v_x] -
323 particle_two_properties[PropertiesIndex::v_x];
324 contact_relative_velocity[1] =
325 particle_one_properties[PropertiesIndex::v_y] -
326 particle_two_properties[PropertiesIndex::v_y];
327 contact_relative_velocity[2] =
328 particle_one_properties[PropertiesIndex::v_z] -
329 particle_two_properties[PropertiesIndex::v_z];
330
331 particle_one_omega[0] = particle_one_properties[PropertiesIndex::omega_x];
332 particle_one_omega[1] = particle_one_properties[PropertiesIndex::omega_y];
333 particle_one_omega[2] = particle_one_properties[PropertiesIndex::omega_z];
334
335 particle_two_omega[0] = particle_two_properties[PropertiesIndex::omega_x];
336 particle_two_omega[1] = particle_two_properties[PropertiesIndex::omega_y];
337 particle_two_omega[2] = particle_two_properties[PropertiesIndex::omega_z];
338
339 // Calculation of contact relative velocity
340 // v_ij = (v_i - v_j) + (R_i*omega_i + R_j*omega_j) × n_ij
341 contact_relative_velocity += (cross_product_3d(
342 0.5 * (particle_one_properties[PropertiesIndex::dp] * particle_one_omega +
343 particle_two_properties[PropertiesIndex::dp] * particle_two_omega),
344 normal_unit_vector));
345
346 // Calculation of normal relative velocity. Note that in the
347 // following line the product acts as inner product since both
348 // sides are vectors, while in the second line the product is
349 // scalar and vector product
350 normal_relative_velocity_value =
351 contact_relative_velocity * normal_unit_vector;
352
353 // Calculation of tangential relative velocity
354 // v_rt = v_ij - (v_ij⋅n_ij)*n_ij
355 tangential_relative_velocity =
356 contact_relative_velocity -
357 (normal_relative_velocity_value * normal_unit_vector);
358 }
