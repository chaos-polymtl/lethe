36template <int dim, int spacedim>
37SolidBase<dim, spacedim>::SolidBase(
38 std::shared_ptr<Parameters::NitscheObject<spacedim>> &param,
39 std::shared_ptr<parallel::DistributedTriangulationBase<spacedim>> fluid_tria,
40 std::shared_ptr<Mapping<spacedim>> fluid_mapping)
41 : mpi_communicator(MPI_COMM_WORLD)
42 , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
43 , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
44 , fluid_tria(fluid_tria)
45 , fluid_mapping(fluid_mapping)
46 , param(param)
47 , velocity(&param->solid_velocity)
48 , temperature(&param->solid_temperature)
49{
50 if (param->solid_mesh.simplex)
51 {
52 // for simplex meshes
53 fe = std::make_shared<FE_SimplexP<dim, spacedim>>(1);
54 solid_mapping = std::make_shared<MappingFE<dim, spacedim>>(*fe);
55 quadrature =
56 std::make_shared<QGaussSimplex<dim>>(param->number_quadrature_points);
57 solid_tria = std::make_shared<
58 parallel::fullydistributed::Triangulation<dim, spacedim>>(
59 this->mpi_communicator);
60
61 displacement_fe =
62 std::make_shared<FESystem<dim, spacedim>>(FE_SimplexP<dim, spacedim>(1),
63 spacedim);
64 }
65 else
66 {
67 // Usual case, for quad/hex meshes
68 fe = std::make_shared<FE_Q<dim, spacedim>>(1);
69 solid_mapping = std::make_shared<MappingQGeneric<dim, spacedim>>(1);
70 quadrature =
71 std::make_shared<QGauss<dim>>(param->number_quadrature_points);
72 solid_tria =
73 std::make_shared<parallel::distributed::Triangulation<dim, spacedim>>(
74 mpi_communicator,
75 typename Triangulation<dim, spacedim>::MeshSmoothing(
76 Triangulation<dim, spacedim>::smoothing_on_refinement |
77 Triangulation<dim, spacedim>::smoothing_on_coarsening));
78
79 displacement_fe =
80 std::make_shared<FESystem<dim, spacedim>>(FE_Q<dim, spacedim>(1),
81 spacedim);
82 }
83
84 // Dof handler associated with particles
85 solid_dh.clear();
86 solid_dh.reinit(*this->solid_tria);
87
88 // Dof handler associated with mesh displacement
89 displacement_dh.clear();
90 displacement_dh.reinit(*this->solid_tria);
91}

106template <int dim, int spacedim>
107void
108SolidBase<dim, spacedim>::setup_triangulation(const bool restart)
109{
110 if (param->solid_mesh.type == Parameters::Mesh::Type::gmsh)
111 {
112 if (param->solid_mesh.simplex)
113 {
114 auto comm = solid_tria->get_mpi_communicator();
115 std::string file_name = param->solid_mesh.file_name;
116
117 auto construction_data = TriangulationDescription::Utilities::
118 create_description_from_triangulation_in_groups<dim, spacedim>(
119 [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
120 GridIn<dim, spacedim> grid_in;
121 grid_in.attach_triangulation(basetria);
122 std::ifstream input_file(file_name);
123
124 grid_in.read_msh(input_file);
125 },
126 [](dealii::Triangulation<dim, spacedim> &basetria,
127 const MPI_Comm comm,
128 const unsigned int /*group_size*/) {
129 GridTools::partition_triangulation(
130 Utilities::MPI::n_mpi_processes(comm), basetria);
131 },
132 comm,
133 Utilities::MPI::n_mpi_processes(comm) /* group size */,
134 dealii::Triangulation<dim, spacedim>::none);
135
136 solid_tria->create_triangulation(construction_data);
137 }
138 else
139 { // Grid creation
140 GridIn<dim, spacedim> grid_in;
141 // Attach triangulation to the grid
142 grid_in.attach_triangulation(*solid_tria);
143 // Read input gmsh file
144 std::ifstream input_file(param->solid_mesh.file_name);
145 grid_in.read_msh(input_file);
146 }
147 }
148 else if (param->solid_mesh.type == Parameters::Mesh::Type::dealii)
149 {
150 if (param->solid_mesh.simplex)
151 { // TODO Using dealii generated meshes with simplices in Nitsche solver
152 // generates and error. "boundary_id !=
153 // numbers::internal_face_boundary_id"
154 Triangulation<dim, spacedim> temporary_quad_triangulation;
155 GridGenerator::generate_from_name_and_arguments(
156 temporary_quad_triangulation,
157 param->solid_mesh.grid_type,
158 param->solid_mesh.grid_arguments);
159
160 // initial refinement
161 const int initial_refinement = param->solid_mesh.initial_refinement;
162 temporary_quad_triangulation.refine_global(initial_refinement);
163 // flatten the triangulation
164 Triangulation<dim, spacedim> flat_temp_quad_triangulation;
165 GridGenerator::flatten_triangulation(temporary_quad_triangulation,
166 flat_temp_quad_triangulation);
167
168 Triangulation<dim, spacedim> temporary_tri_triangulation(
169 Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
170 GridGenerator::convert_hypercube_to_simplex_mesh(
171 flat_temp_quad_triangulation, temporary_tri_triangulation);
172
173 GridTools::partition_triangulation_zorder(
174 Utilities::MPI::n_mpi_processes(solid_tria->get_mpi_communicator()),
175 temporary_tri_triangulation);
176 GridTools::partition_multigrid_levels(temporary_tri_triangulation);
177
178 // extract relevant information from distributed triangulation
179 auto construction_data = TriangulationDescription::Utilities::
180 create_description_from_triangulation(
181 temporary_tri_triangulation,
182 solid_tria->get_mpi_communicator(),
183 TriangulationDescription::Settings::
184 construct_multigrid_hierarchy);
185 solid_tria->create_triangulation(construction_data);
186 }
187 else
188 { // deal.ii creates grid and attaches the solid triangulation
189 GridGenerator::generate_from_name_and_arguments(
190 *solid_tria,
191 param->solid_mesh.grid_type,
192 param->solid_mesh.grid_arguments);
193 }
194 }
195 else
196 throw std::runtime_error(
197 "Unsupported mesh type - solid mesh will not be created");
198
199 // Rotate the triangulation
200 rotate_grid(param->solid_mesh.rotation_angle,
201 param->solid_mesh.rotation_axis);
202
203 // Translate the triangulation
204 translate_grid(param->solid_mesh.translation);
205 // Refine the solid triangulation to its initial size
206 // NB: solid_tria should not be refined if loaded from a restart file
207 // afterwards
208 if (!restart)
209 {
210 if (param->solid_mesh.simplex)
211 {
212 // Simplex triangulation refinement isn't possible yet
213 }
214 else
215 {
216 solid_tria->refine_global(param->solid_mesh.initial_refinement);
217 }
218 solid_dh.distribute_dofs(*fe);
219 }
220}

298template <int dim, int spacedim>
299void
300SolidBase<dim, spacedim>::setup_particles()
301{
302 setup_particles_handler();
303
304 // Initialize FEValues
305 std::vector<Point<spacedim>> quadrature_points_vec;
306 quadrature_points_vec.reserve(quadrature->size() *
307 solid_tria->n_locally_owned_active_cells());
308 unsigned int n_properties = (dim == 2 && spacedim == 3) ? 1 + spacedim : 1;
309
310 std::vector<std::vector<double>> properties;
311 properties.reserve(quadrature->size() *
312 solid_tria->n_locally_owned_active_cells() * n_properties);
313
314 UpdateFlags update_flags = update_JxW_values | update_quadrature_points;
315 if (dim == 2 && spacedim == 3)
316 update_flags = update_flags | update_normal_vectors;
317
318 FEValues<dim, spacedim> fe_v(*solid_mapping, *fe, *quadrature, update_flags);
319
320 // Fill quadrature points vector and properties, used to fill
321 // solid_particle_handler
322 for (const auto &cell : solid_dh.active_cell_iterators())
323 if (cell->is_locally_owned())
324 {
325 fe_v.reinit(cell);
326 const auto &points = fe_v.get_quadrature_points();
327 const auto &JxW = fe_v.get_JxW_values();
328 if (dim == 2 && spacedim == 3)
329 {
330 const auto &normal_vectors = fe_v.get_normal_vectors();
331 for (unsigned int q = 0; q < points.size(); ++q)
332 {
333 quadrature_points_vec.emplace_back(points[q]);
334 std::vector<double> prop_i = {JxW[q],
335 normal_vectors[q][0],
336 normal_vectors[q][1]};
337 if (spacedim == 3)
338 prop_i.push_back(normal_vectors[q][2]);
339 properties.emplace_back(prop_i);
340 }
341 }
342 else
343 {
344 for (unsigned int q = 0; q < points.size(); ++q)
345 {
346 quadrature_points_vec.emplace_back(points[q]);
347 std::vector<double> prop_i = {JxW[q]};
348 properties.emplace_back(prop_i);
349 }
350 }
351 }
352
353 // Compute fluid bounding box
354 std::vector<std::vector<BoundingBox<spacedim>>> global_fluid_bounding_boxes;
355
356
357
358 // Use the more general boost rtree bounding boxes
359 std::vector<BoundingBox<spacedim>> all_boxes;
360 all_boxes.reserve(fluid_tria->n_locally_owned_active_cells());
361 for (const auto &cell : fluid_tria->active_cell_iterators())
362 if (cell->is_locally_owned())
363 all_boxes.emplace_back(cell->bounding_box());
364 const auto tree = pack_rtree(all_boxes);
365 const auto local_boxes = extract_rtree_level(
366 tree, std::max(int(log10(fluid_tria->n_locally_owned_active_cells())), 1));
367
368 global_fluid_bounding_boxes =
369 Utilities::MPI::all_gather(mpi_communicator, local_boxes);
370
371 // Fill solid particle handler
372 solid_particle_handler->insert_global_particles(quadrature_points_vec,
373 global_fluid_bounding_boxes,
374 properties);
375
376 // Number of particles used to assess that no particle is lost
377 initial_number_of_particles = solid_particle_handler->n_global_particles();
378}

380template <int dim, int spacedim>
381void
382SolidBase<dim, spacedim>::load_particles(const std::string &filename_part)
383{
384 // Setup particles handler
385 setup_particles_handler();
386
387 // Gather particle serialization information
388 std::ifstream input(filename_part.c_str());
389 AssertThrow(input, ExcFileNotOpen(filename_part));
390
391 std::string buffer;
392 std::getline(input, buffer);
393 std::istringstream iss(buffer);
394 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
395
396 ia >> *solid_particle_handler;
397
398 initial_number_of_particles = solid_particle_handler->n_global_particles();
399}

458template <int dim, int spacedim>
459void
460SolidBase<dim, spacedim>::integrate_velocity(double time_step,
461 double initial_time)
462{
463 const unsigned int sub_particles_iterations = param->particles_sub_iterations;
464 AssertThrow(sub_particles_iterations >= 1,
465 ExcMessage("Sub particles iterations must be 1 or larger"));
466
467
468
469 double sub_iteration_relaxation = 1. / sub_particles_iterations;
470 time_step = time_step * sub_iteration_relaxation;
471 // Particle sub iterations divide the time step in a number of "sub
472 // iterations". This allows the solver to use a larger CFL without
473 // necessitating the use of the more complex particle location detection
474 // approaches. The number of sub particles iterations must be chosen so that
475 // the time_step / sub_particles_iterations * velocity / cell size is smaller
476 // than unity
477 for (unsigned int it = 0; it < sub_particles_iterations; ++it)
478 {
479 for (auto particle = solid_particle_handler->begin();
480 particle != solid_particle_handler->end();
481 ++particle)
482 {
483 // Calculate the next position of the particle using the RK4
484 // time integration scheme
485 // x(t+dt) = x(t) + 1/6 (k1+2*k2+2*k3+k4)
486 // k1 = dt * v(t,x(t))
487 // k2 = dt * v(t+0.5*dt,x+k1/2)
488 // k3 = dt * v(t+0.5*dt,x+k2/2)
489 // k4 = dt * vt(t+dt,x+k3)
490 // The four stages (1 to 4) are built successively
491 // For each stage, the time of the function must be "reset"
492 // to ensure adequate evaluation of the RK4 scheme.
493 // See https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods for
494 // more details
495
496 Point<spacedim> particle_location = particle->get_location();
497
498 Tensor<1, spacedim> k1;
499 velocity->set_time(initial_time);
500 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
501 k1[comp_i] = velocity->value(particle_location, comp_i);
502
503 Point<spacedim> p1 = particle_location + time_step / 2 * k1;
504 Tensor<1, spacedim> k2;
505 velocity->set_time(initial_time + time_step / 2);
506 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
507 k2[comp_i] = velocity->value(p1, comp_i);
508
509 Point<spacedim> p2 = particle_location + time_step / 2 * k2;
510 Tensor<1, spacedim> k3;
511 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
512 k3[comp_i] = velocity->value(p2, comp_i);
513
514 Point<spacedim> p3 = particle_location + time_step * k3;
515 Tensor<1, spacedim> k4;
516 velocity->set_time(initial_time + time_step);
517 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
518 k4[comp_i] = velocity->value(p3, comp_i);
519
520 particle_location += time_step / 6 * (k1 + 2 * k2 + 2 * k3 + k4);
521 particle->set_location(particle_location);
522 }
523 solid_particle_handler->sort_particles_into_subdomains_and_cells();
524 initial_time += time_step;
525 }
526
527 if (initial_number_of_particles !=
528 solid_particle_handler->n_global_particles())
529 {
530 if (this_mpi_process == 0)
531 {
532 std::cout << "Warning - Nitsche Particles have been lost"
533 << std::endl;
534 std::cout << "Initial number of particles : "
535 << initial_number_of_particles << std::endl;
536 std::cout << "Current number of particles : "
537 << solid_particle_handler->n_global_particles()
538 << std::endl;
539 std::cout << "Try increasing the number of particles sub iterations"
540 << std::endl;
541
542 // Stops the simulation if particles have been lost
543 // Future improvement would be to dynamically increase the number of
544 // particles subiteration if particles have been lost
545 if (param->stop_particles_lost)
546 {
547 std::cerr
548 << std::endl
549 << std::endl
550 << "----------------------------------------------------"
551 << std::endl;
552 std::cerr
553 << "Exception on processing: " << std::endl
554 << "Nitsche Particles have been lost" << std::endl
555 << "Aborting!" << std::endl
556 << "----------------------------------------------------"
557 << std::endl;
558 std::exit(1);
559 }
560 }
561 }
562}

564template <int dim, int spacedim>
565void
566SolidBase<dim, spacedim>::move_solid_triangulation(const double time_step,
567 const double initial_time)
568{
569 // First we calculate the displacement that the solid triangulation
570 // will feel during this iteration.
571 // Then, we apply this displacement to the triangulation
572 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
573 const std::vector<bool> locally_owned_vertices =
574 GridTools::get_locally_owned_vertices(*this->solid_tria);
575 std::vector<bool> vertex_moved(this->solid_tria->n_vertices(), false);
576 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
577
578
579 for (const auto &cell : displacement_dh.active_cell_iterators())
580 {
581 if (cell->is_locally_owned())
582 {
583 for (unsigned int vertex = 0;
584 vertex < cell->reference_cell().n_vertices();
585 ++vertex)
586 {
587 // Calculate the next position of the particle using the RK4
588 // time integration scheme
589 // x(t+dt) = x(t) + 1/6 (k1+2*k2+2*k3+k4)
590 // k1 = dt * v(t,x(t))
591 // k2 = dt * v(t+0.5*dt,x+k1/2)
592 // k3 = dt * v(t+0.5*dt,x+k2/2)
593 // k4 = dt * vt(t+dt,x+k3)
594 // The four stages (1 to 4) are built successively
595 // For each stage, the time of the function must be "reset"
596 // to ensure adequate evaluation of the RK4 scheme.
597 // See https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods
598 // for more details
599
600 const auto dof_index = cell->vertex_dof_index(vertex, 0);
601 Point<spacedim> &vertex_position = cell->vertex(vertex);
602 const unsigned global_vertex_no = cell->vertex_index(vertex);
603 if (vertex_moved[global_vertex_no] ||
604 !locally_owned_vertices[global_vertex_no])
605 continue;
606
607 Tensor<1, spacedim> k1;
608 velocity->set_time(initial_time);
609 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
610 k1[comp_i] = velocity->value(vertex_position, comp_i);
611
612 Point<spacedim> p1 = vertex_position + time_step / 2 * k1;
613 Tensor<1, spacedim> k2;
614 velocity->set_time(initial_time + time_step / 2);
615 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
616 k2[comp_i] = velocity->value(p1, comp_i);
617
618 Point<spacedim> p2 = vertex_position + time_step / 2 * k2;
619 Tensor<1, spacedim> k3;
620 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
621 k3[comp_i] = velocity->value(p2, comp_i);
622
623 Point<spacedim> p3 = vertex_position + time_step * k3;
624 Tensor<1, spacedim> k4;
625 velocity->set_time(initial_time + time_step);
626 for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
627 k4[comp_i] = velocity->value(p3, comp_i);
628
629 auto vertex_displacement =
630 time_step / 6 * (k1 + 2 * k2 + 2 * k3 + k4);
631
632 vertex_position += vertex_displacement;
633
634 for (unsigned d = 0; d < spacedim; ++d)
635 displacement[dof_index + d] =
636 displacement[dof_index + d] + vertex_displacement[d];
637
638 vertex_moved[global_vertex_no] = true;
639 }
640 }
641 }
642 this->solid_tria->communicate_locally_moved_vertices(locally_owned_vertices);
643}

646template <int dim, int spacedim>
647void
648SolidBase<dim, spacedim>::move_solid_triangulation_with_displacement()
649{
650 const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
651 std::set<unsigned int> dof_vertex_displaced;
652 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
653
654 for (const auto &cell : displacement_dh.active_cell_iterators())
655 {
656 if (cell->is_locally_owned())
657 {
658 for (unsigned int vertex = 0;
659 vertex < cell->reference_cell().n_vertices();
660 ++vertex)
661 {
662 if (!dof_vertex_displaced.count(cell->vertex_index(vertex)))
663 {
664 const auto dof_index = cell->vertex_dof_index(vertex, 0);
665 Point<spacedim> &vertex_position = cell->vertex(vertex);
666
667 for (unsigned d = 0; d < spacedim; ++d)

670 displacement_relevant[dof_index + d];
671 }
672
673 dof_vertex_displaced.insert(cell->vertex_index(vertex));
674 }
675 }
676 }

741template <int dim, int spacedim>
742void
743SolidBase<dim, spacedim>::read_checkpoint(const std::string &prefix)
744{
745 // Setup an un-refined triangulation before loading if we have a fully
746 // distributed triangulation. If we have a fully distributed triangulation,
747 // nothing should be done to setup the triangulation since it will be fully
748 // generated during the restart process.
749 if (!param->solid_mesh.simplex)
750 setup_triangulation(true);
751
752 // Read the triangulation from the checkpoint
753 const std::string filename = prefix + ".triangulation";
754 try
755 {
756 this->solid_tria->load(filename.c_str());
757 }
758 catch (...)
759 {
760 AssertThrow(false,
761 ExcMessage("Cannot open snapshot mesh file or read the "
762 "triangulation stored there."));
763 }
764
765 // Setup dof-handler for solid and displacement
766 solid_dh.distribute_dofs(*fe);
767 setup_displacement();
768
769 // Read displacement vector
770 std::vector<GlobalVectorType *> x_system(1);
771 x_system[0] = &(displacement);
772
773 SolutionTransfer<dim, GlobalVectorType, spacedim> system_trans_vectors(
774 this->displacement_dh);
775
776 system_trans_vectors.deserialize(x_system);
777 displacement_relevant = displacement;
778
779 // Reset triangulation position using displacement vector if is not a simplex
780 // triangulation fullydistributed::triangulation store their displacement when
781 // they are saved wheras distributed::triangulation do not.
782 if (!param->solid_mesh.simplex)
783 move_solid_triangulation_with_displacement();
784
785 // Gather particle serialization information and deserialize them
786 std::string particle_filename = prefix + ".particles";
787 std::ifstream input(particle_filename.c_str());
788 AssertThrow(input, ExcFileNotOpen(particle_filename));
789
790 std::string buffer;
791 std::getline(input, buffer);
792 std::istringstream iss(buffer);
793 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
794
795
796 // At this stage, the pointer for the particle handler has not been allocated
797 // it thus needs to be done by calling the setup
798 // TODO, this function should be called in the constructor
799 this->setup_particles_handler();
800 ia >> *(this->solid_particle_handler);
801
802 this->solid_particle_handler->deserialize();
803
804 // Number of particles is reset to the number of particles in the
805 // solid_particle_handler after deserialization
806 initial_number_of_particles = solid_particle_handler->n_global_particles();
807}
