15/**
16 * @brief Contain all the common elements of a physics solver. It creates
17 * the non-linear solver as specified by the user using the parameters file and
18 * provides all the necessary elements needed by the solver to solve a physics
19 * problem.
20 *
21 * @param[in] non_linear_solver_parameters Set of parameters that will be used
22 * to construct the non-linear solver.
23 */
24template <typename VectorType>
25class PhysicsSolver
26{
27public:
28 PhysicsSolver(const Parameters::NonLinearSolver non_linear_solver_parameters);
29
30 virtual ~PhysicsSolver()
31 {
32 delete non_linear_solver;
33 }

35 /**

38 virtual void

40
41 /**

45 assemble_system_rhs() = 0;
46

48 * @brief Set up the preconditioner.
49 *
50 */
51 virtual void
52 setup_preconditioner() = 0;
53
54 /**
55 * @brief Solve the linear system of equations.
56 *
57 * @param[in] initial_step Provides the linear solver with indication if this
58 * solution is the first one for the system of equation or not.
59 *
60 * @param[in] renewed_matrix Indicates to the linear solve if the system
61 * matrix has been recalculated or not.
62 */
63 virtual void
64 solve_linear_system(const bool initial_step,

68 * @brief Solve the non linear system of equations.

76 /**

141template <typename VectorType>
142PhysicsSolver<VectorType>::PhysicsSolver(
143 const Parameters::NonLinearSolver non_linear_solver_parameters)
144 : pcout(std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
145{
146 switch (non_linear_solver_parameters.solver)
147 {
148 case Parameters::NonLinearSolver::SolverType::newton:
149 non_linear_solver =
150 new NewtonNonLinearSolver<VectorType>(this,
151 non_linear_solver_parameters);
152 break;
153 case Parameters::NonLinearSolver::SolverType::kinsol_newton:
154 non_linear_solver = new KinsolNewtonNonLinearSolver<VectorType>(
155 this, non_linear_solver_parameters);
156 break;
157 case Parameters::NonLinearSolver::SolverType::inexact_newton:
158 non_linear_solver = new InexactNewtonNonLinearSolver<VectorType>(
159 this, non_linear_solver_parameters);
160 break;
161 default:
162 break;
163 }
164}
