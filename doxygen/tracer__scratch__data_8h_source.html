29/**
30 * @brief Class that stores the information required by the assembly procedure
31 * for a Tracer advection-diffusion equation. Consequently, this class
32 *calculates the tracer (values, gradients, laplacians) and the shape function
33 * (values, gradients, laplacians) at all the gauss points for all degrees
34 * of freedom and stores it into arrays. Additionnaly, the use can request
35 * that this class gathers additional fields for physics which are coupled
36 * to the Tracer equation, such as the velocity which is required. This class
37 * serves as a seperation between the evaluation at the gauss point of the
38 * variables of interest and their use in the assembly, which is carried out
39 * by the assembler functions. For more information on this design, the reader
40 * can consult deal.II step-9
41 * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
42 * example, the scratch is a struct instead of a templated class because of the
43 * simplicity of step-9.
44 *
45 * @tparam dim An integer that denotes the dimension of the space in which
46 * the flow is solved
47 * @ingroup solvers
48 **/
49

60 * @param fe The FESystem used to solve the Navier-Stokes equations

138 /** @brief Reinitializes the content of the scratch.
139 *
140 * Using the FeValues and the content of the solutions and previous solutions,
141 * fills all of the class member of the scratch.
142 *
143 * @param[in] cell The cell over which the assembly is being carried.
144 * This cell must be compatible with the fe which is used to fill the
145 * FeValues.
146 *
147 * @param[in] current_solution The present value of the solution.
148 *
149 * @param[in] previous_solutions The solutions at the previous time steps.
150 *
151 * @param[in] source_function The function describing the tracer source term.
152 */
153 template <typename VectorType>
154 void
155 reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
156 const VectorType &current_solution,
157 const std::vector<VectorType> &previous_solutions,
158 const Function<dim> *source_function)
159 {
160 this->fe_values_tracer.reinit(cell);
161
162 quadrature_points = this->fe_values_tracer.get_quadrature_points();
163 auto &fe_tracer = this->fe_values_tracer.get_fe();
164
165 source_function->value_list(quadrature_points, source);
166
167 // Compute cell diameter
168 double cell_measure =
169 compute_cell_measure_with_JxW(this->fe_values_tracer.get_JxW_values());
170 this->cell_size =
171 compute_cell_diameter<dim>(cell_measure, fe_tracer.degree);
172
173 // Gather tracer (values, gradient and laplacian)
174 this->fe_values_tracer.get_function_values(current_solution,
175 this->tracer_values);
176 this->fe_values_tracer.get_function_gradients(current_solution,
177 this->tracer_gradients);
178 this->fe_values_tracer.get_function_laplacians(current_solution,
179 this->tracer_laplacians);
180 this->fe_values_tracer.get_function_gradients(
181 previous_solutions[0], this->previous_tracer_gradients);
182
183 // Gather previous tracer values
184 for (unsigned int p = 0; p < previous_solutions.size(); ++p)
185 {
186 this->fe_values_tracer.get_function_values(previous_solutions[p],
187 previous_tracer_values[p]);
188 }
189
190 for (unsigned int q = 0; q < n_q_points; ++q)
191 {
192 this->JxW[q] = this->fe_values_tracer.JxW(q);
193
194 for (unsigned int k = 0; k < n_dofs; ++k)
195 {
196 // Shape function
197 this->phi[q][k] = this->fe_values_tracer.shape_value(k, q);
198 this->grad_phi[q][k] = this->fe_values_tracer.shape_grad(k, q);
199 this->hess_phi[q][k] = this->fe_values_tracer.shape_hessian(k, q);
200 this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
201 }
202 }
203
204 boundary_index = 0;
205 }

268 /** @brief Reinitialize the velocity, calculated by the fluid dynamics while also taking into account ALE
269 *
270 * @tparam VectorType The Vector type used for the solvers
271 *
272 * @param cell The cell for which the velocity is reinitialized
273 * This cell must be compatible with the Fluid Dynamics FE
274 *
275 * @param current_solution The present value of the solution for [u,p]
276 *
277 * @param ale The ALE parameters which include the ALE function
278 *
279 * @param drift_velocity Function to calculate the drift velocity
280 *
281 */
282 template <typename VectorType>
283 void
284 reinit_velocity(
285 const typename DoFHandler<dim>::active_cell_iterator &cell,
286 const VectorType &velocity_solution,
287 const Parameters::ALE<dim> &ale,
288 std::shared_ptr<Functions::ParsedFunction<dim>> drift_velocity)
289 {
290 this->fe_values_fd.reinit(cell);
291
292 this->fe_values_fd[velocities].get_function_values(velocity_solution,
293 velocity_values);
294
295 // Add the drift velocity to the velocity to account for tracer drift flux
296 // modeling
297 Tensor<1, dim> drift_velocity_tensor;
298 Vector<double> drift_velocity_vector(dim);
299
300 for (unsigned int q = 0; q < n_q_points; ++q)
301 {
302 drift_velocity->vector_value(quadrature_points[q],
303 drift_velocity_vector);
304 for (unsigned int d = 0; d < dim; ++d)
305 drift_velocity_tensor[d] = drift_velocity_vector[d];
306
307 velocity_values[q] += drift_velocity_tensor;
308 }
309
310 if (!ale.enabled())
311 return;
312
313 // ALE enabled, so extract the ALE velocity and subtract it from the
314 // velocity obtained from the fluid dynamics
315 Tensor<1, dim> velocity_ale;
316 std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
317 ale.velocity;
318 Vector<double> velocity_ale_vector(dim);
319
320 for (unsigned int q = 0; q < n_q_points; ++q)
321 {
322 velocity_ale_function->vector_value(quadrature_points[q],
323 velocity_ale_vector);
324 for (unsigned int d = 0; d < dim; ++d)
325 velocity_ale[d] = velocity_ale_vector[d];
326
327 velocity_values[q] -= velocity_ale;
328 }
329 }

332 /** @brief Reinitializes the content of the scratch for the internal faces. This is only used for the DG assemblers.
333 *
334 * @param[in] cell The cell over which the assembly is being carried.
335 *
336 * @param[in] face_no The face index associated with the cell
337 *
338 * @param[in] sub_face_no The subface index associated with the face
339 *
340 * @param[in] neigh_cell The neighboring cell
341 *
342 * @param[in] neigh_face_no The face index associated with the neighboring
343 * cell
344 *
345 * @param[in] neigh_sub_face_no The subface index associated with the
346 * neighboring cell
347 *
348 * @param[in] current_solution The present value of the solution.
349 */
350 template <typename VectorType>
351 void
352 reinit_internal_face(
353 const typename DoFHandler<dim>::active_cell_iterator &cell,
354 const unsigned int &face_no,
355 const unsigned int &sub_face_no,
356 const typename DoFHandler<dim>::active_cell_iterator &neigh_cell,
357 const unsigned int &neigh_face_no,
358 const unsigned int &neigh_sub_face_no,
359 const VectorType &current_solution)
360 {
361 fe_interface_values_tracer.reinit(
362 cell, face_no, sub_face_no, neigh_cell, neigh_face_no, neigh_sub_face_no);
363 face_quadrature_points = fe_interface_values_tracer.get_quadrature_points();
364
365 n_interface_dofs = fe_interface_values_tracer.n_current_interface_dofs();
366
367 const double extent_here = cell->measure() / cell->face(face_no)->measure();
368 const double extent_there =
369 neigh_cell->measure() / neigh_cell->face(neigh_face_no)->measure();
370
371 penalty_factor = get_penalty_factor(fe_values_tracer.get_fe().degree,
372 extent_here,
373 extent_there);
374
375
376 // BB TODO : Preallocate memory here
377 values_here.resize(face_quadrature_points.size());
378 values_there.resize(face_quadrature_points.size());
379 tracer_value_jump.resize(face_quadrature_points.size());
380 tracer_average_gradient.resize(face_quadrature_points.size());
381
382 fe_interface_values_tracer.get_fe_face_values(0).get_function_values(
383 current_solution, values_here);
384 fe_interface_values_tracer.get_fe_face_values(1).get_function_values(
385 current_solution, values_there);
386
387 fe_interface_values_tracer.get_jump_in_function_values(current_solution,
388 tracer_value_jump);
389
390 fe_interface_values_tracer.get_average_of_function_gradients(
391 current_solution, tracer_average_gradient);
392 }
