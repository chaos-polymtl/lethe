8template <int dim>
9void
10IBParticle<dim>::initialize_all()
11{
12 // initialize all the variables associated to an immersed boundary particle.
13 radius = 1.;
14 particle_id = 0.;
15
16 inertia[0][0] = 1.;
17 inertia[0][1] = 0.;
18 inertia[0][2] = 0.;
19 inertia[1][0] = 0.;
20 inertia[1][1] = 1.;
21 inertia[1][2] = 0.;
22 inertia[2][0] = 0.;
23 inertia[2][1] = 0.;
24 inertia[2][2] = 1.;
25
26
27 fluid_forces[0] = 0.;
28 fluid_forces[1] = 0.;
29
30 velocity[0] = 0.;
31 velocity[1] = 0.;
32
33 position[0] = 0.;
34 position[1] = 0.;
35
36 fluid_torque[0] = 0.;
37 fluid_torque[1] = 0.;
38 fluid_torque[2] = 0.;
39
40 orientation[0] = 0.;
41 orientation[1] = 0.;
42 orientation[2] = 0.;
43
44 omega[0] = 0.;
45 omega[1] = 0.;
46 omega[2] = 0.;
47
48 if (dim == 3)
49 {
50 fluid_forces[2] = 0.;
51 velocity[2] = 0.;
52 position[2] = 0.;
53 }
54
55 // Fill the vectors with default value
56 previous_fluid_forces = fluid_forces;
57 previous_fluid_torque = fluid_torque;
58 velocity_iter = velocity;
59
60 omega_iter = omega;
61 omega_impulsion = 0.;
62 omega_impulsion_iter = 0.;
63 impulsion = 0.;
64 impulsion_iter = 0.;
65 contact_impulsion = 0.;
66
67 previous_positions.resize(3);
68 previous_velocity.resize(3);
69 previous_orientation.resize(3);
70 previous_omega.resize(3);
71
72 for (unsigned int i = 0; i < 3; ++i)
73 {
74 previous_positions[i] = position;
75 previous_velocity[i] = velocity;
76 previous_orientation[i] = orientation;
77 previous_omega[i] = omega;
78 }
79
80 f_position = std::make_shared<Functions::ParsedFunction<dim>>(dim);
81 f_velocity = std::make_shared<Functions::ParsedFunction<dim>>(dim);
82 f_omega = std::make_shared<Functions::ParsedFunction<dim>>(3);
83 f_orientation = std::make_shared<Functions::ParsedFunction<dim>>(3);
84
85 rotation_matrix[0][0] = 1.;
86 rotation_matrix[0][1] = 0.;
87 rotation_matrix[0][2] = 0.;
88 rotation_matrix[1][0] = 0.;
89 rotation_matrix[1][1] = 1.;
90 rotation_matrix[1][2] = 0.;
91 rotation_matrix[2][0] = 0.;
92 rotation_matrix[2][1] = 0.;
93 rotation_matrix[2][2] = 1.;
94}

132template <int dim>
133std::vector<double>
134IBParticle<dim>::get_properties()
135{
136 std::vector<double> properties(get_number_properties());
137 properties[DEM::CFDDEMProperties::PropertiesIndex::type] = particle_id;
138 properties[DEM::CFDDEMProperties::PropertiesIndex::dp] = radius * 2.0;
139 properties[DEM::CFDDEMProperties::PropertiesIndex::v_x] = velocity[0];
140 properties[DEM::CFDDEMProperties::PropertiesIndex::v_y] = velocity[1];
141 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_x] = omega[0];
142 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_y] = omega[1];
143 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_z] = omega[2];
144 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_force_x] =
145 fluid_forces[0];
146 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_force_y] =
147 fluid_forces[1];
148 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_x] =
149 fluid_torque[0];
150 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_y] =
151 fluid_torque[1];
152 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_z] =
153 fluid_torque[2];
154 properties[DEM::CFDDEMProperties::PropertiesIndex::mass] = mass;
155
156 if (dim == 2)
157 {
158 properties[DEM::CFDDEMProperties::PropertiesIndex::v_z] = 0.;
159 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_force_z] = 0.;
160 }
161 if (dim == 3)
162 {
163 properties[DEM::CFDDEMProperties::PropertiesIndex::v_z] = velocity[2];
164 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_force_z] =
165 fluid_forces[2];
166 }
167
168 return properties;
169}

218template <int dim>
219bool
220IBParticle<dim>::is_inside_crown(
221 const Point<dim> &evaluation_point,
222 const double outer_radius,
223 const double inside_radius,
224 const bool absolute_distance,
225 const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
226{
227 const double radius = shape->effective_radius;
228
229 double distance = shape->value_with_cell_guess(evaluation_point, cell_guess);
230 bool is_inside_outer_ring;
231 bool is_outside_inner_ring;
232 if (absolute_distance)
233 {
234 is_inside_outer_ring = distance <= outer_radius;
235 is_outside_inner_ring = distance >= inside_radius;
236 }
237 else
238 {
239 is_inside_outer_ring = distance <= radius * (outer_radius - 1);
240 is_outside_inner_ring = distance >= radius * (inside_radius - 1);
241 }
242
243 return is_inside_outer_ring && is_outside_inner_ring;
244}

283IBParticle<dim>::update_precalculations(DoFHandler<dim> &updated_dof_handler,
284 const bool mesh_based_precalculations)
285{
286 if (integrate_motion || velocity.norm() > 1e-16 || omega.norm() > 1e-16)
287 {
288 this->load_data_from_file();
289 }
290 if (typeid(*shape) == typeid(RBFShape<dim>))
291 {
292 std::static_pointer_cast<RBFShape<dim>>(shape)->update_precalculations(
293 updated_dof_handler, mesh_based_precalculations);
294 }
295 else if (typeid(*shape) == typeid(CompositeShape<dim>))
296 {
297 std::static_pointer_cast<CompositeShape<dim>>(shape)
298 ->update_precalculations(updated_dof_handler,
299 mesh_based_precalculations);
