7template <int dim>
8void
9IBParticle<dim>::initialize_all()
10{
11 // initialize all the variables associated to an immersed boundary particle.
12 radius = 1.;
13 particle_id = 0.;
14
15 inertia[0][0] = 1.;
16 inertia[0][1] = 0.;
17 inertia[0][2] = 0.;
18 inertia[1][0] = 0.;
19 inertia[1][1] = 1.;
20 inertia[1][2] = 0.;
21 inertia[2][0] = 0.;
22 inertia[2][1] = 0.;
23 inertia[2][2] = 1.;
24
25
26 fluid_forces[0] = 0.;
27 fluid_forces[1] = 0.;
28
29 velocity[0] = 0.;
30 velocity[1] = 0.;
31
32 position[0] = 0.;
33 position[1] = 0.;
34
35 fluid_torque[0] = 0.;
36 fluid_torque[1] = 0.;
37 fluid_torque[2] = 0.;
38
39 orientation[0] = 0.;
40 orientation[1] = 0.;
41 orientation[2] = 0.;
42
43 omega[0] = 0.;
44 omega[1] = 0.;
45 omega[2] = 0.;
46
47 if (dim == 3)
48 {
49 fluid_forces[2] = 0.;
50 velocity[2] = 0.;
51 position[2] = 0.;
52 }
53
54 // Fill the vectors with default value
55 previous_fluid_forces = fluid_forces;
56 previous_fluid_torque = fluid_torque;
57 velocity_iter = velocity;
58
59 omega_iter = omega;
60 omega_impulsion = 0.;
61 omega_impulsion_iter = 0.;
62 impulsion = 0.;
63 impulsion_iter = 0.;
64 contact_impulsion = 0.;
65
66 previous_positions.resize(3);
67 previous_velocity.resize(3);
68 previous_orientation.resize(3);
69 previous_omega.resize(3);
70
71 for (unsigned int i = 0; i < 3; ++i)
72 {
73 previous_positions[i] = position;
74 previous_velocity[i] = velocity;
75 previous_orientation[i] = orientation;
76 previous_omega[i] = omega;
77 }
78
79 f_position = std::make_shared<Functions::ParsedFunction<dim>>(dim);
80 f_velocity = std::make_shared<Functions::ParsedFunction<dim>>(dim);
81 f_omega = std::make_shared<Functions::ParsedFunction<dim>>(3);
82 f_orientation = std::make_shared<Functions::ParsedFunction<dim>>(3);
83
84 rotation_matrix[0][0] = 1.;
85 rotation_matrix[0][1] = 0.;
86 rotation_matrix[0][2] = 0.;
87 rotation_matrix[1][0] = 0.;
88 rotation_matrix[1][1] = 1.;
89 rotation_matrix[1][2] = 0.;
90 rotation_matrix[2][0] = 0.;
91 rotation_matrix[2][1] = 0.;
92 rotation_matrix[2][2] = 1.;
93}

131template <int dim>
132std::vector<double>
133IBParticle<dim>::get_properties()
134{
135 std::vector<double> properties(get_number_properties());
136 properties[DEM::CFDDEMProperties::PropertiesIndex::type] = particle_id;
137 properties[DEM::CFDDEMProperties::PropertiesIndex::dp] = radius * 2.0;
138 properties[DEM::CFDDEMProperties::PropertiesIndex::v_x] = velocity[0];
139 properties[DEM::CFDDEMProperties::PropertiesIndex::v_y] = velocity[1];
140 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_x] = omega[0];
141 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_y] = omega[1];
142 properties[DEM::CFDDEMProperties::PropertiesIndex::omega_z] = omega[2];
143 properties
144 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_x] =
145 fluid_forces[0];
146 properties
147 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_y] =
148 fluid_forces[1];
149 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_x] =
150 fluid_torque[0];
151 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_y] =
152 fluid_torque[1];
153 properties[DEM::CFDDEMProperties::PropertiesIndex::fem_torque_z] =
154 fluid_torque[2];
155 properties[DEM::CFDDEMProperties::PropertiesIndex::mass] = mass;
156
157 if (dim == 2)
158 {
159 properties[DEM::CFDDEMProperties::PropertiesIndex::v_z] = 0.;
160 properties
161 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_z] =
162 0.;
163 }
164 if (dim == 3)
165 {
166 properties[DEM::CFDDEMProperties::PropertiesIndex::v_z] = velocity[2];
167 properties
168 [DEM::CFDDEMProperties::PropertiesIndex::fem_force_two_way_coupling_z] =
169 fluid_forces[2];
170 }
171
172 return properties;
173}

193template <int dim>
194bool
195IBParticle<dim>::is_inside_crown(
196 const Point<dim> &evaluation_point,
197 const double outer_radius,
198 const double inside_radius,
199 const bool absolute_distance,
200 const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
201{
202 const double effective_radius = shape->effective_radius;
203
204 double distance = shape->value_with_cell_guess(evaluation_point, cell_guess);
205 bool is_inside_outer_ring;
206 bool is_outside_inner_ring;
207 if (absolute_distance)
208 {
209 is_inside_outer_ring = distance <= outer_radius;
210 is_outside_inner_ring = distance >= inside_radius;
211 }
212 else
213 {
214 is_inside_outer_ring = distance <= effective_radius * (outer_radius - 1);
215 is_outside_inner_ring =
216 distance >= effective_radius * (inside_radius - 1);
217 }
218
219 return is_inside_outer_ring && is_outside_inner_ring;
220}

287template <int dim>
288void
289IBParticle<dim>::update_precalculations(DoFHandler<dim> &updated_dof_handler,
290 const bool precalculations)
291{
292 if (integrate_motion || velocity.norm() > 1e-16 || omega.norm() > 1e-16)
293 {
294 this->load_data_from_file();
295 }
296 if (typeid(*shape) == typeid(RBFShape<dim>))
297 {
298 std::static_pointer_cast<RBFShape<dim>>(shape)->update_precalculations(
299 updated_dof_handler, precalculations);
300 }
301 else if (typeid(*shape) == typeid(CompositeShape<dim>))
302 {
303 std::static_pointer_cast<CompositeShape<dim>>(shape)
304 ->update_precalculations(updated_dof_handler, precalculations);
305 }
