9template <int dim, typename PropertiesIndex>
10void
11DEMContactManager<dim, PropertiesIndex>::execute_cell_neighbors_search(
12 const parallel::distributed::Triangulation<dim> &triangulation,
13 const typename dem_data_structures<dim>::periodic_boundaries_cells_info
14 periodic_boundaries_cells_information)
15{
16 // Get action manager
17 auto *action_manager = DEMActionManager::get_action_manager();
18
19 // Find cell neighbors
20 find_cell_neighbors<dim>(triangulation,
21 cells_local_neighbor_list,
22 cells_ghost_neighbor_list);
23
24 // Find cell periodic neighbors
25 if (action_manager->check_periodic_boundaries_enabled())
26 {
27 find_cell_periodic_neighbors<dim>(
28 triangulation,
29 periodic_boundaries_cells_information,
30 cells_local_periodic_neighbor_list,
31 cells_ghost_periodic_neighbor_list,
32 cells_ghost_local_periodic_neighbor_list);
33 }
34
35 // Get total (with repetition) neighbors list for floating mesh.
36 if (action_manager->check_solid_objects_enabled())
37 {
38 find_full_cell_neighbors<dim>(triangulation, total_neighbor_list);
39 }
40}

42template <int dim, typename PropertiesIndex>
43void
44DEMContactManager<dim, PropertiesIndex>::update_contacts()
45{
46 // Update particle-particle contacts in local_adjacent_particles of fine
47 // search step with local_contact_pair_candidates
48 update_fine_search_candidates<
49 dim,
50 typename dem_data_structures<dim>::adjacent_particle_pairs,
51 typename dem_data_structures<dim>::particle_particle_candidates,
52 ContactType::local_particle_particle>(local_adjacent_particles,
53 local_contact_pair_candidates);
54
55 // Update particle-particle contacts in global_adjacent_particles of fine
56 // search step with global_contact_pair_candidates
57 update_fine_search_candidates<
58 dim,
59 typename dem_data_structures<dim>::adjacent_particle_pairs,
60 typename dem_data_structures<dim>::particle_particle_candidates,
61 ContactType::ghost_particle_particle>(ghost_adjacent_particles,
62 ghost_contact_pair_candidates);
63
64 if (DEMActionManager::get_action_manager()
65 ->check_periodic_boundaries_enabled())
66 {
67 // Update periodic particle-particle contacts in
68 // local_local_periodic_adjacent_particles of fine search step with
69 // local_contact_pair_periodic_candidates
70 update_fine_search_candidates<
71 dim,
72 typename dem_data_structures<dim>::adjacent_particle_pairs,
73 typename dem_data_structures<dim>::particle_particle_candidates,
74 ContactType::local_periodic_particle_particle>(
75 local_local_periodic_adjacent_particles,
76 local_contact_pair_periodic_candidates);
77
78 // Update periodic particle-particle contacts in
79 // local_ghost_periodic_adjacent_particles of fine search step with
80 // ghost_contact_pair_periodic_candidates
81 update_fine_search_candidates<
82 dim,
83 typename dem_data_structures<dim>::adjacent_particle_pairs,
84 typename dem_data_structures<dim>::particle_particle_candidates,
85 ContactType::ghost_periodic_particle_particle>(
86 local_ghost_periodic_adjacent_particles,
87 ghost_contact_pair_periodic_candidates);
88
89 // Update periodic particle-particle contacts in
90 // ghost_local_periodic_adjacent_particles of fine search step with
91 // ghost_local_contact_pair_periodic_candidates
92 update_fine_search_candidates<
93 dim,
94 typename dem_data_structures<dim>::adjacent_particle_pairs,
95 typename dem_data_structures<dim>::particle_particle_candidates,
96 ContactType::ghost_local_periodic_particle_particle>(
97 ghost_local_periodic_adjacent_particles,
98 ghost_local_contact_pair_periodic_candidates);
99 }
100
101 // Update particle-wall contacts in particle_wall_pairs_in_contact of fine
102 // search step with particle_wall_contact_candidates
103 update_fine_search_candidates<
104 dim,
105 typename dem_data_structures<dim>::particle_wall_in_contact,
106 typename dem_data_structures<dim>::particle_wall_candidates,
107 ContactType::particle_wall>(particle_wall_in_contact,
108 particle_wall_candidates);
109
110 // Update particle-floating wall contacts in particle_floating_wall_in_contact
111 // of fine search step with particle_floating_wall_contact_candidates
112 update_fine_search_candidates<
113 dim,
114 typename dem_data_structures<dim>::particle_wall_in_contact,
115 typename dem_data_structures<dim>::particle_floating_wall_candidates,
116 ContactType::particle_floating_wall>(particle_floating_wall_in_contact,
117 particle_floating_wall_candidates);
118
119 // Update particle-floating mesh contacts in particle_floating_mesh_in_contact
120 // of fine search step with particle_floating_mesh_contact_candidates
121 for (unsigned int solid_counter = 0;
122 solid_counter < particle_floating_mesh_in_contact.size();
123 ++solid_counter)
124 {
125 update_fine_search_candidates<
126 dim,
127 typename dem_data_structures<
128 dim>::particle_floating_wall_from_mesh_in_contact,
129 typename dem_data_structures<
130 dim>::particle_floating_wall_from_mesh_candidates,
131 ContactType::particle_floating_mesh>(
132 particle_floating_mesh_in_contact[solid_counter],
133 particle_floating_mesh_candidates[solid_counter]);
134 }

141{

147 dim,
148 typename dem_data_structures<dim>::adjacent_particle_pairs,
149 ContactType::local_particle_particle>(local_adjacent_particles,
150 particle_container);
151
152 // Update contact containers for ghost particle-particle pairs in contact
153 update_contact_container_iterators<
154 dim,
155 typename dem_data_structures<dim>::adjacent_particle_pairs,
156 ContactType::ghost_particle_particle>(ghost_adjacent_particles,
157 particle_container);
158
159 if (DEMActionManager::get_action_manager()

161 {
162 // Update contact containers for local-local periodic particle-particle
163 // pairs in contact

165 dim,
166 typename dem_data_structures<dim>::adjacent_particle_pairs,
167 ContactType::local_periodic_particle_particle>(
168 local_local_periodic_adjacent_particles, particle_container);
169
170 // Update contact containers for local-ghost periodic particle-particle
171 // pairs in contact
172 update_contact_container_iterators<
173 dim,
174 typename dem_data_structures<dim>::adjacent_particle_pairs,
175 ContactType::ghost_periodic_particle_particle>(
176 local_ghost_periodic_adjacent_particles, particle_container);
177
178 // Update contact containers for ghost-local periodic particle-particle
179 // pairs in contact
180 update_contact_container_iterators<
181 dim,
182 typename dem_data_structures<dim>::adjacent_particle_pairs,
183 ContactType::ghost_local_periodic_particle_particle>(
184 ghost_local_periodic_adjacent_particles, particle_container);
185 }
186
187 // Update contact containers for particle-wall pairs in contact
188 update_contact_container_iterators<
189 dim,
190 typename dem_data_structures<dim>::particle_wall_in_contact,
191 ContactType::particle_wall>(particle_wall_in_contact, particle_container);
192
193 // Update contact containers for particle-floating wall pairs in contact

195 dim,

206 update_contact_container_iterators<
207 dim,
208 typename dem_data_structures<

216 dim,
217 typename dem_data_structures<dim>::particle_line_in_contact,

222 dim,
223 typename dem_data_structures<dim>::particle_point_in_contact,

228template <int dim, typename PropertiesIndex>
229void
230DEMContactManager<dim, PropertiesIndex>::execute_particle_particle_broad_search(
231 dealii::Particles::ParticleHandler<dim> &particle_handler,
232 const AdaptiveSparseContacts<dim, PropertiesIndex> &sparse_contacts_object)
233{
234 auto *action_manager = DEMActionManager::get_action_manager();
235
236 // Check if sparse contacts are enabled to use proper broad search functions
237 // The first broad search is the default one for sparse contacts
238 if (action_manager->use_default_broad_search_functions())
239 {
240 find_particle_particle_contact_pairs<dim>(particle_handler,
241 cells_local_neighbor_list,
242 cells_ghost_neighbor_list,
243 local_contact_pair_candidates,
244 ghost_contact_pair_candidates);
245
246 if (action_manager->check_periodic_boundaries_enabled())
247 {
248 find_particle_particle_periodic_contact_pairs<dim>(
249 particle_handler,
250 cells_local_periodic_neighbor_list,
251 cells_ghost_periodic_neighbor_list,
252 cells_ghost_local_periodic_neighbor_list,
253 local_contact_pair_periodic_candidates,
254 ghost_contact_pair_periodic_candidates,
255 ghost_local_contact_pair_periodic_candidates);
256 }
257 }
258 else
259 {
260 find_particle_particle_contact_pairs<dim>(particle_handler,
261 cells_local_neighbor_list,
262 cells_ghost_neighbor_list,
263 local_contact_pair_candidates,
264 ghost_contact_pair_candidates,
265 sparse_contacts_object);
266
267 if (action_manager->check_periodic_boundaries_enabled())
268 {
269 find_particle_particle_periodic_contact_pairs<dim>(
270 particle_handler,
271 cells_local_periodic_neighbor_list,
272 cells_ghost_periodic_neighbor_list,
273 cells_ghost_local_periodic_neighbor_list,
274 local_contact_pair_periodic_candidates,
275 ghost_contact_pair_periodic_candidates,
276 ghost_local_contact_pair_periodic_candidates,
277 sparse_contacts_object);
278 }
279 }

282template <int dim, typename PropertiesIndex>
283void
284DEMContactManager<dim, PropertiesIndex>::execute_particle_wall_broad_search(
285 const Particles::ParticleHandler<dim> &particle_handler,
286 BoundaryCellsInformation<dim> &boundary_cell_object,
287 const typename dem_data_structures<dim>::solid_surfaces_mesh_information
288 solid_surfaces_mesh_info,
289 const Parameters::Lagrangian::FloatingWalls<dim> &floating_walls,
290 const double simulation_time,
291 const AdaptiveSparseContacts<dim, PropertiesIndex> &sparse_contacts_object)
292{
293 auto *action_manager = DEMActionManager::get_action_manager();
294
295 if (action_manager->use_default_broad_search_functions())
296 {
297 // Particle-wall contact candidates
298 find_particle_wall_contact_pairs<dim>(
299 boundary_cell_object.get_boundary_cells_information(),
300 particle_handler,
301 particle_wall_candidates);
302
303 // Particle-floating wall contact pairs
304 if (floating_walls.floating_walls_number > 0)
305 {
306 find_particle_floating_wall_contact_pairs<dim>(
307 boundary_cell_object.get_boundary_cells_with_floating_walls(),
308 particle_handler,
309 floating_walls,
310 simulation_time,
311 particle_floating_wall_candidates);
312 }
313
314 // Particle-floating mesh broad search
315 if (action_manager->check_solid_objects_enabled())
316 {
317 particle_solid_surfaces_contact_search<dim>(
318 solid_surfaces_mesh_info,
319 particle_handler,
320 particle_floating_mesh_candidates,
321 total_neighbor_list);
322 }
323
324 find_particle_point_contact_pairs<dim>(
325 particle_handler,
326 boundary_cell_object.get_boundary_cells_with_points(),
327 particle_point_candidates);
328
329 if constexpr (dim == 3)
330 {
331 find_particle_line_contact_pairs<dim>(
332 particle_handler,
333 boundary_cell_object.get_boundary_cells_with_lines(),
334 particle_line_candidates);
335 }
336 }
337 else
338 {
339 // Particle-wall contact candidates
340 find_particle_wall_contact_pairs<dim>(
341 boundary_cell_object.get_boundary_cells_information(),
342 particle_handler,
343 particle_wall_candidates,
344 sparse_contacts_object);
345
346 // Particle-floating wall contact pairs
347 if (floating_walls.floating_walls_number > 0)
348 {
349 find_particle_floating_wall_contact_pairs<dim>(
350 boundary_cell_object.get_boundary_cells_with_floating_walls(),
351 particle_handler,
352 floating_walls,
353 simulation_time,
354 particle_floating_wall_candidates,
355 sparse_contacts_object);
356 }
357
358 // Particle-floating mesh broad search
359 if (action_manager->check_solid_objects_enabled())
360 {
361 particle_solid_surfaces_contact_search<dim>(
362 solid_surfaces_mesh_info,
363 particle_handler,
364 particle_floating_mesh_candidates,
365 total_neighbor_list,
366 sparse_contacts_object);
367 }
368
369 find_particle_point_contact_pairs<dim>(
370 particle_handler,
371 boundary_cell_object.get_boundary_cells_with_points(),
372 particle_point_candidates,
373 sparse_contacts_object);
374
375 if constexpr (dim == 3)
376 {
377 find_particle_line_contact_pairs<dim>(
378 particle_handler,
379 boundary_cell_object.get_boundary_cells_with_lines(),
380 particle_line_candidates,
381 sparse_contacts_object);
382 }
383 }

461 neighborhood_threshold,
462 particle_points_in_contact);
463
464 if constexpr (dim == 3)

466 particle_line_fine_search<dim, PropertiesIndex>(
467 particle_line_candidates,
468 neighborhood_threshold,
469 particle_lines_in_contact);
