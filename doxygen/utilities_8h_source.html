249 /**
250  * @brief Calculate the equivalent properties for a given phase. Method called
251  * in quadrature points loops in VOF simulations.
252  *
253  * @param phase Phase value for the given quadrature point
254  *
255  * @param property0 Property value for the fluid with index 0 (fluid for phase = 0)
256  *
257  * @param property1 Property value for the fluid with index 1 (fluid for phase = 1)
258  */
259 inline double
260 calculate_point_property(const double phase,
261  const double property0,
262  const double property1)
263 {
264  double property_eq = phase * property1 + (1 - phase) * property0;
265 
266  // Limit parameters value (patch)
267  // TODO see if necessary after compression term is added in the
268  // VOF advection equation
269  const double property_min = std::min(property0, property1);
270  const double property_max = std::max(property0, property1);
271  if (property_eq < property_min)
272  {
273  property_eq = property_min;
274  }
275  if (property_eq > property_max)
276  {
277  property_eq = property_max;
278  }
279 
280  return property_eq;
281 }

284  * @brief Used in the calculate_properties_ch to retrieve the sign of the phase parameter
285  * @tparam T val argument's type

287  * @return an integer -1 or 1 depending of the sign of the phase parameter

313 /**
314  * @brief Calculate the equivalent properties for a given phase. Method called
315  * in quadrature points loops in Cahn-Hilliard simulations.
316  *
317  * @param phase_cahn_hilliard Phase value for the given quadrature point
318  *
319  * @param property0 Property value for the fluid with index 0 (fluid for phase = -1)
320  *
321  * @param property1 Property value for the fluid with index 1 (fluid for phase = 1)
322  */
323 inline double
324 calculate_point_property_cahn_hilliard(const double phase_cahn_hilliard,
325  const double property0,
326  const double property1)
327 {
328  double phase = phase_cahn_hilliard;
329 
330  if (std::abs(phase_cahn_hilliard) > 1)
331  {
332  phase = sgn(phase_cahn_hilliard);
333  }
334 
335  double property_avg = (property0 + property1) * 0.5;
336  double property_diff = (property0 - property1) * 0.5;
337 
338  double property_eq = phase * property_diff + property_avg;
339 
340  return property_eq;
341 }

485 /**
486  * @brief Return the tensor corresponding to the @p value_string. If the
487  * dimension correspondence of the @p value_string is not equivalent to
488  * @p spacedim (either 2 or 3), an exception will be thrown. The delimiter
489  * separating the elements of the @p value_string is a comma (",").
490  *
491  * @remark The function can be use to construct Point<spacedim> objects.
492  *
493  * @tparam spacedim Number of spatial dimensions (2D or 3D).
494  *
495  * @param[in] value_string A string in the parameter file corresponding to a
496  * given tensor.
497  *
498  * @return A Tensor<1,spacedim> corresponding to the @p value_string in the
499  * parameter file.
500  */
501 template <int spacedim>
502 inline Tensor<1, spacedim>
503 value_string_to_tensor(const std::string &value_string)
504 {
505  std::vector<std::string> vector_of_string(
506  Utilities::split_string_list(value_string));
507  std::vector<double> vector_of_double =
508  Utilities::string_to_double(vector_of_string);
509 
510  AssertThrow(vector_of_double.size() == 3 || vector_of_double.size() == 2,
511  ExcMessage("Invalid string: " + value_string +
512  ". This should be a two or three dimensional vector "
513  "or point."));
514 
515  Tensor<1, spacedim> output_tensor;
516  for (unsigned int i = 0; i < spacedim; ++i)
517  output_tensor[i] = vector_of_double[i];
518 
519  return output_tensor;
520 }

522 /**
523  * @brief Return the tensor corresponding to the @p value_string_0, but it can
524  * also allow the usage of deprecated parameters that used to be 3 individual
525  * entries instead of a list of values.
526  * In the case of a single entry declaration, the delimiter separating the
527  * elements of the @p value_string_0 is a comma (",").
528  *
529  * @remark The function can be use to construct Point<spacedim> objects.
530  *
531  * @tparam spacedim Number of spatial dimensions (2D or 3D).
532  *
533  * @param[in] value_string_0 A string in the parameter file corresponding to the
534  * first component of the tensor or to the tensor itself.
535  * @param[in] value_1 A double in the parameter file corresponding to the
536  * second component of the tensor.
537  * @param[in] value_2 A double in the parameter file corresponding to the
538  * third component of the tensor. Only specify if @p spacedim = 3.
539  *
540  * @return A Tensor<1,spacedim> corresponding to the input parameters in the
541  * parameter file.
542  */
543 template <int spacedim>
544 inline Tensor<1, spacedim>
545 value_string_to_tensor(const std::string &value_string_0,
546  const double &value_1,
547  const double &value_2 = 0)
548 {
549  std::vector<std::string> vector_of_string(
550  Utilities::split_string_list(value_string_0));
551  Tensor<1, spacedim> output_tensor;
552 
553  // The used parameter is a list of values
554  if (vector_of_string.size() > 1)
555  {
556  std::vector<double> vector_of_double =
557  Utilities::string_to_double(vector_of_string);
558  AssertThrow(vector_of_double.size() == 3 || vector_of_double.size() == 2,
559  ExcMessage(
560  "Invalid string: " + value_string_0 +
561  ". This should be a two or three dimensional vector "
562  "or point."));
563  for (unsigned int i = 0; i < vector_of_double.size(); ++i)
564  output_tensor[i] = vector_of_double[i];
565  }
566  else // Depreciated individual entries
567  {
568  // Since the first parameter is the alias of the new parameter,
569  // the value of the first parameter is obtained for its entry
570  output_tensor[0] = Utilities::string_to_double(value_string_0);
571  output_tensor[1] = value_1;
572  if constexpr (spacedim == 3)
573  output_tensor[2] = value_2;
574  }
575  return output_tensor;
576 }

578 /**
579  * @brief Computes equivalent cell diameter by comparing the area to a disk (2D)
580  * or the volume to a sphere (3D).
581  *
582  * @tparam dim Number of spatial dimensions (2D or 3D).
583  *
584  * @param[in] cell_measure Area (2D) or volume (3D) of the cell.
585  *
586  * @param[in] fe_degree Polynomial degree of the shape function.
587  *
588  * @return Cell diameter value.
589  */
590 template <int dim>
591 inline double
592 compute_cell_diameter(const double cell_measure, const unsigned int fe_degree)
593 {
594  double h;
595  if constexpr (dim == 2)
596  h = std::sqrt(4. * cell_measure / numbers::PI) / fe_degree;
597  else if constexpr (dim == 3)
598  h = std::cbrt(6. * cell_measure / numbers::PI) / fe_degree;
599  else
600  Assert(
601  false,
602  ExcMessage(std::string(
603  "'dim' should have a value of either 2 or 3. Only 2D and 3D simulations "
604  "are supported.")));
605  return h;
606 }

629 /**
630  * @brief Calculate the penalty factor for the SIPG method for the case when two cells are of different sizes. A harmonic mean is used to calculate an average cell extent.
631  *
632  * @param[in] fe_degree Finite element degree
633  * @param[in] cell_extent_here Distance of the here cell. The ratio between the
634  * volume of the cell and the area of the face is generally used as a metric for
635  * this distance.
636  * @param[in] cell_extent_there Distance of the there cell. The ratio between
637  * the volume of the cell and the area of the face is generally used as a metric
638  * for this distance.
639  * @return double
640  */
641 
642 inline double
643 get_penalty_factor(const unsigned int fe_degree,
644  const double cell_extent_here,
645  const double cell_extent_there)
646 {
647  return (fe_degree + 1) * (fe_degree + 1.) * 0.5 *
648  (1. / cell_extent_here + 1. / cell_extent_there);
649 }
