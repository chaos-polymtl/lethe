29/**
30 * @class statistics
31 * @brief A simple container for storing basic statistical values.
32 *
33 * This class holds the minimum, maximum, total, and average values
34 * of a dataset. It is initialized with zeros by default. This small class
35 * allows us to agglomerate the statistics instead of returning tuples. For
36 * example, this class is used to store the kinetic energy of the particles in
37 * the DEM model.
38 */
39
40class statistics
41{
42 /**
43 * @brief Default constructor.
44 *
45 * Initializes all statistical values (min, max, total, average) to 0.
46 */
47public:
48 statistics()
49 : min(0)
50 , max(0)
51 , total(0)
52 , average(0)
53 {}
54
55
56 /// The minimum value in the dataset.
57 double min;
58
59 /// The maximum value in the dataset.
60 double max;
61
62 /// The sum of all values in the dataset.
63 double total;
64
65 /// The average value of the dataset.
66 double average;
67};

281/**
282 * @brief Calculate the equivalent properties for a given phase. Method called
283 * in quadrature points loops in VOF simulations.
284 *
285 * @param phase Phase value for the given quadrature point
286 *
287 * @param property0 Property value for the fluid with index 0 (fluid for phase = 0)
288 *
289 * @param property1 Property value for the fluid with index 1 (fluid for phase = 1)
290 */
291inline double
292calculate_point_property(const double phase,
293 const double property0,
294 const double property1)
295{
296 double property_eq = phase * property1 + (1 - phase) * property0;
297
298 // Limit parameters value (patch)
299 // TODO see if necessary after compression term is added in the
300 // VOF advection equation
301 const double property_min = std::min(property0, property1);
302 const double property_max = std::max(property0, property1);
303 if (property_eq < property_min)
304 {
305 property_eq = property_min;
306 }
307 if (property_eq > property_max)
308 {
309 property_eq = property_max;
310 }
311
312 return property_eq;
313}

315/**
316 * @brief Retrieve the sign of a value.
317 *
318 * Used in calculate_point_property_cahn_hilliard to determine the sign of the
319 * phase parameter.
320 *
321 * @tparam T The argument's type (must support comparison with zero).
322 * @param[in] val Value of the variable for which the sign is evaluated.
323 * @retval -1 if the value is negative.
324 * @retval +1 if the value is positive.
325 * @retval 0 if the value is zero.
326 */
327template <typename T>
328[[nodiscard]] constexpr int
329sgn(const T val) noexcept
330{
331 return (static_cast<T>(0) < val) - (val < static_cast<T>(0));
332}

335/**
336 * @brief Clip a property between a lower and a higher value
337 * @tparam T argument's type
338 * @param[in,out] lower Lower admissible value
339 * @param[in,out] upper Upper admissible value
340 * @param[in,out] n
341 * @return Clipped variable that is not below the lower limit and not above the upper limit
342 *
343 */
344template <typename T>
345T
346clip(const T &n, const T &lower, const T &upper)
347{
348 return std::max(lower, std::min(n, upper));
349}

351/**
352 * @brief Calculate the equivalent property for a given phase.
353 *
354 * This method is called in quadrature point loops in Cahn-Hilliard simulations.
355 * It interpolates between two fluid properties based on the phase value.
356 *
357 * @param[in] phase_cahn_hilliard Phase value for the given quadrature point.
358 * @param[in] property0 Property value for the fluid with index 0 (fluid for
359 * phase = -1).
360 * @param[in] property1 Property value for the fluid with index 1 (fluid for
361 * phase = 1).
362 * @return double Equivalent property for the given phase.
363 */
364[[nodiscard]] inline double
365calculate_point_property_cahn_hilliard(const double phase_cahn_hilliard,
366 const double property0,
367 const double property1)
368{
369 double phase = phase_cahn_hilliard;
370
371 if (std::abs(phase_cahn_hilliard) > 1)
372 {
373 phase = sgn(phase_cahn_hilliard);
374 }
375
376 double property_avg = (property0 + property1) * 0.5;
377 double property_diff = (property0 - property1) * 0.5;
378
379 double property_eq = phase * property_diff + property_avg;
380
381 return property_eq;
382}

574/**
575 * @brief Return the tensor corresponding to the @p value_string. If the
576 * dimension correspondence of the @p value_string is not equivalent to
577 * @p spacedim (either 2 or 3), an exception will be thrown. The delimiter
578 * separating the elements of the @p value_string is a comma (",").
579 *
580 * @remark The function is used to construct Point<spacedim> objects.
581 *
582 * @tparam spacedim Number of spatial dimensions (2D or 3D).
583 *
584 * @param[in] value_string A string in the parameter file corresponding to a
585 * given tensor.
586 *
587 * @return A Tensor<1,spacedim> corresponding to the @p value_string in the
588 * parameter file.
589 */
590template <int spacedim>
591inline Tensor<1, spacedim>
592value_string_to_tensor(const std::string &value_string)
593{
594 std::vector<std::string> vector_of_string(
595 Utilities::split_string_list(value_string));
596 std::vector<double> vector_of_double =
597 Utilities::string_to_double(vector_of_string);
598
599 Assert(vector_of_double.size() == 3 || vector_of_double.size() == 2,
600 ExcMessage("Invalid string: " + value_string +
601 ". This should be a two or three dimensional vector "
602 "or point."));
603
604 Tensor<1, spacedim> output_tensor;
605 for (unsigned int i = 0; i < spacedim; ++i)
606 output_tensor[i] = vector_of_double[i];
607
608 return output_tensor;
609}

611/**
612 * @brief Return the tensor corresponding to the @p value_string_0, but it can
613 * also allow the usage of deprecated parameters that used to be 3 individual
614 * entries instead of a list of values.
615 * In the case of a single entry declaration, the delimiter separating the
616 * elements of the @p value_string_0 is a comma (",").
617 *
618 * @remark The function is used to construct Point<spacedim> objects.
619 *
620 * @tparam spacedim Number of spatial dimensions (2D or 3D).
621 *
622 * @param[in] value_string_0 A string in the parameter file corresponding to the
623 * first component of the tensor or to the tensor itself.
624 * @param[in] value_1 A double in the parameter file corresponding to the
625 * second component of the tensor.
626 * @param[in] value_2 A double in the parameter file corresponding to the
627 * third component of the tensor. Only specify if @p spacedim = 3.
628 *
629 * @return A Tensor<1,spacedim> corresponding to the input parameters in the
630 * parameter file.
631 */
632template <int spacedim>
633inline Tensor<1, spacedim>
634value_string_to_tensor(const std::string &value_string_0,
635 const double &value_1,
636 const double &value_2 = 0)
637{
638 std::vector<std::string> vector_of_string(
639 Utilities::split_string_list(value_string_0));
640 Tensor<1, spacedim> output_tensor;
641
642 // The used parameter is a list of values
643 if (vector_of_string.size() > 1)
644 {
645 std::vector<double> vector_of_double =
646 Utilities::string_to_double(vector_of_string);
647 Assert(vector_of_double.size() == 3 || vector_of_double.size() == 2,
648 ExcMessage("Invalid string: " + value_string_0 +
649 ". This should be a two or three dimensional vector "
650 "or point."));
651 for (unsigned int i = 0; i < vector_of_double.size(); ++i)
652 output_tensor[i] = vector_of_double[i];
653 }
654 else // Depreciated individual entries
655 {
656 // Since the first parameter is the alias of the new parameter,
657 // the value of the first parameter is obtained for its entry
658 output_tensor[0] = Utilities::string_to_double(value_string_0);
659 output_tensor[1] = value_1;
660 if constexpr (spacedim == 3)
661 output_tensor[2] = value_2;
662 }
663 return output_tensor;
664}

666/**
667 * @brief Computes equivalent cell diameter by comparing the area to a disk (2D)
668 * or the volume to a sphere (3D).
669 *
670 * @tparam dim Number of spatial dimensions (2D or 3D).
671 *
672 * @param[in] cell_measure Area (2D) or volume (3D) of the cell.
673 *
674 * @param[in] fe_degree Polynomial degree of the shape function.
675 *
676 * @return Cell diameter value.
677 */
678template <int dim>
679inline double
680compute_cell_diameter(const double cell_measure, const unsigned int fe_degree)
681{
682 double h;
683 if constexpr (dim == 2)
684 h = std::sqrt(4. * cell_measure / numbers::PI) / fe_degree;
685 else if constexpr (dim == 3)
686 h = std::cbrt(6. * cell_measure / numbers::PI) / fe_degree;
687 else
688 Assert(
689 false,
690 ExcMessage(std::string(
691 "'dim' should have a value of either 2 or 3. Only 2D and 3D simulations "
692 "are supported.")));
693 return h;
694}

696/**
697 * @brief Computes the area (2D) or volume (3D) of the cell by integrating 1
698 * over the cell, by summing JxW values (quadrature weights) returned by the
699 * FEValues object.
700 *
701 * @tparam dim Number of spatial dimensions (2D or 3D).
702 *
703 * @param[in] JxW_values Vector of mapped quadrature weights.
704 *
705 * @return Area (2D) volume (3D) of the cell.
706 */
707inline double
708compute_cell_measure_with_JxW(const std::vector<double> &JxW_values)
709{
710 double cell_measure = 0;
711 for (const double &JxW : JxW_values)
712 cell_measure += JxW;
713 return cell_measure;
714}

718 * of a disk (2D) or sphere (3D) of equivalent area (2D) or volume (3D).
719 *
720 * @tparam dim Number of spatial dimensions (2D or 3D).
721 *
722 * @param[in] mapping Interface for transformation from the reference (unit)
723 * cell to the real cell.
724 *
725 * @param[in] dof_handler Describes the layout of DoFs and the type of FE.
726 *
727 * @param[in] cell_quadrature Local cell quadrature information.
728 *
729 * @param[in] mpi_communicator Allows communication between cores.
730 *
731 * @return Smallest cell size value.
732 */
733template <int dim>
734inline double
735identify_minimum_cell_size(const Mapping<dim> &mapping,
736 const DoFHandler<dim> &dof_handler,
737 const Quadrature<dim> &cell_quadrature,
738 const MPI_Comm &mpi_communicator)
739{
740 // Initialize FEValues for interface algebraic reinitialization
741 FEValues<dim> fe_values(mapping,
742 dof_handler.get_fe(),
743 cell_quadrature,
744 update_JxW_values);
745
746 // Initialize cell diameter
747 double h = DBL_MAX;
748
749 // Element degree
750 const double degree = double(fe_values.get_fe().degree);
751
752 for (const auto &cell : dof_handler.active_cell_iterators())
753 {
754 if (cell->is_locally_owned())
755 {
756 fe_values.reinit(cell);
757
758 // Compute cell diameter
759 double cell_measure =
760 compute_cell_measure_with_JxW(fe_values.get_JxW_values());
761 double h_local = compute_cell_diameter<dim>(cell_measure, degree);
762
763 // Update cell diameter to minimum value
764 h = std::min(h, h_local);
765 }
766 }
767
768 // Get the minimum between all processes
769 h = Utilities::MPI::min(h, mpi_communicator);
770
771 return h;
772}

774/**
775 * @brief Calculate the penalty factor for the SIPG method for the case when two
776 * cells are of different sizes. A harmonic mean is used to calculate an average
777 * cell extent.
778 *
779 * @param[in] fe_degree Finite element degree.
780 * @param[in] cell_extent_here Distance of the here cell. The ratio between the
781 * volume of the cell and the area of the face is generally used as a metric for
782 * this distance.
783 * @param[in] cell_extent_there Distance of the there cell. The ratio between
784 * the volume of the cell and the area of the face is generally used as a metric
785 * for this distance.
786 * @return double
787 */
788
789inline double
790get_penalty_factor(const unsigned int fe_degree,
791 const double cell_extent_here,
792 const double cell_extent_there)
793{
794 return (fe_degree + 1) * (fe_degree + 1.) * 0.5 *
795 (1. / cell_extent_here + 1. / cell_extent_there);
796}
