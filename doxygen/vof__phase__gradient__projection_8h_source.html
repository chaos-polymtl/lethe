35template <int dim>
36class VOFPhaseGradientProjection : public VOFLinearSubequationsSolver<dim>
37{
38public:
39 /**
40 * @brief Constructor for the L2 projection of the VOF phase fraction gradient
41 * (pfg).
42 *
43 * @param[in] p_simulation_parameters Simulation parameters.
44 *
45 * @param[in] p_pcout Parallel cout used to print the information.
46 *
47 * @param[in] p_triangulation Distributed mesh information.
48 *
49 * @param[in,out] p_subequations_interface Subequations interface object used
50 * to get information from other subequations and store information from the
51 * current one.
52 */
53 VOFPhaseGradientProjection(
54 const SimulationParameters<dim> &p_simulation_parameters,
55 const ConditionalOStream &p_pcout,
56 std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
57 p_triangulation,
58 VOFSubequationsInterface<dim> &p_subequations_interface)
59 : VOFLinearSubequationsSolver<dim>(
60 VOFSubequationsID::phase_gradient_projection,
61 p_simulation_parameters,
62 ((p_simulation_parameters.multiphysics.vof_parameters
63 .surface_tension_force.verbosity != Parameters::Verbosity::quiet) ||
64 ((p_simulation_parameters.multiphysics.vof_parameters
65 .regularization_method.algebraic_interface_reinitialization
66 .enable) &&
67 (p_simulation_parameters.multiphysics.vof_parameters
68 .regularization_method.verbosity !=
69 Parameters::Verbosity::quiet))) ?
70 Parameters::Verbosity::verbose :
71 Parameters::Verbosity::quiet, // Set to verbose if surface tension
72 // verbosity is enabled or if algebraic
73 // interface reinitialization is enabled
74 // and set to verbose
75 p_pcout,
76 p_triangulation,
77 p_subequations_interface)
78 {
79 if (this->simulation_parameters.mesh.simplex)
80 {
81 // For simplex meshes
82 const FE_SimplexP<dim> subequation_fe(
83 this->simulation_parameters.fem_parameters.VOF_order);
84 this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
85 this->mapping = std::make_shared<MappingFE<dim>>(*this->fe);
86
87 this->cell_quadrature =
88 std::make_shared<QGaussSimplex<dim>>(this->fe->degree + 1);
89 }
90 else
91 {
92 // Usual case, for quad/hex meshes
93 const FE_Q<dim> subequation_fe(
94 this->simulation_parameters.fem_parameters.VOF_order);
95 this->fe = std::make_shared<FESystem<dim>>(subequation_fe, dim);
96 this->mapping = std::make_shared<MappingQ<dim>>(this->fe->degree);
97
98 this->cell_quadrature =
99 std::make_shared<QGauss<dim>>(this->fe->degree + 1);
100 }
101 }
102
103 /**
104 * @brief Default destructor.
105 */
106 ~VOFPhaseGradientProjection() = default;

116 * @brief Check if a new VOF filtered solution has been set.
117 */
118 void
119 check_dependencies_validity() override;
120};
