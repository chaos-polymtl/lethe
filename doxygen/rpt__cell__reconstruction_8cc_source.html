31template <int dim>
32void
33RPTCellReconstruction<dim>::execute_cell_reconstruction()
34{
35 // Assign detector position & create detectors
36 detectors = assign_detector_positions<dim>(detector_parameters);
37
38 // Read reconstruction (experimental) counts of unknown particle positions
39 reconstruction_counts = read_detectors_counts<dim>(
40 reconstruction_parameters.reconstruction_counts_file, detectors.size());
41
42 // Read known positions if analyse positions if enable
43 if (reconstruction_parameters.analyse_positions)
44 {
45 known_positions =
46 read_positions<dim>(reconstruction_parameters.known_positions_file);
47 AssertThrow(
48 known_positions.size() == reconstruction_counts.size(),
49 ExcMessage(
50 "Reconstruction counts and known positions files do not have data for the same number of positions."));
51 }
52
53 // Create a grid for the reactor vessel
54 attach_grid_to_triangulation<dim>(
55 triangulation, parameters, reconstruction_parameters.reactor_refinement);
56
57 // Calculate counts at vertices of the coarse mesh
58 set_coarse_mesh_counts();
59 find_all_positions();
60
61 // Export positions in file
62 export_positions();
63
64 // Show information about cells and vertices
65 if (parameters.verbosity == Parameters::Verbosity::verbose)
66 {
67 std::cout << std::endl;
68 std::cout << "Number of active cells : " << triangulation.n_active_cells()
69 << std::endl;
70 std::cout << "Number of vertices in the mesh : "
71 << triangulation.n_vertices() << std::endl;
72 std::cout << "Number of vertices with calculated counts : "
73 << map_vertices_index.size() << std::endl;
74 }
75
76 // Generate visualisation files
77 visualize_positions();
78
79 // Disable the output of time clock
80 if (!reconstruction_parameters.verbose_clock)
81 computing_timer.disable_output();
82}

107template <int dim>
108void
109RPTCellReconstruction<dim>::find_unknown_position(
110 std::vector<double> &particle_reconstruction_counts,
111 unsigned int id)
112{
113 int level = 0;
114 std::vector<int> cells_indexes, new_cells_indexes;
115 Point<dim> reconstruction_position;
116 double volume;
117 bool still_new_candidates = true;
118 std::string status;
119
120 // Find the first cell candidates from coarse mesh
121 cells_indexes = find_cells(level, particle_reconstruction_counts);
122
123 // Find the positions with the more refine candidate cell
124 while (level < reconstruction_parameters.reactor_refinement &&
125 still_new_candidates)
126 {
127 level++; // Start at coarse mesh level + 1
128
129 // Get vertices positions at current level and calculate their counts
130 find_vertices_positions(level, cells_indexes);
131 calculate_counts();
132
133 // Find cell candidate for the position
134 // (reconstruction counts are in the counts range of the cells for every
135 // detector)
136 new_cells_indexes =
137 find_cells(level, particle_reconstruction_counts, cells_indexes);
138
139 // If no refined cells are candidates at current level, research has to
140 // stop and need to go back to parent level
141 if (new_cells_indexes[0] == cells_indexes[0])
142 {
143 still_new_candidates = false;
144 level--;
145 status = "parent_cell"; // Cell isn't the highest refinement level
146 }
147
148 cells_indexes = new_cells_indexes;
149 }
150
151 // If many cells are candidate at the more refine level, the least squared
152 // method is used to get the best cell
153 int best_cell;
154 if (cells_indexes.size() > 1)
155 {
156 best_cell =
157 find_best_cell(level, particle_reconstruction_counts, cells_indexes);
158
159 // Update status
160 if (status.empty())
161 status = "cost_function";
162 else
163 status += "|cost_function";
164 }
165 else
166 best_cell = cells_indexes[0];
167
168 // Find the volume and the center position of the best cell and modify
169 // status when analyse positions is enable
170 for (const auto &cell : triangulation.cell_iterators_on_level(level))
171 {
172 if (cell->index() == best_cell)
173 {
174 volume = cell->measure();
175 reconstruction_position = cell->center(true);
176
177 // Verify if the best cell is the good cell and update status
178 if (reconstruction_parameters.analyse_positions)
179 {
180 if (cell->point_inside(known_positions[id]))
181 {
182 if (status.empty())
183 status = "right_cell";
184 else
185 status += "|right_cell";
186 }
187 else
188 {
189 if (status.empty())
190 status = "wrong_cell";
191 else
192 status += "|wrong_cell";
193 }
194 }
195 }
196 }
197
198 // Store information prior data exportation
199 reconstruction_positions.push_back(reconstruction_position);
200 cells_volumes.push_back(volume);
201 final_cell_level.push_back(level);
202 cell_status.push_back(status);
203}

205template <int dim>
206void
207RPTCellReconstruction<dim>::find_vertices_positions(
208 unsigned int level,
209 std::vector<int> parent_cell_indexes /* {-1} */)
210{
211 TimerOutput::Scope t(computing_timer, "finding_vertices_positions");
212
213 // Find the positions associated to the vertices of cell
214 for (const auto &cell : triangulation.cell_iterators_on_level(level))
215 {
216 // Stay false if
217 bool parent_is_candidate = false;
218
219 // Check if parent cell of current level cell is a candidate
220 // Stay false at coarsh mesh or if no candidate at refined level
221 if (cell->level() > reconstruction_parameters.coarse_mesh_level)
222 {
223 for (auto &id : parent_cell_indexes)
224 {
225 if (id == cell->parent_index())
226 parent_is_candidate = true;
227 }
228 }
229 else
230 parent_is_candidate = true; // Coarse mesh level : no parent
231
232 if (parent_is_candidate)
233 {
234 // Calculate counts at new vertices
235 for (unsigned int i = 0; i < cell->n_vertices(); i++)
236 {
237 unsigned int vertex_index = cell->vertex_index(i);
238 Point<dim> vertex_location = cell->vertex(i);
239
240 // Check if vertex already exist and store the position with the
241 // key as the vertex id
242 if (map_vertices_index.find(vertex_index) ==
243 map_vertices_index.end())
244 {
245 std::pair<Point<dim>, std::vector<double>> point_empty_vector(
246 vertex_location, {});
247 map_vertices_index[vertex_index] = point_empty_vector;
248 }
249 }
250 }
251 }
252}

254template <int dim>
255void
256RPTCellReconstruction<dim>::calculate_counts()
257{
258 TimerOutput::Scope t(computing_timer, "vertices_calculation_counts");
259
260 // Vector fir calculated counts for one particle position
261 std::vector<double> calculated_counts;
262
263 unsigned int i = 0;
264 for (auto &it : map_vertices_index)
265 {
266 // Check if counts are not already calculated at vertex
267 if (it.second.second.empty())
268 {
269 calculated_counts.clear();
270 RadioParticle<dim> particle(it.second.first, i);
271
272 for (unsigned int i_detector = 0; i_detector < detectors.size();
273 i_detector++)
274 {
275 // Create the particle-detector interaction object
276 ParticleDetectorInteractions<dim> particle_detector_interactions(
277 particle, detectors[i_detector], parameters);
278
279 // Calculate count and store in vector for the position
280 double count = particle_detector_interactions.calculate_count();
281 calculated_counts.push_back(count);
282 }
283 it.second.second = calculated_counts;
284 i++;
285 }
286 }
287}

289template <int dim>
290std::vector<int>
291RPTCellReconstruction<dim>::find_cells(
292 unsigned int level,
293 std::vector<double> &particle_reconstruction_counts,
294 std::vector<int> parent_cell_indexes /* {-1} */)
295{
296 TimerOutput::Scope t(computing_timer, "finding_cell");
297
298 double min, max;
299 std::vector<bool> candidate_detector(detectors.size());
300 std::vector<int> candidates;
301
302 // Check if parent cell of current level cell is a candidate
303 for (const auto &cell : triangulation.cell_iterators_on_level(level))
304 {
305 bool parent_is_candidate = false;
306 if (cell->level() > reconstruction_parameters.coarse_mesh_level)
307 {
308 for (auto &id : parent_cell_indexes)
309 {
310 if (id == cell->parent_index())
311 parent_is_candidate = true;
312 }
313 }
314 else
315 parent_is_candidate = true;
316
317 if (parent_is_candidate)
318 {
319 for (unsigned int j = 0; j < detectors.size(); j++)
320 {
321 std::vector<double> counts_vertices(8);
322 for (unsigned int i = 0; i < cell->n_vertices(); i++)
323 {
324 // Get counts associate to the current vertex and detector
325 unsigned int vertex_index = cell->vertex_index(i);
326 counts_vertices[i] =
327 map_vertices_index[vertex_index].second[j];
328 }
329 // Get the min/max counts values
330 min = *std::min_element(counts_vertices.begin(),
331 counts_vertices.end());
332 max = *std::max_element(counts_vertices.begin(),
333 counts_vertices.end());
334
335 // If counts is in the range, counts for the detector is true
336 if (particle_reconstruction_counts[j] >= min &&
337 particle_reconstruction_counts[j] <= max)
338 candidate_detector[j] = true;
339 else
340 candidate_detector[j] = false;
341 }
342
343 // If reconstruction counts are in ranges of cell for every detector,
344 // cell in candidate
345 if (std::all_of(candidate_detector.begin(),
346 candidate_detector.end(),
347 [](bool v) { return v; }))
348 candidates.push_back(cell->index());
349 }
350 }
351
352 // If there's no new candidates, parent cells are kept as candidate
353 if (candidates.empty())
354 candidates = parent_cell_indexes;
355
356 return candidates;
357}

359template <int dim>
360int
361RPTCellReconstruction<dim>::find_best_cell(
362 unsigned int level,
363 std::vector<double> &particle_reconstruction_counts,
364 std::vector<int> candidate)
365{
366 TimerOutput::Scope t(computing_timer, "finding_cell");
367
368 std::vector<std::pair<int, double>> cellid_error;
369 double error;
370 int best_cell;
371
372 // Use least squared error cell by cell for all counts per detector and
373 // all vertices of the cell
374 for (const auto &cell : triangulation.cell_iterators_on_level(level))
375 {
376 bool is_candidate = false;
377
378 // Check if candidate cell
379 for (auto &id : candidate)
380 {
381 if (id == cell->index())
382 is_candidate = true;
383 }
384 if (is_candidate)
385 {
386 error = 0;
387 for (unsigned int j = 0; j < detectors.size(); j++)
388 {
389 for (unsigned int i = 0; i < cell->n_vertices(); i++)
390 {
391 unsigned int vertex_index = cell->vertex_index(i);
392 error +=
393 std::pow(std::fabs(
394 particle_reconstruction_counts[j] -
395 map_vertices_index[vertex_index].second[j]),
396 2);
397 }
398 }
399 cellid_error.push_back(std::make_pair(cell->index(), error));
400 }
401 }
402
403 // Sorted index of cell by error
404 std::sort(cellid_error.begin(),
405 cellid_error.end(),
406 [](std::pair<int, double> &pair1, std::pair<int, double> &pair2) {
407 return (pair1.second < pair2.second);
408 });
409
410 // Return the best cell index
411 return best_cell = cellid_error[0].first;
412}

414template <int dim>
415void
416RPTCellReconstruction<dim>::export_positions()
417{
418 std::ofstream myfile;
419 std::string sep;
420 std::string filename =
421 reconstruction_parameters.reconstruction_positions_file;
422 myfile.open(filename);
423
424 // Headers with space or comma
425 if (filename.substr(filename.find_last_of('.') + 1) == ".dat")
426 {
427 myfile
428 << "reconstructed_positions_id volumes particle_positions_x particle_positions_y particle_positions_z cell_level";
429 sep = " ";
430 }
431 else // .csv is default
432 {
433 myfile
434 << "reconstructed_positions_id,volumes,particle_positions_x,particle_positions_y,particle_positions_z,cell_level";
435 sep = ",";
436 }
437
438 // Add status columns and distance if analyse positions is enable
439 if (reconstruction_parameters.analyse_positions)
440 myfile << sep << "status" << sep << "status_id" << sep << "distance"
441 << std::endl;
442 else
443 myfile << std::endl;
444
445 for (unsigned int i_particle = 0;
446 i_particle < reconstruction_positions.size();
447 i_particle++)
448 {
449 myfile << i_particle << sep << cells_volumes[i_particle] << sep
450 << reconstruction_positions[i_particle][0] << sep
451 << reconstruction_positions[i_particle][1] << sep
452 << reconstruction_positions[i_particle][2] << sep
453 << final_cell_level[i_particle];
454
455 if (reconstruction_parameters.analyse_positions)
456 {
457 // Calculate distance between known and found positions
458 double distance =
459 std::sqrt(std::pow(reconstruction_positions[i_particle][0] -
460 known_positions[i_particle][0],
461 2) +
462 std::pow(reconstruction_positions[i_particle][1] -
463 known_positions[i_particle][1],
464 2) +
465 std::pow(reconstruction_positions[i_particle][2] -
466 known_positions[i_particle][2],
467 2));
468
469 // Associate status to a status id
470 unsigned int status_id = UINT_MAX;
471 if (cell_status[i_particle] == "right_cell")
472 status_id = 0;
473 else if (cell_status[i_particle] == "cost_function|right_cell")
474 status_id = 1;
475 else if (cell_status[i_particle] == "parent_cell|right_cell")
476 status_id = 2;
477 else if (cell_status[i_particle] ==
478 "parent_cell|cost_function|right_cell")
479 status_id = 3;
480 else if (cell_status[i_particle] == "wrong_cell")
481 status_id = 4;
482 else if (cell_status[i_particle] == "cost_function|wrong_cell")
483 status_id = 5;
484 else if (cell_status[i_particle] == "parent_cell|wrong_cell")
485 status_id = 6;
486 else if (cell_status[i_particle] ==
487 "parent_cell|cost_function|wrong_cell")
488 status_id = 7;
489
490
491 myfile << sep << cell_status[i_particle] << sep << status_id << sep
492 << distance << std::endl;
493 }
494 else
495 myfile << std::endl;
496 }
497}
