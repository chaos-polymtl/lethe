44/** @brief Class that manages the physical properties
45 * model which are required to calculate the various physical properties
46 * This centralizes the place where the models are created.
47 * The class can be constructed empty and initialized from parameters
48 * or it can be constructed directly with a Parameters section.
49 * The architecture of the PhysicalPropertiesManager is made to ensure that it
50 * can store as many physical properties models as required. It first stores
51 * the physical properties of the fluids, which are always assumed to lie in
52 * material_id=0, and then stores the physical properties of the solid, which
53 * may be in cells for which material_id>0.
54 *
55 */

83 void
84 provide_simulation_control(
85 std::shared_ptr<SimulationControl> &simulation_control)
86 {
87 // At the present moment, only SpecificHeatModel can be time-dependent
88 // Consequently we only pass the SimulationControl object to the
89 // SpecificHeatModel.
90 for (unsigned int f = 0; f < number_of_fluids; ++f)
91 specific_heat[f]->provide_simulation_control(simulation_control);
92 for (unsigned int s = 0; s < number_of_solids; ++s)
93 specific_heat[s + number_of_fluids]->provide_simulation_control(
94 simulation_control);
95 }

301 unsigned int
302 get_material_interaction_id(
303 const material_interactions_type material_interaction_type,
304 const unsigned int material_0_id,
305 const unsigned int material_1_id) const
306 {
307 if (material_interaction_type == material_interactions_type::fluid_fluid)
308 {
309 std::pair<unsigned int, unsigned int> fluid_fluid_material_interaction(
310 material_0_id, material_1_id);
311 return fluid_fluid_interactions_with_material_interaction_ids
312 .find(fluid_fluid_material_interaction)
313 ->second;
314 }
315 else if (material_interaction_type ==
316 material_interactions_type::fluid_solid)
317 {
318 std::pair<unsigned int, unsigned int> fluid_solid_material_interaction(
319 material_0_id, material_1_id);
320 return fluid_solid_interactions_with_material_interaction_ids
321 .find(fluid_solid_material_interaction)
322 ->second;
323 }
324 else
325 throw(std::runtime_error(
326 "Invalid type of material interaction. The choices are <fluid-fluid|fluid-solid>"));
327 }

342 /** @brief Calculates the global id of the physical property. By default. Lethe stores all
343 * the properties of the fluids, then the properties of the solid. Fluids
344 * need to have a material id of 0, then solids are consequential.
345 *
346 * @param fluid_id the id of the fluid (0 for single phase simulations, 0 or 1 for VOF simulations)
347 *
348 * @param material_id the material id of the cell. Cells with material_id=0 are fluid cells (0 or 1) and cells with material_id>0 are solid cells
349 */
350 unsigned int
351 calculate_global_id(const unsigned int fluid_id,
352 const unsigned int material_id) const
353 {
354 if (material_id < 1)
355 return fluid_id;
356 else
357 {
358 return number_of_fluids + material_id - 1;
359 }
360 }
