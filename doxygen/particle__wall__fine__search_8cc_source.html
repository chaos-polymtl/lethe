18template <int dim>
19void
20particle_wall_fine_search(
21 const typename dem_data_structures<dim>::particle_wall_candidates
22 &particle_wall_contact_pair_candidates,
23 typename dem_data_structures<dim>::particle_wall_in_contact
24 &particle_wall_pairs_in_contact)
25{
26 // Iterating over contact candidates from broad search and adding the pairs to
27 // the particle_wall_pairs_in_contact
28 for (auto const &[particle_id, particle_pair_candidates] :
29 particle_wall_contact_pair_candidates)
30 {
31 if (!particle_pair_candidates.empty())
32 {
33 for (auto const &[face_id, particle_pair_candidate_content] :
34 particle_pair_candidates)
35 {
36 auto particle = std::get<0>(particle_pair_candidate_content);
37
38 // Normal vector of the boundary and a point on the boudary are
39 // defined as local parameters
40 auto normal_vector = std::get<1>(particle_pair_candidate_content);
41 Point<dim> point_on_boundary =
42 std::get<2>(particle_pair_candidate_content);
43
44 Tensor<1, 3> normal_vector_3d;
45 if constexpr (dim == 3)
46 normal_vector_3d = normal_vector;
47
48 if constexpr (dim == 2)
49 normal_vector_3d = tensor_nd_to_3d(normal_vector);
50
51 Point<3> point_on_boundary_3d;
52 if constexpr (dim == 3)
53 point_on_boundary_3d = point_on_boundary;
54
55 if constexpr (dim == 2)
56 point_on_boundary_3d = point_nd_to_3d(point_on_boundary);
57
58 particle_wall_pairs_in_contact[particle_id].emplace(
59 face_id,
60 particle_wall_contact_info<dim>{
61 particle,
62 normal_vector_3d,
63 point_on_boundary_3d,
64 std::get<3>(particle_pair_candidate_content),
65 Tensor<1, 3>(),
66 Tensor<1, 3>()});
67 // std::get<3> contains the boundary id
68 }
69 }
70 }
71}

73template <int dim>
74void
75particle_floating_wall_fine_search(
76 const typename dem_data_structures<dim>::particle_floating_wall_candidates
77 &particle_floating_wall_candidates,
78 const Parameters::Lagrangian::FloatingWalls<dim> &floating_wall_properties,
79 const double simulation_time,
80 typename dem_data_structures<dim>::particle_wall_in_contact
81 &particle_floating_wall_pairs_in_contact)
82{
83 // Reading floating wall properties
84 std::vector<Point<dim>> point_on_wall =
85 floating_wall_properties.points_on_walls;
86 std::vector<Tensor<1, dim>> wall_normal_vector =
87 floating_wall_properties.floating_walls_normal_vectors;
88
89 // Iterating over contact candidates from broad search and adding the pairs to
90 // the particle_floating_wall_pairs_in_contact
91 for (auto const &[particle_id, particle_pair_candidates] :
92 particle_floating_wall_candidates)
93 {
94 if (!particle_pair_candidates.empty())
95 {
96 for (auto particle_pair_candidate_iterator =
97 particle_pair_candidates.begin();
98 particle_pair_candidate_iterator !=
99 particle_pair_candidates.end();
100 ++particle_pair_candidate_iterator)
101 {
102 // Getting the floating wall id once to improve efficiency
103 auto floating_wall_id = particle_pair_candidate_iterator->first;
104
105 // Checking simulation time for temporary floating walls
106 if (simulation_time >=
107 floating_wall_properties.time_start[floating_wall_id] &&
108 simulation_time <=
109 floating_wall_properties.time_end[floating_wall_id])
110 {
111 // Reading particle, normal vector and point on wall once to
112 // improve efficiency
113 auto particle = particle_pair_candidate_iterator->second;
114 Tensor<1, dim> normal_vector =
115 wall_normal_vector[floating_wall_id];
116 Point<dim> point_on_floating_wall =
117 point_on_wall[floating_wall_id];
118
119 Point<3> point_on_floating_wall_3d;
120 if constexpr (dim == 3)
121 point_on_floating_wall_3d = point_on_floating_wall;
122
123 if constexpr (dim == 2)
124 point_on_floating_wall_3d =
125 point_nd_to_3d(point_on_floating_wall);
126
127 // Check to see on which side of the wall the particle is
128 // located:
129
130 // Finding connecting vector from defined point on the
131 // boundary wall to the particle location
132 Tensor<1, dim> connecting_vector =
133 particle->get_location() - point_on_floating_wall;
134 int inner_product_sign =
135 boost::math::sign(connecting_vector * normal_vector);
136
137 // If the cell is located on the opposite side of the defined
138 // normal vector, the normal vector of the cell should be
139 // reversed
140 if (inner_product_sign < 0)
141 {
142 normal_vector = -1 * normal_vector;
143 }
144
145 Tensor<1, 3> normal_vector_3d;
146 if constexpr (dim == 3)
147 normal_vector_3d = normal_vector;
148
149 if constexpr (dim == 2)
150 normal_vector_3d = tensor_nd_to_3d(normal_vector);
151
152 // The boundary ID of floating walls is set to 100, it should
153 // be modified after adding motion of floating walls
154 particle_floating_wall_pairs_in_contact[particle_id].emplace(
155 floating_wall_id,
156 particle_wall_contact_info<dim>{particle,
157 normal_vector_3d,
158 point_on_floating_wall_3d,
159 100,
160 Tensor<1, 3>(),
161 Tensor<1, 3>()});
162 }
163 }
164 }
165 }
166}

168template <int dim>
169void
170particle_floating_mesh_fine_search(
171 const typename dem_data_structures<dim>::particle_floating_mesh_candidates
172 &particle_floating_mesh_contact_candidates,
173 typename dem_data_structures<dim>::particle_floating_mesh_in_contact
174 &particle_floating_mesh_in_contact)
175{
176 particle_floating_mesh_in_contact.resize(
177 particle_floating_mesh_contact_candidates.size());
178
179 for (unsigned int solid_counter = 0;
180 solid_counter < particle_floating_mesh_contact_candidates.size();
181 ++solid_counter)
182 {
183 auto &particle_floating_mesh_element =
184 particle_floating_mesh_in_contact[solid_counter];
185
186 auto &candidates =
187 particle_floating_mesh_contact_candidates[solid_counter];
188
189 for (auto const &[cut_cell_key, candidate_particles] : candidates)
190 {
191 if (!candidate_particles.empty())
192 {
193 for (auto &particle_floating_mesh_candidate_iterator :
194 candidate_particles)
195 {
196 particle_floating_mesh_element[cut_cell_key].emplace(
197 particle_floating_mesh_candidate_iterator.first,
198 particle_wall_contact_info<dim>{
199 particle_floating_mesh_candidate_iterator.second,
200 Tensor<1, 3>(),
201 Point<3>(),
202 0,
203 Tensor<1, 3>(),
204 Tensor<1, 3>()});
205 }
206 }
207 }
208 }
209}
