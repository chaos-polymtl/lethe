16/**
17 * @brief Assembles the core of the Navier-Stokes equation with
18 * free surface using CLS modeling.

20 * \f$ \nabla \cdot \mathbf{u} + \rho \mathbf{u} \cdot \nabla
21 * \mathbf{u} + \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
22 * (\nabla \mathbf{u})^T) = 0 \f$ with an SUPG and PSPG stabilization
23 *
24 * @tparam dim An integer that denotes the number of spatial dimensions
25 *
26 * @ingroup assemblers

30{
31public:
32 GLSNavierStokesCLSAssemblerCore(
33 const std::shared_ptr<SimulationControl> &simulation_control,
34 const SimulationParameters<dim> &nsparam)
35 : simulation_control(simulation_control)
36 , cls_parameters(nsparam.multiphysics.cls_parameters)
37 {}

106/**
107 * @brief Assembles a phase change Darcy forcing term for a CLS two-fluid
108 * simulation. The term \f$-\phi \beta_D \mathbf{u} \f$ is added to the right
109 * hand-side of the Navier-Stokes equations to prohibit the motion of a
110 * material. In the phase change model, the value of the \f$ \beta_D \f$
111 * coefficient depends on the temperature field and the material (fluid)
112 * properties. Generally, this is used to impose the stasis in the solid
113 * phase. This generally leads to a better conditioning of the linear
114 * system than increasing the viscosity of the solid phase.
115 *
116 *
117 * @tparam dim An integer that denotes the number of spatial dimensions
118 *
119 * @ingroup assemblers
120 */
121template <int dim>
122class PhaseChangeDarcyCLSAssembler : public NavierStokesAssemblerBase<dim>
123{
124public:
125 PhaseChangeDarcyCLSAssembler(
126 const std::vector<Parameters::PhaseChange> &phase_change_parameters_vector,
127 const bool enable_darcy_multiply_by_density)
128 : phase_change_parameters_vector(phase_change_parameters_vector)
129 , enable_darcy_multiply_by_density(enable_darcy_multiply_by_density)
130 {}
131
132 /**
133 * @brief assemble_matrix Assembles the matrix of: \f$-\beta_D \mathbf{u}\f$.
134 * @param scratch_data (see base class)
135 * @param copy_data (see base class)
136 */
137 virtual void
138 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
139 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
140
141
142 /**
143 * @brief assemble_rhs Assembles the weak form of: \f$-\beta_D \mathbf{u} \f$
144 * @param scratch_data (see base class)
145 * @param copy_data (see base class)
146 */
147 virtual void
148 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
149 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
150
151private:
152 /*
153 * Phase change parameters are kept within the assembler and are used to
154 * calculate, on the fly, the inverse permeability (\f$ \beta_D \f$).
155 */
156 const std::vector<Parameters::PhaseChange> phase_change_parameters_vector;
157
158 /**
159 * Enable the multiplication of the Darcy force term
160 * (\f$\vec{F}_\mathrm{Darcy}\f$) by the density for dimensional consistency
161 * when solving the pressure rather than the kinematic pressure in the
162 * momentum balance.
163 *
164 * \f[
165 * \vec{F}_\mathrm{Darcy} = \rho K \vec{u}
166 * \f]
167 *
168 * with \f$\rho\f$ the density, \f$K\f$ the Darcy penalty, and \f$\vec{u}\f$
169 * the velocity.
170 */
171 const bool enable_darcy_multiply_by_density;
172};

174/**
175 * @brief Assembler of the non-linear Carman-Kozeny permeability model for CLS
176 * simulations with phase change.
177 *
178 * @tparam dim An integer that denotes the number of spatial dimensions
179 */
180template <int dim>
181class PhaseChangeCarmanKozenyCLSAssembler
182 : public NavierStokesAssemblerBase<dim>
183{
184public:
185 /**
186 * @brief Assembler for the Carman-Kozeny permeability source term used for
187 * phase change modelling.
188 *
189 * \f[
190 * \boldsymbol{F}_\mathrm{Carman-Kozeny} = \frac{-\mu}{A_\mathrm{perm}}
191 * \left[ \frac{(1-\alpha_\mathrm{l})^2}{(\alpha_\mathrm{l})^3 +
192 * \delta}\right] \boldsymbol{u}
193 * \f]
194 *
195 * with \f$\mu\f$ the dynamic viscosity, \f$A_\mathrm{perm}\f$ the
196 * permeability area, \f$\alpha_\mathrm{l}\f$ the liquid fraction,
197 * \f$\delta\f$ a tolerance to avoid division by zero in the solid, and
198 * \f$\boldsymbol{u}\f$ the velocity.
199 *
200 * @param[in] fluid_with_phase_change Indicates which fluid(s) has/have phase
201 * change.
202 * @param[in] carman_kozeny_permeability_area Vector of permeability area
203 * values of the solids (pseudo-porous media).
204 * @param[in] carman_kozeny_tolerance Vector of tolerances to avoid division
205 * by zero in the solid regions.
206 * @param[in] phase_change_parameters_vector Vector of phase change parameters
207 * of all fluids.
208 */
209 PhaseChangeCarmanKozenyCLSAssembler(
210 const Parameters::FluidIndicator &fluid_with_phase_change,
211 const std::vector<double> &carman_kozeny_permeability_area,
212 const std::vector<double> &carman_kozeny_tolerance,
213 const std::vector<Parameters::PhaseChange> &phase_change_parameters_vector)
214 : fluid_has_phase_change(indentify_fluid_with_phase_change(
215 fluid_with_phase_change,
216 phase_change_parameters_vector.size()))
217 , carman_kozeny_permeability_area_inv(
218 compute_permeability_area_inverse(carman_kozeny_permeability_area))
219 , carman_kozeny_tolerance(carman_kozeny_tolerance)
220 , phase_change_parameters_vector(phase_change_parameters_vector)
221 {}
222
223 /**
224 * @brief Assembles the matrix of:
225 * \f[
226 * \boldsymbol{F}_\mathrm{Carman-Kozeny} = \frac{-\mu}{A_\mathrm{perm}}
227 * \left[ \frac{(1-\alpha_\mathrm{l})^2}{(\alpha_\mathrm{l})^3 +
228 * \delta}\right] \boldsymbol{u} \f]
229 *
230 * with \f$\mu\f$ the dynamic viscosity, \f$A_\mathrm{perm}\f$ the
231 * permeability area, \f$\alpha_\mathrm{l}\f$ the liquid fraction,
232 * \f$\delta\f$ a tolerance to avoid division by zero in the solid, and
233 * \f$\boldsymbol{u}\f$ the velocity.
234 *
235 * @param[in] scratch_data Scratch data containing the information required
236 * for system assembly.
237 * It is important to note that the scratch data has to have been re-inited
238 * before calling for matrix assembly.
239 * @param[in,out] copy_data Destination where the local_matrix is copied to.
240 */
241 virtual void
242 assemble_matrix(const NavierStokesScratchData<dim> &scratch_data,
243 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
244
245 /**
246 * @brief Assemble the right-hand side (rhs) of:
247 * \f[
248 * \boldsymbol{F}_\mathrm{Carman-Kozeny} = \frac{-\mu}{A_\mathrm{perm}}
249 * \left[ \frac{(1-\alpha_\mathrm{l})^2}{(\alpha_\mathrm{l})^3 +
250 * \delta}\right] \boldsymbol{u} \f]
251 *
252 * with \f$\mu\f$ the dynamic viscosity, \f$A_\mathrm{perm}\f$ the
253 * permeability area, \f$\alpha_\mathrm{l}\f$ the liquid fraction,
254 * \f$\delta\f$ a tolerance to avoid division by zero in the solid, and
255 * \f$\boldsymbol{u}\f$ the velocity.
256 *
257 * @param[in] scratch_data Scratch data containing the information required
258 * for system assembly.
259 * It is important to note that the scratch data has to have been re-inited
260 * before calling for rhs assembly.
261 * @param[in,out] copy_data Destination where the local_rhs is copied to.
262 */
263 virtual void
264 assemble_rhs(const NavierStokesScratchData<dim> &scratch_data,
265 StabilizedMethodsTensorCopyData<dim> &copy_data) override;
266
267private:
268 /**
269 * @brief Compute the inverse values of the Carman-Kozeny permeability areas.
270 *
271 * @param[in] carman_kozeny_permeability_area Vector of the Carman-Kozeny
272 * permeability area values.
273 *
274 * @return Vector of inverse values of the Carman-Kozeny permeability areas.
275 */
276 static std::vector<double>
277 compute_permeability_area_inverse(
278 const std::vector<double> &carman_kozeny_permeability_area)
279 {
280 std::vector<double> carman_kozeny_permeability_area_inverse(
281 carman_kozeny_permeability_area.size());
282
283 for (unsigned int i = 0; i < carman_kozeny_permeability_area.size(); ++i)
284 carman_kozeny_permeability_area_inverse[i] =
285 1 / carman_kozeny_permeability_area[i];
286
287 return carman_kozeny_permeability_area_inverse;
288 }
289
290 /**
291 * @brief Checks which fluid(s) has/have phase change and returns a vector of
292 * boolean indicating if each fluid has phase change (@p true) or not
293 * (@p false). The indices of the vector correspond to the fluid ID.
294 *
295 * @param[in] fluid_with_phase_change Indicates which fluid(s) has/have phase
296 * change.
297 * @param[in] n_fluids Number of fluids.
298 *
299 * @return Vector of boolean indicating which fluid(s) has/have phase change.
300 */
301 static std::vector<bool>
302 indentify_fluid_with_phase_change(
303 const Parameters::FluidIndicator fluid_with_phase_change,
304 const unsigned int n_fluids)
305 {
306 std::vector<bool> enable_phase_change_vector(n_fluids, false);
307
308 if (fluid_with_phase_change == Parameters::FluidIndicator::fluid0)
309 enable_phase_change_vector[0] = true;
310 else if (fluid_with_phase_change == Parameters::FluidIndicator::fluid1)
311 enable_phase_change_vector[1] = true;
312 else // fluid_with_phase_change == Parameters::FluidIndicator::both
313 std::fill(enable_phase_change_vector.begin(),
314 enable_phase_change_vector.end(),
315 true);
316 return enable_phase_change_vector;
317 }
318
319 /**
320 * Vector where each bool indicates if the corresponding fluid has phase
321 * change or not.
322 */
323 const std::vector<bool> fluid_has_phase_change;
324
325 /**
326 * Vector of one over the permeability area of the pseudo-porous bed
327 * (solid phase) for each simulated fluid.
328 */
329 const std::vector<double> carman_kozeny_permeability_area_inv;
330
331 /**
332 * Vector of tolerances in the Carman-Kozeny source term that avoids division
333 * by zero in the solid phase for each simulated fluid.
334 */
335 const std::vector<double> carman_kozeny_tolerance;
336
337 /**
338 * Phase change parameters for the liquidus and solidus temperature of each
339 * fluid.
340 */
341 const std::vector<Parameters::PhaseChange> phase_change_parameters_vector;
342};

492/**
493 * @brief Assembles the core of the Navier-Stokes equation
494 * using a Rheological model to predict non-Newtonian behaviors
495 *
496 * @tparam dim An integer that denotes the number of spatial dimensions
497 *
498 * @ingroup assemblers
499 */
500template <int dim>
501class GLSNavierStokesCLSAssemblerNonNewtonianCore
502 : public NavierStokesAssemblerBase<dim>
503{
504public:
505 GLSNavierStokesCLSAssemblerNonNewtonianCore(
506 const std::shared_ptr<SimulationControl> &simulation_control,
507 const SimulationParameters<dim> &nsparam)
508 : simulation_control(simulation_control)
509 , cls_parameters(nsparam.multiphysics.cls_parameters)
510 {}
511
512 /**
513 * @brief Calculates an approximation of the gradient of the viscosity
514 * @param velocity_gradient The velocity gradient tensor on the quadrature point
515 @param velocity_hessians The velocity hessian tensor on the quadrature point
516 @param non_newtonian_viscosity The viscosity at which the gradient is calculated
517 @param d_gamma_dot Th difference in the shear rate magnitude to approximate the
518 viscosity variation with a slight change in the shear_rate magnitude
519 */
520 inline Tensor<1, dim>
521 get_kinematic_viscosity_gradient(
522 const Tensor<2, dim> &velocity_gradient,
523 const Tensor<3, dim> &velocity_hessians,
524 const double shear_rate_magnitude,
525 const double grad_kinematic_viscosity_shear_rate) const
526 {
527 // Calculates an approximation of the shear rate magnitude gradient using
528 // the derived form, since it does not change with rheological models
529 Tensor<1, dim> grad_shear_rate;
530 for (int d = 0; d < dim; ++d)
531 {
532 if (dim == 2)
533 {
534 for (unsigned int k = 0; k < dim; ++k)
535 {
536 grad_shear_rate[d] +=
537 2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
538 shear_rate_magnitude;
539 }
540 grad_shear_rate[d] +=
541 (velocity_gradient[0][1] + velocity_gradient[1][0]) *
542 (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
543 shear_rate_magnitude;
544 }
545 else
546 {
547 for (unsigned int k = 0; k < dim; ++k)
548 {
549 grad_shear_rate[d] +=
550 2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
551 shear_rate_magnitude +
552 (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
553 velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
554 (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
555 velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
556 shear_rate_magnitude;
557 }
558 }
559 }
560 return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
561 };
