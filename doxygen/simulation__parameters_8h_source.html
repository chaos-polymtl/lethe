69 void
70 declare(ParameterHandler &prm,
71 Parameters::SizeOfSubsections size_of_subsections)
72 {
73 prm.declare_entry("dimension",
74 "0",
75 Patterns::Integer(),
76 "Dimension of the problem");

78 prm.declare_entry("print parameters",
79 "none",

81 "Print all the parameters, or only"

84 dimensionality.declare_parameters(prm);

86 physical_properties.declare_parameters(prm);

89 nitsche->declare_parameters(prm);

96 prm, size_of_subsections.boundary_conditions);

100 prm, size_of_subsections.boundary_conditions);

153 void
154 parse(ParameterHandler &prm)
155 {
156 dimensionality.parse_parameters(prm);
157 test.parse_parameters(prm);
158
159 for (auto physics_name : physics_names)
160 {
161 PhysicsID physics_id = get_physics_id(physics_name);
162 linear_solver[physics_id].parse_parameters(prm, physics_name);
163 non_linear_solver[physics_id].parse_parameters(prm, physics_name);
164 }
165 for (const auto &vof_subequation_name : vof_subequations_names)
166 {
167 VOFSubequationsID vof_subequations_id =
168 get_vof_subequation_id(vof_subequation_name);
169 vof_subequations_linear_solvers[vof_subequations_id].parse_parameters(
170 prm, vof_subequation_name);
171 vof_subequations_non_linear_solvers[vof_subequations_id]
172 .parse_parameters(prm, vof_subequation_name);
173 }
174
175 mesh_adaptation.parse_parameters(prm);
176 mesh.parse_parameters(prm);
177 mesh_box_refinement->parse_parameters(prm);
178 nitsche->parse_parameters(prm);
179 physical_properties.parse_parameters(prm, dimensionality);
180 multiphysics.parse_parameters(prm, dimensionality);
181 timer.parse_parameters(prm);
182 fem_parameters.parse_parameters(prm);
183 laser_parameters->parse_parameters(prm);
184 forces_parameters.parse_parameters(prm);
185 post_processing.parse_parameters(prm);
186 flow_control.parse_parameters(prm);
187 restart_parameters.parse_parameters(prm);
188 boundary_conditions.parse_parameters(prm);
189 boundary_conditions_ht.parse_parameters(prm);
190 boundary_conditions_tracer.parse_parameters(prm);
191 boundary_conditions_vof.parse_parameters(prm);
192 boundary_conditions_cahn_hilliard.parse_parameters(prm);
193 manifolds_parameters.parse_parameters(prm);
194 initial_condition->parse_parameters(prm);
195 analytical_solution->parse_parameters(prm);
196 source_term.parse_parameters(prm);
197 simulation_control.parse_parameters(prm);
198 velocity_sources.parse_parameters(prm);
199 particlesParameters->parse_parameters(prm);
200 constrain_solid_domain.parse_parameters(prm);
201 stabilization.parse_parameters(prm);
202 ale.parse_parameters(prm);
203 evaporation.parse_parameters(prm);
204 tracer_drift_velocity.parse_parameters(prm);
205 mortar_parameters.parse_parameters(prm);
206
207 physical_properties_manager.initialize(physical_properties);
208
209
210 // Check consistency of parameters parsed in different subsections
211 if (multiphysics.VOF && physical_properties.number_of_fluids != 2)
212 {
213 throw std::logic_error(
214 "Inconsistency in .prm!\n with VOF = true\n use: number of fluids = 2");
215 }
216
217 if (not(multiphysics.VOF) && post_processing.postprocessed_fluid ==
218 Parameters::FluidIndicator::fluid1)
219 {
220 throw std::logic_error(
221 "Inconsistency in .prm!\n when VOF = false"
222 "\n use (default value): set postprocessed fluid = both"
223 "\n or: set postprocessed fluid = fluid 0");
224 }
225
226 if (physical_properties.number_of_fluids == 2 &&
227 (!multiphysics.VOF && !multiphysics.cahn_hilliard))
228 {
229 throw std::logic_error(
230 "Inconsistency in .prm!\n "
231 "Number of fluids in 'physical properties' was set to 2,\n "
232 "but neither VOF or cahn hilliard is enabled in the 'multiphysics'.\n ");
233 }
234
235 // Interface physical property models consistency check
236 if (multiphysics.vof_parameters.surface_tension_force.enable)
237 {
238 std::string constant_surface_tension_model(
239 " subsection fluid-fluid interaction\n"
240 " set first fluid id = 0\n"
241 " set second fluid id = 1\n"
242 " set surface tension model = constant\n"
243 " set surface tension coefficient = $value_of_coefficient\n"
244 " end\n");
245
246 std::string linear_surface_tension_model(
247 " subsection fluid-fluid interaction\n"
248 " set first fluid id = 0\n"
249 " set second fluid id = 1\n"
250 " set surface tension model = linear\n"
251 " set surface tension coefficient = $value_of_coefficient\n"
252 " set temperature-driven surface tension gradient = $value_of_gradient\n"
253 " end\n");
254
255 std::string phase_change_surface_tension_model(
256 " subsection fluid-fluid interaction\n"
257 " set first fluid id = 0\n"
258 " set second fluid id = 1\n"
259 " set surface tension model = phase change\n"
260 " set surface tension coefficient = $value_of_coefficient\n"
261 " set temperature-driven surface tension gradient = $value_of_gradient\n"
262 " set solidus temperature = $value_of_solidus_temperature\n"
263 " set liquidus temperature = $value_of_liquidus_temperature\n"
264 " end\n");
265 if (!multiphysics.vof_parameters.surface_tension_force
266 .enable_marangoni_effect) // constant surface tension model
267 {
268 if (physical_properties.number_of_material_interactions == 0)
269 {
270 throw std::logic_error(
271 "Inconsistency in .prm!\n "
272 "In subsection VOF, with surface tension force enabled,\n "
273 "but no material interactions specified in\n "
274 "subsection physical properties.\n "
275 "Use:\n\n"
276 " set number of material interactions = 1\n"
277 " subsection material interaction 0\n"
278 " set type = fluid-fluid\n" +
279 constant_surface_tension_model + " end\n");
280 }
281 else if (multiphysics.VOF &&
282 multiphysics
283 .heat_transfer) // disabled Marangoni effect error
284 {
285 if (!is_constant_surface_tension_model(
286 physical_properties.material_interactions))
287 {
288 throw std::logic_error(
289 "Inconsistency in .prm!\n "
290 "In subsection multiphysics, VOF and heat transfer enabled,\n "
291 "and in subsection physical properties, a non-constant surface\n "
292 "tension model, but Marangoni effect disabled in subsection\n "
293 "surface tension force of subsection VOF. This is necessary to account\n "
294 "for Marangoni effect. In subsection VOF, use:\n\n "
295 " subsection surface tension force\n"
296 " set enable = true\n"
297 " set enable marangoni effect = true\n"
298 " end\n");
299 }
300 }
301 else
302 {
303 if (no_fluid_fluid_interaction_error(
304 physical_properties.material_interactions))
305 {
306 throw std::logic_error(
307 "Inconsistency in .prm!\n "
308 "in subsection VOF, surface tension force enabled,\n "
309 "but no fluid-fluid material interactions specified in \n "
310 "subsection physical properties\n "
311 "Use:\n\n"
312 " subsection material interaction $material_interaction_id\n"
313 " set type = fluid-fluid\n" +
314 constant_surface_tension_model + " end\n");
315 }
316 }
317 }
318 else // non-constant surface tension model
319 {
320 if (physical_properties.number_of_material_interactions == 0)
321 {
322 throw std::logic_error(
323 "Inconsistency in .prm!\n "
324 "In subsection VOF, marangoni effect enabled,\n "
325 "but no material interactions specified in subsection physical\n "
326 "properties. This is necessary to account for Marangoni \n "
327 "effect. In subsection physical properties, use:\n\n"
328 " set number of material interactions = 1\n"
329 " subsection material interaction 0\n"
330 " set type = fluid-fluid\n" +
331 linear_surface_tension_model +
332 "\n"
333 "or: \n\n"
334 " set number of material interactions = 1\n"
335 " subsection material interaction 0\n"
336 " set type = fluid-fluid\n" +
337 phase_change_surface_tension_model + " end\n");
338 }
339 else
340 {
341 if (no_fluid_fluid_interaction_error(
342 physical_properties.material_interactions))
343 {
344 throw std::logic_error(
345 "Inconsistency in .prm!\n "
346 "In subsection VOF, Marangoni effect enabled,\n "
347 "but no fluid-fluid material interactions specified in subsection\n "
348 "physical properties. This is necessary to account for Marangoni\n "
349 "effect. In subsection physical properties, use:\n\n"
350 " subsection material interaction $material_interaction_id\n"
351 " set type = fluid-fluid\n" +
352 linear_surface_tension_model +
353 "\n"
354 "or:\n\n"
355 " set number of material interactions = 1\n"
356 " subsection material interaction 0\n"
357 " set type = fluid-fluid\n" +
358 phase_change_surface_tension_model + " end\n");
359 }
360 if (is_constant_surface_tension_model(
361 physical_properties.material_interactions))
362 {
363 throw std::logic_error(
364 "Inconsistency in .prm!\n "
365 "In subsection VOF, Marangoni effect enabled,\n "
366 "but a constant surface tension model is specified in subsection\n "
367 "physical properties. This is necessary to account for Marangoni \n "
368 "effect. In subsection physical properties, use:\n\n"
369 " subsection material interaction $material_interaction_id\n"
370 " set type = fluid-fluid\n" +
371 linear_surface_tension_model +
372 "\n"
373 "or:\n\n"
374 " set number of material interactions = 1\n"
375 " subsection material interaction 0\n"
376 " set type = fluid-fluid\n" +
377 phase_change_surface_tension_model + " end\n");
378 }
379 }
380 }
381 }
382
383 if (multiphysics.cahn_hilliard && multiphysics.VOF)
384 {
385 throw std::runtime_error(
386 "Cannot solve a multiphase problem using VOF and Cahn-Hilliard at the same time");
387 }
388
389 if (multiphysics.cahn_hilliard &&
390 physical_properties.number_of_material_interactions == 0)
391 {
392 throw std::logic_error(
393 "Inconsistency in .prm!\n "
394 "cahn hilliard = true, \n "
395 "but no fluid-fluid material interactions specified in\n "
396 "subsection physical properties\n "
397 "Use:\n\n"
398 " subsection material interaction $material_interaction_id\n"
399 " set type = fluid-fluid\n"
400 " subsection fluid-fluid interaction\n"
401 " set first fluid id = 0\n"
402 " set second fluid id = 1\n"
403 " set cahn hilliard mobility model = constant\n"
404 " set cahn hilliard mobility coefficient = $value_of_coefficient\n"
405 " end\n"
406 " end\n");
407 }
408 if (laser_parameters->activate_laser &&
409 (laser_parameters->laser_type ==
410 Parameters::Laser<
411 dim>::LaserType::gaussian_heat_flux_vof_interface ||
412 laser_parameters->laser_type ==
413 Parameters::Laser<
414 dim>::LaserType::uniform_heat_flux_vof_interface) &&
415 !multiphysics.VOF)
416 {
417 throw std::logic_error(
418 "At the moment, the laser surface heat flux is not implemented for 1 fluid simulations."
419 "Please enable the VOF auxiliary physic in the 'multiphysics' subsection, \n"
420 "specify a 2nd fluid in the 'physical properties' subsection,\n"
421 "and define appropriate initial conditions in the 'initial conditions' subsection.");
422 }
423
424 if (!multiphysics.heat_transfer && constrain_solid_domain.enable)
425 {
426 throw std::logic_error(
427 "Inconsistency in .prm!\n "
428 "The apply constraints on a solid domain feature is enabled, however\n "
429 "'heat transfer' was not set to 'true' in the 'multiphysics' subsection.\n ");
430 }
431
432 if (constrain_solid_domain.enable &&
433 physical_properties_manager.get_number_of_fluids() <
434 constrain_solid_domain.number_of_constraints)
435 {
436 std::string n_constraints =
437 Utilities::to_string(constrain_solid_domain.number_of_constraints);
438 std::string n_fluids = Utilities::to_string(
439 physical_properties_manager.get_number_of_fluids());
440 throw std::logic_error(
441 "Inconsistency in .prm!\n "
442 "The number of constraints (" +
443 n_constraints + ") is greater than the number of fluids (" +
444 n_fluids +
445 ").\n "
446 "Only 1 constraint per fluid can be declared.\n ");
447 }
448
449 if (constrain_solid_domain.enable && multiphysics.cahn_hilliard)
450 {
451 throw std::logic_error(
452 "Inconsistency in .prm!\n "
453 "The current implementation for constraining solid domains with\n "
454 "temperature is not implemented for Cahn Hilliard simulations.\n ");
455 }
456 }

458 inline bool
459 no_fluid_fluid_interaction_error(
460 std::vector<Parameters::MaterialInteractions> &material_interactions)
461 {
462 for (const Parameters::MaterialInteractions &material_interaction :
463 material_interactions)
464 {
465 if (material_interaction.material_interaction_type ==
466 Parameters::MaterialInteractions::MaterialInteractionsType::
467 fluid_fluid)
468 {
469 return false;
470 }
471 }
472 return true;
473 }
474
475 inline bool
476 is_constant_surface_tension_model(
477 std::vector<Parameters::MaterialInteractions> &material_interactions)
478 {
479 for (const Parameters::MaterialInteractions &material_interaction :
480 material_interactions)
481 {
482 if (material_interaction.surface_tension_model !=
483 Parameters::MaterialInteractions::SurfaceTensionModel::constant)
484 {
485 return false;
486 }
487 }
488 return true;
489 }
