12class CheckpointControl

14public:

22 CheckpointControl(Parameters::Restart &param)

30 /**
31 * @brief Return true if a checkpoint should be made at the current time step.
32 * @param time_step_number Current time step.
33 */
34 bool
35 is_checkpoint_time_step(const unsigned int time_step_number)
36 {
37 if (restart && !(time_step_number % checkpointing_frequency))
38 {
39 increment_checkpoint_id();
40 return true;
41 }
42 return false;
43 }

83private:
