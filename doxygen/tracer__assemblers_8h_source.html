27/**
28 * @brief Class that assembles the core of the Tracer equation.
29 * This class assembles the weak form of:
30 * \f$\mathbf{u} \cdot \nabla T - D \nabla^2T = S \f$ with an SUPG
31 * stabilization
32 *
33 * @tparam dim An integer that denotes the number of spatial dimensions
34 *
35 * @ingroup assemblers
36 */
37template <int dim>
38class TracerAssemblerCore : public TracerAssemblerBase<dim>
39{
40public:
41 TracerAssemblerCore(
42 const std::shared_ptr<SimulationControl> &simulation_control)
43 : simulation_control(simulation_control)
44 {}
45
46 /**
47 * @brief Assembles the matrix
48 * @param[in] scratch_data (see base class)
49 * @param[in,out] copy_data (see base class)
50 */
51 virtual void
52 assemble_matrix(const TracerScratchData<dim> &scratch_data,
53 StabilizedMethodsCopyData &copy_data) override;
54
55 /**
56 * @brief Assembles the rhs
57 * @param[in] scratch_data (see base class)
58 * @param[in,out] copy_data (see base class)
59 */
60 virtual void
61 assemble_rhs(const TracerScratchData<dim> &scratch_data,
62 StabilizedMethodsCopyData &copy_data) override;
63
64 const std::shared_ptr<SimulationControl> simulation_control;
65};

148/**
149 * @brief A pure virtual class that serves as an interface for boundary and
150 * internal faces that occur when using a discontinuous Galerkin discretization.
151 * The main difference between the TracerFaceAssembler and the TracerAssembler
152 * is that the TracerFaceAssembler assembles the matrix and rhs for internal
153 * faces and thus requires the StabilizedDGMethodsCopyData class.
154 *
155 * @tparam dim An integer that denotes the number of spatial dimensions
156 *
157 * @ingroup assemblers
158 */
159template <int dim>
160class TracerFaceAssembler
161{
162public:
163 /**
164 * @brief Interface for the call to matrix assembly
165 * @param[in] scratch_data Scratch data containing the Tracer
166 * information. It is important to note that the scratch data has to have been
167 * re-inited before calling for matrix assembly.
168 * @param[in,out] copy_data Destination where the local_rhs and local_matrix
169 * should be copied
170 */
171
172 virtual void
173 assemble_matrix(const TracerScratchData<dim> &scratch_data,
174 StabilizedDGMethodsCopyData &copy_data) = 0;
175
176
177 /**
178 * @brief Interface for the call to rhs
179 * @param[in] scratch_data Scratch data containing the Tracer
180 * information. It is important to note that the scratch data has to have been
181 * re-inited before calling for matrix assembly.
182 * @param[in,out] copy_data Destination where the local_rhs and local_matrix
183 * should be copied
184 */
185
186 virtual void
187 assemble_rhs(const TracerScratchData<dim> &scratch_data,
188 StabilizedDGMethodsCopyData &copy_data) = 0;
189};

283/**
284 * @brief Class that assembles the reaction source term of the Tracer equation.
285 * It assembles the term:
286 * \f$ S = - R \= - k C \f$
287 * Where C is the tracer concentration and k is a reaction rate prefactor that
288 * changes depending on the underlying model. For power law of base rate
289 * $\alpha$ and order $n$, the model is: \f$ S = - R \= - (\alpha C^(n-1)) C \f$
290 *
291 * @tparam dim An integer that denotes the number of spatial dimensions
292 *
293 * @ingroup assemblers
294 */
295template <int dim>
296class TracerAssemblerReaction : public TracerAssemblerBase<dim>
297{
298public:
299 TracerAssemblerReaction(
300 const std::shared_ptr<SimulationControl> &simulation_control)
301 : simulation_control(simulation_control)
302 {}
303

305 * @brief Assembles the matrix

307 * @param[in,out] copy_data (see base class)
308 */

310 assemble_matrix(const TracerScratchData<dim> &scratch_data,

312
313 /**
314 * @brief Assembles the rhs
