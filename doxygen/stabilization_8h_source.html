17/**
18 * @brief Calculate the stabilization parameter for the Navier-Stokes equations
19 * in steady-state
20 * @return Value of the stabilization parameter - tau
21 *
22 * @param u_mag Magnitude of the velocity
23 *
24 * @param kinematic_viscosity Kinematic viscosity
25 *
26 * @param h Cell size; it should be calculated using the diameter of a sphere of
27 * equal volume to that of the cell.
28 */
29inline double
30calculate_navier_stokes_gls_tau_steady(const double u_mag,
31 const double kinematic_viscosity,
32 const double h)
33{
34 return 1. / std::sqrt(Utilities::fixed_power<2>(2. * u_mag / h) +
35 9 * Utilities::fixed_power<2>(4 * kinematic_viscosity /
36 (h * h)));
37}

39/**
40 * @brief Calculate the stabilization parameter for the transient Navier-Stokes
41 * equations
42 * @return Value of the stabilization parameter - tau
43 *
44 * @param u_mag Magnitude of the velocity
45 *
46 * @param kinematic_viscosity Kinematic viscosity
47 *
48 * @param h Cell size; it should be calculated using the diameter of a sphere of
49 * equal volume to that of the cell.
50 *
51 * @param sdt Inverse of the time-step (1/dt)
52 */
53
54inline double
55calculate_navier_stokes_gls_tau_transient(const double u_mag,
56 const double kinematic_viscosity,
57 const double h,
58 const double sdt)
59{
60 return 1. / std::sqrt(Utilities::fixed_power<2>(sdt) +
61 Utilities::fixed_power<2>(2. * u_mag / h) +
62 9 * Utilities::fixed_power<2>(4 * kinematic_viscosity /
63 (h * h)));
64}

69 * James A. Rossmanith, and David C. Seal. "A simple and effective high-order
70 * shock-capturing limiter for discontinuous Galerkin methods." arXiv preprint
71 * arXiv:1507.03024 (2015). https://doi.org/10.48550/arXiv.1507.03024
72 *

74 * value of alpha in the article to 0. This is supposed to lead to a more

77 * integrator.
78 *
79 * @param dof_handler The DoFHandler used in the scalar simulation. This DOFHandler must have only a single component (a scalar equation).
80 *
81 * @param locally_relevant_vector A solution vector that contains the locally_relevant solution. This vector is used to read the solution on the ghost cells and will be modified at the end.
82 *
83 * @param locally_owned_vector A solution vector that contains the locally_owned solution. This vector will be used to write the new limited solution.
