17/**
18 * @brief Class that creates a hyper_rectangle and transforms it to obtain the hill geometry with
19 * the hill_geometry function. It also attaches a manifold to the geometry.
20 */
21
22template <int dim, int spacedim>
23class PeriodicHillsGrid
24{
25public:
26 /**
27 * @brief Constructor for the PeriodicHillsGrid.
28 *
29 * @param grid_arguments. A string with 5 parameters
30 * spacing_y : allows to control the shifting of horizontal line [0 - 1]
31 * alpha : allows to elongate while keeping the same flat region [0.5 - 3]
32 * repetitions_x : number of separation of cells in x before refinement
33 * repetitions_y : number of separation of cells in y before refinement
34 * repetitions_z : number of separation of cells in z before refinement
35 */
36 PeriodicHillsGrid(const std::string &grid_arguments);
37
38 /**
39 * @brief The hill_geometry function calculates all the domain of the geometry with 6
40 * polynomials depending the x position. (See Hill Geometry Definition file :
41 * https://turbmodels.larc.nasa.gov/Other_LES_Data/2dhill_periodic.html)
42 * This code has nondimensionalized geometry, but the coefficients provided
43 * need a hill height of 28.
44 * This function also does a gradual shifting of the horizontal lines
45 * prior to have smaller element on the bottom of the geometry where results
46 * are more important.
47 *
48 * @param p A point in space which will be adapted to the periodic hill geometry
49 *
50 * @param param Non-linear solver parameters
51 *
52 */
53 Point<spacedim> static hill_geometry(const Point<spacedim> &p,
54 double spacing_y,
55 double alpha);
56
57 /**
58 * @brief make_grid. The make_grid function generates a hyper rectangle of the size of the domain
59 * and then transforms it to the hill geometry. It also constructs the
60 * geometry manifold with FunctionManifold and finally sets the manifold.
61 *
62 * @param triangulation. The triangulation object on which the grid is generated
63 */
64 void
65 make_grid(Triangulation<dim, spacedim> &triangulation);
66
67private:
68 std::string grid_arguments;
69 double spacing_y;
70 int repetitions_x;
71 int repetitions_y;
72 int repetitions_z;
73 double alpha;
74};

76/**
77 * @brief The push_forward & the pull_back classes create the vector_value functions
78 * gradient function and because it inherits from Function<spacedim>. (formula
79 * is currently Euler and can be changed. See AutoDerivativeFunction
80 * documentation)
81 */
82template <int dim, int spacedim>
83class PeriodicHillsPushForward : public AutoDerivativeFunction<spacedim>
84{
85public:
86 PeriodicHillsPushForward(double spacing_y, double alpha)
87 : AutoDerivativeFunction<spacedim>(1e-6, spacedim)
88 , spacing_y(spacing_y)
89 , alpha(alpha)
90 {}
91 /**
92 * @brief vector_value. This function is used to construct the geometry manifold.
93 * It changes the original point (op) of the transformed hyper_rectangle
94 * with hill_geometry to a new point (np) of the hill grid with per_hills_grid
95 * function.
96 *
97 * @param p. A point in space
98 *
99 * @param values. The vector of values which will be calculated at the position p.
100 */
101 virtual void
102 vector_value(const Point<spacedim> &p, Vector<double> &values) const override;
103
104 /**
105 * @brief value. The value function does the same thing than vector_value for one component.
106 * This implementation is needed to use the gradient function inherited by
107 * AutoDerivativeFunction.
108 *
109 * @param op. An original point in space
110 *
111 * @param component. The component of the point (x=0, y=1, z=2)
112 */
113 virtual double
114 value(const Point<spacedim> &p, const unsigned int component) const override;
115
116private:
117 double spacing_y;
118 double alpha;
119};

122template <int dim, int spacedim>
123class PeriodicHillsPullBack : public AutoDerivativeFunction<spacedim>
124{
125public:
126 PeriodicHillsPullBack(double spacing_y, double alpha)
127 : AutoDerivativeFunction<spacedim>(1e-6, spacedim)
128 , spacing_y(spacing_y)
129 , alpha(alpha)
130 {}
131
132 /**
133 * \brief vector_value. This vector_value function is used to construct the
134 * geometry manifold. It changes the new point (np) of the hill grid to the
135 * original point (op) of the transformed hyper_rectangle grid. This function
136 * is mandatory to use FunctionManifold. First, it finds the minimum value of
137 * y depending the x position and then calculates the op with the inverse of
138 * the transformation done by hill_geometry function.
139 *
140 * \param np. A point in space.
141 *
142 * @param values. The vector of values which will be calculated at the position p.
143 */
144 virtual void
145 vector_value(const Point<spacedim> &np,
146 Vector<double> &values) const override;
147
148 /**
149 * @brief value. The value function does the same thing than vector_value for one component.
150 * This implementation is needed to use the gradient function inherited by
151 * AutoDerivativeFunction.
152 *
153 * @param p. An original point in space
154 *
155 * @param component. The component of the point (x=0, y=1, z=2)
156 *
157 */
158 virtual double
159 value(const Point<spacedim> &np, const unsigned int component) const override;
160
161private:
162 double spacing_y;
163 double alpha;
164};
