91/**
92 * @brief Constructor for the PeriodicHillsGrid.
93 *
94 * @param grid_arguments. A string with 5 parameters
95 * spacing_y : allows to control the shifting of horizontal line [0 - 1]
96 * alpha : allows to elongate while keeping the same flat region [0.5 - 3]
97 * repetitions_x : number of separation of cells in x before refinement
98 * repetitions_y : number of separation of cells in y before refinement
99 * repetitions_z : number of separation of cells in z before refinement
100 */
101
102template <int dim, int spacedim>
103PeriodicHillsGrid<dim, spacedim>::PeriodicHillsGrid(
104 const std::string &grid_arguments)
105{
106 this->grid_arguments = grid_arguments;
107
108 // Separate arguments of the string
109 std::vector<std::string> arguments;
110 std::stringstream s_stream(grid_arguments);
111 while (s_stream.good())
112 {
113 std::string substr;
114 getline(s_stream, substr, ';');
115 arguments.push_back(substr);
116 }
117
118 std::vector<double> arguments_double =
119 dealii::Utilities::string_to_double(arguments);
120 spacing_y = arguments_double[0];
121 alpha = arguments_double[1];
122 repetitions_x = static_cast<int>(arguments_double[2]);
123 repetitions_y = static_cast<int>(arguments_double[3]);
124 if (dim == 3)
125 repetitions_z = static_cast<int>(arguments_double[4]);
126
127 if (abs(alpha - 1) < 1e-6)
128 alpha = int(alpha);
129}

131/**
132 * @brief vector_value. This function is used to construct the geometry manifold.
133 * It changes the original point (op) of the transformed hyper_rectangle
134 * with hill_geometry to a new point (np) of the hill grid with per_hills_grid
135 * function.
136 *
137 * @param p. A point in space
138 *
139 * @param values. The vector of values which will be calculated at the position p.
140 */
141template <int dim, int spacedim>
142void
143PeriodicHillsPushForward<dim, spacedim>::vector_value(
144 const Point<spacedim> &op,
145 Vector<double> &values) const
146{
147 const Point<spacedim> np =
148 PeriodicHillsGrid<dim, spacedim>::hill_geometry(op, spacing_y, alpha);
149
150 values(0) = np[0];
151 values(1) = np[1];
152
153 if (spacedim == 3)
154 values(2) = np[2];
155}

177/**
178 * \brief vector_value. This vector_value function is used to construct the
179 * geometry manifold. It changes the new point (np) of the hill grid to the
180 * original point (op) of the transformed hyper_rectangle grid. This function
181 * is mandatory to use FunctionManifold. First, it finds the minimum value of
182 * y depending the x position and then calculates the op with the inverse of
183 * the transformation done by hill_geometry function.
184 *
185 * \param np. A point in space.
186 *
187 * @param values. The vector of values which will be calculated at the position p.
188 */
189template <int dim, int spacedim>
190void
191PeriodicHillsPullBack<dim, spacedim>::vector_value(const Point<spacedim> &np,
192 Vector<double> &values) const
193{
194 double x = np[0];
195 double max_y = 3.035;
196 double max_x = 9.0;
197 double flat_region_length = 5.142;
198 double left_hill = 1.929;
199 double right_hill = 7.071;
200 double min_y;
201
202 // Reversing elongation and shifting of x lines
203 if (alpha != 1)
204 {
205 if (x < left_hill * alpha)
206 x = (x / alpha);
207 else if (x > alpha * left_hill + flat_region_length)
208 x = (x - flat_region_length - alpha * left_hill) / alpha + right_hill;
209 else
210 x = x - (alpha * left_hill) + left_hill;
211 }
212
213 if (alpha > 1)
214 {
215 if (x < max_x / 2)
216 x = (-(1 - 0.5) +
217 std::sqrt(std::pow((1 - 0.5), 2) - (4 * (1 / max_x) * -x))) /
218 (2 / max_x);
219 else if (x > max_x / 2 && x < max_x)
220 x = (-(1 + 1.5) + std::sqrt(std::pow((1 + 1.5), 2) -
221 (4 * (-1 / max_x) * (-0.5 * max_x - x)))) /
222 (2 * -1 / max_x);
223 }
224
225 // Reversing polynomial transformation and shifting of y lines
226 if (spacedim == 2)
227 {
228 min_y =
229 PeriodicHillsGrid<dim, spacedim>::hill_geometry(Point<spacedim>(x, 0),
230 spacing_y,
231 alpha)[1];
232 }
233 else if (spacedim == 3)
234 {
235 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
236 Point<spacedim>(x, 0, np[2]), spacing_y, alpha)[1];
237 values(2) = np[2];
238 }
239
240 double y = (np[1] - min_y) / (1 - min_y / max_y);
241
242 if (y < max_y / 2 && spacing_y != 0)
243 y = (-(1 - 0.5 * spacing_y) + std::sqrt(std::pow((1 - 0.5 * spacing_y), 2) -
244 (4 * (spacing_y / max_y) * -y))) /
245 (2 * spacing_y / max_y);
246 else if (y > max_y / 2 && y < max_y && spacing_y != 0)
247 y =
248 (-(1 + 1.5 * spacing_y) +
249 std::sqrt(std::pow((1 + 1.5 * spacing_y), 2) -
250 (4 * (-spacing_y / max_y) * (-0.5 * spacing_y * max_y - y)))) /
251 (2 * -spacing_y / max_y);
252
253 values(0) = x;
254 values(1) = y;
255}

257/**
258 * @brief value. The value function does the same thing than vector_value for one component.
259 * This implementation is needed to use the gradient function inherited by
260 * AutoDerivativeFunction.
261 *
262 * @param p. An original point in space
263 *
264 * @param component. The component of the point (x=0, y=1, z=2)
265 *
266 */
267template <int dim, int spacedim>
268double
269PeriodicHillsPullBack<dim, spacedim>::value(const Point<spacedim> &np,
270 const unsigned int component) const
271{
272 double x = np[0];
273 double max_y = 3.035;
274 double max_x = 9.0;
275 double flat_region_length = 5.142;
276 double left_hill = 1.929;
277 double right_hill = 7.071;
278 double min_y;
279
280 if (alpha != 1)
281 {
282 if (x < left_hill * alpha)
283 x = (x / alpha);
284 else if (x > alpha * left_hill + flat_region_length)
285 x = (x - flat_region_length - alpha * left_hill) / alpha + right_hill;
286 else
287 x = x - (alpha * left_hill) + left_hill;
288 }
289
290 if (alpha > 1)
291 {
292 if (x < max_x / 2)
293 x = (-(1 - 0.5) +
294 std::sqrt(std::pow((1 - 0.5), 2) - (4 * (1 / max_x) * -x))) /
295 (2 / max_x);
296 else if (x > max_x / 2 && x <= max_x)
297 x = (-(1 + 1.5) + std::sqrt(std::pow((1 + 1.5), 2) -
298 (4 * (-1 / max_x) * (-0.5 * max_x - x)))) /
299 (2 * -1 / max_x);
300 }
301
302 if (spacedim == 2)
303 min_y =
304 PeriodicHillsGrid<dim, spacedim>::hill_geometry(Point<spacedim>(x, 0),
305 spacing_y,
306 alpha)[1];
307 else if (spacedim == 3)
308 min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
309 Point<spacedim>(x, 0, np[2]), spacing_y, alpha)[1];
310
311
312 double y = (np[1] - min_y) / (1 - min_y / max_y);
313
314 if (y < max_y / 2 && spacing_y != 0)
315 y = (-(1 - 0.5 * spacing_y) + std::sqrt(std::pow((1 - 0.5 * spacing_y), 2) -
316 (4 * (spacing_y / max_y) * -y))) /
317 (2 * spacing_y / max_y);
318 else if (y > max_y / 2 && y < max_y && spacing_y != 0)
319 y =
320 (-(1 + 1.5 * spacing_y) +
321 std::sqrt(std::pow((1 + 1.5 * spacing_y), 2) -
322 (4 * (-spacing_y / max_y) * (-0.5 * spacing_y * max_y - y)))) /
323 (2 * -spacing_y / max_y);
324
325 Point<spacedim> op;
326 if (spacedim == 2)
327 op = {x, y};
328 else if (spacedim == 3)
329 op = {x, y, np[2]};
330
331 return op[component];
332}

334/**
335 * @brief The hill_geometry function calculates all the domain of the geometry with 6
336 * polynomials depending the x position. (See Hill Geometry Definition file :
337 * https://turbmodels.larc.nasa.gov/Other_LES_Data/2dhill_periodic.html)
338 * This code has nondimensionalized geometry, but the coefficients provided
339 * need a hill height of 28.
340 * This function also does a gradual shifting of the horizontal lines
341 * prior to have smaller element on the bottom of the geometry where results
342 * are more important.
343 *
344 * @param p A point in space which will be adapted to the periodic hill geometry
345 *
346 * @param param Non-linear solver parameters
347 *
348 */
349template <int dim, int spacedim>
350Point<spacedim>
351PeriodicHillsGrid<dim, spacedim>::hill_geometry(const Point<spacedim> &p,
352 double spacing_y,
353 double alpha)
354{
355 double H = 28; // Height dimension to use with polynomials
356 double x = p[0] * H, y = p[1] * H;
357 double max_y = 3.035 * H;
358 double max_x = 9.0 * H;
359 double flat_region_length = 5.142 * H;
360 double left_hill = 1.929 * H;
361 double right_hill = 7.071 * H;
362 double pos_x_left;
363 double pos_x_right;
364 double pos_y_bottom;
365 double pos_y_top;
366 double pos_y;
367 double y_0;
368 double diff;
369
370 // Gradual spacing and swifting depending on x position if the geometry
371 // is elongated
372 if (alpha > 1)
373 {
374 if (x < max_x / 2)
375 {
376 pos_x_left = x / -max_x + 0.5;
377 x -= pos_x_left * x;
378 }
379 else if (x > max_x / 2 && x < max_x)
380 {
381 pos_x_right = x / max_x - 0.5;
382 x += pos_x_right * (max_x - x);
383 }
384 }
385
386 // Polynomial coefficients :
387 const double a1 = 2.800000000000E+01, b1 = 0.000000000000E+00,
388 c1 = 6.775070969851E-03, d1 = -2.124527775800E-03;
389 const double a2 = 2.507355893131E+01, b2 = 9.754803562315E-01,
390 c2 = -1.016116352781E-01, d2 = 1.889794677828E-03;
391 const double a3 = 2.579601052357E+01, b3 = +8.206693007457E-01,
392 c3 = -9.055370274339E-02, d3 = 1.626510569859E-03;
393 const double a4 = 4.046435022819E+01, b4 = -1.379581654948E+00,
394 c4 = 1.945884504128E-02, d4 = -2.070318932190E-04;
395 const double a5 = 1.792461334664E+01, b5 = +8.743920332081E-01,
396 c5 = -5.567361123058E-02, d5 = 6.277731764683E-04;
397 const double a6 = 5.639011190988E+01, b6 = -2.010520359035E+00,
398 c6 = 1.644919857549E-02, d6 = 2.674976141766E-05;
399
400 double new_x = (max_x - x); // x for the left side of the geometry
401
402 // Gradual spacing and swifting depending on y position
403 if (y < max_y / 2 && (std::abs(spacing_y) > 0))
404 {
405 pos_y_bottom = y / -max_y + 0.5;
406 y -= spacing_y * pos_y_bottom * y;
407 }
408 else if (y > max_y / 2 && y < max_y && (std::abs(spacing_y) > 0))
409 {
410 pos_y_top = y / max_y - 0.5;
411 y += spacing_y * pos_y_top * (max_y - y);
412 }
413
414 pos_y = y / -max_y + 1;
415
416 // Polynomial equations :
417 if (x >= 0 && x < 9)
418 {
419 y += pos_y * (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
420
421 // Checking if y is under 28 and correction
422 y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
423 diff = y_0 - 28.0;
424 if (diff > 0)
425 y -= pos_y * diff;
426 }
427 else if (x >= 9 && x < 14)
428 y += pos_y * (a2 + b2 * x + c2 * std::pow(x, 2) + d2 * std::pow(x, 3));
429 else if (x >= 14 && x < 20)
430 y += pos_y * (a3 + b3 * x + c3 * std::pow(x, 2) + d3 * std::pow(x, 3));
431 else if (x >= 20 && x < 30)
432 y += pos_y * (a4 + b4 * x + c4 * std::pow(x, 2) + d4 * std::pow(x, 3));
433 else if (x >= 30 && x < 40)
434 y += pos_y * (a5 + b5 * x + c5 * std::pow(x, 2) + d5 * std::pow(x, 3));
435 else if (x >= 40 && x < 54)
436 {
437 y += pos_y * (a6 + b6 * x + c6 * std::pow(x, 2) + d6 * std::pow(x, 3));
438 if (y < 0)
439 y = 0;
440 }
441 else if (x <= 252 && x > 243)
442 {
443 y += pos_y * (a1 + b1 * new_x + c1 * std::pow(new_x, 2) +
444 d1 * std::pow(new_x, 3));
445 y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
446 diff = y_0 - 28.0;
447 if (diff > 0)
448 y -= pos_y * diff;
449 }
450 else if (x <= 243 && x > 238)
451 y += pos_y *
452 (a2 + b2 * new_x + c2 * std::pow(new_x, 2) + d2 * std::pow(new_x, 3));
453 else if (x <= 238 && x > 232)
454 y += pos_y *
455 (a3 + b3 * new_x + c3 * std::pow(new_x, 2) + d3 * std::pow(new_x, 3));
456 else if (x <= 232 && x > 222)
457 y += pos_y *
458 (a4 + b4 * new_x + c4 * std::pow(new_x, 2) + d4 * std::pow(new_x, 3));
459 else if (x <= 222 && x > 212)
460 y += pos_y *
461 (a5 + b5 * new_x + c5 * std::pow(new_x, 2) + d5 * std::pow(new_x, 3));
462 else if (x <= 212 && x > 198)
463 {
464 y += pos_y * (a6 + b6 * new_x + c6 * std::pow(new_x, 2) +
465 d6 * std::pow(new_x, 3));
466 if (y < 0)
467 y = 0;
468 }
469 else
470 y += 0;
471
472 // Elongation of the geometry with the alpha factor
473 // Note : The length of the flat region is always the same length
474 if (x < left_hill)
475 x = alpha * x;
476 else if (x > right_hill)
477 x = alpha * (x - right_hill) + flat_region_length + alpha * left_hill;
478 else
479 x = (x - left_hill) + (alpha * left_hill);
480
481 Point<spacedim> q;
482 q[0] = (x / H);
483 q[1] = (y / H);
484
485 if (spacedim == 3)
486 q[2] = p[2];
487
488 return q;
489}

491/**
492 * @brief make_grid. The make_grid function generates a hyper rectangle of the size of the domain
493 * and then transforms it to the hill geometry. It also constructs the
494 * geometry manifold with FunctionManifold and finally sets the manifold.
495 *
496 * @param triangulation. The triangulation object on which the grid is generated
497 */
498template <int dim, int spacedim>
499void
500PeriodicHillsGrid<dim, spacedim>::make_grid(
501 Triangulation<dim, spacedim> &triangulation)
502{
503 std::vector<unsigned int> repetitions(2);
504 repetitions[0] = repetitions_x;
505 repetitions[1] = repetitions_y;
506
507 if (dim == 2)
508 {
509 if (alpha != 1 && (repetitions_x > 1 || repetitions_y > 1))
510 throw std::logic_error(
511 "When alpha parameter is not 1, repetition parameters should be all set to 1");
512
513 GridGenerator::subdivided_hyper_rectangle(triangulation,
514 repetitions,
515 Point<dim>(0.0, 0.0),
516 Point<dim>(9.0, 3.035),
517 true);
518 }
519 else if (dim == 3)
520 {
521 if (alpha != 1 &&
522 (repetitions_x > 1 || repetitions_y > 1 || repetitions_z > 1))
523 throw std::logic_error(
524 "When alpha parameter is not 1, repetition parameters should be all set to 1");
525 repetitions.push_back(repetitions_z);
526 GridGenerator::subdivided_hyper_rectangle(triangulation,
527 repetitions,
528 Point<dim>(0.0, 0.0, 0.0),
529 Point<dim>(9.0, 3.035, 4.5),
530 true);
531 }
532
533 // Transformation of the geometry with the hill geometry
534 GridTools::transform(
535 [this](const Point<spacedim> &p) {
536 return this->hill_geometry(p, spacing_y, alpha);
537 },
538 triangulation);
539
540 // Manifold construction
541 static const FunctionManifold<dim, spacedim, spacedim> manifold_func(
542 std::make_unique<PeriodicHillsPushForward<dim, spacedim>>(spacing_y, alpha),
543 std::make_unique<PeriodicHillsPullBack<dim, spacedim>>(spacing_y, alpha));
544 triangulation.set_manifold(1, manifold_func);
545 triangulation.set_all_manifold_ids(1);
546}
