18template <int dim, typename PropertiesIndex>
19InsertionList<dim, PropertiesIndex>::InsertionList(
20 const std::vector<std::shared_ptr<Distribution>>
21 &size_distribution_object_container,
22 const parallel::distributed::Triangulation<dim> &triangulation,
23 const DEMSolverParameters<dim> &dem_parameters)
24 : Insertion<dim, PropertiesIndex>(size_distribution_object_container,
25 triangulation,
26 dem_parameters)
27 , remaining_particles_of_each_type(
28 dem_parameters.lagrangian_physical_properties.number.at(0))
29{
30 // Initializing current inserting particle type
31 current_inserting_particle_type = 0;
32
33 const std::vector<double> &list_x = dem_parameters.insertion_info.list_x;
34 const std::vector<double> &list_y = dem_parameters.insertion_info.list_y;
35 const std::vector<double> &list_z = dem_parameters.insertion_info.list_z;
36 const std::vector<double> &list_vx = dem_parameters.insertion_info.list_vx;
37 const std::vector<double> &list_vy = dem_parameters.insertion_info.list_vy;
38 const std::vector<double> &list_vz = dem_parameters.insertion_info.list_vz;
39 const std::vector<double> &list_wx = dem_parameters.insertion_info.list_wx;
40 const std::vector<double> &list_wy = dem_parameters.insertion_info.list_wy;
41 const std::vector<double> &list_wz = dem_parameters.insertion_info.list_wz;
42 std::vector<double> list_d = dem_parameters.insertion_info.list_d;
43 std::vector<double> list_T = dem_parameters.insertion_info.list_T;
44
45 // If the default diameter is negative, the diameter list is resized to the
46 // number of particles and assigned the average particle diameter to all
47 // particles
48 if (list_d[0] < 0)
49 {
50 double particle_average_diameter =
51 dem_parameters.lagrangian_physical_properties.particle_average_diameter
52 .at(current_inserting_particle_type);
53 list_d.assign(list_x.size(), particle_average_diameter);
54 }
55
56 // Throw an error if the number diameter provided is not coherent with the
57 // number of particles.
58 AssertThrow(list_x.size() == list_d.size(),
59 DiameterSizeCoherence(list_x.size(), list_d.size()));
60
61 // Generate vector of insertion position
62 for (unsigned int i = 0; i < list_x.size(); ++i)
63 {
64 if constexpr (dim == 2)
65 {
66 insertion_points.emplace_back(Point<dim>({list_x[i], list_y[i]}));
67 velocities.emplace_back(Tensor<1, 3>({list_vx[i], list_vy[i], 0.}));
68 angular_velocities.emplace_back(Tensor<1, 3>({0., 0., list_wz[i]}));
69 }
70 if constexpr (dim == 3)
71 {
72 insertion_points.emplace_back(
73 Point<dim>({list_x[i], list_y[i], list_z[i]}));
74 velocities.emplace_back(
75 Tensor<1, 3>({list_vx[i], list_vy[i], list_vz[i]}));
76 angular_velocities.emplace_back(
77 Tensor<1, 3>({list_wx[i], list_wy[i], list_wz[i]}));
78 }
79 }
80 diameters = list_d;
81 temperatures = list_T;
82}

84// The main insertion function. Insert_global_function is used to insert the
85// particles
86template <int dim, typename PropertiesIndex>
87void
88InsertionList<dim, PropertiesIndex>::insert(
89 Particles::ParticleHandler<dim> &particle_handler,
90 const parallel::distributed::Triangulation<dim> &triangulation,
91 const DEMSolverParameters<dim> &dem_parameters)
92{
93 // TODO refactor into a function call
94 if (remaining_particles_of_each_type == 0 &&
95 current_inserting_particle_type !=
96 dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
97 {
98 remaining_particles_of_each_type =
99 dem_parameters.lagrangian_physical_properties.number.at(
100 ++current_inserting_particle_type);
101 }
102
103 if (remaining_particles_of_each_type > 0)
104 {
105 if (this->removing_particles_in_region)
106 {
107 if (this->mark_for_update)
108 {
109 this->find_cells_in_removing_box(triangulation);
110 this->mark_for_update = false;
111 }
112 this->remove_particles_in_box(particle_handler);
113 }
114
115 unsigned int n_total_particles_to_insert = insertion_points.size();
116 n_total_particles_to_insert =
117 std::min(remaining_particles_of_each_type, n_total_particles_to_insert);
118
119 // All processors except 0 will not insert particles
120 MPI_Comm communicator = triangulation.get_mpi_communicator();
121 auto this_mpi_process = Utilities::MPI::this_mpi_process(communicator);
122 const unsigned int n_particles_to_insert_this_proc =
123 this_mpi_process == 0 ? n_total_particles_to_insert : 0;
124
125 std::vector<Point<dim>> insertion_points_on_proc_this_step;
126 insertion_points_on_proc_this_step.reserve(
127 n_particles_to_insert_this_proc);
128
129 // Because the list insertion is made to insert only a few particles
130 // only processor 0 manages the insertion of these particles
131 if (this_mpi_process == 0)
132 {
133 for (unsigned int p = 0; p < n_particles_to_insert_this_proc; ++p)
134 {
135 insertion_points_on_proc_this_step.emplace_back(
136 insertion_points[p]);
137 }
138 }
139
140 // Obtain global bounding boxes
141 const auto my_bounding_box =
142 GridTools::compute_mesh_predicate_bounding_box(
143 triangulation, IteratorFilters::LocallyOwnedCell());
144 const auto global_bounding_boxes =
145 Utilities::MPI::all_gather(communicator, my_bounding_box);
146
147 // A vector of vectors, which contains all the properties of all inserted
148 // particles at each insertion step
149 std::vector<std::vector<double>> particle_properties;
150
151 // Assign inserted particles properties
152 this->assign_particle_properties_for_list_insertion(
153 dem_parameters,
154 n_particles_to_insert_this_proc,
155 current_inserting_particle_type,
156 particle_properties);
157
158 // Insert the particles using the points and assigned properties
159 particle_handler.insert_global_particles(
160 insertion_points_on_proc_this_step,
161 global_bounding_boxes,
162 particle_properties);
163
164 // Update number of particles remaining to be inserted
165 remaining_particles_of_each_type -= n_total_particles_to_insert;
166
167
168 ConditionalOStream pcout(
169 std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
170 this->print_insertion_info(n_total_particles_to_insert,
171 remaining_particles_of_each_type,
172 current_inserting_particle_type,
173 pcout);
174 }
175}

177template <int dim, typename PropertiesIndex>
178void
179InsertionList<dim, PropertiesIndex>::
180 assign_particle_properties_for_list_insertion(
181 const DEMSolverParameters<dim> &dem_parameters,
182 const unsigned int &inserted_this_step_this_proc,
183 const unsigned int &current_inserting_particle_type,
184 std::vector<std::vector<double>> &particle_properties)
185{
186 // Clearing and resizing particle_properties
187 particle_properties.reserve(inserted_this_step_this_proc);
188
189 // Getting properties as local parameters
190 // TODO: MAYBE CHANGE THE INPUT TO PHYSICAL PROPERTIES DIRECTLY
191 auto physical_properties = dem_parameters.lagrangian_physical_properties;
192
193 // A loop is defined over the number of particles which are going to be
194 // inserted at this step
195 for (unsigned int particle_counter = 0;
196 particle_counter < inserted_this_step_this_proc;
197 ++particle_counter)
198 {
199 double type = current_inserting_particle_type;
200 double diameter = this->diameters[particle_counter];
201 double density =
202 physical_properties.density_particle[current_inserting_particle_type];
203 double vel_x = this->velocities[particle_counter][0];
204 double vel_y = this->velocities[particle_counter][1];
205 double vel_z = this->velocities[particle_counter][2];
206 double omega_x = this->angular_velocities[particle_counter][0];
207 double omega_y = this->angular_velocities[particle_counter][1];
208 double omega_z = this->angular_velocities[particle_counter][2];
209 double mass = density * 4. / 3. * M_PI *
210 Utilities::fixed_power<3, double>(diameter * 0.5);
211
212 std::vector<double> properties_of_one_particle{
213 type, diameter, mass, vel_x, vel_y, vel_z, omega_x, omega_y, omega_z};
214
215 if constexpr (std::is_same_v<PropertiesIndex,
216 DEM::DEMMPProperties::PropertiesIndex>)
217 {
218 double T = this->temperatures[particle_counter];
219 double specific_heat =
220 physical_properties
221 .specific_heat_particle[current_inserting_particle_type];
222 properties_of_one_particle.push_back(T);
223 properties_of_one_particle.push_back(specific_heat);
224 }
225
226 if constexpr (std::is_same_v<PropertiesIndex,
227 DEM::CFDDEMProperties::PropertiesIndex>)
228 {
229 // Push back all zero variables for the CFD-DEM coupling properties
230 // fem_force variable
231 for (unsigned int d = 0; d < dim; ++d)
232 properties_of_one_particle.push_back(0.);
233
234 // fem_torque variable
235 for (unsigned int d = 0; d < dim; ++d)
236 properties_of_one_particle.push_back(0.);
237
238 // volumetric contribution
239 properties_of_one_particle.push_back(0.);
240 }
241
242 particle_properties.push_back(properties_of_one_particle);
243 properties_of_one_particle.clear();
244 }
245}
