35void
36write_forces_torques_output_results(
37 const std::string &filename,
38 const unsigned int output_frequency,
39 const std::vector<unsigned int> &boundary_index,
40 const double time_step,
41 DEM::dem_data_structures<3>::vector_on_boundary &forces_boundary_information,
42 DEM::dem_data_structures<3>::vector_on_boundary &torques_boundary_information)
43{
44 unsigned int this_mpi_process =
45 Utilities::MPI::this_mpi_process(MPI_COMM_WORLD);
46 unsigned int n_mpi_processes =
47 Utilities::MPI::n_mpi_processes(MPI_COMM_WORLD);
48 for (unsigned int i = 0; i < boundary_index.size(); i++)
49 {
50 std::string filename_boundary_description =
51 std::to_string(boundary_index[i]);
52 std::string filename_output =
53 filename + "_boundary_" + filename_boundary_description;
54 if (this_mpi_process == i || (i + 1) > n_mpi_processes)
55 {
56 TableHandler table;
57
58 for (unsigned int j = 0; j < forces_boundary_information.size();
59 j += output_frequency)
60 {
61 table.add_value("Force x", forces_boundary_information[j][i][0]);
62 table.add_value("Force y", forces_boundary_information[j][i][1]);
63 table.add_value("Force z", forces_boundary_information[j][i][2]);
64 table.add_value("Torque x",
65 torques_boundary_information[j][i][0]);
66 table.add_value("Torque y",
67 torques_boundary_information[j][i][1]);
68 table.add_value("Torque z",
69 torques_boundary_information[j][i][2]);
70 table.add_value("Current time", j * time_step);
71 }
72
73 table.set_precision("Force x", 9);
74 table.set_precision("Force y", 9);
75 table.set_precision("Force z", 9);
76 table.set_precision("Torque x", 9);
77 table.set_precision("Torque y", 9);
78 table.set_precision("Torque z", 9);
79 table.set_precision("Current time", 9);
80
81 // output
82 std::ofstream out_file(filename_output);
83 table.write_text(out_file);
84 out_file.close();
85 }
86 }
87}
