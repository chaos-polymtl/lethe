102  /**
103  * @brief Carries out calculation of the rolling resistance torque using the constant model
104  *
105  * @param particle_properties Particle properties
106  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
107  * @param normal_force_norm Normal force norm
108  *
109  * @return Rolling resistance torque
110  */
111  inline Tensor<1, 3>
112  constant_resistance(const ArrayView<const double> &particle_properties,
113  const double effective_rolling_friction_coefficient,
114  const double normal_force_norm,
115  const Tensor<1, 3> & /*normal_contact_vector*/)
116  {
117  // Getting the angular velocity of particle in the vector format
118  Tensor<1, 3> angular_velocity;
119  for (int d = 0; d < 3; ++d)
120  {
121  angular_velocity[d] =
122  particle_properties[DEM::PropertiesIndex::omega_x + d];
123  }
124 
125  // Calculation of particle-wall angular velocity (norm of the
126  // particle angular velocity)
127  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
128 
129  double omega_value = angular_velocity.norm();
130  if (omega_value != 0)
131  {
132  particle_wall_angular_velocity = angular_velocity / omega_value;
133  }
134 
135  // Calculation of rolling resistance torque
136  Tensor<1, 3> rolling_resistance_torque =
137  -effective_rolling_friction_coefficient *
138  (particle_properties[DEM::PropertiesIndex::dp] * 0.5) *
139  normal_force_norm * particle_wall_angular_velocity;
140 
141  return rolling_resistance_torque;
142  }

144  /**
145  * @brief Carries out calculation of the rolling resistance torque using the viscous model
146  *
147  * @param particle_properties Particle properties
148  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
149  * @param normal_force_norm Normal force norm
150  *
151  * @return rolling resistance torque
152  */
153  inline Tensor<1, 3>
154  viscous_resistance(const ArrayView<const double> &particle_properties,
155  const double effective_rolling_friction_coefficient,
156  const double normal_force_norm,
157  const Tensor<1, 3> &normal_contact_vector)
158  {
159  // Getting the angular velocity of particle in the vector format
160  Tensor<1, 3> angular_velocity;
161  for (int d = 0; d < 3; ++d)
162  {
163  angular_velocity[d] =
164  particle_properties[DEM::PropertiesIndex::omega_x + d];
165  }
166 
167  // Calculation of particle-wall angular velocity (norm of the
168  // particle angular velocity)
169  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
170 
171  double omega_value = angular_velocity.norm();
172  if (omega_value != 0)
173  {
174  particle_wall_angular_velocity = angular_velocity / omega_value;
175  }
176 
177  Tensor<1, 3> v_omega =
178  cross_product_3d(angular_velocity,
179  particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
180  normal_contact_vector);
181 
182  // Calculation of rolling resistance torque
183  Tensor<1, 3> rolling_resistance_torque =
184  -effective_rolling_friction_coefficient *
185  particle_properties[DEM::PropertiesIndex::dp] * 0.5 * normal_force_norm *
186  v_omega.norm() * particle_wall_angular_velocity;
187 
188  return rolling_resistance_torque;
189  }
