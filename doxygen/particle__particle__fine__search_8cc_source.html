16template <int dim>
17void
18particle_particle_fine_search(
19 const typename DEM::dem_data_structures<dim>::particle_index_iterator_map
20 &particle_container,
21 typename DEM::dem_data_structures<dim>::adjacent_particle_pairs
22 &adjacent_particles,
23 const typename DEM::dem_data_structures<dim>::particle_particle_candidates
24 &contact_pair_candidates,
25 const double neighborhood_threshold,
26 const Tensor<1, dim> periodic_offset)
27{
28 // First iterating over adjacent_particles
29 for (auto &&adjacent_particles_list :
30 adjacent_particles | boost::adaptors::map_values)
31 {
32 if (adjacent_particles_list.empty())
33 continue;
34
35 // Gather information about particle 1
36 auto &particle_one = adjacent_particles_list.begin()->second.particle_one;
37 Point<dim, double> particle_one_location = particle_one->get_location();
38
39 // Iterating over each map which contains the contact information
40 for (auto adjacent_particles_list_iterator =
41 adjacent_particles_list.begin();
42 adjacent_particles_list_iterator != adjacent_particles_list.end();)
43 {
44 // Getting contact information and particle 2 as local
45 // variables
46 auto &adjacent_pair_information =
47 adjacent_particles_list_iterator->second;
48 auto &particle_two = adjacent_pair_information.particle_two;
49
50 // Finding the properties of the particles in contact
51 Point<dim, double> particle_two_location =
52 particle_two->get_location() - periodic_offset;
53
54 // Finding distance
55 const double square_distance =
56 particle_one_location.distance_square(particle_two_location);
57 if (square_distance > neighborhood_threshold)
58 {
59 adjacent_particles_list_iterator =
60 adjacent_particles_list.erase(adjacent_particles_list_iterator);
61 }
62 else
63 {
64 ++adjacent_particles_list_iterator;
65 }
66 }
67 }
68
69 // Now iterating over contact_pair_candidates (maps of pairs), which
70 // is the output of broad search. If a pair is in vicinity (distance <
71 // threshold), it is added to the adjacent_particles
72 for (auto &[particle_one_id, second_particle_container] :
73 contact_pair_candidates)
74 {
75 if (second_particle_container.empty())
76 continue;
77
78 auto particle_one = particle_container.at(particle_one_id);
79 Point<dim, double> particle_one_location = particle_one->get_location();
80
81 for (const types::particle_index &particle_two_id :
82 second_particle_container)
83 {
84 auto particle_two = particle_container.at(particle_two_id);
85 Point<dim, double> particle_two_location =
86 particle_two->get_location() - periodic_offset;
87
88 // Finding distance
89 const double square_distance =
90 particle_one_location.distance_square(particle_two_location);
91
92 // If the particles distance is less than the threshold
93 if (square_distance < neighborhood_threshold)
94 {
95 // Getting the particle one contact list and particle two id
96 auto &particle_one_contact_list =
97 adjacent_particles[particle_one_id];
98
99 particle_one_contact_list.emplace(
100 particle_two_id,
101 particle_particle_contact_info<dim>{
102 particle_one, particle_two, Tensor<1, 3>(), Tensor<1, 3>()});
103 }
104 }
105 }
106}
