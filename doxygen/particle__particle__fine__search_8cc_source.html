19template <int dim, ContactType contact_type>
20void
21particle_particle_fine_search(
22 const typename DEM::dem_data_structures<dim>::particle_index_iterator_map
23 &particle_container,
24 adjacent_pairs_for_contact_t<dim, contact_type> &adjacent_particles,
25 const typename DEM::dem_data_structures<dim>::particle_particle_candidates
26 &contact_pair_candidates,
27 const double neighborhood_threshold,
28 const std::vector<Tensor<1, dim>> &combined_periodic_offsets)
29{
30 // First iterating over adjacent_particles
31 for (auto &&adjacent_particles_list :
32 adjacent_particles | boost::adaptors::map_values)
33 {
34 if (adjacent_particles_list.empty())
35 continue;
36
37 // Gather information about particle 1
38 auto &particle_one = adjacent_particles_list.begin()->second.particle_one;
39 Point<dim, double> particle_one_location = particle_one->get_location();
40
41 // For non-periodic contacts
42 if constexpr (contact_type == local_particle_particle ||
43 contact_type == ghost_particle_particle)
44 {
45 // Iterating over each map which contains the contact information
46 for (auto adjacent_particles_list_iterator =
47 adjacent_particles_list.begin();
48 adjacent_particles_list_iterator !=
49 adjacent_particles_list.end();)
50 {
51 // Getting contact information and particle 2 as local variables
52 auto &adjacent_pair_information =
53 adjacent_particles_list_iterator->second;
54 auto &particle_two = adjacent_pair_information.particle_two;
55
56 // Finding the properties of the particles in contact
57 Point<dim, double> particle_two_location =
58 particle_two->get_location();
59
60 const double square_distance =
61 particle_one_location.distance_square(particle_two_location);
62 if (square_distance > neighborhood_threshold)
63 {
64 adjacent_particles_list_iterator =
65 adjacent_particles_list.erase(
66 adjacent_particles_list_iterator);
67 }
68 else
69 {
70 ++adjacent_particles_list_iterator;
71 }
72 }
73 }
74
75 // For periodic contacts
76 if constexpr (contact_type == local_periodic_particle_particle ||
77 contact_type == ghost_periodic_particle_particle ||
78 contact_type == ghost_local_periodic_particle_particle)
79 {
80 // Iterating over each map which contains the contact information
81 for (auto adjacent_particles_list_iterator =
82 adjacent_particles_list.begin();
83 adjacent_particles_list_iterator !=
84 adjacent_particles_list.end();)
85 {
86 // Getting contact information and particle 2 as local variables
87 auto &adjacent_pair_information =
88 adjacent_particles_list_iterator->second;
89 auto &particle_two = adjacent_pair_information.particle_two;
90
91 // Finding the properties of the particles in contact
92 Point<dim, double> particle_two_real_location =
93 particle_two->get_location();
94
95 // Find minimum periodic distance squared between particle one and
96 // the set of periodic images of particle two. The nearest
97 // periodic image corresponds to a given periodic offset,
98 // indicating if the particles are in contact through a periodic
99 // edge, face, or corner.
100 double min_square_distance = std::numeric_limits<double>::max();
101 Tensor<1, dim> nearest_translation;
102
103 for (const auto &translation : combined_periodic_offsets)
104 {
105 // Check particle 1 against every periodic image of particle 2
106 double current_square_distance =
107 particle_one_location.distance_square(
108 particle_two_real_location + translation);
109
110 if (current_square_distance < min_square_distance)
111 {
112 min_square_distance = current_square_distance;
113 nearest_translation = translation;
114 }
115 }
116
117
118 // If simulation is well defined, there should be at most one
119 // translation from combined_periodic_offsets that brings a
120 // periodic image within a neighborhood threshold.
121 if (min_square_distance > neighborhood_threshold)
122 {
123 adjacent_particles_list_iterator =
124 adjacent_particles_list.erase(
125 adjacent_particles_list_iterator);
126 }
127 else
128 {
129 // Save a translation that falls within the threshold
130 Tensor<1, 3> offset_3d;
131 for (int d = 0; d < dim; ++d)
132 offset_3d[d] = nearest_translation[d];
133
134 adjacent_pair_information.periodic_offset = offset_3d;
135
136 ++adjacent_particles_list_iterator;
137 }
138 }
139 }
140 }
141
142 // Now iterating over contact_pair_candidates (maps of pairs), which
143 // is the output of broad search. If a pair is in vicinity (distance <
144 // threshold), it is added to the adjacent_particles
145 for (auto &[particle_one_id, second_particle_container] :
146 contact_pair_candidates)
147 {
148 if (second_particle_container.empty())
149 continue;
150
151 auto particle_one = particle_container.at(particle_one_id);
152 Point<dim, double> particle_one_location = particle_one->get_location();
153
154 for (const types::particle_index &particle_two_id :
155 second_particle_container)
156 {
157 auto particle_two = particle_container.at(particle_two_id);
158
159 // For non-periodic contacts
160 if constexpr (contact_type == local_particle_particle ||
161 contact_type == ghost_particle_particle)
162 {
163 Point<dim, double> particle_two_location =
164 particle_two->get_location();
165
166 const double square_distance =
167 particle_one_location.distance_square(particle_two_location);
168
169 if (square_distance < neighborhood_threshold)
170 {
171 auto &particle_one_contact_list =
172 adjacent_particles[particle_one_id];
173
174 particle_one_contact_list.emplace(
175 particle_two_id,
176 particle_particle_contact_info<dim>{particle_one,
177 particle_two,
178 Tensor<1, 3>(),
179 Tensor<1, 3>()});
180 }
181 }
182
183 // For periodic contacts
184 if constexpr (contact_type == local_periodic_particle_particle ||
185 contact_type == ghost_periodic_particle_particle ||
186 contact_type == ghost_local_periodic_particle_particle)
187 {
188 Point<dim, double> particle_two_real_location =
189 particle_two->get_location();
190
191 // Find minimum periodic distance squared between particle one and
192 // the set of periodic images of particle two. The nearest
193 // periodic image corresponds to a given periodic offset,
194 // indicating if the particles are in contact through a periodic
195 // edge, face, or corner.
196 double min_square_distance = std::numeric_limits<double>::max();
197 Tensor<1, dim> nearest_translation;
198
199 for (const auto &translation : combined_periodic_offsets)
200 {
201 double current_square_distance =
202 particle_one_location.distance_square(
203 particle_two_real_location + translation);
204
205 if (current_square_distance < min_square_distance)
206 {
207 min_square_distance = current_square_distance;
208 nearest_translation = translation;
209 }
210 }
211
212 // If simulation is well defined, there should be at most one
213 // translation from combined_periodic_offsets that brings a
214 // periodic image within a neighborhood threshold.
215 if (min_square_distance < neighborhood_threshold)
216 {
217 // Save a translation that falls within the threshold
218 Tensor<1, 3> offset_3d;
219 for (int d = 0; d < dim; ++d)
220 offset_3d[d] = nearest_translation[d];
221
222 auto &particle_one_contact_list =
223 adjacent_particles[particle_one_id];
224
225 particle_one_contact_list.emplace(
226 particle_two_id,
227 periodic_particle_particle_contact_info<dim>{
228 {particle_one,
229 particle_two,
230 Tensor<1, 3>(),
231 Tensor<1, 3>()},
232 offset_3d});
233 }
234 }
235 }
236 }
237}
