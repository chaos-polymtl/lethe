17template <int dim>
18void
19read_checkpoint(
20 TimerOutput &computing_timer,
21 const DEMSolverParameters<dim> &parameters,
22 std::shared_ptr<SimulationControl> &simulation_control,
23 PVDHandler &particles_pvdhandler,
24 PVDHandler &grid_pvdhandler,
25 parallel::distributed::Triangulation<dim> &triangulation,
26 Particles::ParticleHandler<dim> &particle_handler,
27 std::shared_ptr<Insertion<dim>> &insertion_object,
28 std::vector<std::shared_ptr<SerialSolid<dim - 1, dim>>> &solid_surfaces,
29 CheckpointControl &checkpoint_controller)
30{
31 if (!DEMActionManager::get_action_manager()->check_restart_simulation())
32 return;
33
34 TimerOutput::Scope timer(computing_timer, "Read checkpoint");
35 std::string prefix = checkpoint_controller.get_filename();
36
37 // Load checkpoint controller
38 std::string checkpoint_controller_object_filename =
39 prefix + ".checkpoint_controller";
40 std::ifstream iss_checkpoint_controller_obj(
41 checkpoint_controller_object_filename);
42 boost::archive::text_iarchive ia_checkpoint_controller_obj(
43 iss_checkpoint_controller_obj, boost::archive::no_header);
44 checkpoint_controller.deserialize(ia_checkpoint_controller_obj);
45
46 // Update prefix with the checkpoint id
47 prefix =
48 prefix + "_" +
49 Utilities::int_to_string(checkpoint_controller.get_next_checkpoint_id());
50
51 simulation_control->read(prefix);
52 particles_pvdhandler.read(prefix);
53
54 if (parameters.post_processing.lagrangian_post_processing_enabled)
55 {
56 grid_pvdhandler.read(prefix + "_lagrangian_postprocessing");
57 }
58
59 // Gather particle serialization information
60 std::string particle_filename = prefix + ".particles";
61 std::ifstream input(particle_filename.c_str());
62 AssertThrow(input, ExcFileNotOpen(particle_filename));
63
64 std::string buffer;
65 std::getline(input, buffer);
66 std::istringstream iss(buffer);
67 boost::archive::text_iarchive ia(iss, boost::archive::no_header);
68
69 ia >> particle_handler;
70
71 const std::string filename = prefix + ".triangulation";
72 std::ifstream in(filename.c_str());
73 if (!in)
74 AssertThrow(false,
75 ExcMessage(
76 std::string(
77 "You are trying to restart a previous computation, "
78 "but the restart file <") +
79 filename + "> does not appear to exist!"));
80
81 try
82 {
83 triangulation.load(filename.c_str());
84 }
85 catch (...)
86 {
87 AssertThrow(false,
88 ExcMessage("Cannot open snapshot mesh file or read the "
89 "triangulation stored there."));
90 }
91
92
93 // Unpack the information in the particle handler
94 particle_handler.deserialize();
95
96
97 // Load insertion object
98 std::string insertion_object_filename = prefix + ".insertion_object";
99 std::ifstream iss_insertion_obj(insertion_object_filename);
100 boost::archive::text_iarchive ia_insertion_obj(iss_insertion_obj,
101 boost::archive::no_header);
102 insertion_object->deserialize(ia_insertion_obj);
103
104 // Load solid surfaces
105 for (unsigned int i = 0; i < solid_surfaces.size(); ++i)
106 {
107 solid_surfaces[i]->read_checkpoint(prefix);
108 }
109}
