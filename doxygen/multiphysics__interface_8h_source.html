63 * @brief Write physic solved in the terminal

65 * @param physics_id number associated with auxiliary physics in multiphysics.h

69 {
70 if (physics_id == PhysicsID::heat_transfer)
71 {
72 announce_string(pcout, "Heat Transfer");
73 }
74 else if (physics_id == PhysicsID::tracer)
75 {

77 }
78 else if (physics_id == PhysicsID::VOF)
79 {
80 announce_string(pcout, "VOF");
81 }

83 {
84 announce_string(pcout, "Cahn-Hilliard");
85 }

88 /**
89 * @brief Call for the solution of the physics that should be solved
90 *
91 * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
92 * already solved or not. See the map `solve_pre_fluid` to know which
93 * subphysics are solved before the fluid dynamics.
94 *
95 * @param time_stepping_method Time-Stepping method with which the assembly is called
96 */
97 void
98 solve(const bool fluid_dynamics_has_been_solved,
99 const Parameters::SimulationControl::TimeSteppingMethod
100 time_stepping_method)
101 {
102 // Loop through all the elements in the physics map. Consequently, iphys is
103 // an std::pair where iphys.first is the PhysicsID and iphys.second is the
104 // AuxiliaryPhysics pointer. This is how the map can be traversed
105 // sequentially.
106 for (auto &iphys : physics)
107 {
108 // If iphys.first should be solved BEFORE fluid dynamics
109 if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
110 solve_physics(iphys.first, time_stepping_method);
111
112 // If iphys.first should be solved AFTER fluid dynamics OR if is not
113 // present in solve_pre_fluid map
114 else if (fluid_dynamics_has_been_solved &&
115 (!solve_pre_fluid[iphys.first] ||
116 solve_pre_fluid.count(iphys.first) == 0))
117 solve_physics(iphys.first, time_stepping_method);
118 }
119
120 for (auto &iphys : block_physics)
121 {
122 // If iphys.first should be solved BEFORE fluid dynamics
123 if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
124 solve_block_physics(iphys.first, time_stepping_method);
125
126 // If iphys.first should be solved AFTER fluid dynamics OR if is not
127 // present in solve_pre_fluid map
128 else if (fluid_dynamics_has_been_solved &&
129 (!solve_pre_fluid[iphys.first] ||
130 solve_pre_fluid.count(iphys.first) == 0))
131 solve_block_physics(iphys.first, time_stepping_method);
132 }
133 }

135 /**
136 * @brief Call for the solution of a single physic
137 *
138 * @param time_stepping_method Time-Stepping method with which the assembly is called
139 */
140 void
141 solve_physics(const PhysicsID physics_id,
142 const Parameters::SimulationControl::TimeSteppingMethod
143 time_stepping_method)
144 {
145 // Announce physic solved (verbosity = non_linear_solver.verbosity)
146 if (verbosity.at(physics_id) != Parameters::Verbosity::quiet)
147 announce_physics(physics_id);
148
149 AssertThrow(std::find(active_physics.begin(),
150 active_physics.end(),
151 physics_id) != active_physics.end(),
152 ExcInternalError());
153
154 physics[physics_id]->time_stepping_method = time_stepping_method;
155 physics[physics_id]->solve_non_linear_system(false);
156 physics[physics_id]->modify_solution();
157 }
158

219 /**
220 * @brief Rearrange vector solution correctly for transient simulations for
221 * all auxiliary physics.
222 *
223 * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
224 * already solved or not. See the map `solve_pre_fluid` to know which
225 * subphysics are solved before the fluid dynamics.
226 */
227 void
228 percolate_time_vectors(const bool fluid_dynamics_has_been_solved)
229 {
230 // Loop through all the elements in the physics map. Consequently, iphys is
231 // an std::pair where iphys.first is the PhysicsID and iphys.second is the
232 // AuxiliaryPhysics pointer. This is how the map can be traversed
233 // sequentially.
234 for (auto &iphys : physics)
235 {
236 // If iphys.first should be percolated BEFORE fluid dynamics is solved
237 if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
238 iphys.second->percolate_time_vectors();
239
240 // If iphys.first should be percolated AFTER fluid dynamics is solved OR
241 // if is not present in solve_pre_fluid map
242 else if (fluid_dynamics_has_been_solved &&
243 (!solve_pre_fluid[iphys.first] ||
244 solve_pre_fluid.count(iphys.first) == 0))
245 iphys.second->percolate_time_vectors();
246 }
247 for (auto &iphys : block_physics)
248 {
249 // If iphys.first should be percolated BEFORE fluid dynamics is solved
250 if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
251 iphys.second->percolate_time_vectors();
252
253 // If iphys.first should be percolated AFTER fluid dynamics is solved OR
254 // if is not present in solve_pre_fluid map
255 else if (fluid_dynamics_has_been_solved &&
256 (!solve_pre_fluid[iphys.first] ||
257 solve_pre_fluid.count(iphys.first) == 0))
258 iphys.second->percolate_time_vectors();
259 }
