108/**
109 * @brief A solver for the Volume-Averaged incompressible Navier-Stokes equations
110 * implemented in a matrix-free fashion.
111 *
112 * A matrix-free stabilized solver for the Volume-Averaged incompressible
113 * Navier-Stokes (VANS) equations implemented in a matrix-free fashion with a
114 * sum-factorization approach. It uses a continuous Galerkin discretization and
115 * solves the fully-coupled discretized problem in a monolithic way using
116 * multigrid preconditioners. This class greatly inherits from the matrix-free
117 * implementation of the Navier-Stokes equations.
118 *
119 * @tparam dim An integer that denotes the dimension of the space in which
120 * the flow is solved.
121 */
122template <int dim>
123class FluidDynamicsVANSMatrixFree : public FluidDynamicsMatrixFree<dim>
124{
125public:
126 /**
127 * @brief Constructor that sets the finite element degree and system operator
128 * according to simulation parameters. It initializes the CFD-DEM parameters,
129 * the particle mapping, the particle handler, and the void fraction manager.
130 * It also checks whether there are periodic boundaries.
131 *
132 * @param[in] param Relevant parameters for the solver.
133 */
134 FluidDynamicsVANSMatrixFree(CFDDEMSimulationParameters<dim> &param);
135
136 /**
137 * @brief Solve the problem defined by simulation parameters by iterating
138 * through time or through the mesh refinements.
139 */
140 void
141 solve() override;
142
143protected:
144 /**
145 * @brief Read particles arising from a DEM simulation
146 */
147 void
148 read_dem();
149
150 /**
151 * @brief Setup the degree of freedom of the flow solver and the degrees of freedom
152 * and constraints for the void fraction.
153 */
154 void
155 setup_dofs() override;
156
157 /**
158 * @brief Finish the time-step and manage the time-step end for the void
159 * fraction.
160 */
161 virtual void
162 finish_time_step_fd();
163
164 /**
165 * @brief Add void fraction field to output files.
166 *
167 * @return Vector of OutputStructs that will be used to write the output results as VTU files.
168 */
169 std::vector<OutputStruct<dim, LinearAlgebra::distributed::Vector<double>>>
170 gather_output_hook() override;
171
172 /**
173 * @brief Create geometric multigrid preconditioner.
174 */
175 void
176 create_GMG() override;
177
178 /**
179 * Initializes (or re-initializes) geometric multigrid preconditioner
180 */
181 void
182 initialize_GMG() override;
183
184 /// Simulation parameters for CFD-DEM simulations
185 CFDDEMSimulationParameters<dim> cfd_dem_simulation_parameters;
186
187 /// Mapping used for the particles
188 MappingQGeneric<dim> particle_mapping;
189
190 /// Particle-handler used to store particles in CFD-DEM simulations
191 Particles::ParticleHandler<dim, dim> particle_handler;
192
193 /// Object that manages the void fraction calculation from functions
194 /// or from parameters.
195 ParticleProjector<dim> particle_projector;
196
197 /// Member variables which are used to manage boundary conditions
198 bool has_periodic_boundaries;
199 Tensor<1, dim> periodic_offset;
200 unsigned int periodic_direction;
