134/**
135 * @brief A solver for the Volume-Averaged incompressible Navier-Stokes equations
136 * implemented in a matrix-free fashion.
137 *
138 * A matrix-free stabilized solver for the Volume-Averaged incompressible
139 * Navier-Stokes (VANS) equations implemented in a matrix-free fashion with a
140 * sum-factorization approach. It uses a continuous Galerkin discretization and
141 * solves the fully-coupled discretized problem in a monolithic way using
142 * multigrid preconditioners. This class greatly inherits from the matrix-free
143 * implementation of the Navier-Stokes equations.
144 *
145 * @tparam dim An integer that denotes the dimension of the space in which
146 * the flow is solved.
147 */
148template <int dim>
149class FluidDynamicsVANSMatrixFree : public FluidDynamicsMatrixFree<dim>
150{
151 using VectorType = LinearAlgebra::distributed::Vector<double>;
152
153public:
154 /**
155 * @brief Constructor that sets the finite element degree and system operator
156 * according to simulation parameters. It initializes the CFD-DEM parameters,
157 * the particle mapping, the particle handler, and the void fraction manager.
158 * It also checks whether there are periodic boundaries.
159 *
160 * @param[in] param Relevant parameters for the solver.
161 */
162 FluidDynamicsVANSMatrixFree(CFDDEMSimulationParameters<dim> &param);
163
164 /**
165 * @brief Solve the problem defined by simulation parameters by iterating
166 * through time or through the mesh refinements.
167 */
168 void
169 solve() override;
170
171protected:
172 /**
173 * @brief Read particles arising from a DEM simulation
174 */
175 virtual void
176 read_dem();
177
178 /**
179 * @brief Setup the degree of freedom of the flow solver and the degrees of freedom
180 * and constraints for the void fraction.
181 */
182 void
183 setup_dofs() override;
184
185 /**
186 * @brief Assembles the RHS of the VANS system of equation. This function
187 * also recalculates the particle-fluid interaction when the drag coupling
188 * is implicit.
189 */
190 void
191 assemble_system_rhs() override;
192
193 /**
194 * @brief Finish the time-step and manage the time-step end for the void
195 * fraction.
196 */
197 virtual void
198 finish_time_step_fd();
199
200 /**
201 * @brief Add void fraction field to output files.
202 *
203 * @return Vector of OutputStructs that will be used to write the output results as VTU files.
204 */
205 std::vector<OutputStruct<dim, LinearAlgebra::distributed::Vector<double>>>
206 gather_output_hook() override;
207
208 /**
209 * @brief Create geometric multigrid preconditioner.
210 */
211 void
212 create_GMG() override;
213
214 /**
215 * Initializes (or re-initializes) geometric multigrid preconditioner
216 */
217 void
218 initialize_GMG() override;
219
220 /**
221 * @brief Evaluate the time derivative of the void fraction.
222 * This evaluates the time-derivative as a solution vector which can be passed
223 * to the matrix-free operator.
224 */
225 void
226 evaluate_time_derivative_void_fraction();
227
228 /// Simulation parameters for CFD-DEM simulations
229 CFDDEMSimulationParameters<dim> cfd_dem_simulation_parameters;
230
231 /// Mapping used for the particles
232 MappingQGeneric<dim> particle_mapping;
233
234 /// Particle-handler used to store particles in CFD-DEM simulations
235 Particles::ParticleHandler<dim, dim> particle_handler;
236
237 /// Object that manages the void fraction calculation from functions
238 /// or from parameters.
239 ParticleProjector<dim> particle_projector;
240
241 /// Member variables which are used to manage boundary conditions
242 bool has_periodic_boundaries;
243 Tensor<1, dim> periodic_offset;
244 unsigned int periodic_direction;
245
246 /// Vector to store the time derivative of the void fraction
247 VectorType time_derivative_void_fraction;
248};
