17/**
18 * @brief Generates a 3D Fichera oven mesh by removing cells from a
19 * subdivided hyper-rectangle.
20 *
21 * The geometry is a staircase-shaped domain obtained from a 2x2x3 subdivided
22 * hyper-rectangle by removing three groups of cells at increasing z-levels.
23 *
24 * When @p colorize is true, boundary faces at the top of the remaining
25 * staircase (z = top_right[2]) are assigned boundary_id = 1 (e.g., inlet);
26 * all other boundary faces retain the default boundary_id = 0.
27 */
28template <int dim, int spacedim>
29class FicheraOvenGrid
30{
31public:
32 /**
33 * @brief Constructor that parses geometry parameters from a colon-separated
34 * string.
35 *
36 * @param[in] grid_arguments A colon-separated string, @code
37 * "bottom_left:top_right:colorize", with the following fields (coordinates
38 * are comma-separated):
39 *
40 * | # | Field | Format | Required | Description |
41 * |----|-------------|------------|----------|-----------------------------------------------|
42 * | 0 | bottom_left | x,y,z | yes | Bottom-left corner of the bounding box |
43 * | 1 | top_right | x,y,z | yes | Top-right corner of the bounding box |
44 * | 2 | colorize | true/false | no | Assign distinct boundary IDs (default: false) |
45 *
46 * Example: @code "0,0,0 :2,2,3 : true" @endcode
47 *
48 */
49 FicheraOvenGrid(const std::string &grid_arguments);
50
51 /**
52 * @brief Generate the Fichera oven mesh by removing cells from a subdivided
53 * hyper-rectangle.
54 *
55 * Starting from a 2x2x3 subdivided hyper-rectangle spanning from
56 * @p bottom_left to @p top_right, three groups of cells are removed to create
57 * a staircase geometry:
58 *
59 * @verbatim
60 * Level 1 (z > 1/3 height): remove (+x, +y) quadrant
61 * Level 2 (z > 2/3 height): additionally remove (+x, -y) and (-x, +y)
62 * @endverbatim
63 *
64 * The midpoints along x and y are at (bottom_left + top_right) / 2, and the
65 * z-thresholds are at bottom_left[2] + height/3 and bottom_left[2] +
66 * 2*height/3.
67 *
68 * When @p colorize is true, boundary faces at z = top_right[2] are assigned
69 * boundary_id = 1 (e.g., inlet/outlet); all other faces keep the default
70 * boundary_id = 0.
71 *
72 * @param[out] triangulation The triangulation to fill with the oven mesh.
73 */
74 void
75 make_grid(Triangulation<dim, spacedim> &triangulation);
76
77private:
78 std::string grid_arguments;
79 Point<dim> bottom_left;
80 Point<dim> top_right;
81 bool colorize;
82};
